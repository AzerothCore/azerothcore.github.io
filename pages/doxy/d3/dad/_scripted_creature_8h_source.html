103 void DoAction(int32 info, uint16 max = 0)
104 {
105 if (max)
106 RemoveNotExisting(); // pussywizard: when max is set, non existing can be chosen and nothing will happen
107
108 StorageType listCopy = storage_;
109 for (StorageType::const_iterator i = listCopy.begin(); i != listCopy.end(); ++i)
110 {
111 if (Creature* summon = ObjectAccessor::GetCreature(*me, *i))
112 if (summon->IsAIEnabled)
113 summon->AI()->DoAction(info);
114 }
115 }
116
117 template <class Predicate>
118 void DoAction(int32 info, Predicate&& predicate, uint16 max = 0)
119 {
120 if (max)
121 RemoveNotExisting(); // pussywizard: when max is set, non existing can be chosen and nothing will happen
122
123 // We need to use a copy of SummonList here, otherwise original SummonList would be modified
124 StorageType listCopy = storage_;
125 Acore::Containers::RandomResize<StorageType, Predicate>(listCopy, std::forward<Predicate>(predicate), max);
126
127 for (auto const& guid : listCopy)
128 {
129 Creature* summon = ObjectAccessor::GetCreature(*me, guid);
130 if (summon && summon->IsAIEnabled)
131 {
132 summon->AI()->DoAction(info);
133 }
134 else if (!summon)
135 {
136 storage_.remove(guid);
137 }
138 }
139 }

217 enum class Axis
218 {
219 AXIS_X,
220 AXIS_Y
221 };

223 /* This is called for bosses whenever an encounter is happening.
224 * - Arguments:
225 * - Position has to be passed as a constant pointer (&Position)
226 * - Axis is the X or Y axis that is used to decide the position threshold
227 * - Above decides if the boss position should be above the passed position
228 * or below.
229 * Example:
230 * Hodir is in room until his Y position is below the Door position:
231 * IsInRoom(doorPosition, AXIS_Y, false);
232 */
233 bool IsInRoom(const Position* pos, Axis axis, bool above)
234 {
235 if (!pos)
236 {
237 return true;
238 }
239
240 switch (axis)
241 {
242 case Axis::AXIS_X:
243 if ((!above && me->GetPositionX() < pos->GetPositionX()) || me->GetPositionX() > pos->GetPositionX())
244 {
245 EnterEvadeMode();
246 return false;
247 }
248 break;
249 case Axis::AXIS_Y:
250 if ((!above && me->GetPositionY() < pos->GetPositionY()) || me->GetPositionY() > pos->GetPositionY())
251 {
252 EnterEvadeMode();
253 return false;
254 }
255
256 break;
257 }
258
259 return true;
260 }
