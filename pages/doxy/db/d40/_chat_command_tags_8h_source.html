74 /************************** CONTAINER TAGS **********************************************\
75 |* Simple holder classes to differentiate between extraction methods *|
76 |* Must inherit from Acore::Impl::ChatCommands::ContainerTag *|
77 |* Must implement the following: *|
78 |* - TryConsume: ChatHandler const*, std::string_view -> ChatCommandResult *|
79 |* - on match, returns tail of the provided argument string (as std::string_view) *|
80 |* - on specific error, returns error message (as std::string&& or char const*) *|
81 |* - on generic error, returns std::nullopt (this will print command usage) *|
82 |* *|
83 |* - typedef value_type of type that is contained within the tag *|
84 |* - cast operator to value_type *|
85 |* *|

88 template <char... chars>
89 struct ExactSequence : Acore::Impl::ChatCommands::ContainerTag
90 {
91 using value_type = void;
92
93 ChatCommandResult TryConsume(ChatHandler const* handler, std::string_view args) const
94 {
95 if (args.empty())
96 return std::nullopt;
97 std::string_view start = args.substr(0, _string.length());
98 if (StringEqualI(start, _string))
99 {
100 auto [remainingToken, tail] = Acore::Impl::ChatCommands::tokenize(args.substr(_string.length()));
101 if (remainingToken.empty()) // if this is not empty, then we did not consume the full token
102 return tail;
103 start = args.substr(0, _string.length() + remainingToken.length());
104 }
105 return Acore::Impl::ChatCommands::FormatAcoreString(handler, LANG_CMDPARSER_EXACT_SEQ_MISMATCH, STRING_VIEW_FMT_ARG(_string), STRING_VIEW_FMT_ARG(start));
106 }
107
108 private:
109 static constexpr std::array<char, sizeof...(chars)> _storage = { chars... };
110 static_assert(!_storage.empty() && (_storage.back() == '\0'), "ExactSequence parameters must be null terminated! Use the EXACT_SEQUENCE macro to make this easier!");
111 static constexpr std::string_view _string = { _storage.data(), std::string_view::traits_type::length(_storage.data()) };
112 };

203 template <typename linktag>
204 struct Hyperlink : Acore::Impl::ChatCommands::ContainerTag
205 {
206 using value_type = typename linktag::value_type;
207 using storage_type = advstd::remove_cvref_t<value_type>;
208
209 operator value_type() const { return val; }
210 value_type operator*() const { return val; }
211 storage_type const* operator->() const { return &val; }
212
213 ChatCommandResult TryConsume(ChatHandler const* handler, std::string_view args)
214 {
215 Acore::Hyperlinks::HyperlinkInfo info = Acore::Hyperlinks::ParseSingleHyperlink(args);
216 // invalid hyperlinks cannot be consumed
217 if (!info)
218 return std::nullopt;
219
220 // check if we got the right tag
221 if (info.tag != linktag::tag())
222 return std::nullopt;
223
224 // store value
225 if (!linktag::StoreTo(val, info.data))
226 return Acore::Impl::ChatCommands::GetAcoreString(handler, LANG_CMDPARSER_LINKDATA_INVALID);
227
228 // finally, skip any potential delimiters
229 auto [token, next] = Acore::Impl::ChatCommands::tokenize(info.tail);
230 if (token.empty()) /* empty token = first character is delimiter, skip past it */
231 return next;
232 else
233 return info.tail;
234 }
235
236 private:
237 storage_type val;
238 };
