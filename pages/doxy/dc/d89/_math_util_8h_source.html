26// based on https://stackoverflow.com/questions/7616511/calculate-mean-and-standard-deviation-from-a-vector-of-samples-in-c-using-boos/12405793#comment32490316_12405793
27template <typename Container, typename T = typename std::decay<decltype(*std::begin(std::declval<Container>()))>::type>
28inline T standard_deviation(Container&& c)
29{
30 auto b = std::begin(c), e = std::end(c);
31 auto size = std::distance(b, e);
32 auto sum = std::accumulate(b, e, T());
33 auto mean = sum / size;
34
35 if (size == 1)
36 {
37 return (T) 0;
38 }
39
40 T accum = T();
41 for (const auto d : c)
42 {
43 accum += (d - mean) * (d - mean);
44 }
45 return std::sqrt(accum / (size - 1));
46}

57// based off https://www.geeksforgeeks.org/finding-median-of-unsorted-array-in-linear-time-using-c-stl/
58template <typename T>
59inline T median(std::vector<T> a)
60{
61 size_t n = a.size();
62 // If size of the arr[] is even
63 if (n % 2 == 0)
64 {
65
66 // Applying nth_element
67 // on n/2th index
68 std::nth_element(a.begin(),
69 a.begin() + n / 2,
70 a.end());
71
72 // Applying nth_element
73 // on (n-1)/2 th index
74 std::nth_element(a.begin(),
75 a.begin() + (n - 1) / 2,
76 a.end());
77
78 // Find the average of value at
79 // index N/2 and (N-1)/2
80 return (T)(a[(n - 1) / 2]
81 + a[n / 2])
82 / 2.0;
83 }
84
85 // If size of the arr[] is odd
86 else
87 {
88
89 // Applying nth_element
90 // on n/2
91 std::nth_element(a.begin(),
92 a.begin() + n / 2,
93 a.end());
94
95 // Value at index (N/2)th
96 // is the median
97 return (T)a[n / 2];
98 }
99}
