187struct BattlegroundWGScore final : public BattlegroundScore
188{
189 friend class BattlegroundWS;
190
191protected:
192 BattlegroundWGScore(ObjectGuid playerGuid) : BattlegroundScore(playerGuid) { }
193
194 void UpdateScore(uint32 type, uint32 value) override
195 {
196 switch (type)
197 {
198 case SCORE_FLAG_CAPTURES: // Flags captured
199 FlagCaptures += value;
200 break;
201 case SCORE_FLAG_RETURNS: // Flags returned
202 FlagReturns += value;
203 break;
204 default:
205 BattlegroundScore::UpdateScore(type, value);
206 break;
207 }
208 }
209
210 void BuildObjectivesBlock(WorldPacket& data) final;
211
212 uint32 GetAttr1() const override { return FlagCaptures; }
213 uint32 GetAttr2() const override { return FlagReturns; }
214
215 uint32 FlagCaptures = 0;
216 uint32 FlagReturns = 0;
217};
