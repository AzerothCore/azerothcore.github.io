130 struct boss_four_horsemenAI : public BossAI
131 {
132 explicit boss_four_horsemenAI(Creature* c) : BossAI(c, BOSS_HORSEMAN)
133 {
134 pInstance = me->GetInstanceScript();
135 switch (me->GetEntry())
136 {
137 case NPC_SIR_ZELIEK:
138 horsemanId = HORSEMAN_ZELIEK;
139 break;
140 case NPC_LADY_BLAUMEUX:
141 horsemanId = HORSEMAN_BLAUMEUX;
142 break;
143 case NPC_BARON_RIVENDARE:
144 horsemanId = HORSEMAN_RIVENDARE;
145 break;
146 case NPC_THANE_KORTHAZZ:
147 horsemanId = HORSEMAN_KORTHAZZ;
148 break;
149 }
150 }
151
152 EventMap events;
153 InstanceScript* pInstance;
154 uint8 currentWaypoint{};
155 uint8 movementPhase{};
156 uint8 horsemanId;
157
158 void MoveToCorner()
159 {
160 switch(me->GetEntry())
161 {
162 case NPC_THANE_KORTHAZZ:
163 currentWaypoint = 0;
164 break;
165 case NPC_LADY_BLAUMEUX:
166 currentWaypoint = 3;
167 break;
168 case NPC_BARON_RIVENDARE:
169 currentWaypoint = 6;
170 break;
171 case NPC_SIR_ZELIEK:
172 currentWaypoint = 9;
173 break;
174 }
175 me->GetMotionMaster()->MovePoint(currentWaypoint, WaypointPositions[currentWaypoint]);
176 }
177
178 bool IsInRoom()
179 {
180 if (me->GetExactDist(2535.1f, -2968.7f, 241.3f) > 100.0f)
181 {
182 EnterEvadeMode();
183 return false;
184 }
185 return true;
186 }
187
188 void Reset() override
189 {
190 BossAI::Reset();
191 me->SetPosition(me->GetHomePosition());
192 movementPhase = MOVE_PHASE_NONE;
193 currentWaypoint = 0;
194 me->SetReactState(REACT_AGGRESSIVE);
195 events.Reset();
196 events.RescheduleEvent(EVENT_MARK_CAST, 24000);
197 events.RescheduleEvent(EVENT_BERSERK, 600000);
198 if ((me->GetEntry() != NPC_LADY_BLAUMEUX && me->GetEntry() != NPC_SIR_ZELIEK))
199 {
200 events.RescheduleEvent(EVENT_PRIMARY_SPELL, 10000 + rand() % 5000);
201 }
202 else
203 {
204 events.RescheduleEvent(EVENT_PUNISH, 5000);
205 events.RescheduleEvent(EVENT_SECONDARY_SPELL, 15000);
206 }
207 if (pInstance)
208 {
209 if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_HORSEMEN_GATE)))
210 {
211 if (pInstance->GetBossState(BOSS_GOTHIK) == DONE)
212 {
213 go->SetGoState(GO_STATE_ACTIVE);
214 }
215 }
216 }
217 }
218
219 void MovementInform(uint32 type, uint32 id) override
220 {
221 if (type != POINT_MOTION_TYPE)
222 return;
223
224 // final waypoint
225 if (id % 3 == 2)
226 {
227 movementPhase = MOVE_PHASE_FINISHED;
228 me->SetReactState(REACT_AGGRESSIVE);
229 me->SetInCombatWithZone();
230 if (!UpdateVictim())
231 {
232 EnterEvadeMode();
233 return;
234 }
235 if (me->GetEntry() == NPC_LADY_BLAUMEUX || me->GetEntry() == NPC_SIR_ZELIEK)
236 {
237 me->GetMotionMaster()->Clear(false);
238 me->GetMotionMaster()->MoveIdle();
239 }
240 return;
241 }
242 currentWaypoint = id + 1;
243 }
244
245 void AttackStart(Unit* who) override
246 {
247 if (movementPhase == MOVE_PHASE_FINISHED)
248 {
249 if (me->GetEntry() == NPC_LADY_BLAUMEUX || me->GetEntry() == NPC_SIR_ZELIEK)
250 {
251 me->Attack(who, false);
252 }
253 else
254 {
255 ScriptedAI::AttackStart(who);
256 }
257 }
258 }
259
260 void KilledUnit(Unit* who) override
261 {
262 if (who->GetTypeId() != TYPEID_PLAYER)
263 return;
264
265 Talk(SAY_SLAY);
266 if (pInstance)
267 {
268 pInstance->SetData(DATA_IMMORTAL_FAIL, 0);
269 }
270 }
271
272 void JustDied(Unit* killer) override
273 {
274 BossAI::JustDied(killer);
275 if (pInstance)
276 {
277 if (pInstance->GetBossState(BOSS_HORSEMAN) == DONE)
278 {
279 if (!me->GetMap()->GetPlayers().IsEmpty())
280 {
281 if (Player* player = me->GetMap()->GetPlayers().getFirst()->GetSource())
282 {
283 if (GameObject* chest = player->SummonGameObject(RAID_MODE(GO_HORSEMEN_CHEST_10, GO_HORSEMEN_CHEST_25), 2514.8f, -2944.9f, 245.55f, 5.51f, 0, 0, 0, 0, 0))
284 {
285 chest->SetLootRecipient(me);
286 }
287 }
288 }
289 if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_HORSEMEN_GATE)))
290 {
291 go->SetGoState(GO_STATE_ACTIVE);
292 }
293 }
294 }
295 Talk(SAY_DEATH);
296 }
297
298 void EnterCombat(Unit* who) override
299 {
300 BossAI::EnterCombat(who);
301 if (movementPhase == MOVE_PHASE_NONE)
302 {
303 Talk(SAY_AGGRO);
304 me->SetReactState(REACT_PASSIVE);
305 movementPhase = MOVE_PHASE_STARTED;
306 me->SetSpeed(MOVE_RUN, me->GetSpeedRate(MOVE_RUN), true);
307 MoveToCorner();
308 }
309 if (pInstance)
310 {
311 if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_HORSEMEN_GATE)))
312 {
313 go->SetGoState(GO_STATE_READY);
314 }
315 }
316 }
317
318 void UpdateAI(uint32 diff) override
319 {
320 if (movementPhase == MOVE_PHASE_STARTED && currentWaypoint)
321 {
322 me->GetMotionMaster()->MovePoint(currentWaypoint, WaypointPositions[currentWaypoint]);
323 currentWaypoint = 0;
324 }
325
326 if (!IsInRoom())
327 return;
328
329 if (movementPhase < MOVE_PHASE_FINISHED || !UpdateVictim())
330 return;
331
332 events.Update(diff);
333 if (me->HasUnitState(UNIT_STATE_CASTING))
334 return;
335
336 switch (events.ExecuteEvent())
337 {
338 case EVENT_MARK_CAST:
339 me->CastSpell(me, TABLE_SPELL_MARK[horsemanId], false);
340 events.RepeatEvent((me->GetEntry() == NPC_LADY_BLAUMEUX || me->GetEntry() == NPC_SIR_ZELIEK) ? 15000 : 12000);
341 return;
342 case EVENT_BERSERK:
343 Talk(SAY_SPECIAL);
344 me->CastSpell(me, SPELL_BERSERK, true);
345 return;
346 case EVENT_PRIMARY_SPELL:
347 Talk(SAY_TAUNT);
348 me->CastSpell(me->GetVictim(), RAID_MODE(TABLE_SPELL_PRIMARY_10[horsemanId], TABLE_SPELL_PRIMARY_25[horsemanId]), false);
349 events.RepeatEvent(15000);
350 return;
351 case EVENT_PUNISH:
352 if (!SelectTarget(SelectTargetMethod::MaxDistance, 0, 45.0f, true))
353 {
354 me->CastSpell(me, TABLE_SPELL_PUNISH[horsemanId], false);
355 Talk(EMOTE_RAGECAST);
356 }
357 events.RepeatEvent(2010);
358 return;
359 case EVENT_SECONDARY_SPELL:
360 me->CastSpell(me->GetVictim(), RAID_MODE(TABLE_SPELL_SECONDARY_10[horsemanId], TABLE_SPELL_SECONDARY_25[horsemanId]), false);
361 events.RepeatEvent(15000);
362 return;
363 }
364
365 if ((me->GetEntry() == NPC_LADY_BLAUMEUX || me->GetEntry() == NPC_SIR_ZELIEK))
366 {
367 if (Unit* target = SelectTarget(SelectTargetMethod::MaxDistance, 0, 45.0f, true))
368 {
369 me->CastSpell(target, RAID_MODE(TABLE_SPELL_PRIMARY_10[horsemanId], TABLE_SPELL_PRIMARY_25[horsemanId]), false);
370 }
371 }
372 else
373 {
374 DoMeleeAttackIfReady();
375 }
376 }
377 };
