54struct boss_moam : public BossAI
55{
56 boss_moam(Creature* creature) : BossAI(creature, DATA_MOAM) {}
57
58 void InitializeAI() override
59 {
60 me->m_CombatDistance = 50.0f;
61 Reset();
62 }
63
64 void Reset() override
65 {
66 _Reset();
67 me->SetPower(POWER_MANA, 0);
68 me->SetRegeneratingPower(false);
69 }
70
71 void EnterCombat(Unit* who) override
72 {
73 BossAI::EnterCombat(who);
74 Talk(EMOTE_AGGRO);
75 events.ScheduleEvent(EVENT_STONE_PHASE, 90s);
76 events.ScheduleEvent(EVENT_SPELL_TRAMPLE, 9s);
77 events.ScheduleEvent(EVENT_SPELL_DRAIN_MANA, 3s);
78 }
79
80 void JustDied(Unit* /*killer*/) override
81 {
82 _JustDied();
83 DoCastAOE(SPELL_LARGE_OBSIDIAN_CHUNK, true);
84 }
85
86 void SummonedCreatureDies(Creature* /*creature*/, Unit* /*killer*/) override
87 {
88 if (!summons.IsAnyCreatureAlive() && me->HasAura(SPELL_ENERGIZE))
89 {
90 events.RescheduleEvent(EVENT_STONE_PHASE_END, 1s);
91 }
92 }
93
94 void UpdateAI(uint32 diff) override
95 {
96 if (!UpdateVictim())
97 return;
98
99 events.Update(diff);
100
101 if (me->GetPower(POWER_MANA) == me->GetMaxPower(POWER_MANA))
102 {
103 if (me->HasAura(SPELL_ENERGIZE))
104 {
105 me->RemoveAurasDueToSpell(SPELL_ENERGIZE);
106 events.RescheduleEvent(EVENT_STONE_PHASE_END, 1s);
107 }
108
109 Talk(EMOTE_MANA_FULL);
110 DoCastAOE(SPELL_ARCANE_ERUPTION);
111 }
112
113 while (uint32 eventId = events.ExecuteEvent())
114 {
115 switch (eventId)
116 {
117 case EVENT_STONE_PHASE:
118 Talk(EMOTE_STONE_PHASE);
119 DoCastAOE(SPELL_SUMMON_MANA_FIENDS);
120 DoCastSelf(SPELL_ENERGIZE);
121 events.CancelEvent(EVENT_SPELL_DRAIN_MANA);
122 events.ScheduleEvent(EVENT_STONE_PHASE_END, 90s);
123 break;
124 case EVENT_STONE_PHASE_END:
125 me->RemoveAurasDueToSpell(SPELL_ENERGIZE);
126 events.ScheduleEvent(EVENT_SPELL_DRAIN_MANA, 2s, 6s);
127 events.ScheduleEvent(EVENT_STONE_PHASE, 90s);
128 break;
129 case EVENT_SPELL_DRAIN_MANA:
130 DoCastAOE(SPELL_DRAIN_MANA_SERVERSIDE);
131 events.ScheduleEvent(EVENT_SPELL_DRAIN_MANA, 2s, 6s);
132 break;
133 case EVENT_SPELL_TRAMPLE:
134 DoCastAOE(SPELL_TRAMPLE);
135 events.ScheduleEvent(EVENT_SPELL_TRAMPLE, 15s);
136 break;
137 default:
138 break;
139 }
140 }
141
142 DoMeleeAttackIfReady();
143 }
144};

146// 25676 - Drain Mana (server-side)
147class spell_moam_mana_drain_filter : public SpellScript
148{
149 PrepareSpellScript(spell_moam_mana_drain_filter);
150
151 void FilterTargets(std::list<WorldObject*>& targets)
152 {
153 targets.remove_if([&](WorldObject* target) -> bool
154 {
155 return !target->IsPlayer() || target->ToPlayer()->getPowerType() != POWER_MANA;
156 });
157
158 if (!targets.empty())
159 {
160 Acore::Containers::RandomResize(targets, 6);
161 }
162 }
163
164 void HandleScript(SpellEffIndex /*effIndex*/)
165 {
166 if (Unit* caster = GetCaster())
167 {
168 caster->CastSpell(GetHitUnit(), SPELL_DRAIN_MANA, true);
169 }
170 }
171
172 void Register() override
173 {
174 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_moam_mana_drain_filter::FilterTargets, EFFECT_ALL, TARGET_UNIT_SRC_AREA_ENEMY);
175 OnEffectHitTarget += SpellEffectFn(spell_moam_mana_drain_filter::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
176 }
177};
