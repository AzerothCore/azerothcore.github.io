88 void SpellHitTarget(Unit* target, SpellInfo const* spellInfo) override
89 {
90 if (spellInfo->Id == SPELL_TIME_STEP_H || spellInfo->Id == SPELL_TIME_STEP_N)
91 {
92 if (target == me)
93 return;
94
95 if (warps >= 2)
96 {
97 warps = 0;
98 return;
99 }
100 warps++;
101 me->CastSpell(target, DUNGEON_MODE(SPELL_TIME_STEP_N, SPELL_TIME_STEP_H), true);
102 }
103 }
104
105 void UpdateAI(uint32 diff) override
106 {
107 if (!UpdateVictim())
108 return;
109
110 events.Update(diff);
111 if (me->HasUnitState(UNIT_STATE_CASTING))
112 return;
113
114 switch (events.ExecuteEvent())
115 {
116 case EVENT_SPELL_CURSE_OF_EXERTION:
117 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
118 me->CastSpell(target, SPELL_CURSE_OF_EXERTION, false);
119 events.RepeatEvent(9000);
120 break;
121 case EVENT_SPELL_WOUNDING_STRIKE:
122 me->CastSpell(me->GetVictim(), DUNGEON_MODE(SPELL_WOUNDING_STRIKE_N, SPELL_WOUNDING_STRIKE_H), false);
123 events.RepeatEvent(6000);
124 break;
125 case EVENT_SPELL_TIME_STOP:
126 me->CastSpell(me, SPELL_TIME_STOP, false);
127 events.RepeatEvent(20000);
128 break;
129 case EVENT_SPELL_TIME_WARP:
130 Talk(SAY_TIME_WARP);
131 me->CastSpell(me, SPELL_TIME_WARP, false);
132 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
133 me->CastSpell(target, DUNGEON_MODE(SPELL_TIME_STEP_N, SPELL_TIME_STEP_H), true);
134
135 events.RepeatEvent(25000);
136 break;
137 }
138
139 DoMeleeAttackIfReady();
140 }
141
142 void JustDied(Unit* /*killer*/) override
143 {
144 Talk(SAY_DEATH);
145 }
146
147 void KilledUnit(Unit* /*victim*/) override
148 {
149 if (!urand(0, 1))
150 return;
151
152 Talk(SAY_SLAY);
153 }
154 };
