40struct boss_moira_bronzebeardAI : public BossAI
41{
42 // use a default value so we can inherit for priestess
43 boss_moira_bronzebeardAI(Creature* creature, uint32 data = DATA_MOIRA) : BossAI(creature, data) {}
44 void EnterCombat(Unit* /*who*/) override
45 {
46 _EnterCombat();
47 events.ScheduleEvent(SPELL_MINDBLAST, 0.5 * (int) TIMER_MINDBLAST);
48 events.ScheduleEvent(SPELL_HEAL, 0.5 * (int) TIMER_HEAL);
49 events.ScheduleEvent(SPELL_RENEW, 0.5 * (int) TIMER_RENEW);
50 }
51
52 void UpdateAI(uint32 diff) override
53 {
54 if (!UpdateVictim())
55 {
56 return;
57 }
58 events.Update(diff);
59
60 if (me->HasUnitState(UNIT_STATE_CASTING))
61 {
62 return;
63 }
64 while (uint32 eventId = events.ExecuteEvent())
65 {
66 switch (eventId)
67 {
68 case SPELL_MINDBLAST:
69 DoCastVictim(SPELL_MINDBLAST);
70 events.ScheduleEvent(SPELL_MINDBLAST, urand(TIMER_MINDBLAST - 2000, TIMER_MINDBLAST + 2000));
71 break;
72 case SPELL_HEAL:
73 CastOnEmperorIfPossible(SPELL_HEAL, TIMER_HEAL);
74 break;
75 case SPELL_RENEW:
76 CastOnEmperorIfPossible(SPELL_RENEW, TIMER_RENEW);
77 break;
78 default:
79 break;
80 }
81 }
82 DoMeleeAttackIfReady();
83 }
84
85 void CastOnEmperorIfPossible(uint32 spell, uint32 timer)
86 {
87 Creature* emperor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_EMPEROR));
88 if (emperor && emperor->HealthBelowPct(90))
89 {
90 DoCast(emperor, spell);
91 }
92 else if (HealthBelowPct(90))
93 {
94 DoCastSelf(spell);
95 }
96 events.ScheduleEvent(spell, urand(timer - 2000, timer + 2000));
97 }
98};
