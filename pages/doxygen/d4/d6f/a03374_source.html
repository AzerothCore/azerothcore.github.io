97 void ScheduleEvents(bool troll)
98 {
99 events.Reset();
100 if (troll)
101 {
102 events.RescheduleEvent(EVENT_STAMPEDE, 10000);
103 events.RescheduleEvent(EVENT_WHIRLING_SLASH, 21000);

108 events.RescheduleEvent(EVENT_ENRAGE, 15000);

119 ScheduleEvents(true);

121 me->InterruptNonMeleeSpells(true);

124 void JustSummoned(Creature* summon) override
125 {
126 uint32 despawnTime = 0;
127 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 60.0f, true))
128 {
129 summon->CastSpell(target, SPELL_STAMPEDE_DMG, true);

131 }
132
133 summon->DespawnOrUnsummon(despawnTime);
134 }

146
147 void KilledUnit(Unit*) override
148 {
149 if (events.GetNextEventTime(EVENT_KILL_TALK) == 0)
150 {
151 Talk(SAY_SLAY);
152 events.ScheduleEvent(EVENT_KILL_TALK, 6000);
153 }
154 }
155
156 void UpdateAI(uint32 diff) override
157 {
158 if (!UpdateVictim())
159 return;
160
161 events.Update(diff);
162 if (me->HasUnitState(UNIT_STATE_CASTING | UNIT_STATE_CHARGING))
163 return;
164
165 switch (events.ExecuteEvent())
166 {
167 case EVENT_STAMPEDE:
168 Talk(SAY_SUMMON_RHINO);
169 me->CastSpell(me->GetVictim(), SPELL_STAMPEDE, false);
170 events.ScheduleEvent(EVENT_STAMPEDE, 15000);
171 break;
172 case EVENT_WHIRLING_SLASH:
173 if (++phaseCounter >= 3)
174 {
175 ScheduleEvents(false);
176 me->CastSpell(me, SPELL_TRANSFORM_TO_RHINO, false);
177 Talk(SAY_TRANSFORM_1);
178 phaseCounter = 0;
179 return;
180 }
181 events.ScheduleEvent(EVENT_WHIRLING_SLASH, 21000);
182 me->CastSpell(me, SPELL_WHIRLING_SLASH, false);
183 break;
184 case EVENT_PUNCTURE:
185 me->CastSpell(me->GetVictim(), SPELL_PUNCTURE, false);
186 events.ScheduleEvent(EVENT_PUNCTURE, 8000);
187 break;
188 case EVENT_ENRAGE:
189 me->CastSpell(me, SPELL_ENRAGE, false);
190 events.ScheduleEvent(EVENT_ENRAGE, 20000);
191 break;
192 case EVENT_STOMP:
193 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 30.0f, true))
194 me->CastSpell(target, SPELL_STOMP, false);
195 events.ScheduleEvent(EVENT_STOMP, 20000);
196 break;
197 case EVENT_IMPALING_CHARGE:
198 if (++phaseCounter >= 3)
199 {
200 ScheduleEvents(true);
201 me->CastSpell(me, SPELL_TRANSFORM_TO_TROLL, false);
202 Talk(SAY_TRANSFORM_2);
203 phaseCounter = 0;
204 return;
205 }
206 events.ScheduleEvent(EVENT_IMPALING_CHARGE, 21000);
207 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 100.0f, true))
208 {
209 me->CastSpell(target, SPELL_IMPALING_CHARGE, false);
210 impaledList.insert(target->GetGUID());
211 }
212 break;
213 }
214
215 DoMeleeAttackIfReady();
216 }
217 };
