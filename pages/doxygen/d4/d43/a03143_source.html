161 bool CheckCaravan()
162 {
163 for (uint8 i = 0; i < MAX_CARAVAN_SUMMONS; ++i)
164 {
165 if (!summons[i])
166 {
167 SummonHelpers();
168 return false;
169 }
170
171 Creature* summon = ObjectAccessor::GetCreature(*me, summons[i]);
172 if (!summon || me->GetDistance2d(summon) > 25.0f)
173 {
174 SummonHelpers();
