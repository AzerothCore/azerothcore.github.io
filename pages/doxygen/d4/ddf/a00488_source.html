110inline uint32 getMSTimeDiff(uint32 oldMSTime, uint32 newMSTime)
111{
112 // getMSTime() have limited data range and this is case when it overflow in this tick
113 if (oldMSTime > newMSTime)
114 {
115 return (0xFFFFFFFF - oldMSTime) + newMSTime;
116 }
117 else
118 {
119 return newMSTime - oldMSTime;
120 }
121}

123inline uint32 getMSTimeDiff(uint32 oldMSTime, TimePoint newTime)
124{
125 using namespace std::chrono;

148{
149public:
150 IntervalTimer() = default;
151
152 void Update(time_t diff)
153 {
154 _current += diff;
155 if (_current < 0)
156 {
157 _current = 0;
158 }
159 }
160
161 bool Passed()
162 {
163 return _current >= _interval;
164 }
165
166 void Reset()
167 {
168 if (_current >= _interval)

170 _current %= _interval;
171 }
172 }
173

177 }
178
179 void SetInterval(time_t interval)

181 _interval = interval;
182 }
183

188
189 [[nodiscard]] time_t GetCurrent() const

263struct PeriodicTimer
264{
265public:
266 PeriodicTimer(int32 period, int32 start_time)
267 : i_period(period), i_expireTime(start_time)
268 {
269 }
270
271 bool Update(const uint32 diff)
272 {
273 if ((i_expireTime -= diff) > 0)
274 {
275 return false;
276 }
277
278 i_expireTime += i_period > int32(diff) ? i_period : diff;
279 return true;
280 }
281
282 void SetPeriodic(int32 period, int32 start_time)
283 {
284 i_expireTime = start_time;
285 i_period = period;
286 }
287
288 // Tracker interface
289 void TUpdate(int32 diff) { i_expireTime -= diff; }
290 [[nodiscard]] bool TPassed() const { return i_expireTime <= 0; }
291 void TReset(int32 diff, int32 period) { i_expireTime += period > diff ? period : diff; }
292
293private:
294 int32 i_period;
295 int32 i_expireTime;
296};
