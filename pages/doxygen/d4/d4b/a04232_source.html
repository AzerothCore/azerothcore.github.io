36 {
37 uint32 level = player->getLevel();

42 moneyRew = 2300;
43 else if (level <= 69)
44 moneyRew = 9000;
45 else if (level <= 70)
46 moneyRew = 11200;
47 else if (level <= 79)
48 moneyRew = 12000;
49 else
50 moneyRew = 19000;
51
52 player->ModifyMoney(moneyRew);
53 player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_MONEY_FROM_QUEST_REWARD, uint32(moneyRew));
54 player->SaveToDB(false, false);
55
56 // Send packet with money
