155 boss_nazanAI(Creature* creature) : ScriptedAI(creature)
156 {
157 }
158
159 void Reset() override
160 {
161 me->SetCanFly(true);
162 me->SetDisableGravity(true);
163 events.Reset();
164 }
165
166 void EnterEvadeMode(EvadeReason /*why*/) override
167 {
168 me->DespawnOrUnsummon(1);
169 }
170
171 void EnterCombat(Unit*) override
172 {
173 events.ScheduleEvent(EVENT_CHANGE_POS, 0);
174 events.ScheduleEvent(EVENT_SPELL_FIREBALL, 5000);
175 }
176
177 void AttackStart(Unit* who) override
178 {
179 if (me->IsLevitating())
180 me->Attack(who, true);
181 else
182 ScriptedAI::AttackStart(who);
183 }
184
185 void DoAction(int32 param) override
186 {
187 if (param == ACTION_FLY_DOWN)
188 {
189 Talk(EMOTE_NAZAN);
190 events.Reset();
191 me->GetMotionMaster()->MovePoint(POINT_MIDDLE, -1406.5f, 1746.5f, 81.2f, false);
192 }
193 }
194
195 void MovementInform(uint32 type, uint32 id) override
196 {
197 if (type == POINT_MOTION_TYPE && id == POINT_MIDDLE)
198 {
199 me->SetCanFly(false);
200 me->SetDisableGravity(false);
201 events.ScheduleEvent(EVENT_RESTORE_COMBAT, 0);
202 events.ScheduleEvent(EVENT_SPELL_CONE_OF_FIRE, 5000);
203 if (IsHeroic())
204 events.ScheduleEvent(EVENT_SPELL_BELLOWING_ROAR, 10000);
205 }
206 }
207
208 void UpdateAI(uint32 diff) override
209 {
210 if (!UpdateVictim())
211 return;
212
213 events.Update(diff);
214 if (me->HasUnitState(UNIT_STATE_CASTING))
215 return;
216
217 switch (events.ExecuteEvent())
218 {
219 case EVENT_SPELL_FIREBALL:
220 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
221 me->CastSpell(target, SPELL_FIREBALL, false);
222 events.ScheduleEvent(EVENT_SPELL_FIREBALL, urand(4000, 6000));
223 break;
224 case EVENT_CHANGE_POS:
225 me->GetMotionMaster()->MovePoint(POINT_FLIGHT, NazanPos[urand(0, 2)], false);
226 events.DelayEvents(7000);
227 events.ScheduleEvent(EVENT_CHANGE_POS, 30000);
228 break;
229 case EVENT_RESTORE_COMBAT:
230 me->GetMotionMaster()->MoveChase(me->GetVictim());
231 break;
232 case EVENT_SPELL_CONE_OF_FIRE:
233 me->CastSpell(me->GetVictim(), SPELL_CONE_OF_FIRE, false);
234 events.ScheduleEvent(EVENT_SPELL_CONE_OF_FIRE, 12000);
235 break;
236 case EVENT_SPELL_BELLOWING_ROAR:
237 me->CastSpell(me, SPELL_BELLOWING_ROAR, false);
238 events.ScheduleEvent(EVENT_SPELL_BELLOWING_ROAR, 30000);
239 break;
240 }
241
242 if (!me->IsLevitating())
243 DoMeleeAttackIfReady();
244 }
245
246 private:
247 EventMap events;
248 };
249
250 CreatureAI* GetAI(Creature* creature) const override
251 {
252 return GetHellfireRampartsAI<boss_nazanAI>(creature);
253 }
