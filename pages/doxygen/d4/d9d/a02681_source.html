153 brazier->Respawn();
154 }
155 _kirtonosState = NOT_STARTED;
156 break;
157 case DONE:
158 // open door
159 if (GameObject* gate = instance->GetGameObject(GetGuidData(GO_GATE_KIRTONOS)))
160 {
161 gate->SetGoState(GO_STATE_ACTIVE);
162 }
163 [[fallthrough]];
164 default:
165 _kirtonosState = data;
166 break;
167 }
168 break;
169 case DATA_MINI_BOSSES:
170 ++_miniBosses;
171 if (_miniBosses == 6)
172 {
173 if (Creature* Gandling = instance->GetCreature(GandlingGUID))
174 {
175 Gandling->AI()->Talk(0);
176 Gandling->AI()->Reset();
177 }
178 }
179 break;
180 case DATA_DARKMASTER_GANDLING:
181 switch (data)
182 {
183 case DONE:
184 case NOT_STARTED:
185 case FAIL:
186 HandleGameObject(GandlingGatesGUID[6], true);
187 break;
188 case IN_PROGRESS:
189 HandleGameObject(GandlingGatesGUID[6], false);
190 break;
191 }
192 break;
193 case DATA_RAS_HUMAN:
194 _rasHuman = data;
195 break;
196 }
197 SaveToDB();
