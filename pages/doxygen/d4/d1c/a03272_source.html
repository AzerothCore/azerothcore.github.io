34 bool OnGossipHello(Player* player, Creature* creature) override
35 {
36 if (!creature->HasNpcFlag(UNIT_NPC_FLAG_GOSSIP))
37 return true;
38
39 InstanceScript* pInstance = creature->GetInstanceScript();
40 if (!pInstance)
41 return true;
42
43 uint32 gossipTextId = 0;
44 switch (pInstance->GetData(DATA_INSTANCE_PROGRESS))
45 {
46 case INSTANCE_PROGRESS_INITIAL:
47 if (!player->GetVehicle())
48 {
49 if (pInstance->GetData(DATA_TEAMID_IN_INSTANCE) == TEAM_HORDE)
50 gossipTextId = 15043; //Horde text
51 else
52 gossipTextId = 14757; //Alliance text
53 }
54 else
55 {
56 gossipTextId = 14688;
57 AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_START_EVENT1a, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1338);
58 AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_START_EVENT1b, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1341);
59 }
60 break;
61 case INSTANCE_PROGRESS_CHAMPIONS_DEAD:
62 gossipTextId = 14737;
63 AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_START_EVENT2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1339);
64 break;
65 case INSTANCE_PROGRESS_ARGENT_CHALLENGE_DIED:
66 gossipTextId = 14738;
67 AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_START_EVENT3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1340);
68 break;
69 default:
70 return true;
71 }
72
73 SendGossipMenuFor(player, gossipTextId, creature->GetGUID());
74 return true;
75 }

77 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*uiSender*/, uint32 uiAction) override
78 {
79 if(!creature->HasNpcFlag(UNIT_NPC_FLAG_GOSSIP))
80 return true;
81
82 InstanceScript* pInstance = creature->GetInstanceScript();
83 if(!pInstance)
84 return true;
85
86 if(uiAction == GOSSIP_ACTION_INFO_DEF + 1338 || uiAction == GOSSIP_ACTION_INFO_DEF + 1341 || uiAction == GOSSIP_ACTION_INFO_DEF + 1339 || uiAction == GOSSIP_ACTION_INFO_DEF + 1340)
87 {
88 pInstance->SetData(DATA_ANNOUNCER_GOSSIP_SELECT, (uiAction == GOSSIP_ACTION_INFO_DEF + 1341 ? 1 : 0));
89 creature->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP);
90 }
91
92 CloseGossipMenuFor(player);
93 return true;
94 }

119 }

121 void MovementInform(uint32 type, uint32 /*id*/) override
122 {
