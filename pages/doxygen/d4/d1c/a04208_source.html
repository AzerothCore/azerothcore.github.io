46class item_only_for_flight : public ItemScript
47{
48public:
49 item_only_for_flight() : ItemScript("item_only_for_flight") { }
50
51 bool OnUse(Player* player, Item* item, SpellCastTargets const& /*targets*/) override
52 {
53 uint32 itemId = item->GetEntry();
54 bool disabled = false;
55
56 //for special scripts
57 switch (itemId)
58 {
59 case 24538:
60 if (player->GetAreaId() != 3628)
61 disabled = true;
62 break;
63 case 34489:
64 if (player->GetZoneId() != 4080)
65 disabled = true;
66 break;
67 case 34475:
68 if (SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(SPELL_ARCANE_CHARGES))
69 Spell::SendCastResult(player, spellInfo, 1, SPELL_FAILED_NOT_ON_GROUND);
70 break;
71 }
72
73 // allow use in flight only
74 if (player->IsInFlight() && !disabled)
75 return false;
76
77 // error
78 player->SendEquipError(EQUIP_ERR_CANT_DO_RIGHT_NOW, item, nullptr);
79 return true;
80 }
81};

250class item_generic_limit_chance_above_60 : public ItemScript
251{
252public:
253 item_generic_limit_chance_above_60() : ItemScript("item_generic_limit_chance_above_60") { }
254
255 bool OnCastItemCombatSpell(Player* /*player*/, Unit* victim, SpellInfo const* /*spellInfo*/, Item* /*item*/) override
256 {
257 // spell proc chance gets severely reduced on victims > 60 (formula unknown)
258 if (victim->getLevel() > 60)
259 {
260 // gives ~0.1% proc chance at lvl 70
261 float const lvlPenaltyFactor = 9.93f;
262 float const failureChance = (victim->getLevel() - 60) * lvlPenaltyFactor;
263
264 // base ppm chance was already rolled, only roll success chance
265 return !roll_chance_f(failureChance);
266 }
267
268 return true;
269 }
