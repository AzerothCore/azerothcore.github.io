89/*
90 * ---
91 * --- Emerald Dragons : Base AI-structure used for all the Emerald dragons
92 * ---
93 */
94
95struct emerald_dragonAI : public WorldBossAI
96{
97 emerald_dragonAI(Creature* creature) : WorldBossAI(creature)
98 {
99 }
100
101 void Reset() override
102 {
103 WorldBossAI::Reset();
104 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
105 me->SetReactState(REACT_AGGRESSIVE);
106 DoCast(me, SPELL_MARK_OF_NATURE_AURA, true);
107 events.ScheduleEvent(EVENT_TAIL_SWEEP, 4000);
108 events.ScheduleEvent(EVENT_NOXIOUS_BREATH, urand(7500, 15000));
109 events.ScheduleEvent(EVENT_SEEPING_FOG, urand(12500, 20000));
110 events.ScheduleEvent(EVENT_SUMMON_PLAYER, 1s);
111 }
112
113 // Target killed during encounter, mark them as suspectible for Aura Of Nature
114 void KilledUnit(Unit* who) override
115 {
116 if (who->GetTypeId() == TYPEID_PLAYER)
117 who->CastSpell(who, SPELL_MARK_OF_NATURE, true);
118 }
119
120 // Execute and reschedule base events shared between all Emerald Dragons
121 void ExecuteEvent(uint32 eventId) override
122 {
123 switch (eventId)
124 {
125 case EVENT_SEEPING_FOG:
126 // Seeping Fog appears only as "pairs", and only ONE pair at any given time!
127 // Despawntime is 2 minutes, so reschedule it for new cast after 2 minutes + a minor "random time" (30 seconds at max)
128 DoCast(me, SPELL_SEEPING_FOG_LEFT, true);
129 DoCast(me, SPELL_SEEPING_FOG_RIGHT, true);
130 events.ScheduleEvent(EVENT_SEEPING_FOG, urand(120000, 150000));
131 break;
132 case EVENT_NOXIOUS_BREATH:
133 // Noxious Breath is cast on random intervals, no less than 7.5 seconds between
134 DoCast(me, SPELL_NOXIOUS_BREATH);
135 events.ScheduleEvent(EVENT_NOXIOUS_BREATH, urand(7500, 15000));
136 break;
137 case EVENT_TAIL_SWEEP:
138 // Tail Sweep is cast every two seconds, no matter what goes on in front of the dragon
139 DoCast(me, SPELL_TAIL_SWEEP);
140 events.ScheduleEvent(EVENT_TAIL_SWEEP, 2000);
141 break;
142 case EVENT_SUMMON_PLAYER:
143 if (Unit* target = me->GetVictim())
144 if (!target->IsWithinRange(me, 50.f))
145 DoCast(target, SPELL_SUMMON_PLAYER);
146 events.ScheduleEvent(EVENT_SUMMON_PLAYER, 500ms);
147 break;
148 }
149 }
150
151 void JustSummoned(Creature* summon) override
152 {
153 if (summon->GetEntry() == NPC_DREAM_FOG)
154 {
155 summon->AI()->SetGUID(me->GetGUID(), GUID_DRAGON);
156 }
157 }
158
159 void UpdateAI(uint32 diff) override
160 {
161 if (!UpdateVictim())
162 return;
163
164 events.Update(diff);
165
166 if (me->HasUnitState(UNIT_STATE_CASTING))
167 return;
168
169 while (uint32 eventId = events.ExecuteEvent())
170 ExecuteEvent(eventId);
171
172 DoMeleeAttackIfReady();
173 }
174};

179
180class npc_dream_fog : public CreatureScript
181{
182public:
183 npc_dream_fog() : CreatureScript("npc_dream_fog") { }
184
185 struct npc_dream_fogAI : public ScriptedAI
186 {
187 npc_dream_fogAI(Creature* creature) : ScriptedAI(creature) { }
188
189 void Reset() override
190 {
191 ScheduleEvents();
192 }
193
194 void ScheduleEvents()
195 {
196 _scheduler.CancelAll();
197
198 _scheduler.Schedule(1s, [this](TaskContext context)
199 {
200 // Chase target, but don't attack - otherwise just roam around
201 if (Unit* chaseTarget = GetRandomUnitFromDragonThreatList())
202 {
203 me->GetMotionMaster()->Clear();
204 me->GetMotionMaster()->MoveFollow(chaseTarget, 0.02f, 0.0f);
205 _targetGUID = chaseTarget->GetGUID();
206 context.Repeat(15s, 30s);
207 }
208 else
209 {
210 me->GetMotionMaster()->Clear();
211 me->GetMotionMaster()->MoveRandom(25.0f);
212 context.Repeat(2500ms);
213 }
214
215 // Seeping fog movement is slow enough for a player to be able to walk backwards and still outpace it
216 me->SetWalk(true);
217 me->SetSpeed(MOVE_WALK, 0.75f);
218 });
219 }
220
221 void SetGUID(ObjectGuid guid, int32 type) override
222 {
223 if (type == GUID_DRAGON)
224 {
225 _dragonGUID = guid;
226 }
227 else if (type == GUID_FOG_TARGET)
228 {
229 if (guid == _targetGUID)
230 {
231 ScheduleEvents();
232 }
233 }
234 }
235
236 Unit* GetRandomUnitFromDragonThreatList()
237 {
238 if (Creature* dragon = ObjectAccessor::GetCreature(*me, _dragonGUID))
239 {
240 if (dragon->GetAI())
241 {
242 return dragon->GetAI()->SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true);
243 }
244 }
245
246 return nullptr;
247 }
248
249 void UpdateAI(uint32 diff) override
250 {
251 if (!UpdateVictim())
252 return;
253
254 _scheduler.Update(diff);
255 }
256
257 private:
258 ObjectGuid _targetGUID;
259 ObjectGuid _dragonGUID;
260 TaskScheduler _scheduler;
261 };
262
263 CreatureAI* GetAI(Creature* creature) const override
264 {
265 return new npc_dream_fogAI(creature);
266 }
267};

603class boss_taerar : public CreatureScript
604{
605public:
606 boss_taerar() : CreatureScript("boss_taerar") { }
607
608 struct boss_taerarAI : public emerald_dragonAI
609 {
610 boss_taerarAI(Creature* creature) : emerald_dragonAI(creature)
611 {
612 }
613
614 void Reset() override
615 {
616 me->RemoveAurasDueToSpell(SPELL_SHADE);
617 _stage = 1;
618
619 _shades = 0;
620 _banished = false;
621 _banishedTimer = 0;
622
623 emerald_dragonAI::Reset();
624 events.ScheduleEvent(EVENT_ARCANE_BLAST, 12000);
625 events.ScheduleEvent(EVENT_BELLOWING_ROAR, 30000);
626 }
627
628 void EnterCombat(Unit* who) override
629 {
630 Talk(SAY_TAERAR_AGGRO);
631 emerald_dragonAI::EnterCombat(who);
632 }
633
634 void SummonedCreatureDies(Creature* /*summon*/, Unit*) override
635 {
636 --_shades;
637 }
638
639 void DamageTaken(Unit*, uint32& damage, DamageEffectType, SpellSchoolMask) override
640 {
641 // At 75, 50 or 25 percent health, we need to activate the shades and go "banished"
642 // Note: _stage holds the amount of times they have been summoned
643 if (!_banished && me->HealthBelowPctDamaged(100 - (25 * _stage), damage))
644 {
645 _banished = true;
646 _banishedTimer = 60000;
647
648 me->InterruptNonMeleeSpells(false);
649 DoStopAttack();
650
651 Talk(SAY_TAERAR_SUMMON_SHADES);
652
653 uint32 count = sizeof(TaerarShadeSpells) / sizeof(uint32);
654 for (uint32 i = 0; i < count; ++i)
655 DoCast(TaerarShadeSpells[i]);
656 _shades += count;
657 DoCast(SPELL_SHADE);
658 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
659 me->SetReactState(REACT_PASSIVE);
660
661 ++_stage;
662 }
663 }
664
665 void ExecuteEvent(uint32 eventId) override
666 {
667 switch (eventId)
668 {
669 case EVENT_ARCANE_BLAST:
670 DoCast(SPELL_ARCANE_BLAST);
671 events.ScheduleEvent(EVENT_ARCANE_BLAST, urand(7000, 12000));
672 break;
673 case EVENT_BELLOWING_ROAR:
674 DoCast(SPELL_BELLOWING_ROAR);
675 events.ScheduleEvent(EVENT_BELLOWING_ROAR, urand(20000, 30000));
676 break;
677 default:
678 emerald_dragonAI::ExecuteEvent(eventId);
679 break;
680 }
681 }
682
683 void UpdateAI(uint32 diff) override
684 {
685 if (!me->IsInCombat())
686 return;
687
688 if (_banished)
689 {
690 // If all three shades are dead, OR it has taken too long, end the current event and get Taerar back into business
691 if (_banishedTimer <= diff || !_shades)
692 {
693 _banished = false;
694
695 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
696 me->RemoveAurasDueToSpell(SPELL_SHADE);
697 me->SetReactState(REACT_AGGRESSIVE);
698 }
699 // _banishtimer has not expired, and we still have active shades:
700 else
701 _banishedTimer -= diff;
702
703 // Update the events before we return (handled under emerald_dragonAI::UpdateAI(diff); if we're not inside this check)
704 events.Update(diff);
705
706 return;
707 }
708 emerald_dragonAI::UpdateAI(diff);
709 }
710
711 void JustDied(Unit* /*killer*/) override
712 {
713 _JustDied();
714 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
715 }
716
717 private:
718 bool _banished; // used for shades activation testing
719 uint32 _banishedTimer; // counter for banishment timeout
720 uint8 _shades; // keep track of how many shades are dead
721 uint8 _stage; // check which "shade phase" we're at (75-50-25 percentage counters)
722 };
723
724 CreatureAI* GetAI(Creature* creature) const override
725 {
726 return new boss_taerarAI(creature);
727 }
