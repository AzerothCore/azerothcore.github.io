86 void Load(const char* in) override
87 {

90
91 char dataHead1, dataHead2;
92 std::istringstream loadStream(in);
93 loadStream >> dataHead1 >> dataHead2;
94 if (dataHead1 == 'D' && dataHead2 == 'E')
95 {
96 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
97 {
98 loadStream >> _encounters[i];

102 }
103 }
