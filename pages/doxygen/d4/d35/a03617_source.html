300 void UpdateAI(uint32 diff) override
301 {
302 if (!UpdateVictim())
303 return;
304
305 events.Update(diff);
306 if (me->HasUnitState(UNIT_STATE_CASTING))
307 return;
308
309 switch(events.ExecuteEvent())
310 {
311 // Control events
312 case EVENT_HEALTH_CHECK:
313 if (_hardMode)
314 {
315 return;
316 }
317
318 if (me->HealthBelowPct(_healthCheck))
319 {
320 _healthCheck -= 25;
321 me->SetControlled(true, UNIT_STATE_STUNNED);
322 me->SetByteValue(UNIT_FIELD_BYTES_1, UNIT_BYTES_1_OFFSET_STAND_STATE, UNIT_STAND_STATE_SUBMERGED); // submerge with animation
323
324 Talk(SAY_HEART_OPENED);
325
326 events.CancelEventGroup(1);
327 events.ScheduleEvent(EVENT_START_SECOND_PHASE, 5000);
328 return;
329 }
330 events.RepeatEvent(1000);
331 break;
332 case EVENT_CHECK_ROOM:
333 events.RepeatEvent(5000);
334 if (me->GetPositionX() < 722 || me->GetPositionX() > 987 || me->GetPositionY() < -139 || me->GetPositionY() > 124)
335 EnterEvadeMode();
336
337 return;
338
339 // Abilities events
340 case EVENT_GRAVITY_BOMB:
341 me->CastCustomSpell(SPELL_GRAVITY_BOMB, SPELLVALUE_MAX_TARGETS, 1, me, true);
342 events.ScheduleEvent(EVENT_SEARING_LIGHT, 10000, 1);
343 break;
344 case EVENT_SEARING_LIGHT:
345 me->CastCustomSpell(SPELL_SEARING_LIGHT, SPELLVALUE_MAX_TARGETS, 1, me, true);
346 events.ScheduleEvent(EVENT_GRAVITY_BOMB, 10000, 1);
347 break;
348 case EVENT_TYMPANIC_TANTARUM:
349 Talk(EMOTE_TYMPANIC_TANTRUM);
350 Talk(SAY_TYMPANIC_TANTRUM);
351 me->CastSpell(me, SPELL_TYMPANIC_TANTARUM, true);
352 events.RepeatEvent(60000);
353 return;
354 case EVENT_ENRAGE:
355 Talk(SAY_BERSERK);
356 me->CastSpell(me, SPELL_XT002_ENRAGE, true);
357 break;
358
359 // Animation events
360 case EVENT_START_SECOND_PHASE:
361 Talk(EMOTE_HEART_OPENED);
362 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
363 if (Unit* heart = me->GetVehicleKit() ? me->GetVehicleKit()->GetPassenger(HEART_VEHICLE_SEAT) : nullptr)
364 heart->GetAI()->DoAction(ACTION_AWAKEN_HEART);
365
366 events.ScheduleEvent(EVENT_RESTORE, 30000);
367 return;
368 // Restore from heartbreak
369 case EVENT_RESTORE:
370 if (_hardMode)
371 {
372 return;
373 }
374
375 Talk(SAY_HEART_CLOSED);
376
377 me->SetByteValue(UNIT_FIELD_BYTES_1, UNIT_BYTES_1_OFFSET_STAND_STATE, UNIT_STAND_STATE_STAND); // emerge
378 // Hide heart
379 if (Unit* heart = me->GetVehicleKit() ? me->GetVehicleKit()->GetPassenger(HEART_VEHICLE_SEAT) : nullptr)
380 heart->GetAI()->DoAction(ACTION_HIDE_HEART);
381
382 events.ScheduleEvent(EVENT_REMOVE_EMOTE, 4000);
383 return;
384 case EVENT_REMOVE_EMOTE:
385 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
386 me->SetControlled(false, UNIT_STATE_STUNNED);
387
388 RescheduleEvents();
389 return;
390 }
391
392 // Disabled by stunned state
393 DoMeleeAttackIfReady();
394 }
