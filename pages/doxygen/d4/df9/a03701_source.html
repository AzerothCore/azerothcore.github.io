83 void UpdateAI(uint32 diff) override
84 {
85 if (!UpdateVictim())
86 return;
87
88 events.Update(diff);
89
90 if (me->HasUnitState(UNIT_STATE_CASTING))
91 return;
92
93 switch(events.ExecuteEvent())
94 {
95 case 0:
96 break;
97 case EVENT_SPELL_CORROSIVE_SALIVA:
98 me->CastSpell(me->GetVictim(), SPELL_CORROSIVE_SALIVA, false);
99 events.RepeatEvent(urand(8000, 10000));
100 break;
101 case EVENT_SPELL_OPTIC_LINK:
102 if (Unit* target = SelectTarget(SelectTargetMethod::MinDistance, 0, 40.0f, true))
103 {
104 me->CastSpell(target, SPELL_OPTIC_LINK, false);
105 events.RepeatEvent(urand(18000, 21000));
106 }
107 else
108 events.RepeatEvent(5000);
109 break;
110 }
111
112 DoMeleeAttackIfReady();
113 }
114
