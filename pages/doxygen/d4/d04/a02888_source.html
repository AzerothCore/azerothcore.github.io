134ObjectGuid getOrphanGUID(Player* player, uint32 orphan)
135{
136 if (Aura* orphanOut = player->GetAura(SPELL_ORPHAN_OUT))
137 if (orphanOut->GetCaster() && orphanOut->GetCaster()->GetEntry() == orphan)
138 return orphanOut->GetCaster()->GetGUID();
139
140 return ObjectGuid::Empty;
141}

143/*######
144## npc_winterfin_playmate
145######*/
146struct npc_winterfin_playmate : public ScriptedAI
147{
148 npc_winterfin_playmate(Creature* creature) : ScriptedAI(creature) { }
149
150 void Reset() override
151 {
152 timer = 0;
153 phase = 0;
154 playerGUID.Clear();
155 orphanGUID.Clear();
156 }
157
158 void MoveInLineOfSight(Unit* who) override
159 {
160 if (!phase && who && who->GetDistance2d(me) < 10.0f)
161 if (Player* player = who->ToPlayer())
162 if (player->GetQuestStatus(QUEST_PLAYMATE_ORACLE) == QUEST_STATUS_INCOMPLETE)
163 {
164 playerGUID = player->GetGUID();
165 orphanGUID = getOrphanGUID(player, ORPHAN_ORACLE);
166 if (orphanGUID)
167 phase = 1;
168 }
169 }
170
171 void UpdateAI(uint32 diff) override
172 {
173 if (!phase)
174 return;
175
176 if (timer <= diff)
177 {
178 Player* player = ObjectAccessor::GetPlayer(*me, playerGUID);
179 Creature* orphan = ObjectAccessor::GetCreature(*me, orphanGUID);
180
181 if (!orphan || !player)
182 {
183 Reset();
184 return;
185 }
186
187 switch (phase)
188 {
189 case 1:
190 orphan->GetMotionMaster()->MovePoint(0, me->GetPositionX() + cos(me->GetOrientation()) * 5, me->GetPositionY() + std::sin(me->GetOrientation()) * 5, me->GetPositionZ());
191 orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_1);
192 timer = 3000;
193 break;
194 case 2:
195 orphan->SetFacingToObject(me);
196 Talk(TEXT_WINTERFIN_PLAYMATE_1);
197 me->HandleEmoteCommand(EMOTE_STATE_DANCE);
198 timer = 3000;
199 break;
200 case 3:
201 orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_2);
202 timer = 3000;
203 break;
204 case 4:
205 Talk(TEXT_WINTERFIN_PLAYMATE_2);
206 timer = 5000;
207 break;
208 case 5:
209 orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_3);
210 me->HandleEmoteCommand(EMOTE_STATE_NONE);
211 player->GroupEventHappens(QUEST_PLAYMATE_ORACLE, me);
212 orphan->GetMotionMaster()->MoveFollow(player, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
213 Reset();
214 return;
215 }
216 ++phase;
217 }
218 else
219 timer -= diff;
220 }
221
222private:
223 uint32 timer;
224 int8 phase;
225 ObjectGuid playerGUID;
226 ObjectGuid orphanGUID;
227};
