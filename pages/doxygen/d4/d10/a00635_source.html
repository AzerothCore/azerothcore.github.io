85bool PreparedStatementTask::Execute()
86{
87 if (m_has_result)
88 {
89 PreparedResultSet* result = m_conn->Query(m_stmt);
90 if (!result || !result->GetRowCount())
91 {
92 delete result;
93 m_result->set_value(PreparedQueryResult(nullptr));
94 return false;
95 }
96
97 m_result->set_value(PreparedQueryResult(result));
98 return true;
99 }
100
101 return m_conn->Execute(m_stmt);
102}
