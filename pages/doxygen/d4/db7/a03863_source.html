98 struct boss_shade_of_akamaAI : public BossAI

100 boss_shade_of_akamaAI(Creature* creature) : BossAI(creature, DATA_SHADE_OF_AKAMA), summonsChanneler(me), summonsGenerator(me)

106 SummonList summonsGenerator;
107 EventMap events2;
108
109 void ChannelersAction(int32 action)
110 {
111 for (SummonList::const_iterator i = summonsChanneler.begin(); i != summonsChanneler.end(); ++i)
112 if (Creature* summon = ObjectAccessor::GetCreature(*me, *i))
113 {
114 if (action == ACTION_CHANNELERS_START_CHANNEL)
115 {
116 summon->CastSpell(me, SPELL_SHADE_SOUL_CHANNEL, true);
117 summon->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
118 }
119 else if (action == ACTION_START_ENCOUNTER)
120 {
121 summon->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
122 }
123 else if (action == ACTION_KILL_CHANNELERS)
124 {
125 Unit::Kill(me, summon);
126 }
127 }
128 }
129
130 void Reset() override
131 {
132 BossAI::Reset();

134 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
135 me->SetImmuneToAll(true);
136 me->SetWalk(true);
137 }
138
139 void EnterEvadeMode(EvadeReason why) override
140 {
141 BossAI::EnterEvadeMode(why);
142 summonsGenerator.DoAction(ACTION_DESPAWN_ALL);
143 events2.ScheduleEvent(EVENT_SHADE_RESET_ENCOUNTER, 20000);
144 me->SetVisible(false);
145 ChannelersAction(ACTION_KILL_CHANNELERS);
146 }
147
148 void JustDied(Unit* killer) override
149 {

151 summonsGenerator.DoAction(ACTION_DESPAWN_ALL);

156 akama->SetHomePosition(*akama);
157 akama->AI()->DoAction(ACTION_SHADE_DIED);

159 }
160
161 void EnterCombat(Unit* who) override
162 {
163 BossAI::EnterCombat(who);
164 }
165
166 void DoAction(int32 param) override
167 {
168 if (param == ACTION_START_ENCOUNTER)
169 {
170 summonsGenerator.DoAction(ACTION_START_ENCOUNTER);
171 ChannelersAction(ACTION_START_ENCOUNTER);
172 events.ScheduleEvent(EVENT_SHADE_CHECK_DISTANCE, 1000);
173 }
174 else if (param == ACTION_AKAMA_DIED)
175 {
176 EnterEvadeMode(EVADE_REASON_OTHER);
177 }
178 }
179
180 void UpdateAI(uint32 diff) override
181 {
182 events2.Update(diff);
183 switch (events2.ExecuteEvent())
184 {
185 case EVENT_SHADE_GATHER_NPCS:
186 {
187 std::list<Creature*> ChannelerList;
188 me->GetCreaturesWithEntryInRange(ChannelerList, 100.0f, NPC_ASHTONGUE_CHANNELER);
189 for (std::list<Creature*>::const_iterator itr = ChannelerList.begin(); itr != ChannelerList.end(); ++itr)
190 summonsChanneler.Summon(*itr);
191
192 std::list<Creature*> SpawnerList;
193 me->GetCreaturesWithEntryInRange(SpawnerList, 100.0f, NPC_CREATURE_GENERATOR_AKAMA);
194 for (std::list<Creature*>::const_iterator itr = SpawnerList.begin(); itr != SpawnerList.end(); ++itr)
195 summonsGenerator.Summon(*itr);
196
197 summonsChanneler.Respawn();
198 summonsGenerator.Respawn();
199 ChannelersAction(ACTION_CHANNELERS_START_CHANNEL);
200
201 if (Creature* akama = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_AKAMA_SHADE)))
202 akama->Respawn(true);
203 break;
204 }
205 case EVENT_SHADE_RESET_ENCOUNTER:
206 me->SetVisible(true);
207 summonsGenerator.Respawn();
208 summonsChanneler.Respawn();
209 ChannelersAction(ACTION_CHANNELERS_START_CHANNEL);
210
211 if (Creature* akama = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_AKAMA_SHADE)))
212 akama->Respawn(true);
213 break;
214 }
215
216 if (!UpdateVictim())
217 return;
218
219 events.Update(diff);
220 if (me->HasUnitState(UNIT_STATE_CASTING))
221 return;
222
223 switch (events.ExecuteEvent())
224 {
225 case EVENT_SHADE_CHECK_DISTANCE:
226 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() != POINT_MOTION_TYPE)
227 {
228 int32 slow = me->GetMaxNegativeAuraModifier(SPELL_AURA_MOD_DECREASE_SPEED);
229 if (slow > -100)
230 {
231 me->SetWalk(true);
232 me->GetMotionMaster()->MovePoint(POINT_START, 510.0f, 400.7993f, 112.7837f);
233 }
234 }
235 else
236 {
237 int32 slow = me->GetMaxNegativeAuraModifier(SPELL_AURA_MOD_DECREASE_SPEED);
238 if (slow < -100)
239 me->GetMotionMaster()->Clear();
240 else if (slow == 0)
241 {
242 summonsGenerator.DoAction(ACTION_NO_SORCERERS);
243 me->SetWalk(false);
244 }
245 }
246
247 if (me->IsWithinMeleeRange(me->GetVictim()))
248 {
249 me->SetReactState(REACT_AGGRESSIVE);
250 DoResetThreat();
251 me->GetVictim()->InterruptNonMeleeSpells(false);
252 me->AddThreat(me->GetVictim(), 1000000.0f);
253 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
254 me->SetImmuneToAll(false);
255 summonsGenerator.DoAction(ACTION_STOP_SPAWNING);
256 break;
257 }
258 events.ScheduleEvent(EVENT_SHADE_CHECK_DISTANCE, 1000);
259 break;
260 }
261
262 DoMeleeAttackIfReady();
263 }

268 }
