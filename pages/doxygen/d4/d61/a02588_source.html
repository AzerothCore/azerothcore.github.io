87 void Load(const char* in) override
88 {
89 if (!in)
90 return;
91
92 char dataHead1, dataHead2;
93 std::istringstream loadStream(in);
94 loadStream >> dataHead1 >> dataHead2;
95 if (dataHead1 == 'D' && dataHead2 == 'E')
96 {
97 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
98 {
99 loadStream >> _encounters[i];
100 if (_encounters[i] == IN_PROGRESS)
101 _encounters[i] = NOT_STARTED;
102 }
103 }
