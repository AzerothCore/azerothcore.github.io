168 void Load(char const* str) override
169 {
170 if (!str)
171 {
172 OUT_LOAD_INST_DATA_FAIL;
173 return;
174 }
175
176 OUT_LOAD_INST_DATA(str);
177
178 char dataHead1, dataHead2;
179
180 std::istringstream loadStream(str);
181 loadStream >> dataHead1 >> dataHead2;
182
183 if (dataHead1 == 'A' && dataHead2 == 'Z')
184 {
185 for (uint32 i = 0; i < MAX_ENCOUTER; ++i)
186 {
187 uint32 tmpState;
188 loadStream >> tmpState;
189 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
190 tmpState = NOT_STARTED;
191 SetBossState(i, EncounterState(tmpState));
192 }
193 }
194 else
195 OUT_LOAD_INST_DATA_FAIL;
196
197 OUT_LOAD_INST_DATA_COMPLETE;
198 }
