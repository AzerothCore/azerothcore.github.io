172 void MovementInform(uint32 type, uint32 id) override
173 {
174 if (type != POINT_MOTION_TYPE)
175 {
176 if (type == ESCORT_MOTION_TYPE && me->movespline->Finalized() && !me->IsInCombat())
177 startPath = true;
178 return;
179 }
180
181 if (id == POINT_PLATFORM)
182 me->setAttackTimer(BASE_ATTACK, 1000);
183 else if (id == POINT_QUILL)
184 events.ScheduleEvent(EVENT_START_QUILLS, 1000);
185 else if (id == POINT_DIVE)
186 {
187 events.ScheduleEvent(EVENT_START_DIVE, 1000);
188 events.ScheduleEvent(EVENT_CAST_DIVE_BOMB, 5000);
189 }

192 void UpdateAI(uint32 diff) override
193 {
194 if (startPath)
195 {
196 me->StopMoving();
197 startPath = false;
198 if (WaypointPath const* i_path = sWaypointMgr->GetPath(me->GetWaypointPath()))
199 {
200 Movement::PointsArray pathPoints;
201 pathPoints.push_back(G3D::Vector3(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ()));
202 for (uint8 i = 0; i < i_path->size(); ++i)
203 {
204 WaypointData const* node = i_path->at(i);
205 pathPoints.push_back(G3D::Vector3(node->x, node->y, node->z));
206 }
207 me->GetMotionMaster()->MoveSplinePath(&pathPoints);
208 }
209 }
210
211 if (!UpdateVictim())
212 return;
213
214 events.Update(diff);
215 if (me->HasUnitState(UNIT_STATE_CASTING))
216 return;
217
218 switch (events.ExecuteEvent())
219 {
220 case EVENT_SWITCH_PLATFORM:
221 if (roll_chance_i(20 * noQuillTimes))
222 {
223 noQuillTimes = 0;
224 platform = RAND(0, 3);
225 me->GetMotionMaster()->MovePoint(POINT_QUILL, alarPoints[POINT_QUILL], false, true);
226 events.ScheduleEvent(EVENT_SWITCH_PLATFORM, 16000);
227 }
228 else
229 {
230 if (noQuillTimes++ > 0)
231 {
232 me->SetOrientation(alarPoints[platform].GetOrientation());
233 me->SummonCreature(NPC_EMBER_OF_ALAR, *me, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 6000);
234 }
235 me->GetMotionMaster()->MovePoint(POINT_PLATFORM, alarPoints[platform], false, true);
236 platform = (platform + 1) % 4;
237 events.ScheduleEvent(EVENT_SWITCH_PLATFORM, 30000);
238 }
239 me->setAttackTimer(BASE_ATTACK, 20000);
240 break;
241 case EVENT_START_QUILLS:
242 me->CastSpell(me, SPELL_FLAME_QUILLS, false);
243 break;
244 case EVENT_RELOCATE_MIDDLE:
245 me->SetPosition(alarPoints[POINT_MIDDLE]);
246 break;
247 case EVENT_MOVE_TO_PHASE_2:
248 me->RemoveAurasDueToSpell(SPELL_EMBER_BLAST);
249 me->CastSpell(me, SPELL_REBIRTH_PHASE2, false);
250 break;
251 case EVENT_REBIRTH:
252 me->SetReactState(REACT_AGGRESSIVE);
253 platform = POINT_MIDDLE;
254 me->GetMotionMaster()->MoveChase(me->GetVictim());
255
256 events.ScheduleEvent(EVENT_SPELL_MELT_ARMOR, 67000);
257 events.ScheduleEvent(EVENT_SPELL_CHARGE, 10000);
258 events.ScheduleEvent(EVENT_SPELL_FLAME_PATCH, 20000);
259 events.ScheduleEvent(EVENT_SPELL_DIVE_BOMB, 30000);
260 break;
261 case EVENT_SPELL_MELT_ARMOR:
262 me->CastSpell(me->GetVictim(), SPELL_MELT_ARMOR, false);
263 events.ScheduleEvent(EVENT_SPELL_MELT_ARMOR, 60000);
264 break;
265 case EVENT_SPELL_CHARGE:
266 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
267 me->CastSpell(target, SPELL_CHARGE, false);
268 events.ScheduleEvent(EVENT_SPELL_CHARGE, 30000);
269 break;
270 case EVENT_SPELL_FLAME_PATCH:
271 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
272 me->SummonCreature(NPC_FLAME_PATCH, *target, TEMPSUMMON_TIMED_DESPAWN, 2 * MINUTE * IN_MILLISECONDS);
273 events.ScheduleEvent(EVENT_SPELL_FLAME_PATCH, 30000);
274 break;
275 case EVENT_SPELL_DIVE_BOMB:
276 me->GetMotionMaster()->MovePoint(POINT_DIVE, alarPoints[POINT_DIVE], false, true);
277 events.ScheduleEvent(EVENT_SPELL_DIVE_BOMB, 30000);
278 events.DelayEvents(15000);
279 me->setAttackTimer(BASE_ATTACK, 20000);
280 break;
281 case EVENT_START_DIVE:
282 me->CastSpell(me, SPELL_DIVE_BOMB_VISUAL, false);
283 break;
284 case EVENT_CAST_DIVE_BOMB:
285 events.ScheduleEvent(EVENT_SUMMON_DIVE_PHOENIX, 2000);
286 events.ScheduleEvent(EVENT_REBIRTH_DIVE, 6000);
287 events.ScheduleEvent(EVENT_FINISH_DIVE, 10000);
288 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 90.0f, true))
289 {
290 me->CastSpell(target, SPELL_DIVE_BOMB, false);
291 me->SetPosition(*target);
292 me->StopMovingOnCurrentPos();
293 }
294
295 me->RemoveAurasDueToSpell(SPELL_DIVE_BOMB_VISUAL);
296 break;
297 case EVENT_SUMMON_DIVE_PHOENIX:
298 {
299 Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 10.0f, true);
300 me->SummonCreature(NPC_EMBER_OF_ALAR, target ? *target : *me, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 6000);
301 me->SummonCreature(NPC_EMBER_OF_ALAR, target ? *target : *me, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 6000);
302 break;
303 }
304 case EVENT_REBIRTH_DIVE:
305 me->SetModelVisible(true);
306 me->CastSpell(me, SPELL_REBIRTH_DIVE, false);
307 break;
308 case EVENT_FINISH_DIVE:
309 me->GetMotionMaster()->MoveChase(me->GetVictim());
310 break;
311 case EVENT_SPELL_BERSERK:
312 me->CastSpell(me, SPELL_BERSERK, true);
313 break;
314 }
315
316 if (me->isAttackReady())
317 {
318 if (me->IsWithinMeleeRange(me->GetVictim()))
319 {
320 me->AttackerStateUpdate(me->GetVictim());
321 me->resetAttackTimer();
322 }
323 else
324 {
325 me->resetAttackTimer();
326 ThreatContainer::StorageType const& threatList = me->GetThreatMgr().GetThreatList();
327 for (ThreatContainer::StorageType::const_iterator itr = threatList.begin(); itr != threatList.end(); ++itr)
328 if (Unit* unit = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid()))
329 if (me->IsWithinMeleeRange(unit))
330 {
331 me->AttackerStateUpdate(unit);
332 return;
333 }
334
335 me->CastSpell(me, SPELL_FLAME_BUFFET, false);
336 }
337 }
338 }
