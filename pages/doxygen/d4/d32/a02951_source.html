159 void SetData(uint32 type, uint32 data) override
160 {
161 switch (type)
162 {
163 case DATA_RAGEWINTERCHILLEVENT:
164 m_auiEncounter[0] = data;
165 break;
166 case DATA_ANETHERONEVENT:
167 m_auiEncounter[1] = data;
168 break;
169 case DATA_KAZROGALEVENT:
170 m_auiEncounter[2] = data;
171 break;
172 case DATA_AZGALOREVENT:
173 {
174 m_auiEncounter[3] = data;
175 if (data == DONE)
176 {
177 if (ArchiYell)
178 break;
179
180 ArchiYell = true;
181
182 Creature* creature = instance->GetCreature(Azgalor);
183 if (creature)
184 {
185 Creature* unit = creature->SummonCreature(NPC_WORLD_TRIGGER_TINY, creature->GetPositionX(), creature->GetPositionY(), creature->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN, 10000);
186
187 Map* map = creature->GetMap();
188 if (map->IsDungeon() && unit)
189 {
190 unit->SetVisible(false);
191 Map::PlayerList const& PlayerList = map->GetPlayers();
192 if (PlayerList.IsEmpty())
193 return;
194
195 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
196 {
197 if (Player* player = i->GetSource())
198 {
199 WorldPacket packet;
200 ChatHandler::BuildChatPacket(packet, CHAT_MSG_MONSTER_YELL, LANG_UNIVERSAL, unit, player, YELL_EFFORTS);
201 player->SendDirectMessage(&packet);
202 player->PlayDirectSound(10986, player);
203 }
204 }
205 }
206 }
207 }
208 }
209 break;
210 case DATA_ARCHIMONDEEVENT:
211 m_auiEncounter[4] = data;
212 break;
213 case DATA_RESET_TRASH_COUNT:
214 Trash = 0;
215 break;
216 case DATA_TRASH:
217 if (data)
218 Trash = data;
219 else
220 Trash--;
221 DoUpdateWorldState(WORLD_STATE_ENEMYCOUNT, Trash);
222 break;
223 case TYPE_RETREAT:
224 if (data == SPECIAL)
225 {
226 if (!m_uiAncientGemGUID.empty())
227 {
228 for (ObjectGuid const& guid : m_uiAncientGemGUID)
229 {
230 //don't know how long it expected
231 DoRespawnGameObject(guid, DAY);
232 }
233 }
234 }
235 break;
236 case DATA_ALLIANCE_RETREAT:
237 allianceRetreat = data;
238 HandleGameObject(HordeGate, true);
239 SaveToDB();
240 break;
241 case DATA_HORDE_RETREAT:
242 hordeRetreat = data;
243 HandleGameObject(ElfGate, true);
244 SaveToDB();
245 break;
246 case DATA_RAIDDAMAGE:
247 RaidDamage += data;
248 if (RaidDamage >= MINRAIDDAMAGE)
249 RaidDamage = MINRAIDDAMAGE;
250 break;
251 case DATA_RESET_RAIDDAMAGE:
252 RaidDamage = 0;
253 break;
254 }
255
256 // LOG_DEBUG("scripts", "Instance Hyjal: Instance data updated for event {} (Data={})", type, data);
257
258 if (data == DONE)
259 {
260 OUT_SAVE_INST_DATA;
261
262 std::ostringstream saveStream;
263 saveStream << m_auiEncounter[0] << ' ' << m_auiEncounter[1] << ' ' << m_auiEncounter[2] << ' '
264 << m_auiEncounter[3] << ' ' << m_auiEncounter[4]
265 << ' ' << allianceRetreat << ' ' << hordeRetreat
266 << ' ' << RaidDamage;
267
268 str_data = saveStream.str();
269
270 SaveToDB();
271 OUT_SAVE_INST_DATA_COMPLETE;
272 }
273 }

306 void Load(const char* in) override
307 {
308 if (!in)
309 {
310 OUT_LOAD_INST_DATA_FAIL;
311 return;
312 }
313
314 OUT_LOAD_INST_DATA(in);
315 std::istringstream loadStream(in);
316 loadStream >> m_auiEncounter[0] >> m_auiEncounter[1] >> m_auiEncounter[2] >> m_auiEncounter[3] >> m_auiEncounter[4] >> allianceRetreat >> hordeRetreat >> RaidDamage;
317 for (uint8 i = 0; i < EncounterCount; ++i)
318 if (m_auiEncounter[i] == IN_PROGRESS) // Do not load an encounter as IN_PROGRESS - reset it instead.
319 m_auiEncounter[i] = NOT_STARTED;
320 OUT_LOAD_INST_DATA_COMPLETE;
321 }
