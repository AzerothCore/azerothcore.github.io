202 void Load(const char* in) override
203 {
204 if (!in)
205 {
206 OUT_LOAD_INST_DATA_FAIL;
207 return;
208 }
209
210 OUT_LOAD_INST_DATA(in);
211
212 char dataHead1, dataHead2;
213 uint16 data0, data1, data2, data3;
214
215 std::istringstream loadStream(in);
216 loadStream >> dataHead1 >> dataHead2 >> data0 >> data1 >> data2 >> data3;
217
218 if (dataHead1 == 'U' && dataHead2 == 'P')
219 {
220 Encounters[0] = data0;
221 Encounters[1] = data1;
222 Encounters[2] = data2;
223 Encounters[3] = data3;
224
225 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
226 if (Encounters[i] == IN_PROGRESS)
227 Encounters[i] = NOT_STARTED;
228 }
229 else OUT_LOAD_INST_DATA_FAIL;
230
231 OUT_LOAD_INST_DATA_COMPLETE;
232 }
