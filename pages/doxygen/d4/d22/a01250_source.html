43void CinematicMgr::BeginCinematic()
44{
45 // Sanity check for active camera set
46 if (m_activeCinematicCameraId == 0)
47 return;
48
49 if (std::vector<FlyByCamera> const* flyByCameras = GetFlyByCameras(m_activeCinematicCameraId))
50 {
51 // Initialize diff, and set camera
52 m_cinematicDiff = 0;
53 m_cinematicCamera = flyByCameras;
54
55 auto camitr = m_cinematicCamera->begin();
56 if (camitr != m_cinematicCamera->end())
57 {
58 Position const& pos = camitr->locations;
59 if (!pos.IsPositionValid())
60 return;
61
62 player->GetMap()->LoadGrid(pos.GetPositionX(), pos.GetPositionY());
63 m_CinematicObject = player->SummonCreature(VISUAL_WAYPOINT, pos.m_positionX, pos.m_positionY, pos.m_positionZ, 0.0f, TEMPSUMMON_TIMED_DESPAWN, 120000);
64 if (m_CinematicObject)
65 {
66 m_CinematicObject->setActive(true);
67 player->SetViewpoint(m_CinematicObject, true);
68 }
69
70 // Get cinematic length
71 m_cinematicLength = flyByCameras->back().timeStamp;
72 }
73 }
74}

76void CinematicMgr::EndCinematic()
77{
78 if (m_activeCinematicCameraId == 0)
79 return;
80
81 m_cinematicDiff = 0;
82 m_cinematicCamera = nullptr;
83 m_activeCinematicCameraId = 0;
84 if (m_CinematicObject)
85 {
86 if (WorldObject* vpObject = player->GetViewpoint())
87 if (vpObject == m_CinematicObject)
88 player->SetViewpoint(m_CinematicObject, false);
89
90 m_CinematicObject->AddObjectToRemoveList();
91 }
92}

94void CinematicMgr::UpdateCinematicLocation(uint32 /*diff*/)
95{
96 if (m_activeCinematicCameraId == 0 || !m_cinematicCamera || m_cinematicCamera->size() == 0)
97 return;
98
99 Position lastPosition;
100 uint32 lastTimestamp = 0;
101 Position nextPosition;
102 uint32 nextTimestamp = 0;
103
104 // Obtain direction of travel
105 for (FlyByCamera cam : *m_cinematicCamera)
106 {
107 if (cam.timeStamp > m_cinematicDiff)
108 {
109 nextPosition.Relocate(cam.locations);
110 nextTimestamp = cam.timeStamp;
111 break;
112 }
113 lastPosition.Relocate(cam.locations);
114 lastTimestamp = cam.timeStamp;
115 }
116 float angle = lastPosition.GetAbsoluteAngle(&nextPosition);
117 angle -= lastPosition.GetOrientation();
118 if (angle < 0)
119 angle += 2 * float(M_PI);
120
121 // Look for position around 2 second ahead of us.
122 int32 workDiff = m_cinematicDiff;
123
124 // Modify result based on camera direction (Humans for example, have the camera point behind)
125 workDiff += static_cast<int32>(float(CINEMATIC_LOOKAHEAD) * cos(angle));
126
127 // Get an iterator to the last entry in the cameras, to make sure we don't go beyond the end
128 auto endItr = m_cinematicCamera->rbegin();
129 if (endItr != m_cinematicCamera->rend() && workDiff > static_cast<int32>(endItr->timeStamp))
130 workDiff = endItr->timeStamp;
131
132 // Never try to go back in time before the start of cinematic!
133 if (workDiff < 0)
134 workDiff = m_cinematicDiff;
135
136 // Obtain the previous and next waypoint based on timestamp
137 for (FlyByCamera cam : *m_cinematicCamera)
138 {
139 if (static_cast<int32>(cam.timeStamp) >= workDiff)
140 {
141 nextPosition.Relocate(cam.locations);
142 nextTimestamp = cam.timeStamp;
143 break;
144 }
145 lastPosition.Relocate(cam.locations);
146 lastTimestamp = cam.timeStamp;
147 }
148
149 // Never try to go beyond the end of the cinematic
150 if (workDiff > static_cast<int32>(nextTimestamp))
151 workDiff = static_cast<int32>(nextTimestamp);
152
153 // Interpolate the position for this moment in time (or the adjusted moment in time)
154 uint32 timeDiff = nextTimestamp - lastTimestamp;
155 uint32 interDiff = workDiff - lastTimestamp;
156 float xDiff = nextPosition.m_positionX - lastPosition.m_positionX;
157 float yDiff = nextPosition.m_positionY - lastPosition.m_positionY;
158 float zDiff = nextPosition.m_positionZ - lastPosition.m_positionZ;
159 Position interPosition(lastPosition.m_positionX + (xDiff * (float(interDiff) / float(timeDiff))), lastPosition.m_positionY +
160 (yDiff * (float(interDiff) / float(timeDiff))), lastPosition.m_positionZ + (zDiff * (float(interDiff) / float(timeDiff))));
161
162 // Advance (at speed) to this position. The remote sight object is used
163 // to send update information to player in cinematic
164 if (m_CinematicObject && interPosition.IsPositionValid())
165 m_CinematicObject->MonsterMoveWithSpeed(interPosition.m_positionX, interPosition.m_positionY, interPosition.m_positionZ, 500.0f);
166
167 // If we never received an end packet 10 seconds after the final timestamp then force an end
168 if (m_cinematicDiff > m_cinematicLength + 10 * IN_MILLISECONDS)
169 EndCinematic();
170}
