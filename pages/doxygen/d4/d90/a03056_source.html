81struct boss_ayamiss : public BossAI
82{
83 boss_ayamiss(Creature* creature) : BossAI(creature, DATA_AYAMISS) { homePos = creature->GetHomePosition(); }
84
85 void Reset() override
86 {
87 BossAI::Reset();
88 _phase = PHASE_AIR;
89 _enraged = false;
90 SetCombatMovement(false);
91 _scheduler.CancelAll();
92 me->SetReactState(REACT_AGGRESSIVE);
93 }
94
95 void JustSummoned(Creature* who) override
96 {
97 switch (who->GetEntry())
98 {
99 case NPC_HIVEZARA_SWARMER:
100 who->CastSpell(who, SPELL_HIVEZARA_SWARMER_TELEPORT_TRIGGER, true);
101 _swarmers.push_back(who->GetGUID());
102 break;
103 case NPC_HIVEZARA_LARVA:
104 who->GetMotionMaster()->MovePoint(POINT_PARALYZE, AltarPos);
105 break;
106 }
107
108 summons.Summon(who);
109 }
110
111 void MovementInform(uint32 type, uint32 id) override
112 {
113 if (type == POINT_MOTION_TYPE && id == POINT_AIR)
114 {
115 me->AddUnitState(UNIT_STATE_ROOT);
116 }
117 else if (type == WAYPOINT_MOTION_TYPE && id == POINT_GROUND)
118 {
119 SetCombatMovement(true);
120 me->SetDisableGravity(false);
121
122 me->m_Events.AddEventAtOffset([this]()
123 {
124 me->SetReactState(REACT_AGGRESSIVE);
125 if (me->GetVictim())
126 {
127 me->GetMotionMaster()->MoveChase(me->GetVictim());
128 }
129
130 }, 1s);
131
132 _scheduler.Schedule(5s, 8s, [this](TaskContext context) {
133 DoCastVictim(SPELL_LASH);
134 context.Repeat(8s, 15s);
135 }).Schedule(16s, [this](TaskContext context)
136 {
137 DoCastSelf(SPELL_THRASH);
138 context.Repeat();
139 });
140 }
141 }
142
143 void ScheduleTasks()
144 {
145 _scheduler.Schedule(20s, 30s, [this](TaskContext context)
146 {
147 DoCastSelf(SPELL_STINGER_SPRAY);
148 context.Repeat(15s, 20s);
149 }).Schedule(5s, [this](TaskContext context) {
150 DoCastVictim(SPELL_POISON_STINGER);
151 context.SetGroup(PHASE_AIR);
152 context.Repeat(2s, 3s);
153 }).Schedule(5s, [this](TaskContext context) {
154 DoCastAOE(SPELL_SUMMON_HIVEZARA_SWARMER, true);
155
156 if (_swarmers.size() >= MAX_SWARMER_COUNT)
157 {
158 DoCastAOE(SPELL_HIVEZARA_SWARMER_SWARM, true);
159 }
160
161 context.Repeat(RAND(2400ms, 3600ms));
162 }).Schedule(15s, 28s, [this](TaskContext context) {
163 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0, true))
164 {
165 DoCast(target, SPELL_PARALYZE, true);
166 instance->SetGuidData(DATA_PARALYZED, target->GetGUID());
167 DoCastAOE(RAND(SPELL_SUMMON_LARVA_A, SPELL_SUMMON_LARVA_B), true);
168 }
169 context.Repeat();
170 });
171 }
172
173 void DoAction(int32 action) override
174 {
175 if (action == ACTION_SWARMER_SWARM)
176 {
177 for (ObjectGuid const& guid : _swarmers)
178 {
179 if (Creature* swarmer = me->GetMap()->GetCreature(guid))
180 {
181 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
182 {
183 swarmer->AI()->AttackStart(target);
184 }
185 }
186 }
187
188 _swarmers.clear();
189 }
190 }
191
192 void JustDied(Unit* killer) override
193 {
194 me->GetMotionMaster()->MoveFall();
195 BossAI::JustDied(killer);
196 }
197
198 void EnterEvadeMode(EvadeReason why) override
199 {
200 me->ClearUnitState(UNIT_STATE_ROOT);
201 me->SetHomePosition(homePos);
202 BossAI::EnterEvadeMode(why);
203 }
204
205 void EnterCombat(Unit* attacker) override
206 {
207 BossAI::EnterCombat(attacker);
208 me->SetCanFly(true);
209 me->SetDisableGravity(true);
210 me->GetMotionMaster()->MovePoint(POINT_AIR, AyamissAirPos);
211 ScheduleTasks();
212 }
213
214 void DamageTaken(Unit* /*attacker*/, uint32& damage, DamageEffectType, SpellSchoolMask) override
215 {
216 if (_phase == PHASE_AIR && me->HealthBelowPctDamaged(70, damage))
217 {
218 _phase = PHASE_GROUND;
219 me->ClearUnitState(UNIT_STATE_ROOT);
220 me->SetReactState(REACT_PASSIVE);
221 me->SetCanFly(false);
222 me->SetDisableGravity(false);
223 me->GetMotionMaster()->MovePath(me->GetEntry() * 10, false);
224 DoResetThreat();
225 _scheduler.CancelGroup(PHASE_AIR);
226 }
227
228 if (!_enraged && me->HealthBelowPctDamaged(20, damage))
229 {
230 DoCastSelf(SPELL_FRENZY);
231 Talk(EMOTE_FRENZY);
232 _enraged = true;
233 }
234 }
235
236 void UpdateAI(uint32 diff) override
237 {
238 if (!UpdateVictim())
239 return;
240
241 _scheduler.Update(diff,
242 std::bind(&BossAI::DoMeleeAttackIfReady, this));
243 }
244private:
245 GuidList _swarmers;
246 uint8 _phase;
247 bool _enraged;
248 TaskScheduler _scheduler;
249 Position homePos;
250};
