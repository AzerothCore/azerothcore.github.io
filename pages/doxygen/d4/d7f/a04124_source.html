167 void Reset() override
168 {
169 if (_init)
170 return;
171
172 _init = true;
173 uint32 duration = 0;
174 uint32 aura = 0;
175 me->ReplaceAllNpcFlags(UNIT_NPC_FLAG_NONE);
176
177 if (Unit* owner = me->GetCharmerOrOwner())
178 if (Player* player = owner->ToPlayer())
179 if (player->HasAchieved(ACHIEVEMENT_PONY_UP))
180 {
181 _state = ARGENT_PONY_STATE_ENCH;
182
183 aura = (player->GetTeamId(true) == TEAM_ALLIANCE ? SPELL_AURA_TIRED_S : SPELL_AURA_TIRED_G);
184 duration = player->GetSpellCooldownDelay(aura);
185 me->ReplaceAllNpcFlags(UNIT_NPC_FLAG_GOSSIP);
186
187 for (uint8 i = 0; i < 3; ++i)
188 {
189 if (player->GetTeamId(true) == TEAM_ALLIANCE)
190 {
191 if (uint32 cooldown = player->GetSpellCooldownDelay(SPELL_AURA_POSTMAN_S + i))
192 {
193 duration = cooldown;
194 aura = SPELL_AURA_POSTMAN_S + i;
195 _state = argentPonyService[TEAM_ALLIANCE][i];
196 me->ToTempSummon()->UnSummon(duration);
197 break;
198 }
199 }
200 else
201 {
202 if (uint32 cooldown = player->GetSpellCooldownDelay(SPELL_AURA_BANK_G + i))
203 {
204 duration = cooldown * IN_MILLISECONDS;
205 aura = SPELL_AURA_BANK_G + i;
206 _state = argentPonyService[TEAM_HORDE][i];
207 me->ToTempSummon()->UnSummon(duration);
208 break;
209 }
210 }
211 }
212
213 // Generate Banners
214 uint32 mask = player->GetTeamId(true) ? RACEMASK_HORDE : RACEMASK_ALLIANCE;
215 for (uint8 i = 1; i < MAX_RACES; ++i)
216 if (mask & (1 << (i - 1)) && player->HasAchieved(argentBanners[i].achievement))
217 _banners[i] = true;
218 }
219
220 if (duration && aura)
221 {
222 if (Aura* aur = me->AddAura(aura, me))
223 aur->SetDuration(duration);
224 }
225 }

266 bool OnGossipHello(Player* player, Creature* creature)
267 {
268 if (player->GetGUID() != creature->GetOwnerGUID())
269 return true;
270
271 if (!creature->HasAura(player->GetTeamId(true) ? SPELL_AURA_TIRED_G : SPELL_AURA_TIRED_S))
272 {
273 uint8 _state = creature->AI()->GetData(0 /*GET_DATA_STATE*/);
274 if (_state == ARGENT_PONY_STATE_ENCH || _state == ARGENT_PONY_STATE_VENDOR)
275 AddGossipItemFor(player, GOSSIP_ICON_VENDOR, "Visit a trader.", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_TRADE);
276 if (_state == ARGENT_PONY_STATE_ENCH || _state == ARGENT_PONY_STATE_BANK)
277 AddGossipItemFor(player, GOSSIP_ICON_INTERACT_1, "Visit a bank.", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_BANK);
278 if (_state == ARGENT_PONY_STATE_ENCH || _state == ARGENT_PONY_STATE_MAILBOX)
279 AddGossipItemFor(player, GOSSIP_ICON_CHAT, "Visit a mailbox.", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_MAILBOX);
280 }
281
282 for (uint8 i = RACE_HUMAN; i < MAX_RACES; ++i)
283 if (creature->AI()->GetData(i) == uint32(true))
284 AddGossipItemFor(player, GOSSIP_ICON_CHAT, argentBanners[i].text, GOSSIP_SENDER_MAIN, argentBanners[i].spell);
285
286 SendGossipMenuFor(player, player->GetGossipTextId(creature), creature->GetGUID());
287 return true;
288 }
289
290 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*uiSender*/, uint32 action)
291 {
292 CloseGossipMenuFor(player);
293 uint32 spellId = 0;
294 switch (action)
295 {
296 case GOSSIP_ACTION_TRADE:
297 creature->ReplaceAllNpcFlags(UNIT_NPC_FLAG_VENDOR);
298 player->GetSession()->SendListInventory(creature->GetGUID());
299 spellId = player->GetTeamId(true) ? SPELL_AURA_SHOP_G : SPELL_AURA_SHOP_S;
300 creature->AI()->DoAction(ARGENT_PONY_STATE_VENDOR);
301 break;
302 case GOSSIP_ACTION_BANK:
303 creature->ReplaceAllNpcFlags(UNIT_NPC_FLAG_BANKER);
304 player->GetSession()->SendShowBank(player->GetGUID());
305 spellId = player->GetTeamId(true) ? SPELL_AURA_BANK_G : SPELL_AURA_BANK_S;
306 creature->AI()->DoAction(ARGENT_PONY_STATE_BANK);
307 break;
308 case GOSSIP_ACTION_MAILBOX:
309 {
310 creature->ReplaceAllNpcFlags(UNIT_NPC_FLAG_GOSSIP | UNIT_NPC_FLAG_MAILBOX);
311 player->GetSession()->SendShowMailBox(creature->GetGUID());
312 spellId = player->GetTeamId(true) ? SPELL_AURA_POSTMAN_G : SPELL_AURA_POSTMAN_S;
313 creature->AI()->DoAction(ARGENT_PONY_STATE_MAILBOX);
314 break;
315 }
316 default:
317 if (action > 60000)
318 {
319 creature->AI()->DoAction(action);
320 creature->CastSpell(creature, action, true);
321 }
322 return true;
323 }
324
325 if (spellId && !creature->HasAura(spellId))
326 {
327 creature->CastSpell(creature, spellId, true);
328 player->AddSpellCooldown(spellId, 0, 3 * MINUTE * IN_MILLISECONDS);
329 player->AddSpellCooldown(player->GetTeamId(true) ? SPELL_AURA_TIRED_G : SPELL_AURA_TIRED_S, 0, 3 * MINUTE * IN_MILLISECONDS + 4 * HOUR * IN_MILLISECONDS);
330 creature->DespawnOrUnsummon(3 * MINUTE * IN_MILLISECONDS);
331 }
332 return true;
333 }
334
335private:
336 bool _init;
337 uint8 _state;
338 int32 _mountTimer;
339 bool _banners[MAX_RACES];
340 uint32 _lastAura;
