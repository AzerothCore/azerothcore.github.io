389 void UpdateAI(uint32 diff) override
390 {
391 if (me->GetPositionY() <= ENTRANCE_DOOR.GetPositionY() || me->GetPositionY() >= EXIT_DOOR.GetPositionY())
392 {
393 boss_hodirAI::EnterEvadeMode();
394 return;
395 }
396
397 if (!UpdateVictim())
398 {
399 if (me->IsInCombat())
400 {
401 Map::PlayerList const& pl = me->GetMap()->GetPlayers();
402 for (Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr)
403 itr->GetSource()->CastSpell(itr->GetSource(), SPELL_FLASH_FREEZE_INSTAKILL, true);
404 EnterEvadeMode();
405 }
406 return;
407 }
408
409 events.Update(diff);
410
411 if( me->HasUnitState(UNIT_STATE_CASTING) )
412 return;
413
414 switch (events.ExecuteEvent())
415 {
416 case 0:
417 break;
418 case EVENT_BERSERK:
419 {
420 berserk = true;
421 me->CastSpell(me, SPELL_BERSERK, true);
422 Talk(TEXT_BERSERK);
423 }
424 break;
425 case EVENT_HARD_MODE_MISSED:
426 {
427 Talk(TEXT_HM_MISS);
428 bAchievCacheRare = false;
429 if (pInstance)
430 {
431 if (GameObject* go = pInstance->instance->GetGameObject(pInstance->GetGuidData(GO_HODIR_CHEST_HARD)))
432 {
433 me->CastSpell(go, SPELL_SHATTER_CHEST, false);
434 }
435 }
436 }
437 break;
438 case EVENT_DESPAWN_CHEST:
439 if (pInstance && pInstance->GetData(TYPE_HODIR) != DONE)
440 pInstance->SetData(TYPE_HODIR_HM_FAIL, 0);
441 break;
442 case EVENT_FLASH_FREEZE:
443 {
444 std::list<Unit*> targets;
445 Map::PlayerList const& pl = me->GetMap()->GetPlayers();
446 for (Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr)
447 targets.push_back(itr->GetSource());
448 targets.remove_if(Acore::ObjectTypeIdCheck(TYPEID_PLAYER, false));
449 targets.remove_if(Acore::UnitAuraCheck(true, SPELL_FLASH_FREEZE_TRAPPED_PLAYER));
450 Acore::Containers::RandomResize(targets, (RAID_MODE(2,3)));
451 for (std::list<Unit*>::const_iterator itr = targets.begin(); itr != targets.end(); ++itr)
452 {
453 float prevZ = (*itr)->GetPositionZ();
454 (*itr)->m_positionZ = 432.7f;
455 (*itr)->CastSpell((*itr), SPELL_ICICLE_VISUAL_PACKED, true);
456 (*itr)->m_positionZ = prevZ;
457 }
458
459 me->CastSpell((Unit*)nullptr, SPELL_FLASH_FREEZE_CAST, false);
460 me->PlayDirectSound(SOUND_HODIR_FLASH_FREEZE, 0);
461 Talk(TEXT_FLASH_FREEZE);
462 Talk(TEXT_EMOTE_FREEZE);
463 SmallIcicles(false);
464 events.ScheduleEvent(EVENT_FLASH_FREEZE, urand(48000, 49000));
465 events.ScheduleEvent(EVENT_SMALL_ICICLES_ENABLE, Is25ManRaid() ? 12000 : 24000);
466 events.ScheduleEvent(EVENT_FROZEN_BLOWS, 15000);
467 events.RescheduleEvent(EVENT_FREEZE, urand(17000, 20000));
468 }
469 break;
470 case EVENT_SMALL_ICICLES_ENABLE:
471 {
472 SmallIcicles(true);
473 }
474 break;
475 case EVENT_FROZEN_BLOWS:
476 {
477 Talk(TEXT_EMOTE_BLOW);
478 Talk(TEXT_STALACTITE);
479 me->CastSpell(me, Is25ManRaid()? SPELL_FROZEN_BLOWS_25 : SPELL_FROZEN_BLOWS_10, true);
480 }
481 break;
482 case EVENT_FREEZE:
483 if (Player* plr = SelectTargetFromPlayerList(50.0f, SPELL_FLASH_FREEZE_TRAPPED_PLAYER))
484 {
485 me->CastSpell(plr, SPELL_FREEZE, false);
486 }
487 else if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
488 {
489 me->CastSpell(target, SPELL_FREEZE, false);
490 }
491 events.RescheduleEvent(EVENT_FREEZE, urand(17000, 20000));
492 break;
493 }
494
495 DoMeleeAttackIfReady();
496 }

503 void SpawnHelpers()
504 {
505 char faction = 'A';
506 if( hhd[0][0].id )
507 {
508 Map::PlayerList const& cl = me->GetMap()->GetPlayers();
509 for (Map::PlayerList::const_iterator itr = cl.begin(); itr != cl.end(); ++itr)
510 if (!itr->GetSource()->IsGameMaster())
511 {
512 faction = (itr->GetSource()->GetTeamId() == TEAM_ALLIANCE ? 'A' : 'H');
513 break;
514 }
515 }
516
517 uint8 cnt = 0;
518 if( faction )
519 for( uint8 k = 0; k < 4; ++k )
520 {
521 if( (faction == 'A' && ( k > 1 || (k == 1 && RAID_MODE(1, 0)) )) ||
522 (faction == 'H' && ( k < 2 || (k == 3 && RAID_MODE(1, 0)) )) )
523 continue;
524
525 for( uint8 i = 0; i < 4; ++i )
526 {
527 if( !hhd[k][i].id )
528 continue;
529
530 if( Creature* h_p = me->SummonCreature(hhd[k][i].id, hhd[k][i].x, hhd[k][i].y, 432.69f, M_PI / 2) )
531 {
532 h_p->SetFaction(1665);
533 if( cnt < 8 )
534 Helpers[cnt++] = h_p->GetGUID();
535
536 if( Creature* c = h_p->SummonCreature(NPC_FLASH_FREEZE_NPC, h_p->GetPositionX(), h_p->GetPositionY(), h_p->GetPositionZ(), 0.0f, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 2000) )
537 {
538 c->CastSpell(h_p, SPELL_FLASH_FREEZE_TRAPPED_NPC, true);
539 JustSummoned(c);
540 }
541 }
542 }
