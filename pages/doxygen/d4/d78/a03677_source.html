46class boss_koralon : public CreatureScript
47{
48public:
49 boss_koralon() : CreatureScript("boss_koralon") { }

51 struct boss_koralonAI : public ScriptedAI
52 {
53 boss_koralonAI(Creature* creature) : ScriptedAI(creature)
54 {
55 pInstance = me->GetInstanceScript();
56 }
57
58 InstanceScript* pInstance;
59 EventMap events;
60 uint32 rotateTimer;
61
62 void Reset() override
63 {
64 rotateTimer = 0;
65 events.Reset();
66 if (pInstance)
67 {
68 if (pInstance->GetData(DATA_STONED))
69 {
70 if (Aura* aur = me->AddAura(SPELL_STONED_AURA, me))
71 {
72 aur->SetMaxDuration(60 * MINUTE * IN_MILLISECONDS);
73 aur->SetDuration(60 * MINUTE * IN_MILLISECONDS);
74 }
75 }
76 pInstance->SetData(EVENT_KORALON, NOT_STARTED);
77 }
78 }
79
80 void AttackStart(Unit* who) override
81 {
82 if (me->HasAura(SPELL_STONED_AURA))
83 return;
84
85 ScriptedAI::AttackStart(who);
86 }
87
88 void EnterCombat(Unit* /*who*/) override
89 {
90 me->CastSpell(me, SPELL_BURNING_FURY, true);
91
92 events.ScheduleEvent(EVENT_BURNING_BREATH, 10000);
93 events.ScheduleEvent(EVENT_METEOR_FISTS, 30000);
94 events.ScheduleEvent(EVENT_FLAME_CINDER, 20000);
95
96 if (pInstance)
97 pInstance->SetData(EVENT_KORALON, IN_PROGRESS);
98 }
99
100 void JustDied(Unit* /*killer*/) override
101 {
102 if (pInstance)
103 pInstance->SetData(EVENT_KORALON, DONE);
104 }
105
106 void UpdateAI(uint32 diff) override
107 {
108 if (rotateTimer)
109 {
110 me->SetUInt64Value(UNIT_FIELD_CHANNEL_OBJECT, 0);
111 rotateTimer += diff;
112 if (rotateTimer >= 3000)
113 {
114 if (!me->HasUnitMovementFlag(MOVEMENTFLAG_LEFT))
115 {
116 me->SetUnitMovementFlags(MOVEMENTFLAG_LEFT);
117 me->SendMovementFlagUpdate();
118 rotateTimer = 1;
119 return;
120 }
121 else
122 {
123 me->RemoveUnitMovementFlag(MOVEMENTFLAG_LEFT);
124 me->SendMovementFlagUpdate();
125 rotateTimer = 0;
126 return;
127 }
128 }
129 }
130
131 if (!UpdateVictim())
132 return;
133
134 events.Update(diff);
135 if (me->HasUnitState(UNIT_STATE_CASTING))
136 return;
137
138 switch (events.ExecuteEvent())
139 {
140 case EVENT_BURNING_BREATH:
141 rotateTimer = 1500;
142 me->CastSpell(me, SPELL_BURNING_BREATH, false);
143 events.RepeatEvent(45000);
144 break;
145 case EVENT_METEOR_FISTS:
146 me->CastSpell(me, SPELL_METEOR_FISTS, true);
147 events.RepeatEvent(45000);
148 break;
149 case EVENT_FLAME_CINDER:
150 me->CastSpell(me, SPELL_FLAMING_CINDER, true);
151 events.RepeatEvent(30000);
152 break;
153 default:
154 break;
155 }
156
157 DoMeleeAttackIfReady();
158 }
159 };
160
161 CreatureAI* GetAI(Creature* creature) const override
162 {
163 return GetVaultOfArchavonAI<boss_koralonAI>(creature);
164 }
165};
