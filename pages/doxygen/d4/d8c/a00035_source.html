26template<class T, class BoundsFunc = BoundsTrait<T>>
27class BIHWrap
28{
29 template<class RayCallback>
30 struct MDLCallback
31 {
32 const T* const* objects;
33 RayCallback& _callback;
34 uint32 objects_size;
35
36 MDLCallback(RayCallback& callback, const T* const* objects_array, uint32 objects_size ) : objects(objects_array), _callback(callback), objects_size(objects_size) { }
37
39 bool operator() (const G3D::Ray& ray, uint32 idx, float& maxDist, bool stopAtFirstHit)
40 {
41 if (idx >= objects_size)
42 {
43 return false;
44 }
45 if (const T* obj = objects[idx])
46 {
47 return _callback(ray, *obj, maxDist, stopAtFirstHit);
48 }
49 return false;
50 }
51
53 void operator() (const G3D::Vector3& p, uint32 idx)
54 {
55 if (idx >= objects_size)
56 {
57 return;
58 }
59 if (const T* obj = objects[idx])
60 {
61 _callback(p, *obj);
62 }
63 }
64 };

98 void balance()
99 {
100 if (unbalanced_times == 0)
101 {
102 return;
103 }
104
105 unbalanced_times = 0;
106 m_objects.fastClear();
107 m_obj2Idx.getKeys(m_objects);
108 m_objects_to_push.getMembers(m_objects);
109 //assert that m_obj2Idx has all the keys
110
111 m_tree.build(m_objects, BoundsFunc::GetBounds2);
112 }
