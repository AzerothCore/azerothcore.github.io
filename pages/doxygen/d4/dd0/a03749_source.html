67 void RightClickCauldron(ObjectGuid guid)
68 {
69 if (questList.empty())
70 return;
71
72 std::map<ObjectGuid, uint32>::iterator itr = questList.find(guid);
73 if (itr == questList.end())
74 return;
75
76 Player* player = ObjectAccessor::GetPlayer(*me, guid);
77 if (player)
78 {
79 uint32 itemCode = itr->second;
80
81 uint32 itemEntry = GetTaskItemEntry(itemCode);
82 uint32 auraId = GetTaskAura(itemCode);
83 uint32 counter = GetTaskCounter(itemCode);
84 if (player->HasAura(auraId))
85 {
86 // player still has aura, but no item. Skip
87 if (!player->HasItemCount(itemEntry))
88 return;
89
90 // if we are here, all is ok (aura and item present)
91 player->DestroyItemCount(itemEntry, 1, true);
92 player->RemoveAurasDueToSpell(auraId);
93
94 if (counter < 6)
95 {
96 StartNextTask(player->GetGUID(), counter + 1);
97 return;
98 }
99 else
100 player->KilledMonsterCredit(28248);
101 }
102 else
103 {
104 // if we are here, it means we failed :(
105 player->SetQuestStatus(QUEST_ALCHEMIST_APPRENTICE, QUEST_STATUS_FAILED);
106 }
107 }
108 questList.erase(itr);
109 }

111 // Generate a Task and announce it to the player

113 {

126 //uint32 itemEntry = GetTaskItemEntry(itemCode);

130 switch (counter)
131 {
132 case 1:
133 me->TextEmote("Quickly, get me some...", player, true);

137 case 2:

144 me->TextEmote(itemName, player, true);
145 me->CastSpell(player, auraId, true);
146 break;
147 case 4:
148 me->TextEmote("Alright, now fetch me some...", player, true);
149 me->TextEmote(itemName, player, true);
