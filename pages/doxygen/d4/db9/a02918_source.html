171 InfernoTimer = 45000;
172 Talk(SAY_INFERNO);
173 }
174 else InfernoTimer -= diff;
175
176 DoMeleeAttackIfReady();
177 }

240 {
241 me->setDeathState(JUST_DIED);
242 me->RemoveCorpse();
243 return;
244 }
245 }
246 CheckTimer = 5000;
247 }
248 else CheckTimer -= diff;
249

252 {
253 return;
254 }
255
256 if (ImmolationTimer <= diff)
257 {
258 DoCast(me, SPELL_IMMOLATION);
259 ImmolationTimer = 5000;
260 }

262
263 DoMeleeAttackIfReady();
