74 EventMap events;

77 void Reset() override
78 {
79 events.Reset();
80 if (pInstance)
81 pInstance->SetData(BOSS_KRYSTALLUS, NOT_STARTED);
82 }
83
84 void EnterCombat(Unit* /*who*/) override
85 {
86 events.Reset();
87 events.RescheduleEvent(EVENT_BOULDER, 8000);
88 events.RescheduleEvent(EVENT_STOMP, 5000);
89 events.RescheduleEvent(EVENT_GROUND_SLAM, 15000);
90 if (me->GetMap()->IsHeroic())
91 events.RescheduleEvent(EVENT_GROUND_SPIKE, 10000);
92
93 if (pInstance)
94 pInstance->SetData(BOSS_KRYSTALLUS, IN_PROGRESS);
95
96 Talk(SAY_AGGRO);
97 }
98
99 void RemoveStonedEffect()
100 {
101 Map* map = me->GetMap();
102 if (map->IsDungeon())
103 {
104 Map::PlayerList const& players = map->GetPlayers();
105 for(Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
106 if (itr->GetSource()->IsAlive())
107 itr->GetSource()->RemoveAura(GROUND_SLAM_STONED_EFFECT);
108 }
109 }
110
111 void UpdateAI(uint32 diff) override
112 {
113 if (!UpdateVictim())
114 return;
115
116 events.Update(diff);
117
118 if (me->HasUnitState(UNIT_STATE_CASTING))
119 return;
120
121 switch (events.ExecuteEvent())
122 {
123 case EVENT_BOULDER:
124 {
125 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true, 0))
126 me->CastSpell(target, DUNGEON_MODE(BOULDER_TOSS, BOULDER_TOSS_H), false);
127
128 events.RepeatEvent(5000 + rand() % 2000);
129 break;
130 }
131 case EVENT_GROUND_SPIKE:
132 {
133 me->CastSpell(me->GetVictim(), GROUND_SPIKE_H, false); // current enemy target
134 events.RepeatEvent(8000 + rand() % 3000);
135 break;
136 }
137 case EVENT_STOMP:
138 {
139 me->CastSpell(me, DUNGEON_MODE(STOMP, STOMP_H), false);
140 events.RepeatEvent(13000 + rand() % 5000);
141 break;
142 }
143 case EVENT_GROUND_SLAM:
144 {
145 events.RepeatEvent(10000 + rand() % 3000);
146 me->CastSpell(me->GetVictim(), GROUND_SLAM, true);
147 events.DelayEvents(10000);
148 events.RescheduleEvent(EVENT_SHATTER, 8000);
149 break;
150 }
151 case EVENT_SHATTER:
152 {
153 me->CastSpell((Unit*)nullptr, DUNGEON_MODE(SHATTER, SHATTER_H), false);
154 Talk(SAY_SHATTER);
155 events.RescheduleEvent(EVENT_REMOVE_STONED, 1500);
156 break;
157 }
158 case EVENT_REMOVE_STONED:
159 {
160 RemoveStonedEffect();
161 break;
162 }
163 }
164
165 DoMeleeAttackIfReady();
166 }
167
168 void JustDied(Unit* /*killer*/) override
169 {
170 Talk(SAY_DEATH);
171 if (pInstance)
172 pInstance->SetData(BOSS_KRYSTALLUS, DONE);
173 }
174
175 void KilledUnit(Unit* /*victim*/) override
176 {
177 Talk(SAY_KILL);
178 }
179 };

221 void CalculateDamage()
222 {
223 if (!GetHitUnit())
224 return;
225
226 float radius = GetSpellInfo()->Effects[EFFECT_0].CalcRadius(GetCaster());
227 if (!radius)
228 return;
229
230 float distance = GetCaster()->GetDistance2d(GetHitUnit());
231 if (distance > 1.0f)
232 SetHitDamage(int32(GetHitDamage() * ((radius - distance) / radius)));
233 }
