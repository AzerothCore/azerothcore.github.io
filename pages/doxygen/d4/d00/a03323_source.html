121 Talk(SAY_FACE_AGGRO);
122 DoZoneInCombat();
123 events.Reset();
124 events.RescheduleEvent(EVENT_SPELL_PHANTOM_BLAST, 5000);
125 events.RescheduleEvent(EVENT_SPELL_MIRRORED_SOUL, 9000);

133 // Suport for Quest Tempering the Blade
134 Map::PlayerList const& pList = me->GetMap()->GetPlayers();

137 Player* player = itr->GetSource();
138 if ((player->GetTeamId() == TEAM_ALLIANCE && player->GetQuestStatus(QUEST_TEMPERING_THE_BLADE_A) == QUEST_STATUS_INCOMPLETE) ||
139 (player->GetTeamId() == TEAM_HORDE && player->GetQuestStatus(QUEST_TEMPERING_THE_BLADE_H) == QUEST_STATUS_INCOMPLETE))
140 {
141 if (!me->FindNearestCreature(NPC_CRUCIBLE_OF_SOULS, 100.0f))

170 void UpdateAI(uint32 diff) override
171 {
172 if (!UpdateVictim())
173 return;
174
175 events.Update(diff);
176
177 if (Spell* s = me->GetCurrentSpell(CURRENT_CHANNELED_SPELL))
178 if (s->m_spellInfo->Id == SPELL_MIRRORED_SOUL)
179 {
180 switch (events.ExecuteEvent())
181 {
182 case 0:
183 break;
184 case EVENT_SPELL_PHANTOM_BLAST:
185 me->CastSpell(me->GetVictim(), SPELL_PHANTOM_BLAST, false);
186 events.RepeatEvent(5000);
187 break;
188 default:
189 events.RepeatEvent(1000);
190 break;
191 }
192
193 if (!me->GetCurrentSpell(CURRENT_GENERIC_SPELL))
194 {
195 me->ClearUnitState(UNIT_STATE_CASTING);
196 DoMeleeAttackIfReady();
197 me->AddUnitState(UNIT_STATE_CASTING);
198 }
199
200 return;
201 }
202
203 if (me->HasUnitState(UNIT_STATE_CASTING))
204 return;
205
206 switch(events.ExecuteEvent())
207 {
208 case 0:
209 break;
210 case EVENT_SPELL_PHANTOM_BLAST:
211 me->CastSpell(me->GetVictim(), SPELL_PHANTOM_BLAST, false);
212 events.RepeatEvent(5000);
213 break;
214 case EVENT_SPELL_MIRRORED_SOUL:
215 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 90.0f, true))
216 {
217 me->CastSpell(target, SPELL_MIRRORED_SOUL, false);
218 me->setAttackTimer(BASE_ATTACK, 2500);
219 Talk(EMOTE_MIRRORED_SOUL);
220 }
221 events.RepeatEvent(urand(20000, 30000));
222 break;
223 case EVENT_SPELL_WELL_OF_SOULS:
224 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 40.0f, true))
225 me->CastSpell(target, SPELL_WELL_OF_SOULS, false);
226 events.RepeatEvent(urand(25000, 30000));
227 events.DelayEventsToMax(4000, 0);
228 break;
229 case EVENT_SPELL_UNLEASHED_SOULS:
230 me->CastSpell(me, SPELL_UNLEASHED_SOULS, false);
231 Talk(SAY_FACE_UNLEASH_SOUL);
232 Talk(EMOTE_UNLEASH_SOUL);
233 events.RepeatEvent(urand(30000, 40000));
234 events.DelayEventsToMax(5000, 0);
235 me->setAttackTimer(BASE_ATTACK, 5500);
236 break;
237 case EVENT_SPELL_WAILING_SOULS:
238 Talk(SAY_FACE_WAILING_SOUL);
239 Talk(EMOTE_WAILING_SOUL);
240 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true))
241 me->CastCustomSpell(SPELL_WAILING_SOULS_TARGETING, SPELLVALUE_MAX_TARGETS, 1, target, false);
242 events.RepeatEvent(80000);
243 events.DelayEventsToMax(20000, 0);
244 break;
245 }
246
247 DoMeleeAttackIfReady();
248 }

327 void HandlePeriodicTick(AuraEffect const* aurEff)
328 {
329 PreventDefaultAction();
330 if (Unit* t = GetTarget())
331 {
332 if (aurEff->GetTickNumber() < 30)
333 {
334 // spinning, casting, etc.
335 float diff = (2 * M_PI) / (4 * 30);
336 float new_o = t->GetOrientation() + diff * dir;
337 if (new_o >= 2 * M_PI)
338 new_o -= 2 * M_PI;
339 else if (new_o < 0)
340 new_o += 2 * M_PI;
341 t->UpdateOrientation(new_o);
342 t->SetFacingTo(new_o);
343 t->CastSpell(t, SPELL_WAILING_SOULS_DMG_N, true);
344 }
345 else if (aurEff->GetTickNumber() == 33)
346 {
347 t->SetControlled(false, UNIT_STATE_ROOT);
348 t->DisableRotate(false);
349 if (t->GetTypeId() == TYPEID_UNIT)
350 t->ToCreature()->SetReactState(REACT_AGGRESSIVE);
351 if (t->GetVictim())
352 {
353 t->SetGuidValue(UNIT_FIELD_TARGET, t->GetVictim()->GetGUID());
354 t->GetMotionMaster()->MoveChase(t->GetVictim());
355 }
356 }
357 else if (aurEff->GetTickNumber() >= 34)
358 Remove(AURA_REMOVE_BY_EXPIRE);
359 }
360 }
361
362 void Register() override
363 {
364 OnEffectPeriodic += AuraEffectPeriodicFn(spell_wailing_souls_periodic_AuraScript::HandlePeriodicTick, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);
365 }
