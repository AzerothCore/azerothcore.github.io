40 struct instance_uldaman_InstanceMapScript : public InstanceScript
41 {
42 instance_uldaman_InstanceMapScript(Map* map) : InstanceScript(map) { }
43
44 void Initialize() override
45 {
46 memset(&_encounters, 0, sizeof(_encounters));
47 }
48
49 void OnGameObjectCreate(GameObject* gameobject) override
50 {
51 switch (gameobject->GetEntry())
52 {
53 case GO_IRONAYA_SEAL_DOOR:
54 case GO_KEYSTONE:
55 if (_encounters[DATA_IRONAYA_DOORS] == DONE)
56 {
57 HandleGameObject(ObjectGuid::Empty, true, gameobject);
58 gameobject->SetGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
59 }
60 break;
61 case GO_TEMPLE_DOOR:
62 if (_encounters[DATA_STONE_KEEPERS] == DONE)
63 HandleGameObject(ObjectGuid::Empty, true, gameobject);
64 break;
65 case GO_ANCIENT_VAULT_DOOR:
66 ancientVaultDoorGUID = gameobject->GetGUID();
67 if (_encounters[DATA_ARCHAEDAS] == DONE)
68 HandleGameObject(ObjectGuid::Empty, true, gameobject);
69 break;
70 case GO_ARCHAEDAS_TEMPLE_DOOR:
71 archaedasTempleDoorGUID = gameobject->GetGUID();
72 break;
73 }
74 }
75
76 void SetData(uint32 type, uint32 data) override
77 {
78 switch (type)
79 {
80 case DATA_IRONAYA_DOORS:
81 case DATA_STONE_KEEPERS:
82 _encounters[type] = data;
83 break;
84 case DATA_ARCHAEDAS:
85 _encounters[type] = data;
86 HandleGameObject(ancientVaultDoorGUID, data == DONE, nullptr);
87 HandleGameObject(archaedasTempleDoorGUID, data != IN_PROGRESS, nullptr);
88 break;
89 }
90
91 if (data == DONE)
92 SaveToDB();
93 }
94
95 uint32 GetData(uint32 data) const override
96 {
97 if (data < MAX_ENCOUNTERS)
98 {
99 return _encounters[data];
100 }
101 return 0;
102 }
103
104 std::string GetSaveData() override
105 {
106 std::ostringstream saveStream;
107 saveStream << "U D " << _encounters[DATA_IRONAYA_DOORS] << ' ' << _encounters[DATA_STONE_KEEPERS] << ' ' << _encounters[DATA_ARCHAEDAS];
108 return saveStream.str();
109 }
110
111 void Load(const char* in) override
112 {
113 if (!in)
114 return;
115
116 char dataHead1, dataHead2;
117 std::istringstream loadStream(in);
118 loadStream >> dataHead1 >> dataHead2;
119
120 if (dataHead1 == 'U' && dataHead2 == 'D')
121 {
122 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
123 {
124 loadStream >> _encounters[i];
125 if (_encounters[i] == IN_PROGRESS)
126 _encounters[i] = NOT_STARTED;
127 }
128 }
129 }
130
131 void OnCreatureCreate(Creature* creature) override
132 {
133 switch (creature->GetEntry())
134 {
135 case NPC_STONE_KEEPER:
136 if (_encounters[DATA_STONE_KEEPERS] != DONE && !creature->IsAlive())
137 creature->Respawn();
138 break;
139 }
140 }
141
142 private:
143 uint32 _encounters[MAX_ENCOUNTERS];
144 ObjectGuid archaedasTempleDoorGUID;
145 ObjectGuid ancientVaultDoorGUID;
146 };
147
148 InstanceScript* GetInstanceScript(InstanceMap* map) const override
149 {
150 return new instance_uldaman_InstanceMapScript(map);
151 }
152};
