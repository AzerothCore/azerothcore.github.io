86 void EnterCombat(Unit* who) override
87 {
88 DoZoneInCombat();
89 Talk(SAY_AGGRO);
90 DoCast(me, SPELL_EARTH_SHIELD);
91 events.Reset();
92 events.RescheduleEvent(EVENT_SPELL_BLOODLUST, 15000);
93 events.RescheduleEvent(EVENT_SPELL_BREAK_BONDS, urand(9000, 14000));
94 events.RescheduleEvent(EVENT_SPELL_CHAIN_HEAL, 0);
95 events.RescheduleEvent(EVENT_SPELL_EARTH_SHIELD, 20000);
96 events.RescheduleEvent(EVENT_SPELL_EARTH_SHOCK, urand(2000, 8000));
97 events.RescheduleEvent(EVENT_SPELL_LIGHTNING_BOLT, urand(5000, 10000));
98 if (IsHeroic())
99 events.RescheduleEvent(EVENT_SPELL_STORMSTRIKE, 3000);
100
101 if (Creature* c = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_EREKEM_GUARD_1_GUID)))
102 if (!c->IsInCombat())
103 c->AI()->AttackStart(who);
104 if (Creature* c = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_EREKEM_GUARD_2_GUID)))
105 if (!c->IsInCombat())
106 c->AI()->AttackStart(who);
107 }
108
109 void UpdateAI(uint32 diff) override
110 {
111 if (!UpdateVictim())
112 return;
113
114 events.Update(diff);
115
116 if (me->HasUnitState(UNIT_STATE_CASTING))
117 return;
118
119 switch(events.ExecuteEvent())
120 {
121 case 0:
122 break;
123 case EVENT_SPELL_BLOODLUST:
124 me->CastSpell((Unit*)nullptr, SPELL_BLOODLUST, false);
125 events.RepeatEvent(urand(35000, 45000));
126 break;
127 case EVENT_SPELL_BREAK_BONDS:
128 me->CastSpell((Unit*)nullptr, SPELL_BREAK_BONDS, false);
129 events.RepeatEvent(urand(16000, 22000));
130 break;
131 case EVENT_SPELL_CHAIN_HEAL:
132 if (ObjectGuid TargetGUID = GetChainHealTargetGUID())
133 if (pInstance)
134 {
135 if (Creature* target = pInstance->instance->GetCreature(TargetGUID))
136 me->CastSpell(target, SPELL_CHAIN_HEAL, false);
137
138 Creature* pGuard1 = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_EREKEM_GUARD_1_GUID));
139 Creature* pGuard2 = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_EREKEM_GUARD_2_GUID));
140 if ((pGuard1 && !pGuard1->IsAlive()) || (pGuard2 && !pGuard2->IsAlive()))
141 {
142 events.RepeatEvent(urand(3000, 6000));
143 break;
144 }
145 }
146 events.RepeatEvent(urand(8000, 11000));
147 break;
148 case EVENT_SPELL_EARTH_SHIELD:
149 me->CastSpell(me, SPELL_EARTH_SHIELD, false);
150 events.RepeatEvent(20000);
151 break;
152 case EVENT_SPELL_EARTH_SHOCK:
153 me->CastSpell(me->GetVictim(), SPELL_EARTH_SHOCK, false);
154 events.RepeatEvent(urand(8000, 13000));
155 break;
156 case EVENT_SPELL_LIGHTNING_BOLT:
157 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 35.0f, true))
158 me->CastSpell(target, SPELL_LIGHTNING_BOLT, false);
159 events.RepeatEvent(urand(15000, 25000));
160 break;
161 case EVENT_SPELL_STORMSTRIKE:
162 {
163 Creature* pGuard1 = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_EREKEM_GUARD_1_GUID));
164 Creature* pGuard2 = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_EREKEM_GUARD_2_GUID));
165 if (pGuard1 && !pGuard1->IsAlive() && pGuard2 && !pGuard2->IsAlive()) // both dead
166 me->CastSpell(me->GetVictim(), SPELL_STORMSTRIKE, false);
167 events.RepeatEvent(3000);
168 }
169 break;
170 }
171
172 DoMeleeAttackIfReady();
173 }
174
175 void JustDied(Unit* /*killer*/) override

179 pInstance->SetData(DATA_BOSS_DIED, 0);

182 void KilledUnit(Unit* victim) override

184 if (victim && victim->GetGUID() == me->GetGUID())
185 return;
186 Talk(SAY_SLAY);
187 }
188
189 void MoveInLineOfSight(Unit* /*who*/) override {}
190
191 void EnterEvadeMode(EvadeReason why) override
192 {
193 ScriptedAI::EnterEvadeMode(why);
194 events.Reset();
195 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
196 if (pInstance)
197 pInstance->SetData(DATA_FAILED, 1);
198 }
199
200 ObjectGuid GetChainHealTargetGUID()
201 {
202 if (HealthBelowPct(85))
203 return me->GetGUID();
204
205 if (pInstance)
206 {
207 if (Creature* c = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_EREKEM_GUARD_1_GUID)))
208 if (c->IsAlive() && !c->HealthAbovePct(75))
209 return c->GetGUID();
210
211 if (Creature* c = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_EREKEM_GUARD_2_GUID)))
212 if (c->IsAlive() && !c->HealthAbovePct(75))
213 return c->GetGUID();
214 }
215
216 return me->GetGUID();
217 }
