115 events.ScheduleEvent(EVENT_SPELL_CRUSHING_SHADOWS, 17000);

132

136 Talk(SAY_DEATH);
137 me->CastSpell(me, SPELL_SHADOW_OF_DEATH_REMOVE, true);
138 }
139
140 void UpdateAI(uint32 diff) override
141 {
142 if (!UpdateVictim() )
143 return;
144
145 events.Update(diff);
146 if (me->HasUnitState(UNIT_STATE_CASTING))
147 return;
148
149 switch (events.ExecuteEvent())
150 {
151 case EVENT_SPELL_INCINERATE:
152 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
153 {
154 if (roll_chance_i(50))
155 Talk(SAY_INCINERATE);
156 me->CastSpell(target, SPELL_INCINERATE, false);
157 }
158 events.ScheduleEvent(EVENT_SPELL_INCINERATE, 25000);
159 break;
160 case EVENT_SPELL_DOOM_BLOSSOM:
161 if (roll_chance_i(50))
162 Talk(SAY_BLOSSOM);
163
164 me->CastSpell(me, SPELL_SUMMON_DOOM_BLOSSOM, false);
165 events.ScheduleEvent(EVENT_SPELL_DOOM_BLOSSOM, 40000);
166 break;
167 case EVENT_SPELL_CRUSHING_SHADOWS:
168 if (roll_chance_i(20))
169 Talk(SAY_CRUSHING);
170 me->CastCustomSpell(SPELL_CRUSHING_SHADOWS, SPELLVALUE_MAX_TARGETS, 5, me, false);
171 events.ScheduleEvent(EVENT_SPELL_CRUSHING_SHADOWS, 15000);
172 break;
173 case EVENT_SPELL_SHADOW_OF_DEATH:
174 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, ShadowOfDeathSelector()))
175 me->CastSpell(target, SPELL_SHADOW_OF_DEATH, false);
176 events.ScheduleEvent(EVENT_SPELL_SHADOW_OF_DEATH, 30000);
177 break;
178 }
179
180 DoMeleeAttackIfReady();
181 }
182
183 bool CheckEvadeIfOutOfCombatArea() const override
184 {
185 return me->GetDistance(me->GetHomePosition()) > 100.0f;
186 }
