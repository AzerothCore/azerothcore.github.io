73bool ChatHandler::HasLowerSecurityAccount(WorldSession* target, uint32 target_account, bool strong)
74{
75 uint32 target_sec;
76
77 // allow everything from console and RA console
78 if (!m_session)
79 return false;
80
81 // ignore only for non-players for non strong checks (when allow apply command at least to same sec level)
82 if (!AccountMgr::IsPlayerAccount(m_session->GetSecurity()) && !strong && !sWorld->getBoolConfig(CONFIG_GM_LOWER_SECURITY))
83 return false;
84
85 if (target)
86 target_sec = target->GetSecurity();
87 else if (target_account)
88 target_sec = AccountMgr::GetSecurity(target_account, realm.Id.Realm);
89 else
90 return true; // caller must report error for (target == nullptr && target_account == 0)
91
92 AccountTypes target_ac_sec = AccountTypes(target_sec);
93 if (m_session->GetSecurity() < target_ac_sec || (strong && m_session->GetSecurity() <= target_ac_sec))
94 {
95 SendSysMessage(LANG_YOURS_SECURITY_IS_LOW);
96 SetSentErrorMessage(true);
97 return true;
98 }
99
100 return false;
101}

103void ChatHandler::SendSysMessage(std::string_view str, bool escapeCharacters)
104{
105 std::string msg{ str };
106
107 // Replace every "|" with "||" in msg
108 if (escapeCharacters && msg.find('|') != std::string::npos)
109 {
110 std::vector<std::string_view> tokens = Acore::Tokenize(msg, '|', true);
111 std::ostringstream stream;
112
113 for (size_t i = 0; i < tokens.size() - 1; ++i)
114 stream << tokens[i] << "||";
115
116 stream << tokens[tokens.size() - 1];
117
118 msg = stream.str();
119 }
120
121 WorldPacket data;
122 for (std::string_view line : Acore::Tokenize(str, '\n', true))
123 {
124 BuildChatPacket(data, CHAT_MSG_SYSTEM, LANG_UNIVERSAL, nullptr, nullptr, line);
125 m_session->SendPacket(&data);
126 }
127}

154bool ChatHandler::_ParseCommands(std::string_view text)
155{
156 if (Acore::ChatCommands::TryExecuteCommand(*this, text))
157 return true;
158
159 // Pretend commands don't exist for regular players
160 if (m_session && AccountMgr::IsPlayerAccount(m_session->GetSecurity()) && !sWorld->getBoolConfig(CONFIG_ALLOW_PLAYER_COMMANDS))
161 return false;
162
163 // Send error message for GMs
164 PSendSysMessage(LANG_CMD_INVALID, STRING_VIEW_FMT_ARG(text));
165 SetSentErrorMessage(true);
166 return true;
167}

169bool ChatHandler::ParseCommands(std::string_view text)
170{
171 ASSERT(!text.empty());
172
173 // chat case (.command or !command format)
174 if ((text[0] != '!') && (text[0] != '.'))
175 return false;
176
177 // ignore single . and ! in line
178 if (text.length() < 2)
179 return false;
180
181 // ignore messages staring from many dots.
182 if (text[1] == text[0])
183 return false;
184
185 // ignore messages with separator after .
186 if (text[1] == Acore::Impl::ChatCommands::COMMAND_DELIMITER)
187 return false;
188
189 return _ParseCommands(text.substr(1));
190}

192size_t ChatHandler::BuildChatPacket(WorldPacket& data, ChatMsg chatType, Language language, ObjectGuid senderGUID, ObjectGuid receiverGUID, std::string_view message, uint8 chatTag,
193 std::string const& senderName /*= ""*/, std::string const& receiverName /*= ""*/,
194 uint32 achievementId /*= 0*/, bool gmMessage /*= false*/, std::string const& channelName /*= ""*/)
195{
196 size_t receiverGUIDPos = 0;
197 data.Initialize(!gmMessage ? SMSG_MESSAGECHAT : SMSG_GM_MESSAGECHAT);
198 data << uint8(chatType);
199 data << int32(language);
200 data << senderGUID;
201 data << uint32(0); // some flags
202 switch (chatType)
203 {
204 case CHAT_MSG_MONSTER_SAY:
205 case CHAT_MSG_MONSTER_PARTY:
206 case CHAT_MSG_MONSTER_YELL:
207 case CHAT_MSG_MONSTER_WHISPER:
208 case CHAT_MSG_MONSTER_EMOTE:
209 case CHAT_MSG_RAID_BOSS_EMOTE:
210 case CHAT_MSG_RAID_BOSS_WHISPER:
211 case CHAT_MSG_BATTLENET:
212 data << uint32(senderName.length() + 1);
213 data << senderName;
214 receiverGUIDPos = data.wpos();
215 data << receiverGUID;
216 if (receiverGUID && !receiverGUID.IsPlayer() && !receiverGUID.IsPet())
217 {
218 data << uint32(receiverName.length() + 1);
219 data << receiverName;
220 }
221 break;
222 case CHAT_MSG_WHISPER_FOREIGN:
223 data << uint32(senderName.length() + 1);
224 data << senderName;
225 receiverGUIDPos = data.wpos();
226 data << receiverGUID;
227 break;
228 case CHAT_MSG_BG_SYSTEM_NEUTRAL:
229 case CHAT_MSG_BG_SYSTEM_ALLIANCE:
230 case CHAT_MSG_BG_SYSTEM_HORDE:
231 receiverGUIDPos = data.wpos();
232 data << receiverGUID;
233 if (receiverGUID && !receiverGUID.IsPlayer())
234 {
235 data << uint32(receiverName.length() + 1);
236 data << receiverName;
237 }
238 break;
239 case CHAT_MSG_ACHIEVEMENT:
240 case CHAT_MSG_GUILD_ACHIEVEMENT:
241 receiverGUIDPos = data.wpos();
242 data << receiverGUID;
243 break;
244 default:
245 if (gmMessage)
246 {
247 data << uint32(senderName.length() + 1);
248 data << senderName;
249 }
250
251 if (chatType == CHAT_MSG_CHANNEL)
252 {
253 ASSERT(channelName.length() > 0);
254 data << channelName;
255 }
256
257 receiverGUIDPos = data.wpos();
258 data << receiverGUID;
259 break;
260 }
261
262 data << uint32(message.length() + 1);
263 data << message;
264 data << uint8(chatTag);
265
266 if (chatType == CHAT_MSG_ACHIEVEMENT || chatType == CHAT_MSG_GUILD_ACHIEVEMENT)
267 data << uint32(achievementId);
268
269 return receiverGUIDPos;
270}

272size_t ChatHandler::BuildChatPacket(WorldPacket& data, ChatMsg chatType, Language language, WorldObject const* sender, WorldObject const* receiver, std::string_view message,
273 uint32 achievementId /*= 0*/, std::string const& channelName /*= ""*/, LocaleConstant locale /*= DEFAULT_LOCALE*/)
274{
275 ObjectGuid senderGUID;
276 std::string senderName = "";
277 uint8 chatTag = 0;
278 bool gmMessage = false;
279 ObjectGuid receiverGUID;
280 std::string receiverName = "";
281 if (sender)
282 {
283 senderGUID = sender->GetGUID();
284 senderName = sender->GetNameForLocaleIdx(locale);
285 if (Player const* playerSender = sender->ToPlayer())
286 {
287 chatTag = playerSender->GetChatTag();
288 gmMessage = playerSender->IsGameMaster();
289 }
290 }
291
292 if (receiver)
293 {
294 receiverGUID = receiver->GetGUID();
295 receiverName = receiver->GetNameForLocaleIdx(locale);
296 }
297
298 return BuildChatPacket(data, chatType, language, senderGUID, receiverGUID, message, chatTag, senderName, receiverName, achievementId, gmMessage, channelName);
299}

363char* ChatHandler::extractKeyFromLink(char* text, char const* linkType, char** something1)
364{
365 // skip empty
366 if (!text)
367 return nullptr;
368
369 // skip spaces
370 while (*text == ' ' || *text == '\t' || *text == '\b')
371 ++text;
372
373 if (!*text)
374 return nullptr;
375
376 // return non link case
377 if (text[0] != '|')
378 return strtok(text, " ");
379
380 // [name] Shift-click form |color|linkType:key|h[name]|h|r
381 // or
382 // [name] Shift-click form |color|linkType:key:something1:...:somethingN|h[name]|h|r
383
384 char* check = strtok(text, "|"); // skip color
385 if (!check)
386 return nullptr; // end of data
387
388 char* cLinkType = strtok(nullptr, ":"); // linktype
389 if (!cLinkType)
390 return nullptr; // end of data
391
392 if (strcmp(cLinkType, linkType) != 0)
393 {
394 strtok(nullptr, " "); // skip link tail (to allow continue strtok(nullptr, s) use after retturn from function
395 SendSysMessage(LANG_WRONG_LINK_TYPE);
396 return nullptr;
397 }
398
399 char* cKeys = strtok(nullptr, "|"); // extract keys and values
400 char* cKeysTail = strtok(nullptr, "");
401
402 char* cKey = strtok(cKeys, ":|"); // extract key
403 if (something1)
404 *something1 = strtok(nullptr, ":|"); // extract something
405
406 strtok(cKeysTail, "]"); // restart scan tail and skip name with possible spaces
407 strtok(nullptr, " "); // skip link tail (to allow continue strtok(nullptr, s) use after return from function
408 return cKey;
409}

411char* ChatHandler::extractKeyFromLink(char* text, char const* const* linkTypes, int* found_idx, char** something1)
412{
413 // skip empty
414 if (!text)
415 return nullptr;
416
417 // skip spaces
418 while (*text == ' ' || *text == '\t' || *text == '\b')
419 ++text;
420
421 if (!*text)
422 return nullptr;
423
424 // return non link case
425 if (text[0] != '|')
426 return strtok(text, " ");
427
428 // [name] Shift-click form |color|linkType:key|h[name]|h|r
429 // or
430 // [name] Shift-click form |color|linkType:key:something1:...:somethingN|h[name]|h|r
431 // or
432 // [name] Shift-click form |linkType:key|h[name]|h|r
433
434 char* tail;
435
436 if (text[1] == 'c')
437 {
438 char* check = strtok(text, "|"); // skip color
439 if (!check)
440 return nullptr; // end of data
441
442 tail = strtok(nullptr, ""); // tail
443 }
444 else
445 tail = text + 1; // skip first |
446
447 char* cLinkType = strtok(tail, ":"); // linktype
448 if (!cLinkType)
449 return nullptr; // end of data
450
451 for (int i = 0; linkTypes[i]; ++i)
452 {
453 if (strcmp(cLinkType, linkTypes[i]) == 0)
454 {
455 char* cKeys = strtok(nullptr, "|"); // extract keys and values
456 char* cKeysTail = strtok(nullptr, "");
457
458 char* cKey = strtok(cKeys, ":|"); // extract key
459 if (something1)
460 *something1 = strtok(nullptr, ":|"); // extract something
461
462 strtok(cKeysTail, "]"); // restart scan tail and skip name with possible spaces
463 strtok(nullptr, " "); // skip link tail (to allow continue strtok(nullptr, s) use after return from function
464 if (found_idx)
465 *found_idx = i;
466 return cKey;
467 }
468 }
469
470 strtok(nullptr, " "); // skip link tail (to allow continue strtok(nullptr, s) use after return from function
471 SendSysMessage(LANG_WRONG_LINK_TYPE);
472 return nullptr;
473}

538uint32 ChatHandler::extractSpellIdFromLink(char* text)
539{
540 // number or [name] Shift-click form |color|Henchant:recipe_spell_id|h[prof_name: recipe_name]|h|r
541 // number or [name] Shift-click form |color|Hglyph:glyph_slot_id:glyph_prop_id|h[%s]|h|r
542 // number or [name] Shift-click form |color|Hspell:spell_id|h[name]|h|r
543 // number or [name] Shift-click form |color|Htalent:talent_id, rank|h[name]|h|r
544 // number or [name] Shift-click form |color|Htrade:spell_id, skill_id, max_value, cur_value|h[name]|h|r
545 int type = 0;
546 char* param1_str = nullptr;
547 char* idS = extractKeyFromLink(text, spellKeys, &type, &param1_str);
548 if (!idS)
549 return 0;
550
551 uint32 id = (uint32)atol(idS);
552
553 switch (type)
554 {
555 case SPELL_LINK_SPELL:
556 return id;
557 case SPELL_LINK_TALENT:
558 {
559 // talent
560 TalentEntry const* talentEntry = sTalentStore.LookupEntry(id);
561 if (!talentEntry)
562 return 0;
563
564 int32 rank = param1_str ? (uint32)atol(param1_str) : 0;
565 if (rank >= MAX_TALENT_RANK)
566 return 0;
567
568 if (rank < 0)
569 rank = 0;
570
571 return talentEntry->RankID[rank];
572 }
573 case SPELL_LINK_ENCHANT:
574 case SPELL_LINK_TRADE:
575 return id;
576 case SPELL_LINK_GLYPH:
577 {
578 uint32 glyph_prop_id = param1_str ? (uint32)atol(param1_str) : 0;
579
580 GlyphPropertiesEntry const* glyphPropEntry = sGlyphPropertiesStore.LookupEntry(glyph_prop_id);
581 if (!glyphPropEntry)
582 return 0;
583
584 return glyphPropEntry->SpellId;
585 }
586 }
587
588 // unknown type?
589 return 0;
590}

607ObjectGuid::LowType ChatHandler::extractLowGuidFromLink(char* text, HighGuid& guidHigh)
608{
609 int type = 0;
610
611 // |color|Hcreature:creature_guid|h[name]|h|r
612 // |color|Hgameobject:go_guid|h[name]|h|r
613 // |color|Hplayer:name|h[name]|h|r
614 char* idS = extractKeyFromLink(text, guidKeys, &type);
615 if (!idS)
616 return 0;
617
618 switch (type)
619 {
620 case SPELL_LINK_PLAYER:
621 {
622 guidHigh = HighGuid::Player;
623
624 std::string name = idS;
625 if (!normalizePlayerName(name))
626 return 0;
627
628 if (Player* player = ObjectAccessor::FindPlayerByName(name, false))
629 return player->GetGUID().GetCounter();
630
631 if (ObjectGuid guid = sCharacterCache->GetCharacterGuidByName(name))
632 return guid.GetCounter();
633
634 return 0;
635 }
636 case SPELL_LINK_CREATURE:
637 {
638 guidHigh = HighGuid::Unit;
639
640 ObjectGuid::LowType lowguid = (uint32)atol(idS);
641
642 if (sObjectMgr->GetCreatureData(lowguid))
643 return lowguid;
644 else
645 return 0;
646 }
647 case SPELL_LINK_GAMEOBJECT:
648 {
649 guidHigh = HighGuid::GameObject;
650
651 ObjectGuid::LowType lowguid = (uint32)atol(idS);
652
653 if (sObjectMgr->GetGOData(lowguid))
654 return lowguid;
655 else
656 return 0;
657 }
658 }
659
660 // unknown type?
661 return 0;
662}

678bool ChatHandler::extractPlayerTarget(char* args, Player** player, ObjectGuid* player_guid /*=nullptr*/, std::string* player_name /*= nullptr*/)
679{
680 if (args && *args)
681 {
682 std::string name = extractPlayerNameFromLink(args);
683 if (name.empty())
684 {
685 SendSysMessage(LANG_PLAYER_NOT_FOUND);
686 SetSentErrorMessage(true);
687 return false;
688 }
689
690 Player* pl = ObjectAccessor::FindPlayerByName(name, false);
691
692 // if allowed player pointer
693 if (player)
694 *player = pl;
695
696 // if need guid value from DB (in name case for check player existence)
697 ObjectGuid guid = !pl && (player_guid || player_name) ? sCharacterCache->GetCharacterGuidByName(name) : ObjectGuid::Empty;
698
699 // if allowed player guid (if no then only online players allowed)
700 if (player_guid)
701 *player_guid = pl ? pl->GetGUID() : guid;
702
703 if (player_name)
704 *player_name = pl || guid ? name : "";
705 }
706 else
707 {
708 // populate strtok buffer to prevent crashes
709 static char dummy[1] = "";
710 strtok(dummy, "");
711
712 Player* pl = getSelectedPlayer();
713 // if allowed player pointer
714 if (player)
715 *player = pl;
716
717 // if allowed player guid (if no then only online players allowed)
718 if (player_guid)
719 *player_guid = pl ? pl->GetGUID() : ObjectGuid::Empty;
720
721 if (player_name)
722 *player_name = pl ? pl->GetName() : "";
723 }
724
725 // some from req. data must be provided (note: name is empty if player not exist)
726 if ((!player || !*player) && (!player_guid || !*player_guid) && (!player_name || player_name->empty()))
727 {
728 SendSysMessage(LANG_PLAYER_NOT_FOUND);
729 SetSentErrorMessage(true);
730 return false;
731 }
732
733 return true;
734}

736char* ChatHandler::extractQuotedArg(char* args)
737{
738 if (!*args)
739 return nullptr;
740
741 if (*args == '"')
742 return strtok(args + 1, "\"");
743 else
744 {
745 // skip spaces
746 while (*args == ' ')
747 {
748 args += 1;
749 continue;
750 }
751
752 // return nullptr if we reached the end of the string
753 if (!*args)
754 return nullptr;
755
756 // since we skipped all spaces, we expect another token now
757 if (*args == '"')
758 {
759 // return an empty string if there are 2 "" in a row.
760 // strtok doesn't handle this case
761 if (*(args + 1) == '"')
762 {
763 strtok(args, " ");
764 static char arg[1];
765 arg[0] = '\0';
766 return arg;
767 }
768 else
769 return strtok(args + 1, "\"");
770 }
771 else
772 return nullptr;
773 }

832 player = nullptr;

848 if (offline)

852 }
853 }
854
