74 void Initialize()

82 void Reset() override
83 {
84 _Reset();
85 Initialize();
86
87 me->SetStandState(UNIT_STAND_STATE_STAND);
88 me->SetReactState(REACT_AGGRESSIVE);
89 me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
90 me->LoadEquipment(1, true);
91
92 if (Creature* zealot = instance->GetCreature(DATA_LORKHAN))
93 {
94 zealot->AI()->Reset();
95 }
96
97 if (Creature* zealot = instance->GetCreature(DATA_ZATH))
98 {
99 zealot->AI()->Reset();
100 }
101
102 _scheduler.SetValidator([this]
103 {
104 return !me->HasUnitState(UNIT_STATE_CASTING);
105 });
106 }
107
108 void JustDied(Unit* /*killer*/) override
109 {
110 _JustDied();
111 Talk(SAY_DEATH);
112
113 if (Creature* zealot = instance->GetCreature(DATA_LORKHAN))
114 {
115 zealot->Kill(zealot, zealot);
116 }
117
118 if (Creature* zealot = instance->GetCreature(DATA_ZATH))
119 {
120 zealot->Kill(zealot, zealot);
121 }
122 }
123
124 void EnterCombat(Unit* /*who*/) override
125 {
126 _EnterCombat();
127
128 _scheduler.CancelAll();
129 _scheduler.Schedule(4s, [this](TaskContext context) {
130 DoCastVictim(SPELL_MORTALCLEAVE);
131 context.Repeat(15s, 20s);
132 }).Schedule(9s, [this](TaskContext context) {
133 DoCastVictim(SPELL_SILENCE);
134 context.Repeat(20s, 25s);
135 }).Schedule(16s, [this](TaskContext context) {
136 DoCastSelf(SPELL_BLOODLUST);
137 context.Repeat(20s, 28s);
138 });
139 }
140
141 void SetData(uint32 /*type*/, uint32 data) override
142 {
143 UpdateZealotStatus(data, true);
144 CheckPhaseTransition();
145
146 _scheduler.Schedule(10s, [this, data](TaskContext /*context*/) {
147 if (!_lorkhanDied || !_zathDied || !WasDead)
148 {
149 ReviveZealot(data);
150 }
151 });
152 }
153
154 void DamageTaken(Unit* /*attacker*/, uint32& damage, DamageEffectType, SpellSchoolMask) override
155 {
156 if (!me->HasAura(SPELL_TIGER_FORM) && damage >= me->GetHealth())
157 {
158 damage = me->GetHealth() - 1;
159
160 if (!WasDead)
161 {
162 me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
163 me->SetReactState(REACT_PASSIVE);
164 me->SetStandState(UNIT_STAND_STATE_SLEEP);
165 me->AttackStop();
166 DoResetThreat();
167 WasDead = true;
168 CheckPhaseTransition();
169 Talk(EMOTE_THEKAL_DIES);
170 }
171 }
172
173 if (!Enraged && me->HealthBelowPctDamaged(20, damage) && me->HasAura(SPELL_TIGER_FORM))
174 {
175 DoCastSelf(SPELL_ENRAGE);
176 Enraged = true;
177 }
178 }
179
180 void DoAction(int32 action) override
181 {
182 if (action == ACTION_RESSURRECT)
183 {
184 me->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
185 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
186 me->RestoreFaction();
187 me->SetReactState(REACT_AGGRESSIVE);
188 me->SetFullHealth();
189 WasDead = false;
190 }
191 }
192
193 void UpdateAI(uint32 diff) override
194 {
195 if (me->GetReactState() != REACT_PASSIVE && !UpdateVictim())
196 return;
197
198 _scheduler.Update(diff,
199 std::bind(&BossAI::DoMeleeAttackIfReady, this));
200 }
201
202 void ReviveZealot(uint32 zealotData)
203 {
204 if (Creature* zealot = instance->GetCreature(zealotData))
205 {
206 zealot->Respawn(true);
207 zealot->SetInCombatWithZone();
208 UpdateZealotStatus(zealotData, false);
209 }
210 }
211
212 void UpdateZealotStatus(uint32 data, bool dead)
213 {
214 if (data == DATA_LORKHAN)
215 {
216 _lorkhanDied = dead;
217 }
218 else if (data == DATA_ZATH)
219 {
220 _zathDied = dead;
221 }
222 }
223
224 void CheckPhaseTransition()
225 {
226 if (WasDead && _lorkhanDied && _zathDied)
227 {
228 _scheduler.Schedule(3s, [this](TaskContext /*context*/) {
229 Talk(SAY_AGGRO);
230 me->SetStandState(UNIT_STAND_STATE_STAND);
231 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
232
233 _scheduler.Schedule(6s, [this](TaskContext /*context*/) {
234 DoCastSelf(SPELL_TIGER_FORM);
235 me->LoadEquipment(0, true);
236 me->SetReactState(REACT_AGGRESSIVE);
237
238 _scheduler.Schedule(30s, [this](TaskContext context) {
239 DoCastSelf(SPELL_FRENZY);
240 context.Repeat();
241 }).Schedule(4s, [this](TaskContext context) {
242 DoCastVictim(SPELL_FORCEPUNCH);
243 context.Repeat(16s, 21s);
244 }).Schedule(12s, [this](TaskContext context) {
245 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
246 {
247 DoCast(target, SPELL_CHARGE);
248 DoResetThreat();
249 AttackStart(target);
250 }
251 context.Repeat(15s, 22s);
252 }).Schedule(25s, [this](TaskContext context) {
253 DoCastVictim(SPELL_SUMMONTIGERS, true);
254 context.Repeat(10s, 14s);
255 });
256 });
257 });
258 }
259 else
260 {
261 _scheduler.Schedule(10s, [this](TaskContext /*context*/) {
262 if (!(WasDead && _lorkhanDied && _zathDied))
263 {
264 DoAction(ACTION_RESSURRECT);
265 }
266 });
267 }
268 }
269
270 private:
271 TaskScheduler _scheduler;
272 GuidVector _catGuids;
273 bool _lorkhanDied;
274 bool _zathDied;
275 bool Enraged;
276 bool WasDead;
277 };
278
279 CreatureAI* GetAI(Creature* creature) const override
280 {
281 return GetZulGurubAI<boss_thekalAI>(creature);
282 }
