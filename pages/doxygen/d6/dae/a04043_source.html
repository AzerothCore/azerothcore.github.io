102 void UpdateAI(uint32 diff) override
103 {
104 if (!UpdateVictim())
105 return;
106
107 events.Update(diff);
108 if (me->HasUnitState(UNIT_STATE_CASTING))
109 return;
110
111 switch (events.ExecuteEvent())
112 {
113 case EVENT_SACRIFICE:
114 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 0.0f, true))
115 {
116 Talk(SAY_CAST_SACRIFICE);
117 me->CastSpell(target, SPELL_SACRIFICE, false);
118 }
119 events.ScheduleEvent(EVENT_SACRIFICE, 30000);
120 break;
121 case EVENT_HELLFIRE:
122 if (roll_chance_i(50))
123 Talk(SAY_CAST_HELLFIRE);
124 me->CastSpell(me, SPELL_HELLFIRE, false);
125 events.ScheduleEvent(EVENT_HELLFIRE, 22000);
126 break;
127 case EVENT_ENRAGE:
128 Talk(EMOTE_ENRAGE);
129 me->CastSpell(me, SPELL_ENRAGE, false);
130 events.ScheduleEvent(EVENT_ENRAGE, 30000);
131 break;
132 case EVENT_HEALTH_CHECK_50:
133 if (me->HealthBelowPct(50))
134 {
135 Talk(SAY_50_PERCENT_HP);
136 break;
137 }
138 events.ScheduleEvent(EVENT_HEALTH_CHECK_50, 500);
139 break;
140 case EVENT_HEALTH_CHECK_20:
141 if (me->HealthBelowPct(20))
142 {
143 Talk(SAY_20_PERCENT_HP);
144 break;
145 }
146 events.ScheduleEvent(EVENT_HEALTH_CHECK_20, 500);
147 break;
148 }
149
150 DoMeleeAttackIfReady();
151 }
