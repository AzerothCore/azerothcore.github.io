98 void UpdateAI(uint32 diff) override
99 {
100 if (!UpdateVictim())
101 return;
102
103 events.Update(diff);
104
105 if (me->HasUnitState(UNIT_STATE_CASTING))
106 return;
107
108 while (uint32 eventId = events.ExecuteEvent())
109 {
110 switch (eventId)
111 {
112 case EVENT_MORTAL_WOUND:
113 DoCastVictim(SPELL_MORTAL_WOUND);
114 events.ScheduleEvent(EVENT_MORTAL_WOUND, 8s, 10s);
115 break;
116 case EVENT_SAND_TRAP:
117 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 100.f, true))
118 {
119 target->CastSpell(target, SPELL_SAND_TRAP, true, nullptr, nullptr, me->GetGUID());
120 }
121 events.ScheduleEvent(EVENT_SAND_TRAP, 5s, 15s);
122 break;
123 case EVENT_WIDE_SLASH:
124 DoCastSelf(SPELL_WIDE_SLASH);
125 events.ScheduleEvent(EVENT_WIDE_SLASH, 12s, 15s);
126 break;
127 case EVENT_THRASH:
128 DoCastSelf(SPELL_THRASH);
129 events.ScheduleEvent(EVENT_THRASH, 16s);
130 break;
131 default:
132 break;
133 }
134 }
135
136 DoMeleeAttackIfReady();
137 }
138private:
139 bool _enraged;
140};
