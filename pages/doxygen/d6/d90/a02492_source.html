102 struct boss_thebeastAI : public BossAI
103 {
104 boss_thebeastAI(Creature* creature) : BossAI(creature, DATA_THE_BEAST), _beastReached(false), _orcYelled(false) {}
105
106 void Reset() override
107 {
108 _Reset();
109
110 if (_beastReached)
111 {
112 me->GetMotionMaster()->MovePath(BEAST_MOVEMENT_ID, true);
113 }
114 }
115
116 void EnterCombat(Unit* /*who*/) override
117 {
118 _EnterCombat();
119 events.ScheduleEvent(EVENT_FLAME_BREAK, 12 * IN_MILLISECONDS);
120 events.ScheduleEvent(EVENT_IMMOLATE, 3 * IN_MILLISECONDS);
121 events.ScheduleEvent(EVENT_TERRIFYING_ROAR, 23 * IN_MILLISECONDS);
122 events.ScheduleEvent(EVENT_BERSERKER_CHARGE, 2 * IN_MILLISECONDS);
123 events.ScheduleEvent(EVENT_FIREBALL, 8 * IN_MILLISECONDS, 21 * IN_MILLISECONDS);
124 events.ScheduleEvent(EVENT_FIREBLAST, 5 * IN_MILLISECONDS, 8 * IN_MILLISECONDS);
125 }
126
127 void SetData(uint32 type, uint32 /*data*/) override
128 {
129 switch (type)
130 {
131 case DATA_BEAST_ROOM:
132 {
133 if (!_orcYelled)
134 {
135 if (_nearbyOrcsGUIDs.empty())
136 {
137 FindNearbyOrcs();
138 }
139
141 if (_nearbyOrcsGUIDs.empty())
142 {
143 return;
144 }
145
146 _orcYelled = true;
147
148 bool yelled = false;
149 for (ObjectGuid guid : _nearbyOrcsGUIDs)
150 {
151 if (Creature* orc = ObjectAccessor::GetCreature(*me, guid))
152 {
153 if (!yelled)
154 {
155 yelled = true;
156 orc->AI()->Talk(SAY_BLACKHAND_DOOMED);
157 }
158
159 orc->m_Events.AddEvent(new OrcMoveEvent(orc), me->m_Events.CalculateTime(3 * IN_MILLISECONDS));
160 orc->m_Events.AddEvent(new OrcDeathEvent(orc), me->m_Events.CalculateTime(9 * IN_MILLISECONDS));
161 }
162 }
163 }
164 break;
165 }
166 case DATA_BEAST_REACHED:
167 {
168 if (!_beastReached)
169 {
170 _beastReached = true;
171 me->GetMotionMaster()->MovePath(BEAST_MOVEMENT_ID, true);
172
173 // There is a chance player logged in between areatriggers (realm crash or restart)
174 // executing part of script which happens when player enters boss room
175 // otherwise we will see weird behaviour when someone steps on the previous areatrigger (dead mob yelling/moving)
176 SetData(DATA_BEAST_ROOM, DATA_BEAST_ROOM);
177 }
178 break;
179 }
180 }
181 }
182
183 void UpdateAI(uint32 diff) override
184 {
185 if (!UpdateVictim())
186 {
187 return;
188 }
189
190 if (me->GetPositionY() > FirewalPositionY)
191 {
192 EnterEvadeMode();
193 return;
194 }
195
196 events.Update(diff);
197
198 if (me->HasUnitState(UNIT_STATE_CASTING))
199 {
200 return;
201 }
202
203 while (uint32 eventId = events.ExecuteEvent())
204 {
205 switch (eventId)
206 {
207 case EVENT_FLAME_BREAK:
208 DoCastVictim(SPELL_FLAMEBREAK);
209 events.ScheduleEvent(EVENT_FLAME_BREAK, 10 * IN_MILLISECONDS);
210 break;
211 case EVENT_IMMOLATE:
212 DoCastRandomTarget(SPELL_IMMOLATE, 0, 100.0f);
213 events.ScheduleEvent(EVENT_IMMOLATE, 8 * IN_MILLISECONDS);
214 break;
215 case EVENT_TERRIFYING_ROAR:
216 DoCastVictim(SPELL_TERRIFYINGROAR);
217 events.ScheduleEvent(EVENT_TERRIFYING_ROAR, 20 * IN_MILLISECONDS);
218 break;
219 case EVENT_BERSERKER_CHARGE:
220 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 38.f, true))
221 {
222 DoCast(target, SPELL_BERSERKER_CHARGE);
223 }
224 events.ScheduleEvent(EVENT_BERSERKER_CHARGE, 15 * IN_MILLISECONDS, 23 * IN_MILLISECONDS);
225 break;
226 case EVENT_FIREBALL:
227 DoCastVictim(SPELL_FIREBALL);
228 events.ScheduleEvent(EVENT_FIREBALL, 8 * IN_MILLISECONDS, 21 * IN_MILLISECONDS);
229 if (events.GetNextEventTime(EVENT_FIREBLAST) < 3 * IN_MILLISECONDS)
230 {
231 events.RescheduleEvent(EVENT_FIREBLAST, 3 * IN_MILLISECONDS);
232 }
233 break;
234 case EVENT_FIREBLAST:
235 DoCastVictim(SPELL_FIREBLAST);
236 events.ScheduleEvent(EVENT_FIREBLAST, 5 * IN_MILLISECONDS, 8 * IN_MILLISECONDS);
237 if (events.GetNextEventTime(EVENT_FIREBALL) < 3 * IN_MILLISECONDS)
238 {
239 events.RescheduleEvent(EVENT_FIREBALL, 3 * IN_MILLISECONDS);
240 }
241 break;
242 }
243
244 if (me->HasUnitState(UNIT_STATE_CASTING))
245 {
246 return;
247 }
248 }
249
250 DoMeleeAttackIfReady();
251 }
252
253 void FindNearbyOrcs()
254 {
255 std::list<Creature*> temp;
256 me->GetCreatureListWithEntryInGrid(temp, NPC_BLACKHAND_ELITE, 50.0f);
257 for (Creature* creature : temp)
258 {
259 if (creature->IsAlive())
260 {
261 _nearbyOrcsGUIDs.push_back(creature->GetGUID());
262 }
263 }
264 }
265
266 private:
267 bool _beastReached;
268 bool _orcYelled;
269 GuidVector _nearbyOrcsGUIDs;
270 };
271};
