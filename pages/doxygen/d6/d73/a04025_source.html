178 events2.ScheduleEvent(EVENT_PREFIGHT_8, 4000);

180 case EVENT_PREFIGHT_8:

184 case EVENT_PREFIGHT_9:

228 break;
229 case EVENT_FELFIRE:
230 me->CastSpell(me, SPELL_FELFIRE, true);
231 break;
232 case EVENT_ME_FIRST:
233 if (Creature* dalliah = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DALLIAH)))
234 if (dalliah->IsAlive() && !dalliah->IsInCombat())
235 dalliah->AI()->Talk(SAY_AGGRO_SOCCOTHRATES_FIRST);

240 if (Creature* dalliah = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_DALLIAH)))
241 dalliah->AI()->Talk(SAY_SOCCOTHRATES_25_PERCENT);

243 }
