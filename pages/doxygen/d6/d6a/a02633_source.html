103 void Reset() override

131 void EnterCombat(Unit* /*who*/) override
132 {

134 me->SetInCombatWithZone();

136 events.ScheduleEvent(EVENT_SPELL_FIREBALL, 0);

145 void MoveInLineOfSight(Unit* who) override
146 {

154 ScriptedAI::MoveInLineOfSight(who);

157 void DamageTaken(Unit*, uint32& damage, DamageEffectType, SpellSchoolMask) override
158 {
159 if (damage >= me->GetHealth())
160 {
161 damage = me->GetHealth() - 1;
162 if (me->isRegeneratingHealth())
163 {
164 me->SetRegeneratingHealth(false);
165 me->SetUnitFlag(UNIT_FLAG_DISABLE_MOVE);
166 me->SetImmuneToAll(true);
167 me->CombatStop();
168 me->SetReactState(REACT_PASSIVE);
169 LapseAction(ACTION_REMOVE_FLY);
170 events.Reset();
171 events2.ScheduleEvent(EVENT_FINISH_TALK, 6000);
172 Talk(SAY_DEATH);
173 }
174 }
175 }
176
177 void LapseAction(uint8 action)
178 {
179 uint8 counter = 0;
180 Map::PlayerList const& playerList = me->GetMap()->GetPlayers();
181 for (Map::PlayerList::const_iterator itr = playerList.begin(); itr != playerList.end(); ++itr, ++counter)
182 if (Player* player = itr->GetSource())
183 {
184 if (action == ACTION_TELEPORT_PLAYERS)
185 me->CastSpell(player, SPELL_GRAVITY_LAPSE_PLAYER + counter, true);
186 else if (action == ACTION_KNOCKUP)
187 player->CastSpell(player, SPELL_GRAVITY_LAPSE_DOT, true, nullptr, nullptr, me->GetGUID());
188 else if (action == ACTION_ALLOW_FLY)
189 player->CastSpell(player, SPELL_GRAVITY_LAPSE_FLY, true, nullptr, nullptr, me->GetGUID());
190 else if (action == ACTION_REMOVE_FLY)
191 {
192 player->RemoveAurasDueToSpell(SPELL_GRAVITY_LAPSE_FLY);
193 player->RemoveAurasDueToSpell(SPELL_GRAVITY_LAPSE_DOT);
194 }
195 }
196 }

198 void UpdateAI(uint32 diff) override
199 {
200 events2.Update(diff);
201 switch (events2.ExecuteEvent())
202 {
203 case EVENT_INIT_COMBAT:
204 me->SetImmuneToAll(false);
205 if (Unit* target = SelectTargetFromPlayerList(50.0f))
206 AttackStart(target);
207 return;
208 case EVENT_FINISH_TALK:
209 Unit::Kill(me, me);
210 return;
211 }
212
213 if (!UpdateVictim())
214 return;
215
216 events.Update(diff);
217 if (me->HasUnitState(UNIT_STATE_CASTING))
218 return;
219
220 switch (uint32 eventId = events.ExecuteEvent())
221 {
222 case EVENT_SPELL_FIREBALL:
223 me->CastSpell(me->GetVictim(), DUNGEON_MODE(SPELL_FIREBALL_N, SPELL_FIREBALL_H), false);
224 events.ScheduleEvent(EVENT_SPELL_FIREBALL, urand(3000, 4500));
225 break;
226 case EVENT_SPELL_FLAMESTRIKE:
227 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
228 {
229 me->CastSpell(target, SPELL_FLAMESTRIKE_SUMMON, true);
230 Talk(SAY_FLAMESTRIKE);
231 }
232 events.ScheduleEvent(EVENT_SPELL_FLAMESTRIKE, 25000);
233 break;
234 case EVENT_SPELL_SHOCK_BARRIER:
235 me->CastSpell(me, SPELL_SHOCK_BARRIER, true);
236 me->CastCustomSpell(SPELL_PYROBLAST, SPELLVALUE_MAX_TARGETS, 1, (Unit*)nullptr, false);
237 events.ScheduleEvent(EVENT_SPELL_SHOCK_BARRIER, 50000);
238 break;
239 case EVENT_SPELL_PHOENIX:
240 Talk(SAY_PHOENIX);
241 me->CastSpell(me, SPELL_PHOENIX, false);
242 events.ScheduleEvent(EVENT_SPELL_PHOENIX, 60000);
243 break;
244 case EVENT_CHECK_HEALTH:
245 if (HealthBelowPct(50))
246 {
247 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_INTERRUPT_CAST, true);
248 me->CastSpell(me, SPELL_TELEPORT_CENTER, true);
249 events.Reset();
250
251 me->StopMoving();
252 me->GetMotionMaster()->Clear();
253 me->GetMotionMaster()->MoveIdle();
254
255 events.SetPhase(1);
256 events.ScheduleEvent(EVENT_GRAVITY_LAPSE_1_1, 0);
257 break;
258 }
259 events.ScheduleEvent(EVENT_CHECK_HEALTH, 500);
260 break;
261 case EVENT_GRAVITY_LAPSE_1_1:
262 case EVENT_GRAVITY_LAPSE_1_2:
263 Talk(eventId == EVENT_GRAVITY_LAPSE_1_1 ? SAY_GRAVITY_LAPSE : SAY_RECAST_GRAVITY);
264 me->CastSpell(me, SPELL_GRAVITY_LAPSE_INITIAL, false);
265 events.ScheduleEvent(EVENT_GRAVITY_LAPSE_2, 2000);
266 break;
267 case EVENT_GRAVITY_LAPSE_2:
268 LapseAction(ACTION_TELEPORT_PLAYERS);
269 events.ScheduleEvent(EVENT_GRAVITY_LAPSE_3, 1000);
270 break;
271 case EVENT_GRAVITY_LAPSE_3:
272 LapseAction(ACTION_KNOCKUP);
273 events.ScheduleEvent(EVENT_GRAVITY_LAPSE_4, 1000);
274 break;
275 case EVENT_GRAVITY_LAPSE_4:
276 LapseAction(ACTION_ALLOW_FLY);
277 for (uint8 i = 0; i < 3; ++i)
278 me->CastSpell(me, SPELL_SUMMON_ARCANE_SPHERE, true);
279
280 me->CastSpell(me, SPELL_GRAVITY_LAPSE_CHANNEL, false);
281 events.ScheduleEvent(EVENT_GRAVITY_LAPSE_5, 30000);
282 break;
283 case EVENT_GRAVITY_LAPSE_5:
284 LapseAction(ACTION_REMOVE_FLY);
285 me->InterruptNonMeleeSpells(false);
286 Talk(SAY_TIRED);
287 me->CastSpell(me, SPELL_POWER_FEEDBACK, false);
288 events.ScheduleEvent(EVENT_GRAVITY_LAPSE_1_2, 10000);
289 break;
290 }
291
292 if (events.GetPhaseMask() == 0)
293 DoMeleeAttackIfReady();
294 }
295 };
