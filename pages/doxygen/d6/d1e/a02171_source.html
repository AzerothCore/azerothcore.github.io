74void UpdateTime::UpdateWithDiff(uint32 diff)
75{
76 _totalUpdateTime = _totalUpdateTime - _updateTimeDataTable[_updateTimeTableIndex] + diff;
77 _updateTimeDataTable[_updateTimeTableIndex] = diff;
78
79 if (diff > _maxUpdateTime)
80 _maxUpdateTime = diff;
81
82 if (diff > _maxUpdateTimeOfCurrentTable)
83 _maxUpdateTimeOfCurrentTable = diff;
84
85 if (++_updateTimeTableIndex >= _updateTimeDataTable.size())
86 {
87 _updateTimeTableIndex = 0;
88 _maxUpdateTimeOfLastTable = _maxUpdateTimeOfCurrentTable;
89 _maxUpdateTimeOfCurrentTable = 0;
90 }
91
92 if (_updateTimeDataTable[_updateTimeDataTable.size() - 1])
93 _averageUpdateTime = _totalUpdateTime / _updateTimeDataTable.size();
94 else if (_updateTimeTableIndex)
95 _averageUpdateTime = _totalUpdateTime / _updateTimeTableIndex;
96}
