393 void UpdateAI(uint32 diff) override
394 {
395 if (!UpdateVictim())
396 return;
397
398 events.Update(diff);
399
400 if (_transportCheckTimer <= diff)
401 {
402 _transportCheckTimer = 1000;
403 Map::PlayerList const& pl = me->GetMap()->GetPlayers();
404 for (Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr)
405 if (Player* p = itr->GetSource())
406 if (p->GetTransport())
407 {
408 EnterEvadeMode(EVADE_REASON_OTHER);
409 return;
410 }
411 }
412 else
413 _transportCheckTimer -= diff;
414
415 if (me->HasUnitState(UNIT_STATE_CASTING))
416 return;
417
418 if (uint32 eventId = events.ExecuteEvent())
419 {
420 switch (eventId)
421 {
422 case EVENT_SUMMON_BLOOD_BEAST:
423 for (uint32 i10 = 0; i10 < 2; ++i10)
424 DoCast(me, SPELL_SUMMON_BLOOD_BEAST + i10);
425 if (Is25ManRaid())
426 for (uint32 i25 = 0; i25 < 3; ++i25)
427 DoCast(me, SPELL_SUMMON_BLOOD_BEAST_25_MAN + i25);
428 Talk(SAY_BLOOD_BEASTS);
429 events.ScheduleEvent(EVENT_SUMMON_BLOOD_BEAST, 40000);
430 if (IsHeroic())
431 events.ScheduleEvent(EVENT_BLOOD_BEAST_SCENT_OF_BLOOD, 10000);
432 break;
433 case EVENT_BLOOD_BEAST_SCENT_OF_BLOOD:
434 Talk(EMOTE_SCENT_OF_BLOOD);
435 summons.DoAction(ACTION_GAIN_SCENT_OF_BLOOD);
436 break;
437 case EVENT_BLOOD_NOVA:
438 {
439 me->CastSpell((Unit*)nullptr, SPELL_BLOOD_NOVA_TRIGGER, false);
440 events.ScheduleEvent(EVENT_BLOOD_NOVA, urand(20000, 25000));

442 }

444 DoCastVictim(SPELL_RUNE_OF_BLOOD);
