184void InstanceScript::UpdateMinionState(Creature* minion, EncounterState state)
185{
186 switch (state)
187 {
188 case NOT_STARTED:
189 if (!minion->IsAlive())
190 minion->Respawn();
191 else if (minion->IsInCombat())
192 minion->AI()->EnterEvadeMode();
193 break;
194 case IN_PROGRESS:
195 if (!minion->IsAlive())
196 minion->Respawn();
197 else
198 {
199 minion->AI()->DoZoneInCombat(nullptr, 100.0f);
200 }
201 break;
202 default:
203 break;
204 }

207void InstanceScript::UpdateDoorState(GameObject* door)
208{
209 DoorInfoMapBounds range = doors.equal_range(door->GetEntry());
210 if (range.first == range.second)
211 return;
212
213 // xinef: doors can be assigned to few bosses, if any of them demands doors closed - they should be closed (added & operator for assigment)
214 bool open = true;
215 for (; range.first != range.second && open; ++range.first)
216 {
217 DoorInfo const& info = range.first->second;
218 switch (info.type)
219 {
220 case DOOR_TYPE_ROOM:
221 open &= (info.bossInfo->state != IN_PROGRESS);
222 break;
223 case DOOR_TYPE_PASSAGE:
224 open &= (info.bossInfo->state == DONE);
225 break;
226 case DOOR_TYPE_SPAWN_HOLE:
227 open &= (info.bossInfo->state == IN_PROGRESS);
228 break;
229 default:
230 break;
231 }
232 }
233
234 door->SetGoState(open ? GO_STATE_ACTIVE : GO_STATE_READY);
235}

271void InstanceScript::AddDoor(GameObject* door, bool add)
272{
273 DoorInfoMapBounds range = doors.equal_range(door->GetEntry());
274 if (range.first == range.second)
275 return;
276
277 for (; range.first != range.second; ++range.first)
278 {
279 DoorInfo const& data = range.first->second;
280
281 if (add)
282 {
283 data.bossInfo->door[data.type].insert(door);
284 }
285 else
286 data.bossInfo->door[data.type].erase(door);
287 }
288
289 if (add)
290 UpdateDoorState(door);
291}

305bool InstanceScript::SetBossState(uint32 id, EncounterState state)
306{
307 if (id < bosses.size())
308 {
309 BossInfo* bossInfo = &bosses[id];
310 if (bossInfo->state == TO_BE_DECIDED) // loading
311 {
312 bossInfo->state = state;
313 return false;
314 }
315 else
316 {
317 if (bossInfo->state == state)
318 return false;
319
320 if (state == DONE)
321 for (MinionSet::iterator i = bossInfo->minion.begin(); i != bossInfo->minion.end(); ++i)
322 if ((*i)->isWorldBoss() && (*i)->IsAlive())
323 return false;
324
325 bossInfo->state = state;
326 SaveToDB();
327 }
328
329 for (uint32 type = 0; type < MAX_DOOR_TYPES; ++type)
330 for (DoorSet::iterator i = bossInfo->door[type].begin(); i != bossInfo->door[type].end(); ++i)
331 UpdateDoorState(*i);
332
333 for (MinionSet::iterator i = bossInfo->minion.begin(); i != bossInfo->minion.end(); ++i)
334 UpdateMinionState(*i, state);
335
336 return true;
337 }
338 return false;
339}

341std::string InstanceScript::LoadBossState(const char* data)
342{
343 if (!data)
344 return nullptr;
345 std::istringstream loadStream(data);
346 uint32 buff;
347 uint32 bossId = 0;
348 for (std::vector<BossInfo>::iterator i = bosses.begin(); i != bosses.end(); ++i, ++bossId)
349 {
350 loadStream >> buff;
351 if (buff < TO_BE_DECIDED)
352 SetBossState(bossId, (EncounterState)buff);
353 }
354 return loadStream.str();
355}
