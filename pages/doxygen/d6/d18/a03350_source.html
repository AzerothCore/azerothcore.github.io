210 void UpdateAI(uint32 diff) override
211 {
212 if (!UpdateVictim())
213 return;
214
215 if (me->GetVictim())
216 {
217 float x, y, z;
218 me->GetVictim()->GetPosition(x, y, z);
219 if (x < 600.0f || x > 770.0f || y < -270.0f || y > -137.0f || z < 514.0f || z > 550.0f)
220 {
221 me->SetHealth(me->GetMaxHealth());
222 EnterEvadeMode(EVADE_REASON_OTHER);
223 if (CreatureGroup* f = me->GetFormation())
224 {
225 const CreatureGroup::CreatureGroupMemberType& m = f->GetMembers();
226 for (CreatureGroup::CreatureGroupMemberType::const_iterator itr = m.begin(); itr != m.end(); ++itr)
227 if (itr->first->IsAlive() && itr->first->IsInCombat() && !itr->first->IsInEvadeMode() && itr->first->IsAIEnabled)
228 itr->first->AI()->EnterEvadeMode();
229 }
230 return;
231 }
232 }
233
234 events.Update(diff);
235
236 if (me->HasUnitState(UNIT_STATE_CASTING))
237 return;
238
239 switch(events.ExecuteEvent())
240 {
241 case 0:
242 break;
243 case EVENT_SPELL_THROW_SARONITE:
244 bCanSayBoulderHit = true;
245 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 140.0f, true))
246 {
247 WorldPacket data;
248 ChatHandler::BuildChatPacket(data, CHAT_MSG_RAID_BOSS_EMOTE, LANG_UNIVERSAL, me, nullptr, EMOTE_THROW_SARONITE);
249 target->ToPlayer()->GetSession()->SendPacket(&data);
250 me->CastSpell(target, SPELL_THROW_SARONITE, false);
251 }
252 events.RepeatEvent(urand(12500, 20000));
253 break;
254 case EVENT_JUMP:
255 me->DisableRotate(true);
256 if (phase == 1)
257 me->GetMotionMaster()->MoveJump(northForgePos.GetPositionX(), northForgePos.GetPositionY(), northForgePos.GetPositionZ(), 25.0f, 15.0f, 0);
258 else if (phase == 2)
259 me->GetMotionMaster()->MoveJump(southForgePos.GetPositionX(), southForgePos.GetPositionY(), southForgePos.GetPositionZ(), 25.0f, 15.0f, 0);
260
261 break;
262 case EVENT_SPELL_CHILLING_WAVE:
263 me->CastSpell(me->GetVictim(), SPELL_CHILLING_WAVE, false);
264 events.RepeatEvent(35000);
265 break;
266 case EVENT_SPELL_DEEP_FREEZE:
267 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true))
268 {
269 Talk(EMOTE_DEEP_FREEZE, target);
270 me->CastSpell(target, SPELL_DEEP_FREEZE, false);
271 }
272 events.RepeatEvent(35000);
273 break;
274 }
275
276 DoMeleeAttackIfReady();
277 }
