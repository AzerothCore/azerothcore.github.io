186 void AddItem(ObjectGuid::LowType itemGuidLow, uint32 item_template)
187 {
188 MailItemInfo mii;
189 mii.item_guid = itemGuidLow;
190 mii.item_template = item_template;
191 items.push_back(mii);
192 }
193
194 bool RemoveItem(ObjectGuid::LowType item_guid)
195 {
196 for (MailItemInfoVec::iterator itr = items.begin(); itr != items.end(); ++itr)
197 {
198 if (itr->item_guid == item_guid)
199 {
200 items.erase(itr);
201 return true;
202 }
203 }
204 return false;
205 }
206
207 [[nodiscard]] bool HasItems() const { return !items.empty(); }
208 [[nodiscard]] bool IsSentByPlayer() const { return messageType == MAIL_NORMAL; }
209 [[nodiscard]] bool IsSentByGM() const { return stationery == MAIL_STATIONERY_GM; }
210 [[nodiscard]] bool IsCODPayment() const { return checked & MAIL_CHECK_MASK_COD_PAYMENT; }
211 [[nodiscard]] bool IsReturnedMail() const { return checked & MAIL_CHECK_MASK_RETURNED; }
212};
