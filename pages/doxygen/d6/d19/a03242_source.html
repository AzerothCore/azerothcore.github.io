117class boss_baltharus_the_warborn : public CreatureScript
118{
119public:
120 boss_baltharus_the_warborn() : CreatureScript("boss_baltharus_the_warborn") { }
121
122 struct boss_baltharus_the_warbornAI : public BossAI
123 {
124 boss_baltharus_the_warbornAI(Creature* creature) : BossAI(creature, DATA_BALTHARUS_THE_WARBORN)
125 {
126 _introDone = false;
127 }
128
129 void Reset() override
130 {
131 BossAI::Reset();
132 }
133
134 void InitializeAI() override
135 {
136 BossAI::InitializeAI();
137 me->CastSpell(me, SPELL_BARRIER_CHANNEL, false);
138 }
139
140 void JustReachedHome() override
141 {
142 BossAI::JustReachedHome();
143 me->CastSpell(me, SPELL_BARRIER_CHANNEL, false);
144 }
145
146 void DoAction(int32 action) override
147 {
148 if (action == ACTION_INTRO_BALTHARUS && !_introDone)
149 {
150 _introDone = true;
151 me->m_Events.AddEvent(new DelayedTalk(me, SAY_BALTHARUS_INTRO), me->m_Events.CalculateTime(6000));
152 }
153 else if (action == ACTION_CLONE)
154 {
155 me->CastSpell(me, SPELL_REPELLING_WAVE, false);
156 me->CastSpell(me, SPELL_CLEAR_DEBUFFS, false);
157 events.ScheduleEvent(EVENT_SUMMON_CLONE, 1000);
158 }
159 }
160
161 void EnterCombat(Unit* who) override
162 {
163 Talk(SAY_AGGRO);
164 BossAI::EnterCombat(who);
165 me->InterruptNonMeleeSpells(false);
166
167 events.ScheduleEvent(EVENT_CLEAVE, 11000);
168 events.ScheduleEvent(EVENT_ENERVATING_BRAND, 13000);
169 events.ScheduleEvent(EVENT_BLADE_TEMPEST, 15000);
170 if (!Is25ManRaid())
171 events.ScheduleEvent(EVENT_CHECK_HEALTH1, 1000);
172 else
173 {
174 events.ScheduleEvent(EVENT_CHECK_HEALTH2, 1000);
175 events.ScheduleEvent(EVENT_CHECK_HEALTH3, 1000);
176 }
177 }
178
179 void JustDied(Unit* killer) override
180 {
181 Talk(SAY_DEATH);
182 BossAI::JustDied(killer);
183
184 if (Creature* xerestrasza = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_XERESTRASZA)))
185 xerestrasza->AI()->DoAction(ACTION_BALTHARUS_DEATH);
186 }
187
188 void KilledUnit(Unit* /*victim*/) override
189 {
190 if (events.GetNextEventTime(EVENT_KILL_TALK) == 0)
191 {
192 Talk(SAY_KILL);
193 events.ScheduleEvent(EVENT_KILL_TALK, 6000);
194 }
195 }
196
197 void JustSummoned(Creature* summon) override
198 {
199 summons.Summon(summon);
200 summon->SetHealth(me->GetHealth());
201 summon->CastSpell(summon, SPELL_SPAWN_EFFECT, true);
202 summon->SetReactState(REACT_PASSIVE);
203 summon->m_Events.AddEvent(new RestoreFight(summon), summon->m_Events.CalculateTime(2000));
204 }
205
206 void UpdateAI(uint32 diff) override
207 {
208 if (!UpdateVictim())
209 return;
210
211 events.Update(diff);
212 if (me->HasUnitState(UNIT_STATE_CASTING))
213 return;
214
215 switch (events.ExecuteEvent())
216 {
217 case EVENT_CLEAVE:
218 me->CastSpell(me->GetVictim(), SPELL_CLEAVE, false);
219 events.ScheduleEvent(EVENT_CLEAVE, 24000);
220 break;
221 case EVENT_BLADE_TEMPEST:
222 me->CastSpell(me, SPELL_BLADE_TEMPEST, false);
223 events.ScheduleEvent(EVENT_BLADE_TEMPEST, 24000);
224 break;
225 case EVENT_ENERVATING_BRAND:
226 for (uint8 i = 0; i < RAID_MODE<uint8>(2, 4, 2, 4); i++)
227 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 45.0f, true, -SPELL_ENERVATING_BRAND))
228 me->CastSpell(target, SPELL_ENERVATING_BRAND, true);
229 events.ScheduleEvent(EVENT_ENERVATING_BRAND, 26000);
230 break;
231 case EVENT_CHECK_HEALTH1:
232 if (me->HealthBelowPct(50))
233 {
234 DoAction(ACTION_CLONE);
235 break;
236 }
237 events.ScheduleEvent(EVENT_CHECK_HEALTH1, 1000);
238 break;
239 case EVENT_CHECK_HEALTH2:
240 if (me->HealthBelowPct(66))
241 {
242 DoAction(ACTION_CLONE);
243 break;
244 }
245 events.ScheduleEvent(EVENT_CHECK_HEALTH2, 1000);
246 break;
247 case EVENT_CHECK_HEALTH3:
248 if (me->HealthBelowPct(33))
249 {
250 DoAction(ACTION_CLONE);
251 break;
252 }
253 events.ScheduleEvent(EVENT_CHECK_HEALTH3, 1000);
254 break;
255 case EVENT_SUMMON_CLONE:
256 me->CastSpell(me, SPELL_CLONE, false);
257 Talk(SAY_CLONE);
258 break;
259 }
260
261 DoMeleeAttackIfReady();
262 }
263
264 private:
265 bool _introDone;
266 };
267
268 CreatureAI* GetAI(Creature* creature) const override
269 {
270 return GetRubySanctumAI<boss_baltharus_the_warbornAI>(creature);
271 }
