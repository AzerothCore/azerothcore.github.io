28ChatCommandResult Acore::ChatCommands::QuotedString::TryConsume(ChatHandler const* handler, std::string_view args)
29{
30 if (args.empty())
31 return std::nullopt;
32
33 if ((args[0] != '"') && (args[0] != '\''))
34 return ArgInfo<std::string>::TryConsume(*this, handler, args);
35
36 char const QUOTE = args[0];
37 for (size_t i = 1; i < args.length(); ++i)
38 {
39 if (args[i] == QUOTE)
40 {
41 auto [remainingToken, tail] = tokenize(args.substr(i + 1));
42 if (remainingToken.empty()) // if this is not empty, then we did not consume the full token
43 return tail;
44 else
45 return std::nullopt;
46 }
47
48 if (args[i] == '\\')
49 {
50 ++i;
51 if (!(i < args.length()))
52 break;
53 }
54 std::string::push_back(args[i]);
55 }
56
57 // if we reach this, we did not find a closing quote
58 return std::nullopt;
59}

61ChatCommandResult Acore::ChatCommands::AccountIdentifier::TryConsume(ChatHandler const* handler, std::string_view args)
62{
63 std::string_view text;
64 ChatCommandResult next = ArgInfo<std::string_view>::TryConsume(text, handler, args);
65 if (!next)
66 return next;
67
68 // first try parsing as account name
69 _name.assign(text);
70 if (!Utf8ToUpperOnlyLatin(_name))
71 return GetAcoreString(handler, LANG_CMDPARSER_INVALID_UTF8);
72
73 _id = AccountMgr::GetId(_name);
74 if (_id) // account with name exists, we are done
75 return next;
76
77 // try parsing as account id instead
78 Optional<uint32> id = Acore::StringTo<uint32>(text, 10);
79 if (!id)
80 return FormatAcoreString(handler, LANG_CMDPARSER_ACCOUNT_NAME_NO_EXIST, STRING_VIEW_FMT_ARG(_name));
81
82 _id = *id;
83
84 if (AccountMgr::GetName(_id, _name))
85 return next;
86 else
87 return FormatAcoreString(handler, LANG_CMDPARSER_ACCOUNT_ID_NO_EXIST, _id);
88}

90ChatCommandResult Acore::ChatCommands::PlayerIdentifier::TryConsume(ChatHandler const* handler, std::string_view args)
91{
92 Variant<Hyperlink<player>, ObjectGuid::LowType, std::string_view> val;
93 ChatCommandResult next = ArgInfo<decltype(val)>::TryConsume(val, handler, args);
94 if (!next)
95 return next;
96
97 if (val.holds_alternative<ObjectGuid::LowType>())
98 {
99 _guid = ObjectGuid::Create<HighGuid::Player>(val.get<ObjectGuid::LowType>());
100
101 if ((_player = ObjectAccessor::FindPlayerByLowGUID(_guid.GetCounter())))
102 {
103 _name = _player->GetName();
104 }
105 else if (!sCharacterCache->GetCharacterNameByGuid(_guid, _name))
106 {
107 return FormatAcoreString(handler, LANG_CMDPARSER_CHAR_GUID_NO_EXIST, _guid.ToString().c_str());
108 }
109
110 return next;
111 }
112 else
113 {
114 if (val.holds_alternative<Hyperlink<player>>())
115 _name.assign(static_cast<std::string_view>(val.get<Hyperlink<player>>()));
116 else
117 _name.assign(val.get<std::string_view>());
118
119 if (!normalizePlayerName(_name))
120 return FormatAcoreString(handler, LANG_CMDPARSER_CHAR_NAME_INVALID, STRING_VIEW_FMT_ARG(_name));
121
122 if ((_player = ObjectAccessor::FindPlayerByName(_name)))
123 {
124 _guid = _player->GetGUID();
125 }
126 else if (!(_guid = sCharacterCache->GetCharacterGuidByName(_name)))
127 {
128 return FormatAcoreString(handler, LANG_CMDPARSER_CHAR_NAME_NO_EXIST, STRING_VIEW_FMT_ARG(_name));
129 }
130
131 return next;
132 }
133}
