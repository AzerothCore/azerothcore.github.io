64 {

67 me->RestoreFaction();

87 void EnterCombat(Unit* /*who*/) override
88 {
89 DoCastSelf(SPELL_MARK_OF_FROST_AURA);

92 _scheduler

95 DoCastVictim(SPELL_CLEAVE);

113 .Schedule(30s, [this](TaskContext context)
114 {
115 Talk(SAY_TELEPORT);
116 DoCastAOE(SPELL_ARCANE_VACUUM);
117 context.Repeat(30s);
118 })
119 .Schedule(15s, 30s, [this](TaskContext context)
120 {
121 DoCastSelf(SPELL_REFLECT);
122 context.Repeat(20s, 35s);
123 });
124 }
125
126 void JustDied(Unit* /*killer*/) override
127 {
128 me->RemoveAurasDueToSpell(SPELL_MARK_OF_FROST);
129 me->GetMap()->DoForAllPlayers([&](Player* p)
130 {
131 if (p->GetZoneId() == me->GetZoneId())
132 {
133
134 p->RemoveAurasDueToSpell(SPELL_MARK_OF_FROST);
135 p->RemoveAurasDueToSpell(SPELL_AURA_OF_FROST);
136 p->RemoveAurasDueToSpell(SPELL_CHILL);
137 p->RemoveAurasDueToSpell(SPELL_FROST_BREATH);

140
141 me->SetRespawnTime(urand(2 * DAY, 3 * DAY));
142 me->SaveRespawnTime();
