32template<>
33AC_COMMON_API uint32 Acore::Time::TimeStringTo<Seconds>(std::string_view timestring)
34{
35 uint32 secs = 0;
36 uint32 buffer = 0;
37 uint32 multiplier = 0;
38
39 for (char itr : timestring)
40 {
41 if (isdigit(itr))
42 {
43 buffer *= 10;
44 buffer += itr - '0';
45 }
46 else
47 {
48 switch (itr)
49 {
50 case 'd':
51 multiplier = DAY;
52 break;
53 case 'h':
54 multiplier = HOUR;
55 break;
56 case 'm':
57 multiplier = MINUTE;
58 break;
59 case 's':
60 multiplier = 1;
61 break;
62 default:
63 return 0; // bad format
64 }
65
66 buffer *= multiplier;
67 secs += buffer;
68 buffer = 0;
69 }
70 }
71
72 return secs;
73}

75template<>
76AC_COMMON_API std::string Acore::Time::ToTimeString<Microseconds>(uint64 durationTime, TimeOutput timeOutput /*= TimeOutput::Seconds*/, TimeFormat timeFormat /*= TimeFormat::ShortText*/)
77{
78 uint64 microsecs = durationTime % 1000;
79 uint64 millisecs = (durationTime / TimeDiff::MILLISECONDS) % 1000;
80 uint64 secs = (durationTime / TimeDiff::SECONDS) % 60;
81 uint64 minutes = (durationTime / TimeDiff::MINUTES) % 60;
82 uint64 hours = (durationTime / TimeDiff::HOURS) % 24;
83 uint64 days = durationTime / TimeDiff::DAYS;
84
85 if (timeFormat == TimeFormat::Numeric)
86 {
87 if (days)
88 {
89 return Acore::StringFormatFmt("{}:{:02}:{:02}:{:02}:{:02}:{:02}", days, hours, minutes, secs, millisecs);
90 }
91 else if (hours)
92 {
93 return Acore::StringFormatFmt("{}:{:02}:{:02}:{:02}:{:02}", hours, minutes, secs, millisecs);
94 }
95 else if (minutes)
96 {
97 return Acore::StringFormatFmt("{}:{:02}:{:02}:{:02}", minutes, secs, millisecs);
98 }
99 else if (secs)
100 {
101 return Acore::StringFormatFmt("{}:{:02}:{:02}", secs, millisecs);
102 }
103 else if (millisecs)
104 {
105 return Acore::StringFormatFmt("{}:{:02}", millisecs);
106 }
107 else // microsecs
108 {
109 return Acore::StringFormatFmt("{}", microsecs);
110 }
111 }

156 }
157

159 {
160 stringTime = ss.str();
161 }
162

164 {
165 GetStringFormat(secs, "s ", " Second ", " Seconds ");
166 }
167

169 {
170 stringTime = ss.str();
171 }
172

174 {
175 GetStringFormat(millisecs, "ms ", " Millisecond ", " Milliseconds ");
176 }
177

179 {
180 stringTime = ss.str();
181 }
182

185 GetStringFormat(microsecs, "us ", " Microsecond ", " Microseconds ");
186 }
187
188 if (timeOutput == TimeOutput::Microseconds)
189 {
190 stringTime = ss.str();
191 }
192
193 return Acore::String::TrimRightInPlace(stringTime);

254time_t Acore::Time::GetLocalHourTimestamp(time_t time, uint8 hour, bool onlyAfterTime)
255{
256 tm timeLocal = TimeBreakdown(time);
257 timeLocal.tm_hour = 0;
258 timeLocal.tm_min = 0;
259 timeLocal.tm_sec = 0;
260
261 time_t midnightLocal = mktime(&timeLocal);
262 time_t hourLocal = midnightLocal + hour * HOUR;
263
264 if (onlyAfterTime && hourLocal <= time)
265 {
266 hourLocal += DAY;
267 }
268
269 return hourLocal;
270}

302time_t Acore::Time::GetNextTimeWithDayAndHour(int8 dayOfWeek, int8 hour)
303{
304 if (hour < 0 || hour > 23)
305 {
306 hour = 0;
307 }
308
309 tm localTm = TimeBreakdown();
310 localTm.tm_hour = hour;
311 localTm.tm_min = 0;
312 localTm.tm_sec = 0;
313
314 if (dayOfWeek < 0 || dayOfWeek > 6)
315 {
316 dayOfWeek = (localTm.tm_wday + 1) % 7;
317 }
318
319 uint32 add;
320
321 if (localTm.tm_wday >= dayOfWeek)
322 {
323 add = (7 - (localTm.tm_wday - dayOfWeek)) * DAY;
324 }
325 else
326 {
327 add = (dayOfWeek - localTm.tm_wday) * DAY;
328 }
329
330 return mktime(&localTm) + add;
331}

333time_t Acore::Time::GetNextTimeWithMonthAndHour(int8 month, int8 hour)
334{
335 if (hour < 0 || hour > 23)
336 {
337 hour = 0;
338 }
339
340 tm localTm = TimeBreakdown();
341 localTm.tm_mday = 1;
342 localTm.tm_hour = hour;
343 localTm.tm_min = 0;
344 localTm.tm_sec = 0;
345
346 if (month < 0 || month > 11)
347 {
348 month = (localTm.tm_mon + 1) % 12;
349
350 if (!month)
351 {
352 localTm.tm_year += 1;
353 }
354 }
355 else if (localTm.tm_mon >= month)
356 {
357 localTm.tm_year += 1;
358 }
359
360 localTm.tm_mon = month;
361 return mktime(&localTm);
362}
