30struct GroupQueueInfo // stores information about the group in queue (also used when joined as solo!)
31{
32 GuidSet Players; // player guid set
33 TeamId teamId; // Player team (TEAM_ALLIANCE/TEAM_HORDE)
34 TeamId RealTeamID; // Realm player team (TEAM_ALLIANCE/TEAM_HORDE)
35 BattlegroundTypeId BgTypeId; // battleground type id
36 bool IsRated; // rated
37 uint8 ArenaType; // 2v2, 3v3, 5v5 or 0 when BG
38 uint32 ArenaTeamId; // team id if rated match
39 uint32 JoinTime; // time when group was added
40 uint32 RemoveInviteTime; // time when we will remove invite for players in group
41 uint32 IsInvitedToBGInstanceGUID; // was invited to certain BG
42 uint32 ArenaTeamRating; // if rated match, inited to the rating of the team
43 uint32 ArenaMatchmakerRating; // if rated match, inited to the rating of the team
44 uint32 OpponentsTeamRating; // for rated arena matches
45 uint32 OpponentsMatchmakerRating; // for rated arena matches
46 uint32 PreviousOpponentsTeamId; // excluded from the current queue until the timer is met
47 uint8 BracketId; // BattlegroundBracketId
48 uint8 GroupType; // BattlegroundQueueGroupTypes
49};
