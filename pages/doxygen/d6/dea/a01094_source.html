36 void LfgGroupData::SetState(LfgState state)
37 {
38 switch (state)
39 {
40 case LFG_STATE_DUNGEON:
41 _isLFGGroup = true;
42 break;
43 case LFG_STATE_FINISHED_DUNGEON:
44 _isLFGGroup = false;
45 break;
46 default:
47 break;
48 }
49
50 switch (state)
51 {
52 case LFG_STATE_NONE:
53 m_Dungeon = 0;
54 m_KicksLeft = sWorld->getIntConfig(CONFIG_LFG_MAX_KICK_COUNT);
55 [[fallthrough]];
56 case LFG_STATE_FINISHED_DUNGEON:
57 case LFG_STATE_DUNGEON:
58 m_OldState = state;
59 [[fallthrough]];
60 default:
61 m_State = state;
62 }
63 }
