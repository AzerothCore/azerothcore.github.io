88 void EnterCombat(Unit* /*who*/) override
89 {

92 events.Reset();
93 events.ScheduleEvent(EVENT_SHADOW_BOLT_VOLLEY, 2000);
94 events.ScheduleEvent(EVENT_CURSE_OF_TONGUES, 6000);

103 void EnterEvadeMode(EvadeReason /*why*/) override

108
109 void IsSummonedBy(Unit* /*summoner*/) override
110 {
111 events2.Reset();
112 events2.ScheduleEvent(INTRO_1, 1000);
113 me->SetDisableGravity(true);
114 me->SetReactState(REACT_PASSIVE);
115 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
116 me->SetImmuneToAll(true); // for some reason he aggroes if we don't have this.
117 }

119 void MovementInform(uint32 type, uint32 id) override
120 {
121 if (type == WAYPOINT_MOTION_TYPE && id == POINT_KIRTONOS_LAND)
122 {
123 events2.ScheduleEvent(INTRO_2, 1500);
124 events2.ScheduleEvent(INTRO_3, 2500);
125 events2.ScheduleEvent(INTRO_4, 5500);
126 events2.ScheduleEvent(INTRO_5, 6500);
127 events2.ScheduleEvent(INTRO_6, 11500);

131 void UpdateAI(uint32 diff) override
132 {
133 events2.Update(diff);
134 switch (events2.ExecuteEvent())
135 {
136 case INTRO_1:
137 me->GetMotionMaster()->MovePath(KIRTONOS_PATH, false);
138 Talk(EMOTE_SUMMONED);
139 break;
140 case INTRO_2:
141 me->GetMotionMaster()->MovePoint(0, PosMove[0]);
142 break;
143 case INTRO_3:
144 me->SetFacingTo(0.01745329f);
145 break;
146 case INTRO_4:
147 me->SetWalk(true);
148 me->SetCanFly(false);
149 me->SetDisableGravity(false);
150 me->CastSpell(me, SPELL_KIRTONOS_TRANSFORM, true);
151 break;
152 case INTRO_5:
153 me->HandleEmoteCommand(EMOTE_ONESHOT_ROAR);
154 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_KIRTONOS_STAFF));
155 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
156 me->SetImmuneToAll(false);
157 me->SetReactState(REACT_AGGRESSIVE);
158 break;
159 case INTRO_6:
160 if (!me->IsInCombat())
161 {
162 me->GetMotionMaster()->MovePoint(0, PosMove[1]);
163 }
164 break;
165 }
166
167 if (!UpdateVictim())
168 {
169 return;
170 }
171
172 events.Update(diff);
173 if (me->HasUnitState(UNIT_STATE_CASTING))
174 {
175 return;
176 }
177
178 switch (events.ExecuteEvent())
179 {
180 case EVENT_SWOOP:
181 me->CastSpell(me->GetVictim(), SPELL_SWOOP, false);
182 events.ScheduleEvent(EVENT_SWOOP, 15000);
183 break;
184 case EVENT_WING_FLAP:
185 me->CastSpell(me, SPELL_WING_FLAP, false);
186 events.ScheduleEvent(EVENT_WING_FLAP, 13000);
187 break;
188 case EVENT_PIERCE_ARMOR:
189 me->CastSpell(me->GetVictim(), SPELL_PIERCE_ARMOR, false);
190 events.ScheduleEvent(EVENT_PIERCE_ARMOR, 12000);
191 break;
192 case EVENT_DISARM:
193 me->CastSpell(me->GetVictim(), SPELL_DISARM, false);
194 events.ScheduleEvent(EVENT_DISARM, 11000);
195 break;
196 case EVENT_SHADOW_BOLT_VOLLEY:
197 me->CastSpell(me, SPELL_SHADOW_BOLT_VOLLEY, false);
198 events.ScheduleEvent(EVENT_SHADOW_BOLT_VOLLEY, 10000);
199 break;
200 case EVENT_CURSE_OF_TONGUES:
201 me->CastSpell(me, SPELL_CURSE_OF_TONGUES, false);
202 events.ScheduleEvent(EVENT_CURSE_OF_TONGUES, 20000);
203 break;
204 case EVENT_DOMINATE_MIND:
205 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 20.0f, true))
206 {
207 me->CastSpell(target, SPELL_DOMINATE_MIND, false);
208 }
209 break;
210 case EVENT_KIRTONOS_TRANSFORM:
211 TransformsCount++;
212
213 if (me->HasAura(SPELL_KIRTONOS_TRANSFORM))
214 {
215 events.Reset();
216 events.ScheduleEvent(EVENT_SWOOP, 4000);
217 events.ScheduleEvent(EVENT_WING_FLAP, 7000);
218 events.ScheduleEvent(EVENT_PIERCE_ARMOR, 11000);
219 events.ScheduleEvent(EVENT_DISARM, 15000);
220 // show shape-shift animation before aura removal
221 me->CastSpell(me, SPELL_TRANSFORM_VISUAL, true);
222 me->RemoveAura(SPELL_KIRTONOS_TRANSFORM);
223 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(0));
224 }
225 else
226 {
227 events.Reset();
228 events.ScheduleEvent(EVENT_SHADOW_BOLT_VOLLEY, 2000);
229 events.ScheduleEvent(EVENT_CURSE_OF_TONGUES, 6000);
230 events.ScheduleEvent(EVENT_WING_FLAP, 13000);
231 me->CastSpell(me, SPELL_KIRTONOS_TRANSFORM, true);
232 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_KIRTONOS_STAFF));
233 // Schedule Dominate Mind on every 2nd caster transform
234 if ((TransformsCount - 2) % 4 == 0)
235 {
236 events.ScheduleEvent(EVENT_DOMINATE_MIND, urand(4000, 8000));
237 }
238 }
239
240 events.ScheduleEvent(EVENT_KIRTONOS_TRANSFORM, 20000);
241 break;
242 }
243
244 DoZoneInCombat();
245 DoMeleeAttackIfReady();
246 }
247 };
