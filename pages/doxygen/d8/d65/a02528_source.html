86 void Initialize()
87 {
88 PlayerGUID.Clear();
89 HasYelled = false;
90 _introDone = false;
91 _burningAdrenalineCast = 0;
92 me->SetNpcFlag(UNIT_NPC_FLAG_GOSSIP);
93 me->SetNpcFlag(UNIT_NPC_FLAG_QUESTGIVER);
94 me->SetFaction(FACTION_FRIENDLY);
95 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);

98 void Reset() override
99 {
100 _Reset();
101 me->SetHealth(me->CountPctFromMaxHealth(30));
102
103 if (!_introDone)
104 {
105 me->SetStandState(UNIT_STAND_STATE_DEAD);
106 me->SetReactState(REACT_PASSIVE);
107 Initialize();
108 _eventsIntro.Reset();
109 }
110 else
111 {
112 HasYelled = false;
113 _burningAdrenalineCast = 0;
114 }
115 }

117 void EnterCombat(Unit* victim) override
118 {
119 BossAI::EnterCombat(victim);
120
121 DoCastAOE(SPELL_ESSENCE_OF_THE_RED);
122 // now drop damage requirement to be able to take loot
123 me->ResetPlayerDamageReq();
124
125 events.ScheduleEvent(EVENT_CLEAVE, 10000);
126 events.ScheduleEvent(EVENT_FLAME_BREATH, 15000);
127 events.ScheduleEvent(EVENT_FIRE_NOVA, 5000);
128 events.ScheduleEvent(EVENT_TAIL_SWEEP, 11000);
129 events.ScheduleEvent(EVENT_BURNING_ADRENALINE, 15000);
130 }

132 void BeginSpeech(Unit* target)

136 _eventsIntro.ScheduleEvent(EVENT_SPEECH_1, 1000);

139 void KilledUnit(Unit* victim) override
140 {
141 if (rand32() % 5)
142 return;
143
144 Talk(SAY_KILLTARGET, victim);

146
147 void UpdateAI(uint32 diff) override
148 {
149 events.Update(diff);
150 _eventsIntro.Update(diff);
151
152 // Speech
153 if (!_introDone)
154 {
155 while (uint32 eventId = _eventsIntro.ExecuteEvent())
156 {
157 switch (eventId)
158 {
159 case EVENT_SPEECH_1:
160 me->SetStandState(UNIT_STAND_STATE_STAND);
161 me->SummonCreature(NPC_VICTOR_NEFARIUS, aNefariusSpawnLoc[0], aNefariusSpawnLoc[1], aNefariusSpawnLoc[2], aNefariusSpawnLoc[3], TEMPSUMMON_TIMED_DESPAWN, 26000);
162 _eventsIntro.ScheduleEvent(EVENT_SPEECH_2, 1000);
163 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
164 break;
165 case EVENT_SPEECH_2:
166 if (Creature* nefarius = me->GetMap()->GetCreature(m_nefariusGuid))
167 {
168 nefarius->CastSpell(me, SPELL_NEFARIUS_CORRUPTION, TRIGGERED_CAST_DIRECTLY);
169 nefarius->Yell(SAY_NEFARIAN_VAEL_INTRO);
170 nefarius->SetStandState(UNIT_STAND_STATE_STAND);
171 }
172 _eventsIntro.ScheduleEvent(EVENT_SPEECH_3, 18000);
173 break;
174 case EVENT_SPEECH_3:
175 if (Creature* nefarius = me->GetMap()->GetCreature(m_nefariusGuid))
176 nefarius->CastSpell(me, SPELL_RED_LIGHTNING, TRIGGERED_NONE);
177 _eventsIntro.ScheduleEvent(EVENT_SPEECH_4, 2000);
178 break;
179 case EVENT_SPEECH_4:
180 Talk(SAY_LINE1);
181 me->HandleEmoteCommand(EMOTE_ONESHOT_TALK);
182 _eventsIntro.ScheduleEvent(EVENT_SPEECH_5, 12000);
183 break;
184 case EVENT_SPEECH_5:
185 Talk(SAY_LINE2);
186 me->HandleEmoteCommand(EMOTE_ONESHOT_TALK);
187 _eventsIntro.ScheduleEvent(EVENT_SPEECH_6, 12000);
188 break;
189 case EVENT_SPEECH_6:
190 Talk(SAY_LINE3);
191 me->HandleEmoteCommand(EMOTE_ONESHOT_TALK);
192 _eventsIntro.ScheduleEvent(EVENT_SPEECH_7, 17000);
193 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
194 break;
195 case EVENT_SPEECH_7:
196 me->SetFaction(FACTION_DRAGONFLIGHT_BLACK);
197 if (PlayerGUID && ObjectAccessor::GetUnit(*me, PlayerGUID))
198 AttackStart(ObjectAccessor::GetUnit(*me, PlayerGUID));
199 me->SetReactState(REACT_AGGRESSIVE);
200 _introDone = true;
201 break;
202 }
203 }
204 }
205
206 if (!UpdateVictim() || me->HasUnitState(UNIT_STATE_CASTING))
207 return;
208
209 while (uint32 eventId = events.ExecuteEvent())
210 {
211 switch (eventId)
212 {
213 case EVENT_CLEAVE:
214 events.ScheduleEvent(EVENT_CLEAVE, 15000);
215 DoCastVictim(SPELL_CLEAVE);
216 break;
217 case EVENT_FLAME_BREATH:
218 DoCastVictim(SPELL_FLAME_BREATH);
219 events.ScheduleEvent(EVENT_FLAME_BREATH, 8000, 14000);
220 break;
221 case EVENT_FIRE_NOVA:
222 DoCastVictim(SPELL_FIRE_NOVA);
223 events.ScheduleEvent(EVENT_FIRE_NOVA, urand(3000, 5000));
224 break;
225 case EVENT_TAIL_SWEEP:
226 DoCastAOE(SPELL_TAIL_SWEEP);
227 events.ScheduleEvent(EVENT_TAIL_SWEEP, 15000);
228 break;
229 case EVENT_BURNING_ADRENALINE:
230 {
231 if (_burningAdrenalineCast < 2) // It's better to use TaskScheduler for this, but zzz
232 {
233 //selects a random target that isn't the current victim and is a mana user (selects mana users) but not pets
234 //it also ignores targets who have the aura. We don't want to place the debuff on the same target twice.
235 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, [&](Unit* u) { return u && !u->IsPet() && u->getPowerType() == POWER_MANA && !u->HasAura(SPELL_BURNING_ADRENALINE) && u != me->GetVictim(); }))
236 {
237 me->CastSpell(target, SPELL_BURNING_ADRENALINE, true);
238 }
239
240 _burningAdrenalineCast++;
241 }
242 else
243 {
244 me->CastSpell(me->GetVictim(), SPELL_BURNING_ADRENALINE, true);
245 _burningAdrenalineCast = 0;
246 }
247 events.ScheduleEvent(EVENT_BURNING_ADRENALINE, 15000);
248 break;
249 }
250 }
251 }
252
253 // Yell if hp lower than 15%
254 if (HealthBelowPct(15) && !HasYelled)
255 {
256 Talk(SAY_HALFLIFE);
257 HasYelled = true;
258 }
259
260 DoMeleeAttackIfReady();
261 }
