96template <class T>
97uint32 DatabaseWorkerPool<T>::Open()
98{
99 WPFatal(_connectionInfo.get(), "Connection info was not set!");
100
101 LOG_INFO("sql.driver", "Opening DatabasePool '{}'. Asynchronous connections: {}, synchronous connections: {}.",
102 GetDatabaseName(), _async_threads, _synch_threads);
103
104 uint32 error = OpenConnections(IDX_ASYNC, _async_threads);
105
106 if (error)
107 return error;
108
109 error = OpenConnections(IDX_SYNCH, _synch_threads);
110
111 if (!error)
112 {
113 LOG_INFO("sql.driver", "DatabasePool '{}' opened successfully. {} total connections running.",
114 GetDatabaseName(), (_connections[IDX_SYNCH].size() + _connections[IDX_ASYNC].size()));
115 }
116
117 LOG_INFO("sql.driver", " ");
118
119 return error;
120}

142template <class T>
143bool DatabaseWorkerPool<T>::PrepareStatements()
144{
145 for (auto const& connections : _connections)
146 {
147 for (auto const& connection : connections)
148 {
149 connection->LockIfReady();
150 if (!connection->PrepareStatements())
151 {
152 connection->Unlock();
153 Close();
154 return false;
155 }
156 else
157 connection->Unlock();
158
159 size_t const preparedSize = connection->m_stmts.size();
160 if (_preparedStatementSize.size() < preparedSize)
161 _preparedStatementSize.resize(preparedSize);
162
163 for (size_t i = 0; i < preparedSize; ++i)
164 {
165 // already set by another connection
166 // (each connection only has prepared statements of it's own type sync/async)
167 if (_preparedStatementSize[i] > 0)
168 continue;
169
170 if (MySQLPreparedStatement* stmt = connection->m_stmts[i].get())
171 {
172 uint32 const paramCount = stmt->GetParameterCount();
173
174 // WH only supports uint8 indices.
175 ASSERT(paramCount < std::numeric_limits<uint8>::max());
176
177 _preparedStatementSize[i] = static_cast<uint8>(paramCount);
178 }
179 }
180 }
181 }
182
183 return true;
184}

307template <class T>
308void DatabaseWorkerPool<T>::DirectCommitTransaction(SQLTransaction<T>& transaction)
309{
310 T* connection = GetFreeConnection();
311 int errorCode = connection->ExecuteTransaction(transaction);
312
313 if (!errorCode)
314 {
315 connection->Unlock(); // OK, operation succesful
316 return;
317 }
318
321 if (errorCode == ER_LOCK_DEADLOCK)
322 {
323 //todo: handle multiple sync threads deadlocking in a similar way as async threads
324 uint8 loopBreaker = 5;
325
326 for (uint8 i = 0; i < loopBreaker; ++i)
327 {
328 if (!connection->ExecuteTransaction(transaction))
329 break;
330 }
331 }
332
334 transaction->Cleanup();
335
336 connection->Unlock();
337}

379template <class T>
380uint32 DatabaseWorkerPool<T>::OpenConnections(InternalIndex type, uint8 numConnections)
381{
382 for (uint8 i = 0; i < numConnections; ++i)
383 {
384 // Create the connection
385 auto connection = [&]
386 {
387 switch (type)
388 {
389 case IDX_ASYNC:
390 return std::make_unique<T>(_queue.get(), *_connectionInfo);
391 case IDX_SYNCH:
392 return std::make_unique<T>(*_connectionInfo);
393 default:
394 ABORT();
395 }
396 }();
397
398 if (uint32 error = connection->Open())
399 {
400 // Failed to open a connection or invalid version, abort and cleanup
401 _connections[type].clear();
402 return error;
403 }
404 else if (connection->GetServerVersion() < MIN_MYSQL_SERVER_VERSION)
405 {
406 LOG_ERROR("sql.driver", "AzerothCore does not support MySQL versions below 5.7 or MariaDB versions below 10.5");
407 return 1;
408 }
409 else
410 {
411 _connections[type].push_back(std::move(connection));
412 }
413 }
414
415 // Everything is fine
416 return 0;
417}

440template <class T>
441T* DatabaseWorkerPool<T>::GetFreeConnection()
442{
443#ifdef ACORE_DEBUG
444 if (_warnSyncQueries)
445 {
446 std::ostringstream ss;
447 ss << boost::stacktrace::stacktrace();
448 LOG_WARN("sql.performances", "Sync query at:\n{}", ss.str());
449 }
450#endif
451
452 uint8 i = 0;
453 auto const num_cons = _connections[IDX_SYNCH].size();
454 T* connection = nullptr;
455
457 for (;;)
458 {
459 connection = _connections[IDX_SYNCH][++i % num_cons].get();
461 if (connection->LockIfReady())
462 break;
463 }
464
465 return connection;
466}
