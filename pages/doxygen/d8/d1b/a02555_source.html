130 void Reset() override
131 {
132 _Reset();
133
134 // Never reset intro events!
135 if (_isIntroDone && !(extraEvents.GetPhaseMask() & (1 << (PHASE_INTRO - 1))))
136 {
137 extraEvents.Reset();
138 extraEvents.SetPhase(PHASE_EMERGED);
139 me->SetReactState(REACT_AGGRESSIVE);
140 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
141 me->SetImmuneToAll(false);
142 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, 0);
143 me->HandleEmoteCommand(EMOTE_ONESHOT_EMERGE);
144 }
145
146 _hasYelledMagmaBurst = false;
147 _processingMagmaBurst = false;
148 _hasSubmergedOnce = false;
149 _isKnockbackEmoteAllowed = true;
150 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, 0);
151 me->SetControlled(true, UNIT_STATE_ROOT);
152 _lavaBurstGUIDS.clear();

259 void UpdateAI(uint32 diff) override
260 {
261 if (!extraEvents.Empty())
262 {
263 extraEvents.Update(diff);
264
265 while (uint32 const eventId = extraEvents.ExecuteEvent())
266 {
267 switch (eventId)
268 {
269 // Intro events
270 case EVENT_INTRO_SAY:
271 {
272 Talk(SAY_ARRIVAL5_RAG);
273 extraEvents.ScheduleEvent(EVENT_INTRO_MAKE_ATTACKABLE, 2500, 0, PHASE_INTRO);
274 break;
275 }
276 case EVENT_INTRO_MAKE_ATTACKABLE:
277 {
278 _isIntroDone = true;
279 extraEvents.SetPhase(PHASE_EMERGED);
280 me->RemoveAurasDueToSpell(SPELL_RAGNAROS_SUBMERGE_EFFECT);
281 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
282 me->SetImmuneToAll(false);
283 me->SetReactState(REACT_AGGRESSIVE);
284 DoZoneInCombat();
285 break;
286 }
287 // Submerge events
288 case EVENT_EMERGE:
289 {
290 HandleEmerge();
291 break;
292 }
293 case EVENT_RESET_KNOCKBACK_EMOTE:
294 {
295 _isKnockbackEmoteAllowed = true;
296 break;
297 }
298 case EVENT_LAVA_BURST_TRIGGER:
299 {
300 if (!_lavaBurstGUIDS.empty())
301 {
302 ObjectGuid lavaBurstGUID = Acore::Containers::SelectRandomContainerElement(_lavaBurstGUIDS);
303
304 if (GameObject* go = ObjectAccessor::GetGameObject(*me, lavaBurstGUID))
305 {
306 go->CastSpell(nullptr, SPELL_LAVA_BURST_TRAP);
307 go->SendCustomAnim(0);
308 }
309
310 _lavaBurstGUIDS.erase(lavaBurstGUID);
311 extraEvents.RepeatEvent(1000);
312 }
313 else
314 {
315 events.RescheduleEvent(EVENT_LAVA_BURST, 10000, PHASE_EMERGED, PHASE_EMERGED);
316 }
317
318 break;
319 }
320 }
321 }
322 }
323
324 if (!UpdateVictim())
325 {
326 if (!_processingMagmaBurst)
327 {
328 return;
329 }
330 }
331
332 events.Update(diff);
333
334 if (me->HasUnitState(UNIT_STATE_CASTING))
335 {
336 return;
337 }
338
339 // Base combat events - (mainly emerge phase)
340 while (uint32 const eventId = events.ExecuteEvent())
341 {
342 switch (eventId)
343 {
344 case EVENT_WRATH_OF_RAGNAROS:
345 {
346 DoCastVictim(SPELL_WRATH_OF_RAGNAROS);
347
348 if (urand(0, 1))
349 {
350 Talk(SAY_WRATH);
351 }
352 events.RepeatEvent(25000);
353 break;
354 }
355 case EVENT_HAND_OF_RAGNAROS:
356 {
357 DoCastSelf(SPELL_HAND_OF_RAGNAROS);
358 if (_isKnockbackEmoteAllowed)
359 {
360 Talk(SAY_KNOCKBACK);
361 _isKnockbackEmoteAllowed = false;
362 extraEvents.RescheduleEvent(EVENT_RESET_KNOCKBACK_EMOTE, 5000);
363 }
364 events.RepeatEvent(20000);
365 break;
366 }
367 case EVENT_LAVA_BURST:
368 {
369 DoCastAOE(SPELL_LAVA_BURST);
370 break;
371 }
372 case EVENT_MAGMA_BLAST:
373 {
374 _processingMagmaBurst = false;
375
376 if (!IsVictimWithinMeleeRange())
377 {
378 DoCastRandomTarget(SPELL_MAGMA_BLAST);
379
380 if (!_hasYelledMagmaBurst)
381 {
382 Talk(SAY_MAGMABURST);
383 _hasYelledMagmaBurst = true;
384 }
385 }
386
387 break;
388 }
389 case EVENT_MIGHT_OF_RAGNAROS:
390 {
391 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, [](Unit const* target)
392 {
393 return target->IsPlayer() && target->getPowerType() == POWER_MANA;
394 }))
395 {
396 if (me->CastSpell(target, SPELL_MIGHT_OF_RAGNAROS) == SPELL_CAST_OK && _isKnockbackEmoteAllowed)
397 {
398 Talk(SAY_KNOCKBACK, me);
399 _isKnockbackEmoteAllowed = false;
400 extraEvents.RescheduleEvent(EVENT_RESET_KNOCKBACK_EMOTE, 5000);
401 }
402 }
403 events.RepeatEvent(urand(11000, 30000));
404 break;
405 }
406 case EVENT_SUBMERGE:
407 {
408 events.CancelEventGroup(PHASE_EMERGED);
409 events.SetPhase(PHASE_SUBMERGED);
410 extraEvents.SetPhase(PHASE_SUBMERGED);
411 me->SetReactState(REACT_PASSIVE);
412 me->InterruptNonMeleeSpells(false);
413 me->AttackStop();
414 DoResetThreat();
415 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
416 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_SUBMERGED);
417 DoCastSelf(SPELL_RAGNA_SUBMERGE_VISUAL, true);
418 //me->HandleEmoteCommand(EMOTE_ONESHOT_SUBMERGE);
419
420 Talk(_hasSubmergedOnce ? SAY_REINFORCEMENTS2 : SAY_REINFORCEMENTS1);
421
422 DoCastAOE(SPELL_SUMMON_SONS_FLAME);
423
424 if (!_hasSubmergedOnce)
425 {
426 _hasSubmergedOnce = true;
427 }
428
429 extraEvents.ScheduleEvent(EVENT_EMERGE, 90000, PHASE_SUBMERGED, PHASE_SUBMERGED);
430 break;
431 }
432 }
433
434 if (me->HasUnitState(UNIT_STATE_CASTING))
435 {
436 return;
437 }
438 }
439
440 DoMeleeAttackIfReady();
441 }

453 void HandleEmerge()
454 {
455 if (events.IsInPhase(PHASE_EMERGED))
456 {
457 return;
458 }
459
460 events.SetPhase(PHASE_EMERGED);
461 ScheduleCombatEvents();
462 extraEvents.CancelEventGroup(PHASE_SUBMERGED);
463 extraEvents.SetPhase(PHASE_EMERGED);
464
465 me->SetReactState(REACT_AGGRESSIVE);
466 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
467 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, 0);
468 me->HandleEmoteCommand(EMOTE_ONESHOT_EMERGE);
469
470 me->RemoveAurasDueToSpell(SPELL_RAGNA_SUBMERGE_VISUAL);
471
472 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true))
473 {
474 AttackStart(target);
475 }
476 }
477
478 void ScheduleCombatEvents()
479 {
480 events.RescheduleEvent(EVENT_ERUPTION, 15000, PHASE_EMERGED, PHASE_EMERGED);
481 events.RescheduleEvent(EVENT_WRATH_OF_RAGNAROS, 30000, PHASE_EMERGED, PHASE_EMERGED);
482 events.RescheduleEvent(EVENT_HAND_OF_RAGNAROS, 25000, PHASE_EMERGED, PHASE_EMERGED);
483 events.RescheduleEvent(EVENT_LAVA_BURST, 10000, PHASE_EMERGED, PHASE_EMERGED);
484 events.RescheduleEvent(EVENT_SUBMERGE, 180000, PHASE_EMERGED, PHASE_EMERGED);
485 events.RescheduleEvent(EVENT_MIGHT_OF_RAGNAROS, 11000, PHASE_EMERGED, PHASE_EMERGED);
486 }
