58void BattlegroundEY::PostUpdateImpl(uint32 diff)
59{
60 if (GetStatus() == STATUS_IN_PROGRESS)
61 {
62 _bgEvents.Update(diff);
63 while (uint32 eventId = _bgEvents.ExecuteEvent())
64 switch (eventId)
65 {
66 case BG_EY_EVENT_ADD_POINTS:
67 if (_ownedPointsCount[TEAM_ALLIANCE] > 0)
68 AddPoints(TEAM_ALLIANCE, BG_EY_TickPoints[_ownedPointsCount[TEAM_ALLIANCE] - 1]);
69 if (_ownedPointsCount[TEAM_HORDE] > 0)
70 AddPoints(TEAM_HORDE, BG_EY_TickPoints[_ownedPointsCount[TEAM_HORDE] - 1]);
71 _bgEvents.ScheduleEvent(BG_EY_EVENT_ADD_POINTS, BG_EY_FPOINTS_TICK_TIME - (GameTime::GetGameTimeMS().count() % BG_EY_FPOINTS_TICK_TIME));
72 break;
73 case BG_EY_EVENT_FLAG_ON_GROUND:
74 RespawnFlagAfterDrop();
75 break;
76 case BG_EY_EVENT_RESPAWN_FLAG:
77 RespawnFlag();
78 break;
79 case BG_EY_EVENT_CHECK_CPOINTS:
80 UpdatePointsState();
81 _bgEvents.ScheduleEvent(BG_EY_EVENT_CHECK_CPOINTS, BG_EY_FPOINTS_CHECK_TIME - (GameTime::GetGameTimeMS().count() % BG_EY_FPOINTS_CHECK_TIME));
82 break;
83 }
84 }

113void BattlegroundEY::AddPoints(TeamId teamId, uint32 points)
114{
115 uint8 honorRewards = uint8(m_TeamScores[teamId] / _honorTics);
116 m_TeamScores[teamId] += points;
117
118 for (; honorRewards < uint8(m_TeamScores[teamId] / _honorTics); ++honorRewards)
119 RewardHonorToTeam(GetBonusHonorFromKill(1), teamId);
120
121 UpdateWorldState(teamId == TEAM_ALLIANCE ? EY_ALLIANCE_RESOURCES : EY_HORDE_RESOURCES, std::min<uint32>(m_TeamScores[teamId], BG_EY_MAX_TEAM_SCORE));
122 if (m_TeamScores[teamId] >= BG_EY_MAX_TEAM_SCORE)
123 EndBattleground(teamId);
124}

126void BattlegroundEY::UpdatePointsState()
127{
128 std::vector<GameObject*> pointsVec;
129 for (uint8 point = 0; point < EY_POINTS_MAX; ++point)
130 {
131 pointsVec.push_back(GetBGObject(BG_EY_OBJECT_TOWER_CAP_FEL_REAVER + point));
132 _capturePointInfo[point]._playersCount[TEAM_ALLIANCE] = 0;
133 _capturePointInfo[point]._playersCount[TEAM_HORDE] = 0;
134 }
135
136 const BattlegroundPlayerMap& bgPlayerMap = GetPlayers();
137 for (BattlegroundPlayerMap::const_iterator itr = bgPlayerMap.begin(); itr != bgPlayerMap.end(); ++itr)
138 {
139 itr->second->SendUpdateWorldState(PROGRESS_BAR_SHOW, BG_EY_PROGRESS_BAR_DONT_SHOW);
140 for (uint8 point = 0; point < EY_POINTS_MAX; ++point)
141 if (GameObject* pointObject = pointsVec[point])
142 if (itr->second->CanCaptureTowerPoint() && itr->second->IsWithinDistInMap(pointObject, BG_EY_POINT_RADIUS))
143 {
144 itr->second->SendUpdateWorldState(PROGRESS_BAR_SHOW, BG_EY_PROGRESS_BAR_SHOW);
145 itr->second->SendUpdateWorldState(PROGRESS_BAR_PERCENT_GREY, BG_EY_PROGRESS_BAR_PERCENT_GREY);
146 itr->second->SendUpdateWorldState(PROGRESS_BAR_STATUS, _capturePointInfo[point]._barStatus);
147 ++_capturePointInfo[point]._playersCount[itr->second->GetTeamId()];
148 _capturePointInfo[point].player = itr->second;
149
150 // Xinef: ugly hax... area trigger is no longer called by client...
151 if (pointObject->GetEntry() == BG_OBJECT_FR_TOWER_CAP_EY_ENTRY && itr->second->GetDistance2d(2043.96f, 1729.68f) < 3.0f)
152 HandleAreaTrigger(itr->second, AT_FEL_REAVER_POINT);
153 }
154 }
155
156 for (uint8 point = 0; point < EY_POINTS_MAX; ++point)
157 {
158 _capturePointInfo[point]._barStatus += std::max<int8>(std::min<int8>(_capturePointInfo[point]._playersCount[TEAM_ALLIANCE] - _capturePointInfo[point]._playersCount[TEAM_HORDE], BG_EY_POINT_MAX_CAPTURERS_COUNT), -BG_EY_POINT_MAX_CAPTURERS_COUNT);
159 _capturePointInfo[point]._barStatus = std::max<int8>(std::min<int8>(_capturePointInfo[point]._barStatus, BG_EY_PROGRESS_BAR_ALI_CONTROLLED), BG_EY_PROGRESS_BAR_HORDE_CONTROLLED);
160
161 TeamId pointOwnerTeamId = TEAM_NEUTRAL;
162 if (_capturePointInfo[point]._barStatus <= BG_EY_PROGRESS_BAR_NEUTRAL_LOW)
163 pointOwnerTeamId = TEAM_HORDE;
164 else if (_capturePointInfo[point]._barStatus >= BG_EY_PROGRESS_BAR_NEUTRAL_HIGH)
165 pointOwnerTeamId = TEAM_ALLIANCE;
166
167 if (pointOwnerTeamId != _capturePointInfo[point]._ownerTeamId)
168 {
169 if (_capturePointInfo[point].IsUncontrolled())
170 EventTeamCapturedPoint(_capturePointInfo[point].player, pointOwnerTeamId, point);
171
172 if (pointOwnerTeamId == TEAM_NEUTRAL && _capturePointInfo[point].IsUnderControl())
173 EventTeamLostPoint(_capturePointInfo[point].player, point);
174 }
175 }
176}

220void BattlegroundEY::HandleAreaTrigger(Player* player, uint32 trigger)
221{
222 if (GetStatus() != STATUS_IN_PROGRESS || !player->IsAlive())
223 return;
224
225 switch (trigger)
226 {
227 case AT_BLOOD_ELF_POINT:
228 if (_capturePointInfo[POINT_BLOOD_ELF].IsUnderControl(player->GetTeamId()))
229 if (_flagState == BG_EY_FLAG_STATE_ON_PLAYER && GetFlagPickerGUID() == player->GetGUID())
230 EventPlayerCapturedFlag(player, BG_EY_OBJECT_FLAG_BLOOD_ELF);
231 break;
232 case AT_FEL_REAVER_POINT:
233 if (_capturePointInfo[POINT_FEL_REAVER].IsUnderControl(player->GetTeamId()))
234 if (_flagState == BG_EY_FLAG_STATE_ON_PLAYER && GetFlagPickerGUID() == player->GetGUID())
235 EventPlayerCapturedFlag(player, BG_EY_OBJECT_FLAG_FEL_REAVER);
236 break;
237 case AT_MAGE_TOWER_POINT:
238 if (_capturePointInfo[POINT_MAGE_TOWER].IsUnderControl(player->GetTeamId()))
239 if (_flagState == BG_EY_FLAG_STATE_ON_PLAYER && GetFlagPickerGUID() == player->GetGUID())
240 EventPlayerCapturedFlag(player, BG_EY_OBJECT_FLAG_MAGE_TOWER);
241 break;
242 case AT_DRAENEI_RUINS_POINT:
243 if (_capturePointInfo[POINT_DRAENEI_RUINS].IsUnderControl(player->GetTeamId()))
244 if (_flagState == BG_EY_FLAG_STATE_ON_PLAYER && GetFlagPickerGUID() == player->GetGUID())
245 EventPlayerCapturedFlag(player, BG_EY_OBJECT_FLAG_DRAENEI_RUINS);
246 break;
247 case 4512:
248 case 4515:
249 case 4517:
250 case 4519:
251 case 4530:
252 case 4531:
253 case 5866:
254 case AT_BLOOD_ELF_BUFF:
255 case AT_FEL_REAVER_BUFF:
256 case AT_MAGE_TOWER_BUFF:
257 case AT_DRAENEI_RUINS_BUFF:
258 break;
259 }

592GraveyardStruct const* BattlegroundEY::GetClosestGraveyard(Player* player)
593{
594 GraveyardStruct const* entry = sGraveyard->GetGraveyard(static_cast<uint16>(BG_EY_GRAVEYARD_MAIN_ALLIANCE) + player->GetTeamId());
595 GraveyardStruct const* nearestEntry = entry;
596
597 float pX = player->GetPositionX();
598 float pY = player->GetPositionY();
599 float pZ = player->GetPositionZ();
600 float dist = (entry->x - pX) * (entry->x - pX) + (entry->y - pY) * (entry->y - pY) + (entry->z - pZ) * (entry->z - pZ);
601 float minDist = dist;
602
603 for (uint8 i = 0; i < EY_POINTS_MAX; ++i)
604 if (_capturePointInfo[i].IsUnderControl(player->GetTeamId()))
605 {
606 entry = sGraveyard->GetGraveyard(m_CapturingPointTypes[i].GraveYardId);
607 dist = (entry->x - pX) * (entry->x - pX) + (entry->y - pY) * (entry->y - pY) + (entry->z - pZ) * (entry->z - pZ);
608 if (dist < minDist)
609 {
610 minDist = dist;
611 nearestEntry = entry;
612 }
613 }
614
615 return nearestEntry;
