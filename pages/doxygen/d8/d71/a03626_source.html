58 bool OnGossipHello(Player* player, Creature* creature) override
59 {
60 uint32 gossipMenuId = 0;
61 uint32 gossipTextId = 0;
62 switch (creature->GetEntry())
63 {
64 case NPC_FREYA_GOSSIP:
65 gossipMenuId = GOSSIP_MENU_FREYA;
66 gossipTextId = NPC_TEXT_FREYA;
67 break;
68 case NPC_HODIR_GOSSIP:
69 gossipMenuId = GOSSIP_MENU_HODIR;
70 gossipTextId = NPC_TEXT_HODIR;
71 break;
72 case NPC_MIMIRON_GOSSIP:
73 gossipMenuId = GOSSIP_MENU_MIMIRON;
74 gossipTextId = NPC_TEXT_MIMIRON;
75 break;
76 case NPC_THORIM_GOSSIP:
77 gossipMenuId = GOSSIP_MENU_THORIM;
78 gossipTextId = NPC_TEXT_THORIM;
79 break;
80 }
81
82 AddGossipItemFor(player, gossipMenuId, 0, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
83 SendGossipMenuFor(player, gossipTextId, creature->GetGUID());
84 return true;
85 }
86
87 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
88 {
89 ClearGossipMenuFor(player);
90 uint8 _keeper = 0;
91 switch (action)
92 {
93 case GOSSIP_ACTION_INFO_DEF+1:
94 AddGossipItemFor(player, GOSSIP_MENU_CONFIRM, 0, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
95 SendGossipMenuFor(player, NPC_TEXT_CONFIRM, creature);
96 break;
97 case GOSSIP_ACTION_INFO_DEF+2:
98 {
99 switch (creature->GetEntry())
100 {
101 case NPC_FREYA_GOSSIP:
102 creature->AI()->Talk(SAY_KEEPER_SELECTED);
103 _keeper = KEEPER_FREYA;
104 break;
105 case NPC_HODIR_GOSSIP:
106 creature->AI()->Talk(SAY_KEEPER_SELECTED);
107 _keeper = KEEPER_HODIR;
108 break;
109 case NPC_MIMIRON_GOSSIP:
110 creature->AI()->Talk(SAY_KEEPER_SELECTED);
111 _keeper = KEEPER_MIMIRON;
112 break;
113 case NPC_THORIM_GOSSIP:
114 creature->AI()->Talk(SAY_KEEPER_SELECTED);
115 _keeper = KEEPER_THORIM;
116 break;
117 }
118
119 creature->ReplaceAllNpcFlags(UNIT_NPC_FLAG_NONE);
120 CloseGossipMenuFor(player);
121
122 if (creature->GetInstanceScript())
123 {
124 creature->GetInstanceScript()->SetData(TYPE_WATCHERS, _keeper);
125 creature->DespawnOrUnsummon(6000);
126 }
127 }
128 }
129 return true;
130 }
