49 void SetData(uint32 type, uint32 data) override
50 {
51 switch (type)
52 {
53 case TYPE_LORD_COBRAHN:
54 case TYPE_LORD_PYTHAS:
55 case TYPE_LADY_ANACONDRA:
56 case TYPE_LORD_SERPENTIS:
57 case TYPE_MUTANUS:
58 _encounters[type] = data;
59 break;
60 }
61
62 if (data == DONE)
63 SaveToDB();
64
65 if (type == TYPE_LORD_COBRAHN && _encounters[TYPE_LORD_SERPENTIS] != DONE)
66 {
67 instance->LoadGrid(-120.163f, -24.624f);
68 if (Creature* serpentis = instance->GetCreature(SerpentisGUID))
69 serpentis->AI()->Talk(SAY_SERPENTIS);
70 }
71
72 if (type != TYPE_MUTANUS && _encounters[TYPE_LORD_COBRAHN] == DONE && _encounters[TYPE_LORD_PYTHAS] == DONE &&
73 _encounters[TYPE_LADY_ANACONDRA] == DONE && _encounters[TYPE_LORD_SERPENTIS] == DONE)
74 {
75 instance->LoadGrid(-134.97f, 125.402f);
76 if (Creature* disciple = instance->GetCreature(DiscipleOfNaralexGUID))

101 void Load(const char* in) override
102 {
103 if (!in)

105
106 char dataHead1, dataHead2;
107 std::istringstream loadStream(in);
108 loadStream >> dataHead1 >> dataHead2;
109 if (dataHead1 == 'W' && dataHead2 == 'C')
110 {
111 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
112 {
113 loadStream >> _encounters[i];
114 if (_encounters[i] == IN_PROGRESS)
115 _encounters[i] = NOT_STARTED;
116 }
117 }
118 }
