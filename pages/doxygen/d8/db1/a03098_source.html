57struct boss_sartura : public BossAI
58{
59 boss_sartura(Creature* creature) : BossAI(creature, DATA_SARTURA) {}
60
61 void InitializeAI() override
62 {
63 me->m_CombatDistance = 60.f;
64 me->m_SightDistance = 60.f;
65 Reset();
66 }

68 void Reset() override

77 void EnterCombat(Unit* who) override
78 {
79 BossAI::EnterCombat(who);
80 Talk(SAY_AGGRO);
81 events.ScheduleEvent(EVENT_SARTURA_WHIRLWIND, 12s, 22s);
82 events.ScheduleEvent(EVENT_SPELL_BERSERK, 10min);
83 events.ScheduleEvent(EVENT_SARTURA_SUNDERING_CLEAVE, 2400ms, 3s);

88 _JustDied();
89 Talk(SAY_DEATH);

92 void KilledUnit(Unit* /*victim*/) override
93 {
94 Talk(SAY_SLAY);
95 }
96
97 void DamageTaken(Unit*, uint32& /*damage*/, DamageEffectType, SpellSchoolMask) override
98 {
99 if (!enraged && HealthBelowPct(20))
100 {
101 DoCastSelf(SPELL_ENRAGE);
102 enraged = true;
103 }
104 }
105
106 void UpdateAI(uint32 diff) override
107 {
108 if (!UpdateVictim())
109 return;
110
111 events.Update(diff);
112
113 while (uint32 eventId = events.ExecuteEvent())
114 {
115 switch (eventId)
116 {
117 case EVENT_SARTURA_WHIRLWIND:
118 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 100.0f, true))
119 {
120 me->GetThreatMgr().ResetAllThreat();
121 me->AddThreat(target, 1000.0f);
122 }
123 DoCastSelf(SPELL_WHIRLWIND);
124 events.ScheduleEvent(EVENT_SARTURA_WHIRLWIND_RANDOM, 2s, 7s);
125 events.ScheduleEvent(EVENT_SARTURA_WHIRLWIND_END, 15s);
126 break;
127 case EVENT_SARTURA_WHIRLWIND_RANDOM:
128 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100.0f, true))
129 {
130 me->GetThreatMgr().ResetAllThreat();
131 me->AddThreat(target, 1000.0f);
132 }
133 events.Repeat(2s, 7s);
134 break;
135 case EVENT_SARTURA_WHIRLWIND_END:
136 me->GetThreatMgr().ResetAllThreat();
137 me->SetReactState(REACT_AGGRESSIVE);
138 events.CancelEvent(EVENT_SARTURA_WHIRLWIND_RANDOM);
139 events.ScheduleEvent(EVENT_SARTURA_WHIRLWIND, 5s, 11s);
140 break;
141 case EVENT_SPELL_BERSERK:
142 if (!berserked)
143 {
144 DoCastSelf(SPELL_BERSERK, true);
145 berserked = true;
146 }
147 break;
148 case EVENT_SARTURA_SUNDERING_CLEAVE:
149 if (me->HasReactState(REACT_PASSIVE))
150 {
151 Milliseconds whirlwindTimer = events.GetTimeUntilEvent(EVENT_SARTURA_WHIRLWIND_END);
152 events.RescheduleEvent(EVENT_SARTURA_SUNDERING_CLEAVE, whirlwindTimer + 500ms);
153 }
154 else
155 {
156 DoCastVictim(SPELL_SUNDERING_CLEAVE, false);
157 events.RescheduleEvent(EVENT_SARTURA_SUNDERING_CLEAVE, 2400ms, 3s);
158 }
159 break;
160 default:
161 break;
162 }
163 }
164
165 DoMeleeAttackIfReady();
166 };
167
168 private:
169 bool enraged;
170 bool berserked;
171};
