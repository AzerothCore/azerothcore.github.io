38void BattlegroundDS::PostUpdateImpl(uint32 diff)
39{
40 if (GetStatus() != STATUS_IN_PROGRESS)
41 return;
42
43 _events.Update(diff);
44
45 while (uint32 eventId = _events.ExecuteEvent())
46 {
47 switch (eventId)
48 {
49 case BG_DS_EVENT_WATERFALL_WARNING:
50 // Add the water
51 DoorClose(BG_DS_OBJECT_WATER_2);
52 _events.ScheduleEvent(BG_DS_EVENT_WATERFALL_ON, BG_DS_WATERFALL_WARNING_DURATION);
53 break;
54 case BG_DS_EVENT_WATERFALL_ON:
55 // Active collision and start knockback timer
56 DoorClose(BG_DS_OBJECT_WATER_1);
57 _events.ScheduleEvent(BG_DS_EVENT_WATERFALL_OFF, BG_DS_WATERFALL_DURATION);
58 _events.ScheduleEvent(BG_DS_EVENT_WATERFALL_KNOCKBACK, BG_DS_WATERFALL_KNOCKBACK_TIMER);
59 break;
60 case BG_DS_EVENT_WATERFALL_OFF:
61 // Remove collision and water
62 DoorOpen(BG_DS_OBJECT_WATER_1);
63 DoorOpen(BG_DS_OBJECT_WATER_2);
64 _events.CancelEvent(BG_DS_EVENT_WATERFALL_KNOCKBACK);
65 _events.ScheduleEvent(BG_DS_EVENT_WATERFALL_WARNING, BG_DS_WATERFALL_TIMER_MIN, BG_DS_WATERFALL_TIMER_MAX);
66 break;
67 case BG_DS_EVENT_WATERFALL_KNOCKBACK:
68 // Repeat knockback while the waterfall still active
69 if (Creature* waterSpout = GetBGCreature(BG_DS_NPC_WATERFALL_KNOCKBACK))
70 waterSpout->CastSpell(waterSpout, BG_DS_SPELL_WATER_SPOUT, true);
71 _events.ScheduleEvent(eventId, BG_DS_WATERFALL_KNOCKBACK_TIMER);
72 break;
73 case BG_DS_EVENT_PIPE_KNOCKBACK:
74 for (uint32 i = BG_DS_NPC_PIPE_KNOCKBACK_1; i <= BG_DS_NPC_PIPE_KNOCKBACK_2; ++i)
75 if (Creature* waterSpout = GetBGCreature(i))
76 waterSpout->CastSpell(waterSpout, BG_DS_SPELL_FLUSH, true);
77 break;
78 }
79 }
80
81 if (_pipeKnockBackCount < BG_DS_PIPE_KNOCKBACK_TOTAL_COUNT)
82 {
83 if (_pipeKnockBackTimer < diff)
84 {
85 for (uint32 i = BG_DS_NPC_PIPE_KNOCKBACK_1; i <= BG_DS_NPC_PIPE_KNOCKBACK_2; ++i)
86 if (Creature* waterSpout = GetBGCreature(i))
87 waterSpout->CastSpell(waterSpout, BG_DS_SPELL_FLUSH, true);
88
89 ++_pipeKnockBackCount;
90 _pipeKnockBackTimer = BG_DS_PIPE_KNOCKBACK_DELAY;
91 }
92 else
93 _pipeKnockBackTimer -= diff;
94 }
95}

128 itr->second->RemoveAurasDueToSpell(48018);

131void BattlegroundDS::HandleAreaTrigger(Player* player, uint32 trigger)
132{
133 if (GetStatus() != STATUS_IN_PROGRESS)
134 return;
135
136 switch (trigger)
137 {
138 case 5347:
139 case 5348:
140 // Remove effects of Demonic Circle Summon
141 if (player->HasAura(SPELL_WARL_DEMONIC_CIRCLE))
142 player->RemoveAurasDueToSpell(SPELL_WARL_DEMONIC_CIRCLE);
143
144 // Someone has get back into the pipes and the knockback has already been performed,
145 // so we reset the knockback count for kicking the player again into the arena.
146 if (_pipeKnockBackCount >= BG_DS_PIPE_KNOCKBACK_TOTAL_COUNT)
147 _pipeKnockBackCount = 0;
148 break;
149 // OUTSIDE OF ARENA, TELEPORT!
150 case 5328:
151 player->NearTeleportTo(1290.44f, 744.96f, 3.16f, 1.6f);
152 break;
153 case 5329:
154 player->NearTeleportTo(1292.6f, 837.07f, 3.161f, 4.7f);
155 break;
156 case 5330:
157 player->NearTeleportTo(1250.68f, 790.86f, 3.16f, 0.0f);
158 break;
159 case 5331:
160 player->NearTeleportTo(1332.50f, 790.9f, 3.16f, 3.14f);
161 break;
162 case 5326: // -10
163 case 5343: // -40
164 case 5344: // -60
165 player->NearTeleportTo(1330.0f, 800.0f, 3.16f, player->GetOrientation());
166 break;
167 /*default:
168 Battleground::HandleAreaTrigger(player, trigger);
169 break;*/
170 }
