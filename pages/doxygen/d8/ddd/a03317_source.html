77 void Load(const char* in) override
78 {
79 if( !in )
80 return;
81
82 char dataHead1, dataHead2;
83 std::istringstream loadStream(in);
84 loadStream >> dataHead1 >> dataHead2;
85 if (dataHead1 == 'D' && dataHead2 == 'K')
86 {
87 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
88 {
89 uint32 tmpState;
90 loadStream >> tmpState;
91 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
92 tmpState = NOT_STARTED;
93 SetBossState(i, EncounterState(tmpState));
94 }
95 }
96 }
97 };
