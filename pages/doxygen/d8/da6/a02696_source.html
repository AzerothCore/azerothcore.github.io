50

86 void SchedulePossession()
87 {
88 _scheduler.Schedule(20s, 30s, [this](TaskContext context){
89 if (Unit* possessTarget = SelectTarget(SelectTargetMethod::Random, 1, 0, true, false))
90 {
91 DoCast(possessTarget, SPELL_POSSESS, true);
92 DoCast(possessTarget, SPELL_POSSESSED, true);
93 DoCastSelf(SPELL_POSSESS_INV, true);
94 _possessedTargetGuid = possessTarget->GetGUID();
95
96 // We must keep track of the possessed player, the aura falls off when their health drops below 50%.
97 // The encounter resumes when the aura falls off.
98 _scheduler.Schedule(1s, [this](TaskContext possessionContext) {
99 if (Player* possessedTarget = ObjectAccessor::GetPlayer(*me, _possessedTargetGuid))
100 {
101 if (!possessedTarget->HasAura(SPELL_POSSESSED) || possessedTarget->HealthBelowPct(50))
102 {
103 possessedTarget->RemoveAurasDueToSpell(SPELL_POSSESS);
104 possessedTarget->RemoveAurasDueToSpell(SPELL_POSSESSED);
105 me->RemoveAurasDueToSpell(SPELL_POSSESS_INV);
106 _possessedTargetGuid.Clear();
107 SchedulePossession();
108 }
109 else
110 {
111 possessionContext.Repeat(1s);
112 }
113 }
114 });
115 }
116 else
117 {
118 // No valid possession targets found, retry.
119 context.Repeat(1s);
120 }
121 });
122 }

137 TaskScheduler _scheduler;
