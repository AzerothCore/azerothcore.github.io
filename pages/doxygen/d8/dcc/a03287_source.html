123 void UpdateAI(uint32 diff) override
124 {
125 if( !TargetGUID && !me->GetVehicle() )
126 return;
127
128 Unit* t = ObjectAccessor::GetUnit(*me, TargetGUID);
129 if( !t && !(t = me->GetVehicleBase()) )
130 return;
131
132 if( t->isDead() )
133 {
134 t->RemoveAura(SPELL_CHANGE_VEHICLE);
135 me->RemoveAllAuras();
136 me->GetThreatMgr().ClearAllThreat();
137 me->CombatStop(true);
138 me->SetHealth(me->GetMaxHealth());
139 if( pInstance )
140 if( Creature* gormok = ObjectAccessor::GetCreature(*me, pInstance->GetGuidData(TYPE_GORMOK)) )
141 if( gormok->IsAlive() )
142 if( Vehicle* vk = gormok->GetVehicleKit() )
143 for( uint8 i = 0; i < 4; ++i )
144 if( !vk->GetPassenger(i) )
145 {
146 me->EnterVehicleUnattackable(gormok, i);
147 Reset();
148 break;
149 }
150 TargetGUID.Clear();
151 return;
152 }
153
154 events.Update(diff);
155
156 if( me->HasUnitState(UNIT_STATE_CASTING) )
157 return;
158
159 switch( events.ExecuteEvent() )
160 {
161 case 0:
162 break;
163 case EVENT_SPELL_SNOBOLLED:
164 if( t->GetTypeId() == TYPEID_PLAYER )
165 me->CastSpell((Unit*)nullptr, SPELL_SNOBOLLED, true);
166
167 break;
168 case EVENT_SPELL_BATTER:
169 if( t->GetTypeId() == TYPEID_PLAYER )
170 me->CastSpell(t, SPELL_BATTER);
171 events.RepeatEvent(urand(6000, 8000));
172 break;
173 case EVENT_SPELL_FIRE_BOMB:
174 {
175 if( t->GetTypeId() != TYPEID_PLAYER && pInstance )
176 {
177 GuidVector validPlayers;
178 Map::PlayerList const& pl = me->GetMap()->GetPlayers();
179 Creature* gormok = ObjectAccessor::GetCreature(*me, pInstance->GetGuidData(TYPE_GORMOK));
180
181 for( Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr )
182 {
183 if( Player* p = itr->GetSource() )
184 if( p->IsAlive() && p->GetGUID() != TargetGUID && (!gormok || !p->IsWithinMeleeRange(gormok)) )
185 validPlayers.push_back(p->GetGUID());
186 }
187
188 if( !validPlayers.empty() )
189 if( Player* p = ObjectAccessor::GetPlayer(*me, validPlayers.at(urand(0, validPlayers.size() - 1))) )
190 if( Creature* trigger = me->SummonCreature(NPC_FIRE_BOMB, *p, TEMPSUMMON_TIMED_DESPAWN, 60000) )
191 {
192 me->CastSpell(trigger, SPELL_FIRE_BOMB_AURA, true); // periodic damage aura, speed 14.0f
193 me->CastSpell(trigger, SPELL_FIRE_BOMB); // visual + initial damage 4k
194 }
195 }
196
197 events.RepeatEvent(urand(20000, 30000));
198 }
199 break;
200 case EVENT_SPELL_HEAD_CRACK:
201 if( t->GetTypeId() == TYPEID_PLAYER )
202 me->CastSpell(t, SPELL_HEAD_CRACK);
203 events.RepeatEvent(urand(30000, 35000));
204 break;
205 }
206
207 DoMeleeAttackIfReady();
208 }
209

211 {

213 {
214 t->RemoveAura(SPELL_CHANGE_VEHICLE);
215 if( t->IsAlive() )
216 t->RemoveAurasDueToSpell(SPELL_SNOBOLLED);
217 }
218 }

221 {
222 if( param == 1 && !TargetGUID )
