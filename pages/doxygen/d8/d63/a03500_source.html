80 struct boss_anomalusAI : public BossAI
81 {
82 boss_anomalusAI(Creature* creature) : BossAI(creature, DATA_ANOMALUS_EVENT)
83 {
84 }
85
86 bool achievement;
87 uint16 activeRifts;
88
89 void Reset() override
90 {
91 BossAI::Reset();
92 achievement = true;
93 me->CastSpell(me, SPELL_CLOSE_RIFTS, true);
94 }
95
96 uint32 GetData(uint32 data) const override
97 {
98 if (data == me->GetEntry())
99 return achievement;
100 return 0;
101 }
102
103 void SetData(uint32 type, uint32) override
104 {
105 if (type == me->GetEntry())
106 {
107 if (activeRifts > 0 && --activeRifts == 0 && me->HasAura(SPELL_RIFT_SHIELD))
108 {
109 events.DelayEvents(me->GetAura(SPELL_RIFT_SHIELD)->GetDuration() - 46000);
110 me->RemoveAura(SPELL_RIFT_SHIELD);
111 me->InterruptNonMeleeSpells(false);
112 }
113 achievement = false;
114 }
115 }
116
117 void JustSummoned(Creature* summon) override
118 {
119 summons.Summon(summon);
120 activeRifts++;
121 }
122
123 void EnterCombat(Unit* who) override
124 {
125 Talk(SAY_AGGRO);
126 BossAI::EnterCombat(who);
127
128 activeRifts = 0;
129 events.SetTimer(45000);
130 events.ScheduleEvent(EVENT_ANOMALUS_SPARK, 5000);
131 events.ScheduleEvent(EVENT_ANOMALUS_HEALTH, 1000);
132 events.ScheduleEvent(EVENT_ANOMALUS_SPAWN_RIFT, IsHeroic() ? 15000 : 25000);
133 if (IsHeroic())
134 events.ScheduleEvent(EVENT_ANOMALUS_ARCANE_ATTRACTION, 8000);
135 }
136
137 void JustDied(Unit* killer) override
138 {
139 Talk(SAY_DEATH);
140 BossAI::JustDied(killer);
141 me->CastSpell(me, SPELL_CLOSE_RIFTS, true);
142 }
143
144 void UpdateAI(uint32 diff) override
145 {
146 if (!UpdateVictim())
147 return;
148
149 events.Update(diff);
150 if (me->HasUnitState(UNIT_STATE_CASTING))
151 return;
152
153 switch (events.ExecuteEvent())
154 {
155 case EVENT_ANOMALUS_SPARK:
156 me->CastSpell(me->GetVictim(), SPELL_SPARK, false);
157 events.ScheduleEvent(EVENT_ANOMALUS_SPARK, 5000);
158 break;
159 case EVENT_ANOMALUS_HEALTH:
160 if (me->HealthBelowPct(51))
161 {
162 //First time we reach 51%, the next rift going to be empowered following timings.
163 events.CancelEvent(EVENT_ANOMALUS_SPAWN_RIFT);
164 events.ScheduleEvent(EVENT_ANOMALUS_SPAWN_RIFT_EMPOWERED, 1000);
165 break;
166 }
167 events.ScheduleEvent(EVENT_ANOMALUS_HEALTH, 1000);
168 break;
169 case EVENT_ANOMALUS_ARCANE_ATTRACTION:
170 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
171 me->CastSpell(target, SPELL_ARCANE_ATTRACTION, false);
172 events.ScheduleEvent(EVENT_ANOMALUS_ARCANE_ATTRACTION, 15000);
173 break;
174 case EVENT_ANOMALUS_SPAWN_RIFT:
175 Talk(SAY_RIFT);
176 Talk(EMOTE_RIFT);
177 me->CastSpell(me, SPELL_CREATE_RIFT, false);
178 //Once we hit 51% hp mark, after each rift we spawn an empowered
179 events.ScheduleEvent(me->HealthBelowPct(51) ? EVENT_ANOMALUS_SPAWN_RIFT_EMPOWERED : EVENT_ANOMALUS_SPAWN_RIFT, IsHeroic() ? 15000 : 25000);
180 break;
181 case EVENT_ANOMALUS_SPAWN_RIFT_EMPOWERED:
182 Talk(SAY_RIFT);
183 Talk(EMOTE_RIFT);
184
185 me->CastSpell(me, SPELL_CREATE_RIFT, false);
186 me->CastSpell(me, SPELL_RIFT_SHIELD, true);
187 me->m_Events.AddEvent(new ChargeRifts(me), me->m_Events.CalculateTime(1000));
188 events.DelayEvents(46000);
189 //As we just spawned an empowered spawn a normal one
190 events.ScheduleEvent(EVENT_ANOMALUS_SPAWN_RIFT, IsHeroic() ? 15000 : 25000);
191 break;
192 }
193
194 DoMeleeAttackIfReady();
195 }
196
197 bool CheckEvadeIfOutOfCombatArea() const override
198 {
199 return me->GetHomePosition().GetExactDist2d(me) > 60.0f;
200 }
201 };
