194 {
195 npc_oculus_unstable_sphereAI(Creature* c) : ScriptedAI(c) {}

197 uint32 timer;

199
200 void PickNewLocation()
201 {
202 float dist = rand_norm() * 40.0f;
203 float angle = rand_norm() * 2 * M_PI;
204 me->GetMotionMaster()->MovePoint(1, 961.29f + dist * cos(angle), 1049.0f + dist * std::sin(angle), 360.0f);
205 }
206
207 void MovementInform(uint32 type, uint32 id) override
208 {
209 if( type != POINT_MOTION_TYPE || id != 1 )
210 return;
211
212 if( !located )
213 gonext = true;
214 }
215
216 void Reset() override
217 {
218 me->SetReactState(REACT_PASSIVE);
219 me->SetSpeed(MOVE_RUN, 1.4f, true);
220 me->CastSpell(me, SPELL_UNSTABLE_SPHERE_PASSIVE, true);
221 me->CastSpell(me, SPELL_UNSTABLE_SPHERE_TIMER, true);
222 timer = 0;
223 located = false;
224 gonext = false;
225
226 PickNewLocation();
227 }
228
229 void AttackStart(Unit* /*who*/) override {}
230 void MoveInLineOfSight(Unit* /*who*/) override {}
231
232 void UpdateAI(uint32 diff) override
233 {
234 if( timer == 0 )
235 me->CastSpell(me, SPELL_TELEPORT_VISUAL, true);
236
237 timer += diff;
238
239 if( timer > 10000 )
240 {
241 if( !located )
242 me->GetMotionMaster()->MoveIdle();
243 located = true;
244 me->CastSpell(me, SPELL_UNSTABLE_SPHERE_PULSE, true);
245 timer -= 2000;
246 }
247
248 if( !located && gonext )
249 {
250 PickNewLocation();
251 gonext = false;
252 }
253 }
254 };
