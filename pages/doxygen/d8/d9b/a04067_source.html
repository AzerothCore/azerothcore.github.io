125 void Load(char const* str) override
126 {
127 if (!str)
128 {
129 OUT_LOAD_INST_DATA_FAIL;
130 return;
131 }
132
133 OUT_LOAD_INST_DATA(str);
134
135 char dataHead1, dataHead2;
136
137 std::istringstream loadStream(str);
138 loadStream >> dataHead1 >> dataHead2;
139
140 if (dataHead1 == 'E' && dataHead2 == 'Y')
141 {
142 for (uint32 i = 0; i < MAX_ENCOUNTER; ++i)
143 {
144 uint32 tmpState;
145 loadStream >> tmpState;
146 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
147 tmpState = NOT_STARTED;
148 SetBossState(i, EncounterState(tmpState));
149 }
150 }
151 else
152 OUT_LOAD_INST_DATA_FAIL;
153
154 OUT_LOAD_INST_DATA_COMPLETE;
155 }
156 };
