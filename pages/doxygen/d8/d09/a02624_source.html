79 void OnUnitDeath(Unit* unit) override
80 {
81 Creature* creature = unit->ToCreature();
82 if (!creature)
83 return;
84
85 switch (creature->GetEntry())
86 {
87 case NPC_COLDMIST_WIDOW:
88 case NPC_COLDMIST_STALKER:
89 case NPC_SHADOWBAT:
90 case NPC_VAMPIRIC_SHADOWBAT:
91 case NPC_GREATER_SHADOWBAT:
92 case NPC_PHASE_HOUND:
93 case NPC_DREADBEAST:
94 case NPC_SHADOWBEAST:
95 if (GetBossState(DATA_OPTIONAL_BOSS) == TO_BE_DECIDED)
96 {
97 ++OptionalBossCount;
98 if (OptionalBossCount == OPTIONAL_BOSS_REQUIRED_DEATH_COUNT)
99 {
100 switch (urand(NPC_HYAKISS_THE_LURKER, NPC_ROKAD_THE_RAVAGER))
101 {
102 case NPC_HYAKISS_THE_LURKER:
103 instance->SummonCreature(NPC_HYAKISS_THE_LURKER, OptionalSpawn[0]);
104 break;
105 case NPC_SHADIKITH_THE_GLIDER:
106 instance->SummonCreature(NPC_SHADIKITH_THE_GLIDER, OptionalSpawn[1]);
107 break;
108 case NPC_ROKAD_THE_RAVAGER:
109 instance->SummonCreature(NPC_ROKAD_THE_RAVAGER, OptionalSpawn[2]);
110 break;
111 }
112 }
113 }
114 break;
115 default:
116 break;
117 }
118 }

133 bool SetBossState(uint32 type, EncounterState state) override
134 {
135 if (!InstanceScript::SetBossState(type, state))
136 return false;
137
138 switch (type)
139 {
140 case DATA_OPERA_PERFORMANCE:
141 if (state == DONE)
142 {
143 HandleGameObject(m_uiStageDoorLeftGUID, true);
144 HandleGameObject(m_uiStageDoorRightGUID, true);
145 if (GameObject* sideEntrance = instance->GetGameObject(m_uiSideEntranceDoor))
146 sideEntrance->RemoveGameObjectFlag(GO_FLAG_LOCKED);
147 instance->UpdateEncounterState(ENCOUNTER_CREDIT_KILL_CREATURE, 16812, nullptr);
148 }
149 break;
150 case DATA_CHESS:
151 if (state == DONE)
152 DoRespawnGameObject(DustCoveredChest, DAY);
153 break;
154 default:
155 break;
156 }
157
158 return true;
159 }
