78 struct boss_ionarAI : public ScriptedAI
79 {
80 boss_ionarAI(Creature* creature) : ScriptedAI(creature), summons(creature)
81 {
82 m_pInstance = creature->GetInstanceScript();
83 }
84
85 InstanceScript* m_pInstance;
86 EventMap events;
87 SummonList summons;
88 uint8 HealthCheck;

90 void Reset() override
91 {
92 HealthCheck = 50;
93 events.Reset();
94 summons.DespawnAll();
95
96 me->SetVisible(true);
97
98 if (m_pInstance)
99 m_pInstance->SetData(TYPE_IONAR, NOT_STARTED);
100
101 // Ionar is immune to nature damage
102 me->ApplySpellImmune(0, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_NATURE, true);
103 }
104
105 void ScheduleEvents(bool spark)
106 {
107 events.SetPhase(1);
108 if (!spark)
109 events.RescheduleEvent(EVENT_CHECK_HEALTH, 1000, 0, 1);
110
111 events.RescheduleEvent(EVENT_BALL_LIGHTNING, 10000, 0, 1);
112 events.RescheduleEvent(EVENT_STATIC_OVERLOAD, 5000, 0, 1);
113 }
114
115 void EnterCombat(Unit*) override
116 {
117 me->SetInCombatWithZone();
118 Talk(SAY_AGGRO);
119
120 if (m_pInstance)
121 m_pInstance->SetData(TYPE_IONAR, IN_PROGRESS);
122
123 ScheduleEvents(false);
124 }
125
126 void JustDied(Unit*) override
127 {
128 Talk(SAY_DEATH);
129
130 summons.DespawnAll();
131
132 if (m_pInstance)
133 m_pInstance->SetData(TYPE_IONAR, DONE);
134 }
135
136 void KilledUnit(Unit* victim) override
137 {
138 if (victim->GetTypeId() != TYPEID_PLAYER)
139 return;
140
141 Talk(SAY_SLAY);
142 }
143
144 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
145 {
146 if (spell->Id == SPELL_DISPERSE)
147 Split();
148 }
149
150 void Split()
151 {
152 Talk(SAY_SPLIT);
153
154 Creature* spark;
155 for (uint8 i = 0; i < 5; ++i)
156 {
157 if ((spark = me->SummonCreature(NPC_SPARK_OF_IONAR, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN, 20000)))
158 {
159 summons.Summon(spark);
160 spark->CastSpell(spark, me->GetMap()->IsHeroic() ? SPELL_SPARK_VISUAL_TRIGGER_H : SPELL_SPARK_VISUAL_TRIGGER_N, true);
161 spark->CastSpell(spark, SPELL_RANDOM_LIGHTNING, true);
162 spark->SetUnitFlag(UNIT_FLAG_PACIFIED | UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
163 spark->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0);
164
165 if (Player* tgt = SelectTargetFromPlayerList(100))
166 spark->GetMotionMaster()->MoveFollow(tgt, 0.0f, 0.0f, MOTION_SLOT_CONTROLLED);
167 }
168 }
169
170 me->SetVisible(false);
171 me->SetControlled(true, UNIT_STATE_STUNNED);
172
173 events.SetPhase(2);
174 events.ScheduleEvent(EVENT_CALL_SPARKS, 15000, 0, 2);
175 }
176
177 void UpdateAI(uint32 diff) override
178 {
179 if (!UpdateVictim())
180 return;
181
182 events.Update(diff);
183
184 if (me->HasUnitState(UNIT_STATE_CASTING))
185 return;
186
187 switch (events.ExecuteEvent())
188 {
189 case EVENT_BALL_LIGHTNING:
190 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
191 me->CastSpell(target, me->GetMap()->IsHeroic() ? SPELL_BALL_LIGHTNING_H : SPELL_BALL_LIGHTNING_N, false);
192
193 events.RepeatEvent(10000 + rand() % 1000);
194 break;
195 case EVENT_STATIC_OVERLOAD:
196 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
197 me->CastSpell(target, me->GetMap()->IsHeroic() ? SPELL_STATIC_OVERLOAD_H : SPELL_STATIC_OVERLOAD_N, false);
198
199 events.RepeatEvent(5000 + rand() % 1000);
200 break;
201 case EVENT_CHECK_HEALTH:
202 if (HealthBelowPct(HealthCheck))
203 me->CastSpell(me, SPELL_DISPERSE, false);
204
205 events.RepeatEvent(1000);
206 return;
207 case EVENT_CALL_SPARKS:
208 {
209 EntryCheckPredicate pred(NPC_SPARK_OF_IONAR);
210 summons.DoAction(ACTION_CALLBACK, pred);
211 events.ScheduleEvent(EVENT_RESTORE, 2000, 0, 2);
212 return;
213 }
214 case EVENT_RESTORE:
215 EntryCheckPredicate pred(NPC_SPARK_OF_IONAR);
216 summons.DoAction(ACTION_SPARK_DESPAWN, pred);
217
218 me->SetVisible(true);
219 me->SetControlled(false, UNIT_STATE_STUNNED);
220 ScheduleEvents(true);
221 return;
222 }
223
224 DoMeleeAttackIfReady();
225 }
226 };
227};
