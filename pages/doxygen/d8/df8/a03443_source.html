172 void UpdateAI(uint32 diff) override
173 {
174 if (!me->IsInCombat() && sayGreet)
175 {
176 for (SummonList::iterator itr = summons.begin(); itr != summons.end(); ++itr)
177 {
178 if (pInstance)
179 {
180 if (Creature* cr = pInstance->instance->GetCreature(*itr))
181 {
182 if (cr->IsInCombat())
183 DoZoneInCombat();
184 }
185 }
186 }
187 }
188
189 if (!UpdateVictim())
190 return;
191
192 events.Update(diff);
193 if (me->HasUnitState(UNIT_STATE_CASTING))
194 return;
195
196 switch (events.ExecuteEvent())
197 {
198 case EVENT_POISON_BOLT:
199 if (!me->HasAura(RAID_MODE(SPELL_WIDOWS_EMBRACE, SPELL_MINION_WIDOWS_EMBRACE)))
200 {
201 me->CastCustomSpell(RAID_MODE(SPELL_POISON_BOLT_VOLLEY_10, SPELL_POISON_BOLT_VOLLEY_25), SPELLVALUE_MAX_TARGETS, RAID_MODE(3, 10), me, false);
202 }
203 events.RepeatEvent(urand(7000, 15000));
204 break;
205 case EVENT_RAIN_OF_FIRE:
206 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
207 {
208 me->CastSpell(target, RAID_MODE(SPELL_RAIN_OF_FIRE_10, SPELL_RAIN_OF_FIRE_25), false);
209 }
210 events.RepeatEvent(urand(8000, 18000));
211 break;
212 case EVENT_FRENZY:
213 if (!me->HasAura(RAID_MODE(SPELL_FRENZY_10, SPELL_FRENZY_25)))
214 {
215 Talk(SAY_FRENZY);
216 Talk(EMOTE_FRENZY);
217 me->CastSpell(me, RAID_MODE(SPELL_FRENZY_10, SPELL_FRENZY_25), true);
218 events.RepeatEvent(60000);
219 }
220 else
221 {
222 events.RepeatEvent(30000);
223 }
224 break;
225 }
226 DoMeleeAttackIfReady();
227 }
