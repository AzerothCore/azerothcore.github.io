103 }

105
106 if (InstanceScript* pInstance = me->GetInstanceScript())
107 {
108 pInstance->SetData(DATA_SHOW_INFINITE_TIMER, 0);
109 pInstance->DoRemoveAurasDueToSpellOnPlayers(SPELL_CORRUPTING_BLIGHT);
110 }
111 }

113 void DoAction(int32 param) override
114 {

116 return;
117
118 if (param == ACTION_RUN_OUT_OF_TIME)

120 Talk(SAY_FAIL);

126 void UpdateAI(uint32 diff) override
127 {
128 if (beamTimer)
129 {
130 beamTimer += diff;
131 if (beamTimer >= 2000)
132 {
133 me->CastSpell(me, SPELL_CORRUPTION_OF_TIME_CHANNEL, true);
134 beamTimer = 0;
135 }
136 }
137
138 if (!UpdateVictim())
139 return;
140
141 events.Update(diff);
142 if (me->HasUnitState(UNIT_STATE_CASTING))
143 return;
144
145 switch (events.ExecuteEvent())
146 {
147 case EVENT_SPELL_VOID_STRIKE:
148 me->CastSpell(me->GetVictim(), SPELL_VOID_STRIKE, false);
149 events.RepeatEvent(8000);
150 break;
151 case EVENT_SPELL_CORRUPTING_BLIGHT:
152 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
153 me->CastSpell(target, SPELL_CORRUPTING_BLIGHT, false);
154 events.RepeatEvent(12000);
155 break;
156 }
157
158 DoMeleeAttackIfReady();
159 }
160 };
