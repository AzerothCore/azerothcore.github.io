137 void UpdateAI(uint32 diff) override
138 {
139 if (!UpdateVictim())
140 return;
141
142 events.Update(diff);
143
144 if (me->HasUnitState(UNIT_STATE_CASTING))
145 return;
146
147 while (uint32 eventId = events.ExecuteEvent())
148 {
149 switch (eventId)
150 {
151 case EVENT_SHIMMER:
152 {
153 // Cast new random vulnerabilty on self
154 DoCast(me, SPELL_ELEMENTAL_SHIELD);
155 Talk(EMOTE_SHIMMER);
156 events.ScheduleEvent(EVENT_SHIMMER, urand(17000, 25000));
157 break;
158 }
159 case EVENT_BREATH:
160 DoCastVictim(_breathSpells.front());
161 _breathSpells.reverse();
162 events.ScheduleEvent(EVENT_BREATH, 60000);
163 break;
164 case EVENT_AFFLICTION:
165 {
166 uint32 afflictionSpellID = RAND(SPELL_BROODAF_BLUE, SPELL_BROODAF_BLACK, SPELL_BROODAF_RED, SPELL_BROODAF_BRONZE, SPELL_BROODAF_GREEN);
167 std::vector<Player*> playerTargets;
168 Map::PlayerList const& players = me->GetMap()->GetPlayers();
169 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
170 {
171 if (Player* player = itr->GetSource()->ToPlayer())
172 {
173 if (!player->IsGameMaster() && !player->IsSpectator() && player->IsAlive())
174 {
175 playerTargets.push_back(player);
176 }
177 }
178 }
179
180 if (playerTargets.size() > 12)
181 {
182 Acore::Containers::RandomResize(playerTargets, 12);
183 }
184
185 for (Player* player : playerTargets)
186 {
187 DoCast(player, afflictionSpellID, true);
188
189 if (player->HasAura(SPELL_BROODAF_BLUE) && player->HasAura(SPELL_BROODAF_BLACK) && player->HasAura(SPELL_BROODAF_RED) &&
190 player->HasAura(SPELL_BROODAF_BRONZE) && player->HasAura(SPELL_BROODAF_GREEN))
191 {
192 DoCast(player, SPELL_CHROMATIC_MUT_1);
193 }
194 }
195 }
196 events.ScheduleEvent(EVENT_AFFLICTION, 10000);
197 break;
198 case EVENT_FRENZY:
199 DoCast(me, SPELL_FRENZY);
200 events.ScheduleEvent(EVENT_FRENZY, 10000, 15000);
201 break;
202 }
203
204 if (me->HasUnitState(UNIT_STATE_CASTING))
205 return;
206 }
207
208 // Enrage if not already enraged and below 20%
209 if (!Enraged && HealthBelowPct(20))
210 {
211 DoCast(me, SPELL_ENRAGE);
212 Enraged = true;
213 }
214
215 DoMeleeAttackIfReady();
216 }
217
218 private:
219 std::list<uint32> _breathSpells;
220 bool Enraged;
221 ObjectGuid _playerGUID;
222 };

224 CreatureAI* GetAI(Creature* creature) const override
225 {
226 return GetBlackwingLairAI<boss_chromaggusAI>(creature);
227 }
