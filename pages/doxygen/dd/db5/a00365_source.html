27 // C++ implementation of Dmitry Vyukov's lock free MPSC queue
28 // http://www.1024cores.net/home/lock-free-algorithms/queues/non-intrusive-mpsc-node-based-queue
29 template<typename T>
30 class MPSCQueueNonIntrusive
31 {
32 public:
33 MPSCQueueNonIntrusive() : _head(new Node()), _tail(_head.load(std::memory_order_relaxed))
34 {
35 Node* front = _head.load(std::memory_order_relaxed);
36 front->Next.store(nullptr, std::memory_order_relaxed);
37 }
38
39 ~MPSCQueueNonIntrusive()
40 {
41 T* output;
42 while (Dequeue(output))
43 delete output;
44
45 Node* front = _head.load(std::memory_order_relaxed);
46 delete front;
47 }
48
49 void Enqueue(T* input)
50 {
51 Node* node = new Node(input);
52 Node* prevHead = _head.exchange(node, std::memory_order_acq_rel);
53 prevHead->Next.store(node, std::memory_order_release);
54 }
55
56 bool Dequeue(T*& result)
57 {
58 Node* tail = _tail.load(std::memory_order_relaxed);
59 Node* next = tail->Next.load(std::memory_order_acquire);
60 if (!next)
61 return false;
62
63 result = next->Data;
64 _tail.store(next, std::memory_order_release);
65 delete tail;
66 return true;
67 }
68
69 private:
70 struct Node
71 {
72 Node() = default;
73 explicit Node(T* data) : Data(data)
74 {
75 Next.store(nullptr, std::memory_order_relaxed);
76 }
77
78 T* Data;
79 std::atomic<Node*> Next;
80 };

89 // C++ implementation of Dmitry Vyukov's lock free MPSC queue
90 // http://www.1024cores.net/home/lock-free-algorithms/queues/intrusive-mpsc-node-based-queue
91 template<typename T, std::atomic<T*> T::* IntrusiveLink>
92 class MPSCQueueIntrusive
93 {
94 public:
95 MPSCQueueIntrusive() : _dummyPtr(reinterpret_cast<T*>(std::addressof(_dummy))), _head(_dummyPtr), _tail(_dummyPtr)
96 {
97 // _dummy is constructed from aligned_storage and is intentionally left uninitialized (it might not be default constructible)
98 // so we init only its IntrusiveLink here
99 std::atomic<T*>* dummyNext = new (&(_dummyPtr->*IntrusiveLink)) std::atomic<T*>();
100 dummyNext->store(nullptr, std::memory_order_relaxed);
101 }
102
103 ~MPSCQueueIntrusive()
104 {
105 T* output;
106 while (Dequeue(output))
107 delete output;
108 }
109
110 void Enqueue(T* input)
111 {
112 (input->*IntrusiveLink).store(nullptr, std::memory_order_release);
113 T* prevHead = _head.exchange(input, std::memory_order_acq_rel);
114 (prevHead->*IntrusiveLink).store(input, std::memory_order_release);
115 }
116
117 bool Dequeue(T*& result)
118 {
119 T* tail = _tail.load(std::memory_order_relaxed);
120 T* next = (tail->*IntrusiveLink).load(std::memory_order_acquire);
121 if (tail == _dummyPtr)
122 {
123 if (!next)
124 return false;
125
126 _tail.store(next, std::memory_order_release);
127 tail = next;
128 next = (next->*IntrusiveLink).load(std::memory_order_acquire);
129 }
130
131 if (next)
132 {
133 _tail.store(next, std::memory_order_release);
134 result = tail;
135 return true;
136 }
137
138 T* head = _head.load(std::memory_order_acquire);
139 if (tail != head)
140 return false;
141
142 Enqueue(_dummyPtr);
143 next = (tail->*IntrusiveLink).load(std::memory_order_acquire);
144 if (next)
145 {
146 _tail.store(next, std::memory_order_release);
147 result = tail;
148 return true;
149 }
150 return false;
151 }
152
153 private:
154 std::aligned_storage_t<sizeof(T), alignof(T)> _dummy;
155 T* _dummyPtr;
156 std::atomic<T*> _head;
157 std::atomic<T*> _tail;
158
159 MPSCQueueIntrusive(MPSCQueueIntrusive const&) = delete;
160 MPSCQueueIntrusive& operator=(MPSCQueueIntrusive const&) = delete;
161 };
