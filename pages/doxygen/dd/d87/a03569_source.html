131 void Reset() override
132 {
133 events.Reset();
134 summons.DespawnAll();
135
136 SlugeCount = 0;
137 SummonPhase = PHASE_SUMMON_UNFRIENDLY_DWARFES;
138
139 if (pInstance)
140 {
141 pInstance->SetData(BOSS_SJONNIR, NOT_STARTED);
142 pInstance->SetData(DATA_SJONNIR_ACHIEVEMENT, false);
143
144 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
145 if (pInstance->GetData(BOSS_TRIBUNAL_OF_AGES) == DONE)
146 {
147 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
148 if (GameObject* doors = me->GetMap()->GetGameObject(pInstance->GetGuidData(GO_SJONNIR_DOOR)))
149 doors->SetGoState(GO_STATE_ACTIVE);
150
151 if (GameObject* console = me->GetMap()->GetGameObject( pInstance->GetGuidData(GO_SJONNIR_CONSOLE)))
152 console->SetGoState(GO_STATE_READY);
153
154 if (Creature* brann = ObjectAccessor::GetCreature(*me, pInstance->GetGuidData(NPC_BRANN)))
155 {
156 brann->setDeathState(JUST_DIED);
157 brann->Respawn();
158 brann->AI()->DoAction(5);
159 }
160 }
161 }

164 void EnterCombat(Unit* /*who*/) override
165 {
166 Talk(SAY_AGGRO);
167
168 events.ScheduleEvent(EVENT_CHECK_HEALTH, 1000);
169 events.ScheduleEvent(EVENT_SHIELD, 14000 + rand() % 5000);
170 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, 6000 + rand() % 6000);
171 events.ScheduleEvent(EVENT_STATIC_CHARGE, 24000);
172 events.ScheduleEvent(EVENT_LIGHTNING_RING, 25000 + rand() % 6000);
173 events.ScheduleEvent(EVENT_SUMMON, 20000);
174 events.ScheduleEvent(EVENT_SUMMON, 21500);
175 events.ScheduleEvent(EVENT_SUMMON_SPEACH, 20000);
176
177 if (pInstance)
178 {
179 pInstance->SetData(BOSS_SJONNIR, IN_PROGRESS);
180
181 if (GameObject* doors = me->GetMap()->GetGameObject(pInstance->GetGuidData(GO_SJONNIR_DOOR)))
182 doors->SetGoState(GO_STATE_READY);
183
184 if (pInstance->GetData(BOSS_TRIBUNAL_OF_AGES) == DONE)
185 if (Creature* brann = ObjectAccessor::GetCreature(*me, pInstance->GetGuidData(NPC_BRANN)))
186 brann->AI()->DoAction(3);
187 }
188 }

200 void UpdateAI(uint32 diff) override
201 {
202 if (!UpdateVictim())
203 return;
204
205 events.Update(diff);
206 if (me->HasUnitState(UNIT_STATE_CASTING))
207 return;
208
209 switch (events.ExecuteEvent())
210 {
211 case EVENT_CHECK_HEALTH:
212 {
213 if (SummonPhase == PHASE_SUMMON_UNFRIENDLY_DWARFES && HealthBelowPct(50))
214 {
215 SummonPhase = PHASE_SUMMON_OOZE;
216 events.CancelEvent(EVENT_SUMMON);
217 events.ScheduleEvent(EVENT_SUMMON, 0);
218 events.ScheduleEvent(EVENT_SUMMON, 1500);
219
220 if (pInstance)
221 if (Creature* brann = ObjectAccessor::GetCreature(*me, pInstance->GetGuidData(NPC_BRANN)))
222 {
223 brann->Yell("What in the name o' Madoran did THAT do? Oh! Wait: I just about got it...", LANG_UNIVERSAL);
224 brann->PlayDirectSound(14276);
225 }
226 }
227
228 if (HealthBelowPct(20))
229 {
230 if (Creature* brann = ObjectAccessor::GetCreature(*me, pInstance->GetGuidData(NPC_BRANN)))
231 {
232 brann->Yell("Ha, that did it! Help's a-comin'! Take this, ya glowin' iron brute!", LANG_UNIVERSAL);
233 brann->PlayDirectSound(14277);
234 }
235 SummonPhase = PHASE_SUMMON_FRIENDLY_DWARFES;
236 me->CastSpell(me, SPELL_FRENZY, false);
237
238 events.CancelEvent(EVENT_SUMMON);
239 events.ScheduleEvent(EVENT_SUMMON, 0);
240 break;
241 }
242
243 events.RepeatEvent(1000);
244 break;
245 }
246 case EVENT_SHIELD:
247 {
248 me->CastSpell(me, DUNGEON_MODE(SPELL_LIGHTNING_SHIELD, SPELL_LIGHTNING_SHIELD_H), false);
249 events.RepeatEvent(14000 + rand() % 5000);
250 break;
251 }
252 case EVENT_CHAIN_LIGHTNING:
253 {
254 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true, 0))
255 me->CastSpell(target, DUNGEON_MODE(SPELL_CHAIN_LIGHTNING, SPELL_CHAIN_LIGHTNING_H), false);
256
257 events.RepeatEvent(6000 + rand() % 6000);
258 break;
259 }
260 case EVENT_STATIC_CHARGE:
261 {
262 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true, 0))
263 me->CastSpell(target, DUNGEON_MODE(SPELL_STATIC_CHARGE, SPELL_STATIC_CHARGE_H), false);
264
265 events.RepeatEvent(20000);
266 break;
267 }
268 case EVENT_LIGHTNING_RING:
269 {
270 me->CastSpell(me, DUNGEON_MODE(SPELL_LIGHTNING_RING, SPELL_LIGHTNING_RING_H), false);
271 events.RepeatEvent(25000 + rand() % 6000);
272 events.DelayEvents(10000); // Channel duration
273 break;
274 }
275 case EVENT_SUMMON_SPEACH:
276 {
277 if (Creature* brann = ObjectAccessor::GetCreature(*me, pInstance->GetGuidData(NPC_BRANN)))
278 {
279 brann->Yell("This is a wee bit trickier that before... Oh, bloody--incomin'!", LANG_UNIVERSAL);
280 brann->PlayDirectSound(14275);
281 }
282
283 break;
284 }
285 case EVENT_SUMMON:
286 {
287 switch (SummonPhase)
288 {
289 case PHASE_SUMMON_UNFRIENDLY_DWARFES:
290 {
291 SummonDwarfes(false);
292 events.RepeatEvent(20000);
293 break;
294 }
295 case PHASE_SUMMON_OOZE:
296 {
297 for (uint8 i = POS_GEN_RIGHT; i <= POS_GEN_LEFT; i++)
298 {
299 if (Creature* ooze = me->SummonCreature(NPC_OOZE, RoomPosition[i].GetPositionX(), RoomPosition[i].GetPositionY(), RoomPosition[i].GetPositionZ(), 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20000))
300 {
301 ActivatePipe(i);
302 ooze->GetMotionMaster()->MovePoint(0, RoomPosition[POS_ROOM_CENTER].GetPositionX(), RoomPosition[POS_ROOM_CENTER].GetPositionY(), RoomPosition[POS_ROOM_CENTER].GetPositionZ());
303 summons.Summon(ooze);
304 }
305 }
306 events.RepeatEvent(10000);
307 break;
308 }
309 case PHASE_SUMMON_FRIENDLY_DWARFES:
310 {
311 SummonDwarfes(true);
312 break;
313 }
314 }
315 break;
316 }
317 }
318
319 DoMeleeAttackIfReady();
320 }

353 void SummonDwarfes(bool friendly)
354 {
355 if (friendly)
356 {
357 for (int i = 0; i < 3; i++)
358 {
359 uint8 Pos = urand(POS_GEN_RIGHT, POS_GEN_LEFT);
360 if (Creature* dwarf = me->SummonCreature(NPC_DWARFES_FRIENDLY, RoomPosition[Pos].GetPositionX(), RoomPosition[Pos].GetPositionY(), RoomPosition[Pos].GetPositionZ(), 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20000))
361 {
362 if (Player* plr = SelectTargetFromPlayerList(100.0f))
363 dwarf->SetFaction(plr->GetFaction());
364
365 ActivatePipe(Pos);
366 dwarf->AI()->AttackStart(me);
367 summons.Summon(dwarf);
368 }
369 }
370 }
371 else
372 {
373 for (int i = 0; i < 2; i++)
374 {
375 uint8 Pos = urand(POS_GEN_RIGHT, POS_GEN_LEFT);
376 if (Creature* dwarf = me->SummonCreature(urand(0, 1) ? NPC_FORGED_IRON_TROGG : NPC_FORGED_IRON_DWARF, RoomPosition[Pos].GetPositionX(), RoomPosition[Pos].GetPositionY(), RoomPosition[Pos].GetPositionZ(), 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20000))
377 {
378 ActivatePipe(Pos);
379 dwarf->SetInCombatWithZone();
380 summons.Summon(dwarf);
381 }
382 }
383 }
384 }
