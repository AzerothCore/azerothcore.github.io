119 void UpdateAI(uint32 diff) override
120 {
121 if (!UpdateVictim())
122 return;
123
124 events.Update(diff);
125 if (me->HasUnitState(UNIT_STATE_CASTING))
126 return;
127
128 switch (events.ExecuteEvent())
129 {
130 case EVENT_ORMOROK_HEALTH:
131 if (me->HealthBelowPct(26))
132 {
133 me->CastSpell(me, SPELL_FRENZY, true);
134 Talk(EMOTE_FRENZY);
135 break;
136 }
137 events.ScheduleEvent(EVENT_ORMOROK_HEALTH, 1000);
138 break;
139 case EVENT_ORMOROK_TRAMPLE:
140 me->CastSpell(me, SPELL_TRAMPLE, false);
141 events.ScheduleEvent(EVENT_ORMOROK_TRAMPLE, 10000);
142 break;
143 case EVENT_ORMOROK_SPELL_REFLECTION:
144 Talk(SAY_REFLECT);
145 me->CastSpell(me, SPELL_SPELL_REFLECTION, false);
146 events.ScheduleEvent(EVENT_ORMOROK_SPELL_REFLECTION, 30000);
147 break;
148 case EVENT_ORMOROK_SUMMON:
149 if (Unit* target = SelectTarget(SelectTargetMethod::MinDistance, 0, 50.0f, true))
150 me->CastSpell(target, SPELL_SUMMON_CRYSTALLINE_TANGLER, true);
151 events.ScheduleEvent(EVENT_ORMOROK_SUMMON, 17000);
152 break;
153 case EVENT_ORMOROK_CRYSTAL_SPIKES:
154 Talk(SAY_CRYSTAL_SPIKES);
155 me->CastSpell(me, SPELL_CRYSTAL_SPIKES, false);
156 _spikesCount = 0;
157 events.ScheduleEvent(EVENT_ORMOROK_SUMMON_SPIKES, 300);
158 events.ScheduleEvent(EVENT_ORMOROK_CRYSTAL_SPIKES, 20000);
159 break;
160 case EVENT_ORMOROK_SUMMON_SPIKES:
161 if (++_spikesCount > 9)
162 break;
163 for (uint8 i = 0; i < 4; ++i)
164 {
165 float o = rand_norm() * 2.0f * M_PI;
166 float x = me->GetPositionX() + 5.0f * _spikesCount * cos(o);
167 float y = me->GetPositionY() + 5.0f * _spikesCount * std::sin(o);
168 float h = me->GetMapHeight(x, y, me->GetPositionZ());
169
170 if (h != INVALID_HEIGHT)
171 me->SummonCreature(NPC_CRYSTAL_SPIKE, x, y, h, 0, TEMPSUMMON_TIMED_DESPAWN, 7000);
172 }
173 events.ScheduleEvent(EVENT_ORMOROK_SUMMON_SPIKES, 200);
174 break;
175 }
176
177 DoMeleeAttackIfReady();
178 }
179 };
180};
