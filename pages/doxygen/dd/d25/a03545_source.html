233 void RollStance(uint8 stance, uint8 force = 0)
234 {
235 if (urand(0, 1))
236 stance = (++stance == 4 ? 1 : stance);
237 else
238 stance = (--stance == 0 ? 3 : stance);
239
240 if (force)
241 stance = force;
242
243 switch (stance)
244 {
245 case STANCE_DEFENSIVE:
246 Talk(SAY_DEFENSIVE_STANCE);
247
248 me->CastSpell(me, SPELL_DEFENSIVE_STANCE, true);
249 me->CastSpell(me, SPELL_DEFENSIVE_AURA, true);
250
251 events.DelayEvents(20000, STANCE_BERSERKER);
252 events.DelayEvents(20000, STANCE_BATTLE);
253
254 SetEquipmentSlots(false, EQUIP_SWORD, EQUIP_SHIELD, EQUIP_NO_CHANGE);
255 break;
256 case STANCE_BERSERKER:
257 Talk(SAY_BERSERKER_STANCE);
258
259 me->CastSpell(me, SPELL_BERSERKER_STANCE, true);
260 me->CastSpell(me, SPELL_BERSERKER_AURA, true);
261
262 events.DelayEvents(20000, STANCE_DEFENSIVE);
263 events.DelayEvents(20000, STANCE_BATTLE);
264
265 SetEquipmentSlots(false, EQUIP_SWORD, EQUIP_SWORD, EQUIP_NO_CHANGE);
266 break;
267 case STANCE_BATTLE:
268 Talk(SAY_BATTLE_STANCE);
269
270 me->CastSpell(me, SPELL_BATTLE_STANCE, true);
271 me->CastSpell(me, SPELL_BATTLE_AURA, true);
272
273 events.DelayEvents(20000, STANCE_BERSERKER);
274 events.DelayEvents(20000, STANCE_DEFENSIVE);
275
276 SetEquipmentSlots(false, EQUIP_MACE, EQUIP_UNEQUIP, EQUIP_NO_CHANGE);
277 break;
278 }
279
280 m_uiStance = stance;

291 void UpdateEscortAI(uint32 diff) override
292 {
293 if (!me->IsInCombat())
294 return;
295
296 // Return since we have no target
297 if (!UpdateVictim())
298 {
299 Reset();
300 return;
301 }
302
303 events.Update(diff);
304
305 if (me->HasUnitState(UNIT_STATE_CASTING))
306 return;
307
308 switch (events.ExecuteEvent())
309 {
310 case EVENT_BJARNGRIM_CHANGE_STANCE:
311 // roll new stance
312 RemoveStanceAura(m_uiStance);
313 RollStance(m_uiStance);
314 events.RepeatEvent(20000);
315 break;
316
320 case EVENT_BJARNGRIM_REFLECTION:
321 me->CastSpell(me, SPELL_BJARNGRIM_REFLETION, true);
322 events.RepeatEvent(8000 + rand() % 1000);
323 break;
324 case EVENT_BJARNGRIM_PUMMEL:
325 me->CastSpell(me->GetVictim(), SPELL_PUMMEL, false);
326 events.RepeatEvent(10000 + rand() % 1000);
327 break;
328 case EVENT_BJARNGRIM_KNOCK:
329 me->CastSpell(me, SPELL_KNOCK_AWAY, false);
330 events.RepeatEvent(20000 + rand() % 1000);
331 break;
332 case EVENT_BJARNGRIM_IRONFORM:
333 me->CastSpell(me, SPELL_IRONFORM, true);
334 events.RepeatEvent(18000 + rand() % 5000);
335 break;
336
340 case EVENT_BJARNGRIM_MORTAL_STRIKE:
341 me->CastSpell(me->GetVictim(), SPELL_MORTAL_STRIKE, false);
342 events.RepeatEvent(10000);
343 break;
344 case EVENT_BJARNGRIM_WHIRLWIND:
345 me->CastSpell(me, SPELL_WHIRLWIND, true);
346 events.RepeatEvent(25000);
347 break;
348
352 case EVENT_BJARNGRIM_INTERCEPT:
353 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
354 me->CastSpell(target, SPELL_INTERCEPT, true);
355
356 events.RepeatEvent(30000);
357 break;
358 case EVENT_BJARNGRIM_CLEAVE:
359 me->CastSpell(me->GetVictim(), SPELL_CLEAVE, false);
360 events.RepeatEvent(25000);
361 break;
362 case EVENT_BJARNGRIM_SLAM:
363 me->CastSpell(me->GetVictim(), SPELL_SLAM, false);
364 events.RepeatEvent(10000 + rand() % 2000);
365 break;
366 }
367
368 DoMeleeAttackIfReady();
369 }
370 };

424 me->CastSpell(bjarngrim, me->GetMap()->IsHeroic() ? SPELL_RENEW_STEEL_H : SPELL_RENEW_STEEL_N, true);

426 events.RepeatEvent(10000 + rand() % 4000);
427 break;

429
430 DoMeleeAttackIfReady();
