113 void KilledUnit(Unit*) override
114 {
115 if (events.GetNextEventTime(EVENT_KILL_TALK) == 0)
116 {
117 Talk(SAY_SLAY);
118 events.ScheduleEvent(EVENT_KILL_TALK, 6000);
119 }
120 }
121
122 void UpdateAI(uint32 diff) override
123 {
124 if (!me->IsInCombat())
125 {
126 events2.Update(diff);
127 if (events2.ExecuteEvent() == EVENT_PHANTOM)
128 {
129 me->CastSpell(me, SPELL_SUMMON_PHANTOM, true);
130 events2.ScheduleEvent(EVENT_PHANTOM, urand(20000, 25000));
131 }
132 }
133
134 if (!UpdateVictim())
135 return;
136
137 events.Update(diff);
138 if (me->HasUnitState(UNIT_STATE_CASTING))
139 return;
140
141 switch (events.ExecuteEvent())
142 {
143 case EVENT_GROUND_TREMOR:
144 if (roll_chance_i(50))
145 Talk(SAY_QUAKE);
146 me->CastSpell(me, me->GetDisplayId() != me->GetNativeDisplayId() ? SPELL_QUAKE : SPELL_GROUND_TREMOR, false);
147 events.ScheduleEvent(EVENT_GROUND_TREMOR, 10000);
148 break;
149 case EVENT_NUMBLING_SHOUT:
150 me->CastSpell(me, me->GetDisplayId() != me->GetNativeDisplayId() ? SPELL_NUMBING_ROAR : SPELL_NUMBING_SHOUT, false);
151 events.ScheduleEvent(EVENT_NUMBLING_SHOUT, 10000);
152 break;
153 case EVENT_DETERMINED_STAB:
154 me->CastSpell(me->GetVictim(), me->GetDisplayId() != me->GetNativeDisplayId() ? SPELL_DETERMINED_GORE : SPELL_DETERMINED_STAB, false);
155 events.ScheduleEvent(EVENT_DETERMINED_STAB, 8000);
156 break;
157 case EVENT_TRANSFORMATION:
158 Talk(EMOTE_TRANSFORM);
159 Talk(SAY_TRANSFORM);
160 me->CastSpell(me, SPELL_TRANSFORMATION, false);
161 me->CastSpell(me, SPELL_SUMMON_PHANTOM_TRANSFORM, true);
162 events.ScheduleEvent(EVENT_TRANSFORMATION, 10000);
163 break;
164 }
165
166 DoMeleeAttackIfReady();
167 }
168 };
