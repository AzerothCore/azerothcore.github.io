72void HandleBothDead(Creature* creature, bool jarien, Unit* killer)
73{
74 // Spirit talk
75 if (jarien)
76 {
77 if (Creature* spirit = creature->FindNearestCreature(NPC_SPIRIT_OF_JARIEN, 100.f))
78 {
79 spirit->AttackStop();
80 spirit->SetReactState(REACT_PASSIVE);
81 spirit->AI()->Talk(SAY_SPIRIT_BOTH_DEAD);
82 }
83 }
84 else
85 {
86 if (Creature* spirit = creature->FindNearestCreature(NPC_SPIRIT_OF_SOTHOS, 100.f))
87 {
88 spirit->AttackStop();
89 spirit->SetReactState(REACT_PASSIVE);
90 spirit->AI()->Talk(SAY_SPIRIT_BOTH_DEAD);
91 }
92 }
93
94 // chest spawn
95 if (GameObject* chest = killer->SummonGameObject(GO_JARIEN_AND_SOTHOS_HEIRLOOMS, heirloomsPosition.GetPositionX(), heirloomsPosition.GetPositionY(), heirloomsPosition.GetPositionZ(), heirloomsPosition.GetOrientation(), 0.f, 0.f, 0.f, 0.f, 0))
96 {
97 chest->setActive(true);
98 chest->SetGoState(GO_STATE_READY);
99 chest->SetLootState(GO_READY);
100 chest->SetUInt32Value(GAMEOBJECT_FACTION, 35);
101 }

104struct boss_jarien : public BossAI
105{
106 boss_jarien(Creature* creature) : BossAI(creature, DATA_JARIEN)
107 {
108 _scheduler.SetValidator([this]
109 {
110 return !me->HasUnitState(UNIT_STATE_CASTING);
111 });
112
113 me->SetReactState(REACT_PASSIVE);
114 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
115 me->SetImmuneToNPC(true);
116 _talked = false;
117 _phase = PHASE_TALK;
118 }
119
120 void Reset() override
121 {
122 _scheduler.CancelAll();
123 _phase = _talked ? PHASE_FIGHT : PHASE_TALK;
124 if (_sothosDied)
125 {
126 if (Creature* sothos = me->FindNearestCreature(NPC_SOTHOS, 200.f, false))
127 {
128 sothos->Respawn();
129 }
130 }
131 _sothosDied = false;
132 _Reset();
133 }
134
135 void IsSummonedBy(Unit* /*summoner*/) override
136 {
137 Talk(SAY_JARIEN_ON_SUMMON_0);
138
139 _scheduler.Schedule(6s, [this](TaskContext /*context*/)
140 {
141 Talk(SAY_JARIEN_ON_SUMMON_1);
142 })
143 .Schedule(12s, [this](TaskContext /*context*/)
144 {
145 Talk(SAY_JARIEN_ON_SUMMON_2);
146 _talked = true;
147 _phase = PHASE_FIGHT;
148 me->SetReactState(REACT_AGGRESSIVE);
149 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
150 me->SetImmuneToNPC(false);
151 });
152 }
153
154 void JustDied(Unit* killer) override
155 {
156 _JustDied();
157 if (Creature * sothos = me->FindNearestCreature(NPC_SOTHOS, 200.f))
158 {
159 sothos->AI()->DoAction(ACTION_PARTNER_DEAD);
160 }
161
162 me->SummonCreature(NPC_SPIRIT_OF_JARIEN, me->GetPosition());
163
164 if (_sothosDied && killer)
165 {
166 HandleBothDead(me, true, killer);
167 }
168 }
169
170 void DoAction(int32 action) override
171 {
172 if (action == ACTION_PARTNER_DEAD)
173 {
174 me->SetFullHealth();
175 DoCastSelf(SPELL_VENGEANCE);
176 if (Creature* sothos = me->FindNearestCreature(NPC_SOTHOS, 200.f, false))
177 {
178 Talk(EMOTE_JARIEN_VENGEANCE, sothos);
179 }
180 _sothosDied = true;
181 }
182 }
183
184 void EnterCombat(Unit* /*who*/) override
185 {
186 _EnterCombat();
187 _scheduler.Schedule(5s, [this](TaskContext context)
188 {
189 DoCastVictim(SPELL_SHADOW_SHOCK);
190 context.Repeat(10s, 12s);
191 })
192 .Schedule(3s, [this](TaskContext context)
193 {
194 DoCastVictim(SPELL_CLEAVE);
195 context.Repeat(10s);
196 })
197 .Schedule(11s, [this](TaskContext context)
198 {
199 DoCastRandomTarget(SPELL_CRIPPLE);
200 context.Repeat(30s);
201 })
202 .Schedule(10s, [this](TaskContext context)
203 {
204 DoCastVictim(SPELL_MORTAL_STRIKE);
205 context.Repeat(15s);
206 });
207 }
208
209 void UpdateAI(uint32 diff) override
210 {
211 if (!UpdateVictim() && _phase != PHASE_TALK)
212 {
213 return;
214 }
215
216 _scheduler.Update(diff, [this]
217 {
218 DoMeleeAttackIfReady();
219 });
220 }
221
222 protected:
223 TaskScheduler _scheduler;
224 uint8 _phase;
225 bool _talked;
226 bool _sothosDied;
227};
