76
77 struct boss_muruAI : public BossAI
78 {
79 boss_muruAI(Creature* creature) : BossAI(creature, DATA_MURU) { }
80
81 void Reset() override
82 {
83 BossAI::Reset();
84 me->SetReactState(REACT_AGGRESSIVE);
85 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
86 me->SetVisible(true);
87 }
88
89 void EnterCombat(Unit* who) override
90 {
91 BossAI::EnterCombat(who);
92 me->CastSpell(me, SPELL_NEGATIVE_ENERGY, true);
93 me->CastSpell(me, SPELL_SUMMON_BLOOD_ELVES_PERIODIC, true);
94 me->CastSpell(me, SPELL_OPEN_PORTAL_PERIODIC, true);
95 me->CastSpell(me, SPELL_DARKNESS_PERIODIC, true);
96
97 events.ScheduleEvent(EVENT_SPELL_ENRAGE, 600000);
98 }
99
100 void DamageTaken(Unit*, uint32& damage, DamageEffectType, SpellSchoolMask) override
101 {
102 if (damage >= me->GetHealth())
103 {
104 damage = 0;
105 if (!me->HasUnitFlag(UNIT_FLAG_NOT_SELECTABLE))
106 {
107 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
108 me->RemoveAllAuras();
109 me->CastSpell(me, SPELL_OPEN_ALL_PORTALS, true);
110 events.ScheduleEvent(EVENT_SUMMON_ENTROPIUS, 7000);
111 }
112 }
113 }
114
115 void JustSummoned(Creature* summon) override
116 {
117 if (summon->GetEntry() == NPC_ENTROPIUS)
118 summon->AI()->SetData(DATA_ENRAGE_TIMER, events.GetNextEventTime(EVENT_SPELL_ENRAGE));
119 else
120 {
121 if (!summon->IsTrigger())
122 summon->SetInCombatWithZone();
123 summons.Summon(summon);
124 }
125 }
126
127 void UpdateAI(uint32 diff) override
128 {
129 if (!UpdateVictim())
130 return;
131
132 events.Update(diff);
133 if (me->HasUnitState(UNIT_STATE_CASTING))
134 return;
135
136 switch (events.ExecuteEvent())
137 {
138 case EVENT_SPELL_ENRAGE:
139 me->CastSpell(me, SPELL_ENRAGE, true);
140 break;
141 case EVENT_SUMMON_ENTROPIUS:
142 me->CastSpell(me, SPELL_SUMMON_ENTROPIUS, false);
143 events.ScheduleEvent(EVENT_SET_INVISIBLE, 1000);
144 break;
145 case EVENT_SET_INVISIBLE:
146 me->SetVisible(false);
147 break;
148 }
149 }
150 };
151
152 CreatureAI* GetAI(Creature* creature) const override
153 {
154 return GetSunwellPlateauAI<boss_muruAI>(creature);
155 }

171 {
172 events.Reset();
173 events2.Reset();

179 void EnterEvadeMode(EvadeReason why) override
180 {
181 if (InstanceScript* instance = me->GetInstanceScript())

185

187 }

204 return 1 + uint32(events.GetTimer() / 12000);
205 return 0;
206 }
207
208 void JustDied(Unit* /*killer*/) override
209 {
210 if (InstanceScript* instance = me->GetInstanceScript())
211 if (Creature* muru = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_MURU)))
212 Unit::Kill(muru, muru);
213 }
214
215 void UpdateAI(uint32 diff) override
216 {
217 events2.Update(diff);
218 switch (events2.ExecuteEvent())
219 {
220 case EVENT_ENTROPIUS_AURAS:
221 me->CastSpell(me, SPELL_ENTROPIUS_COSMETIC_SPAWN, false);
222 me->CastSpell(me, SPELL_NEGATIVE_ENERGY_PERIODIC, true);
223 break;
224 case EVENT_ENTROPIUS_COMBAT:
225 me->SetReactState(REACT_AGGRESSIVE);
226 me->SetInCombatWithZone();
227 AttackStart(SelectTargetFromPlayerList(50.0f));
228 break;
229 }
230
231 if (!events2.Empty())
232 return;
233
234 if (!UpdateVictim())
235 return;
236
237 events.Update(diff);
238 if (me->HasUnitState(UNIT_STATE_CASTING))
239 return;
240
241 switch (events.ExecuteEvent())
242 {
243 case EVENT_SPELL_ENRAGE:
244 me->CastSpell(me, SPELL_ENRAGE, true);
245 break;
246 case EVENT_SPAWN_DARKNESS:
247 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
248 me->CastSpell(target, SPELL_DARKNESS, true);
249 events.ScheduleEvent(EVENT_SPAWN_DARKNESS, 15000);
250 break;
251 case EVENT_SPAWN_BLACK_HOLE:
252 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
253 me->CastSpell(target, SPELL_BLACK_HOLE, true);
254 events.ScheduleEvent(EVENT_SPAWN_BLACK_HOLE, 15000);
255 break;
256 }
257
258 DoMeleeAttackIfReady();
259 }
260 };

262 CreatureAI* GetAI(Creature* creature) const override

264 return GetSunwellPlateauAI<boss_entropiusAI>(creature);

304 case EVENT_START_BLACK_HOLE:
305 me->RemoveAurasDueToSpell(SPELL_BLACK_HOLE_SUMMON_VISUAL2);
306 me->CastSpell(me, SPELL_BLACK_HOLE_VISUAL2, true);
307 me->CastSpell(me, SPELL_BLACK_HOLE_PASSIVE, true);
308 break;
309 case EVENT_SWITCH_BLACK_HOLE_TARGET:
310 {
311 Map::PlayerList const& players = me->GetMap()->GetPlayers();
312 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
313 if (Player* player = itr->GetSource())
314 if (me->GetDistance2d(player) < 15.0f && player->GetPositionZ() < 72.0f && player->IsAlive() && !player->HasAura(SPELL_BLACK_HOLE_EFFECT))
315 {
316 me->GetMotionMaster()->MovePoint(0, player->GetPositionX(), player->GetPositionY(), player->GetPositionZ(), false, true);
317 events.ScheduleEvent(EVENT_SWITCH_BLACK_HOLE_TARGET, 5000);
318 return;
319 }
320 events.ScheduleEvent(EVENT_SWITCH_BLACK_HOLE_TARGET, 500);
321 break;
322 }
