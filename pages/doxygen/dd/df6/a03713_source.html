154
155 void UpdateAI(uint32 diff) override
156 {
157 if (!pInstance)
158 return;
159
160 events.Update(diff);
161
162 switch(events.ExecuteEvent())
163 {
164 case 0:
165 break;
166 case EVENT_SUMMON_KEEPER_OR_GUARDIAN:
167 bKorG = true;
168 spawned = true;
169 if (Creature* c = DoSummon(RAND(NPC_PORTAL_GUARDIAN, NPC_PORTAL_KEEPER), me, 2.0f, 0, TEMPSUMMON_DEAD_DESPAWN))
170 me->CastSpell(c, SPELL_PORTAL_CHANNEL, false);
171 events.RescheduleEvent(EVENT_SUMMON_KEEPER_TRASH, 20000);
172 break;
173 case EVENT_SUMMON_KEEPER_TRASH:
174 for (uint8 i = 0; i < 3 + addValue; ++i)
175 {
176 uint32 entry = RAND(NPC_AZURE_INVADER_1, NPC_AZURE_INVADER_2, NPC_AZURE_SPELLBREAKER_1, NPC_AZURE_SPELLBREAKER_2, NPC_AZURE_MAGE_SLAYER_1, NPC_AZURE_MAGE_SLAYER_2, NPC_AZURE_BINDER_1, NPC_AZURE_BINDER_2);
177 DoSummon(entry, me, 2.0f, 20000, TEMPSUMMON_DEAD_DESPAWN);
178 }
179 events.RepeatEvent(20000);
180 break;
181 case EVENT_SUMMON_ELITES:
182 spawned = true;
183 for (uint8 i = 0; i < 2 + addValue; ++i)
184 {
185 uint32 entry = RAND(NPC_AZURE_CAPTAIN, NPC_AZURE_RAIDER, NPC_AZURE_STALKER, NPC_AZURE_SORCEROR);
186 DoSummon(entry, me, 2.0f, 20000, TEMPSUMMON_DEAD_DESPAWN);
187 }
188 me->SetVisible(false);
189 break;
190 case EVENT_SUMMON_SABOTEOUR:
191 DoSummon(NPC_SABOTEOUR, me, 2.0f, 0, TEMPSUMMON_CORPSE_DESPAWN);
192 me->DespawnOrUnsummon(3000);
193 break;
194 }
195
196 if (!spawned)
197 return;
198
199 if (bKorG)
200 {
201 if (!me->IsNonMeleeSpellCast(false)) // keeper/guardian died => channeled spell interrupted
202 {
203 // if keeper/guard lost all victims, in enterevademode linking aura is removed, restore it:
204 if (pInstance)
205 for (SummonList::iterator itr = listOfMobs.begin(); itr != listOfMobs.end(); ++itr)
206 if (Creature* c = pInstance->instance->GetCreature(*itr))
207 if (c->IsAlive() && (c->GetEntry() == NPC_PORTAL_GUARDIAN || c->GetEntry() == NPC_PORTAL_KEEPER))
208 {
209 me->CastSpell(c, SPELL_PORTAL_CHANNEL, false);
210 return;
211 }
212 Unit::Kill(me, me, false);
213 }
214 }
215 else
216 {
217 if (listOfMobs.empty())
218 Unit::Kill(me, me, false);
219 }
220 }

223 {
224 events.Reset();
225 if (wave % 6 == 0) // just to be sure, shouln't occur
