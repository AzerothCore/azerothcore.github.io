40void GuardAI::EnterEvadeMode(EvadeReason /*why*/)
41{
42 if (!me->IsAlive())
43 {
44 me->GetMotionMaster()->MoveIdle();
45 me->CombatStop(true);
46 me->GetThreatMgr().ClearAllThreat();
47 return;
48 }
49
50 LOG_DEBUG("entities.unit", "Guard entry: {} enters evade mode.", me->GetEntry());
51
52 me->RemoveAllAuras();
53 me->GetThreatMgr().ClearAllThreat();
54 me->CombatStop(true);
55
56 // Remove ChaseMovementGenerator from MotionMaster stack list, and add HomeMovementGenerator instead
