91 void SetData(uint32 type, uint32 uiData) override
92 {
93 switch (type)

95 case TYPE_OVERSEER:
96 if (!--m_uiFelOverseerCount)

102 break;
103
104 case DATA_BLACKHEARTTHEINCITEREVENT:
105 if (uiData == DONE)
106 DoUseDoorOrButton(m_uiRefectoryDoorGUID);
107 m_auiEncounter[type] = uiData;
108 break;
109
110 case DATA_GRANDMASTERVORPILEVENT:
111 if (uiData == DONE)
112 DoUseDoorOrButton(m_uiScreamingHallDoorGUID);
113 m_auiEncounter[type] = uiData;
114 break;
115
116 case DATA_MURMUREVENT:
117 case TYPE_HELLMAW:

119 break;
120 }
121
122 if (uiData == DONE)

142 void Load(const char* in) override
143 {

147 return;
148 }
149
150 OUT_LOAD_INST_DATA(in);
151
152 char dataHead1, dataHead2;
153 std::istringstream loadStream(in);
154 loadStream >> dataHead1 >> dataHead2;
155 if (dataHead1 == 'S' && dataHead2 == 'L')
156 {
157 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
158 {
159 loadStream >> m_auiEncounter[i];
160 if (m_auiEncounter[i] == IN_PROGRESS)
161 m_auiEncounter[i] = NOT_STARTED;
162 }
163 }
164
165 OUT_LOAD_INST_DATA_COMPLETE;
