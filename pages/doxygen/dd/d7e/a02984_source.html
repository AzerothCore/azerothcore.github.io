58 boss_lieutenant_drakeAI(Creature* creature) : ScriptedAI(creature)
59 {
60 pathPoints.clear();
61 WPPath* path = sSmartWaypointMgr->GetPath(me->GetEntry());
62 if (!path || path->empty())
63 return;
64
65 pathPoints.push_back(G3D::Vector3(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ()));
66
67 uint32 wpCounter = 1;
68 WPPath::const_iterator itr;
69 while ((itr = path->find(wpCounter++)) != path->end())
70 {
71 WayPoint* wp = itr->second;
72 pathPoints.push_back(G3D::Vector3(wp->x, wp->y, wp->z));
73 }
74 }

121 void UpdateAI(uint32 diff) override
122 {
123 if (!UpdateVictim())
124 return;
125
126 events.Update(diff);
127 if (me->HasUnitState(UNIT_STATE_CASTING))
128 return;
129
130 switch (events.ExecuteEvent())
131 {
132 case EVENT_WHIRLWIND:
133 me->CastSpell(me, SPELL_WHIRLWIND, false);
134 events.ScheduleEvent(EVENT_WHIRLWIND, 25000);
135 break;
136 case EVENT_EXPLODING_SHOT:
137 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 40.0f))
138 me->CastSpell(target, SPELL_EXPLODING_SHOT, false);
139 events.ScheduleEvent(EVENT_EXPLODING_SHOT, 25000);
140 break;
141 case EVENT_MORTAL_STRIKE:
142 if (roll_chance_i(40))
143 Talk(SAY_MORTAL);
144 me->CastSpell(me->GetVictim(), SPELL_MORTAL_STRIKE, false);
145 events.ScheduleEvent(EVENT_MORTAL_STRIKE, 10000);
146 break;
147 case EVENT_FRIGHTENING_SHOUT:
148 if (roll_chance_i(40))
149 Talk(SAY_SHOUT);
150 me->CastSpell(me, SPELL_FRIGHTENING_SHOUT, false);
151 events.ScheduleEvent(EVENT_FRIGHTENING_SHOUT, 25000);
152 break;
153 case EVENT_HAMSTRING:
154 me->CastSpell(me->GetVictim(), SPELL_HAMSTRING, false);
155 events.ScheduleEvent(EVENT_HAMSTRING, 25000);
156 break;
157 }
158
159 DoMeleeAttackIfReady();
160 }
