84template<uint32 LIMIT>
85struct CoordPair
86{
87 CoordPair(uint32 x = 0, uint32 y = 0)
88 : x_coord(x)
89 , y_coord(y)
90 {}
91
92 CoordPair(const CoordPair<LIMIT>& obj)
93 : x_coord(obj.x_coord)
94 , y_coord(obj.y_coord)
95 {}
96
97 CoordPair<LIMIT>& operator=(const CoordPair<LIMIT>& obj)
98 {
99 x_coord = obj.x_coord;
100 y_coord = obj.y_coord;
101 return *this;
102 }
103
104 void dec_x(uint32 val)
105 {
106 if (x_coord > val)
107 x_coord -= val;
108 else
109 x_coord = 0;
110 }
111
112 void inc_x(uint32 val)
113 {
114 if (x_coord + val < LIMIT)
115 x_coord += val;
116 else
117 x_coord = LIMIT - 1;
118 }
119
120 void dec_y(uint32 val)
121 {
122 if (y_coord > val)
123 y_coord -= val;
124 else
125 y_coord = 0;
126 }
127
128 void inc_y(uint32 val)
129 {
130 if (y_coord + val < LIMIT)
131 y_coord += val;
132 else
133 y_coord = LIMIT - 1;
134 }
135
136 [[nodiscard]] bool IsCoordValid() const
137 {
138 return x_coord < LIMIT && y_coord < LIMIT;
139 }
140
141 CoordPair& normalize()
142 {
143 x_coord = std::min(x_coord, LIMIT - 1);
144 y_coord = std::min(y_coord, LIMIT - 1);
145 return *this;
146 }
147
148 [[nodiscard]] uint32 GetId() const
149 {
150 return y_coord * LIMIT + x_coord;
151 }
152
153 uint32 x_coord;
154 uint32 y_coord;
155};

174 template<class RET_TYPE, int CENTER_VAL>
175 inline RET_TYPE Compute(float x, float y, float center_offset, float size)
176 {
177 // calculate and store temporary values in double format for having same result as same mySQL calculations
178 double x_offset = (double(x) - center_offset) / size;
179 double y_offset = (double(y) - center_offset) / size;
180
181 int x_val = int(x_offset + CENTER_VAL + 0.5f);
182 int y_val = int(y_offset + CENTER_VAL + 0.5f);
183 return RET_TYPE(x_val, y_val);
184 }
185
186 inline GridCoord ComputeGridCoord(float x, float y)
187 {
188 return Compute<GridCoord, CENTER_GRID_ID>(x, y, CENTER_GRID_OFFSET, SIZE_OF_GRIDS);
189 }
190
191 inline CellCoord ComputeCellCoord(float x, float y)
192 {
193 return Compute<CellCoord, CENTER_GRID_CELL_ID>(x, y, CENTER_GRID_CELL_OFFSET, SIZE_OF_GRID_CELL);
194 }
195
196 inline CellCoord ComputeCellCoord(float x, float y, float& x_off, float& y_off)
197 {
198 double x_offset = (double(x) - CENTER_GRID_CELL_OFFSET) / SIZE_OF_GRID_CELL;
199 double y_offset = (double(y) - CENTER_GRID_CELL_OFFSET) / SIZE_OF_GRID_CELL;
200
201 int x_val = int(x_offset + CENTER_GRID_CELL_ID + 0.5f);
202 int y_val = int(y_offset + CENTER_GRID_CELL_ID + 0.5f);
203 x_off = (float(x_offset) - x_val + CENTER_GRID_CELL_ID) * SIZE_OF_GRID_CELL;
204 y_off = (float(y_offset) - y_val + CENTER_GRID_CELL_ID) * SIZE_OF_GRID_CELL;
205 return CellCoord(x_val, y_val);
206 }

208 inline void NormalizeMapCoord(float& c)
209 {
210 if (c > MAP_HALFSIZE - 0.5f)
211 c = MAP_HALFSIZE - 0.5f;
212 else if (c < -(MAP_HALFSIZE - 0.5f))
213 c = -(MAP_HALFSIZE - 0.5f);
214 }
215
216 inline bool IsValidMapCoord(float c)
217 {
218 return std::isfinite(c) && (std::fabs(c) <= MAP_HALFSIZE - 0.5f);
219 }
220
221 inline bool IsValidMapCoord(float x, float y)
222 {
223 return IsValidMapCoord(x) && IsValidMapCoord(y);
224 }
225
226 inline bool IsValidMapCoord(float x, float y, float z)
227 {
228 return IsValidMapCoord(x, y) && IsValidMapCoord(z);
229 }
230
231 inline bool IsValidMapCoord(float x, float y, float z, float o)
232 {
233 return IsValidMapCoord(x, y, z) && std::isfinite(o);
