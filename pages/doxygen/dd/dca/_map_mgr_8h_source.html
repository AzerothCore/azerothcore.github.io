116 // modulos a radian orientation to the range of 0..2PI
117 static float NormalizeOrientation(float o)
118 {
119 // fmod only supports positive numbers. Thus we have
120 // to emulate negative numbers
121 if (o < 0)
122 {
123 float mod = o * -1;
124 mod = std::fmod(mod, 2.0f * static_cast<float>(M_PI));
125 mod = -mod + 2.0f * static_cast<float>(M_PI);
126 return mod;
127 }
128 return std::fmod(o, 2.0f * static_cast<float>(M_PI));
129 }

173template<typename Worker>
174void MapMgr::DoForAllMaps(Worker&& worker)
175{
176 std::lock_guard<std::mutex> guard(Lock);
177
178 for (auto& mapPair : i_maps)
179 {
180 Map* map = mapPair.second;
181 if (MapInstanced* mapInstanced = map->ToMapInstanced())
182 {
183 MapInstanced::InstancedMaps& instances = mapInstanced->GetInstancedMaps();
184 for (auto& instancePair : instances)
185 worker(instancePair.second);
186 }
187 else
188 worker(map);
189 }
190}
191
192template<typename Worker>
193inline void MapMgr::DoForAllMapsWithMapId(uint32 mapId, Worker&& worker)
194{
195 std::lock_guard<std::mutex> guard(Lock);
196
197 auto itr = i_maps.find(mapId);
198 if (itr != i_maps.end())
199 {
200 Map* map = itr->second;
201 if (MapInstanced* mapInstanced = map->ToMapInstanced())
202 {
203 MapInstanced::InstancedMaps& instances = mapInstanced->GetInstancedMaps();
204 for (auto& p : instances)
205 worker(p.second);
206 }
207 else
208 worker(map);
209 }
210}
