32template<class T>
33inline void Acore::VisibleNotifier::Visit(GridRefMgr<T>& m)
34{
35 // Xinef: Update gameobjects only
36 if (i_gobjOnly)
37 return;
38
39 for (typename GridRefMgr<T>::iterator iter = m.begin(); iter != m.end(); ++iter)
40 {
41 if (i_largeOnly != iter->GetSource()->IsVisibilityOverridden())
42 continue;
43
44 vis_guids.erase(iter->GetSource()->GetGUID());
45 i_player.UpdateVisibilityOf(iter->GetSource(), i_data, i_visibleNow);
46 }
47}

538template<class Builder>
539void Acore::LocalizedPacketDo<Builder>::operator()(Player* p)
540{
541 LocaleConstant loc_idx = p->GetSession()->GetSessionDbLocaleIndex();
542 uint32 cache_idx = loc_idx + 1;
543 WorldPacket* data;
544
545 // create if not cached yet
546 if (i_data_cache.size() < cache_idx + 1 || !i_data_cache[cache_idx])
547 {
548 if (i_data_cache.size() < cache_idx + 1)
549 i_data_cache.resize(cache_idx + 1);
550
551 data = new WorldPacket();
552
553 i_builder(*data, loc_idx);
554
555 i_data_cache[cache_idx] = data;
556 }
557 else
558 data = i_data_cache[cache_idx];
559
560 p->SendDirectMessage(data);
561}

563template<class Builder>
564void Acore::LocalizedPacketListDo<Builder>::operator()(Player* p)
565{
566 LocaleConstant loc_idx = p->GetSession()->GetSessionDbLocaleIndex();
567 uint32 cache_idx = loc_idx + 1;
568 WorldPacketList* data_list;
569
570 // create if not cached yet
571 if (i_data_cache.size() < cache_idx + 1 || i_data_cache[cache_idx].empty())
572 {
573 if (i_data_cache.size() < cache_idx + 1)
574 i_data_cache.resize(cache_idx + 1);
575
576 data_list = &i_data_cache[cache_idx];
577
578 i_builder(*data_list, loc_idx);
579 }
580 else
581 data_list = &i_data_cache[cache_idx];
582
583 for (size_t i = 0; i < data_list->size(); ++i)
584 p->SendDirectMessage((*data_list)[i]);
