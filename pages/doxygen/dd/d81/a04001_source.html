139 me->SummonCreature(NPC_SHATTERED_ASSASSIN, AssassEntrance[0], AssassEntrance[1] - 8, AssassEntrance[2], 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);

141 me->SummonCreature(NPC_SHATTERED_ASSASSIN, AssassExit[0], AssassExit[1] - 8, AssassExit[2], 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
142 break;
143 case EVENT_SUMMON_ADDS:
144 for (uint8 i = 0; i < 2; ++i)
145 me->SummonCreature(NPC_HEARTHEN_GUARD + urand(0, 2), AddsEntrance[0], AddsEntrance[1], AddsEntrance[2], 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
146
147 events.ScheduleEvent(EVENT_SUMMON_ADDS, 30000);
148 break;
149 case EVENT_BLADE_DANCE:
150 events.DelayEvents(10001);
151 events.ScheduleEvent(EVENT_BLADE_DANCE, 40000);
152 events.ScheduleEvent(EVENT_MOVE_TO_NEXT_POINT, 0);
153 events.ScheduleEvent(EVENT_FINISH_BLADE_DANCE, 10000);
154 events.SetPhase(1);
155 me->CastSpell(me, SPELL_SPRINT, true);
156 break;
157 case EVENT_MOVE_TO_NEXT_POINT:
158 {
159 float x = 210 + frand(0.0f, 35.0f);
160 float y = -65.0f - frand(0.0f, 35.0f);
161 me->GetMotionMaster()->MovePoint(1, x, y, me->GetPositionZ());
162 break;
163 }
164 case EVENT_FINISH_BLADE_DANCE:
165 events.SetPhase(0);
166 me->GetMotionMaster()->Clear();
167 me->GetMotionMaster()->MoveChase(me->GetVictim());
168 if (IsHeroic())
169 events.ScheduleEvent(EVENT_SPELL_CHARGE, 3000);
170 break;
171 case EVENT_SPELL_CHARGE:
172 me->CastSpell(me->GetVictim(), SPELL_CHARGE, false);
173 break;
174 }
175
176 if (!events.IsInPhase(1))
177 DoMeleeAttackIfReady();
178 }
