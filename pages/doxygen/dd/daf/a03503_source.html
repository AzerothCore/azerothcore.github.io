94 void UpdateAI(uint32 diff) override
95 {
96 if (!UpdateVictim())
97 return;
98
99 events.Update(diff);
100 if (me->HasUnitState(UNIT_STATE_CASTING))
101 return;
102
103 switch (events.ExecuteEvent())
104 {
105 case EVENT_BATTLE_SHOUT:
106 me->CastSpell(me, SPELL_BATTLE_SHOUT, true);
107 events.ScheduleEvent(EVENT_BATTLE_SHOUT, 120000);
108 break;
109 case EVENT_FRIGHTENING_SHOUT:
110 me->CastSpell(me->GetVictim(), SPELL_FRIGHTENING_SHOUT, false);
111 events.ScheduleEvent(EVENT_FRIGHTENING_SHOUT, urand(15000, 20000));
112 break;
113 case EVENT_WHIRLWIND:
114 me->CastSpell(me, SPELL_WHIRLWIND, false);
115 events.ScheduleEvent(EVENT_WHIRLWIND, 16000);
116 break;
117 case EVENT_COMMANDER_CHARGE:
118 if (Unit* target = SelectTarget(SelectTargetMethod::MinDistance, 0, 25.0f))
119 me->CastSpell(target, SPELL_CHARGE, false);
120 events.ScheduleEvent(EVENT_COMMANDER_CHARGE, 20000);
121 break;
122 }
123
124 DoMeleeAttackIfReady();
125 }
126 };
