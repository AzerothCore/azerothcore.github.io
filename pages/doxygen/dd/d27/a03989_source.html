130 bool SetBossState(uint32 id, EncounterState state) override
131 {
132 if (!InstanceScript::SetBossState(id, state))
133 return false;
134
135 if (id == TYPE_MAGTHERIDON)
136 {
137 if (state == IN_PROGRESS)
138 {
139 for (ObjectGuid const& guid : _wardersSet)
140 if (Creature* warder = instance->GetCreature(guid))
141 if (warder->IsAlive())
142 {
143 warder->InterruptNonMeleeSpells(true);
144 warder->SetInCombatWithZone();
145 }
146 }
147 else
148 {
149 for (ObjectGuid const& guid : _cubesSet)
150 if (GameObject* cube = instance->GetGameObject(guid))
151 cube->SetGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
152
153 if (state == NOT_STARTED)
154 SetData(DATA_COLLAPSE, GO_READY);
155 }
156 }
157 return true;
158 }

160 void SetData(uint32 type, uint32 data) override
161 {
162 switch (type)
163 {
164 case DATA_CHANNELER_COMBAT:
165 if (GetBossState(TYPE_MAGTHERIDON) != IN_PROGRESS)
166 if (Creature* magtheridon = instance->GetCreature(_magtheridonGUID))
167 magtheridon->SetInCombatWithZone();
168 break;
169 case DATA_ACTIVATE_CUBES:
170 for (ObjectGuid const& guid : _cubesSet)
171 if (GameObject* cube = instance->GetGameObject(guid))
172 cube->RemoveGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
173 break;
174 case DATA_COLLAPSE:
175 for (ObjectGuid const& guid : _columnSet)
176 if (GameObject* column = instance->GetGameObject(guid))
177 column->SetGoState(GOState(data));
178 break;
179 }

193 void Load(char const* str) override
194 {
195 if (!str)
196 {
197 OUT_LOAD_INST_DATA_FAIL;
198 return;
199 }
200
201 OUT_LOAD_INST_DATA(str);
202
203 char dataHead1, dataHead2;
204
205 std::istringstream loadStream(str);
206 loadStream >> dataHead1 >> dataHead2;
207
208 if (dataHead1 == 'M' && dataHead2 == 'L')
209 {
210 for (uint32 i = 0; i < MAX_ENCOUNTER; ++i)
211 {
212 uint32 tmpState;
213 loadStream >> tmpState;
214 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
215 tmpState = NOT_STARTED;
216 SetBossState(i, EncounterState(tmpState));
217 }
218 }
219 else
220 OUT_LOAD_INST_DATA_FAIL;
221
222 OUT_LOAD_INST_DATA_COMPLETE;
223 }
