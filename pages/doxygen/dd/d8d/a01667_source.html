103void MotionMaster::UpdateMotion(uint32 diff)
104{
105 if (!_owner)
106 return;
107
108 ASSERT(!empty());
109
110 _cleanFlag |= MMCF_INUSE;
111
112 _cleanFlag |= MMCF_UPDATE;
113 if (!top()->Update(_owner, diff))
114 {
115 _cleanFlag &= ~MMCF_UPDATE;
116 MovementExpired();
117 }
118 else
119 _cleanFlag &= ~MMCF_UPDATE;
120
121 if (_expList)
122 {
123 for (size_t i = 0; i < _expList->size(); ++i)
124 {
125 MovementGenerator* mg = (*_expList)[i];
126 DirectDelete(mg);
127 }
128
129 delete _expList;
130 _expList = nullptr;
131
132 if (empty())
133 Initialize();
134 else if (needInitTop())
135 InitTop();
136 else if (_cleanFlag & MMCF_RESET)
137 top()->Reset(_owner);
138
139 _cleanFlag &= ~MMCF_RESET;
140 }
141
142 _cleanFlag &= ~MMCF_INUSE;
143}

174{
175 if (size() > 1)
176 {
177 MovementGenerator* curr = top();
178 pop();
179 if (curr) DirectDelete(curr);
180 }
181
182 while (!empty() && !top())

184

189 else if (reset)

227 else if (reset)

321void MotionMaster::MoveBackwards(Unit* target, float dist)
322{
323 if (!target)
324 {
325 return;
326 }
327
328 Position const& pos = target->GetPosition();
329 float angle = target->GetAngle(_owner);
330 G3D::Vector3 point;
331 point.x = pos.m_positionX + dist * cosf(angle);
332 point.y = pos.m_positionY + dist * sinf(angle);
333 point.z = pos.m_positionZ;
334
335 if (!_owner->GetMap()->CanReachPositionAndGetValidCoords(_owner, point.x, point.y, point.z, true, true))
336 {
337 return;
338 }
339
340 Movement::MoveSplineInit init(_owner);
341 init.MoveTo(point.x, point.y, point.z, false);
342 init.SetFacing(target);
343 init.SetOrientationInversed();
344 init.Launch();
345}

485void MotionMaster::MoveKnockbackFrom(float srcX, float srcY, float speedXY, float speedZ)
486{
487 //this function may make players fall below map
488 if (_owner->GetTypeId() == TYPEID_PLAYER)
489 return;
490
491 if (speedXY <= 0.1f)
492 return;
493
494 Position dest = _owner->GetPosition();
495 float moveTimeHalf = speedZ / Movement::gravity;
496 float dist = 2 * moveTimeHalf * speedXY;
497 float max_height = -Movement::computeFallElevation(moveTimeHalf, false, -speedZ);
498
499 // Use a mmap raycast to get a valid destination.
500 _owner->MovePositionToFirstCollision(dest, dist, _owner->GetRelativeAngle(srcX, srcY) + float(M_PI));
501
502 Movement::MoveSplineInit init(_owner);
503 init.MoveTo(dest.GetPositionX(), dest.GetPositionY(), dest.GetPositionZ());
504 init.SetParabolic(max_height, 0);
505 init.SetOrientationFixed(true);
506 init.SetVelocity(speedXY);
507 init.Launch();
508 Mutate(new EffectMovementGenerator(0), MOTION_SLOT_CONTROLLED);
509}

545void MotionMaster::MoveFall(uint32 id /*=0*/, bool addFlagForNPC)
546{
547 // Xinef: do not allow to move with UNIT_FLAG_DISABLE_MOVE
548 if (_owner->HasUnitFlag(UNIT_FLAG_DISABLE_MOVE))
549 return;
550
551 // use larger distance for vmap height search than in most other cases
552 float tz = _owner->GetMapHeight(_owner->GetPositionX(), _owner->GetPositionY(), _owner->GetPositionZ(), true, MAX_FALL_DISTANCE);
553 if (tz <= INVALID_HEIGHT)
554 {
555 LOG_DEBUG("movement.motionmaster", "MotionMaster::MoveFall: unable retrive a proper height at map {} (x: {}, y: {}, z: {}).",
556 _owner->GetMap()->GetId(), _owner->GetPositionX(), _owner->GetPositionX(), _owner->GetPositionZ() + _owner->GetPositionZ());
557 return;
558 }
559
560 // Abort too if the ground is very near
561 if (std::fabs(_owner->GetPositionZ() - tz) < 0.1f)
562 return;
563
564 if (_owner->GetTypeId() == TYPEID_PLAYER)
565 {
566 _owner->AddUnitMovementFlag(MOVEMENTFLAG_FALLING);
567 _owner->m_movementInfo.SetFallTime(0);
568 _owner->ToPlayer()->SetFallInformation(GameTime::GetGameTime().count(), _owner->GetPositionZ());
569 }
570 else if (_owner->GetTypeId() == TYPEID_UNIT && addFlagForNPC) // pussywizard
571 {
572 _owner->RemoveUnitMovementFlag(MOVEMENTFLAG_MASK_MOVING);
573 _owner->RemoveUnitMovementFlag(MOVEMENTFLAG_FLYING | MOVEMENTFLAG_CAN_FLY);
574 _owner->AddUnitMovementFlag(MOVEMENTFLAG_FALLING);
575 _owner->m_movementInfo.SetFallTime(0);
576 _owner->SendMovementFlagUpdate();
577 }
578
579 Movement::MoveSplineInit init(_owner);
580 init.MoveTo(_owner->GetPositionX(), _owner->GetPositionY(), tz + _owner->GetHoverHeight());
581 init.SetFall();
582 init.Launch();
583 Mutate(new EffectMovementGenerator(id), MOTION_SLOT_CONTROLLED);
584}

728void MotionMaster::Mutate(MovementGenerator* m, MovementSlot slot)
729{
730 while (MovementGenerator* curr = Impl[slot])
731 {
732 bool delayed = (_top == slot && (_cleanFlag & MMCF_UPDATE));
733
734 // pussywizard: clear slot AND decrease top immediately to avoid crashes when referencing null top in DirectDelete
735 Impl[slot] = nullptr;
736 while (!empty() && !top())
737 --_top;
738
739 if (delayed)
740 DelayedDelete(curr);
741 else
742 DirectDelete(curr);
743 }
744
745 if (_top < slot)
746 _top = slot;
747
748 Impl[slot] = m;
749 if (_top > slot)
750 _needInit[slot] = true;
751 else
752 {
753 _needInit[slot] = false;
754 m->Initialize(_owner);
755 }
756}
