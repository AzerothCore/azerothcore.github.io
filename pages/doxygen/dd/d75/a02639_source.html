141 void SelectNearestCrystal()
142 {
143 if (summons.empty())
144 return;
145
146 CrystalGUID.Clear();
147 Unit* crystal = nullptr;
148 for (SummonList::const_iterator i = summons.begin(); i != summons.end(); )
149 if (Creature* summon = ObjectAccessor::GetCreature(*me, *i++))
150 if (!crystal || me->GetDistanceOrder(summon, crystal, false))
151 crystal = summon;
152
153 if (crystal)
154 {
155 Talk(SAY_ENERGY);
156 float x, y, z;
157 crystal->GetClosePoint(x, y, z, me->GetObjectSize(), CONTACT_DISTANCE);
158 CrystalGUID = crystal->GetGUID();
159 me->GetMotionMaster()->MovePoint(2, x, y, z);
160 }
161 }
162
163 void MovementInform(uint32 type, uint32 id) override
164 {
165 if (type == POINT_MOTION_TYPE && id == 2)
166 {
167 if (Unit* crystal = ObjectAccessor::GetUnit(*me, CrystalGUID))
168 {
169 Talk(EMOTE_CRYSTAL);
170 crystal->ReplaceAllUnitFlags(UNIT_FLAG_NONE);
171 crystal->CastSpell(me, SPELL_MANA_RAGE, true);
172 me->CastSpell(crystal, SPELL_FEL_CRYSTAL_COSMETIC, true);
173 events.SetPhase(1);
174 events.ScheduleEvent(EVENT_EMPOWER, 0, 0, 1);
175 }
176 else
177 events.ScheduleEvent(EVENT_RESTORE_COMBAT, 0);
178 }
179 }
180
181 void UpdateAI(uint32 diff) override
182 {
183 if (!UpdateVictim())
184 return;
185
186 events.Update(diff);
187 if (me->HasUnitState(UNIT_STATE_CASTING))
188 return;
189
190 switch (events.ExecuteEvent())
191 {
192 case EVENT_SPELL_DRAIN_LIFE:
193 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
194 me->CastSpell(target, DUNGEON_MODE(SPELL_DRAIN_LIFE_N, SPELL_DRAIN_LIFE_H), false);
195 events.ScheduleEvent(EVENT_SPELL_DRAIN_LIFE, 10000, 1);
196 return;
197 case EVENT_SPELL_DRAIN_MANA:
198 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, PowerUsersSelector(me, POWER_MANA, 40.0f, false)))
199 me->CastSpell(target, SPELL_DRAIN_MANA, false);
200 events.ScheduleEvent(EVENT_SPELL_DRAIN_MANA, 10000, 1);
201 return;
202 case EVENT_SPELL_FEL_EXPLOSION:
203 me->RemoveAuraFromStack(SPELL_MANA_RAGE_TRIGGER);
204 me->CastSpell(me, SPELL_FEL_EXPLOSION, false);
205 events.ScheduleEvent(EVENT_SPELL_FEL_EXPLOSION, 2000);
206 break;
207 case EVENT_DRAIN_CRYSTAL:
208 events.DelayEvents(10001);
209 events.ScheduleEvent(EVENT_EMPOWER, 10000);
210 events.ScheduleEvent(EVENT_DRAIN_CRYSTAL, 30000);
211 SelectNearestCrystal();
212 break;
213 case EVENT_EMPOWER:
214 if (me->GetPower(POWER_MANA) == me->GetMaxPower(POWER_MANA))
215 {
216 Talk(SAY_EMPOWERED);
217 if (Unit* crystal = ObjectAccessor::GetUnit(*me, CrystalGUID))
218 Unit::Kill(crystal, crystal);
219 events.DelayEvents(10000, 1);
220 events.ScheduleEvent(EVENT_RESTORE_COMBAT, 0);
221 }
222 else
223 events.ScheduleEvent(EVENT_EMPOWER, 0, 0, 1);
224 break;
225 case EVENT_RESTORE_COMBAT:
226 events.SetPhase(0);
227 me->GetMotionMaster()->MoveChase(me->GetVictim());
228 break;
229 }
230
231 DoMeleeAttackIfReady();
232 }
233 };
