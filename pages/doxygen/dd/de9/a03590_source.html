333 void UpdateAI(uint32 diff) override
334 {
335 // THIS IS USED ONLY FOR FIRST ENGAGE!
336 if (_startTimer)
337 {
338 _startTimer += diff;
339 if (_startTimer >= 4000)
340 {
341 // Colossus dead, players in range
342 if (me->FindNearestCreature(NPC_ULDUAR_COLOSSUS, 250.0f, true) || !SelectTargetFromPlayerList(250.0f))
343 _startTimer = 1;
344 else
345 {
346 _startTimer = 0;
347 _speakTimer = 1;
348 }
349 }
350 return;
351 }
352
353 if (_speakTimer)
354 {
355 _speakTimer += diff;
356 if (_speakTimer <= 10000)
357 {
358 _speakTimer = 10000;
359 RadioSay(BRANN_RADIO_SAY_FL_START_0);
360 }
361 else if (_speakTimer > 16000 && _speakTimer < 20000)
362 {
363 _speakTimer = 20000;
364 RadioSay(BRANN_RADIO_SAY_FL_START_1);
365 }
366 else if (_speakTimer > 24000 && _speakTimer < 40000)
367 {
368 _speakTimer = 40000;
369 RadioSay(BRANN_RADIO_SAY_FL_START_2);
370 }
371 else if (_speakTimer > 41000 && _speakTimer < 60000)
372 {
373 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
374 TurnGates(true, false);
375 me->MonsterMoveWithSpeed(homePos.GetPositionX(), homePos.GetPositionY(), homePos.GetPositionZ(), 100.0f);
376 me->UpdatePosition(homePos);
377 _speakTimer = 60000;
378 }
379 else if (_speakTimer > 63500)
380 {
381 me->SetInCombatWithZone();
382 if (!me->GetVictim())
383 {
384 me->CastSpell(me, SPELL_PURSUED, false);
385 events.RescheduleEvent(EVENT_PURSUE, 31000);
386 }
387 _speakTimer = 0;
388 }
389 return;
390 }
391
392 if (!UpdateVictim())
393 return;
394
395 events.Update(diff);
396 if (me->HasUnitState(UNIT_STATE_CASTING))
397 return;
398
399 switch (events.ExecuteEvent())
400 {
401 case EVENT_POSITION_CHECK:
402 if (me->GetPositionX() > 450 || me->GetPositionX() < 120)
403 {
404 EnterEvadeMode();
405 return;
406 }
407 events.RepeatEvent(5000);
408 break;
409 case EVENT_PURSUE:
410 Talk(FLAME_LEVIATHAN_SAY_PURSUE);
411 me->CastSpell(me, SPELL_PURSUED, false);
412 events.RescheduleEvent(EVENT_PURSUE, 31000);
413 return;
414 case EVENT_SPEED:
415 me->CastSpell(me, SPELL_GATHERING_SPEED, false);
416 events.RepeatEvent(15000);
417 return;
418 case EVENT_MISSILE:
419 me->CastSpell(me, SPELL_MISSILE_BARRAGE, true);
420 events.RepeatEvent(4000);
421 return;
422 case EVENT_VENT:
423 me->CastSpell(me, SPELL_FLAME_VENTS, false);
424 events.RepeatEvent(20000);
425 return;
426 case EVENT_SUMMON:
427 if(summons.size() < 20)
428 if (Creature* lift = DoSummonFlyer(NPC_MECHANOLIFT, me, 30.0f, 50.0f, 0))
429 lift->GetMotionMaster()->MoveRandom(100);
430
431 events.RepeatEvent(4000);
432 return;
433 case EVENT_SOUND_BEGINNING:
434 if (_towersCount)
435 Talk(FLAME_LEVIATHAN_SAY_HARDMODE);
436 else
437 Talk(FLAME_LEVIATHAN_SAY_TOWER_NONE);
438 return;
439 case EVENT_REINSTALL:
440 for (uint8 i = RAID_MODE(0, 2); i < 4; ++i)
441 if (Unit* seat = vehicle->GetPassenger(i))
442 if (seat->GetTypeId() == TYPEID_UNIT)
443 seat->ToCreature()->AI()->EnterEvadeMode();
444 Talk(FLAME_LEVIATHAN_EMOTE_REACTIVATE);
445 return;
446 case EVENT_THORIMS_HAMMER:
447 SummonTowerHelpers(TOWER_OF_STORMS);
448 events.RepeatEvent(60000 + rand() % 60000);
449 Talk(FLAME_LEVIATHAN_EMOTE_STORM);
450 Talk(FLAME_LEVIATHAN_SAY_TOWER_STORM);
451 return;
452 case EVENT_FREYA:
453 SummonTowerHelpers(TOWER_OF_LIFE);
454 Talk(FLAME_LEVIATHAN_EMOTE_NATURE);
455 Talk(FLAME_LEVIATHAN_SAY_TOWER_NATURE);
456 return;
457 case EVENT_MIMIRONS_INFERNO:
458 SummonTowerHelpers(TOWER_OF_FLAMES);
459 Talk(FLAME_LEVIATHAN_EMOTE_FLAME);
460 Talk(FLAME_LEVIATHAN_SAY_TOWER_FLAME);
461 return;
462 case EVENT_HODIRS_FURY:
463 SummonTowerHelpers(TOWER_OF_FROST);
464 Talk(FLAME_LEVIATHAN_EMOTE_FROST);
465 Talk(FLAME_LEVIATHAN_SAY_TOWER_FROST);
466 return;
467 }
468
469 if(me->isAttackReady() && !me->HasUnitState(UNIT_STATE_STUNNED))
470 {
471 if(me->IsWithinCombatRange(me->GetVictim(), 15.0f))
472 {
473 me->CastSpell(me->GetVictim(), SPELL_BATTERING_RAM, false);
474 me->resetAttackTimer();
475 }
476 }

508void boss_flame_leviathan::boss_flame_leviathanAI::ActivateTowers()
509{
510 _towersCount = 0;
511 me->ResetLootMode();
512 for (uint32 i = EVENT_TOWER_OF_LIFE_DESTROYED; i <= EVENT_TOWER_OF_FLAMES_DESTROYED; ++i)
513 {
514 if (m_pInstance->GetData(i))
515 {
516 ++_towersCount;
517
518 me->AddLootMode(1 << _towersCount);
519 switch (i)
520 {
521 case EVENT_TOWER_OF_LIFE_DESTROYED:
522 me->AddAura(SPELL_TOWER_OF_LIFE, me);
523 events.RescheduleEvent(EVENT_FREYA, 30000);
524 break;
525 case EVENT_TOWER_OF_STORM_DESTROYED:
526 me->AddAura(SPELL_TOWER_OF_STORMS, me);
527 events.RescheduleEvent(EVENT_THORIMS_HAMMER, 60000);
528 break;
529 case EVENT_TOWER_OF_FROST_DESTROYED:
530 me->AddAura(SPELL_TOWER_OF_FROST, me);
531 events.RescheduleEvent(EVENT_HODIRS_FURY, 20000);
532 break;
533 case EVENT_TOWER_OF_FLAMES_DESTROYED:
534 me->AddAura(SPELL_TOWER_OF_FLAMES, me);
535 events.RescheduleEvent(EVENT_MIMIRONS_INFERNO, 42000);
