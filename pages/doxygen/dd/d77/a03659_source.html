118 InstanceScript* pInstance;
119 EventMap events;
120 SummonList summons;
121 SummonList summons2;
122 uint8 BoatNum;
123 uint8 BoatOrder[4];
124
125 void Reset() override
126 {
127 for (uint8 i = 0; i < 4; ++i)
128 {
129 bool good;
130 do
131 {
132 good = true;
133 BoatOrder[i] = urand(0, 3);
134
135 for (uint8 j = 0; j < i; ++j)
136 if (BoatOrder[i] == BoatOrder[j])
137 {
138 good = false;
139 break;
140 }
141 } while (!good);
142 }
143
144 events.Reset();
145 summons.DespawnAll();
146 summons2.DespawnAll();
147 BoatNum = 0;
148
149 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_DISABLE_MOVE);
150
151 if(pInstance)
152 {
153 pInstance->SetData(DATA_KING_YMIRON, NOT_STARTED);
154 pInstance->SetData(DATA_YMIRON_ACHIEVEMENT, true);
155
156 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
157 if (pInstance->GetData(DATA_SKADI_THE_RUTHLESS) == DONE)
158 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
159 }
160 }
161
162 void EnterEvadeMode(EvadeReason why) override
163 {
164 me->RemoveUnitFlag(UNIT_FLAG_DISABLE_MOVE);
165 ScriptedAI::EnterEvadeMode(why);
166 }
167
168 void EnterCombat(Unit* /*pWho*/) override
169 {
170 Talk(SAY_AGGRO);
171 if(pInstance)
172 {
173 pInstance->SetData(DATA_KING_YMIRON, IN_PROGRESS);
174 if (pInstance->GetData(DATA_SKADI_THE_RUTHLESS) == DONE)
175 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
176 }
177
178 events.RescheduleEvent(EVENT_YMIRON_BANE, 18000);
179 events.RescheduleEvent(EVENT_YMIRON_FETID_ROT, 8000);
180 events.RescheduleEvent(EVENT_YMIRON_DARK_SLASH, 28000);
181 events.RescheduleEvent(EVENT_YMIRON_HEALTH_CHECK, 1000);
182 }
183
184 void MovementInform(uint32 uiType, uint32 point) override
185 {
186 if(uiType != POINT_MOTION_TYPE)
187 return;
188
189 if (point == 0)
190 {
191 Talk(BoatStructure[BoatOrder[BoatNum - 1]].say);
192 if (Creature* cr = me->FindNearestCreature(BoatStructure[BoatOrder[BoatNum - 1]].trigger, 50.0f))
193 me->CastSpell(cr, SPELL_CHANNEL_YMIRON_TO_SPIRIT, true);
194
195 events.ScheduleEvent(EVENT_YMIRON_ACTIVATE_BOAT, 6000);
196 }
197 }
198
199 void SpellHitTarget(Unit*, SpellInfo const* spellInfo) override
200 {
201 if (spellInfo->Id == 59302 && pInstance) // Bane trigger
202 pInstance->SetData(DATA_YMIRON_ACHIEVEMENT, false);
203 }

205 void UpdateAI(uint32 diff) override
206 {
207 if (!UpdateVictim())
208 return;
209
210 events.Update(diff);
211 if (me->HasUnitState(UNIT_STATE_CASTING))
212 return;
213
214 switch (events.ExecuteEvent())
215 {
216 case EVENT_YMIRON_HEALTH_CHECK:
217 {
218 if (me->GetHealth() < std::max(0.0f, float(me->GetMaxHealth() * (1.0f - (IsHeroic() ? 0.2f : 0.334f)*float(BoatNum + 1)))))
219 {
220 events.DelayEvents(12000);
221 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
222 me->InterruptNonMeleeSpells(true);
223 me->CastSpell(me, SPELL_SCREAMS_OF_THE_DEAD, true);
224 me->GetMotionMaster()->Clear();
225 me->GetMotionMaster()->MovePoint(0, BoatStructure[BoatOrder[BoatNum]].MoveX, BoatStructure[BoatOrder[BoatNum]].MoveY, BoatStructure[BoatOrder[BoatNum]].MoveZ);
226 me->SetUnitFlag(UNIT_FLAG_DISABLE_MOVE);
227 summons.DespawnAll();
228
229 // Spawn flames in previous boat if any
230 if (BoatNum) // different than 0
231 if (Creature* cr = me->SummonTrigger(BoatStructure[BoatOrder[BoatNum - 1]].SpawnX, BoatStructure[BoatOrder[BoatNum - 1]].SpawnY, BoatStructure[BoatOrder[BoatNum - 1]].SpawnZ, 0, 1800000))
232 {
233 cr->AddAura(SPELL_FLAMES, cr);
234 summons2.Summon(cr);
235 }
236
237 BoatNum++;
238 }
239
240 events.RepeatEvent(1000);
241 break;
242 }
243 case EVENT_YMIRON_BANE:
244 {
245 me->CastSpell(me, IsHeroic() ? SPELL_BANE_H : SPELL_BANE_N, false);
246 events.RepeatEvent(20000 + rand() % 5000);
247 break;
248 }
249 case EVENT_YMIRON_FETID_ROT:
250 {
251 me->CastSpell(me->GetVictim(), IsHeroic() ? SPELL_FETID_ROT_H : SPELL_FETID_ROT_N, false);
252 events.RepeatEvent(10000 + rand() % 3000);
253 break;
254 }
255 case EVENT_YMIRON_DARK_SLASH:
256 {
257 int32 dmg = me->GetVictim()->GetHealth() / 2;
258 me->CastCustomSpell(me->GetVictim(), SPELL_DARK_SLASH, &dmg, 0, 0, false);
259 events.RepeatEvent(30000 + rand() % 5000);
260 break;
261 }
262 case EVENT_YMIRON_ACTIVATE_BOAT:
263 {
264 // Spawn it!
265 if (Creature* king = me->SummonCreature(BoatStructure[BoatOrder[BoatNum - 1]].npc, BoatStructure[BoatOrder[BoatNum - 1]].SpawnX, BoatStructure[BoatOrder[BoatNum - 1]].SpawnY, BoatStructure[BoatOrder[BoatNum - 1]].SpawnZ, BoatStructure[BoatOrder[BoatNum - 1]].SpawnO, TEMPSUMMON_CORPSE_DESPAWN, 0))
266 {
267 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
268 king->CastSpell(me, SPELL_CHANNEL_SPIRIT_TO_YMIRON, true);
269 summons.Summon(king);
270 king->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
271 king->SetDisableGravity(true);
272 me->RemoveUnitFlag(UNIT_FLAG_DISABLE_MOVE);
273 me->GetMotionMaster()->MoveChase(me->GetVictim());
274 switch(BoatOrder[BoatNum - 1])
275 {
276 case 0:
277 events.ScheduleEvent(EVENT_YMIRON_RANULF_ABILITY, 3000, 1);
278 break;
279 case 1:
280 events.ScheduleEvent(EVENT_YMIRON_TORGYN_ABILITY, 3000, 1);
281 break;
282 case 2:
283 events.ScheduleEvent(EVENT_YMIRON_BJORN_ABILITY, 3000, 1);
284 break;
285 case 3:
286 events.ScheduleEvent(EVENT_YMIRON_HALDOR_ABILITY, 3000, 1);
287 break;
288 }
289 }
290
291 break;
292 }
293 case EVENT_YMIRON_BJORN_ABILITY:
294 {
295 if (Creature* sf = me->SummonCreature(NPC_SPIRIT_FOUNT, 385 + rand() % 10, -330 + rand() % 10, 104.756f, 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 180000))
296 {
297 summons.Summon(sf);
298 sf->SetSpeed(MOVE_RUN, 0.4f);
299 sf->AddAura(IsHeroic() ? SPELL_SPIRIT_FOUNT_H : SPELL_SPIRIT_FOUNT_N, sf);
300 sf->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
301 sf->GetMotionMaster()->MoveFollow(me->GetVictim(), 0, rand_norm()*M_PI * 2);
302 }
303 break;
304 }
305 case EVENT_YMIRON_HALDOR_ABILITY:
306 {
307 me->CastSpell(me->GetVictim(), IsHeroic() ? SPELL_SPIRIT_STRIKE_H : SPELL_SPIRIT_STRIKE_N, false);
308 events.RepeatEvent(5000);
309 break;
310 }
311 case EVENT_YMIRON_RANULF_ABILITY:
312 {
313 me->CastSpell(me, IsHeroic() ? SPELL_SPIRIT_BURST_H : SPELL_SPIRIT_BURST_N, false);
314 events.RepeatEvent(10000);
315 break;
316 }
317 case EVENT_YMIRON_TORGYN_ABILITY:
318 {
319 for(uint8 i = 0; i < 4; ++i)
320 {
321 if (Creature* as = me->SummonCreature(NPC_AVENGING_SPIRIT, me->GetPositionX() + rand() % 10, me->GetPositionY() + rand() % 10, me->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000))
322 {
323 summons.Summon(as);
324 as->SetInCombatWithZone();
325 }
326 }
327 events.RepeatEvent(15000);
328 break;
329 }
330 }
331
332 DoMeleeAttackIfReady();
333 }
334

348 return;
349
350 Talk(SAY_SLAY);
351 }
352 };
