126 void UpdateAI(uint32 diff) override
127 {
128 if (!UpdateVictim())
129 return;
130
131 if (AvatarTimer <= diff)
132 {
133 DoCastVictim(SPELL_AVATAR);
134 AvatarTimer = urand(15 * IN_MILLISECONDS, 20 * IN_MILLISECONDS);
135 }
136 else AvatarTimer -= diff;
137
138 if (ThunderclapTimer <= diff)
139 {
140 DoCastVictim(SPELL_THUNDERCLAP);
141 ThunderclapTimer = urand(5 * IN_MILLISECONDS, 15 * IN_MILLISECONDS);
142 }
143 else ThunderclapTimer -= diff;
144
145 if (StormboltTimer <= diff)
146 {
147 DoCastVictim(SPELL_STORMBOLT);
148 StormboltTimer = urand(10 * IN_MILLISECONDS, 25 * IN_MILLISECONDS);
149 }
150 else StormboltTimer -= diff;
151
152 if (YellTimer <= diff)
153 {
154 Talk(YELL_RANDOM);
155 YellTimer = urand(20 * IN_MILLISECONDS, 30 * IN_MILLISECONDS); //20 to 30 seconds
156 }
157 else YellTimer -= diff;
158
159 // check if creature is not outside of building
160 if (ResetTimer <= diff)
161 {
162 if (me->GetDistance2d(me->GetHomePosition().GetPositionX(), me->GetHomePosition().GetPositionY()) > 50)
163 {
164 ScriptedAI::EnterEvadeMode();
165 Talk(YELL_EVADE);
166 }
167 ResetTimer = 5 * IN_MILLISECONDS;
168 }
169 else ResetTimer -= diff;
170
171 DoMeleeAttackIfReady();
172 }
173 };
