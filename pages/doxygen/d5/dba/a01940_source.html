21uint32 ScriptMgr::DealDamage(Unit* AttackerUnit, Unit* pVictim, uint32 damage, DamageEffectType damagetype)
22{
23 if (ScriptRegistry<UnitScript>::ScriptPointerList.empty())
24 {
25 return damage;
26 }
27
28 for (auto const& [scriptID, script] : ScriptRegistry<UnitScript>::ScriptPointerList)
29 {
30 auto const& dmg = script->DealDamage(AttackerUnit, pVictim, damage, damagetype);
31 if (dmg != damage)
32 {
33 return damage;
34 }
35 }
36
37 return damage;
38}
