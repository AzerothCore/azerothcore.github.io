123 void DoAction(int32 action) override
124 {
125 if (action == ACTION_START_EVENT && _phase == PHASE_ALL)
126 {
127 _phase = PHASE_INTRO;
128 _scheduler.Schedule(1ms, [this](TaskContext /*context*/)
129 {
130 Talk(SAY_INTRO_0);
131 })
132 .Schedule(4s, [this](TaskContext /*context*/)
133 {
134 Talk(SAY_INTRO_1);
135 })
136 .Schedule(8s, [this](TaskContext /*context*/)
137 {
138 Talk(SAY_INTRO_2);
139 })
140 .Schedule(12s, [this](TaskContext context)
141 {
142 me->SetImmuneToAll(false);
143 _phase = PHASE_COMBAT;
144 DoZoneInCombat();
145
146 context.Schedule(6s, [this](TaskContext /*context*/) // Call Baxter
147 {
148 Talk(SAY_CALL_BAXTER);
149 EntryCheckPredicate pred(NPC_APOTHECARY_BAXTER);
150 summons.DoAction(ACTION_START_FIGHT, pred);
151 summons.DoZoneInCombat(NPC_APOTHECARY_BAXTER);
152 })
153 .Schedule(14s, [this](TaskContext /*context*/) // Call Frye
154 {
155 Talk(SAY_CALL_FRYE);
156 EntryCheckPredicate pred(NPC_APOTHECARY_FRYE);
157 summons.DoAction(ACTION_START_FIGHT, pred);
158 })
159 .Schedule(3640ms, [this](TaskContext context) // Perfume spray
160 {
161 DoCastVictim(SPELL_PERFUME_SPRAY);
162 context.Repeat(3640ms);
163 })
164 .Schedule(15s, [this](TaskContext context) // Chain Reaction
165 {
166 DoCastVictim(SPELL_SUMMON_TABLE, true);
167 DoCastAOE(SPELL_CHAIN_REACTION);
168 context.Repeat(25s);
169 })
170 .Schedule(15s, [this](TaskContext /*context*/) // Call Crazed Apothecary (text)
171 {
172 Talk(SAY_SUMMON_ADDS);
173 })
174 .Schedule(15s, [this](TaskContext context) // Call Crazed Apothecary
175 {
176 instance->SetData(DATA_SPAWN_VALENTINE_ADDS, 0);
177 context.Repeat(4s, 6s);
178 });
179
180 std::list<Creature*> trashs;
181 me->GetCreatureListWithEntryInGrid(trashs, NPC_CROWN_APOTHECARY, 100.f);
182 for (Creature* crea : trashs)
183 {
184 crea->DespawnOrUnsummon();
185 }
186 });
187
188 me->SetImmuneToPC(true);
189 me->SetFaction(FACTION_MONSTER);
190 summons.DoAction(ACTION_START_EVENT);
191 }
192 }

194 void DamageTaken(Unit* /*attacker*/, uint32& damage, DamageEffectType, SpellSchoolMask) override
195 {
196 if (damage >= me->GetHealth())
197 {
198 if (_deadCount < 2)
199 {
200 damage = me->GetHealth() - 1;
201 if (!_isDead)
202 {
203 _isDead = true;
204 me->RemoveAurasDueToSpell(SPELL_ALLURING_PERFUME);
205 DoCastSelf(SPELL_PERMANENT_FEIGN_DEATH, true);
206 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
207 Talk(SAY_HUMMEL_DEATH);
208 }
209 }
210 }
211 }

217 _deadCount++;

225
226 void JustDied(Unit* /*killer*/) override
227 {
228 if (!_isDead)

232
233 _scheduler.CancelAll();
234 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
235 instance->SetBossState(DATA_APOTHECARY_HUMMEL, DONE);
236
237 Map::PlayerList const& players = me->GetMap()->GetPlayers();
238 if (!players.IsEmpty())
239 {
240 if (Group* group = players.begin()->GetSource()->GetGroup())
241 {
242 if (group->isLFGGroup())
243 {
244 sLFGMgr->FinishDungeon(group->GetGUID(), 288, me->GetMap());
245 }
246 }
247 }
248 }
