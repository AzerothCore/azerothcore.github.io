113 void EnterCombat(Unit* /*who*/) override

116 events.ScheduleEvent(EVENT_BLOOD_SIPHON, 90000);
117 events.ScheduleEvent(EVENT_CORRUPTED_BLOOD, 25000);
118 events.ScheduleEvent(EVENT_CAUSE_INSANITY, 17000);
119 events.ScheduleEvent(EVENT_ENRAGE, 600000);
120 if (instance->GetBossState(DATA_JEKLIK) != DONE)
121 events.ScheduleEvent(EVENT_ASPECT_OF_JEKLIK, 21000);
122 if (instance->GetBossState(DATA_VENOXIS) != DONE)

126 if (instance->GetBossState(DATA_THEKAL) != DONE)
127 events.ScheduleEvent(EVENT_ASPECT_OF_THEKAL, 10000);
128 if (instance->GetBossState(DATA_ARLOKK) != DONE)
129 events.ScheduleEvent(EVENT_ASPECT_OF_ARLOKK, 18000);
130 Talk(SAY_AGGRO);

139
140 void UpdateAI(uint32 diff) override
141 {
142 if (!UpdateVictim() || !CheckInRoom())
143 return;
144
145 events.Update(diff);
146
147 if (me->HasUnitState(UNIT_STATE_CASTING))
148 return;
149
150 while (uint32 eventId = events.ExecuteEvent())
151 {
152 switch (eventId)
153 {
154 case EVENT_BLOOD_SIPHON:
155 DoCastAOE(SPELL_BLOOD_SIPHON, true);
156 events.ScheduleEvent(EVENT_BLOOD_SIPHON, 90000);
157 break;
158 case EVENT_CORRUPTED_BLOOD:
159 DoCastVictim(SPELL_CORRUPTED_BLOOD, true);
160 events.ScheduleEvent(EVENT_CORRUPTED_BLOOD, urand(30000, 45000));
161 break;
162 case EVENT_CAUSE_INSANITY:
163 if (me->GetThreatMgr().GetThreatListSize() > 1)
164 {
165 if (Unit* victim = SelectTarget(SelectTargetMethod::MaxThreat, 0, 30.f, true))
166 {
167 DoCast(victim, SPELL_CAUSE_INSANITY);
168 }
169 }
170 events.ScheduleEvent(EVENT_CAUSE_INSANITY, urand(35000, 45000));
171 break;
172 case EVENT_ENRAGE:
173 if (!me->HasAura(SPELL_ENRAGE))
174 DoCastSelf(SPELL_ENRAGE);
175 events.ScheduleEvent(EVENT_ENRAGE, 90000);
176 break;
177 case EVENT_ASPECT_OF_JEKLIK:
178 DoCastVictim(SPELL_ASPECT_OF_JEKLIK, true);
179 events.ScheduleEvent(EVENT_ASPECT_OF_JEKLIK, 24000);
180 break;
181 case EVENT_ASPECT_OF_VENOXIS:
182 DoCastVictim(SPELL_ASPECT_OF_VENOXIS, true);
183 events.ScheduleEvent(EVENT_ASPECT_OF_VENOXIS, urand(16000, 18000));
184 break;
185 case EVENT_ASPECT_OF_MARLI:
186 if (Unit* victim = SelectTarget(SelectTargetMethod::MaxThreat, 0, 5.f, true))
187 {
188 DoCast(victim, SPELL_ASPECT_OF_MARLI, true);
189 me->GetThreatMgr().ModifyThreatByPercent(victim, -100.f);
190 }
191 events.ScheduleEvent(EVENT_ASPECT_OF_MARLI, 45000);
192 break;
193 case EVENT_ASPECT_OF_THEKAL:
194 DoCastVictim(SPELL_ASPECT_OF_THEKAL, true);
195 events.ScheduleEvent(EVENT_ASPECT_OF_THEKAL, 15000);
196 break;
197 case EVENT_ASPECT_OF_ARLOKK:
198 if (Unit* victim = SelectTarget(SelectTargetMethod::MaxThreat, 0, 5.f, true))
199 {
200 DoCast(victim, SPELL_ASPECT_OF_ARLOKK, true);
201 me->GetThreatMgr().ModifyThreatByPercent(victim, -100.f);
202 }
203 events.ScheduleEvent(EVENT_ASPECT_OF_ARLOKK, urand(10000, 15000));
204 break;
205 default:
206 break;
207 }
208 }
209
210 DoMeleeAttackIfReady();
211 }
212 };
