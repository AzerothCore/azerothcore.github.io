121 boss_vezaxAI(Creature* pCreature) : ScriptedAI(pCreature), summons(me)
122 {
123 pInstance = pCreature->GetInstanceScript();

126 EventMap events;
127 SummonList summons;
128 uint8 vaporsCount;
129 bool hardmodeAvailable;

132
133 InstanceScript* pInstance;

135 void Reset() override
136 {
137 vaporsCount = 0;
138 hardmodeAvailable = true;
139 berserk = false;
140 bAchievShadowdodger = true;
141 events.Reset();
142 summons.DespawnAll();
143 me->SetLootMode(1);
144
145 if (pInstance)
146 pInstance->SetData(TYPE_VEZAX, NOT_STARTED);

154 void EnterCombat(Unit* /*pWho*/) override
155 {
156 me->setActive(true);
157 me->SetInCombatWithZone();
158
159 events.Reset();
160 events.RescheduleEvent(EVENT_SPELL_VEZAX_SHADOW_CRASH, 13000);
161 events.RescheduleEvent(EVENT_SPELL_SEARING_FLAMES, 10000, 1);
162 events.RescheduleEvent(EVENT_SPELL_SURGE_OF_DARKNESS, 63000);
163 events.RescheduleEvent(EVENT_SPELL_MARK_OF_THE_FACELESS, 20000);
164 events.RescheduleEvent(EVENT_SPELL_SUMMON_SARONITE_VAPORS, 30000);
165 events.RescheduleEvent(EVENT_BERSERK, 600000);
166
167 Talk(SAY_AGGRO);
168
169 if (pInstance)
170 pInstance->SetData(TYPE_VEZAX, IN_PROGRESS);
171
172 me->CastSpell(me, SPELL_AURA_OF_DESPAIR_1, true);
173 }
174
175 void DoAction(int32 param) override
176 {
177 switch( param )
178 {
179 case 1:
180 hardmodeAvailable = false;
181 break;
182 case 2:
183 me->RemoveAura(SPELL_SARONITE_BARRIER);
184 me->SetLootMode(3);
185 break;
186 }
187 }
188
189 uint32 GetData(uint32 id) const override
190 {
191 switch (id)
192 {
193 case 1:
194 return (me->GetLootMode() == 3 ? 1 : 0);
195 case 2:
196 return (bAchievShadowdodger ? 1 : 0);
197 }
198 return 0;
199 }
200
201 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
202 {
203 if (target && spell && target->GetTypeId() == TYPEID_PLAYER && spell->Id == SPELL_VEZAX_SHADOW_CRASH_DMG)
204 bAchievShadowdodger = false;
205 }
206
207 void UpdateAI(uint32 diff) override
208 {
209 if( !UpdateVictim() )
210 return;
211
212 if( !berserk && (me->GetPositionX() < 1720.0f || me->GetPositionX() > 1940.0f || me->GetPositionY() < 20.0f || me->GetPositionY() > 210.0f) )
213 events.RescheduleEvent(EVENT_BERSERK, 1);
214
215 events.Update(diff);
216
217 if( me->HasUnitState(UNIT_STATE_CASTING) )
218 return;
219
220 switch( events.ExecuteEvent() )
221 {
222 case 0:
223 break;
224 case EVENT_BERSERK:
225 berserk = true;
226 me->CastSpell(me, SPELL_VEZAX_BERSERK, true);
227 Talk(SAY_BERSERK);
228 break;
229 case EVENT_SPELL_VEZAX_SHADOW_CRASH:
230 {
231 events.RepeatEvent(10000);
232
233 std::vector<Player*> players;
234 Map::PlayerList const& pl = me->GetMap()->GetPlayers();
235 for( Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr )
236 {
237 Player* temp = itr->GetSource();
238 if( temp->IsAlive() && temp->GetDistance(me) > 15.0f )
239 players.push_back(temp);
240 }
241 if (!players.empty())
242 {
243 me->setAttackTimer(BASE_ATTACK, 2000);
244 Player* target = players.at(urand(0, players.size() - 1));
245 me->SetGuidValue(UNIT_FIELD_TARGET, target->GetGUID());
246 me->CastSpell(target, SPELL_VEZAX_SHADOW_CRASH, false);
247 events.ScheduleEvent(EVENT_RESTORE_TARGET, 750);
248 }
249 }
250 break;
251 case EVENT_RESTORE_TARGET:
252 if (me->GetVictim())
253 me->SetGuidValue(UNIT_FIELD_TARGET, me->GetVictim()->GetGUID());
254 break;
255 case EVENT_SPELL_SEARING_FLAMES:
256 if(!me->HasAura(SPELL_SARONITE_BARRIER))
257 me->CastSpell(me->GetVictim(), SPELL_SEARING_FLAMES, false);
258 events.RepeatEvent( me->GetMap()->Is25ManRaid() ? 8000 : 15000 );
259 break;
260 case EVENT_SPELL_SURGE_OF_DARKNESS:
261 Talk(SAY_SURGE_OF_DARKNESS);
262 Talk(SAY_EMOTE_SURGE_OF_DARKNESS);
263 me->CastSpell(me, SPELL_SURGE_OF_DARKNESS, false);
264 events.RepeatEvent(63000);
265 events.DelayEvents(10000, 1);
266 break;
267 case EVENT_SPELL_MARK_OF_THE_FACELESS:
268 {
269 std::vector<Player*> outside;
270 std::vector<Player*> inside;
271 Map::PlayerList const& pl = me->GetMap()->GetPlayers();
272 for( Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr )
273 if( Player* tmp = itr->GetSource() )
274 if( tmp->IsAlive() )
275 {
276 if( tmp->GetDistance(me) > 15.0f )
277 outside.push_back(tmp);
278 else
279 inside.push_back(tmp);
280 }
281
282 Player* t = nullptr;
283 if( outside.size() >= uint8(me->GetMap()->Is25ManRaid() ? 9 : 4) )
284 t = outside.at(urand(0, outside.size() - 1));
285 else if( !inside.empty() )
286 t = inside.at(urand(0, inside.size() - 1));
287
288 if (t)
289 me->CastSpell(t, SPELL_MARK_OF_THE_FACELESS_AURA, false);
290
291 events.RepeatEvent(40000);
292 }
293 break;
294 case EVENT_SPELL_SUMMON_SARONITE_VAPORS:
295 {
296 vaporsCount++;
297 me->CastSpell(me, SPELL_SUMMON_SARONITE_VAPORS, false);
298
299 if( vaporsCount < 6 || !hardmodeAvailable )
300 events.RepeatEvent(30000);
301 else
302 {
303 for (ObjectGuid const& guid : summons)
304 if (Creature* sv = ObjectAccessor::GetCreature(*me, guid))
305 {
306 sv->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
307 sv->GetMotionMaster()->MoveIdle();
308 sv->GetMotionMaster()->MoveCharge(1852.78f, 81.38f, 342.461f, 28.0f);
309 }
310
311 events.DelayEvents(12000, 0);
312 events.DelayEvents(12000, 1);
313 events.ScheduleEvent(EVENT_SARONITE_VAPORS_SWIRL, 6000);
314 }
315 }
316 break;
317 case EVENT_SARONITE_VAPORS_SWIRL:
318 if (summons.size())
319 {
320 Talk(SAY_EMOTE_ANIMUS);
321 if( Creature* sv = ObjectAccessor::GetCreature(*me, *(summons.begin())) )
322 sv->CastSpell(sv, SPELL_SARONITE_ANIMUS_FORMATION_VISUAL, true);
323
324 events.ScheduleEvent(EVENT_SPELL_SUMMON_SARONITE_ANIMUS, 2000);
325 break;
326 }
327 break;
328 case EVENT_SPELL_SUMMON_SARONITE_ANIMUS:
329 if (summons.size())
330 {
331 Talk(SAY_HARDMODE);
332 Talk(SAY_EMOTE_BARRIER);
333 me->CastSpell(me, SPELL_SARONITE_BARRIER, true);
334 if( Creature* sv = ObjectAccessor::GetCreature(*me, *(summons.begin())) )
335 sv->CastSpell(sv, SPELL_SUMMON_SARONITE_ANIMUS, true);
336
337 events.ScheduleEvent(EVENT_DESPAWN_SARONITE_VAPORS, 2500);
338 break;
339 }
340 break;
341 case EVENT_DESPAWN_SARONITE_VAPORS:
342 summons.DespawnEntry(NPC_SARONITE_VAPORS);
343 break;
344 }
345
346 DoMeleeAttackIfReady();
347 }
348
349 void JustDied(Unit* /*killer*/) override
350 {
351 summons.DespawnAll();
352 if (pInstance)
353 pInstance->SetData(TYPE_VEZAX, DONE);
354
355 Talk(SAY_DEATH);
356
357 if( GameObject* door = me->FindNearestGameObject(GO_VEZAX_DOOR, 500.0f) )
358 if( door->GetGoState() != GO_STATE_ACTIVE )
359 {
360 door->SetLootState(GO_READY);
361 door->UseDoorOrButton(0, false);
362 }
363 }
364
365 void KilledUnit(Unit* who) override
366 {
367 if( who->GetTypeId() == TYPEID_PLAYER )
368 Talk(SAY_SLAY);
369 }
370
371 void MoveInLineOfSight(Unit* /*who*/) override {}
372
373 void JustSummoned(Creature* summon) override
374 {
375 summons.Summon(summon);
376 }
377
378 void SummonedCreatureDespawn(Creature* s) override
379 {
380 summons.Despawn(s);
381 }
382 };
