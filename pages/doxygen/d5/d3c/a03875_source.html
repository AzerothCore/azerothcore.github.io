148 struct boss_illidari_councilAI : public BossAI
149 {
150 boss_illidari_councilAI(Creature* creature) : BossAI(creature, DATA_ILLIDARI_COUNCIL)
151 {
152 }
153
154 ObjectGuid councilGUIDs[4];
155
156 void Reset() override
157 {
158 BossAI::Reset();
159 Creature* member = nullptr;
160 for (uint8 i = 0; i < 4; ++i)
161 if ((member = ObjectAccessor::GetCreature(*me, councilGUIDs[i])))
162 member->AI()->EnterEvadeMode();
163 }
164
165 void AttackStart(Unit*) override { }
166 void MoveInLineOfSight(Unit*) override { }
167
168 void DoAction(int32 param) override
169 {
170 if (!me->isActiveObject() && param == ACTION_START_ENCOUNTER)
171 {
172 me->setActive(true);
173 councilGUIDs[0] = instance->GetGuidData(NPC_GATHIOS_THE_SHATTERER);
174 councilGUIDs[1] = instance->GetGuidData(NPC_HIGH_NETHERMANCER_ZEREVOR);
175 councilGUIDs[2] = instance->GetGuidData(NPC_LADY_MALANDE);
176 councilGUIDs[3] = instance->GetGuidData(NPC_VERAS_DARKSHADOW);
177
178 bool spoken = false;
179 for (uint8 i = 0; i < 4; ++i)
180 {
181 if (Creature* member = ObjectAccessor::GetCreature(*me, councilGUIDs[i]))
182 {
183 if (!spoken && (roll_chance_i(33) || i == 3))
184 {
185 spoken = true;
186 member->AI()->Talk(SAY_COUNCIL_AGGRO);
187 }
188 member->SetOwnerGUID(me->GetGUID());
189 member->SetInCombatWithZone();
190 }
191 }
192 }
193 else if (param == ACTION_ENRAGE)
194 {
195 Creature* member = nullptr;
196 for (uint8 i = 0; i < 4; ++i)
197 if ((member = ObjectAccessor::GetCreature(*me, councilGUIDs[i])))
198 member->AI()->DoAction(ACTION_ENRAGE);
199 }
200 else if (param == ACTION_END_ENCOUNTER)
201 {
202 me->setActive(false);
203 Creature* member = nullptr;
204 for (uint8 i = 0; i < 4; ++i)
205 if ((member = ObjectAccessor::GetCreature(*me, councilGUIDs[i])))
206 if (member->IsAlive())
207 Unit::Kill(me, member);
208 Unit::Kill(me, me);
209 }
210 }
211
212 void UpdateAI(uint32 /*diff*/) override
213 {
214 if (!me->isActiveObject())
215 return;
216
217 if (!SelectTargetFromPlayerList(115.0f))
218 {
219 EnterEvadeMode();
220 return;
221 }
222
223 me->CastSpell(me, SPELL_EMPYREAL_BALANCE, true);
224 }
225 };

338 break;

340 if (roll_chance_i(50))
341 Talk(SAY_COUNCIL_SPECIAL);
342 me->CastSpell(me, SPELL_CONSECRATION, false);
343 events.ScheduleEvent(EVENT_SPELL_AURA, 30000);
344 break;
345 case EVENT_SPELL_HAMMER_OF_JUSTICE:
346 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, HammerOfJusticeSelector(me)))

348 me->CastSpell(target, SPELL_HAMMER_OF_JUSTICE, false);

617 _targetCount++;
618 _sharedHealth += target->GetHealth();
619 _sharedHealthMax += target->GetMaxHealth();

624 {
625 if (_targetCount != 4)

634 if (Creature* target = ObjectAccessor::GetCreature(*GetCaster(), ihit->targetGUID))
635 {
636 target->LowerPlayerDamageReq(target->GetMaxHealth());
637 target->SetHealth(CalculatePct(target->GetMaxHealth(), pct));
638 }
639 }
