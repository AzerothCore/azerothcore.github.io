26void BIH::buildHierarchy(std::vector<uint32>& tempTree, buildData& dat, BuildStats& stats)
27{
28 // create space for the first node
29 tempTree.push_back(uint32(3 << 30)); // dummy leaf
30 tempTree.insert(tempTree.end(), 2, 0);
31 //tempTree.add(0);
32
33 // seed bbox
34 AABound gridBox = { bounds.low(), bounds.high() };
35 AABound nodeBox = gridBox;
36 // seed subdivide function
37 subdivide(0, dat.numPrims - 1, tempTree, dat, gridBox, nodeBox, 0, 1, stats);

40void BIH::subdivide(int left, int right, std::vector<uint32>& tempTree, buildData& dat, AABound& gridBox, AABound& nodeBox, int nodeIndex, int depth, BuildStats& stats)
41{
42 if ((right - left + 1) <= dat.maxPrims || depth >= MAX_STACK_SIZE)
43 {
44 // write leaf node
45 stats.updateLeaf(depth, right - left + 1);
46 createNode(tempTree, nodeIndex, left, right);
47 return;
48 }
49 // calculate extents
50 int axis = -1, prevAxis, rightOrig;
51 float clipL = G3D::fnan(), clipR = G3D::fnan(), prevClip = G3D::fnan();
52 float split = G3D::fnan(), prevSplit;
53 bool wasLeft = true;
54 while (true)
55 {
56 prevAxis = axis;
57 prevSplit = split;
58 // perform quick consistency checks
59 G3D::Vector3 d( gridBox.hi - gridBox.lo );
60 if (d.x < 0 || d.y < 0 || d.z < 0)
61 {
62 throw std::logic_error("negative node extents");
63 }
64 for (int i = 0; i < 3; i++)
65 {
66 if (nodeBox.hi[i] < gridBox.lo[i] || nodeBox.lo[i] > gridBox.hi[i])
67 {
68 //UI.printError(Module.ACCEL, "Reached tree area in error - discarding node with: %d objects", right - left + 1);
69 throw std::logic_error("invalid node overlap");
70 }
71 }
72 // find longest axis
73 axis = d.primaryAxis();
74 split = 0.5f * (gridBox.lo[axis] + gridBox.hi[axis]);
75 // partition L/R subsets
76 clipL = -G3D::inf();
77 clipR = G3D::inf();
78 rightOrig = right; // save this for later
79 float nodeL = G3D::inf();
80 float nodeR = -G3D::inf();
81 for (int i = left; i <= right;)
82 {
83 int obj = dat.indices[i];
84 float minb = dat.primBound[obj].low()[axis];
85 float maxb = dat.primBound[obj].high()[axis];
86 float center = (minb + maxb) * 0.5f;
87 if (center <= split)
88 {
89 // stay left
90 i++;
91 if (clipL < maxb)
92 {
93 clipL = maxb;
94 }
95 }
96 else
97 {
98 // move to the right most
99 int t = dat.indices[i];
100 dat.indices[i] = dat.indices[right];
101 dat.indices[right] = t;
102 right--;
103 if (clipR > minb)
104 {
105 clipR = minb;
106 }
107 }
108 nodeL = std::min(nodeL, minb);
109 nodeR = std::max(nodeR, maxb);
110 }
111 // check for empty space
112 if (nodeL > nodeBox.lo[axis] && nodeR < nodeBox.hi[axis])
113 {
114 float nodeBoxW = nodeBox.hi[axis] - nodeBox.lo[axis];
115 float nodeNewW = nodeR - nodeL;
116 // node box is too big compare to space occupied by primitives?
117 if (1.3f * nodeNewW < nodeBoxW)
118 {
119 stats.updateBVH2();
120 int nextIndex = tempTree.size();
121 // allocate child
122 tempTree.push_back(0);
123 tempTree.push_back(0);
124 tempTree.push_back(0);
125 // write bvh2 clip node
126 stats.updateInner();
127 tempTree[nodeIndex + 0] = (axis << 30) | (1 << 29) | nextIndex;
128 tempTree[nodeIndex + 1] = floatToRawIntBits(nodeL);
129 tempTree[nodeIndex + 2] = floatToRawIntBits(nodeR);
130 // update nodebox and recurse
131 nodeBox.lo[axis] = nodeL;
132 nodeBox.hi[axis] = nodeR;
133 subdivide(left, rightOrig, tempTree, dat, gridBox, nodeBox, nextIndex, depth + 1, stats);
134 return;
135 }
136 }
137 // ensure we are making progress in the subdivision
138 if (right == rightOrig)
139 {
140 // all left
141 if (prevAxis == axis && G3D::fuzzyEq(prevSplit, split))
142 {
143 // we are stuck here - create a leaf
144 stats.updateLeaf(depth, right - left + 1);
145 createNode(tempTree, nodeIndex, left, right);
146 return;
147 }
148 if (clipL <= split)
149 {
150 // keep looping on left half
151 gridBox.hi[axis] = split;
152 prevClip = clipL;
153 wasLeft = true;
154 continue;
155 }
156 gridBox.hi[axis] = split;
157 prevClip = G3D::fnan();
158 }
159 else if (left > right)
160 {
161 // all right
162 right = rightOrig;
163 if (prevAxis == axis && G3D::fuzzyEq(prevSplit, split))
164 {
165 // we are stuck here - create a leaf
166 stats.updateLeaf(depth, right - left + 1);
167 createNode(tempTree, nodeIndex, left, right);
168 return;
169 }
170 if (clipR >= split)
171 {
172 // keep looping on right half
173 gridBox.lo[axis] = split;
174 prevClip = clipR;
175 wasLeft = false;
176 continue;
177 }
178 gridBox.lo[axis] = split;
179 prevClip = G3D::fnan();
180 }
181 else
182 {
183 // we are actually splitting stuff
184 if (prevAxis != -1 && !isnan(prevClip))
185 {
186 // second time through - lets create the previous split
187 // since it produced empty space
188 int nextIndex = tempTree.size();
189 // allocate child node
190 tempTree.push_back(0);
191 tempTree.push_back(0);
192 tempTree.push_back(0);
193 if (wasLeft)
194 {
195 // create a node with a left child
196 // write leaf node
197 stats.updateInner();
198 tempTree[nodeIndex + 0] = (prevAxis << 30) | nextIndex;
199 tempTree[nodeIndex + 1] = floatToRawIntBits(prevClip);
200 tempTree[nodeIndex + 2] = floatToRawIntBits(G3D::inf());
201 }
202 else
203 {
204 // create a node with a right child
205 // write leaf node
206 stats.updateInner();
207 tempTree[nodeIndex + 0] = (prevAxis << 30) | (nextIndex - 3);
208 tempTree[nodeIndex + 1] = floatToRawIntBits(-G3D::inf());
209 tempTree[nodeIndex + 2] = floatToRawIntBits(prevClip);
210 }
211 // count stats for the unused leaf
212 depth++;
213 stats.updateLeaf(depth, 0);
214 // now we keep going as we are, with a new nodeIndex:
215 nodeIndex = nextIndex;
216 }
217 break;
218 }
219 }
220 // compute index of child nodes
221 int nextIndex = tempTree.size();
222 // allocate left node
223 int nl = right - left + 1;
224 int nr = rightOrig - (right + 1) + 1;
225 if (nl > 0)
226 {
227 tempTree.push_back(0);
228 tempTree.push_back(0);
229 tempTree.push_back(0);
230 }
231 else
232 {
233 nextIndex -= 3;
234 }
235 // allocate right node
236 if (nr > 0)
237 {
238 tempTree.push_back(0);
239 tempTree.push_back(0);
240 tempTree.push_back(0);
241 }
242 // write leaf node
243 stats.updateInner();
244 tempTree[nodeIndex + 0] = (axis << 30) | nextIndex;
245 tempTree[nodeIndex + 1] = floatToRawIntBits(clipL);
246 tempTree[nodeIndex + 2] = floatToRawIntBits(clipR);
247 // prepare L/R child boxes
248 AABound gridBoxL(gridBox), gridBoxR(gridBox);
249 AABound nodeBoxL(nodeBox), nodeBoxR(nodeBox);
250 gridBoxL.hi[axis] = gridBoxR.lo[axis] = split;
251 nodeBoxL.hi[axis] = clipL;
252 nodeBoxR.lo[axis] = clipR;
253 // recurse
254 if (nl > 0)
255 {
256 subdivide(left, right, tempTree, dat, gridBoxL, nodeBoxL, nextIndex, depth + 1, stats);
257 }
258 else
259 {
260 stats.updateLeaf(depth + 1, 0);
261 }
262 if (nr > 0)
263 {
264 subdivide(right + 1, rightOrig, tempTree, dat, gridBoxR, nodeBoxR, nextIndex + 3, depth + 1, stats);
265 }
266 else
267 {
268 stats.updateLeaf(depth + 1, 0);
269 }
270}

272bool BIH::writeToFile(FILE* wf) const
273{
274 uint32 treeSize = tree.size();
275 uint32 check = 0, count;
276 check += fwrite(&bounds.low(), sizeof(float), 3, wf);
277 check += fwrite(&bounds.high(), sizeof(float), 3, wf);
278 check += fwrite(&treeSize, sizeof(uint32), 1, wf);
279 check += fwrite(&tree[0], sizeof(uint32), treeSize, wf);
280 count = objects.size();
281 check += fwrite(&count, sizeof(uint32), 1, wf);
282 check += fwrite(&objects[0], sizeof(uint32), count, wf);
283 return check == (3 + 3 + 2 + treeSize + count);
284}

286bool BIH::readFromFile(FILE* rf)
287{
288 uint32 treeSize;
289 G3D::Vector3 lo, hi;
290 uint32 check = 0, count = 0;
291 check += fread(&lo, sizeof(float), 3, rf);
292 check += fread(&hi, sizeof(float), 3, rf);
293 bounds = G3D::AABox(lo, hi);
294 check += fread(&treeSize, sizeof(uint32), 1, rf);
295 tree.resize(treeSize);
296 check += fread(&tree[0], sizeof(uint32), treeSize, rf);
297 check += fread(&count, sizeof(uint32), 1, rf);
298 objects.resize(count); // = new uint32[nObjects];
299 check += fread(&objects[0], sizeof(uint32), count, rf);
300 return uint64(check) == uint64(3 + 3 + 1 + 1 + uint64(treeSize) + uint64(count));
301}

303void BIH::BuildStats::updateLeaf(int depth, int n)
304{
305 numLeaves++;
306 minDepth = std::min(depth, minDepth);
307 maxDepth = std::max(depth, maxDepth);
308 sumDepth += depth;
309 minObjects = std::min(n, minObjects);
310 maxObjects = std::max(n, maxObjects);
311 sumObjects += n;
312 int nl = std::min(n, 5);
313 ++numLeavesN[nl];
314}
