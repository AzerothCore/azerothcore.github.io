65void TempSummon::Update(uint32 diff)
66{
67 Creature::Update(diff);
68
69 if (m_deathState == DEAD)
70 {
71 UnSummon();
72 return;
73 }
74 switch (m_type)
75 {
76 case TEMPSUMMON_MANUAL_DESPAWN:
77 case TEMPSUMMON_DESPAWNED:
78 break;
79 case TEMPSUMMON_TIMED_DESPAWN:
80 {
81 if (m_timer <= diff)
82 {
83 UnSummon();
84 return;
85 }
86
87 m_timer -= diff;
88 break;
89 }
90 case TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT:
91 {
92 if (!IsInCombat())
93 {
94 if (m_timer <= diff)
95 {
96 UnSummon();
97 return;
98 }
99
100 m_timer -= diff;
101 }
102 else if (m_timer != m_lifetime)
103 m_timer = m_lifetime;
104
105 break;
106 }
107 case TEMPSUMMON_TIMED_DESPAWN_OOC_ALIVE:
108 {
109 if (!IsInCombat() && m_deathState != CORPSE)
110 {
111 if (m_timer <= diff)
112 {
113 UnSummon();
114 return;
115 }
116
117 m_timer -= diff;
118 }
119 else if (m_timer != m_lifetime)
120 m_timer = m_lifetime;
121
122 break;
123 }
124 case TEMPSUMMON_CORPSE_TIMED_DESPAWN:
125 {
126 if (m_deathState == CORPSE)
127 {
128 if (m_timer <= diff)
129 {
130 UnSummon();
131 return;
132 }
133
134 m_timer -= diff;
135 }
136 break;
137 }
138 case TEMPSUMMON_CORPSE_DESPAWN:
139 {
140 // if m_deathState is DEAD, CORPSE was skipped
141 if (m_deathState == CORPSE)
142 {
143 UnSummon();
144 return;
145 }
146
147 break;
148 }
149 case TEMPSUMMON_DEAD_DESPAWN:
150 {
151 break;
152 }
153 case TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN:
154 {
155 // if m_deathState is DEAD, CORPSE was skipped
156 if (m_deathState == CORPSE)
157 {
158 UnSummon();
159 return;
160 }
161
162 if (!IsInCombat())
163 {
164 if (m_timer <= diff)
165 {
166 UnSummon();
167 return;
168 }
169 else
170 m_timer -= diff;
171 }
172 else if (m_timer != m_lifetime)
173 m_timer = m_lifetime;
174 break;
175 }
176 case TEMPSUMMON_TIMED_OR_DEAD_DESPAWN:
177 {
178 if (!IsInCombat() && IsAlive())
179 {
180 if (m_timer <= diff)
181 {
182 UnSummon();
183 return;
184 }
185 else
186 m_timer -= diff;
187 }
188 else if (m_timer != m_lifetime)
189 m_timer = m_lifetime;
190 break;
191 }
192 default:
193 UnSummon();
194 LOG_ERROR("entities.unit", "Temporary summoned creature (entry: {}) have unknown type {} of ", GetEntry(), m_type);
195 break;
196 }
197}

199void TempSummon::InitStats(uint32 duration)
200{
201 ASSERT(!IsPet());
202
203 Unit* owner = GetSummonerUnit();
204 if (owner)
205 if (Player* player = owner->ToPlayer())
206 sScriptMgr->OnBeforeTempSummonInitStats(player, this, duration);
207
208 m_timer = duration;
209 m_lifetime = duration;
210
211 if (m_type == TEMPSUMMON_MANUAL_DESPAWN)
212 m_type = (duration == 0) ? TEMPSUMMON_DEAD_DESPAWN : TEMPSUMMON_TIMED_DESPAWN;
213
214 if (owner)
215 {
216 if (IsTrigger() && m_spells[0])
217 {
218 SetFaction(owner->GetFaction());
219 SetLevel(owner->getLevel());
220 if (owner->GetTypeId() == TYPEID_PLAYER)
221 m_ControlledByPlayer = true;
222 }
223
224 if (owner->GetTypeId() == TYPEID_PLAYER)
225 m_CreatedByPlayer = true;
226 }
227
228 if (!m_Properties)
229 return;
230
231 if (owner)
232 {
233 if (uint32 slot = m_Properties->Slot)
234 {
235 if (owner->m_SummonSlot[slot] && owner->m_SummonSlot[slot] != GetGUID())
236 {
237 Creature* oldSummon = GetMap()->GetCreature(owner->m_SummonSlot[slot]);
238 if (oldSummon && oldSummon->IsSummon())
239 oldSummon->ToTempSummon()->UnSummon();
240 }
241 owner->m_SummonSlot[slot] = GetGUID();
242 }
243 }
244
245 if (m_Properties->Faction)
246 SetFaction(m_Properties->Faction);
247 else if (IsVehicle() && owner) // properties should be vehicle
248 SetFaction(owner->GetFaction());
249}

270void TempSummon::UnSummon(uint32 msTime)
271{
272 if (msTime)
273 {
274 ForcedUnsummonDelayEvent* pEvent = new ForcedUnsummonDelayEvent(*this);
275
276 m_Events.AddEvent(pEvent, m_Events.CalculateTime(msTime));
277 return;
278 }
279
280 // Dont allow to call this function twice (possible)
281 if (m_type == TEMPSUMMON_DESPAWNED)
282 return;
283 SetTempSummonType(TEMPSUMMON_DESPAWNED);
284
285 //ASSERT(!IsPet());
286 if (IsPet())
287 {
288 ToPet()->Remove(PET_SAVE_NOT_IN_SLOT);
289 ASSERT(!IsInWorld());
290 return;
291 }
292
293 Unit* owner = GetSummonerUnit();
294 if (owner && owner->GetTypeId() == TYPEID_UNIT && owner->ToCreature()->IsAIEnabled)
295 owner->ToCreature()->AI()->SummonedCreatureDespawn(this);
296
297 AddObjectToRemoveList();
298}

375void Minion::setDeathState(DeathState s, bool despawn)
376{
377 Creature::setDeathState(s, despawn);
378 if (s == JUST_DIED && IsGuardianPet())
379 if (Unit* owner = GetOwner())
380 if (owner->GetTypeId() == TYPEID_PLAYER && owner->GetMinionGUID() == GetGUID())
381 for (Unit::ControlSet::const_iterator itr = owner->m_Controlled.begin(); itr != owner->m_Controlled.end(); ++itr)
382 if ((*itr)->IsAlive() && (*itr)->GetEntry() == GetEntry())
383 {
384 owner->SetMinionGUID((*itr)->GetGUID());
385 owner->SetPetGUID((*itr)->GetGUID());
386 owner->ToPlayer()->CharmSpellInitialize();
387 }
