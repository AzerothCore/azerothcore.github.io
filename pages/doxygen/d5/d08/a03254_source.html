179 bool SetBossState(uint32 type, EncounterState state) override
180 {
181 if (!InstanceScript::SetBossState(type, state))
182 return false;
183
184 switch (type)
185 {
186 case DATA_SAVIANA_RAGEFIRE:
187 case DATA_BALTHARUS_THE_WARBORN:
188 if (GetBossState(DATA_BALTHARUS_THE_WARBORN) == DONE && GetBossState(DATA_SAVIANA_RAGEFIRE) == DONE)
189 if (Creature* zarithrian = instance->GetCreature(GeneralZarithrianGUID))
190 {
191 zarithrian->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
192 zarithrian->SetImmuneToPC(false);
193 }
194 break;
195 case DATA_GENERAL_ZARITHRIAN:
196 if (state == DONE)
197 if (Creature* halionController = instance->GetCreature(HalionControllerGUID))
198 halionController->AI()->DoAction(ACTION_INTRO_HALION);
199 break;
200 case DATA_HALION:
201 DoUpdateWorldState(WORLDSTATE_CORPOREALITY_TOGGLE, 0);
202 DoUpdateWorldState(WORLDSTATE_CORPOREALITY_TWILIGHT, 0);
203 DoUpdateWorldState(WORLDSTATE_CORPOREALITY_MATERIAL, 0);
204 HandleGameObject(FlameRingGUID, true);
205 break;
206 }
207
208 return true;
209 }

229 void Load(char const* str) override
230 {
231 if (!str)
232 {
233 OUT_LOAD_INST_DATA_FAIL;
234 return;
235 }
236
237 OUT_LOAD_INST_DATA(str);
238
239 char dataHead1, dataHead2;
240
241 std::istringstream loadStream(str);
242 loadStream >> dataHead1 >> dataHead2;
243
244 if (dataHead1 == 'R' && dataHead2 == 'S')
245 {
246 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
247 {
248 uint32 tmpState;
249 loadStream >> tmpState;
250 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
251 tmpState = NOT_STARTED;
252
253 SetBossState(i, EncounterState(tmpState));
254 }
255
256 // Xinef: additional check
257 if (GetBossState(DATA_HALION_INTRO_DONE) != DONE)
258 {
259 SetBossState(DATA_HALION_INTRO1, NOT_STARTED);
260 SetBossState(DATA_HALION_INTRO2, NOT_STARTED);
261 }
262 }
263 else
264 OUT_LOAD_INST_DATA_FAIL;
265
266 OUT_LOAD_INST_DATA_COMPLETE;
267 }
