79 void JustSummoned(Creature* creature) override
80 {
81 BossAI::JustSummoned(creature);
82
83 float ImageHealthPct = 0.f;
84 if (me->GetHealthPct() < 25.0f)
85 ImageHealthPct = 0.50f;

87 ImageHealthPct = 0.20f;

102 void DoTeleport(Creature* creature)
103 {
104 // Shift the boss and images (Get it? *Shift*?)
105 uint8 rand = 0;
106 if (_flag != 0)
107 {
108 while (_flag & (1 << rand))
109 rand = urand(0, 2);
110 DoCast(me, BlinkSpells[rand]);
111 _flag |= (1 << rand);
112 _flag |= (1 << 7);
113 }
114
115 while (_flag & (1 << rand))
116 rand = urand(0, 2);
117
118 creature->SetReactState(REACT_AGGRESSIVE);
119 creature->SetImmuneToAll(false);
120 creature->SetControlled(false, UNIT_STATE_ROOT);
121 creature->CastSpell(creature, BlinkSpells[rand], true);
122
123 _flag |= (1 << rand);
124
125 if (_flag & (1 << 7))
126 _flag = 0;
127
128 events.ScheduleEvent(EVENT_INIT_IMAGE, 400ms);
129 }

160 void UpdateAI(uint32 diff) override
161 {
162 if (!UpdateVictim())
163 return;
164
165 events.Update(diff);
166
167 while (uint32 eventId = events.ExecuteEvent())
168 {
169 switch (eventId)
170 {
171 case EVENT_ARCANE_EXPLOSION:
172 DoCastAOE(SPELL_ARCANE_EXPLOSION, false);
173 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, 8s, 18s);
174 break;
175 case EVENT_FULLFILMENT:
176 DoCast(SelectTarget(SelectTargetMethod::MinDistance, 1, 0.0f, true), SPELL_TRUE_FULFILLMENT, false);
177 events.ScheduleEvent(EVENT_FULLFILMENT, 20s, 30s);
178 break;
179 case EVENT_BLINK:
180 DoCast(me, BlinkSpells[urand(0, 2)]);
181 DoResetThreat();
182 events.ScheduleEvent(EVENT_BLINK, 10s, 30s);
183 break;
184 case EVENT_EARTH_SHOCK:
185 DoCastVictim(SPELL_EARTH_SHOCK);
186 events.ScheduleEvent(EVENT_EARTH_SHOCK, 1200ms);
187 break;
188 case EVENT_TELEPORT:
189 me->SetReactState(REACT_AGGRESSIVE);
190 me->SetImmuneToAll(false);
191 me->SetControlled(false, UNIT_STATE_ROOT);
192 for (ObjectGuid const& guid : _copiesGUIDs)
193 {
194 if (Creature* image = ObjectAccessor::GetCreature(*me, guid))
195 {
196 DoTeleport(image);
197 }
198 }
199 DoResetThreat();
200 events.RescheduleEvent(EVENT_BLINK, 10s, 30s);
201 break;
202 case EVENT_INIT_IMAGE:
203 me->CastSpell(me, SPELL_INITIALIZE_IMAGE, true);
204 break;
205 }
206 }
207
208 if (!me->IsSummon() && me->GetHealthPct() < _hpct)
209 {
210 _copiesGUIDs.clear();
211 DoCast(me, SPELL_SUMMON_IMAGES, true);
212 me->SetReactState(REACT_PASSIVE);
213 me->SetImmuneToAll(true);
214 me->SetControlled(true, UNIT_STATE_ROOT);
215 Talk(SAY_SPLIT);
216 _hpct -= 25.0f;
217 events.ScheduleEvent(EVENT_TELEPORT, 2s);
218 }
219
220 if (Unit* myVictim = me->GetVictim())
221 {
222 if (me->IsWithinMeleeRange(myVictim))
223 {
224 DoMeleeAttackIfReady();
225
226 if (Unit* victimTarget = myVictim->GetVictim())
227 {
228 if (victimTarget->GetGUID() == me->GetGUID())
229 {
230 events.RescheduleEvent(EVENT_EARTH_SHOCK, 1200ms);
231 }
232 }
233 }
234 }
235 }

237private:
238 float _hpct;
239 uint8 _flag;
240 GuidVector _copiesGUIDs;
241};
