136 events.ScheduleEvent(EVENT_SPELL_CATACLYSMIC_BOLT, 10000);
137 events.ScheduleEvent(EVENT_SPELL_ENRAGE, 600000);
138 events.ScheduleEvent(EVENT_SPELL_SEAR_NOVA, 25000);
139 events.ScheduleEvent(EVENT_HEALTH_CHECK, 1000);
140 }
141
142 void UpdateAI(uint32 diff) override
143 {
144 if (!UpdateVictim())
145 return;
146
147 events.Update(diff);
148 if (me->HasUnitState(UNIT_STATE_CASTING))
149 return;
150
151 switch (events.ExecuteEvent())
152 {
153 case EVENT_SPELL_ENRAGE:
154 me->CastSpell(me, SPELL_ENRAGE, true);
155 break;
156 case EVENT_SPELL_CATACLYSMIC_BOLT:
157 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, PowerUsersSelector(me, POWER_MANA, 50.0f, true)))
158 me->CastSpell(target, SPELL_CATACLYSMIC_BOLT, false);
159 events.ScheduleEvent(EVENT_SPELL_CATACLYSMIC_BOLT, 6000);
160 break;
161 case EVENT_SPELL_SEAR_NOVA:
162 me->CastSpell(me, SPELL_SEAR_NOVA, false);
163 events.ScheduleEvent(EVENT_SPELL_SEAR_NOVA, 20000 + urand(0, 20000));
164 break;
165 case EVENT_HEALTH_CHECK:
166 if (me->HealthBelowPct(76))
167 {
168 for (SummonList::const_iterator itr = summons.begin(); itr != summons.end(); ++itr)
169 if (Creature* summon = ObjectAccessor::GetCreature(*me, *itr))
170 if (summon->GetMaxHealth() > 500000)
171 summon->CastSpell(me, SPELL_BLESSING_OF_THE_TIDES, true);
172
173 if (me->HasAura(SPELL_BLESSING_OF_THE_TIDES))
174 Talk(SAY_GAIN_BLESSING);
175 break;
176 }
177 events.ScheduleEvent(EVENT_HEALTH_CHECK, 1000);
178 break;
179 }
180
181 DoMeleeAttackIfReady();
182 }
183 };
