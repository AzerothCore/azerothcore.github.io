50struct boss_huhuran : public BossAI
51{
52 boss_huhuran(Creature* creature) : BossAI(creature, DATA_HUHURAN)
53 {
54 me->m_CombatDistance = 90.f;
55 }
56
57 void Reset() override
58 {
59 BossAI::Reset();
60 _berserk = false;
61 _hardEnrage = false;
62 }
63
64 void EnterCombat(Unit* who) override
65 {
66 BossAI::EnterCombat(who);
67 events.ScheduleEvent(EVENT_FRENZY, 12s, 21s);
68 events.ScheduleEvent(EVENT_WYVERN_STING, 25s, 43s);
69 events.ScheduleEvent(EVENT_ACID_SPIT, 1s, 20s);
70 events.ScheduleEvent(EVENT_NOXIOUS_POISON, 10s, 22s);
71 events.ScheduleEvent(EVENT_HARD_ENRAGE, 5min);
72 }
73
74 void DamageTaken(Unit*, uint32& /*damage*/, DamageEffectType, SpellSchoolMask) override
75 {
76 if (!_berserk && HealthBelowPct(30))
77 {
78 DoCastSelf(SPELL_BERSERK, true);
79 me->TextEmote(EMOTE_BERSERK);
80 events.CancelEvent(EVENT_FRENZY);
81 _berserk = true;
82 }
83 }
84
85 void UpdateAI(uint32 diff) override
86 {
87 if (!UpdateVictim())
88 return;
89
90 events.Update(diff);
91 while (uint32 eventid = events.ExecuteEvent())
92 {
93 switch (eventid)
94 {
95 case EVENT_FRENZY:
96 DoCastSelf(SPELL_FRENZY, true);
97 Talk(EMOTE_FRENZY_KILL);
98 events.Repeat(12s, 21s);
99 break;
100 case EVENT_WYVERN_STING:
101 me->CastCustomSpell(SPELL_WYVERN_STING, SPELLVALUE_MAX_TARGETS, 10, me, true);
102 events.Repeat(25s, 43s);
103 break;
104 case EVENT_ACID_SPIT:
105 DoCastVictim(SPELL_ACID_SPIT);
106 events.Repeat(1s, 20s);
107 break;
108 case EVENT_NOXIOUS_POISON:
109 DoCastRandomTarget(SPELL_NOXIOUS_POISON, 0, 100.f, true);
110 events.Repeat(10s, 22s);
111 break;
112 case EVENT_HARD_ENRAGE:
113 if (!_hardEnrage)
114 {
115 DoCastSelf(SPELL_BERSERK, true);
116 events.CancelEvent(EVENT_FRENZY);
117 _hardEnrage = true;
118 }
119 else
120 {
121 DoCastAOE(SPELL_POISON_BOLT);
122 }
123 events.Repeat(2s);
124 break;
125 default:
126 break;
127 }
128 }
129 DoMeleeAttackIfReady();
130 }
131
132private:
133 bool _berserk;
134 bool _hardEnrage;
135};
