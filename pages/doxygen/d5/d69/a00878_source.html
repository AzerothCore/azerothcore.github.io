243{
244 // pussywizard:
245 m_UpdateTimer += diff;
246 if (m_UpdateTimer < BATTLEGROUND_UPDATE_INTERVAL)
247 return;
248

251
252 if (!PreUpdateImpl(diff))
253 return;
254
255 if (!GetPlayersSize())
256 {
257 //BG is empty

270 }

273 }
274
275 switch (GetStatus())
276 {
277 case STATUS_WAIT_JOIN:
278 if (GetPlayersSize())
279 {
280 _ProcessJoin(diff);
281 _CheckSafePositions(diff);
282 }
283 break;
284 case STATUS_IN_PROGRESS:
285 if (isArena())
286 {
287 if (GetStartTime() >= 46 * MINUTE * IN_MILLISECONDS) // pussywizard: 1min startup + 45min allowed duration
288 {
289 EndBattleground(PVP_TEAM_NEUTRAL);
290 return;
291 }
292 }
293 else

295 _ProcessResurrect(diff);
296 if (sBattlegroundMgr->GetPrematureFinishTime() && (GetPlayersCountByTeam(TEAM_ALLIANCE) < GetMinPlayersPerTeam() || GetPlayersCountByTeam(TEAM_HORDE) < GetMinPlayersPerTeam()))
297 _ProcessProgress(diff);
298 else if (m_PrematureCountDown)
299 m_PrematureCountDown = false;
300 }
301 break;
302 case STATUS_WAIT_LEAVE:
303 _ProcessLeave(diff);
304 break;
305 default:
306 break;
307 }
308
309 // Update start time and reset stats timer
310 m_StartTime += diff;
311 m_ResetStatTimer += diff;
312
313 PostUpdateImpl(diff);
314
315 sScriptMgr->OnBattlegroundUpdate(this, diff);
316}

344inline void Battleground::_ProcessResurrect(uint32 diff)
345{
346 // *********************************************************
347 // *** BATTLEGROUND RESURRECTION SYSTEM ***
348 // *********************************************************
349 // this should be handled by spell system
350
351 _reviveEvents.Update(diff);
352
353 m_LastResurrectTime += diff;
354 if (m_LastResurrectTime >= RESURRECTION_INTERVAL)
355 {
356 if (GetReviveQueueSize())
357 {
358 for (std::map<ObjectGuid, GuidVector>::iterator itr = m_ReviveQueue.begin(); itr != m_ReviveQueue.end(); ++itr)
359 {
360 Creature* sh = nullptr;
361 for (ObjectGuid const& guid : itr->second)
362 {
363 Player* player = ObjectAccessor::FindPlayer(guid);
364 if (!player)
365 continue;
366
367 if (!sh && player->IsInWorld())
368 {
369 sh = player->GetMap()->GetCreature(itr->first);
370 // only for visual effect
371 if (sh)
372 // Spirit Heal, effect 117
373 sh->CastSpell(sh, SPELL_SPIRIT_HEAL, true);
374 }
375
376 // Resurrection visual
377 player->CastSpell(player, SPELL_RESURRECTION_VISUAL, true);
378 m_ResurrectQueue.push_back(guid);
379 }
380
381 itr->second.clear();
382 }
383
384 m_ReviveQueue.clear();
385 m_LastResurrectTime = 0;
386 }
387 else
388 // queue is clear and time passed, just update last resurrection time
389 m_LastResurrectTime = 0;
390 }
391 else if (m_LastResurrectTime > 500) // Resurrect players only half a second later, to see spirit heal effect on NPC
392 {
393 for (ObjectGuid const& guid : m_ResurrectQueue)
394 {
395 Player* player = ObjectAccessor::FindPlayer(guid);
396 if (!player)
397 continue;
398 player->ResurrectPlayer(1.0f);
399 player->CastSpell(player, 6962, true);
400 player->CastSpell(player, SPELL_SPIRIT_HEAL_MANA, true);
401 player->SpawnCorpseBones(false);
402 }
403 m_ResurrectQueue.clear();
404 }

417inline void Battleground::_ProcessProgress(uint32 diff)
418{
419 // *********************************************************
420 // *** BATTLEGROUND BALLANCE SYSTEM ***
421 // *********************************************************
422 // if less then minimum players are in on one side, then start premature finish timer
423 if (!m_PrematureCountDown)
424 {
425 m_PrematureCountDown = true;
426 m_PrematureCountDownTimer = sBattlegroundMgr->GetPrematureFinishTime();
427 }
428 else if (m_PrematureCountDownTimer < diff)
429 {
430 // time's up!
431 EndBattleground(GetPrematureWinner());
432 m_PrematureCountDown = false;
433 }
434 else if (!sBattlegroundMgr->isTesting())
435 {
436 uint32 newtime = m_PrematureCountDownTimer - diff;
437 // announce every minute
438 if (newtime > (MINUTE * IN_MILLISECONDS))
439 {
440 if (newtime / (MINUTE * IN_MILLISECONDS) != m_PrematureCountDownTimer / (MINUTE * IN_MILLISECONDS))
441 PSendMessageToAll(LANG_BATTLEGROUND_PREMATURE_FINISH_WARNING, CHAT_MSG_SYSTEM, nullptr, (uint32)(m_PrematureCountDownTimer / (MINUTE * IN_MILLISECONDS)));
442 }
443 else
444 {
445 //announce every 15 seconds
446 if (newtime / (15 * IN_MILLISECONDS) != m_PrematureCountDownTimer / (15 * IN_MILLISECONDS))
447 PSendMessageToAll(LANG_BATTLEGROUND_PREMATURE_FINISH_WARNING_SECS, CHAT_MSG_SYSTEM, nullptr, (uint32)(m_PrematureCountDownTimer / IN_MILLISECONDS));
448 }
449 m_PrematureCountDownTimer = newtime;
450 }

453inline void Battleground::_ProcessJoin(uint32 diff)
454{
455 // *********************************************************
456 // *** BATTLEGROUND STARTING SYSTEM ***
457 // *********************************************************
458 ModifyStartDelayTime(diff);
459
460 if (m_ResetStatTimer > 5000)
461 {
462 m_ResetStatTimer = 0;
463 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
464 itr->second->ResetAllPowers();
465 }
466
467 if (!(m_Events & BG_STARTING_EVENT_1))
468 {
469 m_Events |= BG_STARTING_EVENT_1;
470
471 if (!FindBgMap())
472 {
473 LOG_ERROR("bg.battleground", "Battleground::_ProcessJoin: map (map id: {}, instance id: {}) is not created!", m_MapId, m_InstanceID);
474 EndNow();
475 return;
476 }
477
478 // Setup here, only when at least one player has ported to the map
479 if (!SetupBattleground())
480 {
481 EndNow();
482 return;
483 }
484
485 StartingEventCloseDoors();
486 SetStartDelayTime(StartDelayTimes[BG_STARTING_EVENT_FIRST]);
487
488 // First start warning - 2 or 1 minute
489 if (StartMessageIds[BG_STARTING_EVENT_FIRST])
490 SendBroadcastText(StartMessageIds[BG_STARTING_EVENT_FIRST], CHAT_MSG_BG_SYSTEM_NEUTRAL);
491 }
492 // After 1 minute or 30 seconds, warning is signaled
493 else if (GetStartDelayTime() <= StartDelayTimes[BG_STARTING_EVENT_SECOND] && !(m_Events & BG_STARTING_EVENT_2))
494 {
495 m_Events |= BG_STARTING_EVENT_2;
496
497 if (StartMessageIds[BG_STARTING_EVENT_SECOND])
498 SendBroadcastText(StartMessageIds[BG_STARTING_EVENT_SECOND], CHAT_MSG_BG_SYSTEM_NEUTRAL);
499 }
500 // After 30 or 15 seconds, warning is signaled
501 else if (GetStartDelayTime() <= StartDelayTimes[BG_STARTING_EVENT_THIRD] && !(m_Events & BG_STARTING_EVENT_3))
502 {
503 m_Events |= BG_STARTING_EVENT_3;
504
505 if (StartMessageIds[BG_STARTING_EVENT_THIRD])
506 SendBroadcastText(StartMessageIds[BG_STARTING_EVENT_THIRD], CHAT_MSG_BG_SYSTEM_NEUTRAL);
507
508 if (isArena())
509 switch (GetBgTypeID())
510 {
511 case BATTLEGROUND_NA:
512 DelObject(BG_NA_OBJECT_READY_MARKER_1);
513 DelObject(BG_NA_OBJECT_READY_MARKER_2);
514 break;
515 case BATTLEGROUND_BE:
516 DelObject(BG_BE_OBJECT_READY_MARKER_1);
517 DelObject(BG_BE_OBJECT_READY_MARKER_2);
518 break;
519 case BATTLEGROUND_RL:
520 DelObject(BG_RL_OBJECT_READY_MARKER_1);
521 DelObject(BG_RL_OBJECT_READY_MARKER_2);
522 break;
523 case BATTLEGROUND_DS:
524 DelObject(BG_DS_OBJECT_READY_MARKER_1);
525 DelObject(BG_DS_OBJECT_READY_MARKER_2);
526 break;
527 case BATTLEGROUND_RV:
528 DelObject(BG_RV_OBJECT_READY_MARKER_1);
529 DelObject(BG_RV_OBJECT_READY_MARKER_2);
530 break;
531 default:
532 break;
533 }
534 }
535 // Delay expired (after 2 or 1 minute)
536 else if (GetStartDelayTime() <= 0 && !(m_Events & BG_STARTING_EVENT_4))
537 {
538 m_Events |= BG_STARTING_EVENT_4;
539
540 StartingEventOpenDoors();
541
542 if (StartMessageIds[BG_STARTING_EVENT_FOURTH])
543 SendBroadcastText(StartMessageIds[BG_STARTING_EVENT_FOURTH], CHAT_MSG_BG_SYSTEM_NEUTRAL);
544
545 SetStatus(STATUS_IN_PROGRESS);
546 SetStartDelayTime(StartDelayTimes[BG_STARTING_EVENT_FOURTH]);
547
548 // Remove preparation
549 if (isArena())
550 {
551 // pussywizard: initial visibility range is 30yd, set it to a proper value now:
552 if (BattlegroundMap* map = GetBgMap())
553 map->SetVisibilityRange(World::GetMaxVisibleDistanceInBGArenas());
554
555 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
556 if (Player* player = itr->second)
557 {
558 WorldPacket status;
559 sBattlegroundMgr->BuildBattlegroundStatusPacket(&status, this, player->GetCurrentBattlegroundQueueSlot(), STATUS_IN_PROGRESS, 0, GetStartTime(), GetArenaType(), player->GetBgTeamId());
560 player->GetSession()->SendPacket(&status);
561
562 player->RemoveAurasDueToSpell(SPELL_ARENA_PREPARATION);
563 player->ResetAllPowers();
564 // remove auras with duration lower than 30s
565 Unit::AuraApplicationMap& auraMap = player->GetAppliedAuras();
566 for (Unit::AuraApplicationMap::iterator iter = auraMap.begin(); iter != auraMap.end();)
567 {
568 AuraApplication* aurApp = iter->second;
569 Aura* aura = aurApp->GetBase();
570 if (!aura->IsPermanent()
571 && aura->GetDuration() <= 30 * IN_MILLISECONDS
572 && aurApp->IsPositive()
573 // && (!aura->GetSpellInfo()->HasAttribute(SPELL_ATTR0_NO_IMMUNITIES)) Xinef: condition, ALL buffs should be removed
574 && (!aura->HasEffectType(SPELL_AURA_MOD_INVISIBILITY)))
575 player->RemoveAura(iter);
576 else
577 ++iter;
578 }
579
580 player->UpdateObjectVisibility(true);
581 }
582
583 for (SpectatorList::const_iterator itr = m_Spectators.begin(); itr != m_Spectators.end(); ++itr)
584 ArenaSpectator::HandleResetCommand(*itr);
585
586 CheckWinConditions();
587
588 // pussywizard: arena spectator stuff
589 if (GetStatus() == STATUS_IN_PROGRESS)
590 {
591 for (ToBeTeleportedMap::const_iterator itr = m_ToBeTeleported.begin(); itr != m_ToBeTeleported.end(); ++itr)
592 if (Player* p = ObjectAccessor::FindConnectedPlayer(itr->first))
593 if (Player* t = ObjectAccessor::FindPlayer(itr->second))
594 {
595 if (!t->FindMap() || t->FindMap() != GetBgMap())
596 continue;
597
598 p->SetSummonPoint(t->GetMapId(), t->GetPositionX(), t->GetPositionY(), t->GetPositionZ(), 15, true);
599
600 WorldPacket data(SMSG_SUMMON_REQUEST, 8 + 4 + 4);
601 data << t->GetGUID();
602 data << uint32(t->GetZoneId());
603 data << uint32(15 * IN_MILLISECONDS);
604 p->GetSession()->SendPacket(&data);
605 }
606 m_ToBeTeleported.clear();
607 }
608 }
609 else
610 {
611 PlaySoundToAll(SOUND_BG_START);
612
613 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
614 {
615 itr->second->RemoveAurasDueToSpell(SPELL_PREPARATION);
616 itr->second->ResetAllPowers();
617 }
618
619 // Announce BG starting
620 if (sWorld->getBoolConfig(CONFIG_BATTLEGROUND_QUEUE_ANNOUNCER_ENABLE))
621 sWorld->SendWorldText(LANG_BG_STARTED_ANNOUNCE_WORLD, GetName().c_str(), std::min(GetMinLevel(), (uint32)80), std::min(GetMaxLevel(), (uint32)80));
622
623 sScriptMgr->OnBattlegroundStart(this);
624 }
625 }

1194uint32 Battleground::GetFreeSlotsForTeam(TeamId teamId) const
1195{
1196 if (!(GetStatus() == STATUS_IN_PROGRESS || GetStatus() == STATUS_WAIT_JOIN))
1197 return 0;
1198
1199 // if CONFIG_BATTLEGROUND_INVITATION_TYPE == BG_QUEUE_INVITATION_TYPE_NO_BALANCE, invite everyone unless the BG is full
1200 if (sWorld->getIntConfig(CONFIG_BATTLEGROUND_INVITATION_TYPE) == BG_QUEUE_INVITATION_TYPE_NO_BALANCE)
1201 return (GetInvitedCount(teamId) < GetMaxPlayersPerTeam()) ? GetMaxPlayersPerTeam() - GetInvitedCount(teamId) : 0;
1202
1203 // if BG is already started or CONFIG_BATTLEGROUND_INVITATION_TYPE != BG_QUEUE_INVITATION_TYPE_NO_BALANCE, do not allow to join too many players of one faction
1204 uint32 thisTeamInvitedCount = teamId == TEAM_ALLIANCE ? GetInvitedCount(TEAM_ALLIANCE) : GetInvitedCount(TEAM_HORDE);
1205 uint32 thisTeamPlayersCount = teamId == TEAM_ALLIANCE ? GetPlayersCountByTeam(TEAM_ALLIANCE) : GetPlayersCountByTeam(TEAM_HORDE);
1206 uint32 otherTeamInvitedCount = teamId == TEAM_ALLIANCE ? GetInvitedCount(TEAM_HORDE) : GetInvitedCount(TEAM_ALLIANCE);
1207 uint32 otherTeamPlayersCount = teamId == TEAM_ALLIANCE ? GetPlayersCountByTeam(TEAM_HORDE) : GetPlayersCountByTeam(TEAM_ALLIANCE);
1208
1209 // difference based on ppl invited (not necessarily entered battle)
1210 // default: allow 0
1211 uint32 diff = 0;
1212 uint32 maxPlayersPerTeam = GetMaxPlayersPerTeam();
1213 uint32 minPlayersPerTeam = GetMinPlayersPerTeam();
1214
1215 // allow join one person if the sides are equal (to fill up bg to minPlayerPerTeam)
1216 if (otherTeamInvitedCount == thisTeamInvitedCount)
1217 diff = 1;
1218 else if (otherTeamInvitedCount > thisTeamInvitedCount) // allow join more ppl if the other side has more players
1219 diff = otherTeamInvitedCount - thisTeamInvitedCount;
1220
1221 // difference based on max players per team (don't allow inviting more)
1222 uint32 diff2 = (thisTeamInvitedCount < maxPlayersPerTeam) ? maxPlayersPerTeam - thisTeamInvitedCount : 0;
1223
1224 // difference based on players who already entered
1225 // default: allow 0
1226 uint32 diff3 = 0;
1227
1228 // allow join one person if the sides are equal (to fill up bg minPlayerPerTeam)
1229 if (otherTeamPlayersCount == thisTeamPlayersCount)
1230 diff3 = 1;
1231 else if (otherTeamPlayersCount > thisTeamPlayersCount) // allow join more ppl if the other side has more players
1232 diff3 = otherTeamPlayersCount - thisTeamPlayersCount;
1233 else if (thisTeamInvitedCount <= minPlayersPerTeam) // or other side has less than minPlayersPerTeam
1234 diff3 = minPlayersPerTeam - thisTeamInvitedCount + 1;
1235
1236 // return the minimum of the 3 differences
1237 // min of diff, diff2 and diff3
1238 return std::min({ diff, diff2, diff3 });
1239}
