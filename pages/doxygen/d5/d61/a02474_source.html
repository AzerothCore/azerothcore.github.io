75 boss_mor_grayhoof(Creature* creature) : BossAI(creature, DATA_MOR_GRAYHOOF)
76 {
77 _scheduler.SetValidator([this]

85 void Reset() override
86 {
87 _scheduler.CancelAll();
88 _phase = PHASE_HUMAN;
89 }

91 void CastRandomSpell(uint8 phase)
92 {
93 uint32 spell;
94 switch (phase)
95 {
96 case PHASE_HUMAN:
97 spell = Acore::Containers::SelectRandomContainerElement(humanSpells);
98 if (spell == SPELL_REJUVENATION || spell == SPELL_HEALING_TOUCH)
99 {
100 DoCastSelf(spell);
101 }
102 else
103 {
104 DoCastAOE(spell);
105 }
106 break;
107 case PHASE_BEAR:
108 spell = Acore::Containers::SelectRandomContainerElement(bearSpells);
109 DoCastVictim(spell);
110 break;
111 case PHASE_CAT:
112 spell = Acore::Containers::SelectRandomContainerElement(catSpells);
113 DoCastVictim(spell);
114 break;
115 case PHASE_FAERIE:
116 spell = Acore::Containers::SelectRandomContainerElement(faerieSpells);
117 if (spell == SPELL_SLEEP)
118 {
119 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.f, true))
120 {
121 me->CastSpell(target, SPELL_SLEEP);
122
123 // Sleep can target tank, we need to drop threat temporarily on the target.
124 _sleepTargetGUID = target->GetGUID();
125 _sleepTargetThreat = me->GetThreatMgr().GetThreat(target);
126 me->GetThreatMgr().ModifyThreatByPercent(target, -100);
127 _scheduler.Schedule(10s, [this](TaskContext /*context*/)
128 {
129 if (Unit* sleepTarget = ObjectAccessor::GetUnit(*me, _sleepTargetGUID))
130 {
131 me->GetThreatMgr().AddThreat(sleepTarget, _sleepTargetThreat);
132 }
133 });
134 }
135 }
136 else
137 {
138 DoCastVictim(spell);
139 }
140 break;
141 default:
142 break;
143 }
144 }
145
146 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/, DamageEffectType /*type*/, SpellSchoolMask /*school*/) override
147 {
148 if (_phase == PHASE_HUMAN && me->HealthBelowPct(75.f))
149 {
150 _phase = PHASE_BEAR;
151 me->CastStop();
152 _scheduler.DelayGroup(PHASE_HUMAN, 10s);
153 DoCastSelf(SPELL_BEAR_FORM);
154 _scheduler.Schedule(3s, PHASE_BEAR, [this](TaskContext context)
155 {
156 CastRandomSpell(PHASE_BEAR);
157 if (context.GetRepeatCounter() <= 3)
158 {
159 context.Repeat();
160 }
161 });
162 }
163 else if (_phase == PHASE_BEAR && me->HealthBelowPct(50.f))
164 {
165 _phase = PHASE_CAT;
166 _scheduler.DelayGroup(PHASE_HUMAN, 10s);
167 me->CastStop();
168 DoCastSelf(SPELL_CAT_FORM);
169 _scheduler.Schedule(3s, PHASE_CAT, [this](TaskContext context)
170 {
171 CastRandomSpell(PHASE_CAT);
172 if (context.GetRepeatCounter() <= 3)
173 {
174 context.Repeat();
175 }
176 });
177 }
178 else if (_phase == PHASE_CAT && me->HealthBelowPct(25.f))
179 {
180 _phase = PHASE_FAERIE;
181 _scheduler.CancelGroup(PHASE_HUMAN);
182 me->CastStop();
183 DoCastSelf(SPELL_FAERIE_DRAGON_FORM);
184 _scheduler.Schedule(5s, 10s, PHASE_FAERIE, [this](TaskContext context)
185 {
186 CastRandomSpell(PHASE_FAERIE);
187 context.Repeat(5s, 10s);
188 });
189 }
190 }
191
192 void EnterCombat(Unit* /*who*/) override
193 {
194 _EnterCombat();
195 Talk(SAY_AGGRO);
196 _scheduler.Schedule(5s, 10s, PHASE_HUMAN, [this](TaskContext context)
197 {
198 CastRandomSpell(PHASE_HUMAN);
199 context.Repeat();
200 });
201 }
202
203 void UpdateAI(uint32 diff) override
204 {
205 if (!UpdateVictim())
206 {
207 return;
208 }
209
210 _scheduler.Update(diff, [this]
211 {
212 DoMeleeAttackIfReady();
213 });
214 }
215
216 void JustDied(Unit* /*killer*/) override
217 {
218 _JustDied();
219 Talk(SAY_DEATH);
220 }

222protected:

225 ObjectGuid _sleepTargetGUID;
226 float _sleepTargetThreat;
