218 void Load(const char* strIn) override
219 {
220 if (!strIn)
221 {
222 OUT_LOAD_INST_DATA_FAIL;
223 return;
224 }
225
226 OUT_LOAD_INST_DATA(strIn);
227
228 char dataHead1, dataHead2;
229
230 std::istringstream loadStream(strIn);
231 loadStream >> dataHead1 >> dataHead2;
232
233 if (dataHead1 == 'O' && dataHead2 == 'S')
234 {
235 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
236 {
237 uint32 temp;
238 loadStream >> temp;
239 if (temp == IN_PROGRESS)
240 temp = NOT_STARTED;
241
242 SetBossState(i, static_cast<EncounterState>(temp));
243 }
244 }
245
246 OUT_LOAD_INST_DATA_COMPLETE;
