65void BattlegroundIC::DoAction(uint32 action, ObjectGuid guid)
66{
67 if (action != ACTION_TELEPORT_PLAYER_TO_TRANSPORT)
68 return;
69
70 if (!gunshipAlliance || !gunshipHorde)
71 return;
72
73 Player* player = ObjectAccessor::GetPlayer(*gunshipAlliance, guid);
74 if (!player)
75 return;
76
77 MotionTransport* transport = player->GetTeamId() == TEAM_ALLIANCE ? gunshipAlliance : gunshipHorde;
78 float x = BG_IC_HangarTrigger[player->GetTeamId()].GetPositionX();
79 float y = BG_IC_HangarTrigger[player->GetTeamId()].GetPositionY();
80 float z = BG_IC_HangarTrigger[player->GetTeamId()].GetPositionZ();
81 transport->CalculatePassengerPosition(x, y, z);
82
83 player->TeleportTo(GetMapId(), x, y, z, player->GetOrientation(), TELE_TO_NOT_LEAVE_TRANSPORT);
84}

214 siegeEngineWorkshopTimer = WORKSHOP_UPDATE_TIME;
215 }
216 else
217 siegeEngineWorkshopTimer -= diff;
218 }
219 }
220
221 // the point is waiting for a change on its banner
222 if (nodePoint[i].needChange)
223 {
224 if (nodePoint[i].timer <= diff)
225 {
226 uint32 nextBanner = GetNextBanner(&nodePoint[i], nodePoint[i].faction, true);
227
228 nodePoint[i].last_entry = nodePoint[i].gameobject_entry;
229 nodePoint[i].gameobject_entry = nextBanner;
230 // nodePoint[i].faction = the faction should be the same one...
231
232 GameObject* banner = GetBGObject(nodePoint[i].gameobject_type);
233
234 if (!banner) // this should never happen
235 return;
236
237 float cords[4] = {banner->GetPositionX(), banner->GetPositionY(), banner->GetPositionZ(), banner->GetOrientation() };
238
239 DelObject(nodePoint[i].gameobject_type);
240 AddObject(nodePoint[i].gameobject_type, nodePoint[i].gameobject_entry, cords[0], cords[1], cords[2], cords[3], 0, 0, 0, 0, RESPAWN_ONE_DAY);
241
242 GetBGObject(nodePoint[i].gameobject_type)->SetUInt32Value(GAMEOBJECT_FACTION, nodePoint[i].faction == TEAM_ALLIANCE ? BG_IC_Factions[1] : BG_IC_Factions[0]);
243
244 UpdateNodeWorldState(&nodePoint[i]);
245 HandleCapturedNodes(&nodePoint[i], false);
246
247 if (nodePoint[i].faction == TEAM_ALLIANCE)
248 {
249 SendBroadcastText(ICNodes[i].TextAllianceTaken, CHAT_MSG_BG_SYSTEM_ALLIANCE);
250 }
251 else
252 {
253 SendBroadcastText(ICNodes[i].TextHordeTaken, CHAT_MSG_BG_SYSTEM_HORDE);
254 }
255
256 nodePoint[i].needChange = false;
257 nodePoint[i].timer = BANNER_STATE_CHANGE_TIME;
258 }
259 else nodePoint[i].timer -= diff;
260 }
261 }
262
263 if (resourceTimer <= diff)
264 {
265 for (uint8 i = 0; i < NODE_TYPE_DOCKS; ++i)
266 {
267 if (nodePoint[i].nodeState == NODE_STATE_CONTROLLED_A ||
268 nodePoint[i].nodeState == NODE_STATE_CONTROLLED_H)
269 {
270 factionReinforcements[nodePoint[i].faction] += 1;
271 RewardHonorToTeam(RESOURCE_HONOR_AMOUNT, nodePoint[i].faction);
272 UpdateWorldState((nodePoint[i].faction == TEAM_ALLIANCE ? BG_IC_ALLIANCE_RENFORT : BG_IC_HORDE_RENFORT), factionReinforcements[nodePoint[i].faction]);
273 }
274 }
275 resourceTimer = IC_RESOURCE_TIME;
276 }
277 else resourceTimer -= diff;
278}

340{
341 // this is wrong way to implement these things. On official it done by gameobject spell cast.
342 if (GetStatus() != STATUS_IN_PROGRESS)
343 return;
344
345 switch (trigger)
346 {
347 case AREA_TRIGGER_HORDE_KEEP:
348 if (player->GetTeamId() != TEAM_ALLIANCE)
349 return;
350 for (uint8 i = BG_IC_H_FRONT; i < BG_IC_A_FRONT; ++i)
351 if (GateStatus[i] == BG_IC_GATE_DESTROYED)
352 return;
353 if (!player->HasAchieved(3854)) // ACHIEVEMENT_BACK_DOOR_JOB
354 player->CastSpell(player, SPELL_BACK_DOOR_JOB, true);
355 break;
356 case AREA_TRIGGER_ALLIANCE_KEEP:
357 if (player->GetTeamId() != TEAM_HORDE)
358 return;
359 for (uint8 i = BG_IC_A_FRONT; i < BG_IC_MAXDOOR; ++i)
360 if (GateStatus[i] == BG_IC_GATE_DESTROYED)
361 return;
362 if (!player->HasAchieved(3854)) // ACHIEVEMENT_BACK_DOOR_JOB
363 player->CastSpell(player, SPELL_BACK_DOOR_JOB, true);
364 break;
365 }

689{

705 if (nodePoint->gameobject_entry == nodePoint->banners[BANNER_A_CONTESTED] || nodePoint->banners[BANNER_H_CONTESTED])
706 return nodePoint->last_entry;
