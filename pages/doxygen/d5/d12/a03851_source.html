131 void UpdateAI(uint32 diff) override
132 {
133 if (!UpdateVictim())
134 return;
135
136 events.Update(diff);
137 if (me->HasUnitState(UNIT_STATE_CASTING))
138 return;
139
140 switch (events.ExecuteEvent())
141 {
142 case EVENT_SPELL_BERSERK:
143 Talk(SAY_ENRAGE);
144 me->CastSpell(me, SPELL_BERSERK, true);
145 break;
146 case EVENT_SPELL_BLOOD_BOIL:
147 me->CastCustomSpell(SPELL_BLOODBOIL, SPELLVALUE_MAX_TARGETS, 5, me, false);
148 events.ScheduleEvent(EVENT_SPELL_BLOOD_BOIL, 10000);
149 break;
150 case EVENT_SPELL_BEWILDERING_STRIKE:
151 me->CastSpell(me->GetVictim(), SPELL_BEWILDERING_STRIKE, false);
152 events.ScheduleEvent(EVENT_SPELL_BEWILDERING_STRIKE, 30000, GROUP_DELAY);
153 break;
154 case EVENT_SPELL_FEL_ACID_BREATH:
155 me->CastSpell(me->GetVictim(), me->HasAura(SPELL_FEL_RAGE_SELF) ? SPELL_FEL_ACID_BREATH2 : SPELL_FEL_ACID_BREATH1, false);
156 events.ScheduleEvent(EVENT_SPELL_FEL_ACID_BREATH, 30000);
157 break;
158 case EVENT_SPELL_EJECT:
159 me->CastSpell(me->GetVictim(), me->HasAura(SPELL_FEL_RAGE_SELF) ? SPELL_EJECT2 : SPELL_EJECT1, false);
160 events.ScheduleEvent(EVENT_SPELL_EJECT, 20000);
161 break;
162 case EVENT_SPELL_ARCING_SMASH:
163 me->CastSpell(me->GetVictim(), me->HasAura(SPELL_FEL_RAGE_SELF) ? SPELL_ARCING_SMASH2 : SPELL_ARCING_SMASH1, false);
164 events.ScheduleEvent(EVENT_SPELL_ARCING_SMASH, 15000);
165 break;
166 case EVENT_SPELL_FEL_GEYSER:
167 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 40.0f, true))
168 {
169 me->RemoveAurasByType(SPELL_AURA_MOD_TAUNT);
170 me->CastSpell(me, SPELL_FEL_RAGE_SELF, true);
171 me->CastSpell(target, SPELL_FEL_RAGE_TARGET, true);
172 me->CastSpell(target, SPELL_FEL_RAGE_2, true);
173 me->CastSpell(target, SPELL_FEL_RAGE_3, true);
174 me->CastSpell(target, SPELL_FEL_RAGE_SIZE, true);
175 target->CastSpell(me, SPELL_TAUNT_GURTOGG, true);
176
177 me->CastSpell(target, SPELL_FEL_GEYSER_SUMMON, true);
178 me->CastSpell(me, SPELL_FEL_GEYSER_STUN, true);
179 me->CastSpell(me, SPELL_INSIGNIFICANCE, true);
180 events.ScheduleEvent(EVENT_SPELL_CHARGE, 2000);
181 events.DelayEvents(30000, GROUP_DELAY);
182 }
183 events.ScheduleEvent(EVENT_SPELL_FEL_GEYSER, 90000);
184 break;
185 case EVENT_SPELL_CHARGE:
186 me->CastSpell(me->GetVictim(), SPELL_CHARGE, true);
187 break;
188 }
189
190 DoMeleeAttackIfReady();
191 }
192
193 bool CheckEvadeIfOutOfCombatArea() const override

195 return me->GetHomePosition().GetExactDist2d(me) > 105.0f;
196 }
