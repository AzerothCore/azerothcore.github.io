43void FollowerAI::AttackStart(Unit* who)
44{
45 if (!who)
46 return;
47
48 if (me->Attack(who, true))
49 {
50 // This is done in Unit::Attack function which wont bug npcs by not adding threat upon combat start...
51 //me->AddThreat(who, 0.0f);
52 //me->SetInCombatWith(who);
53 //who->SetInCombatWith(me);
54
55 if (me->HasUnitState(UNIT_STATE_FOLLOW))
56 me->ClearUnitState(UNIT_STATE_FOLLOW);
57
58 if (IsCombatMovementAllowed())
59 me->GetMotionMaster()->MoveChase(who);
60 }
61}

63//This part provides assistance to a player that are attacked by who, even if out of normal aggro range
64//It will cause me to attack who that are attacking _any_ player (which has been confirmed may happen also on offi)
65//The flag (type_flag) is unconfirmed, but used here for further research and is a good candidate.
66bool FollowerAI::AssistPlayerInCombatAgainst(Unit* who)
67{
68 if (!who || !who->GetVictim())
69 return false;
70
71 //experimental (unknown) flag not present
72 if (!(me->GetCreatureTemplate()->type_flags & CREATURE_TYPE_FLAG_CAN_ASSIST))
73 return false;
74
75 //not a player
76 if (!who->GetVictim()->GetCharmerOrOwnerPlayerOrPlayerItself())
77 return false;
78
79 //never attack friendly
80 if (me->IsFriendlyTo(who))
81 return false;
82
83 //too far away and no free sight?
84 if (me->IsWithinDistInMap(who, MAX_PLAYER_DISTANCE) && me->IsWithinLOSInMap(who))
85 {
86 AttackStart(who);
87 return true;
88 }
89
90 return false;
91}

93void FollowerAI::MoveInLineOfSight(Unit* who)
94{
95 if (me->GetVictim())
96 return;
97
98 if (!me->HasUnitState(UNIT_STATE_STUNNED) && who->isTargetableForAttack(true, me) && who->isInAccessiblePlaceFor(me))
99 if (HasFollowState(STATE_FOLLOW_INPROGRESS) && AssistPlayerInCombatAgainst(who))
100 return;
101
102 if (me->CanStartAttack(who))
103 {
104 if (me->HasUnitState(UNIT_STATE_DISTRACTED))
105 {
106 me->ClearUnitState(UNIT_STATE_DISTRACTED);
107 me->GetMotionMaster()->Clear();
108 }
109 AttackStart(who);
110 }

113void FollowerAI::JustDied(Unit* /*pKiller*/)
114{
115 if (!HasFollowState(STATE_FOLLOW_INPROGRESS) || !m_uiLeaderGUID || !m_pQuestForFollow)
116 return;
117
118 //TODO: need a better check for quests with time limit.
119 if (Player* player = GetLeaderForFollower())
120 {

180void FollowerAI::UpdateAI(uint32 uiDiff)
181{
182 if (HasFollowState(STATE_FOLLOW_INPROGRESS) && !me->GetVictim())
183 {
184 if (m_uiUpdateFollowTimer <= uiDiff)
185 {
186 if (HasFollowState(STATE_FOLLOW_COMPLETE) && !HasFollowState(STATE_FOLLOW_POSTEVENT))
187 {
188 LOG_DEBUG("scripts.ai", "FollowerAI is set completed, despawns.");
189 me->DespawnOrUnsummon();
190 return;
191 }
192
193 bool bIsMaxRangeExceeded = true;
194
195 if (Player* player = GetLeaderForFollower())
196 {
197 if (HasFollowState(STATE_FOLLOW_RETURNING))
198 {
199 LOG_DEBUG("scripts.ai", "FollowerAI is returning to leader.");
200
201 RemoveFollowState(STATE_FOLLOW_RETURNING);
202 me->GetMotionMaster()->MoveFollow(player, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
203 return;
204 }
205
206 if (Group* group = player->GetGroup())
207 {
208 for (GroupReference* groupRef = group->GetFirstMember(); groupRef != nullptr; groupRef = groupRef->next())
209 {
210 Player* member = groupRef->GetSource();
211
212 if (member && me->IsWithinDistInMap(member, MAX_PLAYER_DISTANCE, true, false))
213 {
214 bIsMaxRangeExceeded = false;
215 break;
216 }
217 }
218 }
219 else
220 {
221 if (me->IsWithinDistInMap(player, MAX_PLAYER_DISTANCE, true, false))
222 bIsMaxRangeExceeded = false;
223 }
224 }
225
226 if (bIsMaxRangeExceeded)
227 {
228 LOG_DEBUG("scripts.ai", "FollowerAI failed because player/group was to far away or not found");
229 me->DespawnOrUnsummon();
230 return;
231 }
232
233 m_uiUpdateFollowTimer = 1000;
234 }
235 else
236 m_uiUpdateFollowTimer -= uiDiff;
237 }
238
239 UpdateFollowerAI(uiDiff);
240}

267void FollowerAI::StartFollow(Player* player, uint32 factionForFollower, const Quest* quest)
268{
269 if (me->GetVictim())
270 {
271 LOG_DEBUG("scripts.ai", "FollowerAI attempt to StartFollow while in combat.");
272 return;
273 }
274
275 if (HasFollowState(STATE_FOLLOW_INPROGRESS))
276 {
277 LOG_ERROR("entities.unit.ai", "FollowerAI attempt to StartFollow while already following.");
278 return;
279 }
280
281 //set variables
282 m_uiLeaderGUID = player->GetGUID();
283
284 if (factionForFollower)
285 me->SetFaction(factionForFollower);
286
287 m_pQuestForFollow = quest;
288
289 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == WAYPOINT_MOTION_TYPE)
290 {
291 me->GetMotionMaster()->Clear();
292 me->GetMotionMaster()->MoveIdle();
293 LOG_DEBUG("scripts.ai", "FollowerAI start with WAYPOINT_MOTION_TYPE, set to MoveIdle.");
294 }
295
296 me->ReplaceAllNpcFlags(UNIT_NPC_FLAG_NONE);
297
298 AddFollowState(STATE_FOLLOW_INPROGRESS);
299
300 me->GetMotionMaster()->MoveFollow(player, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
301
302 LOG_DEBUG("scripts.ai", "FollowerAI start follow {} ({})", player->GetName(), m_uiLeaderGUID.ToString());
303}

305Player* FollowerAI::GetLeaderForFollower()
306{
307 if (Player* player = ObjectAccessor::GetPlayer(*me, m_uiLeaderGUID))
308 {
309 if (player->IsAlive())
310 return player;
311 else
312 {
313 if (Group* group = player->GetGroup())
314 {
315 for (GroupReference* groupRef = group->GetFirstMember(); groupRef != nullptr; groupRef = groupRef->next())
316 {
317 Player* member = groupRef->GetSource();
318
319 if (member && me->IsWithinDistInMap(member, MAX_PLAYER_DISTANCE) && member->IsAlive())
320 {
321 LOG_DEBUG("scripts.ai", "FollowerAI GetLeader changed and returned new leader.");
322 m_uiLeaderGUID = member->GetGUID();
323 return member;
324 }
325 }
326 }
327 }
328 }
329
330 LOG_DEBUG("scripts.ai", "FollowerAI GetLeader can not find suitable leader.");
331 return nullptr;
332}

335{

356void FollowerAI::SetFollowPaused(bool paused)

359 return;
360
