74 boss_elder_nadox(Creature* creature) : BossAI(creature, DATA_PRINCE_TALDARAM),

80 void Reset() override
81 {
82 _Reset();
83
84 // Clear eggs data
85 swarmEggs.clear();
86 guardianEggs.clear();
87 previousSwarmEgg_GUID.Clear();
88 guardianSummoned = false;
89 respectYourElders = true;
90 }
91
92 void EnterCombat(Unit * /*who*/) override
93 {
94 _EnterCombat();
95 Talk(SAY_AGGRO);
96
97 events.ScheduleEvent(EVENT_SWARMER, 10000);
98 events.ScheduleEvent(EVENT_CHECK_HOME, 2000);
99 events.ScheduleEvent(EVENT_PLAGUE, urand(5000, 8000));
100
101 if (IsHeroic())
102 {
103 events.ScheduleEvent(EVENT_BROOD_RAGE, 5000);
104 }
105
106 // Cache eggs
107 std::list<Creature*> eggs;
108 // Swarm eggs
109 me->GetCreatureListWithEntryInGrid(eggs, NPC_AHNKAHAR_SWARM_EGG, 250.0f);
110 if (!eggs.empty())
111 {
112 for (Creature* const egg : eggs)
113 {
114 if (egg)
115 {
116 swarmEggs.push_back(egg->GetGUID());
117 }
118 }
119 }
120
121 eggs.clear();
122
123 // Guardian eggs
124 me->GetCreatureListWithEntryInGrid(eggs, NPC_AHNKAHAR_GUARDIAN_EGG, 250.0f);
125 if (!eggs.empty())
126 {
127 for (Creature* const egg : eggs)
128 {
129 if (egg)
130 {
131 guardianEggs.push_back(egg->GetGUID());
132 }
133 }
134 }
135 }
136
137 void SummonedCreatureDies(Creature* summon, Unit* /*killer*/) override
138 {
139 if (summon->GetEntry() == NPC_AHNKAHAR_GUARDIAN)
140 {
141 respectYourElders = false;
142 }
143 }
144
145 void KilledUnit(Unit* victim) override

147 if (victim->GetTypeId() == TYPEID_PLAYER)

149 Talk(SAY_SLAY);

159 void DamageTaken(Unit* /*attacker*/, uint32& damage, DamageEffectType /*damageType*/, SpellSchoolMask /*school*/) override
160 {
161 if (!guardianSummoned && me->HealthBelowPctDamaged(55, damage))
162 {
163 SummonHelpers(false);
164 guardianSummoned = true;
165 }
166 }

168 uint32 GetData(uint32 type) const override
169 {
170 if (type == DATA_RESPECT_YOUR_ELDERS)
171 {
172 return respectYourElders ? 1 : 0;
173 }
174
175 return 0;
176 }

178 void UpdateAI(uint32 diff) override
179 {
180 if (!UpdateVictim())
181 {
182 return;
183 }
184
185 events.Update(diff);
186
187 if (me->HasUnitState(UNIT_STATE_CASTING))
188 {
189 return;
190 }
191
192 while (uint32 const eventId = events.ExecuteEvent())
193 {
194 switch (eventId)
195 {
196 case EVENT_BROOD_RAGE:
197 {
198 if (Creature* pSwarmer = me->FindNearestCreature(NPC_AHNKAHAR_SWARMER, 40, true))
199 DoCast(pSwarmer, SPELL_BROOD_RAGE_H, true);
200
201 events.RepeatEvent(10000);
202 break;
203 }
204 case EVENT_PLAGUE:
205 {
206 DoCastVictim(SPELL_BROOD_PLAGUE, false);
207 events.RepeatEvent(urand(12000, 17000));
208 break;
209 }
210 case EVENT_SWARMER:
211 {
212 SummonHelpers(true);
213 events.RepeatEvent(10000);
214 break;
215 }
216 case EVENT_CHECK_HOME:
217 {
218 if (!me->HasAura(SPELL_ENRAGE) && (me->GetPositionZ() < 24.0f || !me->GetHomePosition().IsInDist(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 110.0f)))
219 {
220 DoCastSelf(SPELL_ENRAGE, true);
221 }
222 break;
223 }
224 }
225 }
226
227 DoMeleeAttackIfReady();
228 }
229
230private:
231 GuidList swarmEggs;
232 GuidList guardianEggs;
233 ObjectGuid previousSwarmEgg_GUID; // This will prevent casting summoning spells on same egg twice
234 bool guardianSummoned;
235 bool respectYourElders;
236
237 void SummonHelpers(bool swarm)
238 {
239 if (swarm)
240 {
241 if (swarmEggs.empty())
242 {
243 return;
244 }
245
246 // Make a copy of guid list
247 GuidList swarmEggs2 = swarmEggs;
248
249 // Remove previous egg
250 if (previousSwarmEgg_GUID)
251 {
252 std::list<ObjectGuid>::iterator itr = std::find(swarmEggs2.begin(), swarmEggs2.end(), previousSwarmEgg_GUID);
253 if (itr != swarmEggs2.end())
254 {
255 swarmEggs2.erase(itr);
256 }
257 }
258
259 if (swarmEggs2.empty())
260 {
261 return;
262 }
263
264 previousSwarmEgg_GUID = Acore::Containers::SelectRandomContainerElement(swarmEggs2);
265
266 if (Creature* egg = ObjectAccessor::GetCreature(*me, previousSwarmEgg_GUID))
267 {
268 egg->CastSpell(egg, SPELL_SUMMON_SWARMERS, true, nullptr, nullptr, me->GetGUID());
269 }
270
271 if (roll_chance_f(33))
272 {
273 Talk(SAY_EGG_SAC);
274 }
275 }
276 else
277 {
278 if (guardianEggs.empty())
279 {
280 return;
281 }
282
283 ObjectGuid const& guardianEggGUID = Acore::Containers::SelectRandomContainerElement(guardianEggs);
284 if (Creature* egg = ObjectAccessor::GetCreature(*me, guardianEggGUID))
285 {
286 egg->CastSpell(egg, SPELL_SUMMON_SWARM_GUARD, true, nullptr, nullptr, me->GetGUID());
287 }
288
289 Talk(EMOTE_HATCHES, me);
290
291 if (roll_chance_f(33))
292 {
293 Talk(SAY_EGG_SAC);
294 }
295 }
296 }
