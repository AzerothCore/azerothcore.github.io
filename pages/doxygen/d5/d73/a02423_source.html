70 void UpdateAI(uint32 diff) override
71 {
72 //Return since we have no target
73 if (!UpdateVictim())
74 {
75 return;
76 }
77 events.Update(diff);
78
79 if (me->HasUnitState(UNIT_STATE_CASTING))
80 {
81 return;
82 }
83 while (uint32 eventId = events.ExecuteEvent())
84 {
85 switch (eventId)
86 {
87 case SPELL_SHADOWBOLT_VOLLEY:
88 DoCastVictim(SPELL_SHADOWBOLT_VOLLEY);
89 events.ScheduleEvent(SPELL_SHADOWBOLT_VOLLEY, urand(TIMER_SHADOWBOLT_VOLLEY - 2000, TIMER_SHADOWBOLT_VOLLEY + 2000));
90 break;
91 case SPELL_REND:
92 DoCastVictim(SPELL_REND);
93 events.ScheduleEvent(SPELL_REND, urand(TIMER_REND - 2000, TIMER_REND + 2000));
94 break;
95 case SPELL_SHIELD:
96 SpellShieldReady = true;
97 break;
98 default:
99 break;
100 }
101 }
102 DoMeleeAttackIfReady();
103 }
