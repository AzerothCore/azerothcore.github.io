170 events.Update(diff);
171 switch (events.ExecuteEvent())
172 {
173 case EVENT_SPELL_SHADOWBOLT:
174 me->CastSpell(me, SPELL_SHADOWBOLT_VOLLEY, false);
175 events.RepeatEvent(urand(15000, 30000));
176 break;
177 case EVENT_SPELL_BANISH:
178 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 30, false))
179 me->CastSpell(target, SPELL_BANISH, false);
180 events.RepeatEvent(16000);
181 break;
182 case EVENT_SUMMON_TRAVELER:
183 spawnVoidTraveler();
184 events.RepeatEvent(HealthBelowPct(20) ? 5000 : 10000);
185 break;
186 case EVENT_SPELL_DRAWSHADOWS:
187 {
188 Map* map = me->GetMap();
189 Map::PlayerList const& PlayerList = map->GetPlayers();
190 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
191 if (Player* player = i->GetSource())
192 if (player->IsAlive() && !player->HasAura(SPELL_BANISH))
193 player->TeleportTo(me->GetMapId(), VorpilPosition[0], VorpilPosition[1], VorpilPosition[2], 0, TELE_TO_NOT_LEAVE_COMBAT);
194
195 me->NearTeleportTo(VorpilPosition[0], VorpilPosition[1], VorpilPosition[2], 0.0f);
196 me->CastSpell(me, SPELL_DRAW_SHADOWS, true);
197 me->CastSpell(me, SPELL_RAIN_OF_FIRE_N);
198
199 events.RepeatEvent(24000);
200 events.DelayEvents(6000);
201 break;
202 }
203 }
204
205 DoMeleeAttackIfReady();
206 }
207 };

237 void UpdateAI(uint32 diff) override
238 {

240 if (moveTimer >= 1000)
241 {
242 moveTimer = 0;
243 Creature* Vorpil = ObjectAccessor::GetCreature(*me, VorpilGUID);
244 if (!Vorpil)

249 me->GetMotionMaster()->MoveFollow(Vorpil, 0.0f, 0.0f);
250
251 if (sacrificed)
252 {
253 me->AddAura(DUNGEON_MODE(SPELL_EMPOWERING_SHADOWS_N, SPELL_EMPOWERING_SHADOWS_H), Vorpil);
254 Vorpil->ModifyHealth(int32(Vorpil->CountPctFromMaxHealth(4)));
255 me->CastSpell(me, SPELL_SHADOW_NOVA, true);
256 Unit::Kill(me, me);
257 return;
258 }
259
260 if (me->IsWithinDist(Vorpil, 3.0f))
261 {
262 me->CastSpell(me, SPELL_SACRIFICE, false);
263 sacrificed = true;
264 moveTimer = 500;
265 }
266 }
