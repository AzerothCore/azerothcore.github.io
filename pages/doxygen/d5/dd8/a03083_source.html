143 void DoAction(int32 action) override
144 {
145 if (action == ACTION_CONSUME && !_dying)
146 {
147 _isEating = true;
148 me->SetSpeed(MOVE_RUN, 45.f / 7.f); // From sniffs
149 me->SetReactState(REACT_PASSIVE);
150 _scheduler.DelayAll(6s);
151 }
152
153 if (action == ACTION_EXPLODE && _dying)
154 {
155 DoCastSelf(SPELL_BLOODY_DEATH);
156 _dying = false;
157 }

160 void MovementInform(uint32 type, uint32 id) override
161 {
162 if (type != POINT_MOTION_TYPE || id != POINT_CONSUME)
163 return;
164
165 me->GetMotionMaster()->MoveIdle();
166 me->SetSpeed(MOVE_RUN, 15.f/7.f); // From sniffs
167 DoCastSelf(SPELL_FULL_HEAL, true);
168 if (me->GetThreatMgr().GetThreatListSize())
169 DoResetThreat();
170 if (Creature* dying = instance->GetCreature(_creatureDying))
171 {
172 dying->AI()->DoAction(ACTION_EXPLODE);
173 me->SetTarget(dying->GetGUID());

175
176 _scheduler.Schedule(2s, [this](TaskContext /*context*/)
177 {
178 me->SetReactState(REACT_AGGRESSIVE);
179 _isEating = false;

223 void DamageTaken(Unit* who, uint32& damage, DamageEffectType, SpellSchoolMask) override
224 {
225 if (_dying && who->GetGUID() != me->GetGUID())
226 damage = 0;
227
228 if (me->HealthBelowPctDamaged(0, damage) && instance->GetData(DATA_BUG_TRIO_DEATH) < 2 && !_dying)
229 {
230 damage = 0;
231 if (_isEating)
232 return;
233
234 _scheduler.CancelAll();
235 me->SetReactState(REACT_PASSIVE);
236 _dying = true;
237 float x, y, z;
238 // Move the other bugs to this bug position
239 if (Creature* vem = instance->GetCreature(DATA_VEM))
240 {
241 if (vem->GetGUID() != me->GetGUID())
242 {
243 if (vem->IsAlive())
244 {
245 vem->AI()->DoAction(ACTION_CONSUME);
246 me->GetRandomContactPoint(vem, x, y, z);
247 vem->GetMotionMaster()->MovePoint(POINT_CONSUME, x, y, z);
248 }
249 }
250 else _creatureDying = DATA_VEM;

281 void DoFinalSpell()
282 {
283 switch (me->GetEntry())
284 {
285 case NPC_KRI:
286 DoCastSelf(SPELL_POISON_CLOUD, true);
287 break;
288 case NPC_VEM:
289 DoCastSelf(SPELL_VENGEANCE, true);
290 break;
291 case NPC_YAUJ:
292 for (uint8 i = 0; i < 10; ++i)
293 {
294 Position randomPos;
295 me->GetRandomContactPoint(me, randomPos.m_positionX, randomPos.m_positionY, randomPos.m_positionZ);
296 if (Creature* summon = me->SummonCreature(NPC_YAUJ_BROOD, randomPos))
297 DoZoneInCombat(summon);
298 }
299 break;
300 default:
301 break;
302 }
303 }

305 void JustDied(Unit* killer) override
306 {
307 instance->SetData(DATA_BUG_TRIO_DEATH, 1);
308 if (instance->GetData(DATA_BUG_TRIO_DEATH) < 3)
309 {
310 me->RemoveDynamicFlag(UNIT_DYNFLAG_LOOTABLE);
311 DoFinalSpell();
312 Talk(EMOTE_DEVOURED);
313 me->DespawnOrUnsummon(3000);
314 return;
315 }
316
317 BossAI::JustDied(killer);
318 }
