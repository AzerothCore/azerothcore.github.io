226public:
227 boss_isle_of_conquest() : CreatureScript("boss_isle_of_conquest") {}
228
229 struct boss_isle_of_conquestAI : public ScriptedAI
230 {
231 boss_isle_of_conquestAI(Creature* creature) : ScriptedAI(creature) { }
232
233 EventMap events;
234 bool rage;
235 void Reset() override
236 {
237 events.Reset();
238 rage = false;
239 }
240
241 void CheckRageBuff()
242 {
243 if (!rage)
244 {
245 if (me->GetDistance(me->GetHomePosition()) > 40.0f)
246 {
247 rage = true;
248 me->CastSpell(me, SPELL_IOCBOSS_RAGE, true);
249 }
250 }
251 else
252 {
253 if (me->GetDistance(me->GetHomePosition()) < 40.0f && std::abs(me->GetPositionZ() - me->GetHomePosition().GetPositionZ()) < 5.0f)
254 {
255 rage = false;
256 me->RemoveAurasDueToSpell(SPELL_IOCBOSS_RAGE);
257 }
258 }
259 }
260
261 void EnterCombat(Unit* /*who*/) override
262 {
263 events.ScheduleEvent(EVENT_CHECK_RAGE, 2000);
264 events.ScheduleEvent(EVENT_BRUTAL_STRIKE, 6000);
265 events.ScheduleEvent(EVENT_CRUSHING_LEAP, 22000);
266 events.ScheduleEvent(EVENT_DAGGER_THROW, 10000);
267 }
268
269 void UpdateAI(uint32 diff) override
270 {
271 if (!UpdateVictim())
272 return;
273
274 events.Update(diff);
275 if (me->HasUnitState(UNIT_STATE_CASTING))
276 return;
277
278 switch (events.ExecuteEvent())
279 {
280 case EVENT_CHECK_RAGE:
281 CheckRageBuff();
282 events.RepeatEvent(2000);
283 break;
284 case EVENT_BRUTAL_STRIKE:
285 me->CastSpell(me->GetVictim(), SPELL_IOCBOSS_BRUTAL_STRIKE, false);
286 events.RepeatEvent(6000);
287 break;
288 case EVENT_CRUSHING_LEAP:
289 me->CastSpell(me, SPELL_IOCBOSS_CRUSHING_LEAP, false);
290 events.RepeatEvent(22000);
291 break;
292 case EVENT_DAGGER_THROW:
293 if (Unit* tgt = SelectTarget(SelectTargetMethod::Random))
294 me->CastSpell(tgt, SPELL_IOCBOSS_DAGGER_THROW, false);
295
296 events.RepeatEvent(10000);
297 break;
298 }
299
300 DoMeleeAttackIfReady();
301 }
302 };
303
304 CreatureAI* GetAI(Creature* creature) const override
305 {
306 return new boss_isle_of_conquestAI(creature);
307 }
