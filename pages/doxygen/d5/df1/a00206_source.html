27/*static*/ uint32 Acore::Crypto::TOTP::GenerateToken(Secret const& secret, time_t timestamp)
28{
29 timestamp /= TOTP_INTERVAL;
30 unsigned char challenge[8];
31
32 for (int i = 8; i--; timestamp >>= 8)
33 {
34 challenge[i] = timestamp;
35 }
36
37 unsigned char digest[HMAC_RESULT_SIZE];
38 uint32 digestSize = HMAC_RESULT_SIZE;
39 HMAC(EVP_sha1(), secret.data(), secret.size(), challenge, 8, digest, &digestSize);
40
41 uint32 offset = digest[19] & 0xF;
42 uint32 truncated = (digest[offset] << 24) | (digest[offset + 1] << 16) | (digest[offset + 2] << 8) | (digest[offset + 3]);
43 truncated &= 0x7FFFFFFF;
44
45 return (truncated % 1000000);
46}

48/*static*/ bool Acore::Crypto::TOTP::ValidateToken(Secret const& secret, uint32 token)
49{
50 time_t now = GetEpochTime().count();
51 return (
52 (token == GenerateToken(secret, now - TOTP_INTERVAL)) ||
53 (token == GenerateToken(secret, now)) ||
54 (token == GenerateToken(secret, now + TOTP_INTERVAL))
55 );
