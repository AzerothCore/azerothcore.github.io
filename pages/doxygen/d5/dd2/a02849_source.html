81 bool CanAIAttack(Unit const* target) const override
82 {
83 if (me->GetThreatMgr().GetThreatListSize() > 1 && me->GetThreatMgr().GetOnlineContainer().getMostHated()->getTarget() == target)
84 return !target->HasAura(SPELL_SLEEP);
85
86 return true;
87 }
88
89 void UpdateAI(uint32 diff) override
90 {
91 if (!UpdateVictim())
92 return;
93
94 events.Update(diff);
95
96 if (me->HasUnitState(UNIT_STATE_CASTING))
97 return;
98
99 while (uint32 eventId = events.ExecuteEvent())
100 {
101 switch (eventId)
102 {
103 case EVENT_SLEEP:
104 DoCastVictim(SPELL_SLEEP, true);
105 events.ScheduleEvent(EVENT_SLEEP, 24s, 32s);
106 return;
107 case EVENT_EARTH_SHOCK:
108 DoCastVictim(SPELL_EARTH_SHOCK);
109 events.ScheduleEvent(EVENT_EARTH_SHOCK, 8s, 18s);
110 break;
111 case EVENT_CHAIN_BURN:
112 if (me->GetPowerPct(POWER_MANA) > 5.f) // totally guessed
113 {
114 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, [&](Unit* u) { return u && !u->IsPet() && u->getPowerType() == POWER_MANA; }))
115 {
116 DoCast(target, SPELL_CHAIN_BURN);
117 }
118 }
119 events.ScheduleEvent(EVENT_CHAIN_BURN, 12s, 28s);
120 break;
121 case EVENT_ILLUSIONS:
122 DoCastSelf(SPELL_SUMMON_NIGHTMARE_ILLUSION_LEFT, true);
123 DoCastSelf(SPELL_SUMMON_NIGHTMARE_ILLUSION_BACK, true);
124 DoCastSelf(SPELL_SUMMON_NIGHTMARE_ILLUSION_RIGHT, true);
125 events.ScheduleEvent(EVENT_ILLUSIONS, 15s, 25s);
126 break;
127 default:
128 break;
129 }
130 }
131
132 DoMeleeAttackIfReady();
133 }
134};
