40void UnitAI::DoMeleeAttackIfReady()
41{
42 if (me->HasUnitState(UNIT_STATE_CASTING))
43 return;
44
45 Unit* victim = me->GetVictim();
46 if (!victim || !victim->IsInWorld())
47 return;
48
49 if (!me->IsWithinMeleeRange(victim))
50 return;
51
52 //Make sure our attack is ready and we aren't currently casting before checking distance
53 if (me->isAttackReady())
54 {
55 // xinef: prevent base and off attack in same time, delay attack at 0.2 sec
56 if (me->haveOffhandWeapon())
57 if (me->getAttackTimer(OFF_ATTACK) < ATTACK_DISPLAY_DELAY)
58 me->setAttackTimer(OFF_ATTACK, ATTACK_DISPLAY_DELAY);
59
60 me->AttackerStateUpdate(victim);
61 me->resetAttackTimer();
62 }
63
64 if (me->haveOffhandWeapon() && me->isAttackReady(OFF_ATTACK))
65 {
66 // xinef: delay main hand attack if both will hit at the same time (players code)
67 if (me->getAttackTimer(BASE_ATTACK) < ATTACK_DISPLAY_DELAY)
68 me->setAttackTimer(BASE_ATTACK, ATTACK_DISPLAY_DELAY);
69
70 me->AttackerStateUpdate(victim, OFF_ATTACK);
71 me->resetAttackTimer(OFF_ATTACK);
72 }
73}

178SpellCastResult UnitAI::DoCast(uint32 spellId)
179{
180 Unit* target = nullptr;
181
182 switch (AISpellInfo[spellId].target)
183 {
184 default:
185 case AITARGET_SELF:
186 target = me;
187 break;
188 case AITARGET_VICTIM:
189 target = me->GetVictim();
190 break;
191 case AITARGET_ENEMY:
192 {
193 if (SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId))
194 {
195 bool playerOnly = spellInfo->HasAttribute(SPELL_ATTR3_ONLY_ON_PLAYER);
196 target = SelectTarget(SelectTargetMethod::Random, 0, spellInfo->GetMaxRange(false), playerOnly);
197 }
198 break;
199 }
200 case AITARGET_ALLY:
201 target = me;
202 break;
203 case AITARGET_BUFF:
204 target = me;
205 break;
206 case AITARGET_DEBUFF:
207 {
208 if (SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId))
209 {
210 bool playerOnly = spellInfo->HasAttribute(SPELL_ATTR3_ONLY_ON_PLAYER);
211 float range = spellInfo->GetMaxRange(false);
212
213 DefaultTargetSelector targetSelector(me, range, playerOnly, -(int32)spellId);
214 if (!(spellInfo->AuraInterruptFlags & AURA_INTERRUPT_FLAG_NOT_VICTIM)
215 && targetSelector(me->GetVictim()))
216 target = me->GetVictim();
217 else
218 target = SelectTarget(SelectTargetMethod::Random, 0, targetSelector);
219 }
220 break;
221 }
222 }
223
224 if (target)
225 me->CastSpell(target, spellId, false);
226
227 return SPELL_FAILED_BAD_TARGETS;
228}

269void UnitAI::FillAISpellInfo()
270{
271 AISpellInfo = new AISpellInfoType[sSpellMgr->GetSpellInfoStoreSize()];
272
273 AISpellInfoType* AIInfo = AISpellInfo;
274 SpellInfo const* spellInfo;
275
276 for (uint32 i = 0; i < sSpellMgr->GetSpellInfoStoreSize(); ++i, ++AIInfo)
277 {
278 spellInfo = sSpellMgr->GetSpellInfo(i);
279 if (!spellInfo)
280 continue;
281
282 if (spellInfo->HasAttribute(SPELL_ATTR0_ALLOW_CAST_WHILE_DEAD))
283 AIInfo->condition = AICOND_DIE;
284 else if (spellInfo->IsPassive() || spellInfo->GetDuration() == -1)
285 AIInfo->condition = AICOND_AGGRO;
286 else
287 AIInfo->condition = AICOND_COMBAT;
288
289 if (AIInfo->cooldown < spellInfo->RecoveryTime)
290 AIInfo->cooldown = spellInfo->RecoveryTime;
291
292 if (!spellInfo->GetMaxRange(false))
293 UPDATE_TARGET(AITARGET_SELF)
294 else
295 {
296 for (uint32 j = 0; j < MAX_SPELL_EFFECTS; ++j)
297 {
298 uint32 targetType = spellInfo->Effects[j].TargetA.GetTarget();
299
300 if (targetType == TARGET_UNIT_TARGET_ENEMY
301 || targetType == TARGET_DEST_TARGET_ENEMY)
302 UPDATE_TARGET(AITARGET_VICTIM)
303 else if (targetType == TARGET_UNIT_DEST_AREA_ENEMY)
304 UPDATE_TARGET(AITARGET_ENEMY)
305
306 if (spellInfo->Effects[j].Effect == SPELL_EFFECT_APPLY_AURA)
307 {
308 if (targetType == TARGET_UNIT_TARGET_ENEMY)
309 UPDATE_TARGET(AITARGET_DEBUFF)
310 else if (spellInfo->IsPositive())
311 UPDATE_TARGET(AITARGET_BUFF)
312 }
313 }
314 }
315 AIInfo->realCooldown = spellInfo->RecoveryTime + spellInfo->StartRecoveryTime;
316 AIInfo->maxRange = spellInfo->GetMaxRange(false) * 3 / 4;
317 }
318}

326void SimpleCharmedAI::UpdateAI(uint32 /*diff*/)
327{
328 Creature* charmer = me->GetCharmer()->ToCreature();
329
330 //kill self if charm aura has infinite duration
331 if (charmer->IsInEvadeMode())
332 {
333 Unit::AuraEffectList const& auras = me->GetAuraEffectsByType(SPELL_AURA_MOD_CHARM);
334 for (Unit::AuraEffectList::const_iterator iter = auras.begin(); iter != auras.end(); ++iter)
335 if ((*iter)->GetCasterGUID() == charmer->GetGUID() && (*iter)->GetBase()->IsPermanent())
336 {
337 Unit::Kill(charmer, me);
338 return;
339 }
340 }
341
342 if (!charmer->IsInCombat())
343 me->GetMotionMaster()->MoveFollow(charmer, PET_FOLLOW_DIST, me->GetFollowAngle());
344
345 Unit* target = me->GetVictim();
346 if (!target || !charmer->IsValidAttackTarget(target))
347 AttackStart(charmer->SelectNearestTargetInAttackDistance(ATTACK_DISTANCE));
348}

356bool SpellTargetSelector::operator()(Unit const* target) const
357{
358 if (!target)
359 return false;
360
361 if (_spellInfo->CheckTarget(_caster, target) != SPELL_CAST_OK)
362 return false;
363
364 // copypasta from Spell::CheckRange
365 uint32 range_type = _spellInfo->RangeEntry ? _spellInfo->RangeEntry->Flags : 0;
366 float max_range = _caster->GetSpellMaxRangeForTarget(target, _spellInfo);
367 float min_range = _caster->GetSpellMinRangeForTarget(target, _spellInfo);
368
369 if (target && target != _caster)
370 {
371 if (range_type == SPELL_RANGE_MELEE)
372 {
373 // Because of lag, we can not check too strictly here.
374 if (!_caster->IsWithinMeleeRange(target, max_range))
375 return false;
376 }
377 else if (!_caster->IsWithinCombatRange(target, max_range))
378 return false;
379
380 if (range_type == SPELL_RANGE_RANGED)
381 {
382 if (_caster->IsWithinMeleeRange(target))
383 return false;
384 }
385 else if (min_range && _caster->IsWithinCombatRange(target, min_range)) // skip this check if min_range = 0
386 return false;
387 }
388
389 return true;
390}
