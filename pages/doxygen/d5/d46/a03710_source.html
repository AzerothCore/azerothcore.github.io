202 void SetData(uint32 type, uint32 data) override
203 {
204 switch(type)
205 {
206 case DATA_ACTIVATE_DEFENSE_SYSTEM:
207 {
208 if (data)
209 bDefensesUsed = true;
210 const Position pos = {1919.09546f, 812.29724f, 86.2905f, M_PI};
211 instance->SummonCreature(NPC_DEFENSE_SYSTEM, pos, 0, 6499);
212 }
213 break;
214 case DATA_START_INSTANCE:
215 if (EncounterStatus == NOT_STARTED)
216 {
217 EncounterStatus = IN_PROGRESS;
218 if (Creature* c = instance->GetCreature(NPC_SinclariGUID))
219 c->AI()->Talk(SAY_SINCLARI_1);
220 events.RescheduleEvent(EVENT_GUARDS_FALL_BACK, 4000);
221 }
222 break;
223 case DATA_PORTAL_DEFEATED:
224 events.RescheduleEvent(EVENT_SUMMON_PORTAL, 3000);
225 break;
226 case DATA_PORTAL_LOCATION:
227 PortalLocation = data;
228 break;
229 case DATA_DECRASE_DOOR_HEALTH:
230 if (GateHealth > 0)
231 --GateHealth;
232 if (GateHealth == 0)
233 {
234 CLEANED = false;
235 InstanceCleanup();
236 }
237 DoUpdateWorldState(WORLD_STATE_VH_PRISON_STATE, (uint32)GateHealth);
238 break;
239 case DATA_RELEASE_BOSS:
240 if (WaveCount == 6)
241 StartBossEncounter(uiFirstBoss);
242 else
243 StartBossEncounter(uiSecondBoss);
244 break;
245 case DATA_BOSS_DIED:
246 if (WaveCount == 6)
247 m_auiEncounter[0] = DONE;
248 else if (WaveCount == 12)
249 m_auiEncounter[1] = DONE;
250 else if (WaveCount == 18)
251 {
252 m_auiEncounter[2] = DONE;
253 EncounterStatus = DONE;
254 HandleGameObject(GO_MainGateGUID, true);
255 DoUpdateWorldState(WORLD_STATE_VH_SHOW, 0);
256 if (Creature* c = instance->GetCreature(NPC_SinclariGUID)) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); }
257 }
258 SaveToDB();
259 if (WaveCount < 18)
260 events.RescheduleEvent(EVENT_SUMMON_PORTAL, 35000);
261 break;
262 case DATA_FAILED:
263 CLEANED = false;
264 InstanceCleanup();
265 break;
266 case DATA_ACHIEV:
267 bAchiev = !!data;
268 break;
269 }

326 void StartBossEncounter(uint8 uiBoss)
327 {
328 Creature* pBoss = nullptr;
329
330 switch(uiBoss)
331 {
332 case BOSS_MORAGG:
333 HandleGameObject(GO_MoraggCellGUID, true);
334 pBoss = instance->GetCreature(NPC_MoraggGUID);
335 if (pBoss)
336 pBoss->GetMotionMaster()->MovePoint(0, BossStartMove1);
337 break;
338 case BOSS_EREKEM:
339 HandleGameObject(GO_ErekemCellGUID, true);
340 HandleGameObject(GO_ErekemRightGuardCellGUID, true);
341 HandleGameObject(GO_ErekemLeftGuardCellGUID, true);
342 pBoss = instance->GetCreature(NPC_ErekemGUID);
343 if (pBoss)
344 pBoss->GetMotionMaster()->MovePoint(0, BossStartMove2);
345 if (Creature* pGuard1 = instance->GetCreature(NPC_ErekemGuardGUID[0]))
346 {
347 pGuard1->RemoveUnitMovementFlag(MOVEMENTFLAG_WALKING);
348 pGuard1->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
349 pGuard1->SetImmuneToNPC(false);
350 pGuard1->GetMotionMaster()->MovePoint(0, BossStartMove21);
351 }
352 if (Creature* pGuard2 = instance->GetCreature(NPC_ErekemGuardGUID[1]))
353 {
354 pGuard2->RemoveUnitMovementFlag(MOVEMENTFLAG_WALKING);
355 pGuard2->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
356 pGuard2->SetImmuneToNPC(false);
357 pGuard2->GetMotionMaster()->MovePoint(0, BossStartMove22);
358 }
359 break;
360 case BOSS_ICHORON:
361 HandleGameObject(GO_IchoronCellGUID, true);
362 pBoss = instance->GetCreature(NPC_IchoronGUID);
363 if (pBoss)
364 pBoss->GetMotionMaster()->MovePoint(0, BossStartMove3);
365 break;
366 case BOSS_LAVANTHOR:
367 HandleGameObject(GO_LavanthorCellGUID, true);
368 pBoss = instance->GetCreature(NPC_LavanthorGUID);
369 if (pBoss)
370 pBoss->GetMotionMaster()->MovePoint(0, BossStartMove4);
371 break;
372 case BOSS_XEVOZZ:
373 HandleGameObject(GO_XevozzCellGUID, true);
374 pBoss = instance->GetCreature(NPC_XevozzGUID);
375 if (pBoss)
376 pBoss->GetMotionMaster()->MovePoint(0, BossStartMove5);

378 case BOSS_ZURAMAT:
379 HandleGameObject(GO_ZuramatCellGUID, true);

397 void Update(uint32 diff) override
398 {
399 events.Update(diff);
400 switch( events.ExecuteEvent() )
401 {
402 case 0:
403 break;
404 case EVENT_CHECK_PLAYERS:
405 {
406 if( DoNeedCleanup(false) )
407 InstanceCleanup();
408 events.RepeatEvent(CLEANUP_CHECK_INTERVAL);
409 }
410 break;
411 case EVENT_GUARDS_FALL_BACK:
412 {
413 for (uint8 i = 0; i < 4; ++i)
414 if (Creature* c = instance->GetCreature(NPC_GuardGUID[i]))
415 {
416 c->SetReactState(REACT_PASSIVE);
417 c->CombatStop();
418 c->RemoveUnitMovementFlag(MOVEMENTFLAG_WALKING);
419 c->GetMotionMaster()->MovePoint(0, guardMovePosition);
420 }
421 events.RescheduleEvent(EVENT_GUARDS_DISAPPEAR, 5000);
422 }
423 break;
424 case EVENT_GUARDS_DISAPPEAR:
425 {
426 for (uint8 i = 0; i < 4; ++i)
427 if (Creature* c = instance->GetCreature(NPC_GuardGUID[i]))
428 c->SetVisible(false);
429 events.RescheduleEvent(EVENT_SINCLARI_FALL_BACK, 2000);
430 }
431 break;
432 case EVENT_SINCLARI_FALL_BACK:
433 {
434 if (Creature* c = instance->GetCreature(NPC_SinclariGUID))
435 {
436 c->RemoveUnitMovementFlag(MOVEMENTFLAG_WALKING);
437 c->GetMotionMaster()->MovePoint(0, sinclariOutsidePosition);
438 }
439 SetData(DATA_ACTIVATE_DEFENSE_SYSTEM, 0);
440 events.RescheduleEvent(EVENT_START_ENCOUNTER, 4000);
441 }
442 break;
443 case EVENT_START_ENCOUNTER:
444 {
445 if (Creature* c = instance->GetCreature(NPC_DoorSealGUID))
446 c->RemoveAllAuras(); // just to be sure...
447 GateHealth = 100;
448 HandleGameObject(GO_MainGateGUID, false);
449 DoUpdateWorldState(WORLD_STATE_VH_SHOW, 1);
450 DoUpdateWorldState(WORLD_STATE_VH_PRISON_STATE, (uint32)GateHealth);
451 DoUpdateWorldState(WORLD_STATE_VH_WAVE_COUNT, (uint32)WaveCount);
452
453 for (ObjectGuid const& guid : GO_ActivationCrystalGUID)
454 if (GameObject* go = instance->GetGameObject(guid))
455 {
456 HandleGameObject(ObjectGuid::Empty, false, go); // not used yet
457 go->RemoveGameObjectFlag(GO_FLAG_NOT_SELECTABLE); // make it useable
458 }
459 events.RescheduleEvent(EVENT_SUMMON_PORTAL, 4000);
460 }
461 break;
462 case EVENT_SUMMON_PORTAL:
463 ++WaveCount;
464 DoUpdateWorldState(WORLD_STATE_VH_WAVE_COUNT, (uint32)WaveCount);
465 SetData(DATA_PORTAL_LOCATION, (GetData(DATA_PORTAL_LOCATION) + urand(1, 5)) % 6);
466 if (Creature* c = instance->GetCreature(NPC_SinclariGUID))
467 {
468 if (WaveCount % 6 != 0)
469 c->SummonCreature(NPC_TELEPORTATION_PORTAL, PortalLocations[GetData(DATA_PORTAL_LOCATION)], TEMPSUMMON_CORPSE_DESPAWN);
470 else if (WaveCount == 6 || WaveCount == 12) // first or second boss
471 {
472 if (!uiFirstBoss || !uiSecondBoss)
473 {
474 uiFirstBoss = urand(1, 6);
475 do { uiSecondBoss = urand(1, 6); }
476 while (uiFirstBoss == uiSecondBoss);
477 SaveToDB();
478 }
479 c->SummonCreature(NPC_TELEPORTATION_PORTAL, MiddleRoomPortalSaboLocation, TEMPSUMMON_CORPSE_DESPAWN);
480 }
481 else // cyanigossa
482 {
483 if (Creature* cyanigosa = c->SummonCreature(NPC_CYANIGOSA, CyanigosasSpawnLocation, TEMPSUMMON_DEAD_DESPAWN))
484 {
485 cyanigosa->CastSpell(cyanigosa, SPELL_CYANIGOSA_BLUE_AURA, false);
486 cyanigosa->AI()->Talk(CYANIGOSA_SAY_SPAWN);
487 cyanigosa->GetMotionMaster()->MoveJump(MiddleRoomLocation.GetPositionX(), MiddleRoomLocation.GetPositionY(), MiddleRoomLocation.GetPositionZ(), 10.0f, 20.0f);
488 }
489 events.RescheduleEvent(EVENT_CYANIGOSSA_TRANSFORM, 10000);
490 }
491 }
492 break;
493 case EVENT_CYANIGOSSA_TRANSFORM:
494 if (Creature* c = instance->GetCreature(NPC_CyanigosaGUID))
495 {
496 c->RemoveAurasDueToSpell(SPELL_CYANIGOSA_BLUE_AURA);
497 c->CastSpell(c, SPELL_CYANIGOSA_TRANSFORM, 0);
498 events.RescheduleEvent(EVENT_CYANIGOSA_ATTACK, 2500);
499 }
500 break;
501 case EVENT_CYANIGOSA_ATTACK:
502 if (Creature* c = instance->GetCreature(NPC_CyanigosaGUID))
503 {
504 c->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
505 c->SetImmuneToNPC(false);
506 }
507 break;
508 }

511 void OnPlayerEnter(Player* plr) override
512 {
513 if( DoNeedCleanup(plr->IsAlive()) )
514 InstanceCleanup();
515
516 if (EncounterStatus == IN_PROGRESS)
517 {
518 plr->SendUpdateWorldState(WORLD_STATE_VH_SHOW, 1);
519 plr->SendUpdateWorldState(WORLD_STATE_VH_PRISON_STATE, (uint32)GateHealth);
520 plr->SendUpdateWorldState(WORLD_STATE_VH_WAVE_COUNT, (uint32)WaveCount);
521 }
522 else
523 plr->SendUpdateWorldState(WORLD_STATE_VH_SHOW, 0);
524
525 events.RescheduleEvent(EVENT_CHECK_PLAYERS, CLEANUP_CHECK_INTERVAL);
526 }
527
528 bool DoNeedCleanup(bool enter)
529 {
530 uint8 aliveCount = 0;
531 Map::PlayerList const& pl = instance->GetPlayers();
532 for( Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr )
533 if( Player* plr = itr->GetSource() )
534 if( plr->IsAlive() && !plr->IsGameMaster() && !plr->HasAura(27827)/*spirit of redemption aura*/ )
535 ++aliveCount;
536
537 bool need = enter ? aliveCount <= 1 : aliveCount == 0;
538 if( !need && CLEANED )
539 CLEANED = false;
540 return need;
541 }

543 void InstanceCleanup()
544 {
545 if( CLEANED )
546 return;
547 CLEANED = true;
548
549 // reset defense crystals
550 for (ObjectGuid const& guid : GO_ActivationCrystalGUID)
551 if (GameObject* go = instance->GetGameObject(guid))
552 {
553 HandleGameObject(ObjectGuid::Empty, false, go); // not used yet
554 go->SetGameObjectFlag(GO_FLAG_NOT_SELECTABLE); // not useable at the beginning
555 }
556
557 // reset positions of Sinclari and Guards
558 if (Creature* c = instance->GetCreature(NPC_SinclariGUID)) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); }
559 for (uint8 i = 0; i < 4; ++i)
560 if (Creature* c = instance->GetCreature(NPC_GuardGUID[i]))
561 {
562 c->DespawnOrUnsummon();
563 c->SetRespawnTime(3);
564 if (m_auiEncounter[MAX_ENCOUNTER - 1] == DONE)
565 c->SetVisible(false);
566 else
567 c->SetVisible(true);
568 c->SetReactState(REACT_AGGRESSIVE);
569 }
570
571 // remove portal if any
572 if (Creature* c = instance->GetCreature(NPC_PortalGUID))
573 c->DespawnOrUnsummon();
574 NPC_PortalGUID.Clear();
575
576 // remove trash
577 for (ObjectGuid const& guid : trashMobs)
578 if (Creature* c = instance->GetCreature(guid))
579 c->DespawnOrUnsummon();
580
581 trashMobs.clear();
582
583 // clear door seal damaging auras:
584 if (Creature* c = instance->GetCreature(NPC_DoorSealGUID))
585 c->RemoveAllAuras();
586
587 // open main gate
588 HandleGameObject(GO_MainGateGUID, true);
589
590 if (m_auiEncounter[MAX_ENCOUNTER - 1] != DONE) // instance not finished
591 {
592 // close all cells
593 HandleGameObject(GO_MoraggCellGUID, false);
594 HandleGameObject(GO_ErekemCellGUID, false);
595 HandleGameObject(GO_ErekemRightGuardCellGUID, false);
596 HandleGameObject(GO_ErekemLeftGuardCellGUID, false);
597 HandleGameObject(GO_IchoronCellGUID, false);
598 HandleGameObject(GO_LavanthorCellGUID, false);
599 HandleGameObject(GO_XevozzCellGUID, false);
600 HandleGameObject(GO_ZuramatCellGUID, false);
601
602 // respawn bosses
603 if (Creature* c = instance->GetCreature(NPC_MoraggGUID)) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); c->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE); c->SetImmuneToNPC(true); }
604 if (Creature* c = instance->GetCreature(NPC_MoraggGUID)) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); c->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE); c->SetImmuneToNPC(true); }
605 if (Creature* c = instance->GetCreature(NPC_ErekemGUID)) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); c->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE); c->SetImmuneToNPC(true); }
606 if (Creature* c = instance->GetCreature(NPC_ErekemGuardGUID[0])) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); c->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE); c->SetImmuneToNPC(true); }
607 if (Creature* c = instance->GetCreature(NPC_ErekemGuardGUID[1])) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); c->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE); c->SetImmuneToNPC(true); }
608 if (Creature* c = instance->GetCreature(NPC_IchoronGUID)) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); c->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE); c->SetImmuneToNPC(true); }
609 if (Creature* c = instance->GetCreature(NPC_LavanthorGUID)) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); c->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE); c->SetImmuneToNPC(true); }
610 if (Creature* c = instance->GetCreature(NPC_XevozzGUID)) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); c->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE); c->SetImmuneToNPC(true); }
611 if (Creature* c = instance->GetCreature(NPC_ZuramatGUID)) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); c->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE); c->SetImmuneToNPC(true); }
612 if (Creature* c = instance->GetCreature(NPC_CyanigosaGUID)) { c->DespawnOrUnsummon(); }
613 }
614
615 // reinitialize variables and events
616 DoUpdateWorldState(WORLD_STATE_VH_SHOW, 0);
617 EncounterStatus = NOT_STARTED;
618 GateHealth = 100;
619 WaveCount = 0;
620 bDefensesUsed = false;
621 if (m_auiEncounter[MAX_ENCOUNTER - 1] == DONE)
622 EncounterStatus = DONE;
623 events.Reset();
624 events.RescheduleEvent(EVENT_CHECK_PLAYERS, CLEANUP_CHECK_INTERVAL);
