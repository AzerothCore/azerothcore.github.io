187 void Load(const char* in) override
188 {
189 if (!in)
190 {
191 OUT_LOAD_INST_DATA_FAIL;
192 return;
193 }
194
195 OUT_LOAD_INST_DATA(in);
196
197 char dataHead1, dataHead2;
198 uint32 data0, data1, data2, data3, data4;
199
200 std::istringstream loadStream(in);
201 loadStream >> dataHead1 >> dataHead2 >> data0 >> data1 >> data2 >> data3 >> data4;
202
203 if (dataHead1 == 'H' && dataHead2 == 'L')
204 {
205 m_auiEncounter[0] = data0;
206 m_auiEncounter[1] = data1;
207 m_auiEncounter[2] = data2;
208 m_auiEncounter[3] = data3;
209 m_auiEncounter[4] = data4;
210
211 for (uint8 i = 0; i < TYPE_LOKEN_INTRO; ++i)
212 if (m_auiEncounter[i] == IN_PROGRESS)
213 m_auiEncounter[i] = NOT_STARTED;
214
215 OUT_LOAD_INST_DATA_COMPLETE;
216 }
217 else
218 OUT_LOAD_INST_DATA_FAIL;
219 }
