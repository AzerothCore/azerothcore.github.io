103 void UpdateAI(uint32 diff) override
104 {
105 //Return since we have no target
106 if (!UpdateVictim())
107 return;
108
109 events.Update(diff);
110
111 if (me->HasUnitState(UNIT_STATE_CASTING))
112 return;
113
114 while (uint32 eventId = events.ExecuteEvent())
115 {
116 switch (eventId)
117 {
118 case SPELL_HANDOFTHAURISSAN:
119 DoCast(SelectTarget(SelectTargetMethod::Random), SPELL_HANDOFTHAURISSAN);
120 //DoCastVictim(SPELL_HANDOFTHAURISSAN);
121 events.ScheduleEvent(SPELL_HANDOFTHAURISSAN, urand(4000, 7000));
122 break;
123 case SPELL_AVATAROFFLAME:
124 DoCastSelf(SPELL_AVATAROFFLAME);
125 events.ScheduleEvent(SPELL_AVATAROFFLAME, urand(23000, 27000));
126 break;
127 default:
128 break;
129 }
130 }
131 DoMeleeAttackIfReady();
132 }
133 };
