117 void JustSummoned(Creature* summon) override

133 }
134
135 void UpdateAI(uint32 diff) override
136 {
137 if (!UpdateVictim())
138 return;
139
140 events.Update(diff);
141 if (me->HasUnitState(UNIT_STATE_CASTING))
142 return;
143
144 switch (events.ExecuteEvent())
145 {
146 case EVENT_HEALTH_CHECK:
147 //used for check, when Vexallus cast adds 85%, 70%, 55%, 40%
148 if (!HealthAbovePct(100 - INTERVAL_MODIFIER * IntervalHealthAmount))
149 {
150 if (IntervalHealthAmount++ == INTERVAL_SWITCH)
151 {
152 events.Reset();
153 me->CastSpell(me, SPELL_OVERLOAD, true);
154 return;
155 }
156
157 Talk(SAY_ENERGY);
158 Talk(EMOTE_DISCHARGE_ENERGY);
159
160 if (IsHeroic())
161 {
162 me->CastSpell(me, SPELL_SUMMON_PURE_ENERGY_H1, false);
163 me->CastSpell(me, SPELL_SUMMON_PURE_ENERGY_H2, false);
164 }
165 else
166 me->CastSpell(me, SPELL_SUMMON_PURE_ENERGY_N, false);
167 }
168 events.ScheduleEvent(EVENT_HEALTH_CHECK, 0);
169 break;
170 case EVENT_SPELL_CHAIN_LIGHTNING:
171 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
172 me->CastSpell(target, DUNGEON_MODE(SPELL_CHAIN_LIGHTNING_N, SPELL_CHAIN_LIGHTNING_H), false);
173 events.ScheduleEvent(EVENT_SPELL_CHAIN_LIGHTNING, 8000);
174 break;
175 case EVENT_SPELL_ARCANE_SHOCK:
176 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 20.0f))
177 me->CastSpell(target, DUNGEON_MODE(SPELL_ARCANE_SHOCK_N, SPELL_ARCANE_SHOCK_H), false);
178 events.ScheduleEvent(EVENT_SPELL_ARCANE_SHOCK, 8000);
179 break;
180 }
181
182 DoMeleeAttackIfReady();
183 }
184 };
185};
