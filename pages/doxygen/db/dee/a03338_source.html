63 void Reset() override

72 void EnterCombat(Unit* /*who*/) override
73 {
74 me->SetImmuneToAll(false);
75
76 events.ScheduleEvent(EVENT_OBLITERATE, 15000);
77 events.ScheduleEvent(EVENT_WELL_OF_CORRUPTION, 13000);
78 events.ScheduleEvent(EVENT_CORRUPTED_FLESH, 20000);

84 if (a == 1)

86 Talk(SAY_AGGRO);

91 void UpdateAI(uint32 diff) override
92 {
93 if (startFightTimer)
94 {
95 if (startFightTimer <= diff)
96 {
97 startFightTimer = 0;
98 me->SetInCombatWithZone();
99 }
100 else
101 startFightTimer -= diff;
102 }
103
104 if (!UpdateVictim())
105 return;
106
107 events.Update(diff);
108
109 if (me->HasUnitState(UNIT_STATE_CASTING))
110 return;
111
112 switch (events.ExecuteEvent())
113 {
114 case EVENT_OBLITERATE:
115 if (me->IsWithinMeleeRange(me->GetVictim()))
116 {
117 me->CastSpell(me->GetVictim(), SPELL_OBLITERATE, false);
118 events.ScheduleEvent(EVENT_OBLITERATE, 15000);
119 }
120 else
121 events.ScheduleEvent(EVENT_OBLITERATE, 3000);
122 break;
123 case EVENT_WELL_OF_CORRUPTION:
124 if (Unit* target = SelectTargetFromPlayerList(40.0f, 0, true))
125 me->CastSpell(target, SPELL_WELL_OF_CORRUPTION, false);
126 events.ScheduleEvent(EVENT_WELL_OF_CORRUPTION, 13000);
127 break;
128 case EVENT_CORRUPTED_FLESH:
129 Talk(RAND(SAY_CORRUPTED_FLESH_1, SAY_CORRUPTED_FLESH_2));
130 me->CastSpell((Unit*)nullptr, SPELL_CORRUPTED_FLESH, false);
131 events.ScheduleEvent(EVENT_CORRUPTED_FLESH, 20000);
132 break;
133 case EVENT_SHARED_SUFFERING:
134 if (Unit* target = SelectTargetFromPlayerList(200.0f, 0, true))
135 me->CastSpell(target, SPELL_SHARED_SUFFERING, true);
136 events.ScheduleEvent(EVENT_SHARED_SUFFERING, 15000);
137 break;
138 }
139
140 DoMeleeAttackIfReady();
141 }
142
143 void JustDied(Unit* /*killer*/) override
144 {
145 Talk(SAY_DEATH);
146 if (pInstance)
147 pInstance->SetData(DATA_MARWYN, DONE);
148 }
149
150 void KilledUnit(Unit* who) override
151 {
152 if (who->GetTypeId() == TYPEID_PLAYER)
153 Talk(RAND(SAY_SLAY_1, SAY_SLAY_2));
154 }
155
156 void EnterEvadeMode(EvadeReason why) override

158 ScriptedAI::EnterEvadeMode(why);
159 if (startFightTimer)
160 Reset();
161 }
162 };

179 void OnRemove(AuraEffect const* aurEff, AuraEffectHandleModes /*mode*/)
180 {
181 if (GetTargetApplication()->GetRemoveMode() == AURA_REMOVE_BY_ENEMY_SPELL) // dispelled
182 if (Unit* caster = GetCaster())
183 if (Map* map = caster->FindMap())
184 if (Aura* a = aurEff->GetBase())
185 {
186 uint32 count = 0;
187 uint32 ticks = 0;
188 uint32 dmgPerTick = a->GetSpellInfo()->Effects[0].BasePoints;
189 Map::PlayerList const& pl = map->GetPlayers();
190 for (Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr)
191 if (Player* p = itr->GetSource())
192 if (p->IsAlive())
193 ++count;
194 ticks = (a->GetDuration() / int32(a->GetSpellInfo()->Effects[0].Amplitude)) + 1;
195 int32 dmg = (ticks * dmgPerTick) / count;
196 caster->CastCustomSpell(GetTarget(), 72373, nullptr, &dmg, nullptr, true);
197 }
198 }
