45class boss_baron_geddon : public CreatureScript
46{
47public:
48 boss_baron_geddon() : CreatureScript("boss_baron_geddon") { }
49
50 struct boss_baron_geddonAI : public BossAI
51 {
52 boss_baron_geddonAI(Creature* creature) : BossAI(creature, DATA_GEDDON),
53 armageddonCasted(false)
54 {
55 }
56
57 void Reset() override
58 {
59 _Reset();
60 armageddonCasted = false;
61 }
62
63 void EnterCombat(Unit* /*attacker*/) override
64 {
65 _EnterCombat();
66 events.ScheduleEvent(EVENT_INFERNO, urand(13000, 15000));
67 events.ScheduleEvent(EVENT_IGNITE_MANA, urand(7000, 19000));
68 events.ScheduleEvent(EVENT_LIVING_BOMB, urand(11000, 16000));
69 }
70
71 void DamageTaken(Unit* /*attacker*/, uint32& damage, DamageEffectType /*dmgType*/, SpellSchoolMask /*school*/) override
72 {
73 // If boss is below 2% hp - cast Armageddon
74 if (!armageddonCasted && damage < me->GetHealth() && me->HealthBelowPctDamaged(2, damage))
75 {
76 me->RemoveAurasDueToSpell(SPELL_INFERNO);
77 me->StopMoving();
78 if (me->CastSpell(me, SPELL_ARMAGEDDON, TRIGGERED_FULL_MASK) == SPELL_CAST_OK)
79 {
80 Talk(EMOTE_SERVICE);
81 armageddonCasted = true;
82 }
83 }
84 }
85
86 void ExecuteEvent(uint32 eventId) override
87 {
88 switch (eventId)
89 {
90 case EVENT_INFERNO:
91 {
92 DoCastAOE(SPELL_INFERNO);
93 events.RepeatEvent(urand(21000, 26000));
94 break;
95 }
96 case EVENT_IGNITE_MANA:
97 {
98 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true, -SPELL_IGNITE_MANA))
99 {
100 DoCast(target, SPELL_IGNITE_MANA);
101 }
102
103 events.RepeatEvent(urand(27000, 32000));
104 break;
105 }
106 case EVENT_LIVING_BOMB:
107 {
108 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true))
109 {
110 DoCast(target, SPELL_LIVING_BOMB);
111 }
112
113 events.RepeatEvent(urand(11000, 16000));
114 break;
115 }
116 }
117 }
118
119 private:
120 bool armageddonCasted;
121 };
122
123 CreatureAI* GetAI(Creature* creature) const override
124 {
125 return GetMoltenCoreAI<boss_baron_geddonAI>(creature);
126 }
127};

172 case 3:
