64 void OnGameObjectCreate(GameObject* gameobject) override

70 gameobject->SetGoState(GetBossState(DATA_SLAD_RAN) == DONE ? GO_STATE_ACTIVE : GO_STATE_READY);
71 break;
72 case GO_ALTAR_OF_DRAKKARI:
73 _drakkariAltarGUID = gameobject->GetGUID();

76 case GO_ALTAR_OF_MOORABI:

128 void SetData(uint32 type, uint32) override
129 {
130 switch (type)
131 {
132 case NPC_ECK_THE_FEROCIOUS:
133 if (GetBossState(DATA_ECK_THE_FEROCIOUS_INIT) != DONE)
134 {
135 SetBossState(DATA_ECK_THE_FEROCIOUS_INIT, NOT_STARTED);
136 SetBossState(DATA_ECK_THE_FEROCIOUS_INIT, DONE);
137 }
138 break;
139 case GO_ALTAR_OF_SLAD_RAN:
140 if (GameObject* statue = instance->GetGameObject(_bridgeGUIDs[0]))
141 statue->SetGoState(GO_STATE_READY);
142 break;
143 case GO_ALTAR_OF_DRAKKARI:
144 if (GameObject* statue = instance->GetGameObject(_bridgeGUIDs[1]))
145 statue->SetGoState(GO_STATE_READY);
146 break;
147 case GO_ALTAR_OF_MOORABI:
148 if (GameObject* statue = instance->GetGameObject(_bridgeGUIDs[2]))
149 statue->SetGoState(GO_STATE_READY);
150 break;
151 }
152
153 if (type >= GO_ALTAR_OF_SLAD_RAN)
154 {
155 for (uint8 i = 0; i < 3; ++i)
156 if (GameObject* statue = instance->GetGameObject(_bridgeGUIDs[i]))
157 if (statue->GetGoState() != GO_STATE_READY)
158 return;
159 _activateTimer = 1;
160 }
161 }

163 bool SetBossState(uint32 type, EncounterState state) override
164 {
165 if (!InstanceScript::SetBossState(type, state))
166 {
167 if (state == DONE && (type == DATA_SLAD_RAN || type == DATA_MOORABI || type == DATA_DRAKKARI_COLOSSUS))
168 ++_keysInCount;
169 return false;
170 }
171
172 if (state != DONE)
173 return true;
174
175 switch (type)
176 {
177 case DATA_SLAD_RAN:
178 if (GameObject* altar = instance->GetGameObject(_sladRanAltarGUID))
179 altar->RemoveGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
180 break;
181 case DATA_MOORABI:
182 if (GameObject* altar = instance->GetGameObject(_moorabiAltarGUID))
183 altar->RemoveGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
184 break;
185 case DATA_DRAKKARI_COLOSSUS:
186 if (GameObject* altar = instance->GetGameObject(_drakkariAltarGUID))
187 altar->RemoveGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
188 break;
189 case DATA_ECK_THE_FEROCIOUS_INIT:
190 {
191 Position pos = {1624.70f, 891.43f, 95.08f, 1.2f};
192 if (instance->IsHeroic())
193 instance->SummonCreature(NPC_ECK_THE_FEROCIOUS, pos);
194 break;
195 }
196 }
197 return true;
198 }

207 void Load(const char* in) override
208 {
209 if (!in)
210 return;
211
212 char dataHead1, dataHead2;
213 std::istringstream loadStream(in);
214 loadStream >> dataHead1 >> dataHead2;
215 if (dataHead1 == 'G' && dataHead2 == 'D')
216 {
217 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
218 {
219 uint32 tmpState;
220 loadStream >> tmpState;
221 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
222 tmpState = NOT_STARTED;
223 SetBossState(i, EncounterState(tmpState));
224 }
225 }
226 }
227
228 void Update(uint32 diff) override
229 {
230 if (!_activateTimer)
231 return;
232
233 _activateTimer += diff;
234 if (_activateTimer >= 5000)
235 {
236 _activateTimer = 0;
237 for (uint8 i = 0; i < 5; ++i)
238 if (GameObject* go = instance->GetGameObject(_bridgeGUIDs[i]))
239 go->SetGoState(GO_STATE_ACTIVE_ALTERNATIVE);
240 }
241 }
242 };
