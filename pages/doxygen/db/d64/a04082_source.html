98 void UpdateAI(uint32 diff) override
99 {
100 if (!UpdateVictim())
101 return;
102
103 events.Update(diff);
104 if (me->HasUnitState(UNIT_STATE_CASTING))
105 return;
106

109 case EVENT_FROST_ATTACK:

114 me->CastSpell(me->GetVictim(), SPELL_ARCANE_BLAST, false);
115 events.ScheduleEvent(EVENT_ARCANE_BLAST, 12000);
116 break;
117 case EVENT_DRAGONS_BREATH:
118 me->CastSpell(me->GetVictim(), SPELL_DRAGONS_BREATH, true);
119 events.ScheduleEvent(EVENT_DRAGONS_BREATH, 16000);
120 if (roll_chance_i(50))
121 Talk(SAY_DRAGONS_BREATH);
122 break;
123 }
124
125 DoMeleeAttackIfReady();
