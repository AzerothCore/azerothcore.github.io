124 void WaypointReached(uint32 waypointId) override
125 {
126 if (Player* player = GetPlayerForEscort())
127 {
128 switch (waypointId)
129 {
130 case 0:
131 Talk(SAY_MUG_START2, player);
132 break;
133 case 24:
134 Talk(SAY_MUG_BRAZIER, player);
135
136 if (GameObject* go = GetClosestGameObjectWithEntry(me, GO_NAGA_BRAZIER, INTERACTION_DISTANCE * 2))
137 {
138 go->RemoveGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
139 SetEscortPaused(true);
140 }
141 break;
142 case 25:
143 Talk(SAY_MUG_GRATITUDE);
144 player->GroupEventHappens(QUEST_VORSHA, me);
145 break;
146 case 26:
147 Talk(SAY_MUG_PATROL);
148 break;
149 case 27:
150 Talk(SAY_MUG_RETURN);
151 break;
152 }
153 }

156 void DoWaveSummon()
157 {
158 switch (waveId)
159 {
160 case 1:
161 me->SummonCreature(NPC_WRATH_RIDER, FirstNagaCoord[0], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
162 me->SummonCreature(NPC_WRATH_SORCERESS, FirstNagaCoord[1], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
163 me->SummonCreature(NPC_WRATH_RAZORTAIL, FirstNagaCoord[2], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
164 break;
165 case 2:
166 me->SummonCreature(NPC_WRATH_PRIESTESS, SecondNagaCoord[0], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
167 me->SummonCreature(NPC_WRATH_MYRMIDON, SecondNagaCoord[1], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
168 me->SummonCreature(NPC_WRATH_SEAWITCH, SecondNagaCoord[2], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
169 break;
170 case 3:
171 me->SummonCreature(NPC_VORSHA, VorshaCoord, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 60000);
172 break;
173 case 4:
174 SetEscortPaused(false);
175 Talk(SAY_MUG_DONE);
176 break;
177 }
178 }
179
180 void UpdateAI(uint32 diff) override
181 {
182 npc_escortAI::UpdateAI(diff);
183
184 if (!me->GetVictim())
185 {
186 if (HasEscortState(STATE_ESCORT_PAUSED) && _isBrazierExtinguished)
187 {
188 if (eventTimer < diff)
189 {
190 ++waveId;
191 DoWaveSummon();
192 eventTimer = 10000;
193 }
194 else
195 eventTimer -= diff;
196 }
197 return;
198 }
199 DoMeleeAttackIfReady();
200 }
201
202 private:
203 uint32 eventTimer;
204 uint8 waveId;
205 public:
206 bool _isBrazierExtinguished;
207 };
