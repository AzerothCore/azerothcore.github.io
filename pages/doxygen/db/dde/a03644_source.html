34 struct npc_dragonflayer_forge_masterAI : public ScriptedAI
35 {
36 npc_dragonflayer_forge_masterAI(Creature* c) : ScriptedAI(c)
37 {
38 pInstance = c->GetInstanceScript();
39
40 float x = me->GetHomePosition().GetPositionX();
41 float y = me->GetHomePosition().GetPositionY();
42 if (x > 344.0f && x < 357.0f && y < -35.0f && y > -44.0f)
43 {
44 dataId = DATA_FORGE_1;
45 prevDataId = 0;
46 }
47 else if (x > 380.0f && x < 389.0f && y < -12.0f && y > -21.0f)
48 {
49 dataId = DATA_FORGE_2;
50 prevDataId = DATA_FORGE_1;
51 }
52 else
53 {
54 dataId = DATA_FORGE_3;
55 prevDataId = DATA_FORGE_2;
56 }
57 }
58

60 uint32 dataId;
61 uint32 prevDataId;
62
63 void Reset() override
64 {
65 if (pInstance)
66 pInstance->SetData(dataId, NOT_STARTED);
67 }
68
69 void JustDied(Unit* /*killer*/) override
70 {
71 if (pInstance)
72 pInstance->SetData(dataId, DONE);
73 me->SaveRespawnTime();
74 }
75
76 void EnterCombat(Unit* /*who*/) override
77 {
78 if (pInstance)

82 EnterEvadeMode();

87 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_ONESHOT_NONE);

124 void Reset() override
125 {
126 _events.Reset();
127 _events.ScheduleEvent(EVENT_REND, urand(2000, 3000));
128 _events.ScheduleEvent(EVENT_FLAME_BREATH, urand(5500, 7000));
129 _events.ScheduleEvent(EVENT_KNOCKAWAY, urand(3500, 6000));

131
132 void MovementInform(uint32 type, uint32 id) override
133 {
134 if (type == WAYPOINT_MOTION_TYPE && id == POINT_LAST)
135 {
136 me->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0.25f);
137 if (Vehicle* v = me->GetVehicleKit())
138 if (Unit* p = v->GetPassenger(0))
139 if (Creature* rider = p->ToCreature())
140 rider->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0.25f);
141
142 me->SetCanFly(false);
143 me->SetDisableGravity(false);
144 me->SetFacingTo(0.25f);
145 me->SetImmuneToAll(false);
146 }
147 }
148
149 void SetData(uint32 type, uint32 data) override
150 {
151 if (type == TYPE_PROTODRAKE_AT && data == DATA_PROTODRAKE_MOVE && !_setData && me->IsAlive() && me->GetDistance(protodrakeCheckPos) < 10.0f)
152 {
153 _setData = true;
154 me->SetCanFly(true);
155 me->SetDisableGravity(true);
156 me->GetMotionMaster()->MovePath(PATH_PROTODRAKE, false);
157 }
158 }
159
160 void UpdateAI(uint32 diff) override
161 {
162 if (!UpdateVictim())
163 return;
164
165 _events.Update(diff);
166
167 if (me->HasUnitState(UNIT_STATE_CASTING))
168 return;
169
170 while (uint32 eventid = _events.ExecuteEvent())
171 {
172 switch (eventid)
173 {
174 case EVENT_REND:
175 DoCast(SPELL_REND);
176 _events.ScheduleEvent(EVENT_REND, urand(15000, 20000));
177 break;
178 case EVENT_FLAME_BREATH:
179 DoCast(SPELL_FLAME_BREATH);
180 _events.ScheduleEvent(EVENT_FLAME_BREATH, urand(11000, 12000));
181 break;
182 case EVENT_KNOCKAWAY:
183 DoCast(SPELL_KNOCK_AWAY);
184 _events.ScheduleEvent(EVENT_KNOCKAWAY, urand(7000, 8500));
185 break;
186 default:
187 break;
188 }
189 }
190
191 DoMeleeAttackIfReady();
192 }
193
194 private:
195 bool _setData;
196 EventMap _events;
197 };
