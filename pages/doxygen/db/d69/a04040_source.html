67 void Reset() override
68 {
69 _Reset();
70 me->SetDisplayId(MODEL_DEFAULT);
71 _lastTransform = SPELL_DAMAGE_IMMUNE_SHADOW;
72 me->CastSpell(me, SPELL_DAMAGE_IMMUNE_SHADOW, true);
73 }
74
75 void DoTransform()
76 {
77 me->RemoveAurasDueToSpell(_lastTransform);
78
79 switch (_lastTransform = RAND(SPELL_DAMAGE_IMMUNE_SHADOW, SPELL_DAMAGE_IMMUNE_FIRE, SPELL_DAMAGE_IMMUNE_FROST, SPELL_DAMAGE_IMMUNE_NATURE, SPELL_DAMAGE_IMMUNE_ARCANE))
80 {
81 case SPELL_DAMAGE_IMMUNE_SHADOW:
82 me->SetDisplayId(MODEL_DEFAULT);
83 break;
84 case SPELL_DAMAGE_IMMUNE_ARCANE:
85 me->SetDisplayId(MODEL_ARCANE);
86 break;
87 case SPELL_DAMAGE_IMMUNE_FIRE:
88 me->SetDisplayId(MODEL_FIRE);
89 break;
90 case SPELL_DAMAGE_IMMUNE_FROST:
91 me->SetDisplayId(MODEL_FROST);
92 break;
93 case SPELL_DAMAGE_IMMUNE_NATURE:
94 me->SetDisplayId(MODEL_NATURE);
95 break;
96 }
97
98 me->CastSpell(me, _lastTransform, true);
99 }

110 void UpdateAI(uint32 diff) override
111 {
112 if (!UpdateVictim())
113 return;
114
115 events.Update(diff);
116 if (me->HasUnitState(UNIT_STATE_CASTING))
117 return;
118
119 switch (events.ExecuteEvent())
120 {
121 case EVENT_ALERGIC_REACTION:
122 me->CastSpell(me->GetVictim(), SPELL_ALLERGIC_REACTION, false);
123 events.ScheduleEvent(EVENT_ALERGIC_REACTION, 25000);
124 break;
125 case EVENT_TELEPORT:
126 me->CastSpell(me, SPELL_TELEPORT_SELF, false);
127 events.ScheduleEvent(EVENT_SUMMON, 2500);
128 events.ScheduleEvent(EVENT_TELEPORT, 30000);
129 break;
130 case EVENT_SUMMON:
131 Talk(EMOTE_SUMMON);
132 me->CastSpell(me, SPELL_SUMMON_LASHER_1, true);
133 me->CastSpell(me, SPELL_SUMMON_FLAYER_1, true);
134 break;
135 case EVENT_TRANSFORM:
136 DoTransform();
137 events.ScheduleEvent(EVENT_TRANSFORM, 35000);
138 break;
139 }
140
141 DoMeleeAttackIfReady();
142 }
143 private:
144 uint32 _lastTransform;
145 };
