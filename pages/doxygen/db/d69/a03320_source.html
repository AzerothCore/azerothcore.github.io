66 struct boss_bronjahmAI : public ScriptedAI

68 boss_bronjahmAI(Creature* creature) : ScriptedAI(creature), summons(me)
69 {
70 pInstance = creature->GetInstanceScript();
71 }
72
73 InstanceScript* pInstance;
74 EventMap events;
75 SummonList summons;
76
77 void JustReachedHome() override
78 {
79 me->CastSpell(me, SPELL_SOULSTORM_CHANNEL_OOC, true);
80 }
81
82 void Reset() override
83 {
84 me->RemoveUnitFlag(UNIT_FLAG_DISABLE_MOVE);
85 me->CastSpell(me, SPELL_SOULSTORM_CHANNEL_OOC, true);
86 events.Reset();
87 summons.DespawnAll();
88 if (pInstance)
89 pInstance->SetData(DATA_BRONJAHM, NOT_STARTED);
90 }
91
92 void EnterCombat(Unit* /*who*/) override
93 {
94 Talk(SAY_AGGRO);
95 me->RemoveAurasDueToSpell(SPELL_SOULSTORM_CHANNEL_OOC);
96
97 DoZoneInCombat();
98 events.Reset();
99 events.RescheduleEvent(EVENT_SPELL_SHADOW_BOLT, 2000);
100 events.RescheduleEvent(EVENT_SPELL_MAGICS_BANE, urand(5000, 10000));
101 events.RescheduleEvent(EVENT_SPELL_CORRUPT_SOUL, urand(14000, 20000));
102
103 if (pInstance)
104 pInstance->SetData(DATA_BRONJAHM, IN_PROGRESS);
105 }
106
107 void DamageTaken(Unit*, uint32& damage, DamageEffectType, SpellSchoolMask) override
108 {
109 if (!me->HasUnitFlag(UNIT_FLAG_DISABLE_MOVE) && me->HealthBelowPctDamaged(35, damage))
110 {
111 me->SetUnitFlag(UNIT_FLAG_DISABLE_MOVE);
112 me->GetMotionMaster()->Clear();
113 me->GetMotionMaster()->MoveIdle();
114 me->CastSpell(me, SPELL_TELEPORT, false);
115 events.CancelEvent(EVENT_SPELL_CORRUPT_SOUL);
116 events.DelayEvents(6000);
117 events.RescheduleEvent(EVENT_SPELL_FEAR, urand(8000, 14000));
118 }
119 }
120
121 void SpellHitTarget(Unit* /*target*/, SpellInfo const* spell) override
122 {
123 if (spell->Id == SPELL_TELEPORT)
124 {
125 me->CastSpell(me, SPELL_TELEPORT_VISUAL, true);
126 events.RescheduleEvent(EVENT_START_SOULSTORM, 1);
127 }
128 }
129
130 void UpdateAI(uint32 diff) override
131 {
132 if (!UpdateVictim())
133 return;
134
135 events.Update(diff);
136
137 if (me->HasUnitState(UNIT_STATE_CASTING))
138 return;
139
140 if (me->HasUnitFlag(UNIT_FLAG_DISABLE_MOVE))
141 if (me->isAttackReady())
142 me->SetFacingToObject(me->GetVictim());
143
144 switch(events.ExecuteEvent())
145 {
146 case 0:
147 break;
148 case EVENT_SPELL_SHADOW_BOLT:
149 if (!me->IsWithinMeleeRange(me->GetVictim()))
150 me->CastSpell(me->GetVictim(), SPELL_SHADOW_BOLT, false);
151 events.RepeatEvent(2000);
152 break;
153 case EVENT_SPELL_FEAR:
154 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 10.0f, true))
155 me->CastCustomSpell(SPELL_FEAR, SPELLVALUE_MAX_TARGETS, 1, target, false);
156 events.RepeatEvent(urand(8000, 12000));
157 break;
158 case EVENT_SPELL_MAGICS_BANE:
159 me->CastSpell(me->GetVictim(), SPELL_MAGICS_BANE, false);
160 events.RepeatEvent(urand(10000, 15000));
161 break;
162 case EVENT_SPELL_CORRUPT_SOUL:
163 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100.0f, true))
164 {
165 Talk(SAY_CORRUPT_SOUL);
166 me->CastSpell(target, SPELL_CORRUPT_SOUL, false);
167 }
168 events.RepeatEvent(urand(20000, 25000));
169 break;
170 case EVENT_START_SOULSTORM:
171 Talk(SAY_SOUL_STORM);
172 me->CastSpell(me, SPELL_SOULSTORM, false);
173 me->CastSpell(me, SPELL_TELEPORT_VISUAL, true);
174 me->CastSpell(me, SPELL_SOULSTORM_VISUAL, true);
175
176 break;
177 }
178
179 DoMeleeAttackIfReady();
180 }
181
182 void JustDied(Unit* /*killer*/) override
183 {
184 Talk(SAY_DEATH);
185 if (pInstance)
186 pInstance->SetData(DATA_BRONJAHM, DONE);
187 }
188
189 void KilledUnit(Unit* who) override
190 {
191 if (who->GetTypeId() == TYPEID_PLAYER)
192 Talk(SAY_SLAY);
193 }
194
195 void JustSummoned(Creature* summon) override
196 {
197 summons.Summon(summon);
198 summon->SetReactState(REACT_PASSIVE);
199 }
200
201 void EnterEvadeMode(EvadeReason why) override
202 {
203 me->RemoveUnitFlag(UNIT_FLAG_DISABLE_MOVE);
204 ScriptedAI::EnterEvadeMode(why);
205 }
206 };

208 CreatureAI* GetAI(Creature* creature) const override
209 {
210 return GetForgeOfSoulsAI<boss_bronjahmAI>(creature);
211 }

252 }

254 timer -= diff;
255 }
