108 void DamageTaken(Unit* who, uint32& damage, DamageEffectType, SpellSchoolMask) override
109 {
110 if (!finished && damage >= me->GetHealth())
111 {
112 damage = 0;
113 finished = true;
114 me->SetRegeneratingHealth(false);
115 me->SetImmuneToAll(true);
116 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
117 me->SetReactState(REACT_PASSIVE);
118 if (InstanceScript* pInstance = me->GetInstanceScript())
119 {
120 if (Creature* cr = ObjectAccessor::GetCreature(*me, pInstance->GetGuidData(DATA_ARTHAS)))
121 cr->AI()->DoAction(ACTION_KILLED_MALGANIS);
122
123 // give credit to players
124 me->CastSpell(me, 58630, true);
125 }
126
127 // quest completion
128 if (who)
129 if (Player* player = who->GetCharmerOrOwnerPlayerOrPlayerItself())
130 player->RewardPlayerAndGroupAtEvent(31006, player); // Royal Escort quest, Mal'ganis bunny
131
132 EnterEvadeMode();
133 }
134 }

136 void UpdateAI(uint32 diff) override
137 {
138 if (!UpdateVictim())
139 return;
140
141 events.Update(diff);
142 if (me->HasUnitState(UNIT_STATE_CASTING))
143 return;
144
145 switch (events.ExecuteEvent())
146 {
147 case EVENT_SPELL_CARRION_SWARM:
148 me->CastSpell(me->GetVictim(), DUNGEON_MODE(SPELL_CARRION_SWARM_N, SPELL_CARRION_SWARM_H), false);
149 events.RepeatEvent(7000);
150 break;
151 case EVENT_SPELL_MIND_BLAST:
152 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
153 me->CastSpell(target, DUNGEON_MODE(SPELL_MIND_BLAST_N, SPELL_MIND_BLAST_H), false);
154 events.RepeatEvent(6000);
155 break;
156 case EVENT_SPELL_SLEEP:
157 Talk(SAY_SLEEP);
158 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
159 me->CastSpell(target, DUNGEON_MODE(SPELL_SLEEP_N, SPELL_SLEEP_H), false);
160 events.RepeatEvent(17000);
161 break;
162 case EVENT_SPELL_VAMPIRIC_TOUCH:
163 me->CastSpell(me, SPELL_VAMPIRIC_TOUCH, true);
164 events.RepeatEvent(30000);
165 break;
166 }
167
168 DoMeleeAttackIfReady();
169 }
170 };
