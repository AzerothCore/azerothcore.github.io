79
80 void Reset() override
81 {
82 _Reset();
83 _charmerGUID.Clear();
84 secondPhase = false;
85 summons.DespawnAll();
86 instance->SetData(DATA_EGG_EVENT, NOT_STARTED);
87 }
88
89 void JustDied(Unit* /*killer*/) override
90 {
91 if (secondPhase)
92 {
93 _JustDied();
94 }
95 else
96 {
97 // Respawn shorty in case of failure during phase 1.
98 me->SetCorpseRemoveTime(25);
99 me->SetRespawnTime(30);
100 me->SaveRespawnTime();
101
102 // Might not be required, safe measure.
103 me->SetLootRecipient(nullptr);
104
105 instance->SetData(DATA_EGG_EVENT, FAIL);
106 }
107 }
108
109 bool CanAIAttack(Unit const* target) const override
110 {
111 return !(target->GetTypeId() == TYPEID_UNIT && !secondPhase) && !target->HasAura(SPELL_CONFLAGRATION);
112 }
113
114 void EnterCombat(Unit* /*victim*/) override
115 {
116 _EnterCombat();
117
118 events.ScheduleEvent(EVENT_CLEAVE, 15000);
119 events.ScheduleEvent(EVENT_STOMP, 35000);
120 events.ScheduleEvent(EVENT_FIREBALL, 7000);
121 events.ScheduleEvent(EVENT_CONFLAGRATION, 12000);
122
123 instance->SetData(DATA_EGG_EVENT, IN_PROGRESS);
124 }
125
126 void DoChangePhase()
127 {
128 secondPhase = true;
129 _charmerGUID.Clear();
130 me->RemoveAllAuras();
131
132 DoCastSelf(SPELL_WARMING_FLAMES, true);
133
134 if (Creature* troops = instance->GetCreature(DATA_NEFARIAN_TROOPS))
135 {
136 troops->AI()->Talk(EMOTE_TROOPS_RETREAT);
137 }
138
139 for (ObjectGuid const& guid : _summonGUIDS)
140 {
141 if (Creature* creature = ObjectAccessor::GetCreature(*me, guid))
142 {
143 if (creature->IsAlive())
144 {
145 creature->CombatStop(true);
146 creature->SetReactState(REACT_PASSIVE);
147 creature->GetMotionMaster()->MovePoint(0, Position(-7560.568848f, -1028.553345f, 408.491211f, 0.523858f));
148 }
149 }
150 }
151 }
152
153 void SetGUID(ObjectGuid const guid, int32 /*id*/) override
154 {
155 _charmerGUID = guid;
156 }
157
158 void OnCharmed(bool apply) override
159 {
160 if (apply)
161 {
162 if (Unit* charmer = ObjectAccessor::GetUnit(*me, _charmerGUID))
163 {
164 charmer->CastSpell(charmer, SPELL_MIND_EXHAUSTION, true);
165 charmer->CastSpell(me, SPELL_MINDCONTROL_VISUAL, false);
166 }
167 }
168 else
169 {
170 if (Unit* charmer = ObjectAccessor::GetUnit(*me, _charmerGUID))
171 {
172 charmer->RemoveAurasDueToSpell(SPELL_MINDCONTROL_VISUAL);
173 me->TauntApply(charmer);
174 }
175 }
176 }
177
178 void DoAction(int32 action) override
179 {
180 if (action == ACTION_PHASE_TWO)
181 {
182 DoChangePhase();
183 }
184
185 if (action == TALK_EGG_BROKEN_RAND)
186 {
187 Talk(urand(SAY_EGGS_BROKEN1, SAY_EGGS_BROKEN3));
188 }
189 }
190
191 void JustSummoned(Creature* summon) override
192 {
193 _summonGUIDS.push_back(summon->GetGUID());
194 summon->SetOwnerGUID(me->GetGUID());
195 summons.Summon(summon);
196 }
197
198 void SummonMovementInform(Creature* summon, uint32 movementType, uint32 /*pathId*/) override
199 {
200 if (movementType == POINT_MOTION_TYPE)
201 {
202 summon->DespawnOrUnsummon();
203 }
204 }
205
206 void DamageTaken(Unit*, uint32& damage, DamageEffectType, SpellSchoolMask) override
207 {
208 if (!secondPhase && damage >= me->GetHealth())
209 {
210 Talk(SAY_DEATH);
211 DoCastAOE(SPELL_EXPLODE_ORB);
212 DoCastAOE(SPELL_EXPLOSION);
213 }
214 }
215
216 void UpdateAI(uint32 diff) override
217 {
218 if (!UpdateVictim())
219 return;
220
221 if (!me->IsCharmed())
222 {
223 events.Update(diff);
224 }
225
226 if (me->HasUnitState(UNIT_STATE_CASTING))
227 return;
228
229 while (uint32 eventId = events.ExecuteEvent())
230 {
231 switch (eventId)
232 {
233 case EVENT_CLEAVE:
234 DoCastVictim(SPELL_CLEAVE);
235 events.ScheduleEvent(EVENT_CLEAVE, urand(7000, 10000));
236 break;
237 case EVENT_STOMP:
238 DoCastVictim(SPELL_WARSTOMP);
239 events.ScheduleEvent(EVENT_STOMP, urand(15000, 25000));
240 break;
241 case EVENT_FIREBALL:
242 DoCastVictim(SPELL_FIREBALLVOLLEY);
243 events.ScheduleEvent(EVENT_FIREBALL, urand(12000, 15000));
244 break;
245 case EVENT_CONFLAGRATION:
246 DoCastVictim(SPELL_CONFLAGRATION);
247 if (me->GetVictim() && me->GetVictim()->HasAura(SPELL_CONFLAGRATION))
248 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1))
249 me->TauntApply(target);
250 events.ScheduleEvent(EVENT_CONFLAGRATION, 30000);
251 break;
252 }
253 }
254
255 DoMeleeAttackIfReady();
256 }
257
258 private:
259 bool secondPhase;
260 ObjectGuid _charmerGUID;
261 GuidVector _summonGUIDS;
262 };
263
264 CreatureAI* GetAI(Creature* creature) const override
265 {
266 return GetBlackwingLairAI<boss_razorgoreAI>(creature);
267 }
268};
