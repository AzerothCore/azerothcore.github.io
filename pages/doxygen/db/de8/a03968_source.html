177 void Load(const char* strIn) override
178 {
179 if (!strIn)
180 {
181 OUT_LOAD_INST_DATA_FAIL;
182 return;
183 }
184
185 OUT_LOAD_INST_DATA(strIn);
186
187 char dataHead1, dataHead2;
188
189 std::istringstream loadStream(strIn);
190 loadStream >> dataHead1 >> dataHead2;
191
192 if (dataHead1 == 'B' && dataHead2 == 'F')
193 {
194 loadStream >> _auiEncounter[0] >> _auiEncounter[1] >> _auiEncounter[2];
195
196 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
197 if (_auiEncounter[i] == IN_PROGRESS || _auiEncounter[i] == FAIL)
198 _auiEncounter[i] = NOT_STARTED;
199 }
200
201 OUT_LOAD_INST_DATA_COMPLETE;
202 }

247 void StorePrisoner(Creature* creature)
248 {
249 float posX = creature->GetPositionX();
250 float posY = creature->GetPositionY();
251
252 if (posX >= 405.0f && posX <= 423.0f)
253 {
254 if (posY >= 106.0f && posY <= 123.0f)
255 {
256 _prisonersCell[0].insert(creature->GetGUID());
257 ++_prisonerCounter[0];
258 ResetPrisoner(creature);
259 }
260 else if (posY >= 76.0f && posY <= 91.0f)
261 {
262 _prisonersCell[1].insert(creature->GetGUID());
263 ++_prisonerCounter[1];
264 ResetPrisoner(creature);
265 }
266 }
267 else if (posX >= 490.0f && posX <= 506.0f)
268 {
269 if (posY >= 106.0f && posY <= 123.0f)
270 {
271 _prisonersCell[2].insert(creature->GetGUID());
272 ++_prisonerCounter[2];
273 ResetPrisoner(creature);
274 }
275 else if (posY >= 76.0f && posY <= 91.0f)

284 void PrisonerDied(ObjectGuid guid)
285 {
286 if (_prisonersCell[0].find(guid) != _prisonersCell[0].end() && --_prisonerCounter[0] <= 0)
287 ActivateCell(DATA_PRISON_CELL2);
288 else if (_prisonersCell[1].find(guid) != _prisonersCell[1].end() && --_prisonerCounter[1] <= 0)
289 ActivateCell(DATA_PRISON_CELL3);
290 else if (_prisonersCell[2].find(guid) != _prisonersCell[2].end() && --_prisonerCounter[2] <= 0)
291 ActivateCell(DATA_PRISON_CELL4);
292 else if (_prisonersCell[3].find(guid) != _prisonersCell[3].end() && --_prisonerCounter[3] <= 0)
293 ActivateCell(DATA_DOOR5);
294 }
295
296 void ActivateCell(uint8 id)
297 {
298 switch (id)
299 {
300 case DATA_PRISON_CELL1:
301 case DATA_PRISON_CELL2:
302 case DATA_PRISON_CELL3:
303 case DATA_PRISON_CELL4:
304 HandleGameObject(_prisonGUIDs[id - DATA_PRISON_CELL1], true);
305 ActivatePrisoners(_prisonersCell[id - DATA_PRISON_CELL1]);
306 break;
307 case DATA_DOOR5:
308 HandleGameObject(_doorGUIDs[4], true);
309 if (Creature* broggok = instance->GetCreature(GetGuidData(DATA_BROGGOK)))
310 broggok->AI()->DoAction(ACTION_ACTIVATE_BROGGOK);
311 break;
312 }
313 }
314
315 void ActivatePrisoners(GuidSet prisoners)
316 {
317 for (ObjectGuid const& guid : prisoners)
318 if (Creature* prisoner = instance->GetCreature(guid))
319 {
320 prisoner->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
321 prisoner->SetImmuneToAll(false);
322 prisoner->SetInCombatWithZone();
323 }
324 }
