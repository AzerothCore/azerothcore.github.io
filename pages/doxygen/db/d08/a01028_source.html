49 struct TokenizeResult {
50 explicit operator bool() { return !token.empty(); }
51 std::string_view token;
52 std::string_view tail;
53 };
54
55 inline TokenizeResult tokenize(std::string_view args)
56 {
57 TokenizeResult result;
58 if (size_t delimPos = args.find(COMMAND_DELIMITER); delimPos != std::string_view::npos)
59 {
60 result.token = args.substr(0, delimPos);
61 if (size_t tailPos = args.find_first_not_of(COMMAND_DELIMITER, delimPos); tailPos != std::string_view::npos)
62 result.tail = args.substr(tailPos);
63 }
64 else
65 result.token = args;
66
67 return result;
68 }

94 // this essentially models std::optional<std::string_view>, except it can also hold an error message
95 // it has std::string_view's bool conversion and dereference operators
96 //
97 // monostate <-> unspecified error, typically end-of-string reached or parsing failed
98 // std::string <-> specified error, typically character-not-found or invalid item link
99 // std::string_view <-> success, string_view is remaining argument string
100 struct ChatCommandResult
101 {
102 ChatCommandResult(std::nullopt_t) : _storage() {}
103 ChatCommandResult(std::string const&) = delete;
104 ChatCommandResult(std::string&& s) : _storage(std::in_place_type<std::string>, std::forward<std::string>(s)) {}
105 ChatCommandResult(char const* c) : _storage(std::in_place_type<std::string>, c) {}
106 ChatCommandResult(std::string_view s) : _storage(std::in_place_type<std::string_view>, s) {}
107
108 ChatCommandResult(ChatCommandResult const&) = delete;
109 ChatCommandResult(ChatCommandResult&&) = default;
110 ChatCommandResult& operator=(ChatCommandResult const&) = delete;
111 ChatCommandResult& operator=(ChatCommandResult&&) = default;
112
113 std::string_view operator*() const { return std::get<std::string_view>(_storage); }
114 bool IsSuccessful() const { return std::holds_alternative<std::string_view>(_storage); }
115 explicit operator bool() const { return IsSuccessful(); }
116 bool HasErrorMessage() const { return std::holds_alternative<std::string>(_storage); }
117 std::string const& GetErrorMessage() const { return std::get<std::string>(_storage); }
118
119 private:
120 std::variant<std::monostate, std::string_view, std::string> _storage;
121 };
