163 bool CheckProc(ProcEventInfo& eventInfo)

166 {
167 if ((spellInfo->GetSchoolMask() & _lastSchool) && _swapTime > GameTime::GetGameTime().count())
168 return false;
169
170 uint32 form = 0;
171 switch (GetFirstSchoolInMask(spellInfo->GetSchoolMask()))
172 {
173 case SPELL_SCHOOL_FIRE:
174 form = SPELL_FIRE_FORM;
175 break;
176 case SPELL_SCHOOL_FROST:
177 form = SPELL_FROST_FORM;
178 break;
179 case SPELL_SCHOOL_ARCANE:
180 form = SPELL_ARCANE_FORM;
181 break;
182 case SPELL_SCHOOL_SHADOW:
183 form = SPELL_SHADOW_FORM;
184 break;
185 default:
186 break;
187 }
188
189 if (form)
190 {
191 _swapTime = GameTime::GetGameTime().count() + 6;
192 _lastSchool = spellInfo->GetSchoolMask();
193 GetUnitOwner()->RemoveAurasDueToSpell(_lastForm);
194 _lastForm = form;
195 GetUnitOwner()->CastSpell(GetUnitOwner(), _lastForm, true);
196 }
197 }
198
199 return false;
200 }
201

203 {
204 DoCheckProc += AuraCheckProcFn(spell_botanica_shift_form_AuraScript::CheckProc);
205 }
