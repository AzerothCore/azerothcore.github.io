232 // modulos a radian orientation to the range of 0..2PI
233 static float NormalizeOrientation(float o)
234 {
235 // fmod only supports positive numbers. Thus we have
236 // to emulate negative numbers
237 if (o < 0)
238 {
239 float mod = o * -1;
240 mod = std::fmod(mod, 2.0f * static_cast<float>(M_PI));
241 mod = -mod + 2.0f * static_cast<float>(M_PI);
242 return mod;
243 }
244 return std::fmod(o, 2.0f * static_cast<float>(M_PI));
245 }
