126 Unit* FindHatefulStrikeTarget()
127 {
128 Unit* target = nullptr;
129 ThreatContainer::StorageType const& threatlist = me->GetThreatMgr().GetThreatList();
130 for (ThreatContainer::StorageType::const_iterator i = threatlist.begin(); i != threatlist.end(); ++i)
131 {
132 Unit* unit = ObjectAccessor::GetUnit(*me, (*i)->getUnitGuid());
133 if (unit && me->IsWithinMeleeRange(unit))
134 if (!target || unit->GetHealth() > target->GetHealth())
135 target = unit;
136 }
137
138 return target;
139 }

141 void UpdateAI(uint32 diff) override
142 {
143 if (!UpdateVictim())
144 return;
145
146 events.Update(diff);
147 if (me->HasUnitState(UNIT_STATE_CASTING))
148 return;
149
150 switch (events.ExecuteEvent())
151 {
152 case EVENT_SPELL_BERSERK:
153 me->CastSpell(me, SPELL_BERSERK, true);
154 break;
155 case EVENT_SPELL_HATEFUL_STRIKE:
156 if (Unit* target = FindHatefulStrikeTarget())
157 me->CastSpell(target, SPELL_HATEFUL_STRIKE, false);
158 events.ScheduleEvent(EVENT_SPELL_HATEFUL_STRIKE, urand(1500, 3000), EVENT_GROUP_ABILITIES);
159 break;
160 case EVENT_SPELL_MOLTEN_FLAMES:
161 me->CastSpell(me, SPELL_MOLTEN_PUNCH, false);
162 events.ScheduleEvent(EVENT_SPELL_MOLTEN_FLAMES, 20000, EVENT_GROUP_ABILITIES);
163 break;
164 case EVENT_SWITCH_PHASE:
165 SchedulePhase(!me->HasAura(SPELL_SNARE_SELF));
166 break;
167 case EVENT_SWITCH_TARGET:
168 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
169 {
170 DoResetThreat();
171 me->AddThreat(target, 5000000.0f);
172 Talk(EMOTE_NEW_TARGET);
173 }
174 events.ScheduleEvent(EVENT_SWITCH_TARGET, 10000, EVENT_GROUP_ABILITIES);
175 break;
176 case EVENT_CHECK_DIST:
177 if (me->GetDistance(me->GetVictim()) > 40.0f)
178 {
179 Talk(EMOTE_PUNCH_GROUND);
180 me->CastSpell(me->GetVictim(), SPELL_CHARGE, true);
181 events.ScheduleEvent(EVENT_CHECK_DIST, 5000, EVENT_GROUP_ABILITIES);
182 break;
183 }
184 events.ScheduleEvent(EVENT_CHECK_DIST, 1, EVENT_GROUP_ABILITIES);
185 break;
186 case EVENT_SPELL_VOLCANIC_ERUPTION:
187 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
188 {
189 me->CastSpell(target, SPELL_VOLCANIC_ERUPTION, true);
190 Talk(EMOTE_GROUND_CRACK);
191 }
192 events.ScheduleEvent(EVENT_SPELL_VOLCANIC_ERUPTION, urand(10000, 18000), EVENT_GROUP_ABILITIES);
193 break;
194 }
195
196 DoMeleeAttackIfReady();
197 }
