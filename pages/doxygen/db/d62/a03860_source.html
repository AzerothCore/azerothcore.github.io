111class SuckBackEvent : public BasicEvent
112{
113public:
114 SuckBackEvent(Creature& owner, uint32 action) : BasicEvent(), _owner(owner), _action(action) { }
115
116 bool Execute(uint64 /*eventTime*/, uint32 /*diff*/) override
117 {
118 if (_owner.IsSummon())
119 if (Unit* summoner = _owner.ToTempSummon()->GetSummonerUnit())
120 {
121 summoner->GetAI()->DoAction(_action);
122 _owner.SetStandState(UNIT_STAND_STATE_SUBMERGED);

124 }
125 return true;
126 }

178 events.ScheduleEvent(EVENT_SPAWN_ENSLAVED_SOULS, 8000);

183 }
184 }

187 {
188 BossAI::EnterCombat(who);
189 }

198 summon->CastSpell(summon, SPELL_EMERGE_VISUAL, true);

205 }
206
207 void AttackStart(Unit*) override { }
208

210 {
211 summons.clear();
212 BossAI::JustDied(killer);
213 }
214
215 void UpdateAI(uint32 diff) override
216 {
217 if (me->getStandState() == UNIT_STAND_STATE_SLEEP)
218 return;
219
220 events.Update(diff);
221 switch (events.ExecuteEvent())
222 {
223 case EVENT_SUCK_ESSENCE:
224 me->SetStandState(UNIT_STAND_STATE_STAND);
225 break;
226 case EVENT_ENGAGE_ESSENCE:
227 summons.DoAction(ACTION_ENGAGE_ESSENCE);
228 break;
229 case EVENT_ESSENCE_OF_SUFFERING:
230 me->SetStandState(UNIT_STAND_STATE_SUBMERGED);
231 me->CastSpell(me, SPELL_SUMMON_ESSENCE_OF_SUFFERING, false);
232 break;
233 case EVENT_ESSENCE_OF_DESIRE:
234 summons.DespawnAll();
235 me->SetStandState(UNIT_STAND_STATE_SUBMERGED);
236 me->CastSpell(me, SPELL_SUMMON_ESSENCE_OF_DESIRE, false);
237 break;
238 case EVENT_ESSENCE_OF_ANGER:
239 summons.DespawnAll();

241 me->CastSpell(me, SPELL_SUMMON_ESSENCE_OF_ANGER, false);

244 events.ScheduleEvent(EVENT_SPAWN_SOUL, 0);
245 events.ScheduleEvent(EVENT_SPAWN_SOUL, 0);
246 for (uint8 i = 0; i < 16; ++i)
247 events.ScheduleEvent(EVENT_SPAWN_SOUL, i * 1200);
248 break;
249 case EVENT_SPAWN_SOUL:
250 me->CastCustomSpell(SPELL_SUMMON_ENSLAVED_SOUL, SPELLVALUE_MAX_TARGETS, 1, me, false);
251 break;
252 }
253 }
