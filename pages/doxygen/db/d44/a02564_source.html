244 bool SetBossState(uint32 bossId, EncounterState state) override
245 {
246 if (!InstanceScript::SetBossState(bossId, state))
247 {
248 return false;
249 }
250
251 if (bossId == DATA_MAJORDOMO_EXECUTUS && state == DONE)
252 {
253 if (GameObject* cache = instance->GetGameObject(_cacheOfTheFirelordGUID))
254 {
255 cache->SetRespawnTime(7 * DAY);
256 cache->SetLootRecipient(instance);
257 }
258 }
259 else if (bossId == DATA_GOLEMAGG)
260 {
261 switch (state)
262 {
263 case NOT_STARTED:
264 case FAIL:
265 {
266 if (!_golemaggMinionsGUIDS.empty())
267 {
268 for (ObjectGuid const& minionGuid : _golemaggMinionsGUIDS)
269 {
270 Creature* minion = instance->GetCreature(minionGuid);
271 if (minion && minion->isDead())
272 {
273 minion->Respawn();
274 }
275 }
276 }
277 break;
278 }
279 case IN_PROGRESS:
280 {
281 if (!_golemaggMinionsGUIDS.empty())
282 {
283 for (ObjectGuid const& minionGuid : _golemaggMinionsGUIDS)
284 {
285 if (Creature* minion = instance->GetCreature(minionGuid))
286 {
287 minion->AI()->DoZoneInCombat(nullptr, 150.0f);
288 }
289 }
290 }
291 break;
292 }
293 case DONE:
294 {
295 if (!_golemaggMinionsGUIDS.empty())
296 {
297 for (ObjectGuid const& minionGuid : _golemaggMinionsGUIDS)
298 {
299 if (Creature* minion = instance->GetCreature(minionGuid))
300 {
301 minion->CastSpell(minion, SPELL_CORE_RAGER_QUIET_SUICIDE, true);
302 }
303 }
304 _golemaggMinionsGUIDS.clear();
305 }
306 break;
307 }
308 default:
309 break;
310 }
311 }
312 // Perform needed checks for Majordomu
313 if (bossId < DATA_MAJORDOMO_EXECUTUS && state == DONE)
314 {
315 if (GameObject* circle = instance->GetGameObject(_circlesGUIDs[bossId]))
316 {
317 circle->DespawnOrUnsummon(0ms, Seconds(WEEK));
318 _circlesGUIDs[bossId].Clear();
319 }
320
321 if (GameObject* rune = instance->GetGameObject(_runesGUIDs[bossId]))
322 {
323 rune->UseDoorOrButton(WEEK * IN_MILLISECONDS);
324 _runesGUIDs[bossId].Clear();
325 }
326
327 if (CheckMajordomoExecutus())
328 {
329 SummonMajordomoExecutus();
330 }
331 }
332
333 return true;
334 }

418 void Load(char const* data) override
419 {
420 if (!data)
421 {
422 OUT_LOAD_INST_DATA_FAIL;
423 return;
424 }
425
426 OUT_LOAD_INST_DATA(data);
427
428 char dataHead1, dataHead2;
429
430 std::istringstream loadStream(data);
431 loadStream >> dataHead1 >> dataHead2;
432
433 if (dataHead1 == 'M' && dataHead2 == 'C')
434 {
435 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
436 {
437 uint32 tmpState;
438 loadStream >> tmpState;
439 if (tmpState == IN_PROGRESS || tmpState > TO_BE_DECIDED)
440 {
441 tmpState = NOT_STARTED;
442 }
443
444 SetBossState(i, static_cast<EncounterState>(tmpState));
445 }
446
447 if (CheckMajordomoExecutus())
448 {
449 SummonMajordomoExecutus();
450 }
451 }
452 else
453 {
454 OUT_LOAD_INST_DATA_FAIL;
455 }
456
457 OUT_LOAD_INST_DATA_COMPLETE;
458 }
