286 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
287 {
288 if (creature->HasNpcFlag(UNIT_NPC_FLAG_GOSSIP))
289 {
290 switch (action)
291 {
292 case GOSSIP_ACTION_INFO_DEF+1:
293 creature->AI()->DoAction(ACTION_START_CITY);
294 creature->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP);
295 CloseGossipMenuFor(player);
296 break;
297 case GOSSIP_ACTION_INFO_DEF+2:
298 ClearGossipMenuFor(player);
299 AddGossipItemFor(player, 9681, 0, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
300 SendGossipMenuFor(player, GOSSIP_MENU_ARTHAS_3, creature->GetGUID());
301 break;
302 case GOSSIP_ACTION_INFO_DEF+3:
303 // Start Town Hall part
304 creature->AI()->DoAction(ACTION_START_TOWN_HALL);
305 creature->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP);
306 CloseGossipMenuFor(player);
307 break;
308 case GOSSIP_ACTION_INFO_DEF+4:
309 // After killing epoch
310 creature->AI()->DoAction(ACTION_START_SECRET_PASSAGE);
311 creature->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP);
312 creature->SetTarget();
313 CloseGossipMenuFor(player);
314 break;
315 case GOSSIP_ACTION_INFO_DEF+5:
316 creature->AI()->DoAction(ACTION_START_LAST_CITY);
317 creature->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP);
318 CloseGossipMenuFor(player);
319 break;
320 case GOSSIP_ACTION_INFO_DEF+6:
321 creature->AI()->DoAction(ACTION_START_MALGANIS);
322 creature->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP);
323 CloseGossipMenuFor(player);

325 }
326 }
