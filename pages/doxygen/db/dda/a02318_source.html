26void GetPlayerInfo(ChatHandler* handler, Player* player)
27{
28 if (!player)
29 return;
30
31 ObjectGuid guid = player->GetGUID();
32 lfg::LfgDungeonSet dungeons = sLFGMgr->GetSelectedDungeons(guid);
33
34 std::string const& state = lfg::GetStateString(sLFGMgr->GetState(guid));
35 handler->PSendSysMessage(LANG_LFG_PLAYER_INFO, player->GetName().c_str(),
36 state.c_str(), uint8(dungeons.size()), lfg::ConcatenateDungeons(dungeons).c_str(),
37 lfg::GetRolesString(sLFGMgr->GetRoles(guid)).c_str(), sLFGMgr->GetComment(guid).c_str());
38}

65 static bool HandleLfgPlayerInfoCommand(ChatHandler* handler, Optional<PlayerIdentifier> player)
66 {
67 if (!player)
68 player = PlayerIdentifier::FromTargetOrSelf(handler);
69 if (!player)
70 return false;
71
72 if (Player* target = player->GetConnectedPlayer())
73 {
74 GetPlayerInfo(handler, target);
75 return true;
76 }
77
78 return false;

81 static bool HandleLfgGroupInfoCommand(ChatHandler* handler, Optional<PlayerIdentifier> player)
82 {
83 if (!player)
84 player = PlayerIdentifier::FromTargetOrSelf(handler);
85 if (!player)
86 return false;
87
88 Group* groupTarget = nullptr;
89 if (Player* target = player->GetConnectedPlayer())
90 groupTarget = target->GetGroup();
91 if (!groupTarget)
92 {
93 handler->PSendSysMessage(LANG_LFG_NOT_IN_GROUP, player->GetName().c_str());
94 return true;
95 }
96
97 ObjectGuid guid = groupTarget->GetGUID();
98 std::string const& state = lfg::GetStateString(sLFGMgr->GetState(guid));
99 handler->PSendSysMessage(LANG_LFG_GROUP_INFO, groupTarget->isLFGGroup(),
100 state.c_str(), sLFGMgr->GetDungeon(guid));
101
102 for (GroupReference* itr = groupTarget->GetFirstMember(); itr != nullptr; itr = itr->next())
103 GetPlayerInfo(handler, itr->GetSource());
104
105 return true;
106 }
107
108 static bool HandleLfgOptionsCommand(ChatHandler* handler, Optional<uint32> optionsArg)
109 {
110 if (optionsArg)
111 {
112 sLFGMgr->SetOptions(*optionsArg);
113 handler->PSendSysMessage(LANG_LFG_OPTIONS_CHANGED);
114 }
115 handler->PSendSysMessage(LANG_LFG_OPTIONS, sLFGMgr->GetOptions());
116 return true;
117 }
118
119 static bool HandleLfgQueueInfoCommand(ChatHandler* /*handler*/)
120 {
121 return true;
122 }
123
124 static bool HandleLfgCleanCommand(ChatHandler* handler)
125 {
126 handler->PSendSysMessage(LANG_LFG_CLEAN);
127 sLFGMgr->Clean();
128 return true;
129 }
