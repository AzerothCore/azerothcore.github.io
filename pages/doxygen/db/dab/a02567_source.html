48class npc_mc_core_hound : public CreatureScript

50public:
51 npc_mc_core_hound() : CreatureScript("npc_mc_core_hound") {}
52
53 struct npc_mc_core_houndAI : public CreatureAI
54 {
55 npc_mc_core_houndAI(Creature* creature) :
56 CreatureAI(creature),
57 instance(creature->GetInstanceScript()),
58 serratedBiteTimer(3000)
59 {
60 }
61
62 void Reset() override
63 {
64 serratedBiteTimer = 3000;
65 }
66
67 void DamageTaken(Unit* /*attacker*/, uint32& damage, DamageEffectType /*damagetype*/, SpellSchoolMask /*damageSchoolMask*/) override
68 {
69 // Prevent receiving any extra damage if Hound is playing dead
70 if (me->HasAura(SPELL_PLAY_DEAD))
71 {
72 damage = 0;
73 return;
74 }
75 else if (me->GetHealth() <= damage)
76 {
77 damage = 0;
78 Talk(EMOTE_SMOLDERING);
79 DoCastSelf(SPELL_PLAY_DEAD);
80 }
81 }
82
83 void UpdateAI(uint32 diff) override
84 {
85 if (!UpdateVictim())
86 {
87 return;
88 }
89
90 if (me->HasUnitState(UNIT_STATE_CASTING) || me->HasAura(SPELL_PLAY_DEAD))
91 {
92 return;
93 }
94
95 if (serratedBiteTimer <= diff)
96 {
97 DoCastVictim(SPELL_SERRATED_BITE);
98 serratedBiteTimer = urand(5000, 6000);
99 }
100 else
101 {
102 serratedBiteTimer -= diff;
103 }
104
105 DoMeleeAttackIfReady();
106 }
107
108 private:
109 InstanceScript* instance;
110 uint32 serratedBiteTimer;
111 };
112
113 CreatureAI* GetAI(Creature* creature) const override
114 {
115 return GetMoltenCoreAI<npc_mc_core_houndAI>(creature);
116 }
117};

119// 19822 Play Dead
120class spell_mc_play_dead : public SpellScriptLoader
121{
122public:
123 spell_mc_play_dead() : SpellScriptLoader("spell_mc_play_dead") { }
124
125 class spell_mc_play_dead_AuraScript : public AuraScript
126 {
127 PrepareAuraScript(spell_mc_play_dead_AuraScript);
128
129 bool Load() override
130 {
131 return GetCaster()->GetTypeId() == TYPEID_UNIT;
132 }
133
134 void HandleEffectApply(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
135 {
136 Creature* creatureTarget = GetTarget()->ToCreature();
137 if (!creatureTarget)
138 {
139 return;
140 }
141
142 creatureTarget->CastSpell(creatureTarget, SPELL_PLAY_DEAD_PACIFY, true);
143 creatureTarget->SetDynamicFlag(UNIT_DYNFLAG_DEAD);
144 creatureTarget->SetUnitFlag2(UNIT_FLAG2_FEIGN_DEATH);
145 //creatureTarget->SetUnitFlag(UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
146 creatureTarget->SetReactState(REACT_PASSIVE);
147 creatureTarget->SetControlled(true, UNIT_STATE_ROOT);
148
149 creatureTarget->AttackStop();
150 }
151
152 void HandleEffectRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
153 {
154 Creature* creatureTarget = GetTarget()->ToCreature();
155 if (!creatureTarget)
156 {
157 return;
158 }
159
160 creatureTarget->RemoveAurasDueToSpell(SPELL_PLAY_DEAD_PACIFY);
161 creatureTarget->RemoveDynamicFlag(UNIT_DYNFLAG_DEAD);
162 creatureTarget->RemoveUnitFlag2(UNIT_FLAG2_FEIGN_DEATH);
163 //creatureTarget->RemoveUnitFlag(UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
164 creatureTarget->SetControlled(false, UNIT_STATE_ROOT);
165 creatureTarget->SetReactState(REACT_AGGRESSIVE);
166
167 if (!creatureTarget->IsInCombat())
168 {
169 return;
170 }
171
172 bool shouldDie = true;
173 std::list<Creature*> hounds;
174 creatureTarget->GetCreaturesWithEntryInRange(hounds, 80.0f, NPC_CORE_HOUND);
175
176 // Perform lambda based check to find if there is any nearby
177 if (!hounds.empty())
178 {
179 // Alive hound been found within 80 yards -> cancel suicide
180 if (std::find_if(hounds.begin(), hounds.end(), [creatureTarget](Creature const* hound)
181 {
182 return creatureTarget != hound && creatureTarget->IsWithinLOSInMap(hound) && hound->IsAlive() && hound->IsInCombat() && !hound->HasAura(SPELL_PLAY_DEAD);
183 }) != hounds.end())
184 {
185 shouldDie = false;
186 }
187 }
188
189 if (!shouldDie)
190 {
191 if (CreatureAI* targetAI = creatureTarget->AI())
192 {
193 targetAI->DoCastSelf(SPELL_FIRE_NOVA_VISUAL, true);
194 targetAI->DoCastSelf(SPELL_FULL_HEALTH, true);
195 targetAI->Talk(EMOTE_IGNITE);
196 }
197 }
198 else
199 {
200 Unit::Kill(creatureTarget, creatureTarget);
201 creatureTarget->DespawnOrUnsummon(14000);
202 }
203 }
204
205 void Register() override
206 {
207 AfterEffectApply += AuraEffectApplyFn(spell_mc_play_dead_AuraScript::HandleEffectApply, EFFECT_0, SPELL_AURA_FEIGN_DEATH, AURA_EFFECT_HANDLE_REAL);
208 AfterEffectRemove += AuraEffectApplyFn(spell_mc_play_dead_AuraScript::HandleEffectRemove, EFFECT_0, SPELL_AURA_FEIGN_DEATH, AURA_EFFECT_HANDLE_REAL);
209 }
210 };
211
212 AuraScript* GetAuraScript() const override
213 {
214 return new spell_mc_play_dead_AuraScript();
215 }
216};

218struct npc_lava_spawn : public ScriptedAI
219{
220 npc_lava_spawn(Creature* creature) : ScriptedAI(creature)
221 {
222 }
223
224 void Reset() override
225 {
226 _scheduler.CancelAll();
227 }
228
229 void EnterCombat(Unit* /*who*/) override
230 {
231 _scheduler.Schedule(15s, [this](TaskContext context)
232 {
233 std::list<Creature*> lavaSpawns;
234 me->GetCreatureListWithEntryInGrid(lavaSpawns, me->GetEntry(), 100.f);
235 if (lavaSpawns.size() < 16)
236 {
237 Talk(TALK_0);
238
239 DoCastSelf(SPELL_SPLIT_1, true);
240 DoCastSelf(SPELL_SPLIT_2, true);
241
242 me->DespawnOrUnsummon();
243 }
244 else
245 {
246 context.Repeat(15s);
247 }
248 });
249 }
250
251 void UpdateAI(uint32 diff) override
252 {
253 if (!UpdateVictim())
254 {
255 return;
256 }
257
258 _scheduler.Update(diff);
259
260 DoSpellAttackIfReady(SPELL_FIREBALL);
261 }
262
263private:
264 TaskScheduler _scheduler;
265};
