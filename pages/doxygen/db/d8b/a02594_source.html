106 void UpdateAI(uint32 diff) override
107 {
108 if (!UpdateVictim())
109 return;
110
111 events.Update(diff);
112 if (me->HasUnitState(UNIT_STATE_CASTING))
113 return;
114
115 switch (events.ExecuteEvent())
116 {
117 case EVENT_CHECK_HEALTH:
118 if (me->HealthBelowPct(16))
119 {
120 events.CancelEvent(EVENT_SPELL_ASTRAL_FLARE);
121 me->CastSpell(me, SPELL_ARCANE_INFUSION, true);
122 Talk(SAY_ENRAGE);
123 break;
124 }
125 events.ScheduleEvent(EVENT_CHECK_HEALTH, 1000);
126 break;
127 case EVENT_SPELL_BERSERK:
128 Talk(SAY_ENRAGE);
129 me->InterruptNonMeleeSpells(true);
130 me->CastSpell(me, SPELL_ASTRAL_DECONSTRUCTION, true);
131 break;
132 case EVENT_SPELL_HATEFUL_BOLT:
133 if (Unit* target = SelectTarget(SelectTargetMethod::MaxThreat, urand(1, 2), 40.0f))
134 me->CastSpell(target, SPELL_HATEFUL_BOLT, false);
135 events.ScheduleEvent(EVENT_SPELL_HATEFUL_BOLT, urand(5000, 7500) * (events.GetNextEventTime(EVENT_SPELL_BERSERK) == 0 ? 1 : 2));
136 break;
137 case EVENT_SPELL_ASTRAL_FLARE:
138 {
139 me->CastSpell(me, RAND(SPELL_SUMMON_ASTRAL_FLARE1, SPELL_SUMMON_ASTRAL_FLARE2, SPELL_SUMMON_ASTRAL_FLARE3, SPELL_SUMMON_ASTRAL_FLARE4), false);
140 int32 mana = CalculatePct(me->GetMaxPower(POWER_MANA), 10);
141 me->ModifyPower(POWER_MANA, -mana);
142 if (me->GetPowerPct(POWER_MANA) < 10.0f)
143 {
144 Talk(SAY_EVOCATE);
145 me->CastSpell(me, SPELL_EVOCATION, false);
146
147 events.DelayEvents(20000);
148 events.ScheduleEvent(EVENT_SPELL_ASTRAL_FLARE, 20000);
149 }
150 else
151 {
152 if (roll_chance_i(50))
153 Talk(SAY_SUMMON);
154
155 events.ScheduleEvent(EVENT_SPELL_ASTRAL_FLARE, 10000);
156 }
157
158 break;
159 }
160 }
161
162 DoMeleeAttackIfReady();
163 }
164 };
