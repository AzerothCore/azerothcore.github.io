50 void InitializeAI() override
51 {
52 WPPath* path = sSmartWaypointMgr->GetPath(me->GetEntry());
53 if (!path || path->empty())
54 {
55 me->DespawnOrUnsummon(1);
56 return;
57 }
58
59 Movement::PointsArray pathPoints;
60 pathPoints.push_back(G3D::Vector3(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ()));
61
62 uint32 wpCounter = 1;
63 WPPath::const_iterator itr;
64 while ((itr = path->find(wpCounter++)) != path->end())
65 {
66 WayPoint* wp = itr->second;
67 pathPoints.push_back(G3D::Vector3(wp->x, wp->y, wp->z));
68 }
69
70 me->GetMotionMaster()->MoveSplinePath(&pathPoints);
71
72 NullCreatureAI::InitializeAI();
73 pointId = 0;
74 searchForShipTimer = 0;
75 transportEntry = (me->GetEntry() == NPC_HAMMERHEAD ? TRANSPORT_ORGRIMS_HAMMER : TRANSPORT_THE_SKYBREAKER);
76 }
77
78 void MovementInform(uint32 type, uint32 /*id*/) override
79 {
80 if (type == ESCORT_MOTION_TYPE)
81 if (++pointId == 17) // path size
82 searchForShipTimer = 3000;
83 }

85 void UpdateAI(uint32 diff) override
86 {
87 // horde 7.55f, -0.09, 34.44, 3.13, +20
88 // ally 45.18f, 0.03, 40.09, 3.14 +5
89
90 if (searchForShipTimer)
91 {
92 searchForShipTimer += diff;
93 if (searchForShipTimer >= 3000)
94 {
95 searchForShipTimer = 1;
96 TransportsContainer const& transports = me->GetMap()->GetAllTransports();
97 for (TransportsContainer::const_iterator itr = transports.begin(); itr != transports.end(); ++itr)
98 {
99 if ((*itr)->GetEntry() == transportEntry)
100 {
101 float x, y, z;
102 if (transportEntry == TRANSPORT_ORGRIMS_HAMMER)
103 {
104 x = 7.55f;
105 y = -0.09f;
106 z = 54.44f;
107 }
108 else
109 {
110 x = 45.18f;
111 y = 0.03f;
112 z = 45.09f;
113 }
114
115 (*itr)->CalculatePassengerPosition(x, y, z);
116
117 if (me->GetDistance2d(x, y) < 10.0f)
118 {
119 me->DespawnOrUnsummon(1000);
120 if (Vehicle* vehicle = me->GetVehicleKit())
121 if (Unit* passenger = vehicle->GetPassenger(0))
122 {
123 passenger->NearTeleportTo(x, y, z - (transportEntry == TRANSPORT_ORGRIMS_HAMMER ? 19.0f : 4.0f), M_PI);
124 passenger->RemoveAurasDueToSpell(VEHICLE_SPELL_PARACHUTE); // maybe vehicle / seat flag should be responsible for parachute gain?
125 }
126 }
127 else
128 me->GetMotionMaster()->MovePoint(0, x, y, z, false, false);
129 break;
130 }
131 }
132 }
133 }
134 }
