47bool WinServiceInstall()
48{
49 SC_HANDLE serviceControlMgr = OpenSCManager(0, 0, SC_MANAGER_CREATE_SERVICE);
50
51 if (serviceControlMgr)
52 {
53 char path[_MAX_PATH + 10];
54 if (GetModuleFileName( 0, path, sizeof(path) / sizeof(path[0]) ) > 0)
55 {
56 SC_HANDLE service;
57 std::strcat(path, " --service");
58 service = CreateService(serviceControlMgr,
59 serviceName, // name of service
60 serviceLongName, // service name to display
61 SERVICE_ALL_ACCESS, // desired access
62 // service type
63 SERVICE_WIN32_OWN_PROCESS | SERVICE_INTERACTIVE_PROCESS,
64 SERVICE_AUTO_START, // start type
65 SERVICE_ERROR_IGNORE, // error control type
66 path, // service's binary
67 0, // no load ordering group
68 0, // no tag identifier
69 0, // no dependencies
70 0, // LocalSystem account
71 0); // no password
72 if (service)
73 {
74 HMODULE advapi32 = GetModuleHandle("ADVAPI32.DLL");
75 if (!advapi32)
76 {
77 CloseServiceHandle(service);
78 CloseServiceHandle(serviceControlMgr);
79 return false;
80 }
81
82 CSD_T ChangeService_Config2 = (CSD_T) GetProcAddress(advapi32, "ChangeServiceConfig2A");
83 if (!ChangeService_Config2)
84 {
85 CloseServiceHandle(service);
86 CloseServiceHandle(serviceControlMgr);
87 return false;
88 }
89
90 SERVICE_DESCRIPTION sdBuf;
91 sdBuf.lpDescription = serviceDescription;
92 ChangeService_Config2(
93 service, // handle to service
94 SERVICE_CONFIG_DESCRIPTION, // change: description
95 &sdBuf); // new data
96
97 SC_ACTION _action[1];
98 _action[0].Type = SC_ACTION_RESTART;
99 _action[0].Delay = 10000;
100 SERVICE_FAILURE_ACTIONS sfa;
101 ZeroMemory(&sfa, sizeof(SERVICE_FAILURE_ACTIONS));
102 sfa.lpsaActions = _action;
103 sfa.cActions = 1;
104 sfa.dwResetPeriod = INFINITE;
105 ChangeService_Config2(
106 service, // handle to service
107 SERVICE_CONFIG_FAILURE_ACTIONS, // information level
108 &sfa); // new data
109
110 CloseServiceHandle(service);
111 }
112 }
113 CloseServiceHandle(serviceControlMgr);
114 }
115 return true;
116}

118bool WinServiceUninstall()
119{
120 SC_HANDLE serviceControlMgr = OpenSCManager(0, 0, SC_MANAGER_CONNECT);
121
122 if (serviceControlMgr)
123 {
124 SC_HANDLE service = OpenService(serviceControlMgr,
125 serviceName, SERVICE_QUERY_STATUS | DELETE);
126 if (service)
127 {
128 SERVICE_STATUS serviceStatus2;
129 if (QueryServiceStatus(service, &serviceStatus2))
130 {
131 if (serviceStatus2.dwCurrentState == SERVICE_STOPPED)
132 {
133 DeleteService(service);
134 }
135 }
136 CloseServiceHandle(service);
137 }
138
139 CloseServiceHandle(serviceControlMgr);
140 }
141 return true;

144void WINAPI ServiceControlHandler(DWORD controlCode)
145{
146 switch (controlCode)
147 {
148 case SERVICE_CONTROL_INTERROGATE:
149 break;
150
151 case SERVICE_CONTROL_SHUTDOWN:
152 case SERVICE_CONTROL_STOP:
153 serviceStatus.dwCurrentState = SERVICE_STOP_PENDING;
154 SetServiceStatus(serviceStatusHandle, &serviceStatus);
155
156 m_ServiceStatus = 0;
157 return;
158
159 case SERVICE_CONTROL_PAUSE:
160 m_ServiceStatus = 2;
161 serviceStatus.dwCurrentState = SERVICE_PAUSED;
162 SetServiceStatus(serviceStatusHandle, &serviceStatus);
163 break;
164
165 case SERVICE_CONTROL_CONTINUE:
166 serviceStatus.dwCurrentState = SERVICE_RUNNING;
167 SetServiceStatus(serviceStatusHandle, &serviceStatus);
168 m_ServiceStatus = 1;
169 break;
170
171 default:
172 if ( controlCode >= 128 && controlCode <= 255 )
173 // user defined control code
174 {
175 break;
176 }
177 else
178 // unrecognized control code
179 {
180 break;
181 }
182 }
183
184 SetServiceStatus(serviceStatusHandle, &serviceStatus);
185}

187void WINAPI ServiceMain(DWORD argc, char* argv[])
188{
189 // initialise service status
190 serviceStatus.dwServiceType = SERVICE_WIN32;
191 serviceStatus.dwCurrentState = SERVICE_START_PENDING;
192 serviceStatus.dwControlsAccepted = SERVICE_ACCEPT_STOP | SERVICE_ACCEPT_PAUSE_CONTINUE;
193 serviceStatus.dwWin32ExitCode = NO_ERROR;
194 serviceStatus.dwServiceSpecificExitCode = NO_ERROR;
195 serviceStatus.dwCheckPoint = 0;
196 serviceStatus.dwWaitHint = 0;
197
198 serviceStatusHandle = RegisterServiceCtrlHandler(serviceName, ServiceControlHandler);
199
200 if ( serviceStatusHandle )
201 {
202 char path[_MAX_PATH + 1];
203 unsigned int i, last_slash = 0;
204
205 GetModuleFileName(0, path, sizeof(path) / sizeof(path[0]));
206
207 for (i = 0; i < std::strlen(path); i++)
208 {
209 if (path[i] == '\\') { last_slash = i; }
210 }
211
212 path[last_slash] = 0;
213
214 // service is starting
215 serviceStatus.dwCurrentState = SERVICE_START_PENDING;
216 SetServiceStatus(serviceStatusHandle, &serviceStatus);
217
218 // do initialisation here
219 SetCurrentDirectory(path);
220
221 // running
222 serviceStatus.dwControlsAccepted |= (SERVICE_ACCEPT_STOP | SERVICE_ACCEPT_SHUTDOWN);
223 serviceStatus.dwCurrentState = SERVICE_RUNNING;
224 SetServiceStatus( serviceStatusHandle, &serviceStatus );
225
227 // service main cycle //
229
230 m_ServiceStatus = 1;
231 argc = 1;
232 main(argc, argv);
233
234 // service was stopped
235 serviceStatus.dwCurrentState = SERVICE_STOP_PENDING;
236 SetServiceStatus(serviceStatusHandle, &serviceStatus);
237
238 // do cleanup here
239
240 // service is now stopped
241 serviceStatus.dwControlsAccepted &= ~(SERVICE_ACCEPT_STOP | SERVICE_ACCEPT_SHUTDOWN);
242 serviceStatus.dwCurrentState = SERVICE_STOPPED;
243 SetServiceStatus(serviceStatusHandle, &serviceStatus);
244 }
245}
