147 void SetData(uint32 type, uint32 /*data*/) override
148 {
149 switch (type)
150 {
151 case TYPE_AEONUS:
152 {
153 encounters[type] = DONE;
154 SaveToDB();
155
156 if (Creature* medivh = instance->GetCreature(_medivhGUID))
157 medivh->AI()->DoAction(ACTION_OUTRO);
158
159 Map::PlayerList const& players = instance->GetPlayers();
160 if (!players.IsEmpty())
161 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
162 if (Player* player = itr->GetSource())
163 {
164 if (player->GetQuestStatus(QUEST_OPENING_PORTAL) == QUEST_STATUS_INCOMPLETE)
165 player->AreaExploredOrEventHappens(QUEST_OPENING_PORTAL);
166
167 if (player->GetQuestStatus(QUEST_MASTER_TOUCH) == QUEST_STATUS_INCOMPLETE)
168 player->AreaExploredOrEventHappens(QUEST_MASTER_TOUCH);
169 }
170 break;
171 }
172 case TYPE_CHRONO_LORD_DEJA:
173 case TYPE_TEMPORUS:
174 encounters[type] = DONE;
175 Events.RescheduleEvent(EVENT_NEXT_PORTAL, 60000);
176 Events.SetPhase(1);
177 SaveToDB();
178 break;
179 case DATA_RIFT_KILLED:
180 if (!Events.IsInPhase(1))
181 Events.RescheduleEvent(EVENT_NEXT_PORTAL, 4000);
182 break;
183 case DATA_MEDIVH:
184 DoUpdateWorldState(WORLD_STATE_BM, 1);
185 DoUpdateWorldState(WORLD_STATE_BM_SHIELD, _shieldPercent);
186 DoUpdateWorldState(WORLD_STATE_BM_RIFT, _currentRift);
187 Events.RescheduleEvent(EVENT_NEXT_PORTAL, 3000);
188 break;
189 case DATA_DAMAGE_SHIELD:
190 --_shieldPercent;
191 DoUpdateWorldState(WORLD_STATE_BM_SHIELD, _shieldPercent);
192 if (!_shieldPercent)
193 if (Creature* medivh = instance->GetCreature(_medivhGUID))
194 if (medivh->IsAlive())
195 {
196 Unit::Kill(medivh, medivh);
197
198 // Xinef: delete all spawns
199 GuidSet eCopy = encounterNPCs;
200 for (ObjectGuid const& guid : eCopy)
201 if (Creature* creature = instance->GetCreature(guid))
202 creature->DespawnOrUnsummon();
203 }
204 break;
205 }

240 void SummonPortalKeeper()
241 {
242 Creature* rift = nullptr;
243 for (ObjectGuid const& guid : encounterNPCs)
244 if (Creature* summon = instance->GetCreature(guid))
245 if (summon->GetEntry() == NPC_TIME_RIFT)
246 {
247 rift = summon;
248 break;
249 }
250
251 if (!rift)
252 return;
253
254 int32 entry = 0;
255 switch (_currentRift)
256 {
257 case 6:
258 entry = GetData(TYPE_CHRONO_LORD_DEJA) == DONE ? (instance->IsHeroic() ? NPC_INFINITE_CHRONO_LORD : -NPC_CHRONO_LORD_DEJA) : NPC_CHRONO_LORD_DEJA;
259 break;
260 case 12:
261 entry = GetData(TYPE_TEMPORUS) == DONE ? (instance->IsHeroic() ? NPC_INFINITE_TIMEREAVER : -NPC_TEMPORUS) : NPC_TEMPORUS;
262 break;
263 case 18:
264 entry = NPC_AEONUS;
265 break;
266 default:
267 entry = RAND(NPC_RIFT_KEEPER_WARLOCK, NPC_RIFT_KEEPER_MAGE, NPC_RIFT_LORD, NPC_RIFT_LORD_2);
268 break;
269 }
270
271 Position pos = rift->GetNearPosition(10.0f, 2 * M_PI * rand_norm());
272
273 if (TempSummon* summon = instance->SummonCreature(std::abs(entry), pos))
274 {
275 summon->SetTempSummonType(TEMPSUMMON_CORPSE_TIMED_DESPAWN);
276 summon->SetTimer(3 * MINUTE * IN_MILLISECONDS);
277
278 if (entry < 0)
279 summon->SetLootMode(0);
280
281 if (summon->GetEntry() != NPC_AEONUS)
282 {
283 rift->AI()->SetGUID(summon->GetGUID());
284 rift->CastSpell(summon, SPELL_RIFT_CHANNEL, false);
285 }
286 else
287 summon->SetReactState(REACT_DEFENSIVE);
288 }
289 }

291 void Update(uint32 diff) override
292 {
293 Events.Update(diff);
294 switch (Events.ExecuteEvent())
295 {
296 case EVENT_NEXT_PORTAL:
297 ++_currentRift;
298 DoUpdateWorldState(WORLD_STATE_BM_RIFT, _currentRift);
299 Events.ScheduleEvent(EVENT_SUMMON_KEEPER, 6000);
300 Events.SetPhase(0);
301
302 if (instance->GetCreature(_medivhGUID))
303 {
304 uint8 position = (_currentRift - 1) % 4;
305 instance->SummonCreature(NPC_TIME_RIFT, PortalLocation[position]);
306 }
307 break;
308 case EVENT_SUMMON_KEEPER:
309 SummonPortalKeeper();
310 break;
311 }
