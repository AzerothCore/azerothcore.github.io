53void Appender::write(LogMessage* message)
54{
55 if (!level || level < message->level)
56 {
57 return;
58 }
59
60 std::ostringstream ss;
61
62 if (flags & APPENDER_FLAGS_PREFIX_TIMESTAMP)
63 {
64 ss << message->getTimeStr() << ' ';
65 }
66
67 if (flags & APPENDER_FLAGS_PREFIX_LOGLEVEL)
68 {
69 ss << Acore::StringFormat("%-5s ", Appender::getLogLevelString(message->level));
70 }
71
72 if (flags & APPENDER_FLAGS_PREFIX_LOGFILTERTYPE)
73 {
74 ss << '[' << message->type << "] ";
75 }
76
77 message->prefix = ss.str();
78 _write(message);
79}
