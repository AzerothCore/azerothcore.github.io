84 void UpdateAI(uint32 diff) override
85 {
86 if (!UpdateVictim())
87 return;
88
89 events.Update(diff);
90 if (me->HasUnitState(UNIT_STATE_CASTING))
91 return;
92
93 switch (events.ExecuteEvent())
94 {
95 case EVENT_SPELL_REPENTANCE:
96 me->CastSpell(me, SPELL_REPENTANCE, true);
97 events.ScheduleEvent(EVENT_SPELL_REPENTANCE, urand(25000, 35000));
98 break;
99 case EVENT_SPELL_HOLY_FIRE:
100 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 40.0f, true))
101 me->CastSpell(target, SPELL_HOLY_FIRE, true);
102 events.ScheduleEvent(EVENT_SPELL_HOLY_FIRE, urand(8000, 18000));
103 break;
104 case EVENT_SPELL_HOLY_WRATH:
105 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 40.0f, true))
106 me->CastSpell(target, SPELL_HOLY_WRATH, true);
107 events.ScheduleEvent(EVENT_SPELL_HOLY_WRATH, urand(20000, 25000));
108 break;
109 case EVENT_SPELL_ENRAGE:
110 me->CastSpell(me, SPELL_BERSERK, true);
111 break;
112 }
113
114 DoMeleeAttackIfReady();
115 }
116 };
