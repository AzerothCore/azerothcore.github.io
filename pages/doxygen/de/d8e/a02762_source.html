182 void DoAction(int32 action) override
183 {

188

193 player->AreaExploredOrEventHappens(QUEST_BALANCE_OF_LIGHT_AND_SHADOW);
194 EnterEvadeMode();
195 return;
196 }
197 else if (_deathCount > 14)
198 {
199 Talk(2);
200 if (Player* player = ObjectAccessor::GetPlayer(*me, _playerGUID))
201 player->FailQuest(QUEST_BALANCE_OF_LIGHT_AND_SHADOW);
202 EnterEvadeMode();
203 return;
204 }
205
206 if (action == 1 && !_spoken)
207 {
208 _spoken = true;
209 Talk(0);
210 me->CastSpell(me, SPELL_ERIS_BLESSING, false);
211 }
212 }
213
214 void UpdateAI(uint32 diff) override
215 {
216 events.Update(diff);
217 switch (events.ExecuteEvent())
218 {
219 case EVENT_CHECK_PLAYER:
220 {
221 Player* player = ObjectAccessor::GetPlayer(*me, _playerGUID);
222 if (!player || me->GetDistance2d(player) > 100.0f)
223 {
224 EnterEvadeMode();
225 return;
226 }
227 events.RepeatEvent(2000);
228 break;
229 }
230 case EVENT_SUMMON_ARCHERS:
231 SummonArchers();
232 break;
233 case EVENT_SUMMON_PEASANTS:
234 _spoken = false;
235 SummonPeasants();
236 _spoken = false;
237 events.RepeatEvent(60 * IN_MILLISECONDS);
238 break;
239 }
240 }

280 creature->GetAI()->DoAction(1);
281
282 me->DespawnOrUnsummon(1);
283 }
284
285 void JustDied(Unit*) override
286 {
287 if (TempSummon* summon = me->ToTempSummon())
288 if (Unit* creature = summon->GetSummonerUnit())
289 creature->GetAI()->DoAction(2);
290 }
291
292 void UpdateAI(uint32 diff) override
293 {
294 if (me->GetEntry() != NPC_SCOURGE_ARCHER)
295 return;
296
297 timer += diff;
298 if (timer >= 4000)
299 {
300 Unit* target = _targetGUID ? ObjectAccessor::GetUnit(*me, _targetGUID) : nullptr;
301 if (!target)
302 target = me->FindNearestCreature(NPC_INJURED_PEASANT, 60.0f);
303
304 if (target)
305 {
306 _targetGUID = target->GetGUID();
307 me->CastSpell(target, SPELL_SHOOT, true);
308 }
309
310 timer = urand(0, 3000);
311 }
