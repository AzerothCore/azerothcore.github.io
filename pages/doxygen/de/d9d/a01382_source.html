43void VisibleNotifier::SendToSelf()
44{
45 // at this moment i_clientGUIDs have guids that not iterate at grid level checks
46 // but exist one case when this possible and object not out of range: transports
47 if (Transport* transport = i_player.GetTransport())
48 for (Transport::PassengerSet::const_iterator itr = transport->GetPassengers().begin(); itr != transport->GetPassengers().end(); ++itr)
49 {
50 if (i_largeOnly != (*itr)->IsVisibilityOverridden())
51 continue;
52
53 if (vis_guids.find((*itr)->GetGUID()) != vis_guids.end())
54 {
55 vis_guids.erase((*itr)->GetGUID());
56
57 switch ((*itr)->GetTypeId())
58 {
59 case TYPEID_GAMEOBJECT:
60 i_player.UpdateVisibilityOf((*itr)->ToGameObject(), i_data, i_visibleNow);
61 break;
62 case TYPEID_PLAYER:
63 i_player.UpdateVisibilityOf((*itr)->ToPlayer(), i_data, i_visibleNow);
64 (*itr)->ToPlayer()->UpdateVisibilityOf(&i_player);
65 break;
66 case TYPEID_UNIT:
67 i_player.UpdateVisibilityOf((*itr)->ToCreature(), i_data, i_visibleNow);
68 break;
69 default:
70 break;
71 }
72 }
73 }
74
75 for (GuidUnorderedSet::const_iterator it = vis_guids.begin(); it != vis_guids.end(); ++it)
76 {
77 if (WorldObject* obj = ObjectAccessor::GetWorldObject(i_player, *it))
78 {
79 if (i_largeOnly != obj->IsVisibilityOverridden())
80 continue;
81 }
82
83 // pussywizard: static transports are removed only in RemovePlayerFromMap and here if can no longer detect (eg. phase changed)
84 if ((*it).IsTransport())
85 if (GameObject* staticTrans = i_player.GetMap()->GetGameObject(*it))
86 if (i_player.CanSeeOrDetect(staticTrans, false, true))
87 continue;
88
89 i_player.m_clientGUIDs.erase(*it);
90 i_data.AddOutOfRangeGUID(*it);
91
92 if ((*it).IsPlayer())
93 {
94 Player* player = ObjectAccessor::FindPlayer(*it);
95 if (player && player->IsInMap(&i_player))
96 player->UpdateVisibilityOf(&i_player);
97 }
98 }
99
100 if (!i_data.HasData())
101 return;
102
103 WorldPacket packet;
104 i_data.BuildPacket(&packet);
105 i_player.GetSession()->SendPacket(&packet);
106
107 for (std::vector<Unit*>::const_iterator it = i_visibleNow.begin(); it != i_visibleNow.end(); ++it)
108 {
109 if (i_largeOnly != (*it)->IsVisibilityOverridden())
110 continue;
111
112 i_player.GetInitialVisiblePackets(*it);
113 }
114}

151inline void CreatureUnitRelocationWorker(Creature* c, Unit* u)
152{
153 if (!u->IsAlive() || !c->IsAlive() || c == u || u->IsInFlight())
154 {
155 return;
156 }
157
158 if (!c->HasUnitState(UNIT_STATE_SIGHTLESS))
159 {
160 if (c->IsAIEnabled && c->CanSeeOrDetect(u, false, true))
161 {
162 c->AI()->MoveInLineOfSight_Safe(u);
163 }
164 else if (u->GetTypeId() == TYPEID_PLAYER && u->HasStealthAura() && c->IsAIEnabled && c->CanSeeOrDetect(u, false, true, true))
165 {
166 c->AI()->TriggerAlert(u);
167 }
168 }
169}

182void CreatureRelocationNotifier::Visit(PlayerMapType& m)
183{
184 for (PlayerMapType::iterator iter = m.begin(); iter != m.end(); ++iter)
185 {
186 Player* player = iter->GetSource();
187
188 // NOTIFY_VISIBILITY_CHANGED does not guarantee that player will do it himself (because distance is also checked), but screw it, it's not that important
189 if (!player->m_seer->isNeedNotify(NOTIFY_VISIBILITY_CHANGED))
190 player->UpdateVisibilityOf(&i_creature);
191
192 // NOTIFY_AI_RELOCATION does not guarantee that player will do it himself (because distance is also checked), but screw it, it's not that important
193 if (!player->m_seer->isNeedNotify(NOTIFY_AI_RELOCATION) && !i_creature.IsMoveInLineOfSightStrictlyDisabled())
194 CreatureUnitRelocationWorker(&i_creature, player);
195 }
196}

198void AIRelocationNotifier::Visit(CreatureMapType& m)
199{
200 bool self = isCreature && !((Creature*)(&i_unit))->IsMoveInLineOfSightStrictlyDisabled();
201 for (CreatureMapType::iterator iter = m.begin(); iter != m.end(); ++iter)
202 {
203 Creature* c = iter->GetSource();
204
205 // NOTIFY_VISIBILITY_CHANGED | NOTIFY_AI_RELOCATION does not guarantee that unit will do it itself (because distance is also checked), but screw it, it's not that important
206 if (!c->isNeedNotify(NOTIFY_VISIBILITY_CHANGED | NOTIFY_AI_RELOCATION) && !c->IsMoveInLineOfSightStrictlyDisabled())
207 CreatureUnitRelocationWorker(c, &i_unit);
208
209 if (self)
210 CreatureUnitRelocationWorker((Creature*)&i_unit, c);
211 }

214void MessageDistDeliverer::Visit(PlayerMapType& m)
215{
216 for (PlayerMapType::iterator iter = m.begin(); iter != m.end(); ++iter)
217 {
218 Player* target = iter->GetSource();
219 if (!target->InSamePhase(i_phaseMask))
220 continue;
221
222 if (target->GetExactDist2dSq(i_source) > i_distSq)
223 continue;
224
225 // Send packet to all who are sharing the player's vision
226 if (target->HasSharedVision())
227 {
228 SharedVisionList::const_iterator i = target->GetSharedVisionList().begin();
229 for (; i != target->GetSharedVisionList().end(); ++i)
230 if ((*i)->m_seer == target)
231 SendPacket(*i);
232 }
233
234 if (target->m_seer == target || target->GetVehicle())
235 SendPacket(target);
236 }
237}
