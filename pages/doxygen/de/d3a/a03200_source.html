102struct boss_amanitar : public BossAI
103{
104 boss_amanitar(Creature* creature) : BossAI(creature, DATA_AMANITAR), mushroomsSummoned(false)
105 {
106 creature->ApplySpellImmune(0, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_NATURE, true);
107 creature->SetMeleeDamageSchool(SPELL_SCHOOL_NATURE);
108 }
109
110 void Reset() override
111 {
112 _Reset();
113 _mushroomsDeque.clear();
114 mushroomsSummoned = false;
115 }
116
117 void EnterCombat(Unit* /*attacker*/) override
118 {
119 events.ScheduleEvent(EVENT_ROOTS, urand(5000, 9000));
120 events.ScheduleEvent(EVENT_BASH, urand(10000, 14000));
121 events.ScheduleEvent(EVENT_BOLT, urand(15000, 20000));
122 events.ScheduleEvent(EVENT_MINI, 1000);
123 events.ScheduleEvent(EVENT_RESPAWN, 40000, 60000);
124 }
125
126 void JustDied(Unit* /*killer*/) override
127 {
128 _JustDied();
129 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_MINI);
130 }
131
132 void JustSummoned(Creature* summon) override
133 {
134 summons.Summon(summon);
135 }
136
137 void SummonedCreatureDies(Creature* summon, Unit* killer) override
138 {
139 _mushroomsDeque.push_back(summon->GetPosition());
140 BossAI::SummonedCreatureDies(summon, killer);
141 }
142
143 void EnterEvadeMode(EvadeReason why) override
144 {
145 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_MINI);
146 BossAI::EnterEvadeMode(why);
147 }
148
149 void ExecuteEvent(uint32 eventId) override
150 {
151 switch (eventId)
152 {
153 case EVENT_RESPAWN:
154 {
155 while (!_mushroomsDeque.empty())
156 {
157 SummonMushroom(_mushroomsDeque.front());
158 _mushroomsDeque.pop_front();
159 }
160
161 events.RepeatEvent(urand(40000, 60000));
162 break;
163 }
164 case EVENT_ROOTS:
165 {
166 if (Unit* pTarget = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
167 {
168 DoCast(pTarget, SPELL_ENTANGLING_ROOTS, false);
169 }
170
171 events.RepeatEvent(urand(10000, 15000));
172 break;
173 }
174 case EVENT_BASH:
175 {
176 DoCastVictim(SPELL_BASH, false);
177 events.RepeatEvent(urand(15000, 20000));
178 break;
179 }
180 case EVENT_BOLT:
181 {
182 if (Unit* pTarget = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
183 {
184 DoCast(pTarget, SPELL_VENOM_BOLT_VOLLEY, false);
185 }
186
187 events.RepeatEvent(urand(15000, 20000));
188 break;
189 }
190 case EVENT_REMOVE_MUSHROOM_POWER:
191 {
192 DoCastAOE(SPELL_REMOVE_MUSHROOM_POWER, true);
193 events.RescheduleEvent(EVENT_MINI, 1000);
194 break;
195 }
196 case EVENT_MINI:
197 {
198 if (!mushroomsSummoned)
199 {
200 mushroomsSummoned = true;
201 for (uint8 i = 0; i < MAX_MUSHROOMS_COUNT; ++i)
202 {
203 SummonMushroom(MushroomPositions[i]);
204 }
205 }
206
207 if (SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true, -SPELL_MINI))
208 {
209 DoCastSelf(SPELL_REMOVE_MUSHROOM_POWER, true);
210 DoCastAOE(SPELL_MINI);
211 events.RescheduleEvent(EVENT_REMOVE_MUSHROOM_POWER, 29000);
212 }
213 else
214 {
215 events.RepeatEvent(1000);
216 }
217
218 break;
219 }
220 }
221 }
222
223private:
224 std::deque<Position> _mushroomsDeque;
225 bool mushroomsSummoned;
226
227 void SummonMushroom(Position const& pos)
228 {
229 me->SummonCreature(roll_chance_i(40) ? NPC_HEALTHY_MUSHROOM : NPC_POISONOUS_MUSHROOM, pos, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 4000);
230 }
231};

233struct npc_amanitar_mushrooms : public ScriptedAI
234{
235 npc_amanitar_mushrooms(Creature* pCreature) : ScriptedAI(pCreature)
236 {
237 SetCombatMovement(false);
238
239 //TODO: this prolly needs to be done in database
240 pCreature->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
241 pCreature->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
242 pCreature->SetRegeneratingHealth(false);
243 }
244
245 // Disabled events
246 void EnterCombat(Unit* /*who*/) override {}
247 void AttackStart(Unit* /*victim*/) override {}
248 void EnterEvadeMode(EvadeReason /*why*/) override {}
249
250 void Reset() override
251 {
252 me->SetReactState(REACT_PASSIVE);
253 DoCastSelf(SPELL_PUTRID_MUSHROOM);
254
255 if (me->GetEntry() == NPC_POISONOUS_MUSHROOM)
256 {
257 DoCastSelf(SPELL_POISONOUS_MUSHROOM_VISUAL_AURA, true);
258 }
259 else
260 {
261 DoCastSelf(SPELL_HEALTHY_MUSHROOM_VISUAL_AURA, true);
262 }
263
264 events.ScheduleEvent(EVENT_GROW, 800);
265
266 if (me->GetEntry() == NPC_POISONOUS_MUSHROOM)
267 {
268 events.ScheduleEvent(EVENT_CHECK_PLAYER, 250);
269 }
270 }
271
272 void DamageTaken(Unit* /*attacker*/, uint32& damage, DamageEffectType /*damagetype*/, SpellSchoolMask /*damageSchoolMask*/) override
273 {
274 if (me->GetEntry() == NPC_HEALTHY_MUSHROOM && damage >= me->GetHealth())
275 {
276 DoCastSelf(SPELL_HEALTHY_MUSHROOM_POTENT_FUNGUS, true);
277 }
278 }
279
280 void UpdateAI(uint32 diff) override
281 {
282 if (events.Empty())
283 return;
284
285 events.Update(diff);
286 while (uint32 const eventId = events.ExecuteEvent())
287 {
288 switch (eventId)
289 {
290 case EVENT_GROW:
291 {
292 DoCastSelf(SPELL_GROW);
293 break;
294 }
295 case EVENT_CHECK_PLAYER:
296 {
297 if (Player* plr = me->SelectNearestPlayer(2.0f))
298 {
299 plr->RemoveAurasDueToSpell(SPELL_HEALTHY_MUSHROOM_POTENT_FUNGUS);
300 DoCastSelf(SPELL_POISONOUS_MUSHROOM_VISUAL_AREA);
301 DoCastSelf(SPELL_POISONOUS_MUSHROOM_POISON_CLOUD);
302 DoCastSelf(SPELL_SHRINK);
303 events.ScheduleEvent(EVENT_KILLSELF, 4000);
304 }
305 else
306 {
307 events.RepeatEvent(250);
308 }
309
310 break;
311 }
312 case EVENT_KILLSELF:
313 {
314 me->DisappearAndDie();
315 break;
316 }
317 }
318 }
319 }
320
321private:
322 EventMap events;
323};
