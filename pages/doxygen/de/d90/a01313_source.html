47{

56
57 Object::_Create(guidlow, 0, HighGuid::Mo_Transport);
58
59 GameObjectTemplate const* goinfo = sObjectMgr->GetGameObjectTemplate(entry);

68

70 if (!tInfo)
71 {
72 LOG_ERROR("entities.transport", "Transport {} (name: {}) will not be created, missing `transport_template` entry.", entry, goinfo->name);

74 }
75
76 _transportInfo = tInfo;
77
78 // initialize waypoints
79 _nextFrame = tInfo->keyFrames.begin();
80 _currentFrame = _nextFrame++;
81 _triggeredArrivalEvent = false;
82 _triggeredDepartureEvent = false;
83
84 if (GameObjectTemplateAddon const* addon = GetTemplateAddon())
85 {
86 SetUInt32Value(GAMEOBJECT_FACTION, addon->faction);
87 ReplaceAllGameObjectFlags((GameObjectFlags)addon->flags);

89
90 SetObjectScale(goinfo->size);
91 SetPathProgress(0);

137void MotionTransport::Update(uint32 diff)
138{
139 uint32 const positionUpdateDelay = 1;
140
141 if (AI())
142 AI()->UpdateAI(diff);
143 else if (!AIM_Initialize())
144 LOG_ERROR("entities.transport", "Could not initialize GameObjectAI for Transport");
145
146 if (GetKeyFrames().size() <= 1)
147 return;
148
149 if (IsMoving() || !_pendingStop)
150 SetPathProgress(GetPathProgress() + diff);
151
152 uint32 timer = GetPathProgress() % GetPeriod();
153
154 // Set current waypoint
155 // Desired outcome: _currentFrame->DepartureTime < timer < _nextFrame->ArriveTime
156 // ... arrive | ... delay ... | departure
157 // event / event /
158 for (;;)
159 {
160 if (timer >= _currentFrame->ArriveTime)
161 {
162 if (!_triggeredArrivalEvent)
163 {
164 DoEventIfAny(*_currentFrame, false);
165 _triggeredArrivalEvent = true;
166 }
167
168 if (timer < _currentFrame->DepartureTime)
169 {
170 SetMoving(false);
171 if (_pendingStop && GetGoState() != GO_STATE_READY)
172 {
173 SetGoState(GO_STATE_READY);
174 SetPathProgress(GetPathProgress() / GetPeriod());
175 SetPathProgress(GetPathProgress() * GetPeriod());
176 SetPathProgress(GetPathProgress() + _currentFrame->ArriveTime);
177 }
178 break; // its a stop frame and we are waiting
179 }
180 }
181
182 if (timer >= _currentFrame->DepartureTime && !_triggeredDepartureEvent)
183 {
184 DoEventIfAny(*_currentFrame, true); // departure event
185 _triggeredDepartureEvent = true;
186 }
187
188 // not waiting anymore
189 SetMoving(true);
190
191 // Enable movement
192 if (GetGOInfo()->moTransport.canBeStopped)
193 SetGoState(GO_STATE_ACTIVE);
194
195 if (timer >= _currentFrame->DepartureTime && timer < _currentFrame->NextArriveTime)
196 break; // found current waypoint
197
198 MoveToNextWaypoint();
199
200 sScriptMgr->OnRelocate(this, _currentFrame->Node->index, _currentFrame->Node->mapid, _currentFrame->Node->x, _currentFrame->Node->y, _currentFrame->Node->z);
201
202 //LOG_DEBUG("entities.transport", "Transport {} ({}) moved to node {} {} {} {} {}", GetEntry(), GetName(), _currentFrame->Node->index, _currentFrame->Node->mapid, _currentFrame->Node->x, _currentFrame->Node->y, _currentFrame->Node->z);
203
204 // Departure event
205 if (_currentFrame->IsTeleportFrame())
206 if (TeleportTransport(_nextFrame->Node->mapid, _nextFrame->Node->x, _nextFrame->Node->y, _nextFrame->Node->z, _nextFrame->InitialOrientation))
207 return; // Update more in new map thread
208 }
209
210 // Set position
211 _positionChangeTimer.Update(diff);
212 if (_positionChangeTimer.Passed())
213 {
214 _positionChangeTimer.Reset(positionUpdateDelay);
215 if (IsMoving())
216 {
217 float t = CalculateSegmentPos(float(timer) * 0.001f);
218 G3D::Vector3 pos, dir;
219 _currentFrame->Spline->evaluate_percent(_currentFrame->Index, t, pos);
220 _currentFrame->Spline->evaluate_derivative(_currentFrame->Index, t, dir);
221 UpdatePosition(pos.x, pos.y, pos.z, NormalizeOrientation(std::atan2(dir.y, dir.x) + M_PI));
222 }
223 else
224 {
225 /* There are four possible scenarios that trigger loading/unloading passengers:
226 1. transport moves from inactive to active grid
227 2. the grid that transport is currently in becomes active
228 3. transport moves from active to inactive grid
229 4. the grid that transport is currently in unloads
230 */
231 if (_staticPassengers.empty() && GetMap()->IsGridLoaded(GetPositionX(), GetPositionY())) // 2.
232 LoadStaticPassengers();
233 }
234 }
235
236 sScriptMgr->OnTransportUpdate(this, diff);
237}

311 passenger->ToUnit()->ClearUnitState(UNIT_STATE_IGNORE_PATHFINDING);
312 }
313 }

474float MotionTransport::CalculateSegmentPos(float now)
475{
476 KeyFrame const& frame = *_currentFrame;
477 const float speed = float(m_goInfo->moTransport.moveSpeed);
478 const float accel = float(m_goInfo->moTransport.accelRate);
479 float timeSinceStop = frame.TimeFrom + (now - (1.0f / IN_MILLISECONDS) * frame.DepartureTime);
480 float timeUntilStop = frame.TimeTo - (now - (1.0f / IN_MILLISECONDS) * frame.DepartureTime);
481 float segmentPos, dist;
482 float accelTime = _transportInfo->accelTime;
483 float accelDist = _transportInfo->accelDist;
484 // calculate from nearest stop, less confusing calculation...
485 if (timeSinceStop < timeUntilStop)
486 {
487 if (timeSinceStop < accelTime)
488 dist = 0.5f * accel * timeSinceStop * timeSinceStop;
489 else
490 dist = accelDist + (timeSinceStop - accelTime) * speed;
491 segmentPos = dist - frame.DistSinceStop;
492 }
493 else
494 {
495 if (timeUntilStop < _transportInfo->accelTime)
496 dist = 0.5f * accel * timeUntilStop * timeUntilStop;
497 else
498 dist = accelDist + (timeUntilStop - accelTime) * speed;
499 segmentPos = frame.DistUntilStop - dist;
500 }
501
502 return segmentPos / frame.NextDistFromPrev;
503}

505bool MotionTransport::TeleportTransport(uint32 newMapid, float x, float y, float z, float o)
506{
507 Map const* oldMap = GetMap();
508
509 if (oldMap->GetId() != newMapid)
510 {
511 _delayedTeleport = true;
512 UnloadStaticPassengers();
513 return true;
514 }
515 else
516 {
517 // Teleport players, they need to know it
518 for (PassengerSet::iterator itr = _passengers.begin(); itr != _passengers.end(); ++itr)
519 {
520 if ((*itr)->GetTypeId() == TYPEID_PLAYER)
521 {
522 float destX, destY, destZ, destO;
523 (*itr)->m_movementInfo.transport.pos.GetPosition(destX, destY, destZ, destO);
524 TransportBase::CalculatePassengerPosition(destX, destY, destZ, &destO, x, y, z, o);
525
526 (*itr)->ToUnit()->NearTeleportTo(destX, destY, destZ, destO);
527 }
528 }
529
530 UpdatePosition(x, y, z, o);
531 return false;
532 }
533}

535void MotionTransport::DelayedTeleportTransport()
536{
537 if (!_delayedTeleport)
538 return;
539
540 _delayedTeleport = false;
541
542 uint32 newMapId = _nextFrame->Node->mapid;
543 float x = _nextFrame->Node->x,
544 y = _nextFrame->Node->y,
545 z = _nextFrame->Node->z,
546 o = _nextFrame->InitialOrientation;
547
548 PassengerSet _passengersCopy = _passengers;
549 for (PassengerSet::iterator itr = _passengersCopy.begin(); itr != _passengersCopy.end(); )
550 {
551 WorldObject* obj = (*itr++);
552
553 if (_passengers.find(obj) == _passengers.end())
554 continue;
555
556 switch (obj->GetTypeId())
557 {
558 case TYPEID_UNIT:
559 _passengers.erase(obj);
560 if (!obj->ToCreature()->IsPet())
561 obj->ToCreature()->DespawnOrUnsummon();
562 break;
563 case TYPEID_GAMEOBJECT:
564 _passengers.erase(obj);
565 obj->ToGameObject()->Delete();
566 break;
567 case TYPEID_DYNAMICOBJECT:
568 _passengers.erase(obj);
569 if (Unit* caster = obj->ToDynObject()->GetCaster())
570 if (Spell* s = caster->GetCurrentSpell(CURRENT_CHANNELED_SPELL))
571 if (obj->ToDynObject()->GetSpellId() == s->GetSpellInfo()->Id)
572 {
573 s->SendChannelUpdate(0);
574 s->SendInterrupted(0);
575 caster->RemoveOwnedAura(s->GetSpellInfo()->Id, caster->GetGUID());
576 }
577 obj->AddObjectToRemoveList();
578 break;
579 case TYPEID_PLAYER:
580 {
581 float destX, destY, destZ, destO;
582 obj->m_movementInfo.transport.pos.GetPosition(destX, destY, destZ, destO);
583 TransportBase::CalculatePassengerPosition(destX, destY, destZ, &destO, x, y, z, o);
584 if (!obj->ToPlayer()->TeleportTo(newMapId, destX, destY, destZ, destO, TELE_TO_NOT_LEAVE_TRANSPORT))
585 _passengers.erase(obj);
586 }
587 break;
588 default:
589 break;
590 }
591 }
592
593 Map* newMap = sMapMgr->CreateBaseMap(newMapId);
594 GetMap()->RemoveFromMap<MotionTransport>(this, false);
595 newMap->LoadGrid(x, y); // xinef: load before adding passengers to new map
596 SetMap(newMap);
597
598 Relocate(x, y, z, o);
599 GetMap()->AddToMap<MotionTransport>(this);
600
601 LoadStaticPassengers();
602}

604void MotionTransport::UpdatePassengerPositions(PassengerSet& passengers)
605{
606 for (PassengerSet::iterator itr = passengers.begin(); itr != passengers.end(); ++itr)
607 {
608 WorldObject* passenger = *itr;
609 // transport teleported but passenger not yet (can happen for players)
610 if (passenger->GetMap() != GetMap())
611 continue;
612
613 // if passenger is on vehicle we have to assume the vehicle is also on transport and its the vehicle that will be updating its passengers
614 if (Unit* unit = passenger->ToUnit())
615 if (unit->GetVehicle())
616 continue;
617
618 // Do not use Unit::UpdatePosition here, we don't want to remove auras as if regular movement occurred
619 float x, y, z, o;
620 passenger->m_movementInfo.transport.pos.GetPosition(x, y, z, o);
621 CalculatePassengerPosition(x, y, z, &o);
622
623 // check if position is valid
624 if (!Acore::IsValidMapCoord(x, y, z))
625 continue;
626
627 switch (passenger->GetTypeId())
628 {
629 case TYPEID_UNIT:
630 {
631 Creature* creature = passenger->ToCreature();
632 GetMap()->CreatureRelocation(creature, x, y, z, o);
633
634 creature->GetTransportHomePosition(x, y, z, o);
635 CalculatePassengerPosition(x, y, z, &o);
636 creature->SetHomePosition(x, y, z, o);
637 }
638 break;
639 case TYPEID_PLAYER:
640 if (passenger->IsInWorld())
641 GetMap()->PlayerRelocation(passenger->ToPlayer(), x, y, z, o);
642 break;
643 case TYPEID_GAMEOBJECT:
644 GetMap()->GameObjectRelocation(passenger->ToGameObject(), x, y, z, o);
645 break;
646 case TYPEID_DYNAMICOBJECT:
647 GetMap()->DynamicObjectRelocation(passenger->ToDynObject(), x, y, z, o);
648 break;
649 default:
650 break;
651 }
652 }

677bool StaticTransport::Create(ObjectGuid::LowType guidlow, uint32 name_id, Map* map, uint32 phaseMask, float x, float y, float z, float ang, G3D::Quat const& rotation, uint32 animprogress, GOState go_state, uint32 artKit)
678{
679 ASSERT(map);
680 SetMap(map);
681
682 Relocate(x, y, z, ang);
683 m_stationaryPosition.Relocate(x, y, z, ang);
684 if (!IsPositionValid())
685 {
686 LOG_ERROR("entities.transport", "Gameobject (GUID: {} Entry: {}) not created. Suggested coordinates isn't valid (X: {} Y: {})", guidlow, name_id, x, y);
687 return false;
688 }
689
690 SetPhaseMask(phaseMask, false);
691
692 UpdatePositionData();
693
694 SetZoneScript();
695 if (m_zoneScript)
696 {
697 name_id = m_zoneScript->GetGameObjectEntry(guidlow, name_id);
698 if (!name_id)
699 return false;
700 }
701
702 GameObjectTemplate const* goinfo = sObjectMgr->GetGameObjectTemplate(name_id);
703 if (!goinfo)
704 {
705 LOG_ERROR("sql.sql", "Gameobject (GUID: {} Entry: {}) not created: non-existing entry in `gameobject_template`. Map: {} (X: {} Y: {} Z: {})", guidlow, name_id, map->GetId(), x, y, z);
706 return false;
707 }
708
709 Object::_Create(guidlow, 0, HighGuid::Transport);
710
711 m_goInfo = goinfo;
712

811void StaticTransport::Update(uint32 diff)
812{
813 GameObject::Update(diff);
814
815 if (!IsInWorld())
816 return;
817
818 if (!m_goValue.Transport.AnimationInfo)
819 return;
820

822 {
823 _needDoInitialRelocation = false;
824 RelocateToProgress(GetPathProgress());

831 if (GetPathProgress() == 0) // waiting at it's destination for state change, do nothing
832 return;

834 if (GetPathProgress() < GetPauseTime()) // GOState has changed before previous state was reached, move to new destination immediately
835 SetPathProgress(0);
836 else if (GetPathProgress() + diff < GetPeriod())
837 SetPathProgress(GetPathProgress() + diff);
838 else
839 SetPathProgress(0);
840 }
841 else

843 if (GetPathProgress() == GetPauseTime()) // waiting at it's destination for state change, do nothing

847 SetPathProgress(GetPauseTime());
848 else if (GetPathProgress() + diff < GetPauseTime())
849 SetPathProgress(GetPathProgress() + diff);
850 else
851 SetPathProgress(GetPauseTime());
852 }
853 }
854 else
855 {
856 SetPathProgress(GetPathProgress() + diff);
857 if (GetPathProgress() >= GetPeriod())
858 SetPathProgress(GetPathProgress() % GetPeriod());

861 RelocateToProgress(GetPathProgress());

864void StaticTransport::RelocateToProgress(uint32 progress)
865{
866 TransportAnimationEntry const* curr = nullptr, *next = nullptr;
867 float percPos;
868 if (m_goValue.Transport.AnimationInfo->GetAnimNode(progress, curr, next, percPos))
869 {
870 // curr node offset
871 G3D::Vector3 pos = G3D::Vector3(curr->X, curr->Y, curr->Z);
872
873 // move by percentage of segment already passed
874 pos += G3D::Vector3(percPos * (next->X - curr->X), percPos * (next->Y - curr->Y), percPos * (next->Z - curr->Z));
875
876 // rotate path by PathRotation
877 // pussywizard: PathRotation in db is only simple orientation rotation, so don't use sophisticated and not working code
878 // reminder: WorldRotation only influences model rotation, not the path
879 float sign = GetFloatValue(GAMEOBJECT_PARENTROTATION + 2) >= 0.0f ? 1.0f : -1.0f;
880 float pathRotAngle = sign * 2.0f * acos(GetFloatValue(GAMEOBJECT_PARENTROTATION + 3));
881 float cs = cos(pathRotAngle), sn = std::sin(pathRotAngle);
882 float nx = pos.x * cs - pos.y * sn;
883 float ny = pos.x * sn + pos.y * cs;
884 pos.x = nx;
885 pos.y = ny;
886
887 // add stationary position to the calculated offset
888 pos += G3D::Vector3(GetStationaryX(), GetStationaryY(), GetStationaryZ());
889
890 // rotate by AnimRotation at current segment
891 // pussywizard: AnimRotation in dbc is only simple orientation rotation, so don't use sophisticated and not working code
892 G3D::Quat currRot, nextRot;
893 float percRot;
894 m_goValue.Transport.AnimationInfo->GetAnimRotation(progress, currRot, nextRot, percRot);
895 float signCurr = currRot.z >= 0.0f ? 1.0f : -1.0f;
896 float oriRotAngleCurr = signCurr * 2.0f * acos(currRot.w);
897 float signNext = nextRot.z >= 0.0f ? 1.0f : -1.0f;
898 float oriRotAngleNext = signNext * 2.0f * acos(nextRot.w);
899 float oriRotAngle = oriRotAngleCurr + percRot * (oriRotAngleNext - oriRotAngleCurr);
900
901 // check if position is valid
902 if (!Acore::IsValidMapCoord(pos.x, pos.y, pos.z))
903 return;
904
905 // update position to new one
906 // also adding simplified orientation rotation here
907 UpdatePosition(pos.x, pos.y, pos.z, NormalizeOrientation(GetStationaryO() + oriRotAngle));
908 }
909}

920void StaticTransport::UpdatePassengerPositions()
921{
922 for (PassengerSet::iterator itr = _passengers.begin(); itr != _passengers.end(); ++itr)
923 {
924 WorldObject* passenger = *itr;
925
926 // if passenger is on vehicle we have to assume the vehicle is also on transport and its the vehicle that will be updating its passengers
927 if (Unit* unit = passenger->ToUnit())
928 if (unit->GetVehicle())
929 continue;
930
931 // Do not use Unit::UpdatePosition here, we don't want to remove auras as if regular movement occurred
932 float x, y, z, o;
933 passenger->m_movementInfo.transport.pos.GetPosition(x, y, z, o);
934 CalculatePassengerPosition(x, y, z, &o);
935
936 // check if position is valid
937 if (!Acore::IsValidMapCoord(x, y, z))
938 continue;
939
940 switch (passenger->GetTypeId())
941 {
942 case TYPEID_UNIT:
943 GetMap()->CreatureRelocation(passenger->ToCreature(), x, y, z, o);
944 break;
945 case TYPEID_PLAYER:
946 if (passenger->IsInWorld())
947 {
948 GetMap()->PlayerRelocation(passenger->ToPlayer(), x, y, z, o);
949 passenger->ToPlayer()->SetFallInformation(GameTime::GetGameTime().count(), z);
950 }
951 break;
952 case TYPEID_GAMEOBJECT:
953 GetMap()->GameObjectRelocation(passenger->ToGameObject(), x, y, z, o);
954 break;
955 case TYPEID_DYNAMICOBJECT:
956 GetMap()->DynamicObjectRelocation(passenger->ToDynObject(), x, y, z, o);
957 break;
958 default:
959 break;
960 }
961 }
