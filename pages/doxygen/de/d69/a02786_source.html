230 void SummonCreatureWithRandomTarget(uint32 creatureId, int position)
231 {
232 if (Creature* summoned = me->SummonCreature(creatureId, PyrewoodSpawnPoints[position][0], PyrewoodSpawnPoints[position][1], PyrewoodSpawnPoints[position][2], PyrewoodSpawnPoints[position][3], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 15000))
233 {
234 Unit* target = nullptr;
235 if (PlayerGUID)
236 if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID))
237 if (player->IsAlive() && RAND(0, 1))
238 target = player;
239
240 if (!target)
241 target = me;
242
243 summoned->SetFaction(FACTION_STORMWIND);
244 summoned->AddThreat(target, 32.0f);
245 summoned->AI()->AttackStart(target);
246 }
247 }

254 player->FailQuest(QUEST_PYREWOOD_AMBUSH);
255 }

257 void UpdateAI(uint32 diff) override
258 {
259 //LOG_INFO("scripts", "DEBUG: p({}) k({}) d({}) W({})", Phase, KillCount, diff, WaitTimer);
260
261 if (!QuestInProgress)
262 return;
263
264 if (KillCount && Phase < 6)
265 {
266 if (!UpdateVictim()) //reset() on target Despawn...
267 return;
268
269 DoMeleeAttackIfReady();
270 return;
271 }
272
273 switch (Phase)
274 {
275 case 0:
276 if (WaitTimer == WAIT_SECS)
277 {
278 if (PlayerGUID)
279 {
280 if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID))
281 {
282 me->AI()->Talk(NPCSAY_INIT, player);
283 }
284 }
285 }
286 if (WaitTimer <= diff)
287 {
288 WaitTimer -= diff;
289 return;
290 }
291 break;
292 case 1:
293 SummonCreatureWithRandomTarget(2060, 1);
294 break;
295 case 2:
296 SummonCreatureWithRandomTarget(2061, 2);
297 SummonCreatureWithRandomTarget(2062, 0);
298 break;
299 case 3:
300 SummonCreatureWithRandomTarget(2063, 1);
301 SummonCreatureWithRandomTarget(2064, 2);
302 SummonCreatureWithRandomTarget(2065, 0);
303 break;
304 case 4:
305 SummonCreatureWithRandomTarget(2066, 1);
306 SummonCreatureWithRandomTarget(2066, 1);
307 SummonCreatureWithRandomTarget(2067, 0);
308 SummonCreatureWithRandomTarget(2068, 2);
309 break;
310 case 5: //end
311 if (PlayerGUID)
312 {
313 if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID))
314 {
315 me->AI()->Talk(NPCSAY_END, player);
316 player->GroupEventHappens(QUEST_PYREWOOD_AMBUSH, me);
317 }
318 }
319 QuestInProgress = false;
320 Reset();
321 break;
322 }
323 ++Phase; //prepare next phase
324 }
325 };
