41 void OnCreatureCreate(Creature* creature) override
42 {
43 switch (creature->GetEntry())
44 {
45 case NPC_IMMOL_THAR:
46 _immoltharGUID = creature->GetGUID();
47 if (_pylonsState == ALL_PYLONS_OFF)
48 {
49 creature->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
50 }
51 else
52 {
53 creature->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
54 }
55 break;
56 case NPC_HIGHBORNE_SUMMONER:
57 if (_pylonsState == ALL_PYLONS_OFF)
58 {
59 creature->DespawnOrUnsummon(5000);
60 }
61 else
62 {
63 creature->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
64 }
65 HighborneSummoners.push_back(creature->GetGUID());
66 break;
67 }
68 }

70 void OnGameObjectCreate(GameObject* gameobject) override
71 {
72 switch (gameobject->GetEntry())
73 {
74 case GO_DIRE_MAUL_FORCE_FIELD:
75 if (_pylonsState == ALL_PYLONS_OFF)
76 gameobject->SetGoState(GO_STATE_ACTIVE);
77 break;
78 case GO_GORDOK_TRIBUTE:
79 {
80 uint32 fullLootMode = 0x3F;
81 for (uint32 i = 0; i < _northWingBosses; ++i)
82 fullLootMode >>= 1;
83
84 gameobject->SetLootMode(fullLootMode);
85 break;
86 }
87 }

90 void SetData(uint32 type, uint32 data) override
91 {
92 switch (type)
93 {
94 case TYPE_EAST_WING_PROGRESS:
95 _eastWingProgress = data;
96 instance->LoadGrid(-56.59f, -269.12f);
97 break;
98 case TYPE_WEST_WING_PROGRESS:
99 _westWingProgress = data;
100 instance->LoadGrid(132.626f, 625.913f);
101 break;
102 case TYPE_NORTH_WING_PROGRESS:
103 _northWingProgress = data;
104 break;
105 case TYPE_NORTH_WING_BOSSES:
106 _northWingBosses |= (1 << _northWingBosses);
107 break;
108 case TYPE_PYLONS_STATE:
109 if (_pylonsState & data)
110 return;
111 _pylonsState |= data;
112 if (_pylonsState == ALL_PYLONS_OFF) // all five active, 31
113 {
114 instance->LoadGrid(-38.08f, 812.44f);
115 if (Creature* immol = instance->GetCreature(_immoltharGUID))
116 {
117 immol->setActive(true);
118 immol->GetAI()->SetData(1, 1);
119 immol->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
120 }
121 for (const auto& guid : HighborneSummoners)
122 {
123 if (Creature* summoner = instance->GetCreature(guid))
124 {
125 summoner->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
126 }
127 }
128 }
129 break;
130 }
131
132 SaveToDB();
133 }

154 void Load(const char* in) override
155 {
156 if (!in)
157 return;
158
159 char dataHead1, dataHead2;
160 std::istringstream loadStream(in);
161 loadStream >> dataHead1 >> dataHead2;
162 if (dataHead1 == 'D' && dataHead2 == 'M')
163 {
164 loadStream >> _eastWingProgress;
165 loadStream >> _westWingProgress;
166 loadStream >> _pylonsState;
167 loadStream >> _northWingProgress;
168 loadStream >> _northWingBosses;
169 }
