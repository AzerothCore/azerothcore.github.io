68 void Reset() override
69 {
70 BossAI::Reset();
71 events2.Reset();
72 events2.ScheduleEvent(EVENT_SPAWN_INVADERS, 30000);
73 }
74
75 void EnterCombat(Unit* who) override
76 {
77 events.ScheduleEvent(EVENT_SPELL_INFECTED_WOUND, urand(6000, 10000));
78 events.ScheduleEvent(EVENT_SPELL_CRUSH, urand(3000, 5000));
79 events.ScheduleEvent(EVENT_SPELL_CONSUME, 15000);
80 events.ScheduleEvent(EVENT_SPELL_CORPSE_EXPLODE, 35000);
81 events.ScheduleEvent(EVENT_SPAWN_INVADERS, 20000, 30000);
82
83 me->setActive(true);
84 instance->SetBossState(DATA_TROLLGORE, IN_PROGRESS);
85 if (who->GetTypeId() == TYPEID_PLAYER)
86 {
87 Talk(SAY_AGGRO);
88 me->SetInCombatWithZone();
89 }
90 }
91
92 void JustDied(Unit* killer) override
93 {
94 Talk(SAY_DEATH);
95 BossAI::JustDied(killer);
96 }
97
98 void KilledUnit(Unit* /*victim*/) override
99 {
100 if (events.GetNextEventTime(EVENT_KILL_TALK) == 0)
101 {
102 Talk(SAY_KILL);
103 events.ScheduleEvent(EVENT_KILL_TALK, 6000);
104 }
105 }
106
107 void JustSummoned(Creature* summon) override
108 {
109 summons.Summon(summon);
110 }
111
112 void UpdateAI(uint32 diff) override
113 {
114 events2.Update(diff);
115 switch (events2.ExecuteEvent())
116 {
117 case EVENT_SPAWN_INVADERS:
118 me->CastSpell(me, SPELL_SUMMON_INVADER_A, true);
119 me->CastSpell(me, SPELL_SUMMON_INVADER_B, true);
120 me->CastSpell(me, SPELL_SUMMON_INVADER_C, true);
121 events2.ScheduleEvent(EVENT_SPAWN_INVADERS, 30000);
122 break;
123 }
124
125 if (!UpdateVictim())
126 return;
127
128 events.Update(diff);
129 if (me->HasUnitState(UNIT_STATE_CASTING))
130 return;
131
132 switch(events.ExecuteEvent())
133 {
134 case EVENT_SPELL_INFECTED_WOUND:
135 me->CastSpell(me->GetVictim(), SPELL_INFECTED_WOUND, false);
136 events.ScheduleEvent(EVENT_SPELL_INFECTED_WOUND, urand(25000, 35000));
137 break;
138 case EVENT_SPELL_CRUSH:
139 me->CastSpell(me->GetVictim(), SPELL_CRUSH, false);
140 events.ScheduleEvent(EVENT_SPELL_CRUSH, urand(10000, 15000));
141 break;
142 case EVENT_SPELL_CONSUME:
143 Talk(SAY_CONSUME);
144 me->CastSpell(me, SPELL_CONSUME, false);
145 events.ScheduleEvent(EVENT_SPELL_CONSUME, 15000);
146 break;
147 case EVENT_SPELL_CORPSE_EXPLODE:
148 Talk(SAY_EXPLODE);
149 me->CastSpell(me, SPELL_CORPSE_EXPLODE, false);
150 events.ScheduleEvent(EVENT_SPELL_CORPSE_EXPLODE, urand(15000, 19000));
151 break;
152 }
153
154 DoMeleeAttackIfReady();
155 }
156
157 bool CheckEvadeIfOutOfCombatArea() const override
158 {
159 return me->GetHomePosition().GetExactDist2d(me) > 60.0f;
160 }
161
162 private:
163 EventMap events2;
164 };
165
166 CreatureAI* GetAI(Creature* creature) const override
167 {
168 return GetDraktharonKeepAI<boss_trollgoreAI>(creature);
169 }
