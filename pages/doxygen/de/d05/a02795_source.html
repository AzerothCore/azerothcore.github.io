130 void UpdateAI(uint32 diff) override
131 {
132 if (m_uiPhase)
133 {
134 if (m_uiPhaseTimer <= diff)

137 {

140 }
141

143 {
144 case 1:
145 Talk(SAY_COMPLETE);
146 ++m_uiPhase;
147 break;
148 case 2:
149 if (Player* player = ObjectAccessor::GetPlayer(*me, m_uiPlayerGUID))
150 player->AreaExploredOrEventHappens(QUEST_590);
151
152 DoCast(me, SPELL_DRINK, true);
153 ++m_uiPhase;
154 break;
155 case 3:
156 EnterEvadeMode();
157 break;
158 }
159
