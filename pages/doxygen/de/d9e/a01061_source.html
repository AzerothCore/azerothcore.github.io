32// Checks if object meets the condition
33// Can have CONDITION_SOURCE_TYPE_NONE && !mReferenceId if called from a special event (ie: eventAI)
34bool Condition::Meets(ConditionSourceInfo& sourceInfo)
35{
36 // ASSERT(ConditionTarget < MAX_CONDITION_TARGETS);
37 if (ConditionTarget >= MAX_CONDITION_TARGETS)
38 {
39 LOG_ERROR("condition", "ConditionTarget {} for for condition (Entry: {} Type: {} Group: {}) is greater or equal than MAX_CONDITION_TARGETS", ConditionTarget, SourceEntry, SourceType, SourceGroup);
40 return false;
41 }
42
43 WorldObject* object = sourceInfo.mConditionTargets[ConditionTarget];
44 // object not present, return false
45 if (!object)
46 {
47 LOG_DEBUG("condition", "Condition object not found for condition (Entry: {} Type: {} Group: {})", SourceEntry, SourceType, SourceGroup);
48 return false;
49 }
50 bool condMeets = false;
51 switch (ConditionType)
52 {
53 case CONDITION_NONE:
54 condMeets = true; // empty condition, always met
55 break;
56 case CONDITION_AURA:
57 {
58 if (Unit* unit = object->ToUnit())
59 condMeets = unit->HasAuraEffect(ConditionValue1, ConditionValue2);
60 break;
61 }
62 case CONDITION_ITEM:
63 {
64 if (Unit* unit = object->ToUnit())
65 {
66 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
67 {
68 // don't allow 0 items (it's checked during table load)
69 ASSERT(ConditionValue2);
70 bool checkBank = !!ConditionValue3;
71 condMeets = player->HasItemCount(ConditionValue1, ConditionValue2, checkBank);
72 }
73 }
74 break;
75 }
76 case CONDITION_ITEM_EQUIPPED:
77 {
78 if (Unit* unit = object->ToUnit())
79 {
80 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
81 {
82 condMeets = player->HasItemOrGemWithIdEquipped(ConditionValue1, 1);
83 }
84 }
85 break;
86 }
87 case CONDITION_ZONEID:
88 condMeets = object->GetZoneId() == ConditionValue1;
89 break;
90 case CONDITION_REPUTATION_RANK:
91 {
92 if (Unit* unit = object->ToUnit())
93 {
94 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
95 {
96 if (FactionEntry const* faction = sFactionStore.LookupEntry(ConditionValue1))
97 {
98 condMeets = (ConditionValue2 & (1 << player->GetReputationMgr().GetRank(faction)));
99 }
100 }
101 }
102 break;
103 }
104 case CONDITION_ACHIEVEMENT:
105 {
106 if (Unit* unit = object->ToUnit())
107 {
108 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
109 {
110 condMeets = player->HasAchieved(ConditionValue1);
111 }
112 }
113 break;
114 }
115 case CONDITION_TEAM:
116 {
117 if (Unit* unit = object->ToUnit())
118 {
119 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
120 {
121 // Xinef: DB Data compatibility...
122 uint32 teamOld = player->GetTeamId() == TEAM_ALLIANCE ? ALLIANCE : HORDE;
123 condMeets = teamOld == ConditionValue1;
124 }
125 }
126 break;
127 }
128 case CONDITION_CLASS:
129 {
130 if (Unit* unit = object->ToUnit())
131 condMeets = unit->getClassMask() & ConditionValue1;
132 break;
133 }
134 case CONDITION_RACE:
135 {
136 if (Unit* unit = object->ToUnit())
137 condMeets = unit->getRaceMask() & ConditionValue1;
138 break;
139 }
140 case CONDITION_GENDER:
141 {
142 if (Unit* unit = object->ToUnit())
143 {
144 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
145 {
146 condMeets = player->getGender() == ConditionValue1;
147 }
148 }
149 break;
150 }
151 case CONDITION_SKILL:
152 {
153 if (Unit* unit = object->ToUnit())
154 {
155 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
156 {
157 condMeets = player->HasSkill(ConditionValue1) && player->GetBaseSkillValue(ConditionValue1) >= ConditionValue2;
158 }
159 }
160 break;
161 }
162 case CONDITION_QUESTREWARDED:
163 {
164 if (Unit* unit = object->ToUnit())
165 {
166 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
167 {
168 condMeets = player->GetQuestRewardStatus(ConditionValue1);
169 }
170 }
171 break;
172 }
173 case CONDITION_QUESTTAKEN:
174 {
175 if (Unit* unit = object->ToUnit())
176 {
177 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
178 {
179 QuestStatus status = player->GetQuestStatus(ConditionValue1);
180 condMeets = (status == QUEST_STATUS_INCOMPLETE);
181 }
182 }
183 break;
184 }
185 case CONDITION_QUEST_COMPLETE:
186 {
187 if (Unit* unit = object->ToUnit())
188 {
189 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
190 {
191 QuestStatus status = player->GetQuestStatus(ConditionValue1);
192 condMeets = (status == QUEST_STATUS_COMPLETE && !player->GetQuestRewardStatus(ConditionValue1));
193 }
194 }
195 break;
196 }
197 case CONDITION_QUEST_NONE:
198 {
199 if (Unit* unit = object->ToUnit())
200 {
201 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
202 {
203 QuestStatus status = player->GetQuestStatus(ConditionValue1);
204 condMeets = (status == QUEST_STATUS_NONE);
205 }
206 }
207 break;
208 }
209 case CONDITION_QUEST_SATISFY_EXCLUSIVE:
210 {
211 if (Unit* unit = object->ToUnit())
212 {
213 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
214 {
215 // Xinef: cannot be null, checked at loading
216 const Quest* quest = sObjectMgr->GetQuestTemplate(ConditionValue1);
217 condMeets = !player->IsQuestRewarded(ConditionValue1) && player->SatisfyQuestExclusiveGroup(quest, false);
218 }
219 }
220 break;
221 }
222 case CONDITION_ACTIVE_EVENT:
223 condMeets = sGameEventMgr->IsActiveEvent(ConditionValue1);
224 break;
225 case CONDITION_INSTANCE_INFO:
226 {
227 Map* map = object->GetMap();
228 if (map->IsDungeon())
229 {
230 if (InstanceScript const* instance = map->ToInstanceMap()->GetInstanceScript())
231 {
232 switch (ConditionValue3)
233 {
234 case INSTANCE_INFO_DATA:
235 condMeets = instance->GetData(ConditionValue1) == ConditionValue2;
236 break;
237 case INSTANCE_INFO_GUID_DATA:
238 condMeets = instance->GetGuidData(ConditionValue1) == ObjectGuid(uint64(ConditionValue2));
239 break;
240 case INSTANCE_INFO_BOSS_STATE:
241 condMeets = instance->GetBossState(ConditionValue1) == EncounterState(ConditionValue2);
242 break;
243 case INSTANCE_INFO_DATA64:
244 condMeets = instance->GetData64(ConditionValue1) == ConditionValue2;
245 break;
246 }
247 }
248 }
249 break;
250 }
251 case CONDITION_MAPID:
252 condMeets = object->GetMapId() == ConditionValue1;
253 break;
254 case CONDITION_AREAID:
255 condMeets = object->GetAreaId() == ConditionValue1;
256 break;
257 case CONDITION_SPELL:
258 {
259 if (Unit* unit = object->ToUnit())
260 {
261 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
262 {
263 condMeets = player->HasSpell(ConditionValue1);
264 }
265 }
266 break;
267 }
268 case CONDITION_LEVEL:
269 {
270 if (Unit* unit = object->ToUnit())
271 condMeets = CompareValues(static_cast<ComparisionType>(ConditionValue2), static_cast<uint32>(unit->getLevel()), ConditionValue1);
272 break;
273 }
274 case CONDITION_DRUNKENSTATE:
275 {
276 if (Unit* unit = object->ToUnit())
277 {
278 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
279 {
280 condMeets = (uint32)Player::GetDrunkenstateByValue(player->GetDrunkValue()) >= ConditionValue1;
281 }
282 }
283 break;
284 }
285 case CONDITION_NEAR_CREATURE:
286 {
287 condMeets = static_cast<bool>(GetClosestCreatureWithEntry(object, ConditionValue1, static_cast<float>(ConditionValue2), !ConditionValue3));
288 break;
289 }
290 case CONDITION_NEAR_GAMEOBJECT:
291 {
292 condMeets = static_cast<bool>(GetClosestGameObjectWithEntry(object, ConditionValue1, static_cast<float>(ConditionValue2)));
293 break;
294 }
295 case CONDITION_OBJECT_ENTRY_GUID:
296 {
297 if (ConditionValue3 == 1 && object->ToUnit()) // pussywizard: if == 1, ignore not attackable/selectable targets
298 if (object->ToUnit()->HasUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE))
299 break;
300
301 if (uint32(object->GetTypeId()) == ConditionValue1)
302 {
303 condMeets = !ConditionValue2 || (object->GetEntry() == ConditionValue2);
304
305 if (ConditionValue3 > 1)
306 {
307 switch (object->GetTypeId())
308 {
309 case TYPEID_UNIT:
310 condMeets &= object->ToCreature()->GetSpawnId() == ConditionValue3;
311 break;
312 case TYPEID_GAMEOBJECT:
313 condMeets &= object->ToGameObject()->GetSpawnId() == ConditionValue3;
314 break;
315 default:
316 break;
317 }
318 }
319 }
320 break;
321 }
322 case CONDITION_TYPE_MASK:
323 {
324 condMeets = object->isType(ConditionValue1);
325 break;
326 }
327 case CONDITION_RELATION_TO:
328 {
329 if (WorldObject* toObject = sourceInfo.mConditionTargets[ConditionValue1])
330 {
331 Unit* toUnit = toObject->ToUnit();
332 Unit* unit = object->ToUnit();
333 if (toUnit && unit)
334 {
335 switch (ConditionValue2)
336 {
337 case RELATION_SELF:
338 condMeets = unit == toUnit;
339 break;
340 case RELATION_IN_PARTY:
341 condMeets = unit->IsInPartyWith(toUnit);
342 break;
343 case RELATION_IN_RAID_OR_PARTY:
344 condMeets = unit->IsInRaidWith(toUnit);
345 break;
346 case RELATION_OWNED_BY:
347 condMeets = unit->GetOwnerGUID() == toUnit->GetGUID();
348 break;
349 case RELATION_PASSENGER_OF:
350 condMeets = unit->IsOnVehicle(toUnit);
351 break;
352 case RELATION_CREATED_BY:
353 condMeets = unit->GetCreatorGUID() == toUnit->GetGUID();
354 break;
355 }
356 }
357 }
358 break;
359 }
360 case CONDITION_REACTION_TO:
361 {
362 if (WorldObject* toObject = sourceInfo.mConditionTargets[ConditionValue1])
363 {
364 Unit* toUnit = toObject->ToUnit();
365 Unit* unit = object->ToUnit();
366 if (toUnit && unit)
367 condMeets = (1 << unit->GetReactionTo(toUnit)) & ConditionValue2;
368 }
369 break;
370 }
371 case CONDITION_DISTANCE_TO:
372 {
373 if (WorldObject* toObject = sourceInfo.mConditionTargets[ConditionValue1])
374 condMeets = CompareValues(static_cast<ComparisionType>(ConditionValue3), object->GetDistance(toObject), static_cast<float>(ConditionValue2));
375 break;
376 }
377 case CONDITION_ALIVE:
378 {
379 if (Unit* unit = object->ToUnit())
380 condMeets = unit->IsAlive();
381 break;
382 }
383 case CONDITION_HP_VAL:
384 {
385 if (Unit* unit = object->ToUnit())
386 condMeets = CompareValues(static_cast<ComparisionType>(ConditionValue2), unit->GetHealth(), static_cast<uint32>(ConditionValue1));
387 break;
388 }
389 case CONDITION_HP_PCT:
390 {
391 if (Unit* unit = object->ToUnit())
392 condMeets = CompareValues(static_cast<ComparisionType>(ConditionValue2), unit->GetHealthPct(), static_cast<float>(ConditionValue1));
393 break;
394 }
395 case CONDITION_WORLD_STATE:
396 {
397 condMeets = ConditionValue2 == sWorld->getWorldState(ConditionValue1);
398 break;
399 }
400 case CONDITION_PHASEMASK:
401 {
402 condMeets = object->GetPhaseMask() & ConditionValue1;
403 break;
404 }
405 case CONDITION_TITLE:
406 {
407 if (Unit* unit = object->ToUnit())
408 {
409 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
410 {
411 condMeets = player->HasTitle(ConditionValue1);
412 }
413 }
414 break;
415 }
416 case CONDITION_SPAWNMASK:
417 {
418 condMeets = ((1 << object->GetMap()->GetSpawnMode()) & ConditionValue1);
419 break;
420 }
421 case CONDITION_UNIT_STATE:
422 {
423 if (Unit* unit = object->ToUnit())
424 condMeets = unit->HasUnitState(ConditionValue1);
425 break;
426 }
427 case CONDITION_CREATURE_TYPE:
428 {
429 if (Creature* creature = object->ToCreature())
430 condMeets = creature->GetCreatureTemplate()->type == ConditionValue1;
431 break;
432 }
433 case CONDITION_REALM_ACHIEVEMENT:
434 {
435 AchievementEntry const* achievement = sAchievementStore.LookupEntry(ConditionValue1);
436 if (achievement && sAchievementMgr->IsRealmCompleted(achievement))
437 condMeets = true;
438 break;
439 }
440 case CONDITION_IN_WATER:
441 {
442 if (Unit* unit = object->ToUnit())
443 condMeets = unit->IsInWater();
444 break;
445 }
446 case CONDITION_QUESTSTATE:
447 {
448 if (Unit* unit = object->ToUnit())
449 {
450 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
451 {
452 uint32 queststateConditionValue1 = player->GetQuestStatus(ConditionValue1);
453 if (((ConditionValue2 & (1 << QUEST_STATUS_NONE)) && (queststateConditionValue1 == QUEST_STATUS_NONE)) ||
454 ((ConditionValue2 & (1 << QUEST_STATUS_COMPLETE)) && (queststateConditionValue1 == QUEST_STATUS_COMPLETE)) ||
455 ((ConditionValue2 & (1 << QUEST_STATUS_INCOMPLETE)) && (queststateConditionValue1 == QUEST_STATUS_INCOMPLETE)) ||
456 ((ConditionValue2 & (1 << QUEST_STATUS_FAILED)) && (queststateConditionValue1 == QUEST_STATUS_FAILED)) ||
457 ((ConditionValue2 & (1 << QUEST_STATUS_REWARDED)) && player->GetQuestRewardStatus(ConditionValue1)))
458 {
459 condMeets = true;
460 }
461 }
462 }
463 break;
464 }
465 case CONDITION_DAILY_QUEST_DONE:
466 {
467 if (Unit* unit = object->ToUnit())
468 {
469 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
470 {
471 condMeets = player->IsDailyQuestDone(ConditionValue1);
472 }
473 }
474 break;
475 }
476 case CONDITION_QUEST_OBJECTIVE_PROGRESS:
477 {
478 if (Unit* unit = object->ToUnit())
479 {
480 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
481 {
482 Quest const* quest = ASSERT_NOTNULL(sObjectMgr->GetQuestTemplate(ConditionValue1));
483 uint16 log_slot = player->FindQuestSlot(quest->GetQuestId());
484 if (log_slot >= MAX_QUEST_LOG_SIZE)
485 {
486 break;
487 }
488
489 if (player->GetQuestSlotCounter(log_slot, ConditionValue2) == ConditionValue3)
490 {
491 condMeets = true;
492 }
493 }
494 }
495 break;
496 }
497 case CONDITION_HAS_AURA_TYPE:
498 {
499 if (Unit* unit = object->ToUnit())
500 condMeets = unit->HasAuraType(AuraType(ConditionValue1));
501 break;
502 }
503 case CONDITION_STAND_STATE:
504 {
505 if (Unit* unit = object->ToUnit())
506 {
507 if (ConditionValue1 == 0)
508 {
509 condMeets = (unit->getStandState() == ConditionValue2);
510 }
511 else if (ConditionValue2 == 0)
512 {
513 condMeets = unit->IsStandState();
514 }
515 else if (ConditionValue2 == 1)
516 {
517 condMeets = unit->IsSitState();
518 }
519 }
520
521 break;
522 }
523 case CONDITION_DIFFICULTY_ID:
524 {
525 condMeets = object->GetMap()->GetDifficulty() == ConditionValue1;
526 break;
527 }
528 case CONDITION_PET_TYPE:
529 {
530 if (Unit* unit = object->ToUnit())
531 {
532 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
533 {
534 if (Pet* pet = player->GetPet())
535 {
536 condMeets = (((1 << pet->getPetType()) & ConditionValue1) != 0);
537 }
538 }
539 }
540 break;
541 }
542 case CONDITION_TAXI:
543 {
544 if (Unit* unit = object->ToUnit())
545 {
546 if (Player* player = unit->GetCharmerOrOwnerPlayerOrPlayerItself())
547 {
548 condMeets = player->IsInFlight();
549 }
550 }
551 break;
552 }
553 case CONDITION_CHARMED:
554 {
555 if (Unit* unit = object->ToUnit())
556 condMeets = unit->IsCharmed();
557 break;
558 }
559 default:
560 condMeets = false;
561 break;
562 }
563
564 if (NegativeCondition)
565 condMeets = !condMeets;
566
567 if (!condMeets)
568 sourceInfo.mLastFailedCondition = this;
569
570 // bool script = sScriptMgr->OnConditionCheck(this, sourceInfo); // Returns true by default. // pussywizard: optimization
571 return condMeets; // && script;
572}

575{

728 break;

731 break;

733 mask |= GRID_MAP_TYPE_MASK_PLAYER;
734 break;

748 mask |= GRID_MAP_TYPE_MASK_ALL;
749 break;

752 break;

758 break;
759 default:

761 break;
762 }
763 return mask;
764}

766uint32 Condition::GetMaxAvailableConditionTargets()

810uint32 ConditionMgr::GetSearcherTypeMaskForConditionList(ConditionList const& conditions)
811{
812 if (conditions.empty())
813 return GRID_MAP_TYPE_MASK_ALL;
814 // groupId, typeMask
815 std::map<uint32, uint32> ElseGroupStore;
816 for (ConditionList::const_iterator i = conditions.begin(); i != conditions.end(); ++i)
817 {
818 // no point of having not loaded conditions in list
819 ASSERT((*i)->isLoaded() && "ConditionMgr::GetSearcherTypeMaskForConditionList - not yet loaded condition found in list");
820 std::map<uint32, uint32>::const_iterator itr = ElseGroupStore.find((*i)->ElseGroup);
821 // group not filled yet, fill with widest mask possible
822 if (itr == ElseGroupStore.end())
823 ElseGroupStore[(*i)->ElseGroup] = GRID_MAP_TYPE_MASK_ALL;
824 // no point of checking anymore, empty mask
825 else if (!(*itr).second)
826 continue;
827
828 if ((*i)->ReferenceId) // handle reference
829 {
830 ConditionReferenceContainer::const_iterator ref = ConditionReferenceStore.find((*i)->ReferenceId);
831 ASSERT(ref != ConditionReferenceStore.end() && "ConditionMgr::GetSearcherTypeMaskForConditionList - incorrect reference");
832 ElseGroupStore[(*i)->ElseGroup] &= GetSearcherTypeMaskForConditionList((*ref).second);
833 }
834 else // handle normal condition
835 {
836 // object will match conditions in one ElseGroupStore only when it matches all of them
837 // so, let's find a smallest possible mask which satisfies all conditions
838 ElseGroupStore[(*i)->ElseGroup] &= (*i)->GetSearcherTypeMaskForCondition();
839 }
840 }
841 // object will match condition when one of the checks in ElseGroupStore is matching
842 // so, let's include all possible masks
843 uint32 mask = 0;
844 for (std::map<uint32, uint32>::const_iterator i = ElseGroupStore.begin(); i != ElseGroupStore.end(); ++i) mask |= i->second;
845
846 return mask;
847}

849bool ConditionMgr::IsObjectMeetToConditionList(ConditionSourceInfo& sourceInfo, ConditionList const& conditions)
850{
851 // groupId, groupCheckPassed
852 std::map<uint32, bool> ElseGroupStore;
853 for (ConditionList::const_iterator i = conditions.begin(); i != conditions.end(); ++i)
854 {
855 LOG_DEBUG("condition", "ConditionMgr::IsPlayerMeetToConditionList condType: {} val1: {}", (*i)->ConditionType, (*i)->ConditionValue1);
856 if ((*i)->isLoaded())
857 {
859 std::map<uint32, bool>::const_iterator itr = ElseGroupStore.find((*i)->ElseGroup);
861 if (itr == ElseGroupStore.end())
862 ElseGroupStore[(*i)->ElseGroup] = true;
863 else if (!(*itr).second)
864 continue;
865
866 if ((*i)->ReferenceId) // handle reference
867 {
868 ConditionReferenceContainer::const_iterator ref = ConditionReferenceStore.find((*i)->ReferenceId);
869 if (ref != ConditionReferenceStore.end())
870 {
871 if (!IsObjectMeetToConditionList(sourceInfo, (*ref).second))
872 ElseGroupStore[(*i)->ElseGroup] = false;
873 }
874 else
875 {
876 LOG_DEBUG("condition", "IsPlayerMeetToConditionList: Reference template -{} not found", (*i)->ReferenceId);
877 }
878 }
879 else // handle normal condition
880 {
881 if (!(*i)->Meets(sourceInfo))
882 ElseGroupStore[(*i)->ElseGroup] = false;
883 }
884 }
885 }
886 for (std::map<uint32, bool>::const_iterator i = ElseGroupStore.begin(); i != ElseGroupStore.end(); ++i)
887 if (i->second)
888 return true;
889
890 return false;
891}

1021
1022 // must clear all custom handled cases (groupped types) before reload
1023 if (isReload)

1049 " ConditionValue1, ConditionValue2, ConditionValue3, NegativeCondition, ErrorType, ErrorTextId, ScriptName FROM conditions");
1050
1051 if (!result)
1052 {
1053 LOG_WARN("server.loading", ">> Loaded 0 conditions. DB table `conditions` is empty!");

1055 }
1056
1057 uint32 count = 0;
1058
1059 do
1060 {
1061 Field* fields = result->Fetch();
1062
1063 Condition* cond = new Condition();

1065 cond->SourceGroup = fields[1].Get<uint32>();

1078

1082 if (iSourceTypeOrReferenceId >= 0)

1090 delete cond;
1091 continue;
1092 }
1093 cond->ReferenceId = uint32(std::abs(iConditionTypeOrReference));
1094
1095 const char* rowType = "reference template";
1096 if (iSourceTypeOrReferenceId >= 0)
1097 rowType = "reference";
1098 // check for useless data
1099 if (cond->ConditionTarget)
1100 LOG_ERROR("sql.sql", "Condition {} {} has useless data in ConditionTarget ({})!", rowType, iSourceTypeOrReferenceId, cond->ConditionTarget);

1102 LOG_ERROR("sql.sql", "Condition {} {} has useless data in value1 ({})!", rowType, iSourceTypeOrReferenceId, cond->ConditionValue1);
1103 if (cond->ConditionValue2)
1104 LOG_ERROR("sql.sql", "Condition {} {} has useless data in value2 ({})!", rowType, iSourceTypeOrReferenceId, cond->ConditionValue2);
1105 if (cond->ConditionValue3)
1106 LOG_ERROR("sql.sql", "Condition {} {} has useless data in value3 ({})!", rowType, iSourceTypeOrReferenceId, cond->ConditionValue3);

1108 LOG_ERROR("sql.sql", "Condition {} {} has useless data in NegativeCondition ({})!", rowType, iSourceTypeOrReferenceId, cond->NegativeCondition);

1112 LOG_ERROR("sql.sql", "Condition {} {} has useless data in SourceEntry ({})!", rowType, iSourceTypeOrReferenceId, cond->SourceEntry);
1113 }
1114 else if (!isConditionTypeValid(cond)) // doesn't have reference, validate ConditionType
1115 {
1116 delete cond;
1117 continue;
1118 }
1119
1120 if (iSourceTypeOrReferenceId < 0) // it is a reference template
1121 {
1122 uint32 uRefId = std::abs(iSourceTypeOrReferenceId);
1123 if (ConditionReferenceStore.find(uRefId) == ConditionReferenceStore.end()) // make sure we have a list for our conditions, based on reference id
1124 {
1125 ConditionList mCondList;
1126 ConditionReferenceStore[uRefId] = mCondList;
1127 }
1128 ConditionReferenceStore[uRefId].push_back(cond); // add to reference storage
1129 count++;
1130 continue;
1131 } // end of reference templates
1132
1133 // if not a reference and SourceType is invalid, skip
1134 if (iConditionTypeOrReference >= 0 && !isSourceTypeValid(cond))
1135 {
1136 delete cond;
1137 continue;
1138 }
1139
1140 // Grouping is only allowed for some types (loot templates, gossip menus, gossip items)
1141 if (cond->SourceGroup && !CanHaveSourceGroupSet(cond->SourceType))
1142 {
1143 LOG_ERROR("sql.sql", "Condition type {} has not allowed value of SourceGroup = {}!", uint32(cond->SourceType), cond->SourceGroup);

1145 continue;
1146 }
1147 if (cond->SourceId && !CanHaveSourceIdSet(cond->SourceType))
1148 {
1149 LOG_ERROR("sql.sql", "Condition type {} has not allowed value of SourceId = {}!", uint32(cond->SourceType), cond->SourceId);

1151 continue;
1152 }

1158 }

1163 cond->ErrorTextId = 0;
1164 }
1165
1166 if (cond->SourceGroup || cond->SourceType == CONDITION_SOURCE_TYPE_PLAYER_LOOT_TEMPLATE)

1174 break;
1175 case CONDITION_SOURCE_TYPE_DISENCHANT_LOOT_TEMPLATE:
1176 valid = addToLootTemplate(cond, LootTemplates_Disenchant.GetLootForConditionFill(cond->SourceGroup));
1177 break;
1178 case CONDITION_SOURCE_TYPE_FISHING_LOOT_TEMPLATE:
1179 valid = addToLootTemplate(cond, LootTemplates_Fishing.GetLootForConditionFill(cond->SourceGroup));
1180 break;
1181 case CONDITION_SOURCE_TYPE_GAMEOBJECT_LOOT_TEMPLATE:
1182 valid = addToLootTemplate(cond, LootTemplates_Gameobject.GetLootForConditionFill(cond->SourceGroup));
1183 break;
1184 case CONDITION_SOURCE_TYPE_ITEM_LOOT_TEMPLATE:
1185 valid = addToLootTemplate(cond, LootTemplates_Item.GetLootForConditionFill(cond->SourceGroup));
1186 break;
1187 case CONDITION_SOURCE_TYPE_MAIL_LOOT_TEMPLATE:

1189 break;
1190 case CONDITION_SOURCE_TYPE_MILLING_LOOT_TEMPLATE:

1192 break;
1193 case CONDITION_SOURCE_TYPE_PICKPOCKETING_LOOT_TEMPLATE:

1199 case CONDITION_SOURCE_TYPE_REFERENCE_LOOT_TEMPLATE:

1201 break;
1202 case CONDITION_SOURCE_TYPE_SKINNING_LOOT_TEMPLATE:

1217 valid = true;
1218 ++count;
1219 continue; // do not add to m_AllocatedMemory to avoid double deleting
1220 }
1221 case CONDITION_SOURCE_TYPE_SPELL_IMPLICIT_TARGET:
1222 valid = addToSpellImplicitTargetConditions(cond);
1223 break;
1224 case CONDITION_SOURCE_TYPE_VEHICLE_SPELL:
1225 {
1226 VehicleSpellConditionStore[cond->SourceGroup][cond->SourceEntry].push_back(cond);
1227 valid = true;
1228 ++count;
1229 continue; // do not add to m_AllocatedMemory to avoid double deleting
1230 }
1231 case CONDITION_SOURCE_TYPE_SMART_EVENT:
1232 {
1234 std::pair<int32, uint32> key = std::make_pair(cond->SourceEntry, cond->SourceId);
1235 SmartEventConditionStore[key][cond->SourceGroup].push_back(cond);
1236 valid = true;
1237 ++count;
1238 continue;
1239 }
1240 case CONDITION_SOURCE_TYPE_NPC_VENDOR:
1241 {
1242 NpcVendorConditionContainerStore[cond->SourceGroup][cond->SourceEntry].push_back(cond);
1243 valid = true;
1244 ++count;
1245 continue;
1246 }
1247 case CONDITION_SOURCE_TYPE_PLAYER_LOOT_TEMPLATE:
1248 {
1249 valid = addToLootTemplate(cond, LootTemplates_Player.GetLootForConditionFill(cond->SourceGroup));
1250 break;
1251 }
1252 default:
1253 break;
1254 }
1255
1256 if (!valid)
1257 {
1258 LOG_ERROR("sql.sql", "Not handled grouped condition, SourceGroup {}", cond->SourceGroup);
1259 delete cond;
1260 }
1261 else
1262 {
1263 AllocatedMemoryStore.push_back(cond);
1264 ++count;
1265 }
1266 continue;
1267 }
1268
1269 // handle not grouped conditions
1270 // make sure we have a storage list for our SourceType
1271 if (ConditionStore.find(cond->SourceType) == ConditionStore.end())
1272 {
1273 ConditionTypeContainer mTypeMap;
1274 ConditionStore[cond->SourceType] = mTypeMap; // add new empty list for SourceType
1275 }
1276
1277 // make sure we have a condition list for our SourceType's entry
1278 if (ConditionStore[cond->SourceType].find(cond->SourceEntry) == ConditionStore[cond->SourceType].end())
1279 {
1280 ConditionList mCondList;
1281 ConditionStore[cond->SourceType][cond->SourceEntry] = mCondList;
1282 }
1283
1284 // add new Condition to storage based on Type/Entry
1285 ConditionStore[cond->SourceType][cond->SourceEntry].push_back(cond);
1286 ++count;
1287 } while (result->NextRow());
1288
1289 LOG_INFO("server.loading", ">> Loaded {} conditions in {} ms", count, GetMSTimeDiffToNow(oldMSTime));
1290 LOG_INFO("server.loading", " ");
1291}

1353 {
1354 // check if effect is already a part of some shared mask
1355 bool found = false;
1356 for (std::list<uint32>::iterator itr = sharedMasks.begin(); itr != sharedMasks.end(); ++itr)
1357 {
1358 if ((1 << i) & *itr)
1359 {
1360 found = true;
1361 break;
1362 }
1363 }
1364 if (found)
1365 continue;
1366
1367 // build new shared mask with found effect
1368 uint32 sharedMask = (1 << i);
1369 ConditionList* cmp = spellInfo->Effects[i].ImplicitTargetConditions;
1370 for (uint8 effIndex = i + 1; effIndex < MAX_SPELL_EFFECTS; ++effIndex)
1371 {
1372 if (spellInfo->Effects[effIndex].ImplicitTargetConditions == cmp)
1373 sharedMask |= 1 << effIndex;
1374 }
1375 sharedMasks.push_back(sharedMask);
1376 }
1377
1378 for (std::list<uint32>::iterator itr = sharedMasks.begin(); itr != sharedMasks.end(); ++itr)
1379 {
1380 // some effect indexes should have same data
1381 if (uint32 commonMask = *itr & conditionEffMask)
1382 {
1383 uint8 firstEffIndex = 0;
1384 for (; firstEffIndex < MAX_SPELL_EFFECTS; ++firstEffIndex)
1385 if ((1 << firstEffIndex) & *itr)
1386 break;
1387
1388 if (firstEffIndex >= MAX_SPELL_EFFECTS)
1389 return false;
1390
1391 // get shared data
1392 ConditionList* sharedList = spellInfo->Effects[firstEffIndex].ImplicitTargetConditions;
1393
1394 // there's already data entry for that sharedMask
1395 if (sharedList)
1396 {
1397 // we have overlapping masks in db
1398 if (conditionEffMask != *itr)
1399 {
1400 LOG_ERROR("sql.sql",
1401 "SourceEntry {} in `condition` table, has incorrect SourceGroup {} (spell effectMask) set - "
1402 "effect masks are overlapping (all SourceGroup values having given bit set must be equal) - ignoring.",
1403 cond->SourceEntry, cond->SourceGroup);
1404 return false;
1405 }
1406 }
1407 // no data for shared mask, we can create new submask
1408 else
1409 {
1410 // add new list, create new shared mask
1411 sharedList = new ConditionList();
1412 bool assigned = false;
1413 for (uint8 i = firstEffIndex; i < MAX_SPELL_EFFECTS; ++i)
1414 {
1415 if ((1 << i) & commonMask)
1416 {
1417 spellInfo->Effects[i].ImplicitTargetConditions = sharedList;
1418 assigned = true;
1419 }
1420 }
1421
1422 if (!assigned)
1423 delete sharedList;
1424 }
1425 if (sharedList)
1426 sharedList->push_back(cond);
1427 break;
1428 }
1429 }
1430 return true;
1431}
