27static bool IsMutualChase(Unit* owner, Unit* target)
28{
29 if (target->GetMotionMaster()->GetCurrentMovementGeneratorType() != CHASE_MOTION_TYPE)
30 return false;
31
32 return target->GetVictim() == owner;
33}

35template<class T>
36bool ChaseMovementGenerator<T>::PositionOkay(T* owner, Unit* target, Optional<float> maxDistance, Optional<ChaseAngle> angle)
37{
38 float const distSq = owner->GetExactDistSq(target);
39 if (maxDistance && distSq > G3D::square(*maxDistance))
40 return false;
41 if (angle && !angle->IsAngleOkay(target->GetRelativeAngle(owner)))
42 return false;
43 if (!owner->IsWithinLOSInMap(target))
44 return false;
45 return true;
46}

48template<class T>
49bool ChaseMovementGenerator<T>::DoUpdate(T* owner, uint32 time_diff)
50{
51 if (!i_target.isValid() || !i_target->IsInWorld() || !owner->IsInMap(i_target.getTarget()))
52 return false;
53
54 if (!owner || !owner->IsAlive())
55 return false;
56
57 Creature* cOwner = owner->ToCreature();
58
59 // the owner might be unable to move (rooted or casting), or we have lost the target, pause movement
60 if (owner->HasUnitState(UNIT_STATE_NOT_MOVE) || HasLostTarget(owner) || (cOwner && cOwner->IsMovementPreventedByCasting()))
61 {
62 owner->StopMoving();
63 _lastTargetPosition.reset();
64 if (Creature* cOwner2 = owner->ToCreature())
65 {
66 cOwner2->SetCannotReachTarget();
67 }
68
69 return true;
70 }
71
72 bool forceDest =
73 //(cOwner && (cOwner->isWorldBoss() || cOwner->IsDungeonBoss())) || // force for all bosses, even not in instances
74 (i_target->GetTypeId() == TYPEID_PLAYER && i_target->ToPlayer()->IsGameMaster()) || // for .npc follow
75 (owner->CanFly())
76 ; // closes "bool forceDest", that way it is more appropriate, so we can comment out crap whenever we need to
77
78 Unit* target = i_target.getTarget();
79
80 bool const mutualChase = IsMutualChase(owner, target);
81 float const hitboxSum = owner->GetCombatReach() + target->GetCombatReach();
82 float const minTarget = (_range ? _range->MinTolerance : 0.0f) + hitboxSum;
83 float const maxRange = _range ? _range->MaxRange + hitboxSum : owner->GetMeleeRange(target); // melee range already includes hitboxes
84 float const maxTarget = _range ? _range->MaxTolerance + hitboxSum : CONTACT_DISTANCE + hitboxSum;
85 Optional<ChaseAngle> angle = mutualChase ? Optional<ChaseAngle>() : _angle;
86
87 i_recheckDistance.Update(time_diff);
88 if (i_recheckDistance.Passed())
89 {
90 i_recheckDistance.Reset(100);
91
92 if (i_recalculateTravel && PositionOkay(owner, target, _movingTowards ? maxTarget : Optional<float>(), angle))
93 {
94 if (Creature* cOwner2 = owner->ToCreature())
95 {
96 cOwner2->SetCannotReachTarget();
97 }
98
99 i_recalculateTravel = false;
100 i_path = nullptr;
101
102 owner->StopMoving();
103 owner->SetInFront(target);
104 MovementInform(owner);
105 return true;
106 }
107 }
108
109 if (owner->HasUnitState(UNIT_STATE_CHASE_MOVE) && owner->movespline->Finalized())
110 {
111 owner->ClearUnitState(UNIT_STATE_CHASE_MOVE);
112 owner->SetInFront(target);
113 MovementInform(owner);
114
115 if (owner->IsWithinMeleeRange(this->i_target.getTarget()))
116 {
117 owner->Attack(this->i_target.getTarget(), true);
118 }
119 else if (i_path && i_path->GetPathType() & PATHFIND_INCOMPLETE)
120 {
121 if (Creature* cOwner2 = owner->ToCreature())
122 {
123 cOwner2->SetCannotReachTarget(this->i_target.getTarget()->GetGUID());
124 }
125 }
126
127 i_recalculateTravel = false;
128 i_path = nullptr;
129 }
130
131 if (_lastTargetPosition && i_target->GetPosition() == _lastTargetPosition.value() && mutualChase == _mutualChase)
132 return true;
133
134 _lastTargetPosition = i_target->GetPosition();
135
136 if (PositionOkay(owner, target, maxRange, angle) && !owner->HasUnitState(UNIT_STATE_CHASE_MOVE))
137 return true;
138
139 float tarX, tarY, tarZ;
140 target->GetPosition(tarX, tarY, tarZ);
141
142 bool withinRange = owner->IsInDist(target, maxRange);
143 bool withinLOS = owner->IsWithinLOS(tarX, tarY, tarZ);
144 bool moveToward = !(withinRange && withinLOS);
145
146 _mutualChase = mutualChase;
147
148 if (owner->HasUnitState(UNIT_STATE_CHASE_MOVE))
149 {
150 // can we get to the target?
151 if (cOwner && !target->isInAccessiblePlaceFor(cOwner))
152 {
153 cOwner->SetCannotReachTarget(target->GetGUID());
154 cOwner->StopMoving();
155 i_path = nullptr;
156 return true;
157 }
158 }
159
160 if (!i_path || moveToward != _movingTowards)
161 i_path = std::make_unique<PathGenerator>(owner);
162 else
163 i_path->Clear();
164
165 float x, y, z;
166 bool shortenPath;
167 // if we want to move toward the target and there's no fixed angle...
168 if (moveToward && !angle)
169 {
170 // ...we'll pathfind to the center, then shorten the path
171 target->GetPosition(x, y, z);
172 shortenPath = true;
173 }
174 else
175 {
176 // otherwise, we fall back to nearpoint finding
177 target->GetNearPoint(owner, x, y, z, (moveToward ? maxTarget : minTarget) - hitboxSum, 0, angle ? target->ToAbsoluteAngle(angle->RelativeAngle) : target->GetAngle(owner));
178 shortenPath = false;
179 }
180
181 if (owner->IsHovering())
182 owner->UpdateAllowedPositionZ(x, y, z);
183
184 i_recalculateTravel = true;
185
186 bool success = i_path->CalculatePath(x, y, z, forceDest);
187 if (!success || i_path->GetPathType() & PATHFIND_NOPATH)
188 {
189 if (cOwner)
190 {
191 cOwner->SetCannotReachTarget(target->GetGUID());
192 }
193
194 return true;
195 }
196
197 if (shortenPath)
198 i_path->ShortenPathUntilDist(G3D::Vector3(target->GetPositionX(), target->GetPositionY(), target->GetPositionZ()), maxTarget);
199
200 if (cOwner)
201 {
202 cOwner->SetCannotReachTarget();
203 }
204
205 bool walk = false;
206 if (cOwner && !cOwner->IsPet())
207 {
208 switch (cOwner->GetMovementTemplate().GetChase())
209 {
210 case CreatureChaseMovementType::CanWalk:
211 if (owner->IsWalking())
212 walk = true;
213 break;
214 case CreatureChaseMovementType::AlwaysWalk:
215 walk = true;
216 break;
217 default:
218 break;
219 }
220 }
221
222 owner->AddUnitState(UNIT_STATE_CHASE_MOVE);
223
224 Movement::MoveSplineInit init(owner);
225 init.MovebyPath(i_path->GetPath());
226 init.SetFacing(target);
227 init.SetWalk(walk);
228 init.Launch();
229
230 return true;
231}

282static Optional<float> GetVelocity(Unit* owner, Unit* target, G3D::Vector3 const& dest, bool playerPet)
283{
284 Optional<float> speed = {};
285 if (!owner->IsInCombat() && !owner->IsVehicle() && !owner->HasUnitFlag(UNIT_FLAG_POSSESSED) &&
286 (owner->IsPet() || owner->IsGuardian() || owner->GetGUID() == target->GetCritterGUID() || owner->GetCharmerOrOwnerGUID() == target->GetGUID()))
287 {
288 UnitMoveType moveType = Movement::SelectSpeedType(target->GetUnitMovementFlags());
289 speed = std::max(target->GetSpeed(moveType), owner->GetSpeed(moveType));
290
291 if (playerPet)
292 {
293 float distance = owner->GetDistance2d(dest.x, dest.y) - (*speed / 2.f);
294 if (distance > 0.f)
295 {
296 float multiplier = 1.f + (distance / 10.f);
297 *speed *= multiplier;
298 }
299 else
300 {
301 switch (moveType)
302 {
303 case MOVE_RUN_BACK:
304 case MOVE_SWIM_BACK:
305 case MOVE_FLIGHT_BACK:
306 break;
307 default:
308 *speed *= 0.9f;
309 break;
310 }
311 }
312 }
313 }
314
315 return speed;
316}

318static Position const PredictPosition(Unit* target)
319{
320 Position pos = target->GetPosition();
321
322 // 0.5 - it's time (0.5 sec) between starting movement opcode (e.g. MSG_MOVE_START_FORWARD) and MSG_MOVE_HEARTBEAT sent by client
323 float speed = target->GetSpeed(Movement::SelectSpeedType(target->GetUnitMovementFlags())) * 0.5f;
324 float orientation = target->GetOrientation();
325
326 if (target->m_movementInfo.HasMovementFlag(MOVEMENTFLAG_FORWARD))
327 {
328 pos.m_positionX += cos(orientation) * speed;
329 pos.m_positionY += std::sin(orientation) * speed;
330 }
331 else if (target->m_movementInfo.HasMovementFlag(MOVEMENTFLAG_BACKWARD))
332 {
333 pos.m_positionX -= cos(orientation) * speed;
334 pos.m_positionY -= std::sin(orientation) * speed;
335 }
336
337 if (target->m_movementInfo.HasMovementFlag(MOVEMENTFLAG_STRAFE_LEFT))
338 {
339 pos.m_positionX += cos(orientation + M_PI / 2.f) * speed;
340 pos.m_positionY += std::sin(orientation + M_PI / 2.f) * speed;
341 }
342 else if (target->m_movementInfo.HasMovementFlag(MOVEMENTFLAG_STRAFE_RIGHT))
343 {
344 pos.m_positionX += cos(orientation - M_PI / 2.f) * speed;
345 pos.m_positionY += std::sin(orientation - M_PI / 2.f) * speed;
346 }
347
348 return pos;
349}

351template<class T>
352bool FollowMovementGenerator<T>::PositionOkay(Unit* target, bool isPlayerPet, bool& targetIsMoving, uint32 diff)
353{
354 if (!_lastTargetPosition)
355 return false;
356
357 float exactDistSq = target->GetExactDistSq(_lastTargetPosition->GetPositionX(), _lastTargetPosition->GetPositionY(), _lastTargetPosition->GetPositionZ());
358 float distanceTolerance = 0.25f;
359 // For creatures, increase tolerance
360 if (target->GetTypeId() == TYPEID_UNIT)
361 {
362 distanceTolerance += _range + _range;
363 }
364
365 if (isPlayerPet)
366 {
367 targetIsMoving = target->m_movementInfo.HasMovementFlag(MOVEMENTFLAG_FORWARD | MOVEMENTFLAG_BACKWARD | MOVEMENTFLAG_STRAFE_LEFT | MOVEMENTFLAG_STRAFE_RIGHT);
368 }
369
370 if (exactDistSq > distanceTolerance)
371 return false;
372
373 if (isPlayerPet)
374 {
375 if (!targetIsMoving)
376 {
377 if (i_recheckPredictedDistanceTimer.GetExpiry())
378 {
379 i_recheckPredictedDistanceTimer.Update(diff);
380 if (i_recheckPredictedDistanceTimer.Passed())
381 {
382 i_recheckPredictedDistanceTimer = 0;
383 return false;
384 }
385 }
386
387 return true;
388 }
389
390 return false;
391 }
392
393 return true;
394}

396template<class T>
397bool FollowMovementGenerator<T>::DoUpdate(T* owner, uint32 time_diff)
398{
399 if (!i_target.isValid() || !i_target->IsInWorld() || !owner->IsInMap(i_target.getTarget()))
400 return false;
401
402 if (!owner || !owner->IsAlive())
403 return false;
404
405 Creature* cOwner = owner->ToCreature();
406 Unit* target = i_target.getTarget();
407
408 // the owner might be unable to move (rooted or casting), or we have lost the target, pause movement
409 if (owner->HasUnitState(UNIT_STATE_NOT_MOVE) || (cOwner && owner->ToCreature()->IsMovementPreventedByCasting()))
410 {
411 i_path = nullptr;
412 owner->StopMoving();
413 _lastTargetPosition.reset();
414 return true;
415 }
416
417 bool followingMaster = false;
418 Pet* oPet = owner->ToPet();
419 if (oPet)
420 {
421 if (target->GetGUID() == oPet->GetOwnerGUID())
422 followingMaster = true;
423 }
424
425 bool forceDest =
426 (followingMaster) || // allow pets following their master to cheat while generating paths
427 (i_target->GetTypeId() == TYPEID_PLAYER && i_target->ToPlayer()->IsGameMaster()) // for .npc follow
428 ; // closes "bool forceDest", that way it is more appropriate, so we can comment out crap whenever we need to
429
430 bool targetIsMoving = false;
431 if (PositionOkay(target, owner->IsGuardian() && target->GetTypeId() == TYPEID_PLAYER, targetIsMoving, time_diff))
432 {
433 if (owner->HasUnitState(UNIT_STATE_FOLLOW_MOVE) && owner->movespline->Finalized())
434 {
435 owner->ClearUnitState(UNIT_STATE_FOLLOW_MOVE);
436 i_path = nullptr;
437 MovementInform(owner);
438
439 if (i_recheckPredictedDistance)
440 {
441 i_recheckPredictedDistanceTimer.Reset(1000);
442 }
443
444 owner->SetFacingTo(target->GetOrientation());
445 }
446 }
447 else
448 {
449 Position targetPosition = target->GetPosition();
450 _lastTargetPosition = targetPosition;
451
452 // If player is moving and their position is not updated, we need to predict position
453 if (targetIsMoving)
454 {
455 Position predictedPosition = PredictPosition(target);
456 if (_lastPredictedPosition && _lastPredictedPosition->GetExactDistSq(&predictedPosition) < 0.25f)
457 return true;
458
459 _lastPredictedPosition = predictedPosition;
460 targetPosition = predictedPosition;
461 i_recheckPredictedDistance = true;
462 }
463 else
464 {
465 i_recheckPredictedDistance = false;
466 i_recheckPredictedDistanceTimer.Reset(0);
467 }
468
469 if (!i_path)
470 i_path = std::make_unique<PathGenerator>(owner);
471 else
472 i_path->Clear();
473
474 target->MovePositionToFirstCollision(targetPosition, owner->GetCombatReach() + _range, target->ToAbsoluteAngle(_angle.RelativeAngle) - target->GetOrientation());
475
476 float x, y, z;
477 targetPosition.GetPosition(x, y, z);
478
479 if (owner->IsHovering())
480 owner->UpdateAllowedPositionZ(x, y, z);
481
482 bool success = i_path->CalculatePath(x, y, z, forceDest);
483 if (!success || (i_path->GetPathType() & PATHFIND_NOPATH && !followingMaster))
484 {
485 if (!owner->IsStopped())
486 owner->StopMoving();
487
488 return true;
489 }
490
491 owner->AddUnitState(UNIT_STATE_FOLLOW_MOVE);
492
493 Movement::MoveSplineInit init(owner);
494 init.MovebyPath(i_path->GetPath());
495 init.SetWalk(target->IsWalking());
496 if (Optional<float> velocity = GetVelocity(owner, target, i_path->GetActualEndPosition(), owner->IsGuardian() && target->GetTypeId() == TYPEID_PLAYER))
497 init.SetVelocity(*velocity);
498 init.Launch();
499 }
500
501 return true;
502}
