82 _JustDied();
83
84 std::list<GameObject*> _goList;
85 GetGameObjectListWithEntryInGrid(_goList, me, GO_SUPPRESSION_DEVICE, 200.0f);
86 for (std::list<GameObject*>::const_iterator itr = _goList.begin(); itr != _goList.end(); itr++)
87 {
88 ((*itr)->AI()->DoAction(ACTION_DEACTIVATE));
89 }

92 void UpdateAI(uint32 diff) override
93 {
94 if (!UpdateVictim())
95 return;
96
97 events.Update(diff);
98
99 while (uint32 eventId = events.ExecuteEvent())
100 {
101 switch (eventId)
102 {
103 case EVENT_CLEAVE:
104 DoCastVictim(SPELL_CLEAVE);
105 events.ScheduleEvent(EVENT_CLEAVE, 7000);
106 break;
107 case EVENT_BLASTWAVE:
108 DoCastVictim(SPELL_BLASTWAVE);
109 events.ScheduleEvent(EVENT_BLASTWAVE, 20000, 35000);
110 break;
111 case EVENT_MORTALSTRIKE:
112 DoCastVictim(SPELL_MORTALSTRIKE);
113 events.ScheduleEvent(EVENT_MORTALSTRIKE, 25000, 35000);
114 break;
115 case EVENT_KNOCKBACK:
116 DoCastVictim(SPELL_KNOCKBACK);
117 if (DoGetThreat(me->GetVictim()))
118 DoModifyThreatPercent(me->GetVictim(), -50);
119 events.ScheduleEvent(EVENT_KNOCKBACK, 15000, 30000);
120 break;
121 case EVENT_CHECK:
122 if (me->GetDistance(me->GetHomePosition()) > 150.0f)
123 {
124 Talk(SAY_LEASH);
125 EnterEvadeMode();
126 }
127 events.ScheduleEvent(EVENT_CHECK, 1000);
128 break;
129 }
130 }
131
132 DoMeleeAttackIfReady();
133 }
134 };

142class go_suppression_device : public GameObjectScript
143{
144 public:
145 go_suppression_device() : GameObjectScript("go_suppression_device") { }
146
147 void OnLootStateChanged(GameObject* go, uint32 state, Unit* /*unit*/) override
148 {
149 switch (state)
150 {
151 case GO_JUST_DEACTIVATED: // This case prevents the Gameobject despawn by Disarm Trap
152 go->SetLootState(GO_READY);
153 [[fallthrough]];
154 case GO_ACTIVATED:
155 go->AI()->DoAction(ACTION_DISARMED);
156 break;
157 }
158 }
159
160 struct go_suppression_deviceAI : public GameObjectAI
161 {
162 go_suppression_deviceAI(GameObject* go) : GameObjectAI(go), _instance(go->GetInstanceScript()), _active(true) { }
163
164 void InitializeAI() override
165 {
166 if (_instance->GetBossState(DATA_BROODLORD_LASHLAYER) == DONE)
167 {
168 Deactivate();
169 return;
170 }
171
172 _events.ScheduleEvent(EVENT_SUPPRESSION_CAST, 5000);
173 }
174
175 void UpdateAI(uint32 diff) override
176 {
177 _events.Update(diff);
178
179 while (uint32 eventId = _events.ExecuteEvent())
180 {
181 switch (eventId)
182 {
183 case EVENT_SUPPRESSION_CAST:
184 if (me->GetGoState() == GO_STATE_READY)
185 {
186 me->CastSpell(nullptr, SPELL_SUPPRESSION_AURA);
187 me->SendCustomAnim(0);
188 }
189 _events.ScheduleEvent(EVENT_SUPPRESSION_CAST, 5000);
190 break;
191 case EVENT_SUPPRESSION_RESET:
192 Activate();
193 break;
194 }
195 }
196 }
197
198 void DoAction(int32 action) override
199 {
200 if (action == ACTION_DEACTIVATE)
201 {
202 _events.CancelEvent(EVENT_SUPPRESSION_RESET);
203 }
204 else if (action == ACTION_DISARMED)
205 {
206 Deactivate();
207 _events.CancelEvent(EVENT_SUPPRESSION_CAST);
208
209 if (_instance->GetBossState(DATA_BROODLORD_LASHLAYER) != DONE)
210 {
211 _events.ScheduleEvent(EVENT_SUPPRESSION_RESET, urand(30000, 120000));
212 }
213 }
214 }
215
216 void Activate()
217 {
218 if (_active)
219 return;
220 _active = true;
221 if (me->GetGoState() == GO_STATE_ACTIVE)
222 me->SetGoState(GO_STATE_READY);
223 me->SetLootState(GO_READY);
224 me->RemoveGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
225 _events.ScheduleEvent(EVENT_SUPPRESSION_CAST, 5000);
226 me->Respawn();
227 }
228
229 void Deactivate()
230 {
231 if (!_active)
232 return;
233 _active = false;
234 me->SetGoState(GO_STATE_ACTIVE);
235 me->SetGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
236 _events.CancelEvent(EVENT_SUPPRESSION_CAST);
237 }
238
239 private:
240 InstanceScript* _instance;
241 EventMap _events;
242 bool _active;
243 };
244
245 GameObjectAI* GetAI(GameObject* go) const override
246 {
247 return new go_suppression_deviceAI(go);
248 }
249};
