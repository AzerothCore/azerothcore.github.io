237 void HandlePeriodic(AuraEffect const* aurEff)
238 {
239 PreventDefaultAction();
240 if (aurEff->GetAuraType() == SPELL_AURA_MOD_STAT && (aurEff->GetMiscValue() == STAT_STAMINA || aurEff->GetMiscValue() == STAT_INTELLECT))
241 {
242 int32 currentAmount = aurEff->GetAmount();
243 int32 newAmount = GetEffect(aurEff->GetEffIndex())->CalculateAmount(GetCaster());
244 if (newAmount != currentAmount)
245 {
246 if (aurEff->GetMiscValue() == STAT_STAMINA)
247 {
248 uint32 actStat = GetUnitOwner()->GetHealth();
249 GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
250 GetUnitOwner()->SetHealth(std::min<uint32>(GetUnitOwner()->GetMaxHealth(), actStat));
251 }
252 else
253 {
254 uint32 actStat = GetUnitOwner()->GetPower(POWER_MANA);
255 GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
256 GetUnitOwner()->SetPower(POWER_MANA, std::min<uint32>(GetUnitOwner()->GetMaxPower(POWER_MANA), actStat));
257 }
258 }
259 }
260 else
261 GetEffect(aurEff->GetEffIndex())->RecalculateAmount();
262 }
