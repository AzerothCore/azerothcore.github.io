46 template <GenericHashImpl::HashCreator HashCreator, size_t DigestLength>
47 class GenericHash
48 {
49 public:
50 static constexpr size_t DIGEST_LENGTH = DigestLength;
51 using Digest = std::array<uint8, DIGEST_LENGTH>;
52
53 static Digest GetDigestOf(uint8 const* data, size_t len)
54 {
55 GenericHash hash;
56 hash.UpdateData(data, len);
57 hash.Finalize();
58 return hash.GetDigest();
59 }
60
61 template <typename... Ts>
62 static auto GetDigestOf(Ts&& ... pack) -> std::enable_if_t < !(std::is_integral_v<std::decay_t<Ts>> || ...), Digest >
63 {
64 GenericHash hash;
65 (hash.UpdateData(std::forward<Ts>(pack)), ...);
66 hash.Finalize();
67 return hash.GetDigest();
68 }
69
70 GenericHash() : _ctx(GenericHashImpl::MakeCTX())
71 {
72 int result = EVP_DigestInit_ex(_ctx, HashCreator(), nullptr);
73 ASSERT(result == 1);
74 }
75
76 ~GenericHash()
77 {
78 if (!_ctx)
79 {
80 return;
81 }
82 GenericHashImpl::DestroyCTX(_ctx);
83 _ctx = nullptr;
84 }
85
86 void UpdateData(uint8 const* data, size_t len)
87 {
88 int result = EVP_DigestUpdate(_ctx, data, len);
89 ASSERT(result == 1);
90 }
91 void UpdateData(std::string_view str) { UpdateData(reinterpret_cast<uint8 const*>(str.data()), str.size()); }
92 void UpdateData(std::string const& str) { UpdateData(std::string_view(str)); } /* explicit overload to avoid using the container template */
93 void UpdateData(char const* str) { UpdateData(std::string_view(str)); } /* explicit overload to avoid using the container template */
94 template <typename Container>
95 void UpdateData(Container const& c) { UpdateData(std::data(c), std::size(c)); }
96
97 void Finalize()
98 {
99 uint32 length;
100 int result = EVP_DigestFinal_ex(_ctx, _digest.data(), &length);
101 ASSERT(result == 1);
102 ASSERT(length == DIGEST_LENGTH);
103 GenericHashImpl::DestroyCTX(_ctx);
104 _ctx = nullptr;
105 }
106
107 Digest const& GetDigest() const { return _digest; }
108
109 private:
110 EVP_MD_CTX* _ctx;
111 Digest _digest = { };
112 };
