320 void UpdateAI(uint32 diff) override
321 {
322 if (!UpdateVictim())
323 return;
324
325 if( me->GetPositionX() < 490.0f || me->GetPositionX() > 690.0f || me->GetPositionY() < 130.0f || me->GetPositionY() > 410.0f )
326 {
327 EnterEvadeMode(EVADE_REASON_OTHER);
328 return;
329 }
330
331 events.Update(diff);
332
333 if (me->HasUnitState(UNIT_STATE_CASTING))
334 return;
335
336 switch (events.ExecuteEvent())
337 {
338 case 0:
339 break;
340 case EVENT_ACTIVATE_CONSTRUCT:
341 Talk(SAY_SUMMON);
342 me->CastCustomSpell(SPELL_ACTIVATE_CONSTRUCT, SPELLVALUE_MAX_TARGETS, 1, (Unit*)nullptr, false);
343 if (++counter >= 20)
344 {
345 Talk(SAY_BERSERK);
346 me->CastSpell(me, SPELL_BERSERK, true);
347 break;
348 }
349 events.RepeatEvent(RAID_MODE(40000, 30000));
350 break;
351 case EVENT_SPELL_SCORCH:
352 Talk(SAY_SCORCH);
353 me->SetControlled(true, UNIT_STATE_ROOT);
354 me->DisableRotate(true);
355 me->SendMovementFlagUpdate();
356 me->CastSpell(me->GetVictim(), S_SCORCH, false);
357 events.RepeatEvent(20000);
358 events.RescheduleEvent(EVENT_ENABLE_ROTATE, 3001);
359 break;
360 case EVENT_ENABLE_ROTATE:
361 me->SetControlled(false, UNIT_STATE_ROOT);
362 me->DisableRotate(false);
363 break;
364 case EVENT_SPELL_FLAME_JETS:
365 Talk(EMOTE_JETS);
366 me->CastSpell(me->GetVictim(), S_FLAME_JETS, false);
367 events.RepeatEvent(25000);
368 break;
369 case EVENT_GRAB:
370 {
371 std::list<Creature*> icl;
372 me->GetCreaturesWithEntryInRange(icl, 300.0f, NPC_IRON_CONSTRUCT);
373
374 GuidVector playerGUIDs;
375 Map::PlayerList const& pl = me->GetMap()->GetPlayers();
376 Player* temp = nullptr;
377
378 for( Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr )
379 {
380 temp = itr->GetSource();
381 if( !temp->IsAlive() || temp->GetExactDist2d(me) > 90.0f )
382 continue;
383 if( me->GetVictim() && temp->GetGUID() == me->GetVictim()->GetGUID() )
384 continue;
385 bool found = false;
386 for (std::list<Creature*>::iterator iterator = icl.begin(); iterator != icl.end(); ++iterator)
387 {
388 if ((*iterator)->GetVictim() && (*iterator)->GetVictim()->GetGUID() == temp->GetGUID())
389 {
390 found = true;
391 break;
392 }
393 }
394
395 if( !found )
396 playerGUIDs.push_back(temp->GetGUID());
397 }
398
399 if( !playerGUIDs.empty() )
400 {
401 int8 pos = urand(0, playerGUIDs.size() - 1);
402 if( Player* pTarget = ObjectAccessor::GetPlayer(*me, playerGUIDs.at(pos)) )
403 {
404 Talk(SAY_SLAG_POT);
405 me->CastSpell(pTarget, SPELL_GRAB, false);
406 }
407 }
408
409 events.RepeatEvent(24000); // +6000 below
410 events.DelayEvents(6000);
411 }
412 break;
413 }
414
415 DoMeleeAttackIfReady();
416 }
