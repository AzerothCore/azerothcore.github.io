66 if (bReset || spell->Id != 3607)

68
69 if (Player* player = caster->ToPlayer())
70 {
71 if (player->GetQuestStatus(592) == QUEST_STATUS_INCOMPLETE) //Yenniku's Release
72 {
73 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_STUN);
74 me->CombatStop(); //stop combat
75 me->GetThreatMgr().ClearAllThreat(); //unsure of this

77

79 Reset_Timer = 60000;
80 }
81 }
82 }

86 void UpdateAI(uint32 diff) override
87 {
88 if (bReset)
89 {
90 if (Reset_Timer <= diff)
91 {
92 EnterEvadeMode();
93 bReset = false;
94 me->SetFaction(FACTION_TROLL_BLOODSCALP);
95 return;
96 }
97
98 Reset_Timer -= diff;
99
100 if (me->IsInCombat() && me->GetVictim())
101 {
102 if (Player* player = me->GetVictim()->ToPlayer())
103 {
104 if (player->GetTeamId() == TEAM_HORDE)
105 {
106 me->CombatStop();
107 me->GetThreatMgr().ClearAllThreat();
108 }
109 }
110 }
111 }
112
113 //Return since we have no target
114 if (!UpdateVictim())
115 return;
116
117 DoMeleeAttackIfReady();
118 }
119 };
