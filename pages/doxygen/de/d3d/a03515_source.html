141 bool SetBossState(uint32 id, EncounterState state) override
142 {
143 if (!InstanceScript::SetBossState(id, state))
144 return false;
145
146 if (state != DONE || id > DATA_ORMOROK_EVENT)
147 return true;
148
149 BossInfo const* bossInfo = GetBossInfo(id + DATA_TELESTRA_ORB);
150 for (DoorSet::const_iterator i = bossInfo->door[DOOR_TYPE_PASSAGE].begin(); i != bossInfo->door[DOOR_TYPE_PASSAGE].end(); ++i)
151 (*i)->RemoveGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
152 return true;
153 }

162 void Load(const char* in) override
163 {
164 if( !in )
165 return;
166
167 char dataHead1, dataHead2, dataHead3;
168 std::istringstream loadStream(in);
169 loadStream >> dataHead1 >> dataHead2 >> dataHead3;
170 if (dataHead1 == 'N' && dataHead2 == 'E' && dataHead3 == 'X')
171 {
172 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
173 {
174 uint32 tmpState;
175 loadStream >> tmpState;
176 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
177 tmpState = NOT_STARTED;
178 SetBossState(i, EncounterState(tmpState));
179 }
180 }
181 }
182 };

239 if (!_allowDeath)
240 {
241 me->RemoveAllAuras();
242 me->GetThreatMgr().ClearAllThreat();
243 me->CombatStop(true);
244 damage = 0;
245
246 me->SetReactState(REACT_PASSIVE);
247 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
248 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
249 me->SetRegeneratingHealth(false);
250 me->CastSpell(me, SPELL_SUMMON_SEED_POD, true);
251 me->CastSpell(me, SPELL_SEED_POD, true);
252 me->CastSpell(me, SPELL_AURA_OF_REGENERATION, false);
253 restoreTimer = 1;
254 }
255 }
256 }
257
258 void UpdateAI(uint32 diff) override
259 {
260 if (restoreTimer)
261 {
262 restoreTimer += diff;
263 if (restoreTimer >= 90 * IN_MILLISECONDS)
264 {
265 Talk(0);
266 me->SetRegeneratingHealth(true);
267 restoreTimer = 0;
268 me->SetReactState(REACT_AGGRESSIVE);
269 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
270 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
271 }
272 return;
273 }
274
275 if (!UpdateVictim())
276 return;
277
278 abilityTimer1 += diff;
279 abilityTimer2 += diff;
280
281 if (abilityTimer1 >= 5000)
282 {
283 me->CastSpell(me->GetVictim(), SPELL_ENSNARE, false);
284 abilityTimer1 = 0;
285 }
286
287 if (abilityTimer2 >= 30000)
288 {
289 me->CastSpell(me->GetVictim(), SPELL_CRYSTAL_BLOOM, false);
290 abilityTimer2 = 0;
291 }
292 }
