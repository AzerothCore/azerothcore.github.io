160 void InitializeAI() override
161 {
162 BossAI::InitializeAI();
163 if (instance->GetBossState(DATA_MAJORDOMO_EXECUTUS) != DONE)
164 {
165 events.SetPhase(PHASE_COMBAT);
166
167 std::list<TempSummon*> p_summons;
168 me->SummonCreatureGroup(SUMMON_GROUP_ADDS, &p_summons);
169 if (!p_summons.empty())
170 {
171 for (TempSummon const* summon : p_summons)
172 {
173 if (summon)
174 {
175 static_minionsGUIDS.insert(summon->GetGUID());
176 majordomoSummonsData[summon->GetGUID().GetCounter()] = MajordomoAddData(summon->GetGUID(), summon->GetEntry(), summon->GetPosition());
177 }
178 }
179 }
180 }
181 else
182 {
183 events.SetPhase(PHASE_NONE);
184 me->SetImmuneToAll(true);
185 me->SetNpcFlag(UNIT_NPC_FLAG_GOSSIP);
186 me->SetFaction(FACTION_MAJORDOMO_FRIENDLY);
187 }
188 }
189
190 void Reset() override
191 {
192 me->ResetLootMode();
193 events.Reset();
194 aliveMinionsGUIDS.clear();
195
196 if (instance->GetBossState(DATA_MAJORDOMO_EXECUTUS) != DONE)
197 {
198 events.SetPhase(PHASE_COMBAT);
199 instance->SetBossState(DATA_MAJORDOMO_EXECUTUS, NOT_STARTED);
200
201 for (auto const& summon : majordomoSummonsData)
202 {
203 if (ObjectAccessor::GetCreature(*me, summon.second.guid))
204 {
205 continue;
206 }
207
208 if (Creature* spawn = me->SummonCreature(summon.second.creatureEntry, summon.second.spawnPos))
209 {
210 static_minionsGUIDS.erase(summon.second.guid); // Erase the guid from the previous, no longer existing, spawn.
211 static_minionsGUIDS.insert(spawn->GetGUID());
212 majordomoSummonsData.erase(summon.second.guid.GetCounter());
213 majordomoSummonsData[spawn->GetGUID().GetCounter()] = MajordomoAddData(spawn->GetGUID(), spawn->GetEntry(), spawn->GetPosition());
214 }
215 }
216
217 me->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP);
218 }
219 else
220 {
221 static_minionsGUIDS.clear();
222 majordomoSummonsData.clear();
223 summons.DespawnAll();
224 }
225 }

260 void SummonedCreatureDies(Creature* summon, Unit* /*killer*/) override
261 {
262 aliveMinionsGUIDS.erase(summon->GetGUID());
263 if (summon->GetEntry() == NPC_FLAMEWAKER_HEALER || summon->GetEntry() == NPC_FLAMEWAKER_ELITE)
264 {
265 uint32 const remainingAdds = std::count_if(aliveMinionsGUIDS.begin(), aliveMinionsGUIDS.end(), [](ObjectGuid const& summonGuid)
266 {
267 return summonGuid.GetEntry() == NPC_FLAMEWAKER_HEALER || summonGuid.GetEntry() == NPC_FLAMEWAKER_ELITE;
268 });
269
270 // Last remaining add
271 if (remainingAdds == 1)
272 {
273 Talk(SAY_LAST_ADD);
274 DoCastAOE(SPELL_CHAMPION);
275 }
276 // 50% of adds
277 else if (remainingAdds == 4)
278 {
279 DoCastAOE(SPELL_IMMUNE_POLY);
280 }
281 else if (!remainingAdds)
282 {
283 static_minionsGUIDS.clear();
284
285 instance->SetBossState(DATA_MAJORDOMO_EXECUTUS, DONE);
286 events.CancelEventGroup(PHASE_COMBAT);
287 me->GetMap()->UpdateEncounterState(ENCOUNTER_CREDIT_KILL_CREATURE, me->GetEntry(), me);
288 me->SetImmuneToAll(true);
289 me->SetFaction(FACTION_MAJORDOMO_FRIENDLY);
290 EnterEvadeMode();
291 Talk(SAY_DEFEAT);
292 return;
293 }
294 DoCastAOE(SPELL_ENCOURAGEMENT);
295 }

317 void UpdateAI(uint32 diff) override
318 {
319
320 switch (events.GetPhaseMask())
321 {
322 case (1 << (PHASE_COMBAT - 1)):
323 {
324 if (!UpdateVictim())
325 {
326 return;
327 }
328
329 events.Update(diff);
330
331 if (me->HasUnitState(UNIT_STATE_CASTING))
332 {
333 return;
334 }
335
336 while (uint32 const eventId = events.ExecuteEvent())
337 {
338 switch (eventId)
339 {
340 case EVENT_SHIELD_REFLECTION:
341 {
342 if (rand_chance() <= 50.f)
343 {
344 DoCastSelf(SPELL_MAGIC_REFLECTION);
345 }
346 else
347 {
348 DoCastSelf(SPELL_DAMAGE_REFLECTION);
349 }
350 events.RepeatEvent(30000);
351 break;
352 }
353 case EVENT_TELEPORT_RANDOM:
354 {
355 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 0.0f, true))
356 {
357 DoCastSelf(SPELL_HATE_TO_ZERO, true);
358 DoCast(target, SPELL_TELEPORT_RANDOM);
359 }
360
361 events.RepeatEvent(30000);
362 break;
363 }
364 case EVENT_TELEPORT_TARGET:
365 {
366 DoCastSelf(SPELL_HATE_TO_ZERO, true);
367 DoCastAOE(SPELL_TELEPORT_TARGET);
368 events.RepeatEvent(30000);
369 break;
370 }
371 }
372
373 if (me->HasUnitState(UNIT_STATE_CASTING))
374 {
375 return;
376 }
377 }
378
379 DoMeleeAttackIfReady();
380 break;
381 }
382 case (1 << (PHASE_DEFEAT_OUTRO - 1)):
383 {
384 events.Update(diff);
385 while (uint32 const eventId = events.ExecuteEvent())
386 {
387 switch (eventId)
388 {
389 case EVENT_DEFEAT_OUTRO_1:
390 {
391 Talk(SAY_DEFEAT_2);
392 events.ScheduleEvent(EVENT_DEFEAT_OUTRO_2, 8000, PHASE_DEFEAT_OUTRO, PHASE_DEFEAT_OUTRO);
393 break;
394 }
395 case EVENT_DEFEAT_OUTRO_2:
396 {
397 Talk(SAY_DEFEAT_3);
398 events.ScheduleEvent(EVENT_DEFEAT_OUTRO_3, 21500, PHASE_DEFEAT_OUTRO, PHASE_DEFEAT_OUTRO);
399 break;
400 }
401 case EVENT_DEFEAT_OUTRO_3:
402 {
403 DoCastSelf(SPELL_TELEPORT_SELF);
404 break;
405 }
406 }
407 }
408 break;
409 }
410 case (1 << (PHASE_RAGNAROS_SUMMONING - 1)):
411 {
412 events.Update(diff);
413 while (uint32 const eventId = events.ExecuteEvent())
414 {
415 switch (eventId)
416 {
417 case EVENT_RAGNAROS_SUMMON_1:
418 {
419 if (GameObject* lavaSplash = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_LAVA_SPLASH)))
420 {
421 lavaSplash->SetRespawnTime(900);
422 lavaSplash->Refresh();
423 }
424 if (GameObject* lavaSteam = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_LAVA_STEAM)))
425 {
426 lavaSteam->SetRespawnTime(900);
427 lavaSteam->Refresh();
428 }
429 Talk(SAY_RAG_SUM_2);
430 // Next event will get triggered in MovementInform
431 me->SetWalk(true);
432 me->GetMotionMaster()->MovePoint(POINT_RAGNAROS_SUMMON, MajordomoMoveRagPos, true, false);
433 break;
434 }
435 case EVENT_RAGNAROS_SUMMON_2:
436 {
437 if (GameObject* lavaSteam = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_LAVA_STEAM)))
438 {
439 me->SetFacingToObject(lavaSteam);
440 }
441
442 Talk(SAY_SUMMON_MAJ);
443 events.ScheduleEvent(EVENT_RAGNAROS_SUMMON_3, 16700, PHASE_RAGNAROS_SUMMONING, PHASE_RAGNAROS_SUMMONING);
444 events.ScheduleEvent(EVENT_RAGNAROS_EMERGE, 15000, PHASE_RAGNAROS_SUMMONING, PHASE_RAGNAROS_SUMMONING);
445 break;
446 }
447 case EVENT_RAGNAROS_SUMMON_3:
448 {
449 if (Creature* ragnaros = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_RAGNAROS)))
450 {
451 ragnaros->AI()->Talk(SAY_ARRIVAL1_RAG);
452 }
453 events.ScheduleEvent(EVENT_RAGNAROS_SUMMON_4, 11700, PHASE_RAGNAROS_SUMMONING, PHASE_RAGNAROS_SUMMONING);
454 break;
455 }
456 case EVENT_RAGNAROS_SUMMON_4:
457 {
458 Talk(SAY_ARRIVAL2_MAJ);
459 events.ScheduleEvent(EVENT_RAGNAROS_SUMMON_5, 8700, PHASE_RAGNAROS_SUMMONING, PHASE_RAGNAROS_SUMMONING);
460 break;
461 }
462 case EVENT_RAGNAROS_SUMMON_5:
463 {
464 if (Creature* ragnaros = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_RAGNAROS)))
465 {
466 ragnaros->AI()->Talk(SAY_ARRIVAL3_RAG);
467 }
468
469 events.ScheduleEvent(EVENT_RAGNAROS_SUMMON_6, 16500, PHASE_RAGNAROS_SUMMONING, PHASE_RAGNAROS_SUMMONING);
470 break;
471 }
472 case EVENT_RAGNAROS_SUMMON_6:
473 {
474 if (Creature* ragnaros = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_RAGNAROS)))

476 ragnaros->CastSpell(me, SPELL_ELEMENTAL_FIRE, true);

487 ragnaros->CastSpell(ragnaros, SPELL_RAGNA_EMERGE);
488 }
489 }break;
490 }
491 }
492 break;
493 }
