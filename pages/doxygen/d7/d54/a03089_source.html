59 void Reset() override
60 {
61 _scheduler.CancelAll();
62 summonWormSpells = { SPELL_SUMMON_WORM_1, SPELL_SUMMON_WORM_2, SPELL_SUMMON_WORM_3};
63
64 BossAI::Reset();
65 }
66
67 void SummonWorms()
68 {
69 uint32 amount = urand(1, 3);
70 Acore::Containers::RandomResize(summonWormSpells, amount);
71 for (uint32 summonSpell : summonWormSpells)
72 DoCastAOE(summonSpell, true);
73 summonWormSpells = { SPELL_SUMMON_WORM_1, SPELL_SUMMON_WORM_2, SPELL_SUMMON_WORM_3 };
74 }
75
76 void SummonHatchlingWaves()
77 {
78 for (Position spawnPos : hatchlingsSpawnPoints)
79 {
80 for (uint8 i = 0; i < MAX_HATCHLING_SPAWN; i++)
81 {
82 Position randSpawn = me->GetRandomPoint(spawnPos, 10.f);
83 me->SummonCreature(NPC_VEKNISS_HATCHLING, randSpawn, TEMPSUMMON_CORPSE_DESPAWN);
84 }
85 }
86 }
87
88 void EnterCombat(Unit* who) override
89 {
90 _scheduler.CancelAll();
91 BossAI::EnterCombat(who);
92
93 _scheduler
94 .Schedule(7s, 14s, [this](TaskContext context)
95 {
96 DoCastVictim(SPELL_MORTAL_WOUND);
97 context.Repeat();
98 })
99 .Schedule(30s, 50s, [this](TaskContext context)
100 {
101 SummonWorms();
102 context.Repeat(22s, 70s);
103 })
104 .Schedule(15s, 20s, [this](TaskContext context)
105 {
106 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 0.0f, true))
107 {
108 uint32 spellId = Acore::Containers::SelectRandomContainerElement(entangleSpells);
109 DoCast(target, spellId);
110 }
111
112 SummonHatchlingWaves();
113 context.Repeat(25s, 55s);
114 });
115 }
116
117 void UpdateAI(uint32 diff) override
118 {
119 //Return since we have no target
120 if (!UpdateVictim())
121 return;
122
123 _scheduler.Update(diff,
124 std::bind(&ScriptedAI::DoMeleeAttackIfReady, this));
125 }

127private:
128 TaskScheduler _scheduler;
129 std::vector<uint32> summonWormSpells;
130};
