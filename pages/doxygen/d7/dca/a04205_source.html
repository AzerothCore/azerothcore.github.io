62 void UpdateAI(uint32 diff) override
63 {
64 if (!UpdateVictim())
65 return;
66
67 if (canTeleport)
68 {
69 if (exileTimer <= diff)
70 {
71 if (Unit* temp = ObjectAccessor::GetUnit(*me, playerGUID))
72 {
73 temp->CastSpell(temp, SPELL_EXILE, true);
74 temp->CastSpell(temp, SPELL_BANISH_TELEPORT, true);
75 }
76 playerGUID.Clear();
77 exileTimer = 8500;
78 canTeleport = false;
79 }
80 else exileTimer -= diff;
81 }
82 else if (banishTimer <= diff)
83 {
84 Unit* temp = me->GetVictim();
85 if (temp && temp->GetTypeId() == TYPEID_PLAYER)
86 {
87 DoCast(temp, SPELL_BANISHED_SHATTRATH_A);
88 banishTimer = 9000;
89 playerGUID = temp->GetGUID();
90 if (playerGUID)
91 canTeleport = true;
92 }
93 }
94 else banishTimer -= diff;
95
96 DoMeleeAttackIfReady();
97 }
98
99 private:
100 uint32 exileTimer;
101 uint32 banishTimer;
102 ObjectGuid playerGUID;
103 bool canTeleport;
104 };
