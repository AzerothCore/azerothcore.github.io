59 static bool CheckSpellExistsAndIsValid(ChatHandler* handler, SpellInfo const* spell)
60 {
61 if (!spell)
62 {
63 handler->PSendSysMessage(LANG_COMMAND_NOSPELLFOUND);
64 handler->SetSentErrorMessage(true);
65 return false;
66 }
67
68 if (!SpellMgr::IsSpellValid(spell))
69 {
70 handler->PSendSysMessage(LANG_COMMAND_SPELL_BROKEN, spell->Id);
71 handler->SetSentErrorMessage(true);
72 return false;
73 }

77 static Optional<TriggerCastFlags> GetTriggerFlags(Optional<std::string> triggeredStr)
78 {

89 static bool HandleCastCommand(ChatHandler* handler, SpellInfo const* spell, Optional<std::string> triggeredStr)
90 {
91 Unit* target = handler->getSelectedUnit();
92 if (!target)
93 {
94 handler->SendSysMessage(LANG_SELECT_CHAR_OR_CREATURE);
95 handler->SetSentErrorMessage(true);
96 return false;
97 }
98
99 if (!CheckSpellExistsAndIsValid(handler, spell))
100 return false;
101
102 Optional<TriggerCastFlags> triggerFlags = GetTriggerFlags(triggeredStr);
103 if (!triggerFlags)
104 return false;
105
106 handler->GetSession()->GetPlayer()->CastSpell(target, spell->Id, *triggerFlags);
107
108 return true;
109 }
