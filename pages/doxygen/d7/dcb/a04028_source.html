79 events.Update(diff);
80 if (me->HasUnitState(UNIT_STATE_CASTING))
81 return;
82
83 switch (events.ExecuteEvent())
84 {
85 case EVENT_VOID_ZONE:
86 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 60.0f, true))
87 me->CastSpell(target, SPELL_VOID_ZONE, false);
88 events.ScheduleEvent(EVENT_VOID_ZONE, 15000);
89 break;
90 case EVENT_SHADOW_NOVA:
91 me->CastSpell(me, SPELL_SHADOW_NOVA, false);
92 if (roll_chance_i(50))
93 Talk(SAY_SHADOW_NOVA);
94 events.ScheduleEvent(EVENT_SHADOW_NOVA, 12000);
95 break;
96 case EVENT_SEED_OF_CORRUPTION:
97 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 30.0f, true))
98 me->CastSpell(target, SPELL_SEED_OF_CORRUPTION, false);
99 events.ScheduleEvent(EVENT_SEED_OF_CORRUPTION, 16000);
100 break;
101 }
102
103 DoMeleeAttackIfReady();
104 }
