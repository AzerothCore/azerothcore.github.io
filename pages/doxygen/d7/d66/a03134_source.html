105 {

107 {
108 me->RemoveUnitFlag(UNIT_FLAG_PLAYER_CONTROLLED);
109 me->SetStandState(UNIT_STAND_STATE_STAND);
110
111 DoCast(me, SPELL_STUNNED, true);
112
113 pCaster = Caster->GetGUID();
114
115 SayThanksTimer = 5000;
116 }
117 }
118
119 void UpdateAI(uint32 diff) override
120 {
121 if (SayThanksTimer)
122 {
123 if (SayThanksTimer <= diff)
124 {
125 me->RemoveAurasDueToSpell(SPELL_IRRIDATION);
126
127 if (Player* player = ObjectAccessor::GetPlayer(*me, pCaster))
128 {
129 Talk(SAY_HEAL, player);
130
131 player->TalkedToCreature(me->GetEntry(), me->GetGUID());
132 }
133
134 me->GetMotionMaster()->Clear();
135 me->GetMotionMaster()->MovePoint(0, -4115.053711f, -13754.831055f, 73.508949f);
136
137 RunAwayTimer = 10000;
138 SayThanksTimer = 0;
139 }
140 else SayThanksTimer -= diff;
141
142 return;
143 }
144
145 if (RunAwayTimer)
146 {
147 if (RunAwayTimer <= diff)
148 me->DespawnOrUnsummon();
149 else
150 RunAwayTimer -= diff;
151
152 return;
153 }
154
155 if (SayHelpTimer <= diff)
156 {
157 CanSayHelp = true;
158 SayHelpTimer = 20000;
159 }
160 else SayHelpTimer -= diff;
161 }
162 };

287 void UpdateEscortAI(uint32 diff) override
288 {
289 _events.Update(diff);
290 if (uint32 eventId = _events.ExecuteEvent())
291 {
292 switch (eventId)
293 {
294 case EVENT_ACCEPT_QUEST:
295 if (Player* player = ObjectAccessor::GetPlayer(*me, _player))
296 {
297 Talk(SAY_START, player);
298 }
299 me->SetFaction(FACTION_ESCORTEE_N_NEUTRAL_PASSIVE);
300 _events.ScheduleEvent(EVENT_START_ESCORT, 1000);
301 break;
302 case EVENT_START_ESCORT:
303 if (Player* player = ObjectAccessor::GetPlayer(*me, _player))
304 {
305 npc_escortAI::Start(true, false, player->GetGUID());
306 }
307 _events.ScheduleEvent(EVENT_STAND, 2000);
308 break;
309 case EVENT_STAND: // Remove kneel standstate. Using a separate delayed event because it causes unwanted delay before starting waypoint movement.
310 me->SetByteValue(UNIT_FIELD_BYTES_1, UNIT_BYTES_1_OFFSET_STAND_STATE, UNIT_STAND_STATE_STAND);
311 break;
312 case EVENT_TALK_END:
313 if (Player* player = ObjectAccessor::GetPlayer(*me, _player))
314 {
315 Talk(SAY_END1, player);
316 }
317 _events.ScheduleEvent(EVENT_COWLEN_TALK, 2000);
318 break;
319 case EVENT_COWLEN_TALK:
320 if (Creature* cowlen = me->FindNearestCreature(NPC_COWLEN, 50.0f, true))
321 {
322 cowlen->AI()->Talk(SAY_COWLEN);
323 }
324 break;
325 }
326 }
327 npc_escortAI::UpdateEscortAI(diff);
328 }
329 private:
330 EventMap _events;
331 ObjectGuid _player;
332 };
