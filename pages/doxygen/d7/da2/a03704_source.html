97 void UpdateAI(uint32 diff) override
98 {
99 if (!UpdateVictim())
100 return;
101
102 events.Update(diff);
103
104 if (me->HasUnitState(UNIT_STATE_CASTING))
105 return;
106
107 switch(events.ExecuteEvent())
108 {
109 case 0:
110 break;
111 case EVENT_SPELL_ARCANE_BARRAGE_VOLLEY:
112 me->CastSpell((Unit*)nullptr, SPELL_ARCANE_BARRAGE_VOLLEY, false);
113 events.RepeatEvent(20000);
114 break;
115 case EVENT_SPELL_ARCANE_BUFFET:
116 me->CastSpell(me->GetVictim(), SPELL_ARCANE_BUFFET, false);
117 break;
118 case EVENT_SUMMON_SPHERES:
119 {
120 Talk(SAY_SUMMON_ENERGY);
121 spheres.DespawnAll();
122 uint32 entry1 = RAND(SPELL_SUMMON_ETHEREAL_SPHERE_1, SPELL_SUMMON_ETHEREAL_SPHERE_2, SPELL_SUMMON_ETHEREAL_SPHERE_3);
123 me->CastSpell((Unit*)nullptr, entry1, true);
124 if (IsHeroic())
125 {
126 uint32 entry2;
127 do { entry2 = RAND(SPELL_SUMMON_ETHEREAL_SPHERE_1, SPELL_SUMMON_ETHEREAL_SPHERE_2, SPELL_SUMMON_ETHEREAL_SPHERE_3); }
128 while (entry1 == entry2);
129 me->CastSpell((Unit*)nullptr, entry2, true);
130 }
131 events.RepeatEvent(45000);
132 events.RescheduleEvent(EVENT_SPELL_ARCANE_BUFFET, 5000);
133 events.RescheduleEvent(EVENT_CHECK_DISTANCE, 6000);
134 }
135 break;
136 case EVENT_CHECK_DISTANCE:
137 {
138 bool found = false;
139 if (pInstance)
140 for (ObjectGuid const& guid : spheres)
141 if (Creature* c = pInstance->instance->GetCreature(guid))
142 if (me->GetDistance(c) < 3.0f)
143 {
144 c->CastSpell(me, SPELL_ARCANE_POWER, false);
145 c->DespawnOrUnsummon(8000);
146 found = true;
147 }
148 if (found)
149 {
150 Talk(SAY_CHARGED);
151 events.RepeatEvent(9000);
152 events.RescheduleEvent(EVENT_SUMMON_SPHERES, 10000);
153 }
154 else
155 events.RepeatEvent(2000);
156 }
157 break;
158 }
159
160 DoMeleeAttackIfReady();
161 }
