81 struct boss_hadronoxAI : public BossAI

83 boss_hadronoxAI(Creature* creature) : BossAI(creature, DATA_HADRONOX_EVENT)
84 {
85 }
86
87 void Reset() override
88 {
89 summons.DoAction(ACTION_DESPAWN_ADDS);
90 BossAI::Reset();
91 me->SummonCreature(NPC_ANUB_AR_CRUSHER, 542.9f, 519.5f, 741.24f, 2.14f);
92 }
93
94 void DoAction(int32 param) override
95 {
96 if (param == ACTION_START_EVENT)
97 {
98 instance->SetBossState(DATA_HADRONOX_EVENT, IN_PROGRESS);
99 me->setActive(true);
100 events.ScheduleEvent(EVENT_HADRONOX_MOVE1, 20000);
101 events.ScheduleEvent(EVENT_HADRONOX_MOVE2, 40000);
102 events.ScheduleEvent(EVENT_HADRONOX_MOVE3, 60000);
103 events.ScheduleEvent(EVENT_HADRONOX_MOVE4, 80000);
104 }
105 }
106
107 uint32 GetData(uint32 data) const override
108 {
109 if (data == me->GetEntry())
110 return !me->isActiveObject() || events.GetNextEventTime(EVENT_HADRONOX_MOVE4) != 0;
111 return 0;
112 }
113
114 void JustSummoned(Creature* summon) override
115 {
116 summons.Summon(summon);
117
118 // Xinef: cannot use pathfinding...
119 if (summon->GetDistance(477.0f, 618.0f, 771.0f) < 5.0f)
120 summon->GetMotionMaster()->MovePath(3000012, false);
121 else if (summon->GetDistance(583.0f, 617.0f, 771.0f) < 5.0f)
122 summon->GetMotionMaster()->MovePath(3000013, false);
123 else if (summon->GetDistance(581.0f, 608.5f, 739.0f) < 5.0f)
124 summon->GetMotionMaster()->MovePath(3000014, false);
125 }
126
127 void KilledUnit(Unit* victim) override
128 {
129 if (!me->IsAlive() || !victim->HasAura(SPELL_LEECH_POISON))
130 return;
131
132 me->ModifyHealth(int32(me->CountPctFromMaxHealth(10)));
133 }
134
135 void JustDied(Unit* killer) override
136 {
137 BossAI::JustDied(killer);
138 }
139
140 void EnterCombat(Unit*) override
141 {
142 events.RescheduleEvent(EVENT_HADRONOX_ACID, 10000);
143 events.RescheduleEvent(EVENT_HADRONOX_LEECH, 4000);
144 events.RescheduleEvent(EVENT_HADRONOX_PIERCE, 1000);
145 events.RescheduleEvent(EVENT_HADRONOX_GRAB, 15000);
146 }
147
148 bool AnyPlayerValid() const
149 {
150 Map::PlayerList const& playerList = me->GetMap()->GetPlayers();
151 for(Map::PlayerList::const_iterator itr = playerList.begin(); itr != playerList.end(); ++itr)
152 if (me->GetDistance(itr->GetSource()) < 130.0f && itr->GetSource()->IsAlive() && !itr->GetSource()->IsGameMaster() && me->CanCreatureAttack(itr->GetSource()))
153 return true;
154
155 return false;
156 }
157
158 void UpdateAI(uint32 diff) override
159 {
160 if (!UpdateVictim())
161 return;
162
163 events.Update(diff);
164 if (me->HasUnitState(UNIT_STATE_CASTING))
165 return;
166
167 switch (uint32 eventId = events.ExecuteEvent())
168 {
169 case EVENT_HADRONOX_PIERCE:
170 me->CastSpell(me->GetVictim(), SPELL_PIERCE_ARMOR, false);
171 events.ScheduleEvent(EVENT_HADRONOX_PIERCE, 8000);
172 break;
173 case EVENT_HADRONOX_ACID:
174 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100, false))
175 me->CastSpell(target, SPELL_ACID_CLOUD, false);
176 events.ScheduleEvent(EVENT_HADRONOX_ACID, 25000);
177 break;
178 case EVENT_HADRONOX_LEECH:
179 me->CastSpell(me, SPELL_LEECH_POISON, false);
180 events.ScheduleEvent(EVENT_HADRONOX_LEECH, 12000);
181 break;
182 case EVENT_HADRONOX_GRAB:
183 me->CastSpell(me, SPELL_WEB_GRAB, false);
184 events.ScheduleEvent(EVENT_HADRONOX_GRAB, 25000);
185 break;
186 case EVENT_HADRONOX_MOVE4:
187 me->CastSpell(me, SPELL_WEB_FRONT_DOORS, true);
188 [[fallthrough]]; // TODO: Not sure whether the fallthrough was a mistake (forgetting a break) or intended. This should be double-checked.
189 case EVENT_HADRONOX_MOVE1:
190 case EVENT_HADRONOX_MOVE2:
191 case EVENT_HADRONOX_MOVE3:
192 Talk(SAY_HADRONOX_EMOTE);
193 me->GetMotionMaster()->MoveCharge(hadronoxSteps[eventId - 1].GetPositionX(), hadronoxSteps[eventId - 1].GetPositionY(), hadronoxSteps[eventId - 1].GetPositionZ(), 10.0f, 0, nullptr, true);
194 break;
195 }
196
197 DoMeleeAttackIfReady();
198 }
199
200 bool CheckEvadeIfOutOfCombatArea() const override
201 {
202 return me->isActiveObject() && !AnyPlayerValid();
203 }
204 };
205
206 CreatureAI* GetAI(Creature* creature) const override
207 {
208 return GetAzjolNerubAI<boss_hadronoxAI>(creature);
209 }
