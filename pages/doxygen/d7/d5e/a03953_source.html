141 void Load(char const* str) override
142 {
143 if (!str)
144 {
145 OUT_LOAD_INST_DATA_FAIL;
146 return;
147 }
148
149 OUT_LOAD_INST_DATA(str);
150
151 char dataHead1, dataHead2;
152
153 std::istringstream loadStream(str);
154 loadStream >> dataHead1 >> dataHead2;
155
156 if (dataHead1 == 'G' && dataHead2 == 'L')
157 {
158 for (uint32 i = 0; i < MAX_ENCOUNTER; ++i)
159 {
160 uint32 tmpState;
161 loadStream >> tmpState;
162 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
163 tmpState = NOT_STARTED;
164 SetBossState(i, EncounterState(tmpState));
165 }
166 }
167 else
168 OUT_LOAD_INST_DATA_FAIL;
169
170 OUT_LOAD_INST_DATA_COMPLETE;
171 }
172
173 protected:
174 uint32 _addsKilled;
175 ObjectGuid _maulgarGUID;
176 };
