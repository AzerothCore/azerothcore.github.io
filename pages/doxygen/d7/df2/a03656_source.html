125 {

128 instance->SetData(DATA_SVALA_SORROWGRAVE, NOT_STARTED);
129 instance->SetData(DATA_SVALA_ACHIEVEMENT, false);

151 void SetData(uint32 data, uint32 param) override
152 {
153 if (data != 1 || param != 1 || Started || (instance && instance->GetData(DATA_SVALA_SORROWGRAVE) == DONE))
154 return;
155
156 me->SetImmuneToAll(true);
157 Started = true;
158 me->setActive(true);
159 events2.ScheduleEvent(EVENT_SVALA_START, 5000);
160 if (Creature* pArthas = me->SummonCreature(NPC_ARTHAS, 295.81f, -366.16f, 92.57f, 1.58f, TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 59000))

173 summons.Summon(summon);

175 summon->CastSpell(summon, SPELL_TELEPORT_VISUAL, true);
176 }

180 me->SetInCombatWithZone();

185 events.ScheduleEvent(EVENT_SORROWGRAVE_RITUAL, 25000);

206
207 if (victim->GetTypeId() == TYPEID_PLAYER)
208 Talk(SAY_SLAY);

211 void UpdateAI(uint32 diff) override
212 {
213 events2.Update(diff);
214 switch (events2.ExecuteEvent())
215 {
216 case EVENT_SVALA_START:
217 Talk(TALK_INTRO_S1);
218 events2.ScheduleEvent(EVENT_SVALA_TALK1, 8000);
219 break;
220 case EVENT_SVALA_TALK1:
221 if (Creature* Arthas = ObjectAccessor::GetCreature(*me, ArthasGUID))
222 Arthas->AI()->Talk(TALK_INTRO_A1);
223 events2.ScheduleEvent(EVENT_SVALA_TALK2, 9000);
224 break;
225 case EVENT_SVALA_TALK2:
226 if (Creature* Arthas = ObjectAccessor::GetCreature(*me, ArthasGUID))
227 Arthas->CastSpell(me, SPELL_ARTHAS_TRANSFORMING_SVALA, false);
228 me->CastSpell(me, SPELL_SVALA_TRANSFORMING2, true);
229 events2.ScheduleEvent(EVENT_SVALA_TALK3, 3000);
230 break;
231 case EVENT_SVALA_TALK3:
232 me->SetFloatValue(UNIT_FIELD_HOVERHEIGHT, 6.0f);
233 me->SetHover(true);
234 me->AddUnitState(UNIT_STATE_NO_ENVIRONMENT_UPD);
235 events2.ScheduleEvent(30, 1000);
236 events2.ScheduleEvent(EVENT_SVALA_TALK4, 9000);
237 break;
238 case 30:
239 {
240 WorldPacket data(SMSG_SPLINE_MOVE_SET_HOVER, 9);
241 data << me->GetPackGUID();
242 me->SendMessageToSet(&data, false);
243 break;
244 }
245 case EVENT_SVALA_TALK4:
246 {
247 me->CastSpell(me, SPELL_SVALA_TRANSFORMING1, true);
248 me->UpdateEntry(NPC_SVALA_SORROWGRAVE);
249 me->SetCorpseDelay(sWorld->getIntConfig(CONFIG_CORPSE_DECAY_ELITE));
250 me->SetFloatValue(UNIT_FIELD_HOVERHEIGHT, 6.0f);
251 me->SetImmuneToAll(true);
252 if (Creature* Arthas = ObjectAccessor::GetCreature(*me, ArthasGUID))
253 Arthas->InterruptNonMeleeSpells(false);
254 me->RemoveAllAuras();
255 me->SetWalk(false);
256 events2.ScheduleEvent(EVENT_SVALA_TALK5, 2000);
257
258 std::list<Creature*> creatureList;
259 me->GetCreaturesWithEntryInRange(creatureList, 100.0f, NPC_DRAGONFLAYER_SPECTATOR);
260 for (std::list<Creature*>::const_iterator itr = creatureList.begin(); itr != creatureList.end(); ++itr)
261 (*itr)->AI()->SetData(1, 2);
262
263 break;
264 }
265 case EVENT_SVALA_TALK5:
266 Talk(TALK_INTRO_S2);
267 events2.ScheduleEvent(EVENT_SVALA_TALK6, 12000);
268 break;
269 case EVENT_SVALA_TALK6:
270 if (Creature* Arthas = ObjectAccessor::GetCreature(*me, ArthasGUID))
271 Arthas->AI()->Talk(TALK_INTRO_A2);
272 events2.ScheduleEvent(EVENT_SVALA_TALK7, 9000);
273 break;
274 case EVENT_SVALA_TALK7:
275 me->SetFacingTo(M_PI / 2.0f);
276 Talk(TALK_INTRO_S3);
277 if (GameObject* mirror = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(GO_SVALA_MIRROR)))
278 mirror->SetGoState(GO_STATE_ACTIVE);
279 events2.ScheduleEvent(EVENT_SVALA_TALK8, 13000);
280 break;
281 case EVENT_SVALA_TALK8:
282 me->GetMotionMaster()->MoveFall(0, true);
283 events2.ScheduleEvent(EVENT_SVALA_TALK9, 2000);
284 break;
285 case EVENT_SVALA_TALK9:
286 me->SetFloatValue(UNIT_FIELD_HOVERHEIGHT, 3.0f);
287 me->SetImmuneToAll(false);
288 me->LoadEquipment(1, true);
289 me->setActive(false);
290 if (Player* target = SelectTargetFromPlayerList(100.0f))
291 AttackStart(target);
292 return;
293 }
294
295 if (!UpdateVictim())
296 return;
297
298 events.Update(diff);
299 if (me->HasUnitState(UNIT_STATE_CASTING))
300 return;
301
302 switch (events.ExecuteEvent())
303 {
304 case EVENT_SORROWGRAVE_SS:
305 me->CastSpell(me->GetVictim(), IsHeroic() ? SPELL_SINSTER_STRIKE_H : SPELL_SINSTER_STRIKE_N, false);
306 events.ScheduleEvent(EVENT_SORROWGRAVE_SS, urand(3000, 5000));
307 break;
308 case EVENT_SORROWGRAVE_FLAMES:
309 summons.DespawnAll();
310 me->CastSpell(me, SPELL_CALL_FLAMES, false);
311 events.ScheduleEvent(EVENT_SORROWGRAVE_FLAMES2, 500);
312 events.ScheduleEvent(EVENT_SORROWGRAVE_FLAMES2, 1000);
313 events.ScheduleEvent(EVENT_SORROWGRAVE_FLAMES, urand(8000, 12000));
314 break;
315 case EVENT_SORROWGRAVE_FLAMES2:
316 {
317 std::list<Creature*> braziers;
318 me->GetCreaturesWithEntryInRange(braziers, 100.0f, NPC_FLAME_BRAZIER);
319 if (!braziers.empty())
320 {
321 for (std::list<Creature*>::const_iterator itr = braziers.begin(); itr != braziers.end(); ++itr)
322 (*itr)->CastCustomSpell(SPELL_BALL_OF_FLAME, SPELLVALUE_MAX_TARGETS, 1, (*itr), true);
323 }
324 break;
325 }
326 case EVENT_SORROWGRAVE_RITUAL:
327 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true))
328 {
329 Talk(SAY_SACRIFICE_PLAYER);
330
331 for (uint8 i = 0; i < 3; ++i)
332 if (Creature* cr = me->SummonCreature(NPC_RITUAL_CHANNELER, RitualChannelerLoc[i], TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 360000))
333 cr->AI()->AttackStart(target);
334
335 me->GetMotionMaster()->MoveIdle();
336 DoTeleportPlayer(target, 296.632f, -346.075f, 90.63f, 4.6f);
337 me->NearTeleportTo(296.632f, -346.075f, 110.0f, 4.6f, false);
338 me->SetControlled(true, UNIT_STATE_ROOT);
339 me->RemoveAurasByType(SPELL_AURA_PERIODIC_DAMAGE);
340 me->RemoveAurasByType(SPELL_AURA_PERIODIC_DAMAGE_PERCENT);
341 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
342 }
343
344 events.DelayEvents(25001); // +1 just to be sure
345 events.ScheduleEvent(EVENT_SORROWGRAVE_RITUAL_SPELLS, 0);
346 events.ScheduleEvent(EVENT_SORROWGRAVE_FINISH_RITUAL, 25000);
347 return;
348 case EVENT_SORROWGRAVE_RITUAL_SPELLS:
349 //me->CastSpell(me, SPELL_RITUAL_OF_THE_SWORD, false);
350 me->CastSpell(me, SPELL_RITUAL_STRIKE, true);
351 return;
352 case EVENT_SORROWGRAVE_FINISH_RITUAL:
353 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
354 me->SetControlled(false, UNIT_STATE_ROOT);
355 AttackStart(me->GetVictim());
356 me->GetMotionMaster()->MoveFall(0, true);
357 summons.DespawnAll();
358 break;
359 }
360
361 DoMeleeAttackIfReady();
362 }
363 };
