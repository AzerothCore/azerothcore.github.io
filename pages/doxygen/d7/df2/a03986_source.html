105 {
106 boss_magtheridonAI(Creature* creature) : BossAI(creature, TYPE_MAGTHERIDON) { }
107
108 EventMap events2;
109
110 void Reset() override
111 {
112 events2.Reset();
113 events2.ScheduleEvent(EVENT_RANDOM_TAUNT, 90000);
114 _Reset();
115 me->CastSpell(me, SPELL_SHADOW_CAGE, true);
116 me->SetReactState(REACT_PASSIVE);
117 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
118 me->SetImmuneToPC(true);
119 }
120
121 void KilledUnit(Unit* /*victim*/) override
122 {
123 if (events.GetNextEventTime(EVENT_RECENTLY_SPOKEN) == 0)
124 {
125 events.ScheduleEvent(EVENT_RECENTLY_SPOKEN, 5000);
126 Talk(SAY_SLAY);
127 }
128 }
129
130 void JustDied(Unit* /*killer*/) override
131 {
132 _JustDied();
133 Talk(SAY_DEATH);
134 }
135
136 void MoveInLineOfSight(Unit* /*who*/) override { }
137
138 void EnterCombat(Unit* /*who*/) override
139 {
140 events2.Reset();
141 _EnterCombat();
142 events.ScheduleEvent(EVENT_EMOTE1, 0);
143 events.ScheduleEvent(EVENT_EMOTE2, 60000);
144 events.ScheduleEvent(EVENT_EMOTE3, 120000);
145 events.ScheduleEvent(EVENT_ENTER_COMBAT, 123000);
146 }
147
148 void UpdateAI(uint32 diff) override
149 {
150 events2.Update(diff);
151 switch (events2.ExecuteEvent())
152 {
153 case EVENT_RANDOM_TAUNT:
154 Talk(SAY_TAUNT);
155 events2.ScheduleEvent(EVENT_RANDOM_TAUNT, 90000);
156 break;
157 case EVENT_CHECK_GRASP:
158 if (me->GetAuraCount(SPELL_SHADOW_GRASP_VISUAL) == 5)
159 {
160 Talk(SAY_BANISH);
161 me->InterruptNonMeleeSpells(true);
162 break;
163 }
164 events2.ScheduleEvent(EVENT_CHECK_GRASP, 0);
165 break;
166 }
167
168 if (!UpdateVictim())
169 return;
170
171 events.Update(diff);
172 if (me->HasUnitState(UNIT_STATE_CASTING))
173 return;
174
175 switch (events.ExecuteEvent())
176 {
177 case EVENT_EMOTE1:
178 Talk(SAY_EMOTE_BEGIN);
179 break;
180 case EVENT_EMOTE2:
181 Talk(SAY_EMOTE_NEARLY);
182 break;
183 case EVENT_EMOTE3:
184 Talk(SAY_EMOTE_FREE);
185 Talk(SAY_FREE);
186 break;
187 case EVENT_ENTER_COMBAT:
188 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
189 me->SetImmuneToPC(false);
190 me->SetReactState(REACT_AGGRESSIVE);
191 events.ScheduleEvent(EVENT_CLEAVE, 9000);
192 events.ScheduleEvent(EVENT_BLAST_NOVA, 60000);
193 events.ScheduleEvent(EVENT_BLAZE, 10000);
194 events.ScheduleEvent(EVENT_QUAKE, 40000);
195 events.ScheduleEvent(EVENT_CHECK_HEALTH, 500);
196 events.ScheduleEvent(EVENT_ENRAGE, 22 * MINUTE * IN_MILLISECONDS);
197
198 instance->SetData(DATA_ACTIVATE_CUBES, 1);
199 me->RemoveAurasDueToSpell(SPELL_SHADOW_CAGE);
200 break;
201 case EVENT_CLEAVE:
202 me->CastSpell(me->GetVictim(), SPELL_CLEAVE, false);
203 events.ScheduleEvent(EVENT_CLEAVE, 10000);
204 break;
205 case EVENT_BLAST_NOVA:
206 me->CastSpell(me, SPELL_BLAST_NOVA, false);
207 events.ScheduleEvent(EVENT_BLAST_NOVA, 60000);
208 events.ScheduleEvent(EVENT_CANCEL_GRASP_CHECK, 12000);
209 events2.ScheduleEvent(EVENT_CHECK_GRASP, 0);
210 break;
211 case EVENT_BLAZE:
212 me->CastCustomSpell(SPELL_BLAZE, SPELLVALUE_MAX_TARGETS, 1);
213 events.ScheduleEvent(EVENT_BLAZE, 30000);
214 break;
215 case EVENT_ENRAGE:
216 me->CastSpell(me, SPELL_BERSERK, true);
217 break;
218 case EVENT_CANCEL_GRASP_CHECK:
219 events2.Reset();
220 break;
221 case EVENT_QUAKE:
222 me->CastSpell(me, SPELL_QUAKE, false);
223 events.ScheduleEvent(EVENT_QUAKE, 50000);
224 break;
225 case EVENT_CHECK_HEALTH:
226 if (me->HealthBelowPct(30))
227 {
228 Talk(SAY_PHASE3);
229 events.SetPhase(1);
230 events.DelayEvents(18000);
231 events.ScheduleEvent(EVENT_COLLAPSE_CEIL, 8000);
232 events.ScheduleEvent(EVENT_COLLAPSE_DAMAGE, 15000);
233 break;
234 }
235 events.ScheduleEvent(EVENT_CHECK_HEALTH, 500);
236 break;
237 case EVENT_COLLAPSE_CEIL:
238 me->CastSpell(me, SPELL_CAMERA_SHAKE, true);
239 instance->SetData(DATA_COLLAPSE, GO_STATE_ACTIVE);
240 break;
241 case EVENT_COLLAPSE_DAMAGE:
242 me->CastSpell(me, SPELL_COLLAPSE_DAMAGE, true);
243 me->resetAttackTimer();
244 events.SetPhase(0);
245 events.ScheduleEvent(EVENT_DEBRIS, 20000);
246 break;
247 case EVENT_DEBRIS:
248 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
249 {
250 target->CastSpell(target, SPELL_DEBRIS_VISUAL, true, nullptr, nullptr, me->GetGUID());
251 me->m_Events.AddEvent(new DealDebrisDamage(*me, target->GetGUID()), me->m_Events.CalculateTime(5000));
252 }
253 events.ScheduleEvent(EVENT_DEBRIS, 20000);
254 break;
255 }
256
257 if (!events.IsInPhase(1))
258 DoMeleeAttackIfReady();
259 }
260 };
