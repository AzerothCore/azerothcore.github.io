66InstanceSave* InstanceSaveMgr::AddInstanceSave(uint32 mapId, uint32 instanceId, Difficulty difficulty, bool startup /*=false*/)
67{
68 ASSERT(!GetInstanceSave(instanceId));
69
70 MapEntry const* entry = sMapStore.LookupEntry(mapId);
71 if (!entry)
72 {
73 LOG_ERROR("instance.save", "InstanceSaveMgr::AddInstanceSave: wrong mapid = {}, instanceid = {}!", mapId, instanceId);
74 return nullptr;
75 }
76
77 if (instanceId == 0)
78 {
79 LOG_ERROR("instance.save", "InstanceSaveMgr::AddInstanceSave: mapid = {}, wrong instanceid = {}!", mapId, instanceId);
80 return nullptr;
81 }
82
83 if (difficulty >= (entry->IsRaid() ? MAX_RAID_DIFFICULTY : MAX_DUNGEON_DIFFICULTY))
84 {
85 LOG_ERROR("instance.save", "InstanceSaveMgr::AddInstanceSave: mapid = {}, instanceid = {}, wrong dificalty {}!", mapId, instanceId, difficulty);
86 return nullptr;
87 }
88
89 time_t resetTime, extendedResetTime;
90 if (entry->map_type == MAP_RAID || difficulty > DUNGEON_DIFFICULTY_NORMAL)
91 {
92 resetTime = GetResetTimeFor(mapId, difficulty);
93 extendedResetTime = GetExtendedResetTimeFor(mapId, difficulty);
94 }
95 else
96 {
97 resetTime = GameTime::GetGameTime().count() + static_cast<long long>(3) * DAY; // normals expire after 3 days even if someone is still bound to them, cleared on startup
98 extendedResetTime = 0;
99 }
100 InstanceSave* save = new InstanceSave(mapId, instanceId, difficulty, resetTime, extendedResetTime);
101 if (!startup)
102 save->InsertToDB();
103
104 m_instanceSaveById[instanceId] = save;
105 return save;
106}

270

273 CharacterDatabase.DirectExecute("DELETE FROM gameobject_respawn WHERE instanceId > 0 AND instanceId NOT IN (SELECT id FROM instance)");

275
276 // Clean invalid references to instance
277 CharacterDatabase.DirectExecute("UPDATE corpse SET instanceId = 0 WHERE instanceId > 0 AND instanceId NOT IN (SELECT id FROM instance)");
278 CharacterDatabase.DirectExecute("UPDATE characters AS tmp LEFT JOIN instance ON tmp.instance_id = instance.id SET tmp.instance_id = 0 WHERE tmp.instance_id > 0 AND instance.id IS NULL");

297void InstanceSaveMgr::LoadResetTimes()
298{
299 time_t now = GameTime::GetGameTime().count();
300 time_t today = (now / DAY) * DAY;
301
302 // load the global respawn times for raid/heroic instances
303 uint32 diff = sWorld->getIntConfig(CONFIG_INSTANCE_RESET_TIME_HOUR) * HOUR;
304 QueryResult result = CharacterDatabase.Query("SELECT mapid, difficulty, resettime FROM instance_reset");
305 if (result)
306 {
307 do
308 {
309 Field* fields = result->Fetch();
310 uint32 mapid = fields[0].Get<uint16>();
311 Difficulty difficulty = Difficulty(fields[1].Get<uint8>());
312 uint64 resettime = fields[2].Get<uint32>();
313
314 MapDifficulty const* mapDiff = GetMapDifficultyData(mapid, difficulty);
315 if (!mapDiff)
316 {
317 LOG_ERROR("instance.save", "InstanceSaveMgr::LoadResetTimes: invalid mapid({})/difficulty({}) pair in instance_reset!", mapid, difficulty);
318 CharacterDatabase.DirectExecute("DELETE FROM instance_reset WHERE mapid = '{}' AND difficulty = '{}'", mapid, difficulty);
319 continue;
320 }
321
322 SetResetTimeFor(mapid, difficulty, resettime);
323 } while (result->NextRow());
324 }
325
326 // calculate new global reset times for expired instances and those that have never been reset yet
327 // add the global reset times to the priority queue
328 for (MapDifficultyMap::const_iterator itr = sMapDifficultyMap.begin(); itr != sMapDifficultyMap.end(); ++itr)
329 {
330 uint32 map_diff_pair = itr->first;
331 uint32 mapid = PAIR32_LOPART(map_diff_pair);
332 Difficulty difficulty = Difficulty(PAIR32_HIPART(map_diff_pair));
333 MapDifficulty const* mapDiff = &itr->second;
334 if (!mapDiff->resetTime)
335 continue;
336
337 // the reset_delay must be at least one day
338 uint32 period = uint32(((mapDiff->resetTime * sWorld->getRate(RATE_INSTANCE_RESET_TIME)) / DAY) * DAY);
339 if (period < DAY)
340 period = DAY;
341
342 time_t t = GetResetTimeFor(mapid, difficulty);
343 if (!t)
344 {
345 // initialize the reset time
346 t = today + period + diff;
347 SetResetTimeFor(mapid, difficulty, t);
348 CharacterDatabase.DirectExecute("INSERT INTO instance_reset VALUES ('{}', '{}', '{}')", mapid, difficulty, (uint32)t);
349 }
350
351 if (t < now)
352 {
353 // assume that expired instances have already been cleaned
354 // calculate the next reset time
355 t = (t * DAY) / DAY;
356 t += ((today - t) / period + 1) * period + diff;
357 CharacterDatabase.DirectExecute("UPDATE instance_reset SET resettime = '{}' WHERE mapid = '{}' AND difficulty = '{}'", (uint32)t, mapid, difficulty);
358 }
359
360 SetExtendedResetTimeFor(mapid, difficulty, t);
361
362 // schedule the global reset/warning
363 uint8 type;
364 for (type = 1; type < 5; ++type)
365 if (now + ResetTimeDelay[type - 1] < t)
366 break;
367
368 ScheduleReset(t - ResetTimeDelay[type - 1], InstResetEvent(type, mapid, difficulty));
369 }
370}

460void InstanceSaveMgr::Update()
461{
462 time_t now = GameTime::GetGameTime().count();
463 time_t t;
464 bool resetOccurred = false;
465
466 while (!m_resetTimeQueue.empty())
467 {
468 t = m_resetTimeQueue.begin()->first;
469 if (t >= now)
470 break;
471
472 InstResetEvent& event = m_resetTimeQueue.begin()->second;
473 if (event.type)
474 {
475 // global reset/warning for a certain map
476 time_t resetTime = GetResetTimeFor(event.mapid, event.difficulty);
477 bool warn = event.type < 5;
478 _ResetOrWarnAll(event.mapid, event.difficulty, warn, resetTime);
479 if (warn)
480 {
481 // schedule the next warning/reset
482 ++event.type;
483 ScheduleReset(resetTime - ResetTimeDelay[event.type - 1], event);
484 }
485 else
486 resetOccurred = true;
487 }
488 m_resetTimeQueue.erase(m_resetTimeQueue.begin());
489 }
490
491 // pussywizard: send updated calendar and raid info
492 if (resetOccurred)
493 {
494 LOG_INFO("instance.save", "Instance ID reset occurred, sending updated calendar and raid info to all players!");
495 WorldPacket dummy;
496 for (SessionMap::const_iterator itr = sWorld->GetAllSessions().begin(); itr != sWorld->GetAllSessions().end(); ++itr)
497 if (Player* plr = itr->second->GetPlayer())
498 {
499 itr->second->HandleCalendarGetCalendar(dummy);
500 plr->SendRaidInfo();
501 }
502 }

505void InstanceSaveMgr::_ResetSave(InstanceSaveHashMap::iterator& itr)
506{
507 lock_instLists = true;
508
509 GuidList& pList = itr->second->m_playerList;
510 for (GuidList::iterator iter = pList.begin(), iter2; iter != pList.end(); )
511 {
512 iter2 = iter++;
513 PlayerUnbindInstanceNotExtended(*iter2, itr->second->GetMapId(), itr->second->GetDifficulty(), ObjectAccessor::FindConnectedPlayer(*iter2));
514 }
515
516 // delete stuff if no players left (noone extended id)
517 if (pList.empty())
518 {
519 // delete character_instance per id, delete instance per id
520 CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CHAR_INSTANCE_BY_INSTANCE);
521 stmt->SetData(0, itr->second->GetInstanceId());
522 CharacterDatabase.Execute(stmt);
523 stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_INSTANCE_BY_INSTANCE);
524 stmt->SetData(0, itr->second->GetInstanceId());
525 CharacterDatabase.Execute(stmt);
526 DeleteInstanceSavedData(itr->second->GetInstanceId());
527
528 // clear respawn times if the map is already unloaded and won't do it by itself
529 if (!sMapMgr->FindMap(itr->second->GetMapId(), itr->second->GetInstanceId()))
530 Map::DeleteRespawnTimesInDB(itr->second->GetMapId(), itr->second->GetInstanceId());
531
532 delete itr->second;
533 m_instanceSaveById.erase(itr);
534 }
535 else
536 {
537 // delete character_instance per id where extended = 0, transtaction with set extended = 0, transaction is used to avoid mysql thread races
538 CharacterDatabaseTransaction trans = CharacterDatabase.BeginTransaction();
539 CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CHAR_INSTANCE_BY_INSTANCE_NOT_EXTENDED);
540 stmt->SetData(0, itr->second->GetInstanceId());
541 trans->Append(stmt);
542 stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_CHAR_INSTANCE_SET_NOT_EXTENDED);
543 stmt->SetData(0, itr->second->GetInstanceId());
544 trans->Append(stmt);
545 CharacterDatabase.CommitTransaction(trans);
546
547 // update reset time and extended reset time for instance save
548 itr->second->SetResetTime(GetResetTimeFor(itr->second->GetMapId(), itr->second->GetDifficulty()));
549 itr->second->SetExtendedResetTime(GetExtendedResetTimeFor(itr->second->GetMapId(), itr->second->GetDifficulty()));
550 }
551
552 lock_instLists = false;
553}

555void InstanceSaveMgr::_ResetOrWarnAll(uint32 mapid, Difficulty difficulty, bool warn, time_t resetTime)
556{
557 // global reset for all instances of the given map
558 MapEntry const* mapEntry = sMapStore.LookupEntry(mapid);
559 if (!mapEntry->Instanceable())
560 return;
561
562 time_t now = GameTime::GetGameTime().count();
563
564 if (!warn)
565 {
566 MapDifficulty const* mapDiff = GetMapDifficultyData(mapid, difficulty);
567 if (!mapDiff || !mapDiff->resetTime)
568 {
569 LOG_ERROR("instance.save", "InstanceSaveMgr::ResetOrWarnAll: not valid difficulty or no reset delay for map {}", mapid);
570 return;
571 }
572
573 // calculate the next reset time
574 uint32 diff = sWorld->getIntConfig(CONFIG_INSTANCE_RESET_TIME_HOUR) * HOUR;
575
576 uint32 period = uint32(((mapDiff->resetTime * sWorld->getRate(RATE_INSTANCE_RESET_TIME)) / DAY) * DAY);
577 if (period < DAY)
578 period = DAY;
579
580 uint32 next_reset = uint32(((resetTime + MINUTE) / DAY * DAY) + period + diff);
581 SetResetTimeFor(mapid, difficulty, next_reset);
582 SetExtendedResetTimeFor(mapid, difficulty, next_reset + period);
583 ScheduleReset(time_t(next_reset - 3600), InstResetEvent(1, mapid, difficulty));
584
585 // update it in the DB
586 CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_GLOBAL_INSTANCE_RESETTIME);
587 stmt->SetData(0, next_reset);
588 stmt->SetData(1, uint16(mapid));
589 stmt->SetData(2, uint8(difficulty));
590 CharacterDatabase.Execute(stmt);
591
592 // remove all binds to instances of the given map and delete from db (delete per instance id, no mass deletion!)
593 // do this after new reset time is calculated
594 for (InstanceSaveHashMap::iterator itr = m_instanceSaveById.begin(), itr2; itr != m_instanceSaveById.end(); )
595 {
596 itr2 = itr++;
597 if (itr2->second->GetMapId() == mapid && itr2->second->GetDifficulty() == difficulty)
598 _ResetSave(itr2);
599 }
600 }
601
602 // now loop all existing maps to warn / reset
603 Map const* map = sMapMgr->CreateBaseMap(mapid);
604 MapInstanced::InstancedMaps& instMaps = ((MapInstanced*)map)->GetInstancedMaps();
605 MapInstanced::InstancedMaps::iterator mitr;
606 uint32 timeLeft;
607
608 for (mitr = instMaps.begin(); mitr != instMaps.end(); ++mitr)
609 {
610 Map* map2 = mitr->second;
611 if (!map2->IsDungeon() || map2->GetDifficulty() != difficulty)
612 continue;
613
614 if (warn)
615 {
616 if (now >= resetTime)
617 timeLeft = 0;
618 else
619 timeLeft = uint32(resetTime - now);
620
621 map2->ToInstanceMap()->SendResetWarnings(timeLeft);
622 }
623 else
624 {
625 InstanceSave* save = GetInstanceSave(map2->GetInstanceId());
626 map2->ToInstanceMap()->Reset(INSTANCE_RESET_GLOBAL, (save ? & (save->m_playerList) : nullptr));
627 }
628 }
629}
