103 void EnterCombat(Unit* /*who*/) override
104 {
105 _scheduler.Schedule(6s, [this](TaskContext context)
106 {
107 std::list<Unit*> targets;
108 SelectTargetList(targets, [&](Unit* target)
109 {
110 return target && target->IsPlayer() && target->GetPower(POWER_MANA) > 0;
111 }, 6, SelectTargetMethod::Random);
112
113 for (Unit* target : targets)
114 {
115 DoCast(target, SPELL_DRAIN_MANA, true);
116 }
117
118 if (me->GetPowerPct(POWER_MANA) >= 100.f)
119 {
120 DoCastAOE(SPELL_PURGE, true);
121 }
122
123 context.Repeat(6s);
124 });
125 }
