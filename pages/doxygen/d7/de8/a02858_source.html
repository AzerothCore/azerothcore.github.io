117void RevivePlayer(Unit* victim, ObjectGuid& reviveGUID)
118{
119 std::list<Creature*> chainedSpirits;
120 GetCreatureListWithEntryInGrid(chainedSpirits, victim, NPC_CHAINED_SPIRIT, 200.f);
121 if (chainedSpirits.empty())
122 return;
123
124 // Sort the list by distance to the victim.
125 chainedSpirits.sort([victim](Creature const* c1, Creature const* c2)
126 {
127 return c1->GetDistance2d(victim) < c2->GetDistance2d(victim);
128 });
129
130 // Now we have to check if the spirit is already reviving someone...
131 for (Creature* spirit : chainedSpirits)
132 {
133 if (!spirit->isMoving() && !spirit->HasUnitState(UNIT_STATE_CASTING))
134 {
135 spirit->AI()->SetGUID(reviveGUID);
136 spirit->AI()->DoAction(ACTION_REVIVE);
137 reviveGUID.Clear();
138 break;
139 }
140 }

322 void DoMeleeAttackIfReady(bool ignoreCasting)
323 {
324 if (!ignoreCasting && me->HasUnitState(UNIT_STATE_CASTING))
325 {
326 return;
327 }
328
329 Unit* victim = me->GetVictim();
330 if (!victim || !victim->IsInWorld())
331 return;
332
333 if (!me->IsWithinMeleeRange(victim))
334 return;
335
336 //Make sure our attack is ready and we aren't currently casting before checking distance
337 if (me->isAttackReady())
338 {
339 // xinef: prevent base and off attack in same time, delay attack at 0.2 sec
340 if (me->haveOffhandWeapon())
341 if (me->getAttackTimer(OFF_ATTACK) < ATTACK_DISPLAY_DELAY)
342 me->setAttackTimer(OFF_ATTACK, ATTACK_DISPLAY_DELAY);
343
344 me->AttackerStateUpdate(victim, BASE_ATTACK, false, ignoreCasting);
345 me->resetAttackTimer();
346 }
347
348 if (me->haveOffhandWeapon() && me->isAttackReady(OFF_ATTACK))
349 {
350 // xinef: delay main hand attack if both will hit at the same time (players code)
351 if (me->getAttackTimer(BASE_ATTACK) < ATTACK_DISPLAY_DELAY)
352 me->setAttackTimer(BASE_ATTACK, ATTACK_DISPLAY_DELAY);
353
354 me->AttackerStateUpdate(victim, OFF_ATTACK, false, ignoreCasting);
355 me->resetAttackTimer(OFF_ATTACK);
356 }
357 }

359 void UpdateAI(uint32 diff) override
360 {
361 events.Update(diff);
362
363 if (!UpdateVictim())
364 {
365 if (instance->GetBossState(DATA_MANDOKIR) == NOT_STARTED || instance->GetBossState(DATA_MANDOKIR) == SPECIAL)
366 {
367 while (uint32 eventId = events.ExecuteEvent())
368 {
369 switch (eventId)
370 {
371 case EVENT_CHECK_START:
372 if (instance->GetBossState(DATA_MANDOKIR) == SPECIAL)
373 {
374 me->GetMotionMaster()->MovePoint(0, PosMandokir[1].m_positionX, PosMandokir[1].m_positionY, PosMandokir[1].m_positionZ);
375 events.ScheduleEvent(EVENT_STARTED, 6000);
376 }
377 else
378 {
379 events.ScheduleEvent(EVENT_CHECK_START, 1000);
380 }
381 break;
382 case EVENT_STARTED:
383 me->SetImmuneToAll(false);
384 me->SetInCombatWithZone();
385 break;
386 default:
387 break;
388 }
389 }
390 }
391 return;
392 }
393
394 if (me->HasUnitState(UNIT_STATE_CASTING) || me->HasUnitState(UNIT_STATE_CHARGING))
395 {
396 if (me->GetCurrentSpellCastTime(SPELL_WATCH) >= 0)
397 {
398 DoMeleeAttackIfReady(true);
399 }
400
401 return;
402 }
403
404 while (uint32 eventId = events.ExecuteEvent())
405 {
406 switch (eventId)
407 {
408 case EVENT_OVERPOWER:
409 if (DoCastVictim(SPELL_OVERPOWER) == SPELL_CAST_OK)
410 {
411 events.ScheduleEvent(EVENT_OVERPOWER, urand(6000, 8000));
412 }
413 else
414 {
415 events.ScheduleEvent(EVENT_OVERPOWER, 1000);
416 }
417 break;
418 case EVENT_MORTAL_STRIKE:
419 DoCastVictim(SPELL_MORTAL_STRIKE);
420 events.ScheduleEvent(EVENT_MORTAL_STRIKE, urand(14000, 28000));
421 break;
422 case EVENT_WHIRLWIND:
423 DoCast(me, SPELL_WHIRLWIND);
424 events.ScheduleEvent(EVENT_WHIRLWIND, urand(22000, 26000));
425 break;
426 case EVENT_CHECK_OHGAN:
427 if (instance->GetBossState(DATA_OHGAN) == DONE)
428 {
429 DoCast(me, SPELL_FRENZY);
430 Talk(SAY_OHGAN_DEAD);
431 }
432 else
433 {
434 events.ScheduleEvent(EVENT_CHECK_OHGAN, 1000);
435 }
436 break;
437 case EVENT_WATCH_PLAYER:
438 if (Unit* player = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
439 {
440 DoCast(player, SPELL_WATCH);
441 Talk(SAY_WATCH, player);
442 _chargeTarget = std::make_pair(player->GetGUID(), 0.f);
443 }
444 events.ScheduleEvent(EVENT_WATCH_PLAYER, urand(12000, 24000));
445 break;
446 case EVENT_CHARGE_PLAYER:
447 if (Unit* target = SelectTarget(SelectTargetMethod::MinDistance, 0, [this](Unit const* target)
448 {
449 if (!me || !target)
450 return false;
451 if (target->GetTypeId() != TYPEID_PLAYER || !me->IsWithinLOSInMap(target))
452 return false;
453 return true;
454 }))
455 {
456 DoCast(target, SPELL_CHARGE);
457 events.DelayEvents(1500);
458 if (Unit* mainTarget = SelectTarget(SelectTargetMethod::MaxThreat, 0, 100.0f))
459 {
460 me->GetThreatMgr().ModifyThreatByPercent(mainTarget, -100);
461 }
462 }
463 events.ScheduleEvent(EVENT_CHARGE_PLAYER, urand(30000, 40000));
464 break;
465 case EVENT_EXECUTE:
466 DoCastVictim(SPELL_EXECUTE, true);
467 events.ScheduleEvent(EVENT_EXECUTE, urand(7000, 14000));
468 break;
469 case EVENT_CLEAVE:
470 {
471 std::list<Unit*> meleeRangeTargets;
472 auto i = me->GetThreatMgr().GetThreatList().begin();
473 for (; i != me->GetThreatMgr().GetThreatList().end(); ++i)
474 {
475 Unit* target = (*i)->getTarget();
476 if (me->IsWithinMeleeRange(target))
477 {
478 meleeRangeTargets.push_back(target);
479 }
480 }
481 if (meleeRangeTargets.size() >= 5)
482 {
483 DoCastVictim(SPELL_MANDOKIR_CLEAVE);
484 events.ScheduleEvent(EVENT_CLEAVE, urand(10000, 20000));
485 }
486 else
487 {
488 events.ScheduleEvent(EVENT_CLEAVE, 1000);
489 }
490 break;
491 }
492 default:
493 break;
494 }
495 }
496
497 DoMeleeAttackIfReady(false);
498 }
499
500 private:
501 uint8 killCount;
502 ObjectGuid reviveGUID;
503 bool _useExecute;
504 std::pair<ObjectGuid, float> _chargeTarget;
505 };
