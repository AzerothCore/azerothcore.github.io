57void BattlegroundWS::PostUpdateImpl(uint32 diff)
58{
59 if (GetStatus() == STATUS_IN_PROGRESS)
60 {
61 _bgEvents.Update(diff);
62 switch (_bgEvents.ExecuteEvent())
63 {
64 case BG_WS_EVENT_UPDATE_GAME_TIME:
65 UpdateWorldState(BG_WS_STATE_TIMER, GetMatchTime());
66 _bgEvents.ScheduleEvent(BG_WS_EVENT_UPDATE_GAME_TIME, ((BG_WS_TOTAL_GAME_TIME - GetStartTime()) % (MINUTE * IN_MILLISECONDS)) + 1);
67 break;
68 case BG_WS_EVENT_NO_TIME_LEFT:
69 if (GetTeamScore(TEAM_ALLIANCE) == GetTeamScore(TEAM_HORDE))
70 EndBattleground(_lastFlagCaptureTeam);
71 else
72 EndBattleground(GetTeamScore(TEAM_HORDE) > GetTeamScore(TEAM_ALLIANCE) ? TEAM_HORDE : TEAM_ALLIANCE);
73 break;
74 case BG_WS_EVENT_RESPAWN_BOTH_FLAGS:
75 SpawnBGObject(BG_WS_OBJECT_H_FLAG, RESPAWN_IMMEDIATELY);
76 SpawnBGObject(BG_WS_OBJECT_A_FLAG, RESPAWN_IMMEDIATELY);
77 SendBroadcastText(BG_WS_TEXT_FLAGS_PLACED, CHAT_MSG_BG_SYSTEM_NEUTRAL);
78 PlaySoundToAll(BG_WS_SOUND_FLAGS_RESPAWNED);
79 break;
80 case BG_WS_EVENT_ALLIANCE_DROP_FLAG:
81 RespawnFlagAfterDrop(TEAM_ALLIANCE);
82 break;
83 case BG_WS_EVENT_HORDE_DROP_FLAG:
84 RespawnFlagAfterDrop(TEAM_HORDE);
85 break;
86 case BG_WS_EVENT_BOTH_FLAGS_KEPT10:
87 if (Player* player = ObjectAccessor::GetPlayer(FindBgMap(), GetFlagPickerGUID(TEAM_ALLIANCE)))
88 player->CastSpell(player, BG_WS_SPELL_FOCUSED_ASSAULT, true);
89 if (Player* player = ObjectAccessor::GetPlayer(FindBgMap(), GetFlagPickerGUID(TEAM_HORDE)))
90 player->CastSpell(player, BG_WS_SPELL_FOCUSED_ASSAULT, true);
91 break;
92 case BG_WS_EVENT_BOTH_FLAGS_KEPT15:
93 if (Player* player = ObjectAccessor::GetPlayer(FindBgMap(), GetFlagPickerGUID(TEAM_ALLIANCE)))
94 {
95 player->RemoveAurasDueToSpell(BG_WS_SPELL_FOCUSED_ASSAULT);
96 player->CastSpell(player, BG_WS_SPELL_BRUTAL_ASSAULT, true);
97 }
98 if (Player* player = ObjectAccessor::GetPlayer(FindBgMap(), GetFlagPickerGUID(TEAM_HORDE)))
99 {
100 player->RemoveAurasDueToSpell(BG_WS_SPELL_FOCUSED_ASSAULT);
101 player->CastSpell(player, BG_WS_SPELL_BRUTAL_ASSAULT, true);
102 }
103 break;
104 case BG_WS_EVENT_DESPAWN_DOORS:
105 SpawnBGObject(BG_WS_OBJECT_DOOR_H_1, RESPAWN_ONE_DAY);
106 SpawnBGObject(BG_WS_OBJECT_DOOR_H_2, RESPAWN_ONE_DAY);
107 break;
108 }
109 }
110}

151void BattlegroundWS::RespawnFlagAfterDrop(TeamId teamId)
152{

173{
174 if (GetStatus() != STATUS_IN_PROGRESS || GetFlagState(teamId) != BG_WS_FLAG_STATE_ON_PLAYER)
175 {
176 return;

179 uint32 triggerId = teamId == TEAM_ALLIANCE ? BG_WS_TRIGGER_HORDE_FLAG_SPAWN : BG_WS_TRIGGER_ALLIANCE_FLAG_SPAWN;

181 if (Player* player = ObjectAccessor::GetPlayer(FindBgMap(), GetFlagPickerGUID(teamId)))
182 {
183 if (areaTrigger && player->IsInAreaTriggerRadius(areaTrigger))
184 {
185 HandleAreaTrigger(player, triggerId);
186 }

190void BattlegroundWS::EventPlayerCapturedFlag(Player* player)
191{
192 if (GetStatus() != STATUS_IN_PROGRESS)
193 return;
194
195 player->RemoveAurasWithInterruptFlags(AURA_INTERRUPT_FLAG_ENTER_PVP_COMBAT);
196 RemoveAssaultAuras();
197

199 SetFlagPicker(ObjectGuid::Empty, GetOtherTeamId(player->GetTeamId()));
200 UpdateFlagState(GetOtherTeamId(player->GetTeamId()), BG_WS_FLAG_STATE_ON_BASE);
201
202 if (player->GetTeamId() == TEAM_ALLIANCE)
203 {
204 player->RemoveAurasDueToSpell(BG_WS_SPELL_WARSONG_FLAG);
205 PlaySoundToAll(BG_WS_SOUND_FLAG_CAPTURED_ALLIANCE);
206 SendBroadcastText(BG_WS_TEXT_CAPTURED_HORDE_FLAG, CHAT_MSG_BG_SYSTEM_ALLIANCE, player);
207 RewardReputationToTeam(890, _reputationCapture, TEAM_ALLIANCE);
208 }
209 else
210 {
211 player->RemoveAurasDueToSpell(BG_WS_SPELL_SILVERWING_FLAG);
212 PlaySoundToAll(BG_WS_SOUND_FLAG_CAPTURED_HORDE);
213 SendBroadcastText(BG_WS_TEXT_CAPTURED_ALLIANCE_FLAG, CHAT_MSG_BG_SYSTEM_HORDE, player);
214 RewardReputationToTeam(889, _reputationCapture, TEAM_HORDE);
215 }
216
217 SpawnBGObject(BG_WS_OBJECT_H_FLAG, BG_WS_FLAG_RESPAWN_TIME);
218 SpawnBGObject(BG_WS_OBJECT_A_FLAG, BG_WS_FLAG_RESPAWN_TIME);
219
220 UpdateWorldState(player->GetTeamId() == TEAM_ALLIANCE ? BG_WS_FLAG_CAPTURES_ALLIANCE : BG_WS_FLAG_CAPTURES_HORDE, GetTeamScore(player->GetTeamId()));
221 UpdatePlayerScore(player, SCORE_FLAG_CAPTURES, 1); // +1 flag captures
222 _lastFlagCaptureTeam = player->GetTeamId();
223
224 RewardHonorToTeam(GetBonusHonorFromKill(2), player->GetTeamId());
225
226 if (GetTeamScore(TEAM_ALLIANCE) == BG_WS_MAX_TEAM_SCORE || GetTeamScore(TEAM_HORDE) == BG_WS_MAX_TEAM_SCORE)
227 {
228 UpdateWorldState(BG_WS_STATE_TIMER_ACTIVE, 0);
229 EndBattleground(GetTeamScore(TEAM_HORDE) == BG_WS_MAX_TEAM_SCORE ? TEAM_HORDE : TEAM_ALLIANCE);
230 }
231 else
232 _bgEvents.ScheduleEvent(BG_WS_EVENT_RESPAWN_BOTH_FLAGS, BG_WS_FLAG_RESPAWN_TIME);
233
234 _bgEvents.CancelEvent(BG_WS_EVENT_BOTH_FLAGS_KEPT10);
235 _bgEvents.CancelEvent(BG_WS_EVENT_BOTH_FLAGS_KEPT15);
236}

398void BattlegroundWS::HandleAreaTrigger(Player* player, uint32 trigger)
399{
400 if (GetStatus() != STATUS_IN_PROGRESS || !player->IsAlive())
401 return;
402
403 switch (trigger)
404 {
405 case BG_WS_TRIGGER_ALLIANCE_FLAG_SPAWN: // Alliance Flag spawn
406 if (GetFlagState(TEAM_ALLIANCE) == BG_WS_FLAG_STATE_ON_BASE && GetFlagPickerGUID(TEAM_HORDE) == player->GetGUID())
407 EventPlayerCapturedFlag(player);
408 break;
409 case BG_WS_TRIGGER_HORDE_FLAG_SPAWN: // Horde Flag spawn
410 if (GetFlagState(TEAM_HORDE) == BG_WS_FLAG_STATE_ON_BASE && GetFlagPickerGUID(TEAM_ALLIANCE) == player->GetGUID())
411 EventPlayerCapturedFlag(player);
412 break;
413 case 3649: // Not used
414 case 3688: // Not used
415 case 4628: // Not used
416 case 4629: // Not used
417 case BG_WS_TRIGGER_ALLIANCE_ELIXIR_SPEED_SPAWN: // Alliance elixir of speed spawn
418 case BG_WS_TRIGGER_HORDE_ELIXIR_SPEED_SPAWN: // Horde elixir of speed spawn
419 case BG_WS_TRIGGER_ALLIANCE_ELIXIR_REGEN_SPAWN: // Alliance elixir of regeneration spawn
420 case BG_WS_TRIGGER_HORDE_ELIXIR_REGEN_SPAWN: // Horde elixir of regeneration spawn
421 case BG_WS_TRIGGER_ALLIANCE_ELIXIR_BERSERK_SPAWN: // Alliance elixir of berserk spawn
422 case BG_WS_TRIGGER_HORDE_ELIXIR_BERSERK_SPAWN: // Horde elixir of berserk spawn
423 break;
424 }
