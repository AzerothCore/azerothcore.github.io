140
141 void UpdateAI(uint32 diff) override
142 {
143 if (!UpdateVictim())
144 return;
145
146 events.Update(diff);
147 if (me->HasUnitState(UNIT_STATE_CASTING))
148 return;
149
150 switch (events.ExecuteEvent())
151 {
152 case EVENT_HEALTH_CHECK:
153 if (me->HealthBelowPct(26))
154 {
155 Talk(SAY_ENRAGE);
156 Talk(EMOTE_FRENZY);
157 me->CastSpell(me, SPELL_ENRAGE, true);
158 break;
159 }
160 events.ScheduleEvent(EVENT_HEALTH_CHECK, 1000);
161 break;
162 case EVENT_ACHIEVEMENT_CHECK:
163 {
164 Map::PlayerList const& pList = me->GetMap()->GetPlayers();
165 for(Map::PlayerList::const_iterator itr = pList.begin(); itr != pList.end(); ++itr)
166 if (Aura* aur = itr->GetSource()->GetAura(SPELL_INTENSE_COLD_TRIGGER))
167 if (aur->GetStackAmount() > 2)
168 aGuids.insert(itr->GetSource()->GetGUID().GetCounter());
169 events.ScheduleEvent(EVENT_ACHIEVEMENT_CHECK, 500);
170 break;
171 }
172 case EVENT_CRYSTALFIRE_BREATH:
173 me->CastSpell(me->GetVictim(), SPELL_CRYSTALFIRE_BREATH, false);
174 events.ScheduleEvent(EVENT_CRYSTALFIRE_BREATH, 14000);
175 break;
176 case EVENT_TAIL_SWEEP:
177 me->CastSpell(me, SPELL_TAIL_SWEEP, false);
178 events.ScheduleEvent(EVENT_TAIL_SWEEP, 5000);
179 break;
180 case EVENT_CRYSTAL_CHAINS:
181 Talk(SAY_CRYSTAL_NOVA);
182 if (IsHeroic())
183 me->CastSpell(me, SPELL_CRYSTALIZE, false);
184 else if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 40.0f, true))
185 me->CastSpell(target, SPELL_CRYSTAL_CHAINS, false);
186 events.ScheduleEvent(EVENT_CRYSTAL_CHAINS, DUNGEON_MODE(20000, 11000));
187 break;
188 }
189
190 DoMeleeAttackIfReady();
191 }
192 };
