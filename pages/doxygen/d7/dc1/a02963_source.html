87 void UpdateAI(uint32 diff) override
88 {
89 if (!UpdateVictim())
90 return;
91
92 events.Update(diff);
93
94 if (me->HasUnitState(UNIT_STATE_CASTING))
95 return;
96
97 switch (events.ExecuteEvent())
98 {
99 case EVENT_SPELL_DISEASE_EXPULSION:
100 me->CastSpell(me, DUNGEON_MODE(SPELL_DISEASE_EXPULSION_N, SPELL_DISEASE_EXPULSION_H), false);
101 events.RepeatEvent(6000);
102 break;
103 case EVENT_SPELL_FRENZY:
104 me->CastSpell(me, SPELL_FRENZY, false);
105 events.RepeatEvent(20000);
106 break;
107 case EVENT_SPELL_CONSTRICTING_CHAINS:
108 if (Unit* pTarget = SelectTarget(SelectTargetMethod::MinThreat, 0, 50.0f, true))
109 me->CastSpell(pTarget, DUNGEON_MODE(SPELL_CONSTRICTING_CHAINS_N, SPELL_CONSTRICTING_CHAINS_H), false);
110 events.RepeatEvent(14000);
111 break;
112 }
113
114 DoMeleeAttackIfReady();
115 }
