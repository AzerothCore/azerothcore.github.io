59void WorldSession::HandleTalentWipeConfirmOpcode(WorldPacket& recvData)
60{
61 LOG_DEBUG("network", "MSG_TALENT_WIPE_CONFIRM");
62 ObjectGuid guid;
63 recvData >> guid;
64
65 Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_TRAINER);
66 if (!unit)
67 {
68 LOG_DEBUG("network", "WORLD: HandleTalentWipeConfirmOpcode - Unit ({}) not found or you can't interact with him.", guid.ToString());
69 return;
70 }
71
72 if (!unit->isCanTrainingAndResetTalentsOf(_player))
73 return;
74
75 // remove fake death
76 if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
77 GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
78
79 if (!(_player->resetTalents()))
80 {
81 WorldPacket data(MSG_TALENT_WIPE_CONFIRM, 8 + 4); //you have not any talent
82 data << uint64(0);
83 data << uint32(0);
84 SendPacket(&data);
85 return;
86 }
87
88 _player->SendTalentsInfoData(false);
89 unit->CastSpell(_player, 14867, true); //spell: "Untalent Visual Effect"
90}
