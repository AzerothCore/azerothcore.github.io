52 void delink()

97 [[nodiscard]] bool IsEmpty() const { return (!iFirst.iNext->isInList()); }

99 LinkedListElement* getFirst() { return (IsEmpty() ? nullptr : iFirst.iNext); }

104
105 void insertFirst(LinkedListElement* pElem)
106 {
107 iFirst.insertAfter(pElem);
108 }
109
110 void insertLast(LinkedListElement* pElem)
111 {
112 iLast.insertBefore(pElem);
113 }
114
115 [[nodiscard]] uint32 getSize() const
116 {
117 if (!iSize)
118 {
119 uint32 result = 0;
120 LinkedListElement const* e = getFirst();
121 while (e)
122 {
123 ++result;
124 e = e->next();
125 }
126 return result;
127 }
128 else
129 {
130 return iSize;
131 }
132 }
133
134 void incSize() { ++iSize; }
