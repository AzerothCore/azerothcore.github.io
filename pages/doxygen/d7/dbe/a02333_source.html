64 static bool HandleMmapPathCommand(ChatHandler* handler, Optional<std::string> para)
65 {
66 if (!MMAP::MMapFactory::createOrGetMMapMgr()->GetNavMesh(handler->GetSession()->GetPlayer()->GetMapId()))
67 {
68 handler->PSendSysMessage("NavMesh not loaded for current map.");
69 return true;
70 }
71
72 handler->PSendSysMessage("mmap path:");
73
74 // units
75 Player* player = handler->GetSession()->GetPlayer();
76 Unit* target = handler->getSelectedUnit();
77 if (!player || !target)
78 {
79 handler->PSendSysMessage("Invalid target/source selection.");
80 return true;
81 }
82
83 bool useStraightPath = false;
84 if (StringStartsWith("true", *para))
85 useStraightPath = true;
86
87 bool useRaycast = false;
88 if (StringStartsWith("line", *para) || StringStartsWith("ray", *para) || StringStartsWith("raycast", *para))
89 {
90 useRaycast = true;
91 }
92
93 // unit locations
94 float x, y, z;
95 player->GetPosition(x, y, z);
96
97 // path
98 PathGenerator path(target);
99 path.SetUseStraightPath(useStraightPath);
100 path.SetUseRaycast(useRaycast);
101 bool result = path.CalculatePath(x, y, z, false);
102
103 Movement::PointsArray const& pointPath = path.GetPath();
104 handler->PSendSysMessage("%s's path to %s:", target->GetName().c_str(), player->GetName().c_str());
105 handler->PSendSysMessage("Building: %s", useStraightPath ? "StraightPath" : useRaycast ? "Raycast" : "SmoothPath");
106 handler->PSendSysMessage(Acore::StringFormatFmt("Result: {} - Length: {} - Type: {}", (result ? "true" : "false"), pointPath.size(), path.GetPathType()).c_str());
107
108 G3D::Vector3 const& start = path.GetStartPosition();
109 G3D::Vector3 const& end = path.GetEndPosition();
110 G3D::Vector3 const& actualEnd = path.GetActualEndPosition();
111
112 handler->PSendSysMessage("StartPosition (%.3f, %.3f, %.3f)", start.x, start.y, start.z);
113 handler->PSendSysMessage("EndPosition (%.3f, %.3f, %.3f)", end.x, end.y, end.z);
114 handler->PSendSysMessage("ActualEndPosition (%.3f, %.3f, %.3f)", actualEnd.x, actualEnd.y, actualEnd.z);
115
116 if (!player->IsGameMaster())
117 handler->PSendSysMessage("Enable GM mode to see the path points.");
118
119 for (auto i : pointPath)
120 player->SummonCreature(VISUAL_WAYPOINT, i.x, i.y, i.z, 0, TEMPSUMMON_TIMED_DESPAWN, 9000);
121
122 return true;
123 }

125 static bool HandleMmapLocCommand(ChatHandler* handler)
126 {
127 handler->PSendSysMessage("mmap tileloc:");
128
129 // grid tile location
130 Player* player = handler->GetSession()->GetPlayer();
131
132 int32 gx = 32 - player->GetPositionX() / SIZE_OF_GRIDS;
133 int32 gy = 32 - player->GetPositionY() / SIZE_OF_GRIDS;
134
135 handler->PSendSysMessage("%03u%02i%02i.mmtile", player->GetMapId(), gx, gy);
136 handler->PSendSysMessage("gridloc [%i, %i]", gy, gx);
137
138 // calculate navmesh tile location
139 dtNavMesh const* navmesh = MMAP::MMapFactory::createOrGetMMapMgr()->GetNavMesh(handler->GetSession()->GetPlayer()->GetMapId());
140 dtNavMeshQuery const* navmeshquery = MMAP::MMapFactory::createOrGetMMapMgr()->GetNavMeshQuery(handler->GetSession()->GetPlayer()->GetMapId(), player->GetInstanceId());
141 if (!navmesh || !navmeshquery)
142 {
143 handler->PSendSysMessage("NavMesh not loaded for current map.");
144 return true;
145 }
146
147 float const* min = navmesh->getParams()->orig;
148 float x, y, z;
149 player->GetPosition(x, y, z);
150 float location[VERTEX_SIZE] = {y, z, x};
151 float extents[VERTEX_SIZE] = {3.0f, 5.0f, 3.0f};
152
153 int32 tilex = int32((y - min[0]) / SIZE_OF_GRIDS);
154 int32 tiley = int32((x - min[2]) / SIZE_OF_GRIDS);
155
156 handler->PSendSysMessage("Calc [%02i, %02i]", tilex, tiley);
157
158 // navmesh poly -> navmesh tile location
159 dtQueryFilterExt filter = dtQueryFilterExt();
160 dtPolyRef polyRef = INVALID_POLYREF;
161 if (dtStatusFailed(navmeshquery->findNearestPoly(location, extents, &filter, &polyRef, nullptr)))
162 {
163 handler->PSendSysMessage("Dt [??,??] (invalid poly, probably no tile loaded)");
164 return true;
165 }
166
167 if (polyRef == INVALID_POLYREF)
168 handler->PSendSysMessage("Dt [??, ??] (invalid poly, probably no tile loaded)");
169 else
170 {
171 dtMeshTile const* tile;
172 dtPoly const* poly;
173 if (dtStatusSucceed(navmesh->getTileAndPolyByRef(polyRef, &tile, &poly)))
174 {
175 if (tile)
176 {
177 handler->PSendSysMessage("Dt [%02i,%02i]", tile->header->x, tile->header->y);
178 return false;
179 }
180 }
181
182 handler->PSendSysMessage("Dt [??,??] (no tile loaded)");
183 }
184
185 return true;
186 }

188 static bool HandleMmapLoadedTilesCommand(ChatHandler* handler)
189 {
190 uint32 mapid = handler->GetSession()->GetPlayer()->GetMapId();
191 dtNavMesh const* navmesh = MMAP::MMapFactory::createOrGetMMapMgr()->GetNavMesh(mapid);
192 dtNavMeshQuery const* navmeshquery = MMAP::MMapFactory::createOrGetMMapMgr()->GetNavMeshQuery(mapid, handler->GetSession()->GetPlayer()->GetInstanceId());
193 if (!navmesh || !navmeshquery)
194 {
195 handler->PSendSysMessage("NavMesh not loaded for current map.");
196 return true;
197 }
198
199 handler->PSendSysMessage("mmap loadedtiles:");
200
201 for (int32 i = 0; i < navmesh->getMaxTiles(); ++i)
202 {
203 dtMeshTile const* tile = navmesh->getTile(i);
204 if (!tile || !tile->header)
205 continue;
206
207 handler->PSendSysMessage("[%02i, %02i]", tile->header->x, tile->header->y);
208 }
209
210 return true;

213 static bool HandleMmapStatsCommand(ChatHandler* handler)
214 {
215 handler->PSendSysMessage("mmap stats:");
216 //handler->PSendSysMessage(" global mmap pathfinding is %sabled", DisableMgr::IsPathfindingEnabled(mapId) ? "en" : "dis");
217
218 MMAP::MMapMgr* manager = MMAP::MMapFactory::createOrGetMMapMgr();
219 handler->PSendSysMessage(" %u maps loaded with %u tiles overall", manager->getLoadedMapsCount(), manager->getLoadedTilesCount());
220
221 dtNavMesh const* navmesh = manager->GetNavMesh(handler->GetSession()->GetPlayer()->GetMapId());
222 if (!navmesh)
223 {
224 handler->PSendSysMessage("NavMesh not loaded for current map.");
225 return true;
226 }
227
228 uint32 tileCount = 0;
229 uint32 nodeCount = 0;
230 uint32 polyCount = 0;
231 uint32 vertCount = 0;
232 uint32 triCount = 0;
233 uint32 triVertCount = 0;
234 uint32 dataSize = 0;
235 for (int32 i = 0; i < navmesh->getMaxTiles(); ++i)
236 {
237 dtMeshTile const* tile = navmesh->getTile(i);
238 if (!tile || !tile->header)
239 continue;
240
241 tileCount++;
242 nodeCount += tile->header->bvNodeCount;
243 polyCount += tile->header->polyCount;
244 vertCount += tile->header->vertCount;
245 triCount += tile->header->detailTriCount;
246 triVertCount += tile->header->detailVertCount;
247 dataSize += tile->dataSize;
248 }
249
250 handler->PSendSysMessage("Navmesh stats:");
251 handler->PSendSysMessage(" %u tiles loaded", tileCount);
252 handler->PSendSysMessage(" %u BVTree nodes", nodeCount);
253 handler->PSendSysMessage(" %u polygons (%u vertices)", polyCount, vertCount);
254 handler->PSendSysMessage(" %u triangles (%u vertices)", triCount, triVertCount);
255 handler->PSendSysMessage(" %.2f MB of data (not including pointers)", ((float)dataSize / sizeof(unsigned char)) / 1048576);
256
257 return true;
258 }

260 static bool HandleMmapTestArea(ChatHandler* handler)
261 {
262 float radius = 40.0f;
263 WorldObject* object = handler->GetSession()->GetPlayer();
264
265 // Get Creatures
266 std::list<Creature*> creatureList;
267 Acore::AnyUnitInObjectRangeCheck go_check(object, radius);
268 Acore::CreatureListSearcher<Acore::AnyUnitInObjectRangeCheck> go_search(object, creatureList, go_check);
269 Cell::VisitGridObjects(object, go_search, radius);
270
271 if (!creatureList.empty())
272 {
273 handler->PSendSysMessage(Acore::StringFormatFmt("Found {} Creatures.", creatureList.size()).c_str());
274
275 uint32 paths = 0;
276 uint32 uStartTime = getMSTime();
277
278 float gx, gy, gz;
279 object->GetPosition(gx, gy, gz);
280 for (std::list<Creature*>::iterator itr = creatureList.begin(); itr != creatureList.end(); ++itr)
281 {
282 PathGenerator path(*itr);
283 path.CalculatePath(gx, gy, gz);
284 ++paths;
285 }
286
287 uint32 uPathLoadTime = getMSTimeDiff(uStartTime, getMSTime());
288 handler->PSendSysMessage("Generated %i paths in %i ms", paths, uPathLoadTime);
289 }
290 else
291 handler->PSendSysMessage("No creatures in %f yard range.", radius);
292
293 return true;
294 }
