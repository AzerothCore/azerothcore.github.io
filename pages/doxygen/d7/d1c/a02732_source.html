247 void Load(char const* str) override
248 {
249 if (!str)
250 {
251 OUT_LOAD_INST_DATA_FAIL;
252 return;
253 }
254
255 OUT_LOAD_INST_DATA(str);
256
257 char dataHead1, dataHead2;
258
259 std::istringstream loadStream(str);
260 loadStream >> dataHead1 >> dataHead2;
261
262 if (dataHead1 == 'S' && dataHead2 == 'P')
263 {
264 for (uint32 i = 0; i < MAX_ENCOUNTERS; ++i)
265 {
266 uint32 tmpState;
267 loadStream >> tmpState;
268 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
269 tmpState = NOT_STARTED;
270 SetBossState(i, EncounterState(tmpState));
271 }
272 }
273 else
274 OUT_LOAD_INST_DATA_FAIL;
275
276 OUT_LOAD_INST_DATA_COMPLETE;
277 }
