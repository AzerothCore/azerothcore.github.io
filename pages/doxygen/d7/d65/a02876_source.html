184 void Load(const char* str) override
185 {
186 if (!str)
187 {
188 OUT_LOAD_INST_DATA_FAIL;
189 return;
190 }
191
192 OUT_LOAD_INST_DATA(str);
193
194 char dataHead1, dataHead2;
195
196 std::istringstream loadStream(str);
197 loadStream >> dataHead1 >> dataHead2;
198
199 if (dataHead1 == 'Z' && dataHead2 == 'G')
200 {
201 for (uint32 i = 0; i < EncounterCount; ++i)
202 {
203 uint32 tmpState;
204 loadStream >> tmpState;
205 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
206 tmpState = NOT_STARTED;
207 SetBossState(i, EncounterState(tmpState));
208 }
209 }
210 else
211 OUT_LOAD_INST_DATA_FAIL;
212
213 OUT_LOAD_INST_DATA_COMPLETE;
214 }

256 bool GossipHello(Player* /*player*/, bool reportUse) override
257 {
258 if (reportUse)
259 {
260 return true;
261 }
262
263 if (InstanceScript* instanceScript = me->GetInstanceScript())
264 {
265 if (instanceScript->GetGuidData(DATA_EDGE_OF_MADNESS))
266 {
267 return false;
268 }
269 }
270
271 uint32 bossEntry = 0;
272 for (uint8 i = 0; i < 4; ++i)
273 {
274 if (sGameEventMgr->IsActiveEvent(BrazierOfMadnessContainer[i].first))
275 {
276 bossEntry = BrazierOfMadnessContainer[i].second;
277 break;
278 }
279 }
280
281 if (bossEntry)
282 {
283 me->SummonCreature(bossEntry, edgeOfMagnessSummonPos, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 2 * HOUR * IN_MILLISECONDS);
284 }
285
286 return false;
