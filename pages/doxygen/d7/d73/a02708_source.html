178 void Load(const char* in) override
179 {
180 if (!in)
181 return;
182
183 char dataHead1, dataHead2;
184 std::istringstream loadStream(in);
185 loadStream >> dataHead1 >> dataHead2;
186 if (dataHead1 == 'T' && dataHead2 == 'A')
187 {
188 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
189 {
190 loadStream >> _encounters[i];
191 if (_encounters[i] == IN_PROGRESS)
192 _encounters[i] = NOT_STARTED;
193 }
194
195 loadStream >> _statuePhase;
196 loadStream >> _defendersKilled;
197 }
