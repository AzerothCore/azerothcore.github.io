157 void UpdateAI(uint32 diff) override
158 {
159 if (!IsInRoom())
160 return;
161
162 if (!UpdateVictim())
163 return;
164
165 events.Update(diff);
166 if (me->HasUnitState(UNIT_STATE_CASTING))
167 return;
168
169 switch (events.ExecuteEvent())
170 {
171 case EVENT_WEB_SPRAY:
172 Talk(EMOTE_WEB_SPRAY);
173 me->CastSpell(me, RAID_MODE(SPELL_WEB_SPRAY_10, SPELL_WEB_SPRAY_25), true);
174 events.RepeatEvent(40000);
175 break;
176 case EVENT_POISON_SHOCK:
177 me->CastSpell(me->GetVictim(), RAID_MODE(SPELL_POISON_SHOCK_10, SPELL_POISON_SHOCK_25), false);
178 events.RepeatEvent(10000);
179 break;
180 case EVENT_NECROTIC_POISON:
181 me->CastSpell(me->GetVictim(), RAID_MODE(SPELL_NECROTIC_POISON_10, SPELL_NECROTIC_POISON_25), false);
182 events.RepeatEvent(30000);
183 break;
184 case EVENT_SUMMON_SPIDERLINGS:
185 Talk(EMOTE_SPIDERS);
186 for (uint8 i = 0; i < 8; ++i)
187 {
188 me->SummonCreature(NPC_MAEXXNA_SPIDERLING, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
189 }
190 events.RepeatEvent(40000);
191 break;
192 case EVENT_HEALTH_CHECK:
193 if (me->GetHealthPct() < 30)
194 {
195 me->CastSpell(me, RAID_MODE(SPELL_FRENZY_10, SPELL_FRENZY_25), true);
196 break;
197 }
198 events.RepeatEvent(1000);
199 break;
200 case EVENT_WEB_WRAP:
201 Talk(EMOTE_WEB_WRAP);
202 for (uint8 i = 0; i < RAID_MODE(1, 2); ++i)
203 {
204 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 0, true, -SPELL_WEB_WRAP))
205 {
206 target->RemoveAura(RAID_MODE(SPELL_WEB_SPRAY_10, SPELL_WEB_SPRAY_25));
207 uint8 pos = urand(0, 2);
208 if (Creature* wrap = me->SummonCreature(NPC_WEB_WRAP, PosWrap[pos].GetPositionX(), PosWrap[pos].GetPositionY(), PosWrap[pos].GetPositionZ(), 0.0f, TEMPSUMMON_TIMED_DESPAWN, 60000))
209 {
210 wrap->AI()->SetGUID(target->GetGUID());
211 target->GetMotionMaster()->MoveJump(PosWrap[pos].GetPositionX(), PosWrap[pos].GetPositionY(), PosWrap[pos].GetPositionZ(), 20, 20);
212 }
213 }
214 }
215 events.RepeatEvent(40000);
216 break;
217 }
218 DoMeleeAttackIfReady();
219 }
220 };
