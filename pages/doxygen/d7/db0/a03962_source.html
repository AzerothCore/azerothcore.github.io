121 void DoAction(int32 param) override
122 {
123 if (param == ACTION_CHANNELER_ENGAGED)
124 {
125 if (!addYell)
126 {

128 Talk(SAY_ADD_AGGRO);
129
130 for (uint8 i = 0; i < 5; ++i)
131 {
132 Creature* channeler = ObjectAccessor::GetCreature(*me, channelers[i]);
133 if (channeler && !channeler->IsInCombat())
134 channeler->SetInCombatWithZone();

136 }
137 }
138 else if (param == ACTION_CHANNELER_DIED)
139 {
140 for (uint8 i = 0; i < 5; ++i)
141 {
142 Creature* channeler = ObjectAccessor::GetCreature(*me, channelers[i]);
143 if (channeler && channeler->IsAlive())
144 return;
145 }
146 me->SetReactState(REACT_AGGRESSIVE);
147 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
148 me->SetImmuneToAll(false);
149 if (Unit* target = me->SelectNearestPlayer(100.0f))
150 AttackStart(target);
151 }

154 void CheckChannelers()
155 {
156 if (addYell)

158 if (!SelectTargetFromPlayerList(100.0f))
159 EnterEvadeMode();
160 return;
161 }
162
163 SummonChannelers();
164 for (uint8 i = 0; i < 5; ++i)
165 {
166 Creature* channeler = ObjectAccessor::GetCreature(*me, channelers[i]);
167 if (channeler && !channeler->HasUnitState(UNIT_STATE_CASTING) && !channeler->IsInCombat())
168 {
169 Creature* target = ObjectAccessor::GetCreature(*me, channelers[(i + 2) % 5]);
170 if (target)
171 channeler->CastSpell(target, SPELL_CHANNELING, false);
172 }
173 }
174 }
175
176 void SummonChannelers()
177 {
178 for (uint8 i = 0; i < 5; ++i)
179 {
180 Creature* channeler = ObjectAccessor::GetCreature(*me, channelers[i]);
181 if (channeler && channeler->isDead())
182 {
183 channeler->DespawnOrUnsummon(1);
184 channeler = nullptr;
185 }
186 if (!channeler)
187 channeler = me->SummonCreature(NPC_CHANNELER, ShadowmoonChannelers[i][0], ShadowmoonChannelers[i][1], ShadowmoonChannelers[i][2], ShadowmoonChannelers[i][3], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 300000);
188
189 channelers[i] = channeler ? channeler->GetGUID() : ObjectGuid::Empty;
190 }
191 }

228 void UpdateAI(uint32 diff) override
229 {
230 if (!UpdateVictim())
231 {
232 checkTimer += diff;
233 if (checkTimer >= 5000)
234 {
235 checkTimer = 0;
236 CheckChannelers();
237 if (!me->HasUnitState(UNIT_STATE_CASTING))
238 me->CastSpell(me, SPELL_EVOCATION, false);
239 }
240 return;
241 }
242
243 events.Update(diff);
244 if (me->HasUnitState(UNIT_STATE_CASTING))
245 return;
246
247 switch (events.ExecuteEvent())
248 {
249 case EVENT_SPELL_VOLLEY:
250 me->CastSpell(me, SPELL_SHADOW_BOLT_VOLLEY, false);
251 events.RepeatEvent(urand(8000, 13000));
252 break;
253 case EVENT_SPELL_CORRUPTION:
254 me->CastSpell(me, SPELL_CORRUPTION, false);
255 events.RepeatEvent(urand(30000, 50000));
256 break;
257 case EVENT_SPELL_BURNING_NOVA:
258 Talk(SAY_NOVA);
259
260 ApplyImmunities(false);
261 me->AddAura(SPELL_BURNING_NOVA, me);
262 ApplyImmunities(true);
263
264 if (IsHeroic())
265 DoTeleportAll(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
266
267 events.DelayEvents(6000, 0);
268 events.RepeatEvent(urand(25000, 32000));
269 events.ScheduleEvent(EVENT_SPELL_FIRE_NOVA, 5000);
270 break;
271 case EVENT_SPELL_FIRE_NOVA:
272 me->CastSpell(me, SPELL_FIRE_NOVA, true);
273 break;
274 }
275
276 DoMeleeAttackIfReady();
277 }
278 };
