56// Selects invalid loot items to be removed from group possible entries (before rolling)
57struct LootGroupInvalidSelector : public Acore::unary_function<LootStoreItem*, bool>
58{
59 explicit LootGroupInvalidSelector(Loot const& loot, uint16 lootMode) : _loot(loot), _lootMode(lootMode) { }
60
61 bool operator()(LootStoreItem* item) const
62 {
63 if (!(item->lootmode & _lootMode))
64 return true;
65
66 if (!item->reference)
67 {
68 ItemTemplate const* _proto = sObjectMgr->GetItemTemplate(item->itemid);
69 if (!_proto)
70 return true;
71
72 uint8 foundDuplicates = 0;
73 for (std::vector<LootItem>::const_iterator itr = _loot.items.begin(); itr != _loot.items.end(); ++itr)
74 if (itr->itemid == item->itemid)
75 {
76 ++foundDuplicates;
77 if (_proto->InventoryType == 0 && foundDuplicates == 3 && _proto->ItemId != 47242 /*Trophy of the Crusade*/) // Non-equippable items are limited to 3 drops
78 return true;
79 else if (_proto->InventoryType != 0 && foundDuplicates == 1) // Equippable item are limited to 1 drop
80 return true;
81 }
82 }
83
84 return false;
85 }
86
87private:
88 Loot const& _loot;
89 uint16 _lootMode;
90};

139// Loads a *_loot_template DB table into loot store
140// All checks of the loaded template are called from here, no error reports at loot generation required
141uint32 LootStore::LoadLootTable()
142{
143 LootTemplateMap::const_iterator tab;
144
145 // Clearing store (for reloading case)
146 Clear();
147
148 // 0 1 2 3 4 5 6
149 QueryResult result = WorldDatabase.Query("SELECT Entry, Item, Reference, Chance, QuestRequired, LootMode, GroupId, MinCount, MaxCount FROM {}", GetName());
150
151 if (!result)
152 return 0;
153
154 uint32 count = 0;
155
156 do
157 {
158 Field* fields = result->Fetch();
159
160 uint32 entry = fields[0].Get<uint32>();
161 uint32 item = fields[1].Get<uint32>();
162 int32 reference = fields[2].Get<int32>();
163 float chance = fields[3].Get<float>();
164 bool needsquest = fields[4].Get<bool>();
165 uint16 lootmode = fields[5].Get<uint16>();
166 uint8 groupid = fields[6].Get<uint8>();
167 int32 mincount = fields[7].Get<uint8>();
168 int32 maxcount = fields[8].Get<uint8>();
169
170 if (maxcount > std::numeric_limits<uint8>::max())
171 {
172 LOG_ERROR("sql.sql", "Table '{}' Entry {} Item {}: MaxCount value ({}) to large. must be less {} - skipped", GetName(), entry, item, maxcount, std::numeric_limits<uint8>::max());
173 continue; // error already printed to log/console.
174 }
175
176 if (lootmode == 0)
177 {
178 LOG_ERROR("sql.sql", "Table '{}' Entry {} Item {}: LootMode is equal to 0, item will never drop - setting mode 1", GetName(), entry, item);
179 lootmode = 1;
180 }
181
182 LootStoreItem* storeitem = new LootStoreItem(item, reference, chance, needsquest, lootmode, groupid, mincount, maxcount);
183
184 if (!storeitem->IsValid(*this, entry)) // Validity checks
185 {
186 delete storeitem;
187 continue;
188 }
189
190 // Looking for the template of the entry
191 // often entries are put together
192 if (m_LootTemplates.empty() || tab->first != entry)
193 {
194 // Searching the template (in case template Id changed)
195 tab = m_LootTemplates.find(entry);
196 if (tab == m_LootTemplates.end())
197 {
198 std::pair< LootTemplateMap::iterator, bool > pr = m_LootTemplates.insert(LootTemplateMap::value_type(entry, new LootTemplate()));
199 tab = pr.first;
200 }
201 }
202 // else is empty - template Id and iter are the same
203 // finally iter refers to already existed or just created <entry, LootTemplate>
204
205 // Adds current row to the template
206 tab->second->AddEntry(storeitem);
207 ++count;
208 } while (result->NextRow());
209
210 Verify(); // Checks validity of the loot store
211
212 return count;
213}

301// Checks if the entry (quest, non-quest, reference) takes it's chance (at loot generation)
302// RATE_DROP_ITEMS is no longer used for all types of entries
303bool LootStoreItem::Roll(bool rate, Player const* player, Loot& loot, LootStore const& store) const
304{
305 float _chance = chance;
306
307 if (!sScriptMgr->OnItemRoll(player, this, _chance, loot, store))
308 return false;
309
310 if (_chance >= 100.0f)
311 return true;
312
313 if (reference) // reference case
314 return roll_chance_f(_chance * (rate ? sWorld->getRate(RATE_DROP_ITEM_REFERENCED) : 1.0f));
315
316 ItemTemplate const* pProto = sObjectMgr->GetItemTemplate(itemid);
317
318 float qualityModifier = pProto && rate ? sWorld->getRate(qualityToRate[pProto->Quality]) : 1.0f;
319
320 return roll_chance_f(_chance * qualityModifier);
321}

323// Checks correctness of values
324bool LootStoreItem::IsValid(LootStore const& store, uint32 entry) const
325{
326 if (groupid >= 1 << 7) // it stored in 7 bit field
327 {
328 LOG_ERROR("sql.sql", "Table '{}' Entry {} Item {}: GroupId ({}) must be less {} - skipped", store.GetName(), entry, itemid, groupid, 1 << 7);
329 return false;
330 }
331
332 if (mincount == 0)
333 {
334 LOG_ERROR("sql.sql", "Table '{}' Entry {} Item {}: wrong MinCount ({}) - skipped", store.GetName(), entry, itemid, mincount);
335 return false;
336 }
337
338 if (!reference) // item (quest or non-quest) entry, maybe grouped
339 {
340 ItemTemplate const* proto = sObjectMgr->GetItemTemplate(itemid);
341 if (!proto)
342 {
343 LOG_ERROR("sql.sql", "Table '{}' Entry {} Item {}: item entry not listed in `item_template` - skipped", store.GetName(), entry, itemid);
344 return false;
345 }
346
347 if (chance == 0 && groupid == 0) // Zero chance is allowed for grouped entries only
348 {
349 LOG_ERROR("sql.sql", "Table '{}' Entry {} Item {}: equal-chanced grouped entry, but group not defined - skipped", store.GetName(), entry, itemid);
350 return false;
351 }
352
353 if (chance != 0 && chance < 0.000001f) // loot with low chance
354 {
355 LOG_ERROR("sql.sql", "Table '{}' Entry {} Item {}: low chance ({}) - skipped",
356 store.GetName(), entry, itemid, chance);
357 return false;
358 }
359
360 if (maxcount < mincount) // wrong max count
361 {
362 LOG_ERROR("sql.sql", "Table '{}' Entry {} Item {}: MaxCount ({}) less that MinCount ({}) - skipped", store.GetName(), entry, itemid, int32(maxcount), mincount);
363 return false;
364 }
365 }
366 else // if reference loot
367 {
368 if (needs_quest)
369 LOG_ERROR("sql.sql", "Table '{}' Entry {} Item {}: quest required will be ignored", store.GetName(), entry, itemid);
370 else if (chance == 0) // no chance for the reference
371 {
372 LOG_ERROR("sql.sql", "Table '{}' Entry {} Item {}: zero chance is specified for a reference, skipped", store.GetName(), entry, itemid);
373 return false;
374 }
375 }
376 return true; // Referenced template existence is checked at whole store level
377}

406// Basic checks for player/item compatibility - if false no chance to see the item in the loot
407bool LootItem::AllowedForPlayer(Player const* player, bool isGivenByMasterLooter /*= false*/, bool allowQuestLoot /*= true*/, ObjectGuid source) const
408{
409 ItemTemplate const* pProto = sObjectMgr->GetItemTemplate(itemid);
410 if (!pProto)
411 {
412 return false;
413 }
414
415 bool isMasterLooter = player->GetGroup() && player->GetGroup()->GetMasterLooterGuid() == player->GetGUID();
416
417 // DB conditions check
418 if (!sConditionMgr->IsObjectMeetToConditions(const_cast<Player*>(player), conditions))
419 {
420 // Master Looter can see conditioned recipes
421 if (!isGivenByMasterLooter && isMasterLooter)
422 {
423 if ((pProto->Flags & ITEM_FLAG_HIDE_UNUSABLE_RECIPE) || (pProto->Class == ITEM_CLASS_RECIPE && pProto->Bonding == BIND_WHEN_PICKED_UP && pProto->Spells[1].SpellId != 0))
424 {
425 return true;
426 }
427 }
428
429 return false;
430 }
431
432 // not show loot for not own team
433 if ((pProto->Flags2 & ITEM_FLAGS_EXTRA_HORDE_ONLY) && player->GetTeamId(true) != TEAM_HORDE)
434 {
435 return false;
436 }
437
438 if ((pProto->Flags2 & ITEM_FLAGS_EXTRA_ALLIANCE_ONLY) && player->GetTeamId(true) != TEAM_ALLIANCE)
439 {
440 return false;
441 }
442
443 // Master looter can see all items even if the character can't loot them
444 if (!isGivenByMasterLooter && isMasterLooter && allowQuestLoot)
445 {
446 return true;
447 }
448
449 // Don't allow loot for players without profession or those who already know the recipe
450 if ((pProto->Flags & ITEM_FLAG_HIDE_UNUSABLE_RECIPE) && (!player->HasSkill(pProto->RequiredSkill) || player->HasSpell(pProto->Spells[1].SpellId)))
451 {
452 return false;
453 }
454
455 // Don't allow to loot soulbound recipes that the player has already learned
456 if (pProto->Class == ITEM_CLASS_RECIPE && pProto->Bonding == BIND_WHEN_PICKED_UP && pProto->Spells[1].SpellId != 0 && player->HasSpell(pProto->Spells[1].SpellId))
457 {
458 return false;
459 }
460
461 // check quest requirements
462 if (!(pProto->FlagsCu & ITEM_FLAGS_CU_IGNORE_QUEST_STATUS) && ((needs_quest || (pProto->StartQuest && player->GetQuestStatus(pProto->StartQuest) != QUEST_STATUS_NONE)) && !player->HasQuestForItem(itemid)))
463 {
464 return false;
465 }
466
467 if (!sScriptMgr->OnAllowedForPlayerLootCheck(player, source))
468 {
469 return false;
470 }
471
472 return true;
473}

484// Inserts the item into the loot (called by LootTemplate processors)
485void Loot::AddItem(LootStoreItem const& item)
486{
487 ItemTemplate const* proto = sObjectMgr->GetItemTemplate(item.itemid);
488 if (!proto)
489 return;
490
491 uint32 count = urand(item.mincount, item.maxcount);
492 uint32 stacks = count / proto->GetMaxStackSize() + (count % proto->GetMaxStackSize() ? 1 : 0);
493
494 std::vector<LootItem>& lootItems = item.needs_quest ? quest_items : items;
495 uint32 limit = item.needs_quest ? MAX_NR_QUEST_ITEMS : MAX_NR_LOOT_ITEMS;
496
497 for (uint32 i = 0; i < stacks && lootItems.size() < limit; ++i)
498 {
499 LootItem generatedLoot(item);
500 generatedLoot.count = std::min(count, proto->GetMaxStackSize());
501 generatedLoot.itemIndex = lootItems.size();
502 lootItems.push_back(generatedLoot);
503 count -= proto->GetMaxStackSize();
504
505 // In some cases, a dropped item should be visible/lootable only for some players in group
506 bool canSeeItemInLootWindow = false;
507 if (auto player = ObjectAccessor::FindPlayer(lootOwnerGUID))
508 {
509 if (auto group = player->GetGroup())
510 {
511 for (auto itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
512 {
513 if (auto member = itr->GetSource())
514 {
515 if (generatedLoot.AllowedForPlayer(member, sourceWorldObjectGUID))
516 {
517 canSeeItemInLootWindow = true;
518 break;
519 }
520 }
521 }
522 }
523 else if (generatedLoot.AllowedForPlayer(player, sourceWorldObjectGUID))
524 {
525 canSeeItemInLootWindow = true;
526 }
527 }
528
529 if (!canSeeItemInLootWindow)
530 {
531 LOG_DEBUG("loot", "Skipping ++unlootedCount for unlootable item: {}", item.itemid);
532 continue;
533 }
534
535 // non-conditional one-player only items are counted here,
536 // free for all items are counted in FillFFALoot(),
537 // non-ffa conditionals are counted in FillNonQuestNonFFAConditionalLoot()
538 if (!item.needs_quest && item.conditions.empty() && !(proto->Flags & ITEM_FLAG_MULTI_DROP))
539 ++unlootedCount;
540 }

543// Calls processor of corresponding LootTemplate (which handles everything including references)
544bool Loot::FillLoot(uint32 lootId, LootStore const& store, Player* lootOwner, bool personal, bool noEmptyError, uint16 lootMode /*= LOOT_MODE_DEFAULT*/, WorldObject* lootSource /*= nullptr*/)
545{
546 // Must be provided
547 if (!lootOwner)
548 return false;
549
550 lootOwnerGUID = lootOwner->GetGUID();
551
552 LootTemplate const* tab = store.GetLootFor(lootId);
553
554 if (!tab)
555 {
556 if (!noEmptyError)
557 LOG_ERROR("sql.sql", "Table '{}' loot id #{} used but it doesn't have records.", store.GetName(), lootId);
558 return false;
559 }
560
561 items.reserve(MAX_NR_LOOT_ITEMS);
562 quest_items.reserve(MAX_NR_QUEST_ITEMS);
563
564 tab->Process(*this, store, lootMode, lootOwner); // Processing is done there, callback via Loot::AddItem()
565
566 sScriptMgr->OnAfterLootTemplateProcess(this, tab, store, lootOwner, personal, noEmptyError, lootMode);
567
568 // Setting access rights for group loot case
569 Group* group = lootOwner->GetGroup();
570 if (!personal && group)
571 {
572 roundRobinPlayer = lootOwner->GetGUID();
573
574 for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
575 {
576 if (Player* player = itr->GetSource()) // should actually be looted object instead of lootOwner but looter has to be really close so doesnt really matter
577 {
578 if (player->IsAtLootRewardDistance(lootSource ? lootSource : lootOwner))
579 {
580 FillNotNormalLootFor(player);
581 }
582 }
583 }
584
585 for (uint8 i = 0; i < items.size(); ++i)
586 {
587 if (ItemTemplate const* proto = sObjectMgr->GetItemTemplate(items[i].itemid))
588 if (proto->Quality < uint32(group->GetLootThreshold()))
589 items[i].is_underthreshold = true;
590 }
591 }
592 // ... for personal loot
593 else
594 FillNotNormalLootFor(lootOwner);
595
596 return true;
597}

599void Loot::FillNotNormalLootFor(Player* player)
600{
601 ObjectGuid playerGuid = player->GetGUID();
602
603 QuestItemMap::const_iterator qmapitr = PlayerQuestItems.find(playerGuid);
604 if (qmapitr == PlayerQuestItems.end())
605 FillQuestLoot(player);
606
607 qmapitr = PlayerFFAItems.find(playerGuid);
608 if (qmapitr == PlayerFFAItems.end())
609 FillFFALoot(player);
610
611 qmapitr = PlayerNonQuestNonFFAConditionalItems.find(playerGuid);
612 if (qmapitr == PlayerNonQuestNonFFAConditionalItems.end())
613 FillNonQuestNonFFAConditionalLoot(player);
614
615 // Process currency items
616 uint32 max_slot = GetMaxSlotInLootFor(player);
617 LootItem const* item = nullptr;
618 uint32 itemsSize = uint32(items.size());
619 for (uint32 i = 0; i < max_slot; ++i)
620 {
621 if (i < items.size())
622 item = &items[i];
623 else
624 item = &quest_items[i - itemsSize];
625
626 if (!item->is_looted && item->freeforall && item->AllowedForPlayer(player, sourceWorldObjectGUID))
627 if (ItemTemplate const* proto = sObjectMgr->GetItemTemplate(item->itemid))
628 if (proto->IsCurrencyToken())
629 player->StoreLootItem(i, this);
630 }

633QuestItemList* Loot::FillFFALoot(Player* player)
634{
635 QuestItemList* ql = new QuestItemList();
636
637 for (uint8 i = 0; i < items.size(); ++i)

972 b << uint32(l.gold); //gold
973
974 size_t count_pos = b.wpos(); // pos of item count byte

1645// Rolls for every item in the template and adds the rolled items the the loot
1646void LootTemplate::Process(Loot& loot, LootStore const& store, uint16 lootMode, Player const* player, uint8 groupId) const
1647{
1648 bool rate = store.IsRatesAllowed();
1649
1650 if (groupId) // Group reference uses own processing of the group
1651 {
1652 if (groupId > Groups.size())
1653 return; // Error message already printed at loading stage
1654
1655 if (!Groups[groupId - 1])
1656 return;
1657
1658 Groups[groupId - 1]->Process(loot, player, store, lootMode);
1659 return;
1660 }
1661
1662 // Rolling non-grouped items
1663 for (LootStoreItemList::const_iterator i = Entries.begin(); i != Entries.end(); ++i)
1664 {
1665 LootStoreItem* item = *i;
1666 if (!(item->lootmode & lootMode)) // Do not add if mode mismatch
1667 continue;
1668
1669 if (!item->Roll(rate, player, loot, store))
1670 continue; // Bad luck for the entry
1671
1672 if (item->reference) // References processing
1673 {
1674 LootTemplate const* Referenced = LootTemplates_Reference.GetLootFor(std::abs(item->reference));
1675 if (!Referenced)
1676 continue; // Error message already printed at loading stage
1677
1678 uint32 maxcount = uint32(float(item->maxcount) * sWorld->getRate(RATE_DROP_ITEM_REFERENCED_AMOUNT));
1679 sScriptMgr->OnAfterRefCount(player, loot, rate, lootMode, item, maxcount, store);
1680 for (uint32 loop = 0; loop < maxcount; ++loop) // Ref multiplicator
1681 Referenced->Process(loot, store, lootMode, player, item->groupid);
1682 }
1683 else
1684 {
1685 // Plain entries (not a reference, not grouped)
1686 sScriptMgr->OnBeforeDropAddItem(player, loot, rate, lootMode, item, store);
1687 loot.AddItem(*item); // Chance is already checked, just add
1688 }
1689 }
1690
1691 // Now processing groups
1692 for (LootGroups::const_iterator i = Groups.begin(); i != Groups.end(); ++i)
1693 if (LootGroup* group = *i)
1694 group->Process(loot, player, store, lootMode);
1695}
