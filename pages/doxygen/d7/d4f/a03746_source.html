150 return true;
151 }
152
153 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender */, uint32 action) override
154 {
155 CloseGossipMenuFor(player);
156
157 uint32 spellId = 0;
158 switch (action - GOSSIP_ACTION_INFO_DEF)
159 {
160 case 0:
161 spellId = SPELL_BUILD_CATAPULT_FORCE;
162 break;
163 case 1:
164 spellId = SPELL_BUILD_DEMOLISHER_FORCE;
165 break;
166 case 2:
167 spellId = player->GetTeamId() == TEAM_ALLIANCE ? SPELL_BUILD_SIEGE_VEHICLE_FORCE_ALLIANCE : SPELL_BUILD_SIEGE_VEHICLE_FORCE_HORDE;
168 break;
169 }
170
171 if (canBuild(creature))
172 {
173 creature->CastSpell(player, spellId, true);
174 creature->CastSpell(creature, SPELL_ACTIVATE_CONTROL_ARMS, true);
175 }
176 else
177 {
178 WorldPacket data(SMSG_CAST_FAILED, 1 + 4 + 1 + 4);
179 data << uint8(0);
180 data << spellId;
181 data << uint8(SPELL_FAILED_CUSTOM_ERROR);
182 data << uint32(SPELL_CUSTOM_ERROR_CANT_BUILD_MORE_VEHICLES);
183 player->GetSession()->SendPacket(&data);
184 }
185 return true;
186 }
187
188private:
189 bool canBuild(Creature* creature)
190 {
191 Battlefield* wintergrasp = sBattlefieldMgr->GetBattlefieldByBattleId(BATTLEFIELD_BATTLEID_WG);
192 if (!wintergrasp)
193 return false;
194
195 switch (creature->GetEntry())
196 {
197 case NPC_GOBLIN_MECHANIC:
198 return (wintergrasp->GetData(BATTLEFIELD_WG_DATA_MAX_VEHICLE_H) > wintergrasp->GetData(BATTLEFIELD_WG_DATA_VEHICLE_H));
199 case NPC_GNOMISH_ENGINEER:
200 return (wintergrasp->GetData(BATTLEFIELD_WG_DATA_MAX_VEHICLE_A) > wintergrasp->GetData(BATTLEFIELD_WG_DATA_VEHICLE_A));
201 default:
202 return false;
203 }
204 }
