429 }

441 {
442 // Back to Nature achievement
443 if (aur->GetStackAmount() - param < 25)
444 _backToNature = false;

446 if (aur->GetStackAmount() > param)
447 aur->SetStackAmount(aur->GetStackAmount() - param);
448 else // Aura out of stack
449 {
450 events.ScheduleEvent(EVENT_FREYA_NATURE_BOMB, 5000);
451 events.SetPhase(EVENT_PHASE_FINAL);
452 aur->Remove();
453 return;
454 }
