90 {

92 return;
93
94 events.Update(diff);
95 switch (events.ExecuteEvent())
96 {
97 case EVENT_SPELL_LIGHTNING:
98 for (uint8 i = 0; i < DUNGEON_MODE(1, 2); ++i)
99 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
100 me->CastSpell(target, SPELL_LIGHTNING_CLOUD, false);
101 events.RepeatEvent(urand(15000, 25000));
102 break;
103 case EVENT_SPELL_LUNG:
104 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
105 DoCast(target, SPELL_LUNG_BURST);
106 events.RepeatEvent(urand(7000, 12000));
107 break;
108 case EVENT_SPELL_ENVELOPING:
109 for (uint8 i = 0; i < DUNGEON_MODE(1, 2); ++i)
110 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
111 me->CastSpell(target, SPELL_ENVELOPING_WINDS, false);
112 events.RepeatEvent(urand(10000, 15000));
113 break;
114 }
115
116 DoMeleeAttackIfReady();
117 }
