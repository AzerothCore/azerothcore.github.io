303 uint32 AshbringerEvent(uint32 uiSteps)
304 {
305 Creature* mograine = me->FindNearestCreature(NPC_HIGHLORD_MOGRAINE, 200.0f);
306
307 switch (uiSteps)
308 {
309 case 1:
310 me->GetMotionMaster()->MovePoint(0, 1152.039795f, 1398.405518f, 32.527878f);
311 return 2 * IN_MILLISECONDS;
312 case 2:
313 me->SetSheath(SHEATH_STATE_UNARMED);
314 me->SetStandState(UNIT_STAND_STATE_KNEEL);
315 return 2 * IN_MILLISECONDS;
316 case 3:
317 Talk(3);
318 return 10 * IN_MILLISECONDS;
319 case 4:
320 me->SummonCreature(NPC_HIGHLORD_MOGRAINE, 1065.130737f, 1399.350586f, 30.763723f, 6.282961f, TEMPSUMMON_TIMED_DESPAWN, 400000)->SetName("Highlord Mograine");
321 me->FindNearestCreature(NPC_HIGHLORD_MOGRAINE, 200.0f)->SetFaction(FACTION_FRIENDLY);
322 return 30 * IN_MILLISECONDS;
323 case 5:
324 mograine->StopMovingOnCurrentPos();
325 mograine->AI()->Talk(0);
326 mograine->HandleEmoteCommand(EMOTE_ONESHOT_POINT);
327 return 4 * IN_MILLISECONDS;
328 case 6:
329 me->SetStandState(UNIT_STAND_STATE_STAND);
330 return 2 * IN_MILLISECONDS;
331 case 7:
332 Talk(4);
333 return 4 * IN_MILLISECONDS;
334 case 8:
335 mograine->AI()->Talk(1);
336 return 11 * IN_MILLISECONDS;
337 case 9:
338 mograine->HandleEmoteCommand(EMOTE_ONESHOT_BATTLE_ROAR);
339 return 4 * IN_MILLISECONDS;
340 case 10:
341 me->SetSheath(SHEATH_STATE_UNARMED);
342 me->SetStandState(UNIT_STAND_STATE_KNEEL);
343 Talk(5);
344 return 2 * IN_MILLISECONDS;
345 case 11:
346 mograine->CastSpell(me, SPELL_FORGIVENESS, false);
347 return 1 * IN_MILLISECONDS;
348 case 12:
349 mograine->CastSpell(me, SPELL_COSMETIC_CHAIN, true);
350 return 0.5 * IN_MILLISECONDS;
351 case 13:
352 mograine->AI()->Talk(2);
353 mograine->DespawnOrUnsummon(3 * IN_MILLISECONDS);
354 mograine->Kill(me, me, true);
355 return 0;
356 default:

358 mograine->DespawnOrUnsummon(0);
359 return 0;

464 void UpdateAI(uint32 diff) override
465 {
466 timer = timer - diff;
467 if (SayAshbringer && step < 15)
468 {
469 if (timer <= 0)
470 {
471 timer = AshbringerEvent(step);
472 step++;
473 }
474 }
475
476 if (!UpdateVictim())
477 return;
478
479 if (hasDied && !heal && instance->GetData(TYPE_MOGRAINE_AND_WHITE_EVENT) == SPECIAL)
480 {
481 //On resurrection, stop fake death and heal whitemane and resume fight
482 if (Unit* Whitemane = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_WHITEMANE)))
483 {
484 //Incase wipe during phase that mograine fake death
485 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
486 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
487 me->RemoveAurasDueToSpell(SPELL_PERMANENT_FEIGN_DEATH);
488 me->CastSpell(me, SPELL_RETRIBUTION_AURA, true);
489 me->CastSpell(Whitemane, SPELL_LAY_ON_HANDS, true);
490 events.ScheduleEvent(EVENT_SPELL_CRUSADER_STRIKE, urand(1000, 5000));
491 events.ScheduleEvent(EVENT_SPELL_HAMMER_OF_JUSTICE, urand(6000, 11000));
492 if (me->GetVictim())
493 me->GetMotionMaster()->MoveChase(me->GetVictim());
494 heal = true;
495 }
496 }
497
498 //This if-check to make sure mograine does not attack while fake death
499 if (fakeDeath)
500 return;
501
502 events.Update(diff);
503 if (me->HasUnitState(UNIT_STATE_CASTING))
504 return;
505
506 while (uint32 eventId = events.ExecuteEvent())
507 {
508 switch(eventId)
509 {
510 case EVENT_SPELL_CRUSADER_STRIKE:
511 me->CastSpell(me->GetVictim(), SPELL_CRUSADER_STRIKE, true);
512 events.ScheduleEvent(EVENT_SPELL_CRUSADER_STRIKE, 10000);
513 break;
514 case EVENT_SPELL_HAMMER_OF_JUSTICE:
515 me->CastSpell(me->GetVictim(), SPELL_HAMMER_OF_JUSTICE, true);
516 events.ScheduleEvent(EVENT_SPELL_HAMMER_OF_JUSTICE, 60000);
517 break;
518 case EVENT_PULL_CATHEDRAL:
519 PullCathedral();
520 break;
521 }
522 }
523 DoMeleeAttackIfReady();
524 }

587 void UpdateAI(uint32 diff) override
588 {
589 if (!UpdateVictim())
590 return;
591
592 if (canResurrect)
593 {
594 //When casting resuruction make sure to delay so on rez when reinstate battle deepsleep runs out
595 if (Wait_Timer <= diff)
596 {
597 if (ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_MOGRAINE)))
598 {
599 DoCast(SPELL_SCARLET_RESURRECTION);
600 Talk(SAY_WH_RESURRECT);
601 canResurrect = false;
602 }
603 }
604 else Wait_Timer -= diff;
605 }
606
607 //Cast Deep sleep when health is less than 50%
608 if (!canResurrectCheck && !HealthAbovePct(50))
609 {
610 if (me->IsNonMeleeSpellCast(false))
611 me->InterruptNonMeleeSpells(false);
612
613 me->CastSpell(me->GetVictim(), SPELL_DEEP_SLEEP, true);
614 canResurrectCheck = true;
615 canResurrect = true;
616 return;
617 }
618
619 //while in "resurrect-mode", don't do anything
620 if (canResurrect)
621 return;
622
623 //If we are <75% hp cast healing spells at self or Mograine
624 if (Heal_Timer <= diff)
625 {
626 Creature* target = nullptr;
627
628 if (!HealthAbovePct(75))
629 target = me;
630
631 if (Creature* mograine = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_MOGRAINE)))
632 {
633 // checking canResurrectCheck prevents her healing Mograine while he is "faking death"
634 if (canResurrectCheck && mograine->IsAlive() && !mograine->HealthAbovePct(75))
635 target = mograine;
636 }
637
638 if (target)
639 me->CastSpell(target, SPELL_HEAL, false);
640
641 Heal_Timer = 13000;
642 }
643 else Heal_Timer -= diff;
644
645 events.Update(diff);
646 if (me->HasUnitState(UNIT_STATE_CASTING))
647 return;
648
649 while (uint32 eventId = events.ExecuteEvent())
650 {
651 switch (eventId)
652 {
653 case EVENT_SPELL_POWER_WORLD_SHIELD:
654 me->CastSpell(me, SPELL_POWER_WORD_SHIELD, false);
