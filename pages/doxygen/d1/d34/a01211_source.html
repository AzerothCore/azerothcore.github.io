38ObjectPosSelector::UsedPosList::value_type const* ObjectPosSelector::nextUsedPos(UsedPosType uptype)
39{
40 UsedPosList::const_iterator itr = m_nextUsedPos[uptype];

75bool ObjectPosSelector::FirstAngle(float& angle)
76{
77 if (m_UsedPosLists[USED_POS_PLUS].empty() && !m_UsedPosLists[USED_POS_MINUS].empty())
78 return NextAngleFor(*m_UsedPosLists[USED_POS_MINUS].begin(), 1.0f, USED_POS_PLUS, angle);
79 else if (m_UsedPosLists[USED_POS_MINUS].empty() && !m_UsedPosLists[USED_POS_PLUS].empty())
80 return NextAngleFor(*m_UsedPosLists[USED_POS_PLUS].begin(), -1.0f, USED_POS_MINUS, angle);

85bool ObjectPosSelector::NextAngle(float& angle)
86{
87 while (m_nextUsedPos[USED_POS_PLUS] != m_UsedPosLists[USED_POS_PLUS].end() ||
88 m_nextUsedPos[USED_POS_MINUS] != m_UsedPosLists[USED_POS_MINUS].end() ||
89 m_smallStepOk[USED_POS_PLUS] || m_smallStepOk[USED_POS_MINUS] )

112bool ObjectPosSelector::NextPosibleAngle(float& angle)
113{
114 // ++ direction less updated
115 if (m_nextUsedPos[USED_POS_PLUS] != m_UsedPosLists[USED_POS_PLUS].end() &&
116 (m_nextUsedPos[USED_POS_MINUS] == m_UsedPosLists[USED_POS_MINUS].end() || m_nextUsedPos[USED_POS_PLUS]->first <= m_nextUsedPos[USED_POS_MINUS]->first))
117 {
118 bool ok;
119 if (m_smallStepOk[USED_POS_PLUS])
120 ok = NextSmallStepAngle(1.0f, USED_POS_PLUS, angle);
121 else
122 ok = NextAngleFor(*m_nextUsedPos[USED_POS_PLUS], 1.0f, USED_POS_PLUS, angle);
123
124 if (!ok)
125 ++m_nextUsedPos[USED_POS_PLUS]; // increase. only at fail (original or checked)
126 return ok;
127 }
128 // -- direction less updated
129 else if (m_nextUsedPos[USED_POS_MINUS] != m_UsedPosLists[USED_POS_MINUS].end())
130 {
131 bool ok;
132 if (m_smallStepOk[USED_POS_MINUS])
133 ok = NextSmallStepAngle(-1.0f, USED_POS_MINUS, angle);
134 else
135 ok = NextAngleFor(*m_nextUsedPos[USED_POS_MINUS], -1.0f, USED_POS_MINUS, angle);
136
137 if (!ok)
138 ++m_nextUsedPos[USED_POS_MINUS];
139 return ok;
140 }
141 else // both list empty
142 {
143 if (m_smallStepOk[USED_POS_PLUS] && (!m_smallStepOk[USED_POS_MINUS] || m_smallStepAngle[USED_POS_PLUS] <= m_smallStepAngle[USED_POS_MINUS]))
144 return NextSmallStepAngle(1.0f, USED_POS_PLUS, angle);
145 // -- direction less updated
146 else if (m_smallStepOk[USED_POS_MINUS])
147 return NextSmallStepAngle(-1.0f, USED_POS_MINUS, angle);
148 }
149
150 // no angles
151 return false;
