290// Defines base stats for creatures (used to calculate HP/mana/armor/attackpower/rangedattackpower/all damage).
291struct CreatureBaseStats
292{
293 uint32 BaseHealth[MAX_EXPANSIONS];
294 uint32 BaseMana;
295 uint32 BaseArmor;
296 uint32 AttackPower;
297 uint32 RangedAttackPower;
298 float BaseDamage[MAX_EXPANSIONS];
299
300 // Helpers
301
302 uint32 GenerateHealth(CreatureTemplate const* info) const
303 {
304 return uint32(std::ceil(BaseHealth[info->expansion] * info->ModHealth));
305 }
306
307 uint32 GenerateMana(CreatureTemplate const* info) const
308 {
309 // Mana can be 0.
310 if (!BaseMana)
311 return 0;
312
313 return uint32(std::ceil(BaseMana * info->ModMana));
314 }
315
316 uint32 GenerateArmor(CreatureTemplate const* info) const
317 {
318 return uint32(std::ceil(BaseArmor * info->ModArmor));
319 }
320
321 float GenerateBaseDamage(CreatureTemplate const* info) const
322 {
323 return BaseDamage[info->expansion];
324 }
325
326 static CreatureBaseStats const* GetBaseStats(uint8 level, uint8 unitClass);
327};
