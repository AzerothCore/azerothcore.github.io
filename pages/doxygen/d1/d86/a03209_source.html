226 void SummonedCreatureDies(Creature* summon, Unit* killer) override
227 {
228 switch (summon->GetEntry())
229 {
230 case NPC_TWILIGHT_INITIATE:
231 {
232 GuidList::iterator itr = std::find(oocSummons.begin(), oocSummons.end(), summon->GetGUID());
233 if (itr == oocSummons.end())
234 {
235 break;
236 }

240 {
241 break;
242 }
243
244 DespawnOOCSummons();
245 DoCastSelf(SPELL_HOVER_FALL);
246 me->GetMotionMaster()->MoveIdle();
247 me->GetMotionMaster()->MovePoint(POINT_DOWN, JedogaPosition[1], false);
248
249 if (!combatSummonsSummoned)
250 {
251 summons.DespawnEntry(NPC_TWILIGHT_VOLUNTEER);
252 for (uint8 i = 0; i < MAX_COMBAT_INITIATES; ++i)
253 {
254 if (TempSummon* summon = me->SummonCreature(NPC_TWILIGHT_VOLUNTEER, VolunteerSpotPositions[i][0], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 5000))
255 {
256 summon->GetMotionMaster()->MovePoint(POINT_INITIAL, VolunteerSpotPositions[i][1]);
257 summon->SetReactState(REACT_PASSIVE);
258 summon->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
259 summon->SetImmuneToAll(true);
260 summons.Summon(summon);
261 }
262 }
263 combatSummonsSummoned = true;
264 }
265
266 break;
267 }
268 case NPC_TWILIGHT_VOLUNTEER:
269 {
270 if (sacraficeTarget_GUID && summon->GetGUID() != sacraficeTarget_GUID)
271 {
272 break;
273 }
274
275 if (killer != me && killer->GetGUID() != sacraficeTarget_GUID)
276 {
277 volunteerWork = false;
278 }
279 else
280 {
281 DoCastSelf(SPELL_GIFT_OF_THE_HERALD, true);
282 }
283 events.ScheduleEvent(EVENT_JEDOGA_MOVE_DOWN, 1000, 0, PHASE_RITUAL);
284 break;
285 }
286 }

288 summons.Despawn(summon);

349 {
350 if (!(type == POINT_MOTION_TYPE || type == EFFECT_MOTION_TYPE))

353 }

357 case POINT_DOWN:
358 {
359 me->ClearUnitState(UNIT_STATE_NO_ENVIRONMENT_UPD);

361 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
362 me->SetImmuneToAll(false);
363 me->SetReactState(REACT_AGGRESSIVE);
364

372
373 me->SetInCombatWithZone();
374 if (Unit* victim = me->GetVictim())

376 me->StopMoving();
377 AttackStart(victim);
378 }
379 break;
380 }
381 case POINT_UP:
382 {
383 me->SetFacingTo(5.66f);
384 if (!summons.empty())
385 {
386 sacraficeTarget_GUID = Acore::Containers::SelectRandomContainerElement(summons);

388 {
389 Talk(SAY_SACRIFICE_1);
390 sacraficeTarget_GUID = volunteer->GetGUID();
391 volunteer->AI()->DoAction(ACTION_RITUAL_BEGIN);
392 }
393 // Something failed, let players continue but do not grant achievement
394 else
395 {
396 volunteerWork = false;
397 me->GetMotionMaster()->Clear();
398 DoCastSelf(SPELL_HOVER_FALL);
399 me->GetMotionMaster()->MovePoint(POINT_DOWN, JedogaPosition[1], false);
400 }
401 }
402 break;
403 }
404 case POINT_RITUAL:
405 {
406 me->SetFacingTo(5.66f);
407 DoCastSelf(SPELL_HOVER_FALL);
408 events.ScheduleEvent(EVENT_JEDOGA_MOVE_UP, 1000, 0, PHASE_RITUAL);
409 break;
410 }

412 {
413 me->SetFacingTo(5.66f);
414 DoCastSelf(SPELL_SPHERE_VISUAL, true);
415 DoCastSelf(SPELL_LIGHTNING_BOLTS, true);
416 if (!oocTriggers.empty())

418 for (ObjectGuid const& guid : oocTriggers)

431 void UpdateAI(uint32 diff) override
432 {
433 if (!UpdateVictim())
434 {
435 if (instance->GetBossState(DATA_PRINCE_TALDARAM) == DONE)
436 {
437 if (sayPreachTimer <= diff)
438 {
439 Talk(SAY_PREACHING);
440 sayPreachTimer = 120000; // 2 min
441 }
442 else
443 {
444 sayPreachTimer -= diff;
445 }
446 }
447 return;
448 }
449
450 events.Update(diff);
451 if (me->HasUnitState(UNIT_STATE_CASTING))
452 {
453 return;
454 }
455
456 while (uint32 const eventId = events.ExecuteEvent())
457 {
458 switch (eventId)
459 {
460 // Normal phase
461 case EVENT_JEDOGA_CYCLONE:
462 {
463 DoCastSelf(DUNGEON_MODE(SPELL_CYCLONE_STRIKE, SPELL_CYCLONE_STRIKE_H), false);
464 events.RepeatEvent(urand(10000, 14000));
465 break;
466 }
467 case EVENT_JEDOGA_LIGHTNING_BOLT:
468 {
469 if (Unit* pTarget = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
470 {
471 DoCast(pTarget, DUNGEON_MODE(SPELL_LIGHTNING_BOLT, SPELL_LIGHTNING_BOLT_H), false);
472 }
473 events.RepeatEvent(urand(11000, 15000));
474 break;
475 }
476 case EVENT_JEDOGA_THUNDERSHOCK:
477 {
478 if (Unit* pTarget = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
479 {
480 DoCast(pTarget, DUNGEON_MODE(SPELL_THUNDERSHOCK, SPELL_THUNDERSHOCK_H), false);
481 }
482
483 events.RepeatEvent(urand(16000, 22000));
484 break;
485 }
486 // Ritual phase
487 case EVENT_JEDOGA_PREPARE_RITUAL:
488 {
489 me->GetMotionMaster()->Clear(true);
490 me->GetMotionMaster()->MovePoint(POINT_RITUAL, JedogaPosition[1]);
491 break;
492 }
493 case EVENT_JEDOGA_MOVE_UP:
494 {
495 me->GetMotionMaster()->Clear(true);
496 me->SetDisableGravity(true);
497 me->SetHover(true);
498 me->GetMotionMaster()->MoveTakeoff(POINT_UP, JedogaPosition[0], 7.0f);
499 break;
500 }
501 case EVENT_JEDOGA_MOVE_DOWN:
502 {
503 summons.DespawnEntry(NPC_JEDOGA_CONTROLLER);
504 DoCastSelf(SPELL_HOVER_FALL);
505 me->GetMotionMaster()->Clear();
506 me->GetMotionMaster()->MovePoint(POINT_DOWN, JedogaPosition[1], false);
507 break;
508 }
509 }
510 }
511
512 DoMeleeAttackIfReady();
513 }
