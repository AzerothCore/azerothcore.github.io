56 void Reset() override

69 if (!enraged && me->HealthBelowPctDamaged(10, damage))

75 }

78 void UpdateAI(uint32 diff) override
79 {
80 if (!UpdateVictim())
81 {
82 return;
83 }
84
85 // Should not get impact by cast state (cast should always happen)
86 if (earthquakeTimer)
87 {
88 if (earthquakeTimer <= diff)
89 {
90 DoCastSelf(SPELL_EARTHQUAKE, true);
91 earthquakeTimer = 5000;
92 }
93 else
94 {
95 earthquakeTimer -= diff;
96 }
97 }
98
99 if (me->HasUnitState(UNIT_STATE_CASTING))
100 {
101 return;
102 }
103
104 if (pyroblastTimer <= diff)
105 {
106 DoCastRandomTarget(SPELL_PYROBLAST);
107
108 pyroblastTimer = 7000;
109 }
110 else
111 {
112 pyroblastTimer -= diff;
113 }
114
115 DoMeleeAttackIfReady();
116 }
117
118 private:
119 uint32 earthquakeTimer;
120 uint32 pyroblastTimer;
121 bool enraged;
