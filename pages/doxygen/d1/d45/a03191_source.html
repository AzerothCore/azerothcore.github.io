106 void UpdateAI(uint32 diff) override
107 {
108 if (postGossipStep > 0 && postGossipStep < 4)
109 {
110 if (Text_Timer <= diff)
111 {
112 switch (postGossipStep)
113 {
114 case 1:
115 //weegli doesn't fight - he goes & blows up the door
116 if (Creature* pWeegli = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_WEEGLI)))
117 {
118 pWeegli->AI()->DoAction(0);
119 }
120 Talk(SAY_1);
121 Text_Timer = 5000;
122 break;
123 case 2:
124 Talk(SAY_2);
125 Text_Timer = 5000;
126 break;
127 case 3:
128 me->SetFaction(FACTION_MONSTER);
129 Player* target = ObjectAccessor::GetPlayer(*me, PlayerGUID);
130
131 switchFactionIfAlive(NPC_RAVEN, target);
132 switchFactionIfAlive(NPC_ORO, target);
133 switchFactionIfAlive(NPC_MURTA, target);
134
135 if (target)
136 {
137 AttackStart(target);
138 }
139 }
140
141 postGossipStep++;
142 }
143 else
144 {
145 Text_Timer -= diff;
146 }
147 }
148
149 if (!UpdateVictim())
150 {
151 return;
152 }
153
154 if (ShieldBash_Timer <= diff)
155 {
156 DoCastVictim(SPELL_SHIELD_BASH);
157 ShieldBash_Timer = 15000;
158 }
159 else
160 {
161 ShieldBash_Timer -= diff;
162 }
163
164 if (Revenge_Timer <= diff)
165 {
166 DoCastVictim(SPELL_REVENGE);
167 Revenge_Timer = 10000;
168 }

170 {
171 Revenge_Timer -= diff;
172 }
173
174 DoMeleeAttackIfReady();
