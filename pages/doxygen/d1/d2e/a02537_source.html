54class boss_garr : public CreatureScript
55{
56public:
57 boss_garr() : CreatureScript("boss_garr") {}
58
59 struct boss_garrAI : public BossAI
60 {
61 boss_garrAI(Creature* creature) : BossAI(creature, DATA_GARR),
62 massEruptionTimer(600000) // 10 mins
63 {
64 }
65
66 void Reset() override
67 {
68 _Reset();
69 massEruptionTimer = 600000;
70 }
71
72 void EnterCombat(Unit* /*attacker*/) override
73 {
74 _EnterCombat();
75 DoCastSelf(SPELL_SEPARATION_ANXIETY, true);
76 events.ScheduleEvent(EVENT_ANTIMAGIC_PULSE, 15000);
77 events.ScheduleEvent(EVENT_MAGMA_SHACKLES, 10000);
78 massEruptionTimer = 600000; // 10 mins
79 }
80
81 void UpdateAI(uint32 diff) override
82 {
83 if (!UpdateVictim())
84 {
85 return;
86 }
87
88 // This should always process
89 if (massEruptionTimer <= diff)
90 {
91 Talk(EMOTE_MASS_ERRUPTION, me);
92 DoCastAOE(SPELL_ERUPTION_TRIGGER, true);
93 massEruptionTimer = 20000;
94 }
95 else
96 {
97 massEruptionTimer -= diff;
98 }
99
100 events.Update(diff);
101
102 if (me->HasUnitState(UNIT_STATE_CASTING))
103 {
104 return;
105 }
106
107 while (uint32 const eventId = events.ExecuteEvent())
108 {
109 switch (eventId)
110 {
111 case EVENT_ANTIMAGIC_PULSE:
112 {
113 DoCastSelf(SPELL_ANTIMAGIC_PULSE);
114 events.RepeatEvent(20000);
115 break;
116 }
117 case EVENT_MAGMA_SHACKLES:
118 {
119 DoCastSelf(SPELL_MAGMA_SHACKLES);
120 events.RepeatEvent(15000);
121 break;
122 }
123 }
124
125 if (me->HasUnitState(UNIT_STATE_CASTING))
126 {
127 return;
128 }
129 }
130
131 DoMeleeAttackIfReady();
132 }
133
134 private:
135 uint32 massEruptionTimer;
136 };
137
138 CreatureAI* GetAI(Creature* creature) const override
139 {
140 return GetMoltenCoreAI<boss_garrAI>(creature);
141 }
142};

180 class spell_garr_separation_nexiety_AuraScript : public AuraScript
181 {
182 PrepareAuraScript(spell_garr_separation_nexiety_AuraScript);
183
184 bool Validate(SpellInfo const* /*spell*/) override
185 {
186 return ValidateSpellInfo({ SPELL_SEPARATION_ANXIETY_MINION });
187 }
188
189 void HandlePeriodic(AuraEffect const* aurEff)
190 {
191 Unit const* caster = GetCaster();
192 Unit* target = GetTarget();
193 if (caster && target && target->GetDistance(caster) > 40.0f && !target->HasAura(SPELL_SEPARATION_ANXIETY_MINION))
194 {
195 target->CastSpell(target, SPELL_SEPARATION_ANXIETY_MINION, true, nullptr, aurEff);
196 }
197 }
198
199 void Register() override
200 {
201 OnEffectPeriodic += AuraEffectPeriodicFn(spell_garr_separation_nexiety_AuraScript::HandlePeriodic, EFFECT_0, SPELL_AURA_PERIODIC_DUMMY);
202 }
203 };
