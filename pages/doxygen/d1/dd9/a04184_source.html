561 void CalculateAmount(AuraEffect const* aurEff, int32& amount, bool& canBeRecalculated)
562 {
563 if (Unit* caster = GetCaster())
564 {
565 canBeRecalculated = false;
566
567 // $0.2 * (($MWB + $mwb) / 2 + $AP / 14 * $MWS) bonus per tick
568 float ap = caster->GetTotalAttackPowerValue(BASE_ATTACK);
569 int32 mws = caster->GetAttackTime(BASE_ATTACK);
570 float mwbMin = 0.f;
571 float mwbMax = 0.f;
572 for (uint8 i = 0; i < MAX_ITEM_PROTO_DAMAGES; ++i)
573 {
574 mwbMin += caster->GetWeaponDamageRange(BASE_ATTACK, MINDAMAGE, i);
575 mwbMax += caster->GetWeaponDamageRange(BASE_ATTACK, MAXDAMAGE, i);
576 }
577
578 float mwb = ((mwbMin + mwbMax) / 2 + ap * mws / 14000) * 0.2f;
579 amount += int32(caster->ApplyEffectModifiers(GetSpellInfo(), aurEff->GetEffIndex(), mwb));
580
581 // "If used while your target is above 75% health, Rend does 35% more damage."
582 // as for 3.1.3 only ranks above 9 (wrong tooltip?)
583 if (GetSpellInfo()->GetRank() >= 9)
584 {
585 if (GetUnitOwner()->HasAuraState(AURA_STATE_HEALTH_ABOVE_75_PERCENT, GetSpellInfo(), caster))
586 AddPct(amount, GetSpellInfo()->Effects[EFFECT_2].CalcValue(caster));
587 }
588 }
