122 void SetData(uint32 type, uint32 data) override
123 {

194 void Load(const char* strIn) override
195 {

198 OUT_LOAD_INST_DATA_FAIL;
199 return;
200 }
201
202 char dataHead1, dataHead2;
203 std::istringstream loadStream(strIn);
204 loadStream >> dataHead1 >> dataHead2;
205
206 if (dataHead1 == 'S' && dataHead2 == 'V')
207 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
208 {
209 loadStream >> m_auiEncounter[i];
210 if (m_auiEncounter[i] == IN_PROGRESS)
211 m_auiEncounter[i] = NOT_STARTED;
212 }
213
214 OUT_LOAD_INST_DATA_COMPLETE;
215 }
