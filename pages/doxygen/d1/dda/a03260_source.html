90class boss_eadric : public CreatureScript
91{
92public:
93 boss_eadric() : CreatureScript("boss_eadric") { }
94 struct boss_eadricAI : public ScriptedAI
95 {
96 boss_eadricAI(Creature* pCreature) : ScriptedAI(pCreature)
97 {
98 pInstance = pCreature->GetInstanceScript();
99 }
100
101 InstanceScript* pInstance;
102 EventMap events;
103
104 void Reset() override
105 {
106 events.Reset();
107 me->SetReactState(REACT_PASSIVE);
108 if( pInstance )
109 pInstance->SetData(BOSS_ARGENT_CHALLENGE, NOT_STARTED);
110 }
111
112 void MovementInform(uint32 type, uint32 id) override
113 {
114 if( type == POINT_MOTION_TYPE && id == 1 )
115 me->SetFacingTo(3 * M_PI / 2);
116 }
117
118 void KilledUnit(Unit* who) override
119 {
120 if( who->GetTypeId() == TYPEID_PLAYER )
121 {
122 if( urand(0, 1) )
123 Talk(TEXT_EADRIC_SLAIN_1);
124 else
125 Talk(TEXT_EADRIC_SLAIN_2);
126 }
127 }
128
129 void EnterCombat(Unit* /*who*/) override
130 {
131 events.Reset();
132 events.ScheduleEvent(EVENT_SPELL_RADIANCE, 16000);
133 events.ScheduleEvent(EVENT_SPELL_HAMMER_RIGHTEOUS, 25000);
134 Talk(TEXT_EADRIC_AGGRO);
135 me->CastSpell(me, SPELL_VENGEANCE, false);
136 if( pInstance )
137 pInstance->SetData(BOSS_ARGENT_CHALLENGE, IN_PROGRESS);
138 }
139
140 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
141 {
142 if (spell->Id == 66905 && me->GetHealth() == 1) // hammer throw back damage (15k)
143 me->CastSpell(me, 68197, true);
144 }
145
146 void DamageTaken(Unit*, uint32& damage, DamageEffectType, SpellSchoolMask) override
147 {
148 if( damage >= me->GetHealth() )
149 {
150 damage = me->GetHealth() - 1;
151 if (me->GetFaction() != FACTION_FRIENDLY)
152 {
153 me->CastSpell((Unit*)nullptr, 68575, true); // achievements
154 me->GetMap()->UpdateEncounterState(ENCOUNTER_CREDIT_CAST_SPELL, 68574, me); // paletress' spell credits encounter, but shouldn't credit achievements
155 me->SetFaction(FACTION_FRIENDLY);
156 events.Reset();
157 Talk(TEXT_EADRIC_DEATH);
158 me->GetThreatMgr().clearReferences();
159 me->SetRegeneratingHealth(false);
160 _EnterEvadeMode();
161 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
162 me->SetImmuneToAll(true);
163 if( pInstance )
164 pInstance->SetData(BOSS_ARGENT_CHALLENGE, DONE);
165 }
166 }
167 }
168
169 void UpdateAI(uint32 diff) override
170 {
171 if( !UpdateVictim() )
172 return;
173
174 events.Update(diff);
175
176 if( me->HasUnitState(UNIT_STATE_CASTING) )
177 return;
178
179 switch( events.ExecuteEvent() )
180 {
181 case 0:
182 break;
183 case EVENT_SPELL_RADIANCE:
184 me->CastSpell((Unit*)nullptr, SPELL_RADIANCE, false);
185 me->TextEmote(TEXT_RADIATE, nullptr, true);
186 events.RepeatEvent(16000);
187 break;
188 case EVENT_SPELL_HAMMER_RIGHTEOUS:
189 if( Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 55.0f, true) )
190 {
191 char buffer[100];
192 sprintf(buffer, "Eadric the Pure targets %s with the Hammer of the Righteous!", target->GetName().c_str());
193 me->TextEmote(buffer, nullptr, true);
194 Talk(TEXT_EADRIC_HAMMER);
195 me->CastSpell(target, SPELL_HAMMER_JUSTICE, true);
196 me->CastSpell(target, SPELL_HAMMER_RIGHTEOUS, false);
197 }
198 events.RepeatEvent(25000);
199 break;
200 }
201
202 DoMeleeAttackIfReady();
203 }
204 };
205
206 CreatureAI* GetAI(Creature* pCreature) const override
207 {
208 return GetTrialOfTheChampionAI<boss_eadricAI>(pCreature);
209 }

327 void SummonMemory()
328 {
329 uint8 uiRandom = urand(0, 25);
330 uint32 uiSpells[26] = {66704, 66705, 66706, 66707, 66709, 66710, 66711, 66712, 66713, 66714, 66715, 66708, 66708, 66691, 66692, 66694, 66695, 66696, 66697, 66698, 66699, 66700, 66701, 66702, 66703, 66543};
331 me->CastSpell(me, uiSpells[uiRandom], true);
332 }
333
334 void UpdateAI(uint32 diff) override
335 {
336 if( !UpdateVictim() )
337 return;
338
339 events.Update(diff);
340
341 if( !summoned && HealthBelowPct(25) )
342 {
343 me->InterruptNonMeleeSpells(true);
344 Talk(TEXT_PALETRESS_MEMORY_SUMMON);
345 me->CastSpell((Unit*)nullptr, SPELL_HOLY_NOVA, false);
346 me->CastSpell(me, SPELL_SHIELD, false);
347 me->CastSpell((Unit*)nullptr, SPELL_SUMMON_MEMORY, false);
348 SummonMemory();
349 me->CastSpell((Unit*)nullptr, SPELL_CONFESS, false);
350 events.ScheduleEvent(EVENT_SPELL_RENEW, urand(6000, 8000));
351 summoned = true;
352 return;
353 }
354
355 if( me->HasUnitState(UNIT_STATE_CASTING) )
356 return;
357
358 switch( events.ExecuteEvent() )
359 {
360 case 0:
361 break;
362 case EVENT_SPELL_SMITE:
363 if( Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true) )
364 me->CastSpell(target, SPELL_SMITE, false);
365 events.RepeatEvent(urand(3000, 4000));
366 break;
367 case EVENT_SPELL_HOLY_FIRE:
368 if( Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 30.0f, true) )
369 me->CastSpell(target, SPELL_HOLY_FIRE, false);
370 events.RepeatEvent(urand(9000, 12000));
371 break;
372 case EVENT_SPELL_RENEW:
373 if( !MemoryGUID )
374 {
375 break;
376 }
377 if( urand(0, 1) )
378 me->CastSpell(me, SPELL_RENEW, false);
379 else if( Creature* memory = ObjectAccessor::GetCreature(*me, MemoryGUID) )
380 if( memory->IsAlive() )
381 me->CastSpell(memory, SPELL_RENEW, false);
382 events.RepeatEvent(urand(15000, 17000));
383 break;
384 }
385
386 DoMeleeAttackIfReady();
387 }
388 };

390 CreatureAI* GetAI(Creature* pCreature) const override
391 {
392 return GetTrialOfTheChampionAI<boss_paletressAI>(pCreature);
393 }
