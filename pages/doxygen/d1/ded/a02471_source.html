139 void UpdateAI(uint32 diff) override
140 {
141 if (!UpdateVictim() && !events.IsInPhase(EVENT_PHASE_TALK))
142 {
143 return;
144 }
145
146 events.Update(diff);
147
148 if (me->HasUnitState(UNIT_STATE_CASTING))
149 {
150 return;
151 }
152
153 while (uint32 eventId = events.ExecuteEvent())
154 {
155 switch (eventId)
156 {
157 case EVENT_FIGHT:
158 StartFighting();
159 break;
160 case EVENT_SUMMON_SPECTRAL_ASSASSIN:
161 DoCast(me, SPELL_SUMMON_SPECTRAL_ASSASSIN);
162 events.ScheduleEvent(EVENT_SUMMON_SPECTRAL_ASSASSIN, urand(30 * IN_MILLISECONDS, 35 * IN_MILLISECONDS), 0, EVENT_PHASE_FIGHT);
163 break;
164 case EVENT_SHADOW_BOLT_VOLLEY:
165 DoCastVictim(SPELL_SHADOW_BOLT_VOLLEY);
166 events.ScheduleEvent(EVENT_SHADOW_BOLT_VOLLEY, urand(4 * IN_MILLISECONDS, 6 * IN_MILLISECONDS), 0, EVENT_PHASE_FIGHT);
167 break;
168 case EVENT_SHADOW_WRATH:
169 DoCastVictim(SPELL_SHADOW_WRATH);
170 events.ScheduleEvent(EVENT_SHADOW_WRATH, urand(19 * IN_MILLISECONDS, 24 * IN_MILLISECONDS), 0, EVENT_PHASE_FIGHT);
171 break;
172 default:
173 break;
174 }
175 }
176
177 DoMeleeAttackIfReady();
178 }
179 private:
180 bool frenzy40;
181 bool frenzy15;
182 };
