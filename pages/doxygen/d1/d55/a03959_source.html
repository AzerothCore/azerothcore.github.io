45 struct boss_broggokAI : public ScriptedAI
46 {
47 boss_broggokAI(Creature* creature) : ScriptedAI(creature)
48 {
49 instance = creature->GetInstanceScript();
50 }
51
52 InstanceScript* instance;
53 EventMap events;
54 bool canAttack;
55
56 void Reset() override
57 {
58 events.Reset();
59
60 me->SetReactState(REACT_PASSIVE);
61 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
62 me->SetImmuneToAll(true);
63 canAttack = false;
64
65 if (instance)
66 instance->SetData(DATA_BROGGOK, NOT_STARTED);
67 }
68
69 void EnterCombat(Unit* /*who*/) override
70 {
71 Talk(SAY_AGGRO);
72 }
73
74 void JustSummoned(Creature* summoned) override
75 {
76 summoned->SetFaction(FACTION_MONSTER_2);
77 summoned->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
78 summoned->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
79 summoned->CastSpell(summoned, SPELL_POISON, false, 0, 0, me->GetGUID());
80 }
81
82 void UpdateAI(uint32 diff) override
83 {
84 if (!UpdateVictim() || !canAttack)
85 return;
86
87 events.Update(diff);
88 if (me->HasUnitState(UNIT_STATE_CASTING))
89 return;
90
91 switch (events.ExecuteEvent())
92 {
93 case EVENT_SPELL_SLIME:
94 me->CastSpell(me->GetVictim(), SPELL_SLIME_SPRAY, false);
95 events.RepeatEvent(urand(7000, 12000));
96 break;
97 case EVENT_SPELL_BOLT:
98 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
99 me->CastSpell(target, SPELL_POISON_BOLT, false);
100 events.RepeatEvent(urand(6000, 11000));
101 break;
102 case EVENT_SPELL_POISON:
103 me->CastSpell(me, SPELL_POISON_CLOUD, false);
104 events.RepeatEvent(20000);
105 break;
106 }
107
108 DoMeleeAttackIfReady();
109 }
110
111 void JustDied(Unit* /*killer*/) override
112 {
113 if (instance)
114 {
115 instance->HandleGameObject(instance->GetGuidData(DATA_DOOR4), true);
116 instance->HandleGameObject(instance->GetGuidData(DATA_DOOR5), true);
117 instance->SetData(DATA_BROGGOK, DONE);
118 }
119 }
120
121 void DoAction(int32 action) override
122 {
123 switch (action)
124 {
125 case ACTION_PREPARE_BROGGOK:
126 me->SetInCombatWithZone();
127 break;
128 case ACTION_ACTIVATE_BROGGOK:
129 events.ScheduleEvent(EVENT_SPELL_SLIME, 10000);
130 events.ScheduleEvent(EVENT_SPELL_POISON, 5000);
131 events.ScheduleEvent(EVENT_SPELL_BOLT, 7000);
132
133 me->SetReactState(REACT_AGGRESSIVE);
134 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
135 me->SetImmuneToAll(false);
136 canAttack = true;
137 break;
138 }
139 }
140 };
141
142 CreatureAI* GetAI(Creature* creature) const override
143 {
144 return GetBloodFurnaceAI<boss_broggokAI>(creature);
145 }
146};

154 {
155 if (InstanceScript* instance = go->GetInstanceScript())
156 if (instance->GetData(DATA_BROGGOK) != DONE && instance->GetData(DATA_BROGGOK) != IN_PROGRESS)
157 if (Creature* broggok = ObjectAccessor::GetCreature(*go, instance->GetGuidData(DATA_BROGGOK)))
158 {
159 instance->SetData(DATA_BROGGOK, IN_PROGRESS);
160 broggok->AI()->DoAction(ACTION_PREPARE_BROGGOK);
161 }
162
163 go->UseDoorOrButton();
