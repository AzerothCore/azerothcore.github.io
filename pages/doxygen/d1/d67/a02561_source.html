55 struct boss_sulfuronAI : public BossAI
56 {
57 boss_sulfuronAI(Creature* creature) : BossAI(creature, DATA_SULFURON) {}
58
59 void EnterCombat(Unit* /*victim*/) override
60 {
61 _EnterCombat();
62 events.ScheduleEvent(EVENT_DEMORALIZING_SHOUT, urand(6000, 20000));
63 events.ScheduleEvent(EVENT_INSPIRE, urand(7000, 10000));
64 events.ScheduleEvent(EVENT_KNOCKDOWN, 6000);
65 events.ScheduleEvent(EVENT_FLAMESPEAR, 2000);
66 }
67
68 void ExecuteEvent(uint32 eventId) override
69 {
70 switch (eventId)
71 {
72 case EVENT_DEMORALIZING_SHOUT:
73 {
74 DoCastVictim(SPELL_DEMORALIZING_SHOUT);
75 events.RepeatEvent(urand(12000, 18000));
76 break;
77 }
78 case EVENT_INSPIRE:
79 {
80 std::list<Creature*> healers = DoFindFriendlyMissingBuff(45.0f, SPELL_INSPIRE);
81 if (!healers.empty())
82 {
83 DoCast(Acore::Containers::SelectRandomContainerElement(healers), SPELL_INSPIRE);
84 }
85
86 DoCastSelf(SPELL_INSPIRE);
87 events.RepeatEvent(urand(13000, 20000));
88 break;
89 }
90 case EVENT_KNOCKDOWN:
91 {
92 DoCastVictim(SPELL_KNOCKDOWN);
93 events.RepeatEvent(urand(10000, 20000));
94 break;
95 }
96 case EVENT_FLAMESPEAR:
97 {
98 DoCastRandomTarget(SPELL_FLAMESPEAR);
99 events.RepeatEvent(urand(12000, 16000));
100 break;
101 }
102 }
103 }
104 };
105
106 CreatureAI* GetAI(Creature* creature) const override
107 {
108 return GetMoltenCoreAI<boss_sulfuronAI>(creature);
109 }

119 npc_flamewaker_priestAI(Creature* creature) : ScriptedAI(creature) {}
120
121 void Reset() override
122 {
123 events.Reset();
124 }
125
126 void JustDied(Unit* /*killer*/) override
127 {
128 events.Reset();
129 }
130
131 void EnterCombat(Unit* /*victim*/) override
132 {
133 events.ScheduleEvent(EVENT_DARK_STRIKE, urand(4000, 7000));
134 events.ScheduleEvent(EVENT_DARK_MENDING, urand(15000, 30000));
135 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(2000, 4000));
136 events.ScheduleEvent(EVENT_IMMOLATE, urand(3500, 6000));
137 }
138
139 void UpdateAI(uint32 diff) override
140 {
141 if (!UpdateVictim())
142 {
143 return;
144 }
145
146 events.Update(diff);
147
148 if (me->HasUnitState(UNIT_STATE_CASTING))
149 {
150 return;
151 }
152
153 while (uint32 const eventId = events.ExecuteEvent())
154 {
155 switch (eventId)
156 {
157 case EVENT_DARK_STRIKE:
158 {
159 DoCastVictim(SPELL_DARK_STRIKE);
160 events.RepeatEvent(urand(4000, 7000));
161 break;
162 }
163 case EVENT_DARK_MENDING:
164 {
165 if (Unit* target = DoSelectLowestHpFriendly(60.0f, 1))
166 {
167 if (target->GetGUID() != me->GetGUID())
168 {
169 DoCast(target, SPELL_DARK_MENDING);
170 }
171 }
172 events.RepeatEvent(urand(15000, 20000));
173 break;
174 }
175 case EVENT_SHADOW_WORD_PAIN:
176 {
177 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true, -SPELL_SHADOW_WORD_PAIN))
178 {
179 DoCast(target, SPELL_SHADOW_WORD_PAIN);
180 }
181 events.RepeatEvent(urand(2500, 5000));
182 break;
183 }
184 case EVENT_IMMOLATE:
185 {
186 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true, -SPELL_IMMOLATE))
187 {
188 DoCast(target, SPELL_IMMOLATE);
189 }
190 events.RepeatEvent(urand(5000, 7000));
191 break;
192 }
193 }
194
195 if (me->HasUnitState(UNIT_STATE_CASTING))
196 {
197 return;
198 }
199 }
200
201 DoMeleeAttackIfReady();
202 }
203
204 private:
205 EventMap events;
206 };
207
208 CreatureAI* GetAI(Creature* creature) const override
209 {
210 return GetMoltenCoreAI<npc_flamewaker_priestAI>(creature);
211 }
212};
