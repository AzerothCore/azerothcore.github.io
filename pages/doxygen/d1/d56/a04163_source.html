235 void CalculateResistanceAmount(AuraEffect const* aurEff, int32& amount, bool& /*canBeRecalculated*/)
236 {
237 // xinef: mage pet inherits 40% of resistance from owner and 35% of armor (guessed)
238 if (Unit* owner = GetUnitOwner()->GetOwner())

242 amount = CalculatePct(std::max<int32>(0, owner->GetResistance(schoolMask)), modifier);
243 }
244 }
245
246 void CalculateStatAmount(AuraEffect const* aurEff, int32& amount, bool& /*canBeRecalculated*/)

248 // xinef: mage pet inherits 30% of intellect / stamina
249 if (Unit* owner = GetUnitOwner()->GetOwner())
250 {
251 Stats stat = Stats(aurEff->GetSpellInfo()->Effects[aurEff->GetEffIndex()].MiscValue);
252 amount = CalculatePct(std::max<int32>(0, owner->GetStat(stat)), 30);
253 }
254 }
255
256 void CalculateAPAmount(AuraEffect const* /*aurEff*/, int32& /*amount*/, bool& /*canBeRecalculated*/)
257 {
258 // xinef: mage pet inherits 0% AP
259 }
260
261 void CalculateSPAmount(AuraEffect const* /*aurEff*/, int32& amount, bool& /*canBeRecalculated*/)
262 {
263 // xinef: mage pet inherits 33% of SP
264 if (Unit* owner = GetUnitOwner()->GetOwner())
265 {
266 int32 frost = owner->SpellBaseDamageBonusDone(SPELL_SCHOOL_MASK_FROST);
267 amount = CalculatePct(std::max<int32>(0, frost), 33);
268
269 // xinef: Update appropriate player field
270 if (owner->GetTypeId() == TYPEID_PLAYER)
271 owner->SetUInt32Value(PLAYER_PET_SPELL_POWER, (uint32)amount);
272 }
273 }
274
275 void HandleEffectApply(AuraEffect const* aurEff, AuraEffectHandleModes /*mode*/)
276 {
277 if (GetUnitOwner()->IsPet())
278 return;
279
280 GetUnitOwner()->ApplySpellImmune(0, IMMUNITY_STATE, aurEff->GetAuraType(), true, SPELL_BLOCK_TYPE_POSITIVE);
281 if (aurEff->GetAuraType() == SPELL_AURA_MOD_ATTACK_POWER)
282 GetUnitOwner()->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_ATTACK_POWER_PCT, true, SPELL_BLOCK_TYPE_POSITIVE);
283 else if (aurEff->GetAuraType() == SPELL_AURA_MOD_STAT)
284 GetUnitOwner()->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_TOTAL_STAT_PERCENTAGE, true, SPELL_BLOCK_TYPE_POSITIVE);
285 }
286
287 void CalcPeriodic(AuraEffect const* /*aurEff*/, bool& isPeriodic, int32& amplitude)
288 {
289 if (!GetUnitOwner()->IsPet())
290 return;
291
292 isPeriodic = true;
293 amplitude = 2 * IN_MILLISECONDS;
294 }
295
296 void HandlePeriodic(AuraEffect const* aurEff)
297 {
298 PreventDefaultAction();
299 if (aurEff->GetAuraType() == SPELL_AURA_MOD_STAT && (aurEff->GetMiscValue() == STAT_STAMINA || aurEff->GetMiscValue() == STAT_INTELLECT))
300 {
301 int32 currentAmount = aurEff->GetAmount();
302 int32 newAmount = GetEffect(aurEff->GetEffIndex())->CalculateAmount(GetCaster());
303 if (newAmount != currentAmount)
304 {
305 if (aurEff->GetMiscValue() == STAT_STAMINA)
306 {
307 uint32 actStat = GetUnitOwner()->GetHealth();
308 GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
309 GetUnitOwner()->SetHealth(std::min<uint32>(GetUnitOwner()->GetMaxHealth(), actStat));
310 }
311 else
312 {
313 uint32 actStat = GetUnitOwner()->GetPower(POWER_MANA);
314 GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
315 GetUnitOwner()->SetPower(POWER_MANA, std::min<uint32>(GetUnitOwner()->GetMaxPower(POWER_MANA), actStat));
316 }
317 }
318 }
319 else
320 GetEffect(aurEff->GetEffIndex())->RecalculateAmount();
321 }
322
323 void Register() override
324 {
325 if (m_scriptSpellId != 35657)
326 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_mage_pet_scaling::CalculateResistanceAmount, EFFECT_ALL, SPELL_AURA_MOD_RESISTANCE);
327
328 if (m_scriptSpellId == 35657 || m_scriptSpellId == 35658)
329 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_mage_pet_scaling::CalculateStatAmount, EFFECT_ALL, SPELL_AURA_MOD_STAT);
330
331 if (m_scriptSpellId == 35657)
332 {
333 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_mage_pet_scaling::CalculateAPAmount, EFFECT_ALL, SPELL_AURA_MOD_ATTACK_POWER);
334 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_mage_pet_scaling::CalculateSPAmount, EFFECT_ALL, SPELL_AURA_MOD_DAMAGE_DONE);
335 }
336
337 OnEffectApply += AuraEffectApplyFn(spell_mage_pet_scaling::HandleEffectApply, EFFECT_ALL, SPELL_AURA_ANY, AURA_EFFECT_HANDLE_REAL);
338 DoEffectCalcPeriodic += AuraEffectCalcPeriodicFn(spell_mage_pet_scaling::CalcPeriodic, EFFECT_ALL, SPELL_AURA_ANY);
339 OnEffectPeriodic += AuraEffectPeriodicFn(spell_mage_pet_scaling::HandlePeriodic, EFFECT_ALL, SPELL_AURA_ANY);
340 }
