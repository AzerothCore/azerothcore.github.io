72{

105void CasterAI::InitializeAI()
106{
107 CombatAI::InitializeAI();
108
109 m_attackDist = 30.0f;
110 for (SpellVct::iterator itr = spells.begin(); itr != spells.end(); ++itr)
111 if (AISpellInfo[*itr].condition == AICOND_COMBAT && m_attackDist > GetAISpellInfo(*itr)->maxRange)
112 m_attackDist = GetAISpellInfo(*itr)->maxRange;
113 if (m_attackDist == 30.0f)
114 m_attackDist = MELEE_RANGE;
115}

117void CasterAI::EnterCombat(Unit* who)
118{
119 if (spells.empty())
120 return;
121
122 uint32 spell = rand() % spells.size();
123 uint32 count = 0;
124 for (SpellVct::iterator itr = spells.begin(); itr != spells.end(); ++itr, ++count)
125 {
126 if (AISpellInfo[*itr].condition == AICOND_AGGRO)
127 me->CastSpell(who, *itr, false);
128 else if (AISpellInfo[*itr].condition == AICOND_COMBAT)
129 {
130 uint32 cooldown = GetAISpellInfo(*itr)->realCooldown;
131 if (count == spell)
132 {
133 DoCast(spells[spell]);
134 cooldown += me->GetCurrentSpellCastTime(*itr);
135 }
136 events.ScheduleEvent(*itr, cooldown);
137 }
138 }
139}

141void CasterAI::UpdateAI(uint32 diff)
142{
143 if (!UpdateVictim())
144 return;
145
146 events.Update(diff);
147
148 if (me->GetVictim()->HasBreakableByDamageCrowdControlAura(me))
149 {
150 me->InterruptNonMeleeSpells(false);
151 return;
152 }
153
154 if (me->HasUnitState(UNIT_STATE_CASTING))
155 return;
156
157 if (uint32 spellId = events.ExecuteEvent())
158 {
159 DoCast(spellId);
160 uint32 casttime = me->GetCurrentSpellCastTime(spellId);
161 events.ScheduleEvent(spellId, (casttime ? casttime : 500) + GetAISpellInfo(spellId)->realCooldown);
162 }
163}

169ArcherAI::ArcherAI(Creature* c) : CreatureAI(c)
170{
171 if (!me->m_spells[0])
172 LOG_ERROR("entities.unit.ai", "ArcherAI set for creature (entry = {}) with spell1=0. AI will do nothing", me->GetEntry());
173
174 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(me->m_spells[0]);
175 m_minRange = spellInfo ? spellInfo->GetMinRange(false) : 0;
176
177 if (!m_minRange)
178 m_minRange = MELEE_RANGE;
179 me->m_CombatDistance = spellInfo ? spellInfo->GetMaxRange(false) : 0;
180 me->m_SightDistance = me->m_CombatDistance;
181}
182
183void ArcherAI::AttackStart(Unit* who)
184{
185 if (!who)
186 return;
187
188 if (me->IsWithinCombatRange(who, m_minRange))
189 {
190 if (me->Attack(who, true) && !who->IsFlying())
191 me->GetMotionMaster()->MoveChase(who);
192 }
193 else
194 {
195 if (me->Attack(who, false) && !who->IsFlying())
196 me->GetMotionMaster()->MoveChase(who, me->m_CombatDistance);
197 }
198
199 if (who->IsFlying())
200 me->GetMotionMaster()->MoveIdle();
201}

209 DoSpellAttackIfReady(me->m_spells[0]);
210 else
211 DoMeleeAttackIfReady();
212}

298void VehicleAI::CheckConditions(uint32 diff)
299{
300 if (m_ConditionsTimer < diff)
301 {
302 if (!conditions.empty())
303 {
304 if (Vehicle* vehicleKit = me->GetVehicleKit())
305 for (SeatMap::iterator itr = vehicleKit->Seats.begin(); itr != vehicleKit->Seats.end(); ++itr)
306 if (Unit* passenger = ObjectAccessor::GetUnit(*me, itr->second.Passenger.Guid))
307 {
308 if (Player* player = passenger->ToPlayer())
309 {
310 if (!sConditionMgr->IsObjectMeetToConditions(player, me, conditions))
311 {
312 player->ExitVehicle();
313 return; // check other pessanger in next tick
314 }
315 }
316 }
317 }
318 m_ConditionsTimer = VEHICLE_CONDITION_CHECK_TIME;
319 }
320 else
321 m_ConditionsTimer -= diff;
322}
