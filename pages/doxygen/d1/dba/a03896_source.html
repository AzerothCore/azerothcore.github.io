99 void Reset() override
100 {
101 BossAI::Reset();
102 me->CastSpell(me, SPELL_CLEAR_CONSUMING_MADNESS, true);
103 me->CastSpell(me, SPELL_DUAL_WIELD, true);
104 me->SetStandState(UNIT_STAND_STATE_KNEEL);
105 me->LoadEquipment(0, true);
106 me->SetReactState(REACT_PASSIVE);
107 }
108
109 void InitializeAI() override
110 {
111 BossAI::InitializeAI();
112 SummonChannelers();
113 }
114
115 void JustReachedHome() override
116 {
117 BossAI::JustReachedHome();
118 SummonChannelers();
119 }
120
121 void SummonChannelers()
122 {
123 me->ApplySpellImmune(0, IMMUNITY_MECHANIC, MECHANIC_BANISH, false);
124 me->CastSpell(me, SPELL_BANISH, true);
125 me->ApplySpellImmune(0, IMMUNITY_MECHANIC, MECHANIC_BANISH, true);
126
127 summons.DespawnAll();
128 for (uint8 i = 0; i < MAX_CHANNELERS; ++i)
129 me->SummonCreature(NPC_GREYHEART_SPELLBINDER, channelersPos[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 5000);
130 }
131
132 void MoveInLineOfSight(Unit* /*who*/) override { }
133
134 void JustSummoned(Creature* summon) override
135 {
136 summons.Summon(summon);
137 }
138
139 void SummonedCreatureDies(Creature* summon, Unit*) override
140 {
141 me->SetInCombatWithZone();
142 summons.Despawn(summon);
143 if (summon->GetEntry() == NPC_GREYHEART_SPELLBINDER)
144 if (!summons.HasEntry(NPC_GREYHEART_SPELLBINDER))
145 {
146 me->RemoveAllAuras();
147 me->LoadEquipment();
148 me->SetReactState(REACT_AGGRESSIVE);
149 me->SetStandState(UNIT_STAND_STATE_STAND);
150 Talk(SAY_AGGRO);
151
152 events.ScheduleEvent(EVENT_SPELL_BERSERK, 600000);
153 events.ScheduleEvent(EVENT_HEALTH_CHECK, 1000);
154 events.ScheduleEvent(EVENT_SWITCH_TO_DEMON, 55000);
155 events.ScheduleEvent(EVENT_SPELL_WHIRLWIND, 10000);
156 }
157 }
158
159 void KilledUnit(Unit* /*victim*/) override
160 {
161 if (events.GetNextEventTime(EVENT_KILL_TALK) == 0)
162 {
163 Talk(me->GetDisplayId() != me->GetNativeDisplayId() ? SAY_DEMON_SLAY : SAY_NIGHTELF_SLAY);
164 events.ScheduleEvent(EVENT_KILL_TALK, 6000);
165 }
166 }
167
168 void JustDied(Unit* killer) override
169 {
170 me->CastSpell(me, SPELL_CLEAR_CONSUMING_MADNESS, true);
171 Talk(SAY_DEATH);
172 BossAI::JustDied(killer);
173 }
174
175 void EnterCombat(Unit* who) override
176 {
177 BossAI::EnterCombat(who);
178 me->SetStandState(UNIT_STAND_STATE_KNEEL);
179 }
180
181 void AttackStart(Unit* who) override
182 {
183 if (who && me->Attack(who, true))
184 me->GetMotionMaster()->MoveChase(who, me->GetDisplayId() == me->GetNativeDisplayId() ? 0.0f : 25.0f);
185 }
186
187 void UpdateAI(uint32 diff) override
188 {
189 if (!UpdateVictim())
190 return;
191
192 events.Update(diff);
193 if (me->HasUnitState(UNIT_STATE_CASTING))
194 return;
195
196 switch (events.ExecuteEvent())
197 {
198 case EVENT_SPELL_BERSERK:
199 me->CastSpell(me, SPELL_BERSERK, true);
200 break;
201 case EVENT_HEALTH_CHECK:
202 if (me->HealthBelowPct(15))
203 {
204 if (me->GetDisplayId() != me->GetNativeDisplayId())
205 {
206 DoResetThreat();
207 me->LoadEquipment();
208 me->RemoveAurasDueToSpell(SPELL_METAMORPHOSIS);
209 events.ScheduleEvent(EVENT_SPELL_WHIRLWIND, 10000);
210 }
211 events.CancelEvent(EVENT_SWITCH_TO_DEMON);
212 events.CancelEvent(EVENT_SPELL_INSIDIOUS_WHISPER);
213 events.DelayEvents(10000);
214 events.ScheduleEvent(EVENT_SUMMON_DEMON, 4000);
215 events.ScheduleEvent(EVENT_RESTORE_FIGHT, 6000);
216 me->SetStandState(UNIT_STAND_STATE_KNEEL);
217 me->SetReactState(REACT_PASSIVE);
218 me->GetMotionMaster()->Clear();
219 me->StopMoving();
220 Talk(SAY_FINAL_FORM);
221 break;
222 }
223 events.ScheduleEvent(EVENT_HEALTH_CHECK, 1000);
224 break;
225 case EVENT_SWITCH_TO_DEMON:
226 DoResetThreat();
227 Talk(SAY_SWITCH_TO_DEMON);
228 me->LoadEquipment(0, true);
229 me->GetMotionMaster()->MoveChase(me->GetVictim(), 25.0f);
230 me->CastSpell(me, SPELL_METAMORPHOSIS, true);
231
232 events.CancelEvent(EVENT_SPELL_WHIRLWIND);
233 events.ScheduleEvent(EVENT_SPELL_INSIDIOUS_WHISPER, 25000);
234 events.ScheduleEvent(EVENT_SWITCH_TO_ELF, 60000);
235 break;
236 case EVENT_SWITCH_TO_ELF:
237 DoResetThreat();
238 me->LoadEquipment();
239 me->GetMotionMaster()->MoveChase(me->GetVictim(), 0.0f);
240 me->RemoveAurasDueToSpell(SPELL_METAMORPHOSIS);
241 events.ScheduleEvent(EVENT_SWITCH_TO_DEMON, 55000);
242 events.ScheduleEvent(EVENT_SPELL_WHIRLWIND, 10000);
243 break;
244 case EVENT_SPELL_WHIRLWIND:
245 me->CastSpell(me, SPELL_WHIRLWIND, false);
246 events.ScheduleEvent(EVENT_SPELL_WHIRLWIND, 27000);
247 break;
248 case EVENT_SPELL_INSIDIOUS_WHISPER:
249 Talk(SAY_INNER_DEMONS);
250 me->CastCustomSpell(SPELL_INSIDIOUS_WHISPER, SPELLVALUE_MAX_TARGETS, 5, me, false);
251 break;
252 case EVENT_SUMMON_DEMON:
253 me->CastSpell(me, SPELL_SUMMON_SHADOW_OF_LEOTHERAS, true);
254 break;
255 case EVENT_RESTORE_FIGHT:
256 me->SetStandState(UNIT_STAND_STATE_STAND);
257 me->SetReactState(REACT_AGGRESSIVE);
258 me->GetMotionMaster()->MoveChase(me->GetVictim());
259 break;
260 }
261
262 if (me->GetDisplayId() == me->GetNativeDisplayId())
263 DoMeleeAttackIfReady();
264 else if (me->isAttackReady(BASE_ATTACK))
265 {
266 me->CastSpell(me->GetVictim(), SPELL_CHAOS_BLAST, false);
267 me->setAttackTimer(BASE_ATTACK, 2000);
268 }
269 }
270 };
