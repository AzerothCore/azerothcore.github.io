61 void ReadCompleted(size_type bytes) { _rpos += bytes; }

63
64 [[nodiscard]] size_type GetActiveSize() const { return _wpos - _rpos; }
65 [[nodiscard]] size_type GetRemainingSpace() const { return _storage.size() - _wpos; }
66 [[nodiscard]] size_type GetBufferSize() const { return _storage.size(); }
67
68 // Discards inactive data
69 void Normalize()
70 {
71 if (_rpos)
72 {
73 if (_rpos != _wpos)
74 {
75 memmove(GetBasePointer(), GetReadPointer(), GetActiveSize());
76 }
77
78 _wpos -= _rpos;
79 _rpos = 0;
80 }
81 }
82
83 // Ensures there's "some" free space, make sure to call Normalize() before this
84 void EnsureFreeSpace()
85 {
86 // resize buffer if it's already full
87 if (GetRemainingSpace() == 0)
88 {
89 _storage.resize(_storage.size() * 3 / 2);
90 }
91 }
92
93 void Write(void const* data, std::size_t size)
94 {
95 if (size)
96 {
97 memcpy(GetWritePointer(), data, size);
98 WriteCompleted(size);
99 }
100 }
101
102 std::vector<uint8>&& Move()
103 {
104 _wpos = 0;
105 _rpos = 0;
106
107 return std::move(_storage);
108 }

122 MessageBuffer& operator=(MessageBuffer&& right) noexcept
123 {

127 _rpos = right._rpos;
128 _storage = right.Move();
129 }
130
131 return *this;
132 }
133
134private:
135 size_type _wpos{0};
136 size_type _rpos{0};
137 std::vector<uint8> _storage;
138};
