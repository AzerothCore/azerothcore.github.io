206class achievement_tilted : public AchievementCriteriaScript
207{
208public:
209 achievement_tilted() : AchievementCriteriaScript("achievement_tilted") {}
210
211 bool OnCheck(Player* player, Unit* /*target*/, uint32 /*criteria_id*/) override
212 {
213 if (!player)
214 return false;
215
216 uint32 areaid = player->GetAreaId();
217 bool checkArea = areaid == AREA_ARGENT_TOURNAMENT_FIELDS ||
218 areaid == AREA_RING_OF_ASPIRANTS ||
219 areaid == AREA_RING_OF_ARGENT_VALIANTS ||
220 areaid == AREA_RING_OF_ALLIANCE_VALIANTS ||
221 areaid == AREA_RING_OF_HORDE_VALIANTS ||
222 areaid == AREA_RING_OF_CHAMPIONS;
223
224 return checkArea && player->duel && player->duel->IsMounted;
225 }
