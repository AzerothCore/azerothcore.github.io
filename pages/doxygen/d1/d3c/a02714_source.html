68
69 void Reset() override
70 {
71 BossAI::Reset();
72 me->CastSpell(me, SPELL_DUAL_WIELD, true);
73 }
74
75 void DamageTaken(Unit* who, uint32& damage, DamageEffectType, SpellSchoolMask) override
76 {
77 if (me->GetReactState() == REACT_PASSIVE && (!who || who->GetEntry() != NPC_MADRIGOSA))
78 {
79 if (who)
80 Unit::Kill(me, who);
81 damage = 0;

83 }
84
85 void EnterCombat(Unit* who) override
86 {
87 if (who->GetEntry() == NPC_MADRIGOSA)
88 return;
89
90 Talk(YELL_AGGRO);
91 BossAI::EnterCombat(who);
92
93 events.ScheduleEvent(EVENT_SPELL_SLASH, 11000);
94 events.ScheduleEvent(EVENT_SPELL_STOMP, 30000);
95 events.ScheduleEvent(EVENT_SPELL_BURN, 45000);
96 events.ScheduleEvent(EVENT_SPELL_BERSERK, 360000);
97 }
98
99 void KilledUnit(Unit* victim) override
100 {
101 if (victim->GetTypeId() == TYPEID_PLAYER && roll_chance_i(50))
102 Talk(YELL_KILL);
103 }
104
105 void JustDied(Unit* killer) override
106 {
107 BossAI::JustDied(killer);
108 Talk(YELL_DEATH);
109
110 me->CastSpell(me, SPELL_SUMMON_BRUTALLUS_DEATH_CLOUD, true);
111 if (Creature* madrigosa = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_MADRIGOSA)))
112 madrigosa->AI()->DoAction(ACTION_SPAWN_FELMYST);
113 }
114
115 void AttackStart(Unit* who) override
116 {
117 if (who->GetEntry() == NPC_MADRIGOSA)
118 return;
119 BossAI::AttackStart(who);
120 }
121
122 void UpdateAI(uint32 diff) override
123 {
124 if (!UpdateVictim())
125 return;
126
127 events.Update(diff);
128 if (me->HasUnitState(UNIT_STATE_CASTING))
129 return;
130
131 switch (events.ExecuteEvent())
132 {
133 case EVENT_SPELL_SLASH:
134 me->CastSpell(me->GetVictim(), SPELL_METEOR_SLASH, false);
135 events.ScheduleEvent(EVENT_SPELL_SLASH, 10000);
136 break;
137 case EVENT_SPELL_STOMP:
138 me->CastSpell(me->GetVictim(), SPELL_STOMP, false);
139 Talk(YELL_LOVE);
140 events.ScheduleEvent(EVENT_SPELL_STOMP, 30000);
141 break;
142 case EVENT_SPELL_BURN:
143 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100.0f, true, -SPELL_BURN_DAMAGE))
144 me->CastSpell(target, SPELL_BURN, false);
145 events.ScheduleEvent(EVENT_SPELL_BURN, 60000);
146 break;
147 case EVENT_SPELL_BERSERK:
148 me->CastSpell(me, SPELL_BERSERK, true);
149 Talk(YELL_BERSERK);
150 break;
151 }
152
153 DoMeleeAttackIfReady();
154 }
155 };
