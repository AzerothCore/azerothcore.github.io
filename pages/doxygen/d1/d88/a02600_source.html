87 struct boss_midnightAI : public BossAI
88 {
89 boss_midnightAI(Creature* creature) : BossAI(creature, DATA_ATTUMEN) { }
90
91 void Reset() override
92 {
93 BossAI::Reset();
94 me->SetVisible(true);
95 _healthPct = 100.0f;
96 }
97
98 void EnterCombat(Unit* who) override
99 {
100 BossAI::EnterCombat(who);
101 events.ScheduleEvent(EVENT_CHECK_HEALTH_95, 0);
102 events.ScheduleEvent(EVENT_SPELL_KNOCKDOWN, 6000);
103 DoZoneInCombat();
104 }
105
106 void KilledUnit(Unit* /*victim*/) override
107 {
108 if (Creature* attumen = summons.GetCreatureWithEntry(NPC_ATTUMEN_THE_HUNTSMAN))
109 attumen->AI()->Talk(SAY_ATTUMEN_MIDNIGHT_KILL);
110 }
111
112 void JustSummoned(Creature* summon) override
113 {
114 summons.Summon(summon);
115 summon->SetInCombatWithZone();
116
117 if (summon->GetEntry() == NPC_ATTUMEN_THE_HUNTSMAN_MOUNTED)
118 {
119 summon->SetHealth(summon->CountPctFromMaxHealth(_healthPct));
120 summon->CastSpell(summon, SPELL_SPAWN_SMOKE2, true);
121 }
122 else
123 summon->CastSpell(summon, SPELL_SPAWN_SMOKE1, true);
124 }
125
126 void SetData(uint32 type, uint32 /*data*/) override
127 {
128 if (type == DATA_ATTUMEN_READY)
129 events.ScheduleEvent(EVENT_SUMMON_ATTUMEN_MOUNTED, 0);
130 }
131
132 void SummonedCreatureDies(Creature* summon, Unit* /*killer*/) override
133 {
134 if (summon->GetEntry() == NPC_ATTUMEN_THE_HUNTSMAN_MOUNTED)
135 {
136 summons.clear();
137 Unit::Kill(me, me);
138 }
139 }
140
141 void UpdateAI(uint32 diff) override
142 {
143 if (!UpdateVictim())
144 return;
145
146 events.Update(diff);
147 if (me->HasUnitState(UNIT_STATE_CASTING))
148 return;
149
150 switch (events.ExecuteEvent())
151 {
152 case EVENT_CHECK_HEALTH_95:
153 if (me->HealthBelowPct(96))
154 {
155 me->CastSpell(me, SPELL_SUMMON_ATTUMEN, true);
156 events.ScheduleEvent(EVENT_CHECK_HEALTH_25, 0);
157 break;
158 }
159 events.ScheduleEvent(EVENT_CHECK_HEALTH_95, 500);
160 break;
161 case EVENT_CHECK_HEALTH_25:
162 if (me->HealthBelowPct(25))
163 {
164 Talk(SAY_MIDNIGHT_EMOTE);
165 me->CastSpell(me, SPELL_MOUNT_TARGET_ATTUMEN, true);
166 break;
167 }
168 events.ScheduleEvent(EVENT_CHECK_HEALTH_25, 500);
169 break;
170 case EVENT_SPELL_KNOCKDOWN:
171 me->CastSpell(me->GetVictim(), SPELL_KNOCKDOWN, false);
172 events.ScheduleEvent(EVENT_SPELL_KNOCKDOWN, 20000);
173 break;
174 case EVENT_SUMMON_ATTUMEN_MOUNTED:
175 if (Creature* attumen = summons.GetCreatureWithEntry(NPC_ATTUMEN_THE_HUNTSMAN))
176 {
177 _healthPct = std::max<float>(me->GetHealthPct(), attumen->GetHealthPct());
178 attumen->DespawnOrUnsummon();
179 }
180
181 me->CastSpell(me, SPELL_SUMMON_ATTUMEN_MOUNTED, true);
182 me->SetVisible(false);
183 break;
184 }
185
186 if (me->IsVisible())
187 DoMeleeAttackIfReady();
188 }
189
190 bool CheckEvadeIfOutOfCombatArea() const override
191 {
192 return me->GetHomePosition().GetExactDist2d(me) > 50.0f || me->GetPositionZ() > 60.0f;
193 }
194
195 private:
196 float _healthPct;
197 };
198
199 CreatureAI* GetAI(Creature* creature) const override
200 {
201 return GetKarazhanAI<boss_midnightAI>(creature);
202 }
