108 void InitializeGuests()
109 {
110 if (!me->IsAlive())
111 return;
112
113 if (_activeGuests == 0)
114 {
115 _activeGuests |= 0x3F;
116 uint8 rand1 = RAND(0x01, 0x02, 0x04);
117 uint8 rand2 = RAND(0x08, 0x10, 0x20);
118 _activeGuests &= ~(rand1 | rand2);
119 }
120
121 for (uint8 i = 0; i < MAX_GUEST_COUNT; ++i)
122 if ((1 << i) & _activeGuests)
123 me->SummonCreature(GuestEntries[i], GuestsPosition[summons.size()], TEMPSUMMON_MANUAL_DESPAWN);
124
125 _events2.Reset();
126 _events2.ScheduleEvent(EVENT_GUEST_TALK, 10000);
127 }
128
129 void Reset() override
130 {
131 BossAI::Reset();
132 me->CastSpell(me, SPELL_DUAL_WIELD, true);
133 }

135 void EnterCombat(Unit* who) override
136 {
137 BossAI::EnterCombat(who);
138 Talk(SAY_AGGRO);
139
140 events.ScheduleEvent(EVENT_SPELL_VANISH, 30000);
141 events.ScheduleEvent(EVENT_SPELL_BLIND, 20000);
142 events.ScheduleEvent(EVENT_SPELL_GOUGE, 13000);
143 events.ScheduleEvent(EVENT_CHECK_HEALTH, 5000);
144 events.ScheduleEvent(EVENT_SPELL_ENRAGE, 600000);

146 _events2.Reset();

149 }

174 Creature* GetRandomGuest()
175 {
176 std::list<Creature*> guestList;
177 for (SummonList::const_iterator i = summons.begin(); i != summons.end(); ++i)
178 if (Creature* summon = ObjectAccessor::GetCreature(*me, *i))
179 guestList.push_back(summon);
180
181 return Acore::Containers::SelectRandomContainerElement(guestList);
182 }
183
184 void UpdateAI(uint32 diff) override
185 {
186 _events2.Update(diff);
187 switch (_events2.ExecuteEvent())
188 {
189 case EVENT_GUEST_TALK:
190 if (Creature* guest = GetRandomGuest())
191 guest->AI()->Talk(SAY_GUEST);
192 _events2.ScheduleEvent(EVENT_GUEST_TALK2, 5000);
193 break;
194 case EVENT_GUEST_TALK2:
195 Talk(SAY_OUT_OF_COMBAT);
196 _events2.ScheduleEvent(EVENT_GUEST_TALK, urand(60000, 120000));
197 break;
198 }
199
200 if (!UpdateVictim())
201 return;
202
203 events.Update(diff);
204 if (me->HasUnitState(UNIT_STATE_CASTING))
205 return;
206
207 switch (events.ExecuteEvent())
208 {
209 case EVENT_CHECK_HEALTH:
210 if (me->HealthBelowPct(31))
211 {
212 me->CastSpell(me, SPELL_FRENZY, true);
213 break;
214 }
215 events.ScheduleEvent(EVENT_CHECK_HEALTH, 1000);
216 break;
217 case EVENT_SPELL_ENRAGE:
218 me->CastSpell(me, SPELL_BERSERK, true);
219 break;
220 case EVENT_SPELL_BLIND:
221 if (Unit* target = SelectTarget(SelectTargetMethod::MaxThreat, 1, 10.0f, true))
222 me->CastSpell(target, SPELL_BLIND, false);
223 events.ScheduleEvent(EVENT_SPELL_BLIND, urand(25000, 40000));
224 break;
225 case EVENT_SPELL_GOUGE:
226 me->CastSpell(me->GetVictim(), SPELL_GOUGE, false);
227 events.ScheduleEvent(EVENT_SPELL_GOUGE, urand(25000, 40000));
228 return;
229 case EVENT_SPELL_VANISH:
230 events.DelayEvents(9000);
231 events.SetPhase(1);
232 me->CastSpell(me, SPELL_VANISH, false);
233 events.ScheduleEvent(EVENT_SPELL_VANISH, 30000);
234 events.ScheduleEvent(EVENT_SPELL_GARROTE, urand(5000, 7000));
235 return;
236 case EVENT_SPELL_GARROTE:
237 Talk(SAY_SPECIAL);
238 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
239 target->CastSpell(target, SPELL_GARROTE, true);
240 me->CastSpell(me, SPELL_VANISH_TELEPORT, false);
241 events.SetPhase(0);
242 break;
243 }
244
245 // Xinef: not in vanish
246 if (events.GetPhaseMask() == 0)
247 DoMeleeAttackIfReady();
248 }
249
250 private:
251 EventMap _events2;
252 uint8 _activeGuests;
253 };

257 return GetKarazhanAI<boss_moroesAI>(creature);
