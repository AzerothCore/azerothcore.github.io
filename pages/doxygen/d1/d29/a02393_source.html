164 void UpdateAI(uint32 diff) override
165 {
166 // I have a feeling this isn't blizzlike, but owell, I'm only passing by and cleaning up.
167 if (!_hasAura)
168 {
169 for (uint8 i = 0; i < MAX_SPELL_PAIRS; ++i)
170 if (_auraPairs[i].npcEntry == me->GetEntry())
171 DoCast(me, _auraPairs[i].spellId);
172
173 _hasAura = true;
174 }
175
176 if (!UpdateVictim())
177 return;
178
179 events.Update(diff);
180
181 if (me->HasUnitState(UNIT_STATE_CASTING))
182 return;
183
184 while (uint32 eventId = events.ExecuteEvent())
185 {
186 switch (eventId)
187 {
188 case EVENT_CHARGE_TARGET:
189 DoCastVictim(SPELL_CHARGE);
190 events.ScheduleEvent(EVENT_CHARGE, urand(10 * IN_MILLISECONDS, 25 * IN_MILLISECONDS));
191 break;
192 case EVENT_CLEAVE:
193 DoCastVictim(SPELL_CLEAVE);
194 events.ScheduleEvent(EVENT_CLEAVE, urand(10 * IN_MILLISECONDS, 16 * IN_MILLISECONDS));
195 break;
196 case EVENT_DEMORALIZING_SHOUT:
197 DoCast(me, SPELL_DEMORALIZING_SHOUT);
198 events.ScheduleEvent(EVENT_DEMORALIZING_SHOUT, urand(10 * IN_MILLISECONDS, 15 * IN_MILLISECONDS));
199 break;
200 case EVENT_WHIRLWIND:
201 DoCast(me, SPELL_WHIRLWIND);
202 events.ScheduleEvent(EVENT_WHIRLWIND, urand(10 * IN_MILLISECONDS, 25 * IN_MILLISECONDS));
203 break;
204 case EVENT_ENRAGE:
205 DoCast(me, SPELL_ENRAGE);
206 events.ScheduleEvent(EVENT_ENRAGE, urand(10 * IN_MILLISECONDS, 30 * IN_MILLISECONDS));
207 break;
208 case EVENT_CHECK_RESET:
209 {
210 Position const& _homePosition = me->GetHomePosition();
211 if (me->GetDistance2d(_homePosition.GetPositionX(), _homePosition.GetPositionY()) > 50.0f)
212 {
213 ScriptedAI::EnterEvadeMode();
214 return;
215 }
216 events.ScheduleEvent(EVENT_CHECK_RESET, 5000);
217 break;
218 }
219 }
220 }
221
222 DoMeleeAttackIfReady();
223 }

225 private:
226 EventMap events;
227 bool _hasAura;
228 bool _attacked;
229 };
