148 void ResetMobs()
149 {
150 std::list<Creature*> templist;
151 float x, y, z;
152 me->GetPosition(x, y, z);
153
154 {
155 CellCoord pair(Acore::ComputeCellCoord(x, y));
156 Cell cell(pair);
157 cell.SetNoCreate();
158
159 Acore::AllFriendlyCreaturesInGrid check(me);
160 Acore::CreatureListSearcher<Acore::AllFriendlyCreaturesInGrid> searcher(me, templist, check);
161
162 TypeContainerVisitor<Acore::CreatureListSearcher<Acore::AllFriendlyCreaturesInGrid>, GridTypeMapContainer> cSearcher(searcher);
163
164 cell.Visit(pair, cSearcher, *(me->GetMap()), *me, me->GetGridActivationRange());
165 }
166
167 if (templist.empty())
168 return;
169
170 for (std::list<Creature*>::const_iterator i = templist.begin(); i != templist.end(); ++i)
171 if ((*i) && me->GetGUID() != (*i)->GetGUID() && me->IsWithinDistInMap((*i), 25))
172 (*i)->AI()->Reset();
173 }

354 void UpdateAI(uint32 diff) override
355 {
356 if (waitTimer && inMove)
357 {
358 if (waitTimer <= diff)
359 {
360 (*me).GetMotionMaster()->MovementExpired();
361 (*me).GetMotionMaster()->MovePoint(MovePhase, NalorakkWay[MovePhase][0], NalorakkWay[MovePhase][1], NalorakkWay[MovePhase][2]);
362 waitTimer = 0;
363 }
364 else waitTimer -= diff;
365 }
366
367 if (!UpdateVictim())
368 return;
369
370 if (Berserk_Timer <= diff)
371 {
372 DoCast(me, SPELL_BERSERK, true);
373 Talk(SAY_BERSERK);
374 Berserk_Timer = 600000;
375 }
376 else Berserk_Timer -= diff;
377
378 if (ShapeShift_Timer <= diff)
379 {
380 if (inBearForm)
381 {
382 // me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 1, 5122);
383 Talk(SAY_SHIFTEDTOTROLL);
384 me->RemoveAurasDueToSpell(SPELL_BEARFORM);
385 Surge_Timer = urand(15000, 20000);
386 BrutalSwipe_Timer = urand(7000, 12000);
387 Mangle_Timer = urand(10000, 15000);
388 ShapeShift_Timer = urand(45000, 50000);
389 inBearForm = false;
390 }
391 else
392 {
393 // me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 1, 0);
394 Talk(SAY_SHIFTEDTOBEAR);
395 DoCast(me, SPELL_BEARFORM, true);
396 LaceratingSlash_Timer = 2000; // dur 18s
397 RendFlesh_Timer = 3000; // dur 5s
398 DeafeningRoar_Timer = urand(5000, 10000); // dur 2s
399 ShapeShift_Timer = urand(20000, 25000); // dur 30s
400 inBearForm = true;
401 }
402 }
403 else ShapeShift_Timer -= diff;
404
405 if (!inBearForm)
406 {
407 if (BrutalSwipe_Timer <= diff)
408 {
409 DoCastVictim(SPELL_BRUTALSWIPE);
410 BrutalSwipe_Timer = urand(7000, 12000);
411 }
412 else BrutalSwipe_Timer -= diff;
413
414 if (Mangle_Timer <= diff)
415 {
416 if (me->GetVictim() && !me->GetVictim()->HasAura(SPELL_MANGLEEFFECT))
417 {
418 DoCastVictim(SPELL_MANGLE);
419 Mangle_Timer = 1000;
420 }
421 else Mangle_Timer = urand(10000, 15000);
422 }
423 else Mangle_Timer -= diff;
424
425 if (Surge_Timer <= diff)
426 {
427 Talk(SAY_SURGE);
428 Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 45, true);
429 if (target)
430 DoCast(target, SPELL_SURGE);
431 Surge_Timer = urand(15000, 20000);
432 }
433 else Surge_Timer -= diff;
434 }
435 else
436 {
437 if (LaceratingSlash_Timer <= diff)
438 {
439 DoCastVictim(SPELL_LACERATINGSLASH);
440 LaceratingSlash_Timer = urand(18000, 23000);
441 }
442 else LaceratingSlash_Timer -= diff;
443
444 if (RendFlesh_Timer <= diff)
445 {
446 DoCastVictim(SPELL_RENDFLESH);
447 RendFlesh_Timer = urand(5000, 10000);
448 }
449 else RendFlesh_Timer -= diff;
450
451 if (DeafeningRoar_Timer <= diff)
452 {
453 DoCastVictim(SPELL_DEAFENINGROAR);
454 DeafeningRoar_Timer = urand(15000, 20000);
455 }
456 else DeafeningRoar_Timer -= diff;
457 }
458
459 DoMeleeAttackIfReady();
460 }
