60bool PathGenerator::CalculatePath(float x, float y, float z, float destX, float destY, float destZ, bool forceDest)
61{
62 if (!Acore::IsValidMapCoord(destX, destY, destZ) || !Acore::IsValidMapCoord(x, y, z))
63 return false;
64
65 METRIC_DETAILED_EVENT("mmap_events", "CalculatePath", "");
66
67 G3D::Vector3 dest(destX, destY, destZ);
68 SetEndPosition(dest);
69
70 G3D::Vector3 start(x, y, z);
71 SetStartPosition(start);
72
73 _forceDestination = forceDest;
74
75 // make sure navMesh works - we can run on map w/o mmap
76 // check if the start and end point have a .mmtile loaded (can we pass via not loaded tile on the way?)
77 Unit const* _sourceUnit = _source->ToUnit();
78 if (!_navMesh || !_navMeshQuery || (_sourceUnit && _sourceUnit->HasUnitState(UNIT_STATE_IGNORE_PATHFINDING)) ||
79 !HaveTile(start) || !HaveTile(dest))
80 {
81 BuildShortcut();
82 _type = PathType(PATHFIND_NORMAL | PATHFIND_NOT_USING_PATH);
83 return true;
84 }
85
86 UpdateFilter();
87
88 BuildPolyPath(start, dest);
89 return true;
90}

92dtPolyRef PathGenerator::GetPathPolyByPosition(dtPolyRef const* polyPath, uint32 polyPathSize, float const* point, float* distance) const
93{
94 if (!polyPath || !polyPathSize)
95 return INVALID_POLYREF;
96
97 dtPolyRef nearestPoly = INVALID_POLYREF;
98 float minDist = FLT_MAX;
99
100 for (uint32 i = 0; i < polyPathSize; ++i)
101 {
102 float closestPoint[VERTEX_SIZE];
103 if (dtStatusFailed(_navMeshQuery->closestPointOnPoly(polyPath[i], point, closestPoint, nullptr)))
104 continue;
105
106 float d = dtVdistSqr(point, closestPoint);
107 if (d < minDist)
108 {
109 minDist = d;
110 nearestPoly = polyPath[i];
111 }
112
113 if (minDist < 1.0f) // shortcut out - close enough for us
114 {
115 break;
116 }
117 }
118
119 if (distance)
120 {
121 *distance = dtMathSqrtf(minDist);
122 }
123
124 return (minDist < 3.0f) ? nearestPoly : INVALID_POLYREF;
125}

127dtPolyRef PathGenerator::GetPolyByLocation(float const* point, float* distance) const
128{
129 // first we check the current path
130 // if the current path doesn't contain the current poly,
131 // we need to use the expensive navMesh.findNearestPoly
132 dtPolyRef polyRef = GetPathPolyByPosition(_pathPolyRefs, _polyLength, point, distance);
133 if (polyRef != INVALID_POLYREF)
134 return polyRef;
135
136 // we don't have it in our old path
137 // try to get it by findNearestPoly()
138 // first try with low search box
139 float extents[VERTEX_SIZE] = { 3.0f, 5.0f, 3.0f }; // bounds of poly search area
140 float closestPoint[VERTEX_SIZE] = { 0.0f, 0.0f, 0.0f };
141 if (dtStatusSucceed(_navMeshQuery->findNearestPoly(point, extents, &_filter, &polyRef, closestPoint)) && polyRef != INVALID_POLYREF)
142 {
143 *distance = dtVdist(closestPoint, point);
144 return polyRef;
145 }
146
147 // still nothing ..
148 // try with bigger search box
149 // Note that the extent should not overlap more than 128 polygons in the navmesh (see dtNavMeshQuery::findNearestPoly)
150 extents[1] = 50.0f;
151
152 if (dtStatusSucceed(_navMeshQuery->findNearestPoly(point, extents, &_filter, &polyRef, closestPoint)) && polyRef != INVALID_POLYREF)
153 {
154 *distance = dtVdist(closestPoint, point);
155 return polyRef;
156 }
157
158 *distance = FLT_MAX;
159 return INVALID_POLYREF;
160}

162void PathGenerator::BuildPolyPath(G3D::Vector3 const& startPos, G3D::Vector3 const& endPos)
163{
164 // *** getting start/end poly logic ***
165
166 float distToStartPoly, distToEndPoly;
167 float startPoint[VERTEX_SIZE] = { startPos.y, startPos.z, startPos.x };
168 float endPoint[VERTEX_SIZE] = { endPos.y, endPos.z, endPos.x };
169
170 dtPolyRef startPoly = GetPolyByLocation(startPoint, &distToStartPoly);
171 dtPolyRef endPoly = GetPolyByLocation(endPoint, &distToEndPoly);
172
173 _type = PathType(PATHFIND_NORMAL);
174
175 Creature const* creature = _source->ToCreature();
176
177 // we have a hole in our mesh
178 // make shortcut path and mark it as NOPATH ( with flying and swimming exception )
179 // its up to caller how he will use this info
180 if (startPoly == INVALID_POLYREF || endPoly == INVALID_POLYREF)
181 {
182 BuildShortcut();
183
184 bool canSwim = creature ? creature->CanSwim() : true;
185 bool path = creature ? creature->CanFly() : true;
186 bool waterPath = IsWaterPath(_pathPoints);
187 if (path || (waterPath && canSwim))
188 {
189 _type = PathType(PATHFIND_NORMAL | PATHFIND_NOT_USING_PATH);
190 return;
191 }
192
193 // raycast doesn't need endPoly to be valid
194 if (!_useRaycast)
195 {
196 _type = PATHFIND_NOPATH;
197 return;
198 }
199 }
200
201 // we may need a better number here
202 bool startFarFromPoly = distToStartPoly > 7.0f;
203 bool endFarFromPoly = distToEndPoly > 7.0f;
204
205 // create a shortcut if the path begins or end too far
206 // away from the desired path points.
207 // swimming creatures should not use a shortcut
208 // because exiting the water must be done following a proper path
209 // we just need to remove/normalize paths between 2 adjacent points
210 if (startFarFromPoly || endFarFromPoly)
211 {
212 bool buildShotrcut = false;
213
214 bool isUnderWaterStart = _source->GetMap()->IsUnderWater(_source->GetPhaseMask(), startPos.x, startPos.y, startPos.z, _source->GetCollisionHeight());
215 bool isUnderWaterEnd = _source->GetMap()->IsUnderWater(_source->GetPhaseMask(), endPos.x, endPos.y, endPos.z, _source->GetCollisionHeight());
216 bool isFarUnderWater = startFarFromPoly ? isUnderWaterStart : isUnderWaterEnd;
217
218 Unit const* _sourceUnit = _source->ToUnit();
219
220 if (_sourceUnit)
221 {
222 bool isUnderWater = (_sourceUnit->CanSwim() && isUnderWaterStart && isUnderWaterEnd) || (isFarUnderWater && _useRaycast);
223
224 if (isUnderWater || _sourceUnit->CanFly() || (_sourceUnit->IsFalling() && endPos.z < startPos.z))
225 {
226 buildShotrcut = true;
227 }
228 }
229
230 if (buildShotrcut)
231 {
232 BuildShortcut();
233 _type = PathType(PATHFIND_NORMAL | PATHFIND_NOT_USING_PATH);
234
235 AddFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
236
237 return;
238 }
239
240 if (!isFarUnderWater)
241 {
242 float closestPoint[VERTEX_SIZE];
243 // we may want to use closestPointOnPolyBoundary instead
244 if (dtStatusSucceed(_navMeshQuery->closestPointOnPoly(endPoly, endPoint, closestPoint, nullptr)))
245 {
246 dtVcopy(endPoint, closestPoint);
247 SetActualEndPosition(G3D::Vector3(endPoint[2], endPoint[0], endPoint[1]));
248 }
249
250 _type = PathType(PATHFIND_INCOMPLETE);
251
252 AddFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
253 }
254 }
255
256 // *** poly path generating logic ***
257
258 // start and end are on same polygon
259 // handle this case as if they were 2 different polygons, building a line path split in some few points
260 if (startPoly == endPoly && !_useRaycast)
261 {
262 _pathPolyRefs[0] = startPoly;
263 _polyLength = 1;
264
265 if (startFarFromPoly || endFarFromPoly)
266 {
267 _type = PathType(PATHFIND_INCOMPLETE);
268
269 AddFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
270 }
271 else
272 _type = PATHFIND_NORMAL;
273
274 BuildPointPath(startPoint, endPoint);
275 return;
276 }
277
278 // look for startPoly/endPoly in current path
280 bool startPolyFound = false;
281 bool endPolyFound = false;
282 uint32 pathStartIndex = 0;
283 uint32 pathEndIndex = 0;
284
285 if (_polyLength)
286 {
287 for (; pathStartIndex < _polyLength; ++pathStartIndex)
288 {
289 // here to catch few bugs
290 if (_pathPolyRefs[pathStartIndex] == INVALID_POLYREF)
291 {
292 break;
293 }
294
295 if (_pathPolyRefs[pathStartIndex] == startPoly)
296 {
297 startPolyFound = true;
298 break;
299 }
300 }
301
302 for (pathEndIndex = _polyLength - 1; pathEndIndex > pathStartIndex; --pathEndIndex)
303 {
304 if (_pathPolyRefs[pathEndIndex] == endPoly)
305 {
306 endPolyFound = true;
307 break;
308 }
309 }
310 }
311
312 if (startPolyFound && endPolyFound)
313 {
314 // we moved along the path and the target did not move out of our old poly-path
315 // our path is a simple subpath case, we have all the data we need
316 // just "cut" it out
317
318 _polyLength = pathEndIndex - pathStartIndex + 1;
319 memmove(_pathPolyRefs, _pathPolyRefs + pathStartIndex, _polyLength * sizeof(dtPolyRef));
320 }
321 else if (startPolyFound && !endPolyFound)
322 {
323 // we are moving on the old path but target moved out
324 // so we have atleast part of poly-path ready
325
326 _polyLength -= pathStartIndex;
327
328 // try to adjust the suffix of the path instead of recalculating entire length
329 // at given interval the target cannot get too far from its last location
330 // thus we have less poly to cover
331 // sub-path of optimal path is optimal
332
333 // take ~80% of the original length
335 uint32 prefixPolyLength = uint32(_polyLength * 0.8f + 0.5f);
336 memmove(_pathPolyRefs, _pathPolyRefs + pathStartIndex, prefixPolyLength * sizeof(dtPolyRef));
337
338 dtPolyRef suffixStartPoly = _pathPolyRefs[prefixPolyLength - 1];
339
340 // we need any point on our suffix start poly to generate poly-path, so we need last poly in prefix data
341 float suffixEndPoint[VERTEX_SIZE];
342 if (dtStatusFailed(_navMeshQuery->closestPointOnPoly(suffixStartPoly, endPoint, suffixEndPoint, nullptr)))
343 {
344 // we can hit offmesh connection as last poly - closestPointOnPoly() don't like that
345 // try to recover by using prev polyref
346 --prefixPolyLength;
347 suffixStartPoly = _pathPolyRefs[prefixPolyLength - 1];
348 if (dtStatusFailed(_navMeshQuery->closestPointOnPoly(suffixStartPoly, endPoint, suffixEndPoint, nullptr)))
349 {
350 // suffixStartPoly is still invalid, error state
351 BuildShortcut();
352 _type = PATHFIND_NOPATH;
353 return;
354 }
355 }
356
357 // generate suffix
358 uint32 suffixPolyLength = 0;
359
360 dtStatus dtResult;
361 if (_useRaycast)
362 {
363 BuildShortcut();
364 _type = PATHFIND_NOPATH;
365 return;
366 }
367 else
368 {
369 dtResult = _navMeshQuery->findPath(
370 suffixStartPoly, // start polygon
371 endPoly, // end polygon
372 suffixEndPoint, // start position
373 endPoint, // end position
374 &_filter, // polygon search filter
375 _pathPolyRefs + prefixPolyLength - 1, // [out] path
376 (int*)&suffixPolyLength,
377 MAX_PATH_LENGTH - prefixPolyLength); // max number of polygons in output path
378 }
379
380 if (!suffixPolyLength || dtStatusFailed(dtResult))
381 {
382 // this is probably an error state, but we'll leave it
383 // and hopefully recover on the next Update
384 // we still need to copy our preffix
385 LOG_ERROR("movement", "PathGenerator::BuildPolyPath: Path Build failed {}", _source->GetGUID().ToString());
386 }
387
388 // new path = prefix + suffix - overlap
389 _polyLength = prefixPolyLength + suffixPolyLength - 1;
390 }
391 else
392 {
393 // either we have no path at all -> first run
394 // or something went really wrong -> we aren't moving along the path to the target
395 // just generate new path
396
397 // free and invalidate old path data
398 Clear();
399
400 dtStatus dtResult;
401 if (_useRaycast)
402 {
403 float hit = 0;
404 float hitNormal[3];
405 memset(hitNormal, 0, sizeof(hitNormal));
406
407 dtResult = _navMeshQuery->raycast(
408 startPoly,
409 startPoint,
410 endPoint,
411 &_filter,
412 &hit,
413 hitNormal,
414 _pathPolyRefs,
415 (int*)&_polyLength,
416 MAX_PATH_LENGTH);
417
418 if (!_polyLength || dtStatusFailed(dtResult))
419 {
420 BuildShortcut();
421 _type = PATHFIND_NOPATH;
422 AddFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
423 return;
424 }
425
426 // raycast() sets hit to FLT_MAX if there is a ray between start and end
427 if (hit != FLT_MAX)
428 {
429 float hitPos[3];
430
431 // Walk back a bit from the hit point to make sure it's in the mesh (sometimes the point is actually outside of the polygons due to float precision issues)
432 hit *= 0.99f;
433 dtVlerp(hitPos, startPoint, endPoint, hit);
434
435 // if it fails again, clamp to poly boundary
436 if (dtStatusFailed(_navMeshQuery->getPolyHeight(_pathPolyRefs[_polyLength - 1], hitPos, &hitPos[1])))
437 _navMeshQuery->closestPointOnPolyBoundary(_pathPolyRefs[_polyLength - 1], hitPos, hitPos);
438
439 _pathPoints.resize(2);
440 _pathPoints[0] = GetStartPosition();
441 _pathPoints[1] = G3D::Vector3(hitPos[2], hitPos[0], hitPos[1]);
442
443 NormalizePath();
444 _type = PATHFIND_INCOMPLETE;
445 AddFarFromPolyFlags(startFarFromPoly, false);
446 return;
447 }
448 else
449 {
450 // clamp to poly boundary if we fail to get the height
451 if (dtStatusFailed(_navMeshQuery->getPolyHeight(_pathPolyRefs[_polyLength - 1], endPoint, &endPoint[1])))
452 _navMeshQuery->closestPointOnPolyBoundary(_pathPolyRefs[_polyLength - 1], endPoint, endPoint);
453
454 _pathPoints.resize(2);
455 _pathPoints[0] = GetStartPosition();
456 _pathPoints[1] = G3D::Vector3(endPoint[2], endPoint[0], endPoint[1]);
457
458 NormalizePath();
459 if (startFarFromPoly || endFarFromPoly)
460 {
461 _type = PathType(PATHFIND_INCOMPLETE);
462
463 AddFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
464 }
465 else
466 _type = PATHFIND_NORMAL;
467 return;
468 }
469 }
470 else
471 {
472 dtResult = _navMeshQuery->findPath(
473 startPoly, // start polygon
474 endPoly, // end polygon
475 startPoint, // start position
476 endPoint, // end position
477 &_filter, // polygon search filter
478 _pathPolyRefs, // [out] path
479 (int*)&_polyLength,
480 MAX_PATH_LENGTH); // max number of polygons in output path
481 }
482
483 if (!_polyLength || dtStatusFailed(dtResult))
484 {
485 // only happens if we passed bad data to findPath(), or navmesh is messed up
486 LOG_ERROR("movement", "PathGenerator::BuildPolyPath: {} Path Build failed: 0 length path", _source->GetGUID().ToString());
487 BuildShortcut();
488 _type = PATHFIND_NOPATH;
489 return;
490 }
491 }
492
493 if (!_polyLength)
494 {
495 LOG_ERROR("movement", "PathGenerator::BuildPolyPath: {} Path Build failed: 0 length path", _source->GetGUID().ToString());
496 BuildShortcut();
497 _type = PATHFIND_NOPATH;
498 return;
499 }
500
501 // by now we know what type of path we can get
502 if (_pathPolyRefs[_polyLength - 1] == endPoly && !(_type & PATHFIND_INCOMPLETE))
503 {
504 _type = PATHFIND_NORMAL;
505 }
506 else
507 {
508 _type = PATHFIND_INCOMPLETE;
509 }
510
511 AddFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
512
513 // generate the point-path out of our up-to-date poly-path
514 BuildPointPath(startPoint, endPoint);
515}

517void PathGenerator::BuildPointPath(const float* startPoint, const float* endPoint)
518{
519 float pathPoints[MAX_POINT_PATH_LENGTH * VERTEX_SIZE];
520 uint32 pointCount = 0;
521 dtStatus dtResult = DT_FAILURE;
522 if (_useRaycast)
523 {
524 // _straightLine uses raycast and it currently doesn't support building a point path, only a 2-point path with start and hitpoint/end is returned
525 LOG_ERROR("movement", "PathGenerator::BuildPointPath() called with _useRaycast for unit {}", _source->GetGUID().ToString());
526 BuildShortcut();
527 _type = PATHFIND_NOPATH;
528 return;
529 }
530 else if (_useStraightPath)
531 {
532 dtResult = _navMeshQuery->findStraightPath(
533 startPoint, // start position
534 endPoint, // end position
535 _pathPolyRefs, // current path
536 _polyLength, // lenth of current path
537 pathPoints, // [out] path corner points
538 nullptr, // [out] flags
539 nullptr, // [out] shortened path
540 (int*)&pointCount,
541 _pointPathLimit); // maximum number of points/polygons to use
542 }
543 else
544 {
545 dtResult = FindSmoothPath(
546 startPoint, // start position
547 endPoint, // end position
548 _pathPolyRefs, // current path
549 _polyLength, // length of current path
550 pathPoints, // [out] path corner points
551 (int*)&pointCount,
552 _pointPathLimit); // maximum number of points
553 }
554
555 // Special case with start and end positions very close to each other
556 if (_polyLength == 1 && pointCount == 1)
557 {
558 // First point is start position, append end position
559 dtVcopy(&pathPoints[1 * VERTEX_SIZE], endPoint);
560 pointCount++;
561 }
562 else if (pointCount < 2 || dtStatusFailed(dtResult))
563 {
564 // only happens if pass bad data to findStraightPath or navmesh is broken
565 // single point paths can be generated here
567 BuildShortcut();
568 _type = PathType(_type | PATHFIND_NOPATH);
569 return;
570 }
571 else if (pointCount >= _pointPathLimit)
572 {
573 BuildShortcut();
574 _type = PathType(_type | PATHFIND_SHORT);
575 return;
576 }
577
578 _pathPoints.resize(pointCount);
579 for (uint32 i = 0; i < pointCount; ++i)
580 _pathPoints[i] = G3D::Vector3(pathPoints[i * VERTEX_SIZE + 2], pathPoints[i * VERTEX_SIZE], pathPoints[i * VERTEX_SIZE + 1]);
581
582 NormalizePath();
583
584 // first point is always our current location - we need the next one
585 SetActualEndPosition(_pathPoints[pointCount - 1]);
586
587 // force the given destination, if needed
588 if (_forceDestination &&
589 (!(_type & PATHFIND_NORMAL) || !InRange(GetEndPosition(), GetActualEndPosition(), 1.0f, 1.0f)))
590 {
591 // we may want to keep partial subpath
592 if (Dist3DSqr(GetActualEndPosition(), GetEndPosition()) < 0.3f * Dist3DSqr(GetStartPosition(), GetEndPosition()))
593 {
594 SetActualEndPosition(GetEndPosition());
595 _pathPoints[_pathPoints.size() - 1] = GetEndPosition();
596 }
597 else
598 {
599 SetActualEndPosition(GetEndPosition());
600 BuildShortcut();
601 }
602
603 _type = PathType(PATHFIND_NORMAL | PATHFIND_NOT_USING_PATH);
604 }
605}

615void PathGenerator::BuildShortcut()
616{
617 Clear();
618
619 // make two point path, our curr pos is the start, and dest is the end
620 _pathPoints.resize(2);
621
622 // set start and a default next position
623 _pathPoints[0] = GetStartPosition();
624 _pathPoints[1] = GetActualEndPosition();
625
626 NormalizePath();
627
628 _type = PATHFIND_SHORTCUT;
629}

631void PathGenerator::CreateFilter()
632{
633 uint16 includeFlags = 0;
634 uint16 excludeFlags = 0;
635
636 if (_source->GetTypeId() == TYPEID_UNIT)
637 {
638 Creature* creature = (Creature*)_source;
639 if (creature->CanWalk())
640 includeFlags |= NAV_GROUND; // walk
641
642 // creatures don't take environmental damage
643 if (creature->CanEnterWater())
644 includeFlags |= (NAV_WATER | NAV_MAGMA);
645 }
646 else // assume Player
647 {
648 // perfect support not possible, just stay 'safe'
649 includeFlags |= (NAV_GROUND | NAV_WATER | NAV_MAGMA);
650 }
651
652 _filter.setIncludeFlags(includeFlags);
653 _filter.setExcludeFlags(excludeFlags);
654
655 UpdateFilter();
656}

658void PathGenerator::UpdateFilter()
659{
660 // allow creatures to cheat and use different movement types if they are moved
661 // forcefully into terrain they can't normally move in
662 if (Unit const* _sourceUnit = _source->ToUnit())
663 {
664 if (_sourceUnit->IsInWater() || _sourceUnit->IsUnderWater())
665 {
666 uint16 includedFlags = _filter.getIncludeFlags();
667 includedFlags |= GetNavTerrain(_source->GetPositionX(),
668 _source->GetPositionY(),
669 _source->GetPositionZ());
670
671 _filter.setIncludeFlags(includedFlags);
672 }
673
674 /*if (Creature const* _sourceCreature = _source->ToCreature())
675 if (_sourceCreature->IsInCombat() || _sourceCreature->IsInEvadeMode())
676 _filter.setIncludeFlags(_filter.getIncludeFlags() | NAV_GROUND_STEEP);*/
677 }

680NavTerrain PathGenerator::GetNavTerrain(float x, float y, float z) const
681{
682 LiquidData data;
683 LiquidData const& liquidData = _source->GetMap()->GetLiquidData(_source->GetPhaseMask(), x, y, z, _source->GetCollisionHeight(), MAP_ALL_LIQUIDS);
684 if (liquidData.Status == LIQUID_MAP_NO_WATER)
685 return NAV_GROUND;
686
687 switch (data.Flags)
688 {
689 case MAP_LIQUID_TYPE_WATER:
690 case MAP_LIQUID_TYPE_OCEAN:
691 return NAV_WATER;
692 case MAP_LIQUID_TYPE_MAGMA:
693 case MAP_LIQUID_TYPE_SLIME:
694 return NAV_MAGMA;
695 default:
696 return NAV_GROUND;
697 }
698}

700bool PathGenerator::HaveTile(const G3D::Vector3& p) const
701{
702 int tx = -1, ty = -1;
703 float point[VERTEX_SIZE] = { p.y, p.z, p.x };
704
705 _navMesh->calcTileLoc(point, &tx, &ty);
706
710 if (tx < 0 || ty < 0)
711 return false;
712
713 return (_navMesh->getTileAt(tx, ty, 0) != nullptr);
714}

716uint32 PathGenerator::FixupCorridor(dtPolyRef* path, uint32 npath, uint32 maxPath, dtPolyRef const* visited, uint32 nvisited)
717{
718 int32 furthestPath = -1;
719 int32 furthestVisited = -1;
720
721 // Find furthest common polygon.
722 for (int32 i = npath - 1; i >= 0; --i)
723 {
724 bool found = false;
725 for (int32 j = nvisited - 1; j >= 0; --j)
726 {
727 if (path[i] == visited[j])
728 {
729 furthestPath = i;
730 furthestVisited = j;
731 found = true;
732 }
733 }
734 if (found)
735 break;
736 }
737
738 // If no intersection found just return current path.
739 if (furthestPath == -1 || furthestVisited == -1)
740 return npath;
741
742 // Concatenate paths.
743
744 // Adjust beginning of the buffer to include the visited.
745 uint32 req = nvisited - furthestVisited;
746 uint32 orig = uint32(furthestPath + 1) < npath ? furthestPath + 1 : npath;
747 uint32 size = npath > orig ? npath - orig : 0;
748 if (req + size > maxPath)
749 size = maxPath - req;
750
751 if (size)
752 memmove(path + req, path + orig, size * sizeof(dtPolyRef));
753
754 // Store visited
755 for (uint32 i = 0; i < req; ++i)
756 path[i] = visited[(nvisited - 1) - i];
757
758 return req + size;
759}

761bool PathGenerator::GetSteerTarget(float const* startPos, float const* endPos,
762 float minTargetDist, dtPolyRef const* path, uint32 pathSize,
763 float* steerPos, unsigned char& steerPosFlag, dtPolyRef& steerPosRef)
764{
765 // Find steer target.
766 static const uint32 MAX_STEER_POINTS = 3;
767 float steerPath[MAX_STEER_POINTS * VERTEX_SIZE];
768 unsigned char steerPathFlags[MAX_STEER_POINTS];
769 dtPolyRef steerPathPolys[MAX_STEER_POINTS];
770 uint32 nsteerPath = 0;
771 dtStatus dtResult = _navMeshQuery->findStraightPath(startPos, endPos, path, pathSize,
772 steerPath, steerPathFlags, steerPathPolys, (int*)&nsteerPath, MAX_STEER_POINTS);
773 if (!nsteerPath || dtStatusFailed(dtResult))
774 return false;
775
776 // Find vertex far enough to steer to.
777 uint32 ns = 0;
778 while (ns < nsteerPath)
779 {
780 // Stop at Off-Mesh link or when point is further than slop away.
781 if ((steerPathFlags[ns] & DT_STRAIGHTPATH_OFFMESH_CONNECTION) ||
782 !InRangeYZX(&steerPath[ns * VERTEX_SIZE], startPos, minTargetDist, 1000.0f))
783 break;
784
785 ns++;
786 }
787 // Failed to find good point to steer to.
788 if (ns >= nsteerPath)
789 return false;
790
791 dtVcopy(steerPos, &steerPath[ns * VERTEX_SIZE]);
792 steerPos[1] = startPos[1]; // keep Z value
793 steerPosFlag = steerPathFlags[ns];
794 steerPosRef = steerPathPolys[ns];
795
796 return true;
797}

799dtStatus PathGenerator::FindSmoothPath(float const* startPos, float const* endPos,
800 dtPolyRef const* polyPath, uint32 polyPathSize,
801 float* smoothPath, int* smoothPathSize, uint32 maxSmoothPathSize)
802{
803 *smoothPathSize = 0;
804 uint32 nsmoothPath = 0;
805
806 dtPolyRef polys[MAX_PATH_LENGTH];
807 memcpy(polys, polyPath, sizeof(dtPolyRef) * polyPathSize);
808 uint32 npolys = polyPathSize;
809
810 float iterPos[VERTEX_SIZE], targetPos[VERTEX_SIZE];
811
812 if (polyPathSize > 1)
813 {
814 // Pick the closest points on poly border
815 if (dtStatusFailed(_navMeshQuery->closestPointOnPolyBoundary(polys[0], startPos, iterPos)))
816 {
817 return DT_FAILURE;
818 }
819
820 if (dtStatusFailed(_navMeshQuery->closestPointOnPolyBoundary(polys[npolys - 1], endPos, targetPos)))
821 {
822 return DT_FAILURE;
823 }
824 }
825 else
826 {
827 // Case where the path is on the same poly
828 dtVcopy(iterPos, startPos);
829 dtVcopy(targetPos, endPos);
830 }
831
832 dtVcopy(&smoothPath[nsmoothPath * VERTEX_SIZE], iterPos);
833 nsmoothPath++;
834
835 // Move towards target a small advancement at a time until target reached or
836 // when ran out of memory to store the path.
837 while (npolys && nsmoothPath < maxSmoothPathSize)
838 {
839 // Find location to steer towards.
840 float steerPos[VERTEX_SIZE];
841 unsigned char steerPosFlag;
842 dtPolyRef steerPosRef = INVALID_POLYREF;
843
844 if (!GetSteerTarget(iterPos, targetPos, SMOOTH_PATH_SLOP, polys, npolys, steerPos, steerPosFlag, steerPosRef))
845 break;
846
847 bool endOfPath = (steerPosFlag & DT_STRAIGHTPATH_END) != 0;
848 bool offMeshConnection = (steerPosFlag & DT_STRAIGHTPATH_OFFMESH_CONNECTION) != 0;
849
850 // Find movement delta.
851 float delta[VERTEX_SIZE];
852 dtVsub(delta, steerPos, iterPos);
853 float len = dtMathSqrtf(dtVdot(delta, delta));
854 // If the steer target is end of path or off-mesh link, do not move past the location.
855 if ((endOfPath || offMeshConnection) && len < SMOOTH_PATH_STEP_SIZE)
856 len = 1.0f;
857 else
858 len = SMOOTH_PATH_STEP_SIZE / len;
859
860 float moveTgt[VERTEX_SIZE];
861 dtVmad(moveTgt, iterPos, delta, len);
862
863 // Move
864 float result[VERTEX_SIZE];
865 const static uint32 MAX_VISIT_POLY = 16;
866 dtPolyRef visited[MAX_VISIT_POLY];
867
868 uint32 nvisited = 0;
869 if (dtStatusFailed(_navMeshQuery->moveAlongSurface(polys[0], iterPos, moveTgt, &_filter, result, visited, (int*)&nvisited, MAX_VISIT_POLY)))
870 {
871 return DT_FAILURE;
872 }
873 npolys = FixupCorridor(polys, npolys, MAX_PATH_LENGTH, visited, nvisited);
874
875 if (dtStatusFailed(_navMeshQuery->getPolyHeight(polys[0], result, &result[1])))
876 LOG_DEBUG("maps", "PathGenerator::FindSmoothPath: Cannot find height at position X: {} Y: {} Z: {} for {}",
877 result[2], result[0], result[1], _source->GetGUID().ToString());
878 result[1] += 0.5f;
879 dtVcopy(iterPos, result);
880
881 bool canCheckSlope = _slopeCheck && (GetPathType() & ~(PATHFIND_NOT_USING_PATH));
882
883 if (canCheckSlope && !IsSwimmableSegment(iterPos, steerPos) && !IsWalkableClimb(iterPos, steerPos))
884 {
885 return DT_FAILURE;
886 }
887
888 // Handle end of path and off-mesh links when close enough.
889 if (endOfPath && InRangeYZX(iterPos, steerPos, SMOOTH_PATH_SLOP, 1.0f))
890 {
891 // Reached end of path.
892 dtVcopy(iterPos, targetPos);
893 if (nsmoothPath < maxSmoothPathSize)
894 {
895 dtVcopy(&smoothPath[nsmoothPath * VERTEX_SIZE], iterPos);
896 nsmoothPath++;
897 }
898 break;
899 }
900 else if (offMeshConnection && InRangeYZX(iterPos, steerPos, SMOOTH_PATH_SLOP, 1.0f))
901 {
902 // Advance the path up to and over the off-mesh connection.
903 dtPolyRef prevRef = INVALID_POLYREF;
904 dtPolyRef polyRef = polys[0];
905 uint32 npos = 0;
906 while (npos < npolys && polyRef != steerPosRef)
907 {
908 prevRef = polyRef;
909 polyRef = polys[npos];
910 npos++;
911 }
912
913 for (uint32 i = npos; i < npolys; ++i)
914 polys[i - npos] = polys[i];
915
916 npolys -= npos;
917
918 // Handle the connection.
919 float connectionStartPos[VERTEX_SIZE], connectionEndPos[VERTEX_SIZE];
920 if (dtStatusSucceed(_navMesh->getOffMeshConnectionPolyEndPoints(prevRef, polyRef, connectionStartPos, connectionEndPos)))
921 {
922 if (nsmoothPath < maxSmoothPathSize)
923 {
924 dtVcopy(&smoothPath[nsmoothPath * VERTEX_SIZE], connectionStartPos);
925 nsmoothPath++;
926 }
927 // Move position at the other side of the off-mesh link.
928 dtVcopy(iterPos, connectionEndPos);
929 if (dtStatusFailed(_navMeshQuery->getPolyHeight(polys[0], iterPos, &iterPos[1])))
930 return DT_FAILURE;
931 iterPos[1] += 0.5f;
932 }
933 }
934
935 // Store results.
936 if (nsmoothPath < maxSmoothPathSize)
937 {
938 dtVcopy(&smoothPath[nsmoothPath * VERTEX_SIZE], iterPos);
939 nsmoothPath++;
940 }
941 }
942
943 *smoothPathSize = nsmoothPath;
944
945 // this is most likely a loop
946 return nsmoothPath < MAX_POINT_PATH_LENGTH ? DT_SUCCESS : DT_FAILURE;
947}

1001bool PathGenerator::InRangeYZX(float const* v1, float const* v2, float r, float h) const
1002{

1004 const float dy = v2[1] - v1[1]; // elevation

1020void PathGenerator::ShortenPathUntilDist(G3D::Vector3 const& target, float dist)
1021{
1022 if (GetPathType() == PATHFIND_BLANK || _pathPoints.size() < 2)
1023 {
1024 LOG_ERROR("movement", "PathGenerator::ReducePathLengthByDist called before path was successfully built");
1025 return;
1026 }
1027
1028 float const distSq = dist * dist;
1029
1030 // the first point of the path must be outside the specified range
1031 // (this should have really been checked by the caller...)
1032 if ((_pathPoints[0] - target).squaredLength() < distSq)
1033 return;
1034
1035 // check if we even need to do anything
1036 if ((*_pathPoints.rbegin() - target).squaredLength() >= distSq)
1037 return;
1038
1039 size_t i = _pathPoints.size() - 1;
1040 float x, y, z, collisionHeight = _source->GetCollisionHeight();
1041 // find the first i s.t.:
1042 // - _pathPoints[i] is still too close
1043 // - _pathPoints[i-1] is too far away
1044 // => the end point is somewhere on the line between the two
1045 while (1)
1046 {
1047 // we know that pathPoints[i] is too close already (from the previous iteration)
1048 if ((_pathPoints[i - 1] - target).squaredLength() >= distSq)
1049 break; // bingo!
1050
1051 bool canCheckSlope = _slopeCheck && (GetPathType() & ~(PATHFIND_NOT_USING_PATH));
1052
1053 // check if the shortened path is still in LoS with the target and it is walkable
1054 _source->GetHitSpherePointFor({ _pathPoints[i - 1].x, _pathPoints[i - 1].y, _pathPoints[i - 1].z + collisionHeight }, x, y, z);
1055 if (!_source->GetMap()->isInLineOfSight(x, y, z, _pathPoints[i - 1].x, _pathPoints[i - 1].y, _pathPoints[i - 1].z + collisionHeight,
1056 _source->GetPhaseMask(), LINEOFSIGHT_ALL_CHECKS, VMAP::ModelIgnoreFlags::Nothing) || (canCheckSlope &&
1057 !IsSwimmableSegment(_source->GetPositionX(), _source->GetPositionY(), _source->GetPositionZ(), _pathPoints[i - 1].x, _pathPoints[i - 1].y, _pathPoints[i - 1].z) &&
1058 !IsWalkableClimb(_source->GetPositionX(), _source->GetPositionY(), _source->GetPositionZ(), _pathPoints[i - 1].x, _pathPoints[i - 1].y, _pathPoints[i - 1].z)))
1059 {
1060 // whenver we find a point that is not valid anymore, simply use last valid path
1061 _pathPoints.resize(i + 1);
1062 return;
1063 }
1064
1065 if (!--i)
1066 {
1067 // no point found that fulfills the condition
1068 _pathPoints[0] = _pathPoints[1];
1069 _pathPoints.resize(2);
1070 return;
1071 }
1072 }
1073
1074 // ok, _pathPoints[i] is too close, _pathPoints[i-1] is not, so our target point is somewhere between the two...
1075 // ... settle for a guesstimate since i'm not confident in doing trig on every chase motion tick...
1076 // (@todo review this)
1077 _pathPoints[i] += (_pathPoints[i - 1] - _pathPoints[i]).direction() * (dist - (_pathPoints[i] - target).length());
1078 _pathPoints.resize(i + 1);
1079}
