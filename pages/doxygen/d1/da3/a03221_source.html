131 void EnterCombat(Unit* ) override
132 {
133 Talk(SAY_AGGRO);
134 instance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_TIMED_START_EVENT);
135
136 events.ScheduleEvent(EVENT_CARRION_BEETELS, 6500);
137 events.ScheduleEvent(EVENT_LEECHING_SWARM, 20000);
138 events.ScheduleEvent(EVENT_POUND, 15000);
139 events.ScheduleEvent(EVENT_CHECK_HEALTH_75, 1000);
140 events.ScheduleEvent(EVENT_CHECK_HEALTH_50, 1000);
141 events.ScheduleEvent(EVENT_CHECK_HEALTH_25, 1000);
142 events.ScheduleEvent(EVENT_CLOSE_DOORS, 5000);
143 }

145 void SummonHelpers(float x, float y, float z, uint32 spellId)
146 {
147 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId);
148 me->SummonCreature(spellInfo->Effects[EFFECT_0].MiscValue, x, y, z);
149 }
150
151 void UpdateAI(uint32 diff) override
152 {
153 if (!UpdateVictim())
154 return;
155
156 events.Update(diff);
157 if (me->HasUnitState(UNIT_STATE_CASTING))
158 return;
159
160 switch (uint32 eventId = events.ExecuteEvent())
161 {
162 case EVENT_CLOSE_DOORS:
163 _EnterCombat();
164 break;
165 case EVENT_CARRION_BEETELS:
166 me->CastSpell(me, SPELL_CARRION_BEETLES, false);
167 events.ScheduleEvent(EVENT_CARRION_BEETELS, 25000);
168 break;
169 case EVENT_LEECHING_SWARM:
170 Talk(SAY_LOCUST);
171 me->CastSpell(me, SPELL_LEECHING_SWARM, false);
172 events.ScheduleEvent(EVENT_LEECHING_SWARM, 20000);
173 break;
174 case EVENT_POUND:
175 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 10.0f))
176 {
177 me->CastSpell(me, SPELL_SELF_ROOT, true);
178 me->DisableRotate(true);
179 me->SendMovementFlagUpdate();
180 events.ScheduleEvent(EVENT_ENABLE_ROTATE, 3300);
181 me->CastSpell(target, SPELL_POUND, false);
182 }
183 events.ScheduleEvent(EVENT_POUND, 18000);
184 break;
185 case EVENT_ENABLE_ROTATE:
186 me->RemoveAurasDueToSpell(SPELL_SELF_ROOT);
187 me->DisableRotate(false);
188 break;
189 case EVENT_CHECK_HEALTH_25:
190 case EVENT_CHECK_HEALTH_50:
191 case EVENT_CHECK_HEALTH_75:
192 if (me->HealthBelowPct(eventId*25))
193 {
194 Talk(SAY_SUBMERGE);
195 me->CastSpell(me, SPELL_IMPALE_PERIODIC, true);
196 me->CastSpell(me, SPELL_SUBMERGE, false);
197 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
198
199 events.DelayEvents(46000, 0);
200 events.ScheduleEvent(EVENT_EMERGE, 45000);
201 events.ScheduleEvent(EVENT_SUMMON_ASSASSINS, 2000);
202 events.ScheduleEvent(EVENT_SUMMON_GUARDIAN, 4000);
203 events.ScheduleEvent(EVENT_SUMMON_ASSASSINS, 15000);
204 events.ScheduleEvent(EVENT_SUMMON_VENOMANCER, 20000);
205 events.ScheduleEvent(EVENT_SUMMON_DARTER, 30000);
206 events.ScheduleEvent(EVENT_SUMMON_ASSASSINS, 35000);
207 break;
208 }
209 events.ScheduleEvent(eventId, 500);
210 break;
211 case EVENT_EMERGE:
212 me->CastSpell(me, SPELL_EMERGE, true);
213 me->RemoveAura(SPELL_SUBMERGE);
214 me->RemoveAura(SPELL_IMPALE_PERIODIC);
215 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE|UNIT_FLAG_NOT_SELECTABLE);
216 break;
217 case EVENT_SUMMON_ASSASSINS:
218 SummonHelpers(509.32f, 247.42f, 239.48f, SPELL_SUMMON_ASSASSIN);
219 SummonHelpers(589.51f, 240.19f, 236.0f, SPELL_SUMMON_ASSASSIN);
220 break;
221 case EVENT_SUMMON_DARTER:
222 SummonHelpers(509.32f, 247.42f, 239.48f, SPELL_SUMMON_DARTER);
223 SummonHelpers(589.51f, 240.19f, 236.0f, SPELL_SUMMON_DARTER);
224 break;
225 case EVENT_SUMMON_GUARDIAN:
226 SummonHelpers(550.34f, 316.00f, 234.30f, SPELL_SUMMON_GUARDIAN);
227 break;
228 case EVENT_SUMMON_VENOMANCER:
229 SummonHelpers(550.34f, 316.00f, 234.30f, SPELL_SUMMON_VENOMANCER);
230 break;
231 }
232
233 if (!me->HasUnitFlag(UNIT_FLAG_NOT_SELECTABLE))
234 DoMeleeAttackIfReady();
235 }
236 };
