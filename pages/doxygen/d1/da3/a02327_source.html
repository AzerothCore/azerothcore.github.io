137 // Enable/Disable accepting whispers (for GM)
138 static bool HandleWhispersCommand(ChatHandler* handler, Optional<Variant<bool, EXACT_SEQUENCE("remove")>> operationArg, Optional<std::string> playerNameArg)
139 {
140 if (!operationArg)
141 {
142 handler->PSendSysMessage(LANG_COMMAND_WHISPERACCEPTING, handler->GetSession()->GetPlayer()->isAcceptWhispers() ? handler->GetAcoreString(LANG_ON) : handler->GetAcoreString(LANG_OFF));
143 return true;
144 }
145
146 if (operationArg->holds_alternative<bool>())
147 {
148 if (operationArg->get<bool>())
149 {
150 handler->GetSession()->GetPlayer()->SetAcceptWhispers(true);
151 handler->SendSysMessage(LANG_COMMAND_WHISPERON);
152 return true;
153 }
154 else
155 {
156 // Remove all players from the Gamemaster's whisper whitelist
157 handler->GetSession()->GetPlayer()->ClearWhisperWhiteList();
158 handler->GetSession()->GetPlayer()->SetAcceptWhispers(false);
159 handler->SendSysMessage(LANG_COMMAND_WHISPEROFF);
160 return true;
161 }
162 }
163
164 if (operationArg->holds_alternative<EXACT_SEQUENCE("remove")>())
165 {
166 if (!playerNameArg)
167 return false;
168
169 if (normalizePlayerName(*playerNameArg))
170 {
171 if (Player* player = ObjectAccessor::FindPlayerByName(*playerNameArg))
172 {
173 handler->GetSession()->GetPlayer()->RemoveFromWhisperWhiteList(player->GetGUID());
174 handler->PSendSysMessage(LANG_COMMAND_WHISPEROFFPLAYER, playerNameArg->c_str());
175 return true;
176 }
177 else
178 {
179 handler->PSendSysMessage(LANG_PLAYER_NOT_FOUND, playerNameArg->c_str());
180 handler->SetSentErrorMessage(true);
181 return false;
182 }
183 }
184 }
185 handler->SendSysMessage(LANG_USE_BOL);
186 handler->SetSentErrorMessage(true);
187 return false;
188 }
