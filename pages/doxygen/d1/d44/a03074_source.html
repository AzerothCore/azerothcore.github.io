258 void Load(char const* data) override
259 {
260 if (!data)
261 {
262 OUT_LOAD_INST_DATA_FAIL;
263 return;
264 }
265
266 OUT_LOAD_INST_DATA(data);
267
268 char dataHead1, dataHead2;
269
270 std::istringstream loadStream(data);
271 loadStream >> dataHead1 >> dataHead2;
272
273 if (dataHead1 == 'R' && dataHead2 == 'A')
274 {
275 for (uint8 i = 0; i < NUM_ENCOUNTER; ++i)
276 {
277 uint32 tmpState;
278 loadStream >> tmpState;
279 if (tmpState == IN_PROGRESS || tmpState > TO_BE_DECIDED)
280 tmpState = NOT_STARTED;
281 SetBossState(i, EncounterState(tmpState));
282 }
283 }
284 else
285 OUT_LOAD_INST_DATA_FAIL;
286
287 OUT_LOAD_INST_DATA_COMPLETE;

290 void CallNextRajaxxLeader(bool announce = false)
291 {
292 ++_rajaxWaveCounter;
293
294 if (Creature* nextLeader = GetCreature(RajaxxWavesData[_rajaxWaveCounter].at(0)))
295 {
296 if (announce)
297 {
298 if (_rajaxWaveCounter >= 2)
299 {
300 if (Creature* rajaxx = GetCreature(DATA_RAJAXX))
301 {
302 rajaxx->AI()->Talk(RajaxxWavesData[_rajaxWaveCounter].at(1));
303 }
304 }
305 }
306
307 if (nextLeader->IsAlive())
308 {
309 Creature* generalAndorov = instance->GetCreature(_andorovGUID);
310 if (generalAndorov && generalAndorov->IsAlive() && generalAndorov->AI()->GetData(DATA_ANDOROV))
311 {
312 nextLeader->AI()->AttackStart(generalAndorov);
313 }
314 else
315 {
316 nextLeader->SetInCombatWithZone();
317 }
318 }
319 else
320 {
321 CallNextRajaxxLeader();
322 }
323 }
324 }
