72 loadStream >> dataHead1 >> dataHead2;

74 if (dataHead1 == 'H' && dataHead2 == 'R')
75 {
76 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)

78 uint32 tmpState;
