194 break;

196 Talk(YELL_SKARVALD_DAL_DIEDFIRST);
197 break;

199 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, (IsHeroic() ? 100.0f : 30.0f), true))

201 ScriptedAI::DoResetThreat();
202 me->AddThreat(target, 10000.0f);
203 me->CastSpell(target, SPELL_CHARGE, false);
204 }

206 break;
207 case EVENT_STONE_STRIKE:
208 if (me->GetVictim() && me->IsWithinMeleeRange(me->GetVictim()))
209 {
210 me->CastSpell(me->GetVictim(), SPELL_STONE_STRIKE, false);
211 events.RepeatEvent(urand(5000, 10000));
212 }
213 else
214 {
215 events.RepeatEvent(3000);
216 }
217 break;
218 case EVENT_ENRAGE:
219 if (me->GetHealthPct() <= 60)
220 {
221 me->CastSpell(me, SPELL_ENRAGE, true);
222 break;
223 }
224 events.RepeatEvent(1000);
225 break;
226 }
227 DoMeleeAttackIfReady();
228 }
