118 void Load(const char* in) override
119 {
120 if( !in )
121 return;
122
123 char dataHead1, dataHead2;
124 std::istringstream loadStream(in);
125 loadStream >> dataHead1 >> dataHead2;
126 if (dataHead1 == 'A' && dataHead2 == 'N')
127 {
128 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
129 {
130 uint32 tmpState;
131 loadStream >> tmpState;
132 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
133 tmpState = NOT_STARTED;
134 SetBossState(i, EncounterState(tmpState));
135 }
136 }
137 }
138 };
