76class spell_pri_shadowfiend_scaling : public AuraScript
77{
78 PrepareAuraScript(spell_pri_shadowfiend_scaling);
79
80 void CalculateResistanceAmount(AuraEffect const* aurEff, int32& amount, bool& /*canBeRecalculated*/)
81 {
82 // xinef: shadowfiend inherits 40% of resistance from owner and 35% of armor (guessed)
83 if (Unit* owner = GetUnitOwner()->GetOwner())
84 {
85 SpellSchoolMask schoolMask = SpellSchoolMask(aurEff->GetSpellInfo()->Effects[aurEff->GetEffIndex()].MiscValue);
86 int32 modifier = schoolMask == SPELL_SCHOOL_MASK_NORMAL ? 35 : 40;
87 amount = CalculatePct(std::max<int32>(0, owner->GetResistance(schoolMask)), modifier);
88 }
89 }
90
91 void CalculateStatAmount(AuraEffect const* aurEff, int32& amount, bool& /*canBeRecalculated*/)
92 {
93 // xinef: shadowfiend inherits 30% of intellect and 65% of stamina (guessed)
94 if (Unit* owner = GetUnitOwner()->GetOwner())
95 {
96 Stats stat = Stats(aurEff->GetSpellInfo()->Effects[aurEff->GetEffIndex()].MiscValue);
97 amount = CalculatePct(std::max<int32>(0, owner->GetStat(stat)), stat == STAT_STAMINA ? 65 : 30);
98 }
99 }
100
101 void CalculateAPAmount(AuraEffect const* /*aurEff*/, int32& amount, bool& /*canBeRecalculated*/)
102 {
103 // xinef: shadowfiend inherits 333% of SP as AP - 35.7% of damage increase per hit
104 if (Unit* owner = GetUnitOwner()->GetOwner())
105 {
106 int32 shadow = owner->SpellBaseDamageBonusDone(SPELL_SCHOOL_MASK_SHADOW);
107 amount = CalculatePct(std::max<int32>(0, shadow), 300); // xinef: deacrased to 300, including 15% from self buff
108 }
109 }
110
111 void CalculateSPAmount(AuraEffect const* /*aurEff*/, int32& amount, bool& /*canBeRecalculated*/)
112 {
113 // xinef: shadowfiend inherits 30% of SP
114 if (Unit* owner = GetUnitOwner()->GetOwner())
115 {
116 int32 shadow = owner->SpellBaseDamageBonusDone(SPELL_SCHOOL_MASK_SHADOW);
117 amount = CalculatePct(std::max<int32>(0, shadow), 30);
118
119 // xinef: Update appropriate player field
120 if (owner->GetTypeId() == TYPEID_PLAYER)
121 owner->SetUInt32Value(PLAYER_PET_SPELL_POWER, (uint32)amount);
122 }
123 }
124
125 void HandleEffectApply(AuraEffect const* aurEff, AuraEffectHandleModes /*mode*/)
126 {
127 GetUnitOwner()->ApplySpellImmune(0, IMMUNITY_STATE, aurEff->GetAuraType(), true, SPELL_BLOCK_TYPE_POSITIVE);
128 if (aurEff->GetAuraType() == SPELL_AURA_MOD_ATTACK_POWER)
129 GetUnitOwner()->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_ATTACK_POWER_PCT, true, SPELL_BLOCK_TYPE_POSITIVE);
130 else if (aurEff->GetAuraType() == SPELL_AURA_MOD_STAT)
131 GetUnitOwner()->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_TOTAL_STAT_PERCENTAGE, true, SPELL_BLOCK_TYPE_POSITIVE);
132 }
133
134 void Register() override
135 {
136 if (m_scriptSpellId != 35661)
137 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_pri_shadowfiend_scaling::CalculateResistanceAmount, EFFECT_ALL, SPELL_AURA_MOD_RESISTANCE);
138
139 if (m_scriptSpellId == 35661 || m_scriptSpellId == 35662)
140 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_pri_shadowfiend_scaling::CalculateStatAmount, EFFECT_ALL, SPELL_AURA_MOD_STAT);
141
142 if (m_scriptSpellId == 35661)
143 {
144 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_pri_shadowfiend_scaling::CalculateAPAmount, EFFECT_ALL, SPELL_AURA_MOD_ATTACK_POWER);
145 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_pri_shadowfiend_scaling::CalculateSPAmount, EFFECT_ALL, SPELL_AURA_MOD_DAMAGE_DONE);
146 }
147
148 OnEffectApply += AuraEffectApplyFn(spell_pri_shadowfiend_scaling::HandleEffectApply, EFFECT_ALL, SPELL_AURA_ANY, AURA_EFFECT_HANDLE_REAL);
