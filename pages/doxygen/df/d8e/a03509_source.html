117 events.ScheduleEvent(EVENT_MAGUS_ICE_NOVA, 10000);
118 events.ScheduleEvent(EVENT_MAGUS_FIREBOMB, 0);
119 events.ScheduleEvent(EVENT_MAGUS_GRAVITY_WELL, 20000);

147 {

149 summon->SetInCombatWithZone();
150 }
151
152 void SpellHit(Unit* caster, SpellInfo const* spellInfo) override
153 {
154 if (spellInfo->Id >= SPELL_FIRE_MAGUS_DEATH && spellInfo->Id <= SPELL_ARCANE_MAGUS_DEATH && caster->ToCreature())
155 {
156 events.ScheduleEvent(EVENT_MAGUS_FAIL_ACHIEVEMENT, 5000);
157 caster->ToCreature()->DespawnOrUnsummon(1000);
158
159 if (++copiesDied >= 3)
160 {
161 copiesDied = 0;
162 events.CancelEvent(EVENT_MAGUS_FAIL_ACHIEVEMENT);
163 events.ScheduleEvent(EVENT_MAGUS_MERGED, 5000);
164 me->CastSpell(me, SPELL_BURNING_WINDS, true);
165 }
166 }
167 }
168
169 void UpdateAI(uint32 diff) override
170 {
171 if (!UpdateVictim())
172 return;
173
174 events.Update(diff);
175 if (me->HasUnitState(UNIT_STATE_CASTING))
176 return;
177
178 switch (events.ExecuteEvent())
179 {
180 case EVENT_MAGUS_HEALTH1:
181 if (me->HealthBelowPct(51))
182 {
183 me->CastSpell(me, SPELL_START_SUMMON_CLONES, false);
184 events.ScheduleEvent(EVENT_MAGUS_RELOCATE, 3500);
185 Talk(SAY_SPLIT);
186 break;
187 }
188 events.ScheduleEvent(EVENT_MAGUS_HEALTH1, 1000);
189 break;
190 case EVENT_MAGUS_HEALTH2:
191 if (me->HealthBelowPct(11))
192 {
193 me->CastSpell(me, SPELL_START_SUMMON_CLONES, false);
194 events.ScheduleEvent(EVENT_MAGUS_RELOCATE, 3500);
195 Talk(SAY_SPLIT);
196 break;
197 }
198 events.ScheduleEvent(EVENT_MAGUS_HEALTH2, 1000);
199 break;
200 case EVENT_MAGUS_FIREBOMB:
201 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
202 me->CastSpell(target, SPELL_FIREBOMB, false);
203 events.ScheduleEvent(EVENT_MAGUS_FIREBOMB, 3000);
204 break;
205 case EVENT_MAGUS_ICE_NOVA:
206 me->CastSpell(me, SPELL_ICE_NOVA, false);
207 events.ScheduleEvent(EVENT_MAGUS_ICE_NOVA, 15000);
208 break;
209 case EVENT_MAGUS_GRAVITY_WELL:
210 me->CastSpell(me, SPELL_GRAVITY_WELL, false);
211 events.ScheduleEvent(EVENT_MAGUS_GRAVITY_WELL, 15000);
212 break;
213 case EVENT_MAGUS_FAIL_ACHIEVEMENT:
214 achievement = false;
215 break;
216 case EVENT_MAGUS_RELOCATE:
217 me->NearTeleportTo(505.04f, 88.915f, -16.13f, 2.98f);
218 break;
219 case EVENT_MAGUS_MERGED:
220 me->CastSpell(me, SPELL_TELESTRA_BACK, true);
221 me->RemoveAllAuras();
222 Talk(SAY_MERGE);
223 break;
224 }
225
226 DoMeleeAttackIfReady();
227 }
228 };

290 void HandlePull(SpellEffIndex effIndex)
291 {
292 PreventHitDefaultEffect(effIndex);
293 Unit* target = GetHitUnit();
294 if (!target)
295 return;
296
297 Position pos;
298 if (target->GetDistance(GetCaster()) < 5.0f)
299 {
300 pos.Relocate(GetCaster()->GetPositionX(), GetCaster()->GetPositionY(), GetCaster()->GetPositionZ() + 1.0f);
301 float o = frand(0, 2 * M_PI);
302 target->MovePositionToFirstCollision(pos, 20.0f, o);
303 pos.m_positionZ += frand(5.0f, 15.0f);
304 }
305 else
306 pos.Relocate(GetCaster()->GetPositionX(), GetCaster()->GetPositionY(), GetCaster()->GetPositionZ() + 1.0f);
307
308 float speedXY = float(GetSpellInfo()->Effects[effIndex].MiscValue) * 0.1f;
309 float speedZ = target->GetDistance(pos) / speedXY * 0.5f * Movement::gravity;
310
311 target->GetMotionMaster()->MoveJump(pos.GetPositionX(), pos.GetPositionY(), pos.GetPositionZ(), speedXY, speedZ);
312 }

315 {
316 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_boss_magus_telestra_gravity_well_SpellScript::SelectTarget, EFFECT_0, TARGET_UNIT_SRC_AREA_ENEMY);
