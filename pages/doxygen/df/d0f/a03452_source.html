175// Predicate function to check that the r efzr unit is NOT on the same side as the source.
176struct NotOnSameSide
177{
178public:
179 explicit NotOnSameSide(Unit* pSource) : m_inLiveSide(IN_LIVE_SIDE(pSource)) { }

354 bool CheckGroupSplitted()
355 {
356 Map::PlayerList const& PlayerList = me->GetMap()->GetPlayers();
357 if (!PlayerList.IsEmpty())
358 {
359 bool checklife = false;
360 bool checkdead = false;
361 for (const auto& i : PlayerList)
362 {
363 Player* player = i.GetSource();
364 if (player->IsAlive() &&
365 player->GetPositionX() <= POS_X_NORTH &&
366 player->GetPositionX() >= POS_X_SOUTH &&
367 player->GetPositionY() <= POS_Y_GATE &&
368 player->GetPositionY() >= POS_Y_EAST)
369 {
370 checklife = true;
371 }
372 else if (player->IsAlive() &&
373 player->GetPositionX() <= POS_X_NORTH &&
374 player->GetPositionX() >= POS_X_SOUTH &&
375 player->GetPositionY() >= POS_Y_GATE &&
376 player->GetPositionY() <= POS_Y_WEST)
377 {
378 checkdead = true;
379 }
380
381 if (checklife && checkdead)
382 return true;
383 }
384 }
385 return false;

415 void UpdateAI(uint32 diff) override
416 {
417 if (!IsInRoom())
418 return;
419
420 if (!UpdateVictim())
421 return;
422
423 events.Update(diff);
424 if (me->HasUnitState(UNIT_STATE_CASTING))
425 return;
426
427 switch (events.ExecuteEvent())
428 {
429 case EVENT_INTRO_2:
430 Talk(SAY_INTRO_2);
431 break;
432 case EVENT_INTRO_3:
433 Talk(SAY_INTRO_3);
434 break;
435 case EVENT_INTRO_4:
436 Talk(SAY_INTRO_4);
437 break;
438 case EVENT_SHADOW_BOLT:
439 me->CastSpell(me->GetVictim(), RAID_MODE(SPELL_SHADOW_BOLT_10, SPELL_SHADOW_BOLT_25), false);
440 events.RepeatEvent(1010);
441 break;
442 case EVENT_HARVEST_SOUL:
443 me->CastSpell(me, SPELL_HARVEST_SOUL, false);
444 events.RepeatEvent(15000);
445 break;
446 case EVENT_TELEPORT:
447 me->AttackStop();
448 if (IN_LIVE_SIDE(me))
449 {
450 me->CastSpell(me, SPELL_TELEPORT_DEAD, false);
451 }
452 else
453 {
454 me->CastSpell(me, SPELL_TELEPORT_LIVE, false);
455 }
456 me->GetThreatMgr().resetAggro(NotOnSameSide(me));
457 if (Unit* pTarget = SelectTarget(SelectTargetMethod::MaxDistance, 0))
458 {
459 me->GetThreatMgr().AddThreat(pTarget, 100.0f);
460 AttackStart(pTarget);
461 }
462 events.RepeatEvent(20000);
463 break;
464 case EVENT_CHECK_HEALTH:
465 if (me->HealthBelowPct(30) && pInstance)
466 {
467 if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_GOTHIK_INNER_GATE)))
468 {
469 go->SetGoState(GO_STATE_ACTIVE);
470 }
471 events.CancelEvent(EVENT_TELEPORT);
472 break;
473 }
474 events.RepeatEvent(1000);
475 break;
476 case EVENT_SUMMON_ADDS:
477 if (gothikWaves[waveCount][0])
478 {
479 SummonHelpers(gothikWaves[waveCount][0]);
480 events.RepeatEvent(gothikWaves[waveCount][1]);
481 }
482 else
483 {
484 secondPhase = true;
485 Talk(SAY_PHASE_TWO);
486 Talk(EMOTE_PHASE_TWO);
487 me->CastSpell(me, SPELL_TELEPORT_LIVE, false);
488 me->SetReactState(REACT_AGGRESSIVE);
489 me->RemoveUnitFlag(UNIT_FLAG_DISABLE_MOVE);
490 me->SetImmuneToPC(false);
491 me->RemoveAllAuras();
492 summons.DoZoneInCombat();
493 events.ScheduleEvent(EVENT_SHADOW_BOLT, 1000);
494 events.ScheduleEvent(EVENT_HARVEST_SOUL, urand(5000, 15000));
495 events.ScheduleEvent(EVENT_TELEPORT, 20000);
496 events.ScheduleEvent(EVENT_CHECK_HEALTH, 1000);
497 }
498 waveCount++;
499 break;
500 case EVENT_CHECK_PLAYERS:
501 if (!CheckGroupSplitted())
502 {
503 if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_GOTHIK_INNER_GATE)))
504 {
505 go->SetGoState(GO_STATE_ACTIVE);
506 }
507 summons.DoZoneInCombat();
508 gateOpened = true;
509 Talk(EMOTE_GATE_OPENED);
510 }
511 break;
512 }
513 DoMeleeAttackIfReady();
514 }
515 };
