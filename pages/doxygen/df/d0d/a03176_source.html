259 void Reset() override
260 {
261 ResetTimer = 120000;
262
263 if (me->GetFaction() != NormFaction)
264 me->SetFaction(NormFaction);
265
266 if (me->HasNpcFlag(UNIT_NPC_FLAG_GOSSIP))
267 me->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP);
268
269 DoCast(me, SPELL_PLUCKY_CHICKEN, false);
270 }
271
272 void ReceiveEmote(Player* player, uint32 TextEmote) override
273 {
274 if (player->GetQuestStatus(QUEST_SCOOP) == QUEST_STATUS_INCOMPLETE)
275 {
276 if (TextEmote == TEXT_EMOTE_BECKON)
277 {
278 me->SetFaction(FACTION_FRIENDLY);
279 me->SetNpcFlag(UNIT_NPC_FLAG_GOSSIP);
280 DoCast(me, SPELL_PLUCKY_HUMAN, false);
281 }
282 }
283
284 if (TextEmote == TEXT_EMOTE_CHICKEN)
285 {
286 if (me->HasNpcFlag(UNIT_NPC_FLAG_GOSSIP))
287 return;
288 else
289 {
290 me->SetFaction(FACTION_FRIENDLY);
291 me->SetNpcFlag(UNIT_NPC_FLAG_GOSSIP);
292 DoCast(me, SPELL_PLUCKY_HUMAN, false);
293 me->HandleEmoteCommand(EMOTE_ONESHOT_WAVE);
294 }
295 }
296 }
297
298 void UpdateAI(uint32 Diff) override
299 {
300 if (me->HasNpcFlag(UNIT_NPC_FLAG_GOSSIP))
301 {
302 if (ResetTimer <= Diff)
303 {
304 if (!me->GetVictim())
305 EnterEvadeMode();
306 else
307 me->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP);
308
309 return;
310 }
311 else
312 ResetTimer -= Diff;
313 }
314
315 if (!UpdateVictim())
316 return;
317
318 DoMeleeAttackIfReady();
319 }
