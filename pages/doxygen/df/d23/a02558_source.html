48 struct boss_shazzrahAI : public BossAI

50 boss_shazzrahAI(Creature* creature) : BossAI(creature, DATA_SHAZZRAH) {}
51
52 void EnterCombat(Unit* /*target*/) override
53 {
54 _EnterCombat();
55 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, urand(2000, 4000));
56 events.ScheduleEvent(EVENT_SHAZZRAH_CURSE, urand(7000, 11000));
57 events.ScheduleEvent(EVENT_MAGIC_GROUNDING, urand(14000, 19000));
58 events.ScheduleEvent(EVENT_COUNTERSPELL, urand(9000, 10000));
59 events.ScheduleEvent(EVENT_SHAZZRAH_GATE, 30000);
60 }
61
62 void ExecuteEvent(uint32 eventId) override
63 {
64 switch (eventId)
65 {
66 case EVENT_ARCANE_EXPLOSION:
67 {
68 DoCastVictim(SPELL_ARCANE_EXPLOSION);
69 events.RepeatEvent(urand(4000, 5000));
70 break;
71 }
72 case EVENT_SHAZZRAH_CURSE:
73 {
74 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true, -SPELL_SHAZZRAH_CURSE))
75 {
76 DoCast(target, SPELL_SHAZZRAH_CURSE);
77 }
78 events.RepeatEvent(urand(23000, 26000));
79 break;
80 }
81 case EVENT_MAGIC_GROUNDING:
82 {
83 DoCastSelf(SPELL_MAGIC_GROUNDING);
84 events.RepeatEvent(urand(7000, 9000));
85 break;
86 }
87 case EVENT_COUNTERSPELL:
88 {
89 DoCastAOE(SPELL_COUNTERSPELL);
90 events.RepeatEvent(urand(15000, 18000));
91 break;
92 }
93 case EVENT_SHAZZRAH_GATE:
94 {
95 DoCastAOE(SPELL_SHAZZRAH_GATE_DUMMY);
96 events.RescheduleEvent(EVENT_ARCANE_EXPLOSION, urand(3000, 6000));
97 events.RepeatEvent(45000);
98 break;
99 }
100 }
101 }
102 };
103
104 CreatureAI* GetAI(Creature* creature) const override
105 {
106 return GetMoltenCoreAI<boss_shazzrahAI>(creature);
107 }
108};

125 void FilterTargets(std::list<WorldObject*>& targets)
126 {
127 Unit* caster = GetCaster();
128 if (!targets.empty())
129 {
130 targets.remove_if([caster](WorldObject const* target) -> bool
131 {
132 Player const* plrTarget = target->ToPlayer();
133 // Should not target non player targets
134 if (!plrTarget)
135 {
136 return true;
137 }
138
139 // Should skip current victim
140 if (caster->GetVictim() == plrTarget)
141 {
142 return true;
143 }
144
145 // Should not target enemies within melee range
146 if (plrTarget->IsWithinMeleeRange(caster))
147 {
148 return true;
149 }
150
151 return false;
152 });
153 }
154
155 if (!targets.empty())
156 {
157 Acore::Containers::RandomResize(targets, 1);
158 }
159 }
160
161 void HandleScript(SpellEffIndex /*effIndex*/)
162 {
163 Unit* caster = GetCaster();
164 Unit* target = GetHitUnit();
165
166 if (caster && target)
167 {
168 target->CastSpell(caster, SPELL_SHAZZRAH_GATE, true);
169 caster->CastSpell(nullptr, SPELL_ARCANE_EXPLOSION);
170
171 if (Creature* creatureCaster = caster->ToCreature())
172 {
173 creatureCaster->GetThreatMgr().ResetAllThreat();
174 creatureCaster->GetThreatMgr().AddThreat(target, 1);
175 creatureCaster->AI()->AttackStart(target); // Attack the target which caster will teleport to.
176 }
177 }
178 }
179

184 }
