65 void DamageTaken(Unit* /*attacker*/, uint32& /*damage*/, DamageEffectType /*type*/, SpellSchoolMask /*school*/) override
66 {
67 if (_phase != PHASE_REGROWTH && me->HealthBelowPct(80.f))
68 {
69 _phase = PHASE_REGROWTH;
70 _scheduler.Schedule(25s, 30s, [this](TaskContext context)
71 {

73 me->GetMotionMaster()->MovePoint(POINT_RANDOM, randomPos);

75 {
