47 void OnCreatureCreate(Creature* creature) override
48 {
49 switch (creature->GetEntry())
50 {
51 case NPC_DND_CRAZED_APOTHECARY_GENERATOR:
52 _crazedApothecaryGeneratorGUIDs.push_back(creature->GetGUID());
53 break;
54 case NPC_APOTHECARY_HUMMEL:
55 _apothecaryHummel = creature->GetGUID();

57 case NPC_CRAZED_APOTHECARY:

61 }
62 break;
63 default:
64 break;
65 }

68 void OnGameObjectCreate(GameObject* gameobject) override
69 {
70 switch (gameobject->GetEntry())
71 {
72 case GO_COURTYARD_DOOR:
73 if (_encounters[TYPE_COURTYARD] == DONE)
74 HandleGameObject(ObjectGuid::Empty, true, gameobject);
75 break;
76 case GO_SORCERER_DOOR:
77 if (_encounters[TYPE_FENRUS_THE_DEVOURER] == DONE)
78 HandleGameObject(ObjectGuid::Empty, true, gameobject);
79 break;
80 case GO_ARUGAL_DOOR:
81 if (_encounters[TYPE_WOLF_MASTER_NANDOS] == DONE)
82 HandleGameObject(ObjectGuid::Empty, true, gameobject);
83 break;
84 default:
85 break;
86 }
87 }
88
89 void SetData(uint32 type, uint32 data) override
90 {
91 switch (type)
92 {
93 case TYPE_COURTYARD:
94 case TYPE_FENRUS_THE_DEVOURER:
95 case TYPE_WOLF_MASTER_NANDOS:
96 _encounters[type] = data;
97 break;
98 case DATA_SPAWN_VALENTINE_ADDS:
99 for (ObjectGuid guid : _crazedApothecaryGeneratorGUIDs)
100 {
101 if (Creature* generator = instance->GetCreature(guid))
102 {
103 generator->CastSpell(nullptr, SPELL_SUMMON_VALENTINE_ADD);
104 }
105 }
106 break;
107 default:
108 break;
109 }
110
111 if (data == DONE)
112 SaveToDB();
113 }
114
115 std::string GetSaveData() override
116 {
117 std::ostringstream saveStream;
118 saveStream << "S K " << _encounters[0] << ' ' << _encounters[1] << ' ' << _encounters[2];
119 return saveStream.str();
120 }
121
122 void Load(const char* in) override
123 {
124 if (!in)
125 return;
126
127 char dataHead1, dataHead2;
128 std::istringstream loadStream(in);
129 loadStream >> dataHead1 >> dataHead2;
130 if (dataHead1 == 'S' && dataHead2 == 'K')
131 {
132 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
133 {
134 loadStream >> _encounters[i];
135 if (_encounters[i] == IN_PROGRESS)
136 _encounters[i] = NOT_STARTED;
137 }
138 }
139 }
140
141 private:
142 uint32 _encounters[MAX_ENCOUNTERS];
143 GuidVector _crazedApothecaryGeneratorGUIDs;
144 ObjectGuid _apothecaryHummel;
145 };
