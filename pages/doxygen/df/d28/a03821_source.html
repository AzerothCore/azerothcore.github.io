199 boss_anzuAI(Creature* creature) : ScriptedAI(creature), summons(me)

210 void Reset() override
211 {
212 summons.DespawnAll();
213 if (InstanceScript* instance = me->GetInstanceScript())

215 instance->SetData(TYPE_ANZU_ENCOUNTER, NOT_STARTED);
216 }

218 void JustSummoned(Creature* summon) override

220 summons.Summon(summon);

223
224 void SummonedCreatureDies(Creature* summon, Unit*) override
225 {
226 summons.Despawn(summon);
227 summons.RemoveNotExisting();
228 if (summons.empty())

230 }

232 void EnterCombat(Unit* /*who*/) override
233 {
234 events.Reset();
235 events.ScheduleEvent(EVENT_SPELL_SCREECH, 14000);
236 events.ScheduleEvent(EVENT_SPELL_BOMB, 5000);
237 events.ScheduleEvent(EVENT_SPELL_CYCLONE, 8000);
238 events.ScheduleEvent(EVENT_ANZU_HEALTH1, 2000);
239 events.ScheduleEvent(EVENT_ANZU_HEALTH2, 2001);
240
241 if (InstanceScript* instance = me->GetInstanceScript())
242 instance->SetData(TYPE_ANZU_ENCOUNTER, IN_PROGRESS);
243 }
244
245 void JustDied(Unit* /*killer*/) override
246 {
247 if (InstanceScript* instance = me->GetInstanceScript())
248 instance->SetData(TYPE_ANZU_ENCOUNTER, DONE);
249 }
250
251 void SummonBroods()
252 {
253 Talk(SAY_SUMMON);
254 me->CastSpell(me, SPELL_BANISH_SELF, true);
255 for (uint8 i = 0; i < 5; ++i)
256 me->SummonCreature(23132 /*NPC_BROOD_OF_ANZU*/, me->GetPositionX() + 20 * cos((float)i), me->GetPositionY() + 20 * std::sin((float)i), me->GetPositionZ() + 25.0f, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 30000);
257 }
258
259 void UpdateAI(uint32 diff) override
260 {
261 if (talkTimer)
262 {
263 talkTimer += diff;
264 if (talkTimer >= 1000 && talkTimer < 10000)
265 {
266 Talk(SAY_ANZU_INTRO1);
267 talkTimer = 10000;
268 }
269 else if (talkTimer >= 16000)
270 {
271 me->ReplaceAllUnitFlags(UNIT_FLAG_NONE);
272 me->RemoveAurasDueToSpell(SPELL_SHADOWFORM);
273 Talk(SAY_ANZU_INTRO2);
274 talkTimer = 0;
275 }
276 }
277
278 if (!UpdateVictim())
279 return;
280
281 events.Update(diff);
282 if (me->HasUnitState(UNIT_STATE_CASTING | UNIT_STATE_STUNNED))
283 return;
284
285 switch (events.ExecuteEvent())
286 {
287 case EVENT_SPELL_SCREECH:
288 me->CastSpell(me, SPELL_PARALYZING_SCREECH, false);
289 events.RepeatEvent(23000);
290 events.DelayEvents(3000);
291 break;
292 case EVENT_SPELL_BOMB:
293 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
294 me->CastSpell(target, SPELL_SPELL_BOMB, false);
295 events.RepeatEvent(urand(16000, 24500));
296 events.DelayEvents(3000);
297 break;
298 case EVENT_SPELL_CYCLONE:
299 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 45.0f, true))
300 me->CastSpell(target, SPELL_CYCLONE, false);
301 events.RepeatEvent(urand(22000, 27000));
302 events.DelayEvents(3000);
303 break;
304 case EVENT_ANZU_HEALTH1:
305 if (me->HealthBelowPct(66))
306 {
307 SummonBroods();
308 events.DelayEvents(10000);
309 return;
310 }
311 events.RepeatEvent(1000);
312 break;
313 case EVENT_ANZU_HEALTH2:
314 if (me->HealthBelowPct(33))
315 {
316 SummonBroods();
317 events.DelayEvents(10000);
318 return;
319 }
320 events.RepeatEvent(1000);
321 break;
322 }
323
324 DoMeleeAttackIfReady();
325 }
326 };
