94bool BattlegroundSA::ResetObjs()
95{
96 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
97 SendTransportsRemove(itr->second);
98
99 uint32 atF = BG_SA_Factions[Attackers];
100 uint32 defF = BG_SA_Factions[Attackers ? TEAM_ALLIANCE : TEAM_HORDE];
101
102 for (uint8 i = 0; i < BG_SA_MAXOBJ; i++)
103 DelObject(i);
104
105 for (uint8 i = 0; i < BG_SA_MAXNPC; i++)
106 DelCreature(i);
107
108 for (uint8 i = BG_SA_MAXNPC; i < static_cast<uint16>(BG_SA_MAXNPC) + BG_SA_MAX_GY; i++)
109 DelCreature(i);
110
111 for (uint8 i = 0; i < 6; ++i)
112 GateStatus[i] = BG_SA_GATE_OK;
113
114 for (uint8 i = 0; i < BG_SA_BOAT_ONE; i++)
115 {
116 if (!AddObject(i, BG_SA_ObjEntries[i], BG_SA_ObjSpawnlocs[i][0], BG_SA_ObjSpawnlocs[i][1], BG_SA_ObjSpawnlocs[i][2], BG_SA_ObjSpawnlocs[i][3], 0, 0, 0, 0, RESPAWN_ONE_DAY))
117 return false;
118 }
119
120 for (uint8 i = BG_SA_BOAT_ONE; i < BG_SA_SIGIL_1; i++)
121 {
122 uint32 boatid = 0;
123 switch (i)
124 {
125 case BG_SA_BOAT_ONE:
126 boatid = Attackers ? BG_SA_BOAT_ONE_H : BG_SA_BOAT_ONE_A;
127 break;
128 case BG_SA_BOAT_TWO:
129 boatid = Attackers ? BG_SA_BOAT_TWO_H : BG_SA_BOAT_TWO_A;
130 break;
131 }
132 if (!AddObject(i, boatid, BG_SA_ObjSpawnlocs[i][0],
133 BG_SA_ObjSpawnlocs[i][1],
134 BG_SA_ObjSpawnlocs[i][2] + (Attackers ? -3.750f : 0),
135 BG_SA_ObjSpawnlocs[i][3], 0, 0, 0, 0, RESPAWN_ONE_DAY))
136 return false;
137 }
138
139 for (uint8 i = BG_SA_SIGIL_1; i < BG_SA_CENTRAL_FLAG; i++)
140 {
141 if (!AddObject(i, BG_SA_ObjEntries[i],
142 BG_SA_ObjSpawnlocs[i][0], BG_SA_ObjSpawnlocs[i][1],
143 BG_SA_ObjSpawnlocs[i][2], BG_SA_ObjSpawnlocs[i][3],
144 0, 0, 0, 0, RESPAWN_ONE_DAY))
145 return false;
146 }
147
148 // MAD props for Kiper for discovering those values - 4 hours of his work.
149 GetBGObject(BG_SA_BOAT_ONE)->SetTransportPathRotation(0.0f, 0.0f, 1.0f, 0.0002f);
150 GetBGObject(BG_SA_BOAT_TWO)->SetTransportPathRotation(0.0f, 0.0f, 1.0f, 0.00001f);
151 SpawnBGObject(BG_SA_BOAT_ONE, RESPAWN_IMMEDIATELY);
152 SpawnBGObject(BG_SA_BOAT_TWO, RESPAWN_IMMEDIATELY);
153
154 //Cannons and demolishers - NPCs are spawned
155 //By capturing GYs.
156 for (uint8 i = 0; i < BG_SA_DEMOLISHER_5; i++)
157 {
158 if (!AddCreature(BG_SA_NpcEntries[i], i,
159 BG_SA_NpcSpawnlocs[i][0], BG_SA_NpcSpawnlocs[i][1],
160 BG_SA_NpcSpawnlocs[i][2], BG_SA_NpcSpawnlocs[i][3], 600))
161 return false;
162 }
163
164 OverrideGunFaction();
165 DemolisherStartState(true);
166
167 for (uint8 i = BG_SA_GREEN_GATE; i <= BG_SA_TITAN_RELIC; i++)
168 {
169 SpawnBGObject(i, RESPAWN_IMMEDIATELY);
170 if (GameObject* go = GetBGObject(i))
171 {
172 go->setActive(true);
173 go->SetUInt32Value(GAMEOBJECT_FACTION, defF);
174 go->SetDestructibleBuildingModifyState(false);
175 }
176 }
177
178 GetBGObject(BG_SA_TITAN_RELIC)->SetUInt32Value(GAMEOBJECT_FACTION, atF);
179 GetBGObject(BG_SA_TITAN_RELIC)->Refresh();
180
181 TotalTime = 0s;
182 ShipsStarted = false;
183
184 //Graveyards!
185 for (uint8 i = 0; i < BG_SA_MAX_GY; i++)
186 {
187 GraveyardStruct const* sg = nullptr;
188 sg = sGraveyard->GetGraveyard(BG_SA_GYEntries[i]);
189
190 if (!sg)
191 {
192 LOG_ERROR("bg.battleground", "SOTA: Can't find GY entry {}", BG_SA_GYEntries[i]);
193 return false;
194 }
195
196 if (i == BG_SA_BEACH_GY)
197 {
198 GraveyardStatus[i] = Attackers;
199 AddSpiritGuide(i + BG_SA_MAXNPC, sg->x, sg->y, sg->z, BG_SA_GYOrientation[i], Attackers);
200 }
201 else
202 {
203 GraveyardStatus[i] = GetOtherTeamId(Attackers);
204 if (!AddSpiritGuide(i + BG_SA_MAXNPC, sg->x, sg->y, sg->z, BG_SA_GYOrientation[i], GetOtherTeamId(Attackers)))
205 LOG_ERROR("bg.battleground", "SOTA: couldn't spawn GY: {}", i);
206 }
207 }
208
209 //GY capture points
210 for (uint8 i = BG_SA_CENTRAL_FLAG; i < BG_SA_PORTAL_DEFFENDER_BLUE; i++)
211 {
212 AddObject(i, (BG_SA_ObjEntries[i] - (Attackers == TEAM_ALLIANCE ? 1 : 0)),
213 BG_SA_ObjSpawnlocs[i][0], BG_SA_ObjSpawnlocs[i][1],
214 BG_SA_ObjSpawnlocs[i][2], BG_SA_ObjSpawnlocs[i][3],
215 0, 0, 0, 0, RESPAWN_ONE_DAY);
216 GetBGObject(i)->SetUInt32Value(GAMEOBJECT_FACTION, atF);
217 }
218
219 for (uint8 i = BG_SA_PORTAL_DEFFENDER_BLUE; i < BG_SA_BOMB; i++)
220 {
221 AddObject(i, BG_SA_ObjEntries[i],
222 BG_SA_ObjSpawnlocs[i][0], BG_SA_ObjSpawnlocs[i][1],
223 BG_SA_ObjSpawnlocs[i][2], BG_SA_ObjSpawnlocs[i][3],
224 0, 0, 0, 0, RESPAWN_ONE_DAY);
225 GetBGObject(i)->SetUInt32Value(GAMEOBJECT_FACTION, defF);
226 }
227
228 UpdateObjectInteractionFlags();
229
230 for (uint8 i = BG_SA_BOMB; i < BG_SA_MAXOBJ; i++)
231 {
232 AddObject(i, BG_SA_ObjEntries[BG_SA_BOMB],
233 BG_SA_ObjSpawnlocs[i][0], BG_SA_ObjSpawnlocs[i][1],
234 BG_SA_ObjSpawnlocs[i][2], BG_SA_ObjSpawnlocs[i][3],
235 0, 0, 0, 0, RESPAWN_ONE_DAY);
236 GetBGObject(i)->SetUInt32Value(GAMEOBJECT_FACTION, atF);
237 }
238
239 //Player may enter BEFORE we set up bG - lets update his worldstates anyway...
240 UpdateWorldState(BG_SA_RIGHT_GY_HORDE, GraveyardStatus[BG_SA_RIGHT_CAPTURABLE_GY] == TEAM_HORDE ? 1 : 0);
241 UpdateWorldState(BG_SA_LEFT_GY_HORDE, GraveyardStatus[BG_SA_LEFT_CAPTURABLE_GY] == TEAM_HORDE ? 1 : 0);
242 UpdateWorldState(BG_SA_CENTER_GY_HORDE, GraveyardStatus[BG_SA_CENTRAL_CAPTURABLE_GY] == TEAM_HORDE ? 1 : 0);
243
244 UpdateWorldState(BG_SA_RIGHT_GY_ALLIANCE, GraveyardStatus[BG_SA_RIGHT_CAPTURABLE_GY] == TEAM_ALLIANCE ? 1 : 0);
245 UpdateWorldState(BG_SA_LEFT_GY_ALLIANCE, GraveyardStatus[BG_SA_LEFT_CAPTURABLE_GY] == TEAM_ALLIANCE ? 1 : 0);
246 UpdateWorldState(BG_SA_CENTER_GY_ALLIANCE, GraveyardStatus[BG_SA_CENTRAL_CAPTURABLE_GY] == TEAM_ALLIANCE ? 1 : 0);
247
248 if (Attackers == TEAM_ALLIANCE)
249 {
250 UpdateWorldState(BG_SA_ALLY_ATTACKS, 1);
251 UpdateWorldState(BG_SA_HORDE_ATTACKS, 0);
252
253 UpdateWorldState(BG_SA_RIGHT_ATT_TOKEN_ALL, 1);
254 UpdateWorldState(BG_SA_LEFT_ATT_TOKEN_ALL, 1);
255 UpdateWorldState(BG_SA_RIGHT_ATT_TOKEN_HRD, 0);
256 UpdateWorldState(BG_SA_LEFT_ATT_TOKEN_HRD, 0);
257
258 UpdateWorldState(BG_SA_HORDE_DEFENCE_TOKEN, 1);
259 UpdateWorldState(BG_SA_ALLIANCE_DEFENCE_TOKEN, 0);
260 }
261 else
262 {
263 UpdateWorldState(BG_SA_HORDE_ATTACKS, 1);
264 UpdateWorldState(BG_SA_ALLY_ATTACKS, 0);
265
266 UpdateWorldState(BG_SA_RIGHT_ATT_TOKEN_ALL, 0);
267 UpdateWorldState(BG_SA_LEFT_ATT_TOKEN_ALL, 0);
268 UpdateWorldState(BG_SA_RIGHT_ATT_TOKEN_HRD, 1);
269 UpdateWorldState(BG_SA_LEFT_ATT_TOKEN_HRD, 1);
270
271 UpdateWorldState(BG_SA_HORDE_DEFENCE_TOKEN, 0);
272 UpdateWorldState(BG_SA_ALLIANCE_DEFENCE_TOKEN, 1);
273 }
274
275 UpdateWorldState(BG_SA_PURPLE_GATEWS, 1);
276 UpdateWorldState(BG_SA_RED_GATEWS, 1);
277 UpdateWorldState(BG_SA_BLUE_GATEWS, 1);
278 UpdateWorldState(BG_SA_GREEN_GATEWS, 1);
279 UpdateWorldState(BG_SA_YELLOW_GATEWS, 1);
280 UpdateWorldState(BG_SA_ANCIENT_GATEWS, 1);
281
282 for (int i = BG_SA_BOAT_ONE; i <= BG_SA_BOAT_TWO; i++)
283 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
284 SendTransportInit(itr->second);
285
286 // set status manually so preparation is cast correctly in 2nd round too
287 SetStatus(STATUS_WAIT_JOIN);
288
289 TeleportPlayers();
290 return true;
291}

315void BattlegroundSA::PostUpdateImpl(uint32 diff)
316{
317 if (InitSecondRound)
318 {
319 if (UpdateWaitTimer < diff)
320 {
321 if (!SignaledRoundTwo)
322 {
323 SignaledRoundTwo = true;
324 InitSecondRound = false;
325 SendBroadcastText(BG_SA_TEXT_ROUND_TWO_START_ONE_MINUTE, CHAT_MSG_BG_SYSTEM_NEUTRAL);
326 }
327 }
328 else
329 {
330 UpdateWaitTimer -= diff;
331 return;
332 }
333 }
334 TotalTime += Milliseconds(diff);
335
336 if (Status == BG_SA_WARMUP)
337 {
338 EndRoundTimer = BG_SA_ROUNDLENGTH;
339 if (TotalTime >= BG_SA_WARMUPLENGTH)
340 {
341 TotalTime = 0s;
342 ToggleTimer();
343 DemolisherStartState(false);
344 Status = BG_SA_ROUND_ONE;
345 StartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, (Attackers == TEAM_ALLIANCE) ? 23748 : 21702);
346 }
347 if (TotalTime >= BG_SA_BOAT_START)
348 StartShips();
349 return;
350 }
351 else if (Status == BG_SA_SECOND_WARMUP)
352 {
353 if (RoundScores[0].time < BG_SA_ROUNDLENGTH)
354 EndRoundTimer = RoundScores[0].time;
355 else
356 EndRoundTimer = BG_SA_ROUNDLENGTH;
357
358 if (TotalTime >= 1min)
359 {
360 SendWarningToAll(LANG_BG_SA_HAS_BEGUN);
361 TotalTime = 0s;
362 ToggleTimer();
363 DemolisherStartState(false);
364 Status = BG_SA_ROUND_TWO;
365 StartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, (Attackers == TEAM_ALLIANCE) ? 23748 : 21702);
366
367 // status was set to STATUS_WAIT_JOIN manually for Preparation, set it back now
368 SetStatus(STATUS_IN_PROGRESS);
369 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
370 itr->second->RemoveAurasDueToSpell(SPELL_PREPARATION);
371 }
372 if (TotalTime >= 30s)
373 {
374 if (!SignaledRoundTwoHalfMin)
375 {
376 SignaledRoundTwoHalfMin = true;
377 SendBroadcastText(BG_SA_TEXT_ROUND_TWO_START_HALF_MINUTE, CHAT_MSG_BG_SYSTEM_NEUTRAL);
378 }
379 }
380 StartShips();
381 return;
382 }
383 else if (GetStatus() == STATUS_IN_PROGRESS)
384 {
385 if (Status == BG_SA_ROUND_ONE)
386 {
387 if (TotalTime >= BG_SA_ROUNDLENGTH || _relicClicked)
388 {
389 if (_relicClicked)
390 {
391 RoundScores[0].winner = Attackers;
392 RoundScores[0].time = TotalTime;
393 //Achievement Storm the Beach (1310)
394 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
395 if (itr->second->GetTeamId() == Attackers)
396 itr->second->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, 65246);
397 }
398 else
399 {
400 // cast this before Attackers variable is switched
401 // cast this spell only upon timer end, no other ability for defenders to win :)
402 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
403 itr->second->CastSpell(itr->second, SPELL_SA_END_OF_ROUND, true);
404
405 RoundScores[0].winner = Attackers;
406 RoundScores[0].time = BG_SA_ROUNDLENGTH;
407 }
408
409 _relicClicked = false;
410 Attackers = (Attackers == TEAM_ALLIANCE) ? TEAM_HORDE : TEAM_ALLIANCE;
411 Status = BG_SA_SECOND_WARMUP;
412 TotalTime = 0s;
413 ToggleTimer();
414 SendWarningToAll(LANG_BG_SA_ROUND_ONE_END);
415 UpdateWaitTimer = 5000;
416 SignaledRoundTwo = false;
417 SignaledRoundTwoHalfMin = false;
418 InitSecondRound = true;
419 ResetObjs();
420
421 return;
422 }
423 }
424 else if (Status == BG_SA_ROUND_TWO)
425 {
426 if (TotalTime >= EndRoundTimer)
427 {
428 // cast this spell only upon timer end, no other ability for defenders to win :)
429 for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
430 itr->second->CastSpell(itr->second, SPELL_SA_END_OF_ROUND, true);
431
432 RoundScores[1].time = BG_SA_ROUNDLENGTH;
433 RoundScores[1].winner = GetOtherTeamId(Attackers);
434
435 if (RoundScores[0].time == RoundScores[1].time)
436 EndBattleground(TEAM_NEUTRAL);
437 else if (RoundScores[0].time < RoundScores[1].time)
438 EndBattleground(RoundScores[0].winner);
439 else
440 EndBattleground(RoundScores[1].winner);
441 return;
442 }
443 }
444 if (Status == BG_SA_ROUND_ONE || Status == BG_SA_ROUND_TWO)
445 {
446 SendTime();
447 UpdateDemolisherSpawns();
448 }
449 }

780 float x, y;

802 mindist = dist;
803 closest = ret;
804 }
805 }
806 if (!closest && GraveyardStatus[BG_SA_BEACH_GY] == player->GetTeamId())
