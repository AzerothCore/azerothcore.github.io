53class boss_toravon : public CreatureScript
54{
55public:
56 boss_toravon() : CreatureScript("boss_toravon") { }
57
58 struct boss_toravonAI : public ScriptedAI
59 {
60 boss_toravonAI(Creature* creature) : ScriptedAI(creature), summons(me)
61 {
62 pInstance = me->GetInstanceScript();
63 }
64
65 InstanceScript* pInstance;
66
67 EventMap events;
68 SummonList summons;
69
70 void Reset() override
71 {
72 events.Reset();
73 summons.DespawnAll();
74 if (pInstance)
75 {
76 if (pInstance->GetData(DATA_STONED))
77 {
78 if (Aura* aur = me->AddAura(SPELL_STONED_AURA, me))
79 {
80 aur->SetMaxDuration(60 * MINUTE * IN_MILLISECONDS);
81 aur->SetDuration(60 * MINUTE * IN_MILLISECONDS);
82 }
83 }
84 pInstance->SetData(EVENT_TORAVON, NOT_STARTED);
85 }
86 }
87
88 void AttackStart(Unit* who) override
89 {
90 if (me->HasAura(SPELL_STONED_AURA))
91 return;
92 ScriptedAI::AttackStart(who);
93 }
94
95 void EnterCombat(Unit* /*who*/) override
96 {
97 me->CastSpell(me, SPELL_FROZEN_MALLET, true);
98
99 events.ScheduleEvent(EVENT_FROZEN_ORB_STALKER, 12000);
100 events.ScheduleEvent(EVENT_FREEZING_GROUND, 7000);
101 events.ScheduleEvent(EVENT_CAST_WHITEOUT, 25000); // schedule FIRST whiteout event in 25 seconds -1 for compesate updateai 2seconds check delay
102
103 if (pInstance)
104 pInstance->SetData(EVENT_TORAVON, IN_PROGRESS);
105 }
106
107 void JustDied(Unit*) override
108 {
109 if (pInstance)
110 {
111 pInstance->SetData(EVENT_TORAVON, DONE);
112 pInstance->DoRemoveAurasDueToSpellOnPlayers(SPELL_WHITEOUT);
113 }
114 summons.DespawnAll();
115 }
116
117 void JustSummoned(Creature* cr) override
118 {
119 summons.Summon(cr);
120 }
121
122 void UpdateAI(uint32 diff) override
123 {
124 if (!UpdateVictim())
125 return;
126
127 events.Update(diff);
128 if (me->HasUnitState(UNIT_STATE_CASTING))
129 return;
130
131 switch (events.ExecuteEvent())
132 {
133 case EVENT_FREEZING_GROUND:
134 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
135 me->CastSpell(target, SPELL_FREEZING_GROUND, false);
136 events.RepeatEvent(20000);
137 break;
138 case EVENT_FROZEN_ORB_STALKER:
139 me->CastCustomSpell(SPELL_FROZEN_ORB, SPELLVALUE_MAX_TARGETS, RAID_MODE(1, 3), me, false);
140 events.RepeatEvent(30000);
141 break;
142 case EVENT_CAST_WHITEOUT:
143 me->CastSpell(me, SPELL_WHITEOUT, false);
144 events.ScheduleEvent(EVENT_CAST_WHITEOUT_GROUND_EFFECT, 1000); // triggers after 1 sec "plus 1 from trigger to cast visual"
145 events.RepeatEvent(40000); // next whiteout instead first 25 SEC is now 45 SEC
146 break;
147 case EVENT_CAST_WHITEOUT_GROUND_EFFECT: // Whiteout Ground effect trigger
148 if (Unit* whiteOutGround = me->SummonCreature(NPC_WHITEOUT_GROUND_EFFECT, -43.3316, -288.708, 92.2511, 1.58825, TEMPSUMMON_TIMED_DESPAWN, 4000))
149 whiteOutGround->CastSpell(whiteOutGround, SPELL_WHITEOUT_VISUAL, false); // Cast the spell
150 break;
151 }
152
153 DoMeleeAttackIfReady();
154 }
155 };
156
157 CreatureAI* GetAI(Creature* creature) const override
158 {
159 return GetVaultOfArchavonAI<boss_toravonAI>(creature);
160 }
161};

174 uint32 switchTimer;
175
176 void Reset() override
177 {
178 switchTimer = 9000;
179 me->CastSpell(me, SPELL_FROZEN_ORB_AURA, true);
180 me->CastSpell(me, SPELL_FROZEN_ORB_DMG, true);
181 }
182
183 void EnterCombat(Unit* /*who*/) override
184 {
185 me->SetInCombatWithZone();
186 }
187
188 void UpdateAI(uint32 diff) override
189 {
190 switchTimer += diff;
191 if (switchTimer >= 10000)
192 {
193 switchTimer = 0;
194 me->GetThreatMgr().ResetAllThreat();
195 if (Player* player = SelectTargetFromPlayerList(100.0f))
196 me->AddThreat(player, 100000.0f);
197 }
198 }
199 };

203 return GetVaultOfArchavonAI<npc_frozen_orbAI>(creature);
