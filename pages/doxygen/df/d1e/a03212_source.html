98 npc_taldaram_flamesphere(Creature *pCreature) : NullCreatureAI(pCreature),

135 uiDespawnTimer = 13000;

139 {
140 DoCastSelf(SPELL_FLAME_SPHERE_DEATH_EFFECT);
141 }
142
143 void UpdateAI(uint32 diff) override
144 {
145 if (moveTimer)
146 {
147 if (moveTimer <= diff)
148 {
149 DoCastSelf(SPELL_FLAME_SPHERE_PERIODIC);
150 float angleOffset = 0.0f;
151
152 switch (me->GetEntry())
153 {
154 case NPC_FLAME_SPHERE_1:
155 break;
156 case NPC_FLAME_SPHERE_2:
157 angleOffset = DATA_SPHERE_ANGLE_OFFSET;
158 break;
159 case NPC_FLAME_SPHERE_3:
160 angleOffset = -DATA_SPHERE_ANGLE_OFFSET;
161 break;
162 default:
163 return;
164 }
165
166 float angle = me->GetAngle(&victimPos) + angleOffset;
167 float x = me->GetPositionX() + DATA_SPHERE_DISTANCE * cos(angle);
168 float y = me->GetPositionY() + DATA_SPHERE_DISTANCE * std::sin(angle);
169 me->GetMotionMaster()->MovePoint(POINT_ORB, x, y, me->GetPositionZ());
170
171 moveTimer = 0;
172 }
173 else
174 {
175 moveTimer -= diff;
176 }
177 }
178
179 if (uiDespawnTimer)
180 {
181 if (uiDespawnTimer <= diff)
182 {
183 me->DisappearAndDie();
184 uiDespawnTimer = 0;
185 }
186 else
187 uiDespawnTimer -= diff;
188 }
189 }
190
191 void SetVictimPos(Position const& pos)
192 {
193 victimPos.Relocate(pos);
194 }
195
196private:
197 Position victimPos;
198 InstanceScript* instance;
199 uint32 uiDespawnTimer;
200 uint32 moveTimer;

210 void InitializeAI() override
211 {
212 BossAI::InitializeAI();
213
214 // Event not started
215 if (instance->GetData(DATA_TELDRAM_SPHERE1) != DONE || instance->GetData(DATA_TELDRAM_SPHERE2) != DONE)
216 {
217 me->SetImmuneToAll(true);

251 {
252 DoCastSelf(SPELL_HOVER_FALL);
253 me->GetMotionMaster()->Clear();

255 Talk(SAY_REMOVE_PRISON);

260 me->SetDisableGravity(false);
261 me->SetHover(false);
262 me->RemoveAllAuras();
263 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE| UNIT_FLAG_NOT_SELECTABLE);
264 me->SetImmuneToAll(false);
265 me->UpdatePosition(me->GetHomePosition(), true);
266 }
267 summons.DespawnEntry(NPC_JEDOGA_CONTROLLER);
268 }
269 }

365 void UpdateAI(uint32 diff) override
366 {
367 if (!UpdateVictim())
368 {
369 return;
370 }
371
372 events.Update(diff);
373
374 if (me->HasUnitState(UNIT_STATE_CASTING))
375 {
376 return;
377 }
378
379 while (uint32 const eventId = events.ExecuteEvent())
380 {
381 switch (eventId)
382 {
383 case EVENT_PRINCE_BLOODTHIRST:
384 {
385 DoCastSelf(SPELL_BLOODTHIRST);
386 events.RepeatEvent(10000);
387 break;
388 }
389 case EVENT_PRINCE_FLAME_SPHERES:
390 {
391 if (Unit* victim = me->GetVictim())
392 {
393 DoCast(victim, SPELL_CONJURE_FLAME_SPHERE);
394 victimSperePos = *victim;
395 }
396
397 if (!events.GetNextEventTime(EVENT_PRINCE_VANISH))
398 {
399 events.RescheduleEvent(EVENT_PRINCE_VANISH, 14000);
400 }
401 else
402 {
403 // Make sure that Vanish won't get triggered at same time as sphere summon
404 events.DelayEvents(4000);
405 }
406
407 events.RepeatEvent(15000);
408 break;
409 }
410 case EVENT_PRINCE_VANISH:
411 {
412 //Count alive players
413 uint8 count = 0;
414 std::list<HostileReference*> const t_list = me->GetThreatMgr().GetThreatList();
415 if (!t_list.empty())
416 {
417 for (HostileReference const* reference : t_list)
418 {
419 if (reference)
420 {
421 Unit const* pTarget = ObjectAccessor::GetUnit(*me, reference->getUnitGuid());
422 if (pTarget && pTarget->GetTypeId() == TYPEID_PLAYER && pTarget->IsAlive())
423 {
424 ++count;
425 }
426 }
427 }
428 }
429
430 // He only vanishes if there are 3 or more alive players
431 if (count > 2)
432 {
433 Talk(SAY_VANISH);
434 DoCastSelf(SPELL_VANISH, false);
435 if (Unit* pEmbraceTarget = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
436 {
437 vanishTarget_GUID = pEmbraceTarget->GetGUID();
438 }
439
440 events.CancelEvent(EVENT_PRINCE_FLAME_SPHERES);
441 events.CancelEvent(EVENT_PRINCE_BLOODTHIRST);
442 events.ScheduleEvent(EVENT_PRINCE_VANISH_RUN, 2499);
443 }
444 break;
445 }
446 case EVENT_PRINCE_VANISH_RUN:
447 {
448 if (Unit* _vanishTarget = ObjectAccessor::GetUnit(*me, vanishTarget_GUID))
449 {
450 vanishDamage = 0;
451 DoCast(_vanishTarget, SPELL_SHADOWSTEP);
452 me->CastSpell(_vanishTarget, SPELL_EMBRACE_OF_THE_VAMPYR, false);
453 me->RemoveAura(SPELL_VANISH);
454 }
455
456 events.ScheduleEvent(EVENT_PRINCE_RESCHEDULE, 20000);
457 break;
458 }
459 case EVENT_PRINCE_RESCHEDULE:
460 {
461 ScheduleCombatEvents();
462 break;
463 }
464 }
465
466 if (me->HasUnitState(UNIT_STATE_CASTING))
467 {
468 return;
469 }
470 }
471
472 if (me->IsVisible())
473 {
474 DoMeleeAttackIfReady();
475 }
476 }
477
478private:
479 Position victimSperePos;
480 ObjectGuid vanishTarget_GUID;
481 uint32 vanishDamage;
482
483 void ScheduleCombatEvents()
484 {
485 events.Reset();
486 events.RescheduleEvent(EVENT_PRINCE_FLAME_SPHERES, 10000);
487 events.RescheduleEvent(EVENT_PRINCE_BLOODTHIRST, 10000);
488 vanishTarget_GUID.Clear();
489 vanishDamage = 0;
490 }
