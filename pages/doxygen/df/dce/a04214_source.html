526 struct npc_air_force_botsAI : public ScriptedAI
527 {
528 npc_air_force_botsAI(Creature* creature) : ScriptedAI(creature)
529 {
530 SpawnAssoc = nullptr;
531 SpawnedGUID.Clear();
532
533 // find the correct spawnhandling
534 static uint32 entryCount = sizeof(spawnAssociations) / sizeof(SpawnAssociation);
535
536 for (uint8 i = 0; i < entryCount; ++i)
537 {
538 if (spawnAssociations[i].thisCreatureEntry == creature->GetEntry())
539 {
540 SpawnAssoc = &spawnAssociations[i];
541 break;
542 }
543 }
544
545 if (!SpawnAssoc)
546 LOG_ERROR("sql.sql", "TCSR: Creature template entry {} has ScriptName npc_air_force_bots, but it's not handled by that script", creature->GetEntry());
547 else
548 {
549 CreatureTemplate const* spawnedTemplate = sObjectMgr->GetCreatureTemplate(SpawnAssoc->spawnedCreatureEntry);
550
551 if (!spawnedTemplate)
552 {
553 LOG_ERROR("sql.sql", "TCSR: Creature template entry {} does not exist in DB, which is required by npc_air_force_bots", SpawnAssoc->spawnedCreatureEntry);
554 SpawnAssoc = nullptr;
555 return;
556 }
557 }

590 void MoveInLineOfSight(Unit* who) override
591
592 {
593 if (!SpawnAssoc)
594 return;
595
596 if (me->IsValidAttackTarget(who))
597 {
598 Player* playerTarget = who->ToPlayer();
599
600 // airforce guards only spawn for players
601 if (!playerTarget)
602 return;
603
604 Creature* lastSpawnedGuard = !SpawnedGUID ? nullptr : GetSummonedGuard();
605
606 // prevent calling ObjectAccessor::GetUnit at next MoveInLineOfSight call - speedup
607 if (!lastSpawnedGuard)
608 SpawnedGUID.Clear();
609
610 switch (SpawnAssoc->spawnType)
611 {
612 case SPAWNTYPE_ALARMBOT:
613 {
614 if (!who->IsWithinDistInMap(me, RANGE_GUARDS_MARK))
615 return;
616
617 Aura* markAura = who->GetAura(SPELL_GUARDS_MARK);
618 if (markAura)
619 {
620 // the target wasn't able to move out of our range within 25 seconds
621 if (!lastSpawnedGuard)
622 {
623 lastSpawnedGuard = SummonGuard();
624
625 if (!lastSpawnedGuard)
626 return;
627 }
628
629 if (markAura->GetDuration() < AURA_DURATION_TIME_LEFT)
630 if (!lastSpawnedGuard->GetVictim())
631 lastSpawnedGuard->AI()->AttackStart(who);
632 }
633 else
634 {
635 if (!lastSpawnedGuard)
636 lastSpawnedGuard = SummonGuard();
637
638 if (!lastSpawnedGuard)
639 return;
640
641 lastSpawnedGuard->CastSpell(who, SPELL_GUARDS_MARK, true);
642 }
643 break;
644 }
645 case SPAWNTYPE_TRIPWIRE_ROOFTOP:
646 {
647 if (!who->IsWithinDistInMap(me, RANGE_TRIPWIRE))
648 return;
649
650 if (!lastSpawnedGuard)
651 lastSpawnedGuard = SummonGuard();
652
653 if (!lastSpawnedGuard)
654 return;
655
656 // ROOFTOP only triggers if the player is on the ground
657 if (!playerTarget->IsFlying() && !lastSpawnedGuard->GetVictim())
658 lastSpawnedGuard->AI()->AttackStart(who);
659
660 break;
661 }
662 }
663 }
664 }
