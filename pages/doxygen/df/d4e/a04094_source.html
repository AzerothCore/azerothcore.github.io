716 Acore::AllWorldObjectsInRange objects(me, searchDistance);
717 Acore::WorldObjectListSearcher<Acore::AllWorldObjectsInRange> searcher(me, ClusterList, objects);
718 Cell::VisitAllObjects(me, searcher, searchDistance);
719
720 for (std::list<WorldObject*>::const_iterator i = ClusterList.begin(); i != ClusterList.end(); ++i)
721 {
722 if (GameObject* go = (*i)->ToGameObject())
723 {
