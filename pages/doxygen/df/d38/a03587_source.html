114 struct boss_auriayaAI : public ScriptedAI
115 {
116 boss_auriayaAI(Creature* pCreature) : ScriptedAI(pCreature), summons(pCreature)
117 {
118 m_pInstance = pCreature->GetInstanceScript();
119 }
120
121 InstanceScript* m_pInstance;
122 EventMap events;
123 SummonList summons;
124
125 bool _feralDied;
126 bool _nineLives;
127
128 void Reset() override
129 {
130 _feralDied = false;
131 _nineLives = false;
132
133 events.Reset();
134 EntryCheckPredicate pred(NPC_FERAL_DEFENDER);
135 summons.DoAction(ACTION_DESPAWN_ADDS, pred);
136 summons.DespawnAll();
137
138 if (m_pInstance)
139 m_pInstance->SetData(TYPE_AURIAYA, NOT_STARTED);
140
141 for (uint8 i = 0; i < RAID_MODE(2, 4); ++i)
142 me->SummonCreature(NPC_SANCTUM_SENTRY, me->GetPositionX() + urand(4, 12), me->GetPositionY() + urand(4, 12), me->GetPositionZ());
143
144 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_INTERRUPT_CAST, false);
145 }
146
147 uint32 GetData(uint32 param) const override
148 {
149 if (param == DATA_CRAZY_CAT)
150 return !_feralDied;
151 else if (param == DATA_NINE_LIVES)
152 return _nineLives;
153
154 return 0;
155 }
156
157 void JustSummoned(Creature* cr) override
158 {
159 if (cr->GetEntry() == NPC_SANCTUM_SENTRY)
160 cr->GetMotionMaster()->MoveFollow(me, 6, rand_norm() * 2 * 3.14f);
161 else
162 cr->SetInCombatWithZone();
163
164 summons.Summon(cr);
165 }
166
167 void SummonedCreatureDies(Creature* cr, Unit*) override
168 {
169 if (cr->GetEntry() == NPC_SANCTUM_SENTRY)
170 _feralDied = true;
171 }
172
173 void JustReachedHome() override { me->setActive(false); }
174
175 void EnterCombat(Unit* /*who*/) override
176 {
177 if (m_pInstance)
178 m_pInstance->SetData(TYPE_AURIAYA, IN_PROGRESS);
179
180 events.ScheduleEvent(EVENT_TERRIFYING_SCREECH, 35000);
181 events.ScheduleEvent(EVENT_SONIC_SCREECH, 45000);
182 events.ScheduleEvent(EVENT_GUARDIAN_SWARM, 70000);
183 events.ScheduleEvent(EVENT_SUMMON_FERAL_DEFENDER, 60000);
184 events.ScheduleEvent(EVENT_SENTINEL_BLAST, 36000);
185 events.ScheduleEvent(EVENT_ENRAGE, 600000);
186
187 summons.DoZoneInCombat(NPC_SANCTUM_SENTRY);
188
189 Talk(SAY_AGGRO);
190 me->setActive(true);
191 }
192
193 void KilledUnit(Unit* victim) override
194 {
195 if (victim->GetTypeId() != TYPEID_PLAYER || urand(0, 2))
196 return;
197
198 Talk(SAY_SLAY);
199 }
200
201 void JustDied(Unit* /*killer*/) override
202 {
203 if (m_pInstance)
204 m_pInstance->SetData(TYPE_AURIAYA, DONE);
205
206 EntryCheckPredicate pred(NPC_FERAL_DEFENDER);
207 summons.DoAction(ACTION_DESPAWN_ADDS, pred);
208 summons.DespawnAll();
209 Talk(EMOTE_DEATH);
210 }
211
212 void DoAction(int32 param) override
213 {
214 if (param == ACTION_FERAL_DEATH_WITH_STACK)
215 events.ScheduleEvent(EVENT_RESPAWN_FERAL_DEFENDER, 25000);
216 else if (param == ACTION_FERAL_DEATH)
217 _nineLives = true;
218 }
219
220 void UpdateAI(uint32 diff) override
221 {
222 if (!UpdateVictim())
223 return;
224
225 events.Update(diff);
226 if (me->HasUnitState(UNIT_STATE_CASTING))
227 return;
228
229 switch (events.ExecuteEvent())
230 {
231 case EVENT_SUMMON_FERAL_DEFENDER:
232 Talk(EMOTE_DEFFENDER);
233 me->CastSpell(me, SPELL_ACTIVATE_FERAL_DEFENDER, true);
234 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_INTERRUPT_CAST, true);
235 events.ScheduleEvent(EVENT_REMOVE_IMMUNE, 3000);
236 break;
237 case EVENT_REMOVE_IMMUNE:
238 me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_INTERRUPT_CAST, false);
239 break;
240 case EVENT_TERRIFYING_SCREECH:
241 Talk(EMOTE_FEAR);
242 me->CastSpell(me, SPELL_TERRIFYING_SCREECH, false);
243 events.RepeatEvent(35000);
244 break;
245 case EVENT_SONIC_SCREECH:
246 me->CastSpell(me, SPELL_SONIC_SCREECH, false);
247 events.RepeatEvent(50000);
248 break;
249 case EVENT_GUARDIAN_SWARM:
250 me->CastSpell(me->GetVictim(), SPELL_GUARDIAN_SWARM, false);
251 events.RepeatEvent(40000);
252 break;
253 case EVENT_SENTINEL_BLAST:
254 me->CastSpell(me, SPELL_SENTINEL_BLAST, false);
255 events.RepeatEvent(35000);
256 events.DelayEvents(5000, 0);
257 break;
258 case EVENT_RESPAWN_FERAL_DEFENDER:
259 {
260 EntryCheckPredicate pred(NPC_FERAL_DEFENDER);
261 summons.DoAction(ACTION_FERAL_RESPAWN, pred);
262 break;
263 }
264 case EVENT_ENRAGE:
265 Talk(SAY_BERSERK);
266 me->CastSpell(me, SPELL_ENRAGE, true);
267 break;
268 }
269
270 DoMeleeAttackIfReady();
271 }
272 };

307 void UpdateAI(uint32 diff) override
308 {

310 return;
311
312 _savagePounceTimer += diff;
313 _ripFleshTimer += diff;
314
315 if (_savagePounceTimer >= 5000)
316 {
317 float dist = me->GetDistance(me->GetVictim());
318 if (dist >= 8 && dist < 25 && me->IsWithinLOSInMap(me->GetVictim()))
319 {
320 me->CastSpell(me->GetVictim(), SPELL_SAVAGE_POUNCE, false);
321 _savagePounceTimer = 0;
322 return;
323 }
324 _savagePounceTimer = 200;
325 }
326 else if (_ripFleshTimer >= 10000)
327 {
328 me->CastSpell(me->GetVictim(), SPELL_RIP_FLESH, false);
329 _ripFleshTimer = 0;
330 }
331
332 DoMeleeAttackIfReady();
333 }

351 int32 _feralRushTimer;
352 int32 _feralPounceTimer;
353 uint8 _feralEssenceStack;
354 SummonList summons;
355
356 void Reset() override
357 {
358 summons.DespawnAll();
359 _feralRushTimer = 3000;
360 _feralPounceTimer = 0;
361 _feralEssenceStack = 8;
362
363 if (Aura* aur = me->AddAura(SPELL_FERAL_ESSENCE, me))
364 aur->SetStackAmount(_feralEssenceStack);
365 }
366
367 void JustDied(Unit*) override
368 {
369 // inform about our death, start timer
370 if (me->GetInstanceScript())
371 if (Creature* cr = ObjectAccessor::GetCreature(*me, me->GetInstanceScript()->GetGuidData(TYPE_AURIAYA)))
372 cr->AI()->DoAction(_feralEssenceStack ? ACTION_FERAL_DEATH_WITH_STACK : ACTION_FERAL_DEATH);
373
374 if (_feralEssenceStack)
375 {
376 if (Creature* cr = me->SummonCreature(NPC_SEEPING_FERAL_ESSENCE, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0.0f))
377 summons.Summon(cr);
378
379 --_feralEssenceStack;
380 }
381 }
382
383 void DoAction(int32 param) override
384 {
385 if (param == ACTION_FERAL_RESPAWN)
386 {
387 me->setDeathState(JUST_RESPAWNED);
388
389 if (Player* target = SelectTargetFromPlayerList(200))
390 AttackStart(target);
391 else
392 {
393 summons.DespawnAll();
394 me->DespawnOrUnsummon(1);
395 }
396
397 if (_feralEssenceStack)
398 if (Aura* aur = me->AddAura(SPELL_FERAL_ESSENCE, me))
399 aur->SetStackAmount(_feralEssenceStack);
400 }
401 else if (param == ACTION_DESPAWN_ADDS)
402 summons.DespawnAll();
403 }
404
405 void UpdateAI(uint32 diff) override
406 {
407 if (!UpdateVictim())
408 return;
409
410 _feralRushTimer += diff;
411 _feralPounceTimer += diff;
412
413 if (_feralRushTimer >= 6000)
414 {
415 DoResetThreat();
416 if (!UpdateVictim())
417 return;
418
419 me->CastSpell(me->GetVictim(), SPELL_FERAL_RUSH, true);
420 _feralRushTimer = 0;
421 }
422 else if (_feralPounceTimer >= 6000)
423 {
424 me->CastSpell(me->GetVictim(), SPELL_FERAL_POUNCE, false);
425 _feralPounceTimer = 0;
426 }
427
428 DoMeleeAttackIfReady();
429 }
