89bool OutdoorPvPSI::HandleAreaTrigger(Player* player, uint32 trigger)
90{
91 std::lock_guard<std::mutex> guard(sOutdoorPvPMgr->_lock);
92
93 switch (trigger)
94 {
95 case SI_AREATRIGGER_A:
96 if (player->GetTeamId() == TEAM_ALLIANCE && player->HasAura(SI_SILITHYST_FLAG))
97 {
98 // remove aura
99 player->RemoveAurasDueToSpell(SI_SILITHYST_FLAG);
100 ++ m_Gathered_A;
101 if (m_Gathered_A >= SI_MAX_RESOURCES)
102 {
103 TeamApplyBuff(TEAM_ALLIANCE, SI_CENARION_FAVOR, 0, player);
104 sWorld->SendZoneText(OutdoorPvPSIBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(LANG_OPVP_SI_CAPTURE_A));
105 m_LastController = TEAM_ALLIANCE;
106 m_Gathered_A = 0;
107 m_Gathered_H = 0;
108 }
109 UpdateWorldState();
110 // reward player, xinef: has no effect on characters above level 70
111 if (player->getLevel() < 70)
112 player->CastSpell(player, SI_TRACES_OF_SILITHYST, true);
113 // add 19 honor
114 player->RewardHonor(nullptr, 1, 19);
115 // add 20 cenarion circle repu
116 player->GetReputationMgr().ModifyReputation(sFactionStore.LookupEntry(609), 20.f);
117 // complete quest
118 player->KilledMonsterCredit(SI_TURNIN_QUEST_CM_A);
119 }
120 return true;
121 case SI_AREATRIGGER_H:
122 if (player->GetTeamId() == TEAM_HORDE && player->HasAura(SI_SILITHYST_FLAG))
123 {
124 // remove aura
125 player->RemoveAurasDueToSpell(SI_SILITHYST_FLAG);
126 ++ m_Gathered_H;
127 if (m_Gathered_H >= SI_MAX_RESOURCES)
128 {
129 TeamApplyBuff(TEAM_HORDE, SI_CENARION_FAVOR, 0, player);
130 sWorld->SendZoneText(OutdoorPvPSIBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(LANG_OPVP_SI_CAPTURE_H));
131 m_LastController = TEAM_HORDE;
132 m_Gathered_A = 0;
133 m_Gathered_H = 0;
134 }
135 UpdateWorldState();
136 // reward player, xinef: has no effect on characters above level 70
137 if (player->getLevel() < 70)
138 player->CastSpell(player, SI_TRACES_OF_SILITHYST, true);
139 // add 19 honor
140 player->RewardHonor(nullptr, 1, 19);
141 // add 20 cenarion circle repu
142 player->GetReputationMgr().ModifyReputation(sFactionStore.LookupEntry(609), 20.f);
143 // complete quest
144 player->KilledMonsterCredit(SI_TURNIN_QUEST_CM_H);
145 }
146 return true;
147 }
148 return false;
149}

151bool OutdoorPvPSI::HandleDropFlag(Player* player, uint32 spellId)
152{
153 if (spellId == SI_SILITHYST_FLAG)
154 {
155 // if it was dropped away from the player's turn-in point, then create a silithyst mound, if it was dropped near the areatrigger, then it was dispelled by the outdoorpvp, so do nothing
156 switch (player->GetTeamId())
157 {
158 case TEAM_ALLIANCE:
159 {
160 AreaTrigger const* atEntry = sObjectMgr->GetAreaTrigger(SI_AREATRIGGER_A);
161 if (atEntry)
162 {
163 // 5.0f is safe-distance
164 if (player->GetDistance(atEntry->x, atEntry->y, atEntry->z) > 5.0f + atEntry->radius)
165 {
166 // he dropped it further, summon mound
167 GameObject* go = sObjectMgr->IsGameObjectStaticTransport(SI_SILITHYST_MOUND) ? new StaticTransport() : new GameObject();
168 Map* map = player->GetMap();
169 if (!map)
170 {
171 delete go;
172 return true;
173 }
174
175 if (!go->Create(map->GenerateLowGuid<HighGuid::GameObject>(), SI_SILITHYST_MOUND, map, player->GetPhaseMask(), player->GetPositionX(), player->GetPositionY(), player->GetPositionZ(), player->GetOrientation(), G3D::Quat(), 100, GO_STATE_READY))
176 {
177 delete go;
178 return true;
179 }
180
181 go->SetRespawnTime(0);
182
183 if (!map->AddToMap(go))
184 {
185 delete go;
186 return true;
187 }
188 }
189 }
190 }
191 break;
192 case TEAM_HORDE:
193 {
194 AreaTrigger const* atEntry = sObjectMgr->GetAreaTrigger(SI_AREATRIGGER_H);
195 if (atEntry)
196 {
197 // 5.0f is safe-distance
198 if (player->GetDistance(atEntry->x, atEntry->y, atEntry->z) > 5.0f + atEntry->radius)
199 {
200 // he dropped it further, summon mound
201 GameObject* go = sObjectMgr->IsGameObjectStaticTransport(SI_SILITHYST_MOUND) ? new StaticTransport() : new GameObject();
202 Map* map = player->GetMap();
203 if (!map)
204 {
205 delete go;
206 return true;
207 }
208
209 if (!go->Create(map->GenerateLowGuid<HighGuid::GameObject>(), SI_SILITHYST_MOUND, map, player->GetPhaseMask(), player->GetPositionX(), player->GetPositionY(), player->GetPositionZ(), player->GetOrientation(), G3D::Quat(), 100, GO_STATE_READY))
210 {
211 delete go;
212 return true;
213 }
214
215 go->SetRespawnTime(0);
216
217 if (!map->AddToMap(go))
218 {
219 delete go;
220 return true;
221 }
222 }
223 }
224 }
225 break;
226 default:
227 break;
228 }
229 return true;
230 }
231 return false;
232}

234bool OutdoorPvPSI::HandleCustomSpell(Player* player, uint32 spellId, GameObject* go)
235{

237 return false;

239 if (go->GetGOInfo()->entry == SI_SILITHYST_MOUND)
240 {
241 // despawn go
242 go->SetRespawnTime(0);
243 go->Delete();
244 }
245 return true;
