48bool PetAI::_needToStop()
49{
50 // This is needed for charmed creatures, as once their target was reset other effects can trigger threat
51 if (me->IsCharmed() && me->GetVictim() == me->GetCharmer())
52 return true;
53
54 // xinef: dont allow to follow targets out of visibility range
55 if (me->GetExactDist(me->GetVictim()) > me->GetVisibilityRange() - 5.0f)
56 return true;
57
58 // dont allow pets to follow targets far away from owner
59 if (Unit* owner = me->GetCharmerOrOwner())
60 if (owner->GetExactDist(me) >= (owner->GetVisibilityRange() - 10.0f))
61 return true;
62
63 return !me->CanCreatureAttack(me->GetVictim());
64}

71void PetAI::_stopAttack()
72{
73 if (!me->IsAlive())
74 {
75 LOG_DEBUG("entities.unit.ai", "Creature stoped attacking cuz his dead [{}]", me->GetGUID().ToString());
76 me->GetMotionMaster()->Clear();
77 me->GetMotionMaster()->MoveIdle();
78 me->CombatStop();
79 me->getHostileRefMgr().deleteReferences();
80 return;
81 }
82
83 me->AttackStop();
84 me->InterruptNonMeleeSpells(false);
85 me->GetCharmInfo()->SetIsCommandAttack(false);
86 ClearCharmInfoFlags();
87 HandleReturnMovement();
88}

99bool PetAI::_canMeleeAttack()
100{
101 combatRange = 0.f;
102 switch (me->GetEntry())
103 {
104 case ENTRY_IMP:
105 case ENTRY_WATER_ELEMENTAL:
106 case ENTRY_WATER_ELEMENTAL_PERM:
107 {
108 for (uint8 i = 0; i < me->GetPetAutoSpellSize(); ++i)
109 {
110 uint32 spellID = me->GetPetAutoSpellOnPos(i);
111 switch (spellID)
112 {
113 case IMP_FIREBOLT_RANK_1:
114 case IMP_FIREBOLT_RANK_2:
115 case IMP_FIREBOLT_RANK_3:
116 case IMP_FIREBOLT_RANK_4:
117 case IMP_FIREBOLT_RANK_5:
118 case IMP_FIREBOLT_RANK_6:
119 case IMP_FIREBOLT_RANK_7:
120 case IMP_FIREBOLT_RANK_8:
121 case IMP_FIREBOLT_RANK_9:
122 case WATER_ELEMENTAL_WATERBOLT_1:
123 case WATER_ELEMENTAL_WATERBOLT_2:
124 {
125 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellID);
126 int32 mana = me->GetPower(POWER_MANA);
127
128 if (mana >= spellInfo->CalcPowerCost(me, spellInfo->GetSchoolMask()))
129 {
130 combatRange = spellInfo->GetMaxRange();
131 return true;
132 }
133 }
134 default:
135 break;
136 }
137 }
138 return false;
139 }
140 default:
141 break;
142 }
143
144 return true;
145}

147void PetAI::UpdateAI(uint32 diff)
148{
149 if (!me->IsAlive() || !me->GetCharmInfo())
150 return;
151
152 Unit* owner = me->GetCharmerOrOwner();
153
154 if (m_updateAlliesTimer <= diff)
155 // UpdateAllies self set update timer
156 UpdateAllies();
157 else
158 m_updateAlliesTimer -= diff;
159
160 if (me->GetVictim() && me->GetVictim()->IsAlive())
161 {
162 // is only necessary to stop casting, the pet must not exit combat
163 if (me->GetVictim()->HasBreakableByDamageCrowdControlAura(me))
164 {
165 me->InterruptNonMeleeSpells(false);
166 return;
167 }
168
169 if (_needToStop())
170 {
171 LOG_DEBUG("entities.unit.ai", "Pet AI stopped attacking [{}]", me->GetGUID().ToString());
172 _stopAttack();
173 return;
174 }
175
176 // Check before attacking to prevent pets from leaving stay position
177 if (me->GetCharmInfo()->HasCommandState(COMMAND_STAY))
178 {
179 if (me->GetCharmInfo()->IsCommandAttack() || (me->GetCharmInfo()->IsAtStay() && me->IsWithinMeleeRange(me->GetVictim())))
180 _doMeleeAttack();
181 }
182 else
183 _doMeleeAttack();
184 }
185 else if (!me->GetCharmInfo() || (!me->GetCharmInfo()->GetForcedSpell() && !(me->IsPet() && me->ToPet()->HasTempSpell()) && !me->HasUnitState(UNIT_STATE_CASTING)))
186 {
187 if (me->HasReactState(REACT_AGGRESSIVE) || me->GetCharmInfo()->IsAtStay())
188 {
189 // Every update we need to check targets only in certain cases
190 // Aggressive - Allow auto select if owner or pet don't have a target
191 // Stay - Only pick from pet or owner targets / attackers so targets won't run by
192 // while chasing our owner. Don't do auto select.
193 // All other cases (ie: defensive) - Targets are assigned by AttackedBy(), OwnerAttackedBy(), OwnerAttacked(), etc.
194 Unit* nextTarget = SelectNextTarget(me->HasReactState(REACT_AGGRESSIVE));
195
196 if (nextTarget)
197 AttackStart(nextTarget);
198 else
199 HandleReturnMovement();
200 }
201 else
202 HandleReturnMovement();
203 }
204
205 // xinef: charm info must be always available
206 if (!me->GetCharmInfo())
207 return;
208
209 // Autocast (casted only in combat or persistent spells in any state)
210 if (!me->HasUnitState(UNIT_STATE_CASTING))
211 {
212 if (owner && owner->GetTypeId() == TYPEID_PLAYER && me->GetCharmInfo()->GetForcedSpell() && me->GetCharmInfo()->GetForcedTarget())
213 {
214 owner->ToPlayer()->GetSession()->HandlePetActionHelper(me, me->GetGUID(), std::abs(me->GetCharmInfo()->GetForcedSpell()), ACT_ENABLED, me->GetCharmInfo()->GetForcedTarget());
215
216 // xinef: if spell was casted properly and we are in passive mode, handle return
217 if (!me->GetCharmInfo()->GetForcedSpell() && me->HasReactState(REACT_PASSIVE))
218 {
219 if (me->HasUnitState(UNIT_STATE_CASTING))
220 {
221 me->GetMotionMaster()->Clear(false);
222 me->StopMoving();
223 }
224 else
225 _stopAttack();
226 }
227 return;
228 }
229
230 // xinef: dont allow ghouls to cast spells below 75 energy
231 if (me->IsPet() && me->ToPet()->IsPetGhoul() && me->GetPower(POWER_ENERGY) < 75)
232 return;
233
234 typedef std::vector<std::pair<Unit*, Spell*> > TargetSpellList;
235 TargetSpellList targetSpellStore;
236
237 for (uint8 i = 0; i < me->GetPetAutoSpellSize(); ++i)
238 {
239 uint32 spellID = me->GetPetAutoSpellOnPos(i);
240 if (!spellID)
241 continue;
242
243 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellID);
244 if (!spellInfo)
245 continue;
246
247 if (me->GetCharmInfo()->GetGlobalCooldownMgr().HasGlobalCooldown(spellInfo))
248 continue;
249
250 // check spell cooldown, this should be checked in CheckCast...
251 if (me->HasSpellCooldown(spellInfo->Id))
252 continue;
253
254 if (spellInfo->IsPositive())
255 {
256 if (spellInfo->CanBeUsedInCombat())
257 {
258 // Check if we're in combat or commanded to attack (exlude auras with infinity duration)
259 if (!me->IsInCombat() && spellInfo->GetMaxDuration() != -1 && !me->IsPetInCombat())
260 {
261 continue;
262 }
263 }
264
265 Spell* spell = new Spell(me, spellInfo, TRIGGERED_NONE);
266 spell->LoadScripts(); // xinef: load for CanAutoCast (calling CheckPetCast)
267 bool spellUsed = false;
268
269 // Some spells can target enemy or friendly (DK Ghoul's Leap)
270 // Check for enemy first (pet then owner)
271 Unit* target = me->getAttackerForHelper();
272 if (!target && owner)
273 target = owner->getAttackerForHelper();
274
275 if (target)
276 {
277 if (CanAttack(target) && spell->CanAutoCast(target))
278 {
279 targetSpellStore.push_back(std::make_pair(target, spell));
280 spellUsed = true;
281 }
282 }
283
284 // No enemy, check friendly
285 if (!spellUsed)
286 {
287 for (ObjectGuid const& guid : m_AllySet)
288 {
289 Unit* ally = ObjectAccessor::GetUnit(*me, guid);
290
291 //only buff targets that are in combat, unless the spell can only be cast while out of combat
292 if (!ally)
293 continue;
294
295 if (spell->CanAutoCast(ally))
296 {
297 targetSpellStore.push_back(std::make_pair(ally, spell));
298 spellUsed = true;
299 break;
300 }
301 }
302 }
303
304 // No valid targets at all
305 if (!spellUsed)
306 delete spell;
307 }
308 else if (me->GetVictim() && CanAttack(me->GetVictim(), spellInfo) && spellInfo->CanBeUsedInCombat())
309 {
310 Spell* spell = new Spell(me, spellInfo, TRIGGERED_NONE);
311 if (spell->CanAutoCast(me->GetVictim()))
312 targetSpellStore.push_back(std::make_pair(me->GetVictim(), spell));
313 else
314 delete spell;
315 }
316 }
317
318 //found units to cast on to
319 if (!targetSpellStore.empty())
320 {
321 uint32 index = urand(0, targetSpellStore.size() - 1);
322
323 Spell* spell = targetSpellStore[index].second;
324 Unit* target = targetSpellStore[index].first;
325
326 targetSpellStore.erase(targetSpellStore.begin() + index);
327
328 SpellCastTargets targets;
329 targets.SetUnitTarget(target);
330
331 if (!me->HasInArc(M_PI, target))
332 {
333 me->SetInFront(target);
334 if (target && target->GetTypeId() == TYPEID_PLAYER)
335 me->SendUpdateToPlayer(target->ToPlayer());
336
337 if (owner && owner->GetTypeId() == TYPEID_PLAYER)
338 me->SendUpdateToPlayer(owner->ToPlayer());
339 }
340
341 me->AddSpellCooldown(spell->m_spellInfo->Id, 0, 0);
342
343 spell->prepare(&targets);
344 }
345
346 // deleted cached Spell objects
347 for (TargetSpellList::const_iterator itr = targetSpellStore.begin(); itr != targetSpellStore.end(); ++itr)
348 delete itr->second;
349 }
350}

352void PetAI::UpdateAllies()
353{
354 Unit* owner = me->GetCharmerOrOwner();
355 Group* group = nullptr;
356
357 m_updateAlliesTimer = 10 * IN_MILLISECONDS; //update friendly targets every 10 seconds, lesser checks increase performance
358
359 if (!owner)
360 return;
361 else if (owner->GetTypeId() == TYPEID_PLAYER)
362 group = owner->ToPlayer()->GetGroup();
363
364 //only pet and owner/not in group->ok
365 if (m_AllySet.size() == 2 && !group)
366 return;
367 //owner is in group; group members filled in already (no raid -> subgroupcount = whole count)
368 if (group && !group->isRaidGroup() && m_AllySet.size() == (group->GetMembersCount() + 2))
369 return;
370
371 m_AllySet.clear();
372 m_AllySet.insert(me->GetGUID());
373 if (group) //add group
374 {
375 for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
376 {
377 Player* Target = itr->GetSource();
378 if (!Target || !Target->IsInMap(owner) || !group->SameSubGroup(owner->ToPlayer(), Target))
379 continue;
380
381 if (Target->GetGUID() == owner->GetGUID())
382 continue;
383
384 m_AllySet.insert(Target->GetGUID());
385 }
386 }
387 else //remove group
388 m_AllySet.insert(owner->GetGUID());
389}

391void PetAI::KilledUnit(Unit* victim)
392{
393 // Called from Unit::Kill() in case where pet or owner kills something
394 // if owner killed this victim, pet may still be attacking something else
395 if (me->GetVictim() && me->GetVictim() != victim)
396 return;
397
398 // Xinef: if pet is channeling a spell and owner killed something different, dont interrupt it
399 if (me->HasUnitState(UNIT_STATE_CASTING) && me->GetGuidValue(UNIT_FIELD_CHANNEL_OBJECT) && me->GetGuidValue(UNIT_FIELD_CHANNEL_OBJECT) != victim->GetGUID())
400 return;
401
402 // Clear target just in case. May help problem where health / focus / mana
403 // regen gets stuck. Also resets attack command.
404 // Can't use _stopAttack() because that activates movement handlers and ignores
405 // next target selection
406 me->AttackStop();
407 me->InterruptNonMeleeSpells(false);
408
409 // Before returning to owner, see if there are more things to attack
410 if (Unit* nextTarget = SelectNextTarget(false))
411 AttackStart(nextTarget);
412 else
413 HandleReturnMovement(); // Return
414}

429{
430 // Called when owner takes damage. This function helps keep pets from running off
431 // simply due to owner gaining aggro.
432

469Unit* PetAI::SelectNextTarget(bool allowAutoSelect) const
470{
471 // Provides next target selection after current target death.
472 // This function should only be called internally by the AI
473 // Targets are not evaluated here for being valid targets, that is done in _CanAttack()
474 // The parameter: allowAutoSelect lets us disable aggressive pet auto targeting for certain situations
475
476 // Passive pets don't do next target selection
477 if (me->HasReactState(REACT_PASSIVE))
478 return nullptr;
479
480 // Check pet attackers first so we don't drag a bunch of targets to the owner
481 if (Unit* myAttacker = me->getAttackerForHelper())
482 if (!myAttacker->HasBreakableByDamageCrowdControlAura() && me->CanCreatureAttack(myAttacker))
483 return myAttacker;
484
485 // Check pet's attackers first to prevent dragging mobs back to owner
486 if (me->HasAuraType(SPELL_AURA_MOD_TAUNT))
487 {
488 const Unit::AuraEffectList& tauntAuras = me->GetAuraEffectsByType(SPELL_AURA_MOD_TAUNT);
489 if (!tauntAuras.empty())
490 for (Unit::AuraEffectList::const_reverse_iterator itr = tauntAuras.rbegin(); itr != tauntAuras.rend(); ++itr)
491 if (Unit* caster = (*itr)->GetCaster())
492 if (me->CanCreatureAttack(caster) && !caster->HasAuraTypeWithCaster(SPELL_AURA_IGNORED, me->GetGUID()))
493 return caster;
494 }
495
496 // Not sure why we wouldn't have an owner but just in case...
497 Unit* owner = me->GetCharmerOrOwner();
498 if (!owner)
499 return nullptr;
500
501 // Check owner attackers
502 if (Unit* ownerAttacker = owner->getAttackerForHelper())
503 if (!ownerAttacker->HasBreakableByDamageCrowdControlAura() && me->CanCreatureAttack(ownerAttacker))
504 return ownerAttacker;
505
506 // Check owner victim
507 // 3.0.2 - Pets now start attacking their owners victim in defensive mode as soon as the hunter does
508 if (Unit* ownerVictim = owner->GetVictim())
509 if (me->CanCreatureAttack(ownerVictim))
510 return ownerVictim;
511
512 // Neither pet or owner had a target and aggressive pets can pick any target
513 // To prevent aggressive pets from chain selecting targets and running off, we
514 // only select a random target if certain conditions are met.
515 if (allowAutoSelect)
516 if (!me->GetCharmInfo()->IsReturning() || me->GetCharmInfo()->IsFollowing() || me->GetCharmInfo()->IsAtStay())
517 if (Unit* nearTarget = me->ToCreature()->SelectNearestTargetInAttackDistance(MAX_AGGRO_RADIUS))
518 return nearTarget;
519
520 // Default - no valid targets
521 return nullptr;
522}

524void PetAI::HandleReturnMovement()
525{
526 // Handles moving the pet back to stay or owner
527
528 // Prevent activating movement when under control of spells
529 // such as "Eyes of the Beast"
530 if (me->isPossessed())
531 return;
532
533 if (me->GetCharmInfo()->HasCommandState(COMMAND_STAY))
534 {
535 if (!me->GetCharmInfo()->IsAtStay() && !me->GetCharmInfo()->IsReturning())
536 {
537 if (me->GetCharmInfo()->HasStayPosition())
538 {
539 // Return to previous position where stay was clicked
540 if (me->GetMotionMaster()->GetMotionSlotType(MOTION_SLOT_CONTROLLED) == NULL_MOTION_TYPE)
541 {
542 float x, y, z;
543 me->GetCharmInfo()->GetStayPosition(x, y, z);
544 ClearCharmInfoFlags();
545 me->GetCharmInfo()->SetIsReturning(true);
546 me->GetMotionMaster()->Clear();
547 me->GetMotionMaster()->MovePoint(me->GetGUID().GetCounter(), x, y, z);
548 }
549 }
550 }
551 }
552 else // COMMAND_FOLLOW
553 {
554 if (!me->GetCharmInfo()->IsFollowing() && !me->GetCharmInfo()->IsReturning())
555 {
556 if (me->GetMotionMaster()->GetMotionSlotType(MOTION_SLOT_CONTROLLED) == NULL_MOTION_TYPE)
557 {
558 ClearCharmInfoFlags();
559 me->GetCharmInfo()->SetIsReturning(true);
560 me->GetMotionMaster()->Clear();
561 me->GetMotionMaster()->MoveFollow(me->GetCharmerOrOwner(), PET_FOLLOW_DIST, me->GetFollowAngle());
562 }
563 }
564 }
565
566 me->GetCharmInfo()->SetForcedSpell(0);
567 me->GetCharmInfo()->SetForcedTargetGUID();
568
569 // xinef: remember that npcs summoned by npcs can also be pets
570 me->GetThreatMgr().ClearAllThreat();
571 me->ClearInPetCombat();
572}

585void PetAI::DoAttack(Unit* target, bool chase)
586{
587 // Handles attack with or without chase and also resets flags
588 // for next update / creature kill
589
590 if (me->Attack(target, true))
591 {
592 // xinef: properly fix fake combat after pet is sent to attack
593 if (Unit* owner = me->GetOwner())
594 owner->SetUnitFlag(UNIT_FLAG_PET_IN_COMBAT);
595
596 me->SetUnitFlag(UNIT_FLAG_PET_IN_COMBAT);
597
598 // Play sound to let the player know the pet is attacking something it picked on its own
599 if (me->HasReactState(REACT_AGGRESSIVE) && !me->GetCharmInfo()->IsCommandAttack())
600 me->SendPetAIReaction(me->GetGUID());
601
602 if (chase)
603 {
604 bool oldCmdAttack = me->GetCharmInfo()->IsCommandAttack(); // This needs to be reset after other flags are cleared
605 ClearCharmInfoFlags();
606 me->GetCharmInfo()->SetIsCommandAttack(oldCmdAttack); // For passive pets commanded to attack so they will use spells
607
608 if (_canMeleeAttack())
609 {
610 float angle = combatRange == 0.f && target->GetTypeId() != TYPEID_PLAYER && !target->IsPet() ? float(M_PI) : 0.f;
611 float tolerance = combatRange == 0.f ? float(M_PI_4) : float(M_PI * 2);
612 me->GetMotionMaster()->MoveChase(target, ChaseRange(0.f, combatRange), ChaseAngle(angle, tolerance));
613 }
614 }
615 else // (Stay && ((Aggressive || Defensive) && In Melee Range)))
616 {
617 ClearCharmInfoFlags();
618
619 me->GetCharmInfo()->SetIsAtStay(true);
620 me->GetMotionMaster()->MovementExpiredOnSlot(MOTION_SLOT_ACTIVE, false);
621 me->GetMotionMaster()->MoveIdle();
622 }
623 }

626void PetAI::MovementInform(uint32 moveType, uint32 data)
627{
628 // Receives notification when pet reaches stay or follow owner
629 switch (moveType)
630 {
631 case POINT_MOTION_TYPE:
632 {
633 // Pet is returning to where stay was clicked. data should be
634 // pet's GUIDLow since we set that as the waypoint ID
635 if (data == me->GetGUID().GetCounter() && me->GetCharmInfo()->IsReturning())
636 {
637 ClearCharmInfoFlags();
638 me->GetCharmInfo()->SetIsAtStay(true);
639 me->GetMotionMaster()->Clear();
640 me->GetMotionMaster()->MoveIdle();
641 }
642 break;
643 }
644 case FOLLOW_MOTION_TYPE:
645 {
646 // If data is owner's GUIDLow then we've reached follow point,
647 // otherwise we're probably chasing a creature
648 if (me->GetCharmerOrOwner() && me->GetCharmInfo() && data == me->GetCharmerOrOwner()->GetGUID().GetCounter() && me->GetCharmInfo()->IsReturning())
649 {
650 ClearCharmInfoFlags();
651 me->GetCharmInfo()->SetIsFollowing(true);
652 }
653 break;
654 }
655 default:
656 break;
657 }
658}

660bool PetAI::CanAttack(Unit* target, SpellInfo const* spellInfo)
661{
662 // Evaluates wether a pet can attack a specific target based on CommandState, ReactState and other flags
663 // IMPORTANT: The order in which things are checked is important, be careful if you add or remove checks
664
665 // Hmmm...
666 if (!target)
667 return false;
668
669 if (!target->IsAlive())
670 {
671 // xinef: if target is invalid, pet should evade automaticly
672 // Clear target to prevent getting stuck on dead targets
673 //me->AttackStop();
674 //me->InterruptNonMeleeSpells(false);
675 return false;
676 }
677
678 // xinef: check unit states of pet
679 if (me->HasUnitState(UNIT_STATE_LOST_CONTROL))
680 return false;
681
682 // xinef: pets of mounted players have stunned flag only, check this also
683 if (me->HasUnitFlag(UNIT_FLAG_STUNNED))
684 return false;
685
686 // pussywizard: TEMP!
687 if (!me->GetCharmInfo())
688 {
689 LOG_INFO("misc", "PetAI::CanAttack (A1) - {}, {}", me->GetEntry(), me->GetOwnerGUID().ToString());
690 return false;
691 }
692
693 // Passive - passive pets can attack if told to
694 if (me->HasReactState(REACT_PASSIVE))
695 return me->GetCharmInfo()->IsCommandAttack();
696
697 // CC - mobs under crowd control can be attacked if owner commanded
698 if (target->HasBreakableByDamageCrowdControlAura() && (!spellInfo || !spellInfo->HasAttribute(SPELL_ATTR4_REACTIVE_DAMAGE_PROC)))
699 return me->GetCharmInfo()->IsCommandAttack();
700
701 // Returning - pets ignore attacks only if owner clicked follow
702 if (me->GetCharmInfo()->IsReturning())
703 return !me->GetCharmInfo()->IsCommandFollow();
704
705 // Stay - can attack if target is within range or commanded to
706 if (me->GetCharmInfo()->HasCommandState(COMMAND_STAY))
707 return (me->IsWithinMeleeRange(target) || me->GetCharmInfo()->IsCommandAttack());
708
709 // Pets attacking something (or chasing) should only switch targets if owner tells them to
710 if (me->GetVictim() && me->GetVictim() != target)
711 {
712 // Check if our owner selected this target and clicked "attack"
713 Unit* ownerTarget = nullptr;
714 if (Player* owner = me->GetCharmerOrOwner()->ToPlayer())
715 ownerTarget = owner->GetSelectedUnit();
716 else
717 ownerTarget = me->GetCharmerOrOwner()->GetVictim();
718
719 if (ownerTarget && me->GetCharmInfo()->IsCommandAttack())
720 return (target->GetGUID() == ownerTarget->GetGUID());
721 }
722
723 // Follow
724 if (me->GetCharmInfo()->HasCommandState(COMMAND_FOLLOW))
725 return !me->GetCharmInfo()->IsReturning();
726
727 // default, though we shouldn't ever get here
728 return false;
729}

731void PetAI::ReceiveEmote(Player* player, uint32 emote)
732{
733 if (me->GetOwnerGUID() && me->GetOwnerGUID() == player->GetGUID())
734 switch (emote)
735 {
736 case TEXT_EMOTE_COWER:
737 if (me->IsPet() && me->ToPet()->IsPetGhoul())
738 me->HandleEmoteCommand(/*EMOTE_ONESHOT_ROAR*/EMOTE_ONESHOT_OMNICAST_GHOUL);
739 break;
740 case TEXT_EMOTE_ANGRY:
741 if (me->IsPet() && me->ToPet()->IsPetGhoul())
742 me->HandleEmoteCommand(/*EMOTE_ONESHOT_COWER*/EMOTE_STATE_STUN);
743 break;
744 case TEXT_EMOTE_GLARE:
745 if (me->IsPet() && me->ToPet()->IsPetGhoul())
746 me->HandleEmoteCommand(EMOTE_STATE_STUN);
747 break;
748 case TEXT_EMOTE_SOOTHE:
749 if (me->IsPet() && me->ToPet()->IsPetGhoul())
750 me->HandleEmoteCommand(EMOTE_ONESHOT_OMNICAST_GHOUL);
751 break;
752 }
753}

771void PetAI::AttackedBy(Unit* attacker)
772{
773 // Called when pet takes damage. This function helps keep pets from running off
774 // simply due to gaining aggro.
775
776 if (!attacker)
777 return;
778
779 // Passive pets don't do anything
780 if (me->HasReactState(REACT_PASSIVE))
781 return;
782
783 // Prevent pet from disengaging from current target
784 if (me->GetVictim() && me->GetVictim()->IsAlive())
785 return;
786
787 // Continue to evaluate and attack if necessary
788 AttackStart(attacker);
789}
