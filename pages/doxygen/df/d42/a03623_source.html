194 {
195 for (uint8 i = 0; i < (MAX_ENCOUNTER - 1); ++i)
196 {
197 if (m_auiEncounter[i] == IN_PROGRESS)
198 return true;
199 }
200
201 // Leviathan does not use IN_PROGRESS type, instead SPECIAL is set and never reset,
202 // Check if he is in combat.
203 if (Unit* l = instance->GetCreature(m_uiLeviathanGUID))
204 if (l->IsInCombat())
205 return true;
206
207 return false;

217 void SpawnHodirChests(Difficulty diff, Creature* hodir)
218 {
219 switch (diff)
220 {
221 case RAID_DIFFICULTY_10MAN_NORMAL: // 10 man chest
222 {
223 if (!m_hodirNormalChest)
224 {
225 if (GameObject* go = hodir->SummonGameObject(
226 GO_HODIR_CHEST_NORMAL,
227 normalChestPosition.GetPositionX(),
228 normalChestPosition.GetPositionY(),
229 normalChestPosition.GetPositionZ(),
230 normalChestPosition.GetOrientation(), 0, 0, 0, 0, 0))
231 {
232 m_hodirNormalChest = go->GetGUID();
233 go->SetGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
234 }
235 }
236 if (!m_hodirHardmodeChest)
237 {
238 if (GameObject* go = hodir->SummonGameObject(
239 GO_HODIR_CHEST_HARD,
240 hardChestPosition.GetPositionX(),
241 hardChestPosition.GetPositionY(),
242 hardChestPosition.GetPositionZ(),
243 hardChestPosition.GetOrientation(), 0, 0, 0, 0, 0))
244 {
245 m_hodirHardmodeChest = go->GetGUID();
246 go->SetGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
247 hmHodir = true;
248 }
249 }
250 break;
251 }
252 case RAID_DIFFICULTY_25MAN_NORMAL: // 25 man chest
253 {
254 if (!m_hodirNormalChest)
255 {
256 if (GameObject* go = hodir->SummonGameObject(
257 GO_HODIR_CHEST_NORMAL_HERO,
258 normalChestPosition.GetPositionX(),
259 normalChestPosition.GetPositionY(),
260 normalChestPosition.GetPositionZ(),
261 normalChestPosition.GetOrientation(), 0, 0, 0, 0, 0))
262 {
263 m_hodirNormalChest = go->GetGUID();
264 go->SetGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
265 }
266 }
267 if (!m_hodirHardmodeChest)
268 {
269 if (GameObject* go = hodir->SummonGameObject(
270 GO_HODIR_CHEST_HARD_HERO,
271 hardChestPosition.GetPositionX(),
272 hardChestPosition.GetPositionY(),
273 hardChestPosition.GetPositionZ(),
274 hardChestPosition.GetOrientation(), 0, 0, 0, 0, 0))
275 {
276 m_hodirHardmodeChest = go->GetGUID();
277 go->SetGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
278 hmHodir = true;
279 }
280 }
281 break;
282 }
283 default:
284 break;
285 }
