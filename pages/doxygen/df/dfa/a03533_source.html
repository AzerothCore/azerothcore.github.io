82 void OnGameObjectCreate(GameObject* pGo) override
83 {
84 switch( pGo->GetEntry() )
85 {
86 case GO_DRAGON_CAGE_DOOR:
87 for( uint8 i = 0; i < 3; ++i )
88 {
89 if( DragonCageDoorGUID[i] )
90 continue;
91
92 DragonCageDoorGUID[i] = pGo->GetGUID();
93 break;
94 }
95 if( m_auiEncounter[DATA_DRAKOS] == DONE )
96 if( pGo->GetGoState() != GO_STATE_ACTIVE )
97 {
98 pGo->SetLootState(GO_READY);
99 pGo->UseDoorOrButton(0, false);
100 }
101 break;
102 case GO_CACHE_OF_EREGOS:
103 case GO_CACHE_OF_EREGOS_HERO:
104 EregosCacheGUID = pGo->GetGUID();
105 break;
106 }

129 void SetData(uint32 type, uint32 data) override
130 {
131 switch( type )
132 {
133 case DATA_DRAKOS:
134 m_auiEncounter[DATA_DRAKOS] = data;
135 if( data == DONE )
136 {
137 DoUpdateWorldState(WORLD_STATE_CENTRIFUGE_CONSTRUCT_SHOW, 1);
138 DoUpdateWorldState(WORLD_STATE_CENTRIFUGE_CONSTRUCT_AMOUNT, 10 - CentrifugeCount);
139
140 if (instance->IsHeroic())
141 DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_MAKE_IT_COUNT_TIMED_EVENT);
142 }
143 break;
144 case DATA_VAROS:
145 m_auiEncounter[DATA_VAROS] = data;
146 if( data == DONE )
147 {
148 DoUpdateWorldState(WORLD_STATE_CENTRIFUGE_CONSTRUCT_SHOW, 0);
149
150 if( Creature* urom = instance->GetCreature(uiUromGUID) )
151 urom->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
152 }
153 break;
154 case DATA_UROM:
155 m_auiEncounter[DATA_UROM] = data;
156 if( data == DONE )
157 if( Creature* eregos = instance->GetCreature(uiEregosGUID) )
158 eregos->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
159 break;
160 case DATA_EREGOS:
161 m_auiEncounter[DATA_EREGOS] = data;
162 if (data == DONE)
163 {
164 DoRespawnGameObject(EregosCacheGUID, 7 * DAY);
165
166 if (GameObject* cache = instance->GetGameObject(EregosCacheGUID))
167 {
168 if (Creature* eregos = instance->GetCreature(uiEregosGUID))
169 {
170 cache->SetLootRecipient(eregos);
171 }
172 }
173 }
174 break;
175 case DATA_CC_COUNT:
176 if( CentrifugeCount < 10 )
177 {
178 ++CentrifugeCount;
179 DoUpdateWorldState(WORLD_STATE_CENTRIFUGE_CONSTRUCT_AMOUNT, 10 - CentrifugeCount);
180 }
181 if( CentrifugeCount >= 10 )
182 if( Creature* varos = instance->GetCreature(uiVarosGUID) )
183 {
184 varos->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
185 varos->InterruptNonMeleeSpells(false);
186 varos->RemoveAura(50053);
187 }
188 break;
189 case DATA_AMBER_VOID:
190 bAmberVoid = !!data;
191 break;
192 case DATA_EMERALD_VOID:
193 bEmeraldVoid = !!data;
194 break;
195 case DATA_RUBY_VOID:
196 bRubyVoid = !!data;
197 break;
198 }
199
200 if( data == DONE )
201 SaveToDB();
202 }

252 void Load(const char* in) override
253 {
254 if( !in )
255 {
256 OUT_LOAD_INST_DATA_FAIL;
257 return;
258 }
259
260 OUT_LOAD_INST_DATA(in);
261
262 char dataHead1, dataHead2;
263 std::istringstream loadStream(in);
264 loadStream >> dataHead1 >> dataHead2;
265
266 if( dataHead1 == 'T' && dataHead2 == 'O' )
267 {
268 loadStream >> m_auiEncounter[0] >> m_auiEncounter[1] >> m_auiEncounter[2] >> m_auiEncounter[3] >> CentrifugeCount;
269
270 for( uint8 i = 0; i < MAX_ENCOUNTER; ++i )
271 if( m_auiEncounter[i] == IN_PROGRESS )
272 m_auiEncounter[i] = NOT_STARTED;
273 }
274 else
275 OUT_LOAD_INST_DATA_FAIL;
276
277 OUT_LOAD_INST_DATA_COMPLETE;
278 }

280 bool CheckAchievementCriteriaMeet(uint32 criteria_id, Player const* source, Unit const* /*target*/, uint32 /*miscvalue1*/) override
281 {
282 switch(criteria_id)
283 {
284 case CRITERIA_EXPERIENCED_AMBER:
285 if( source )
286 if( Unit* drake = source->GetVehicleBase() )
287 if( drake->GetEntry() == NPC_AMBER_DRAKE )
288 return true;
289 break;
290 case CRITERIA_EXPERIENCED_EMERALD:
291 if( source )
292 if( Unit* drake = source->GetVehicleBase() )
293 if( drake->GetEntry() == NPC_EMERALD_DRAKE )
294 return true;
295 break;
296 case CRITERIA_EXPERIENCED_RUBY:
297 if( source )
298 if( Unit* drake = source->GetVehicleBase() )
299 if( drake->GetEntry() == NPC_RUBY_DRAKE )
300 return true;
301 break;
302 case CRITERIA_AMBER_VOID:
303 return bAmberVoid;
304 case CRITERIA_EMERALD_VOID:
305 return bEmeraldVoid;
306 case CRITERIA_RUBY_VOID:
307 return bRubyVoid;
308 }
309 return false;
310 }
