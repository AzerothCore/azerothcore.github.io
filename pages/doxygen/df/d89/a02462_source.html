58 struct boss_gythAI : public BossAI
59 {
60 boss_gythAI(Creature* creature) : BossAI(creature, DATA_GYTH) { }
61
62 void Reset() override
63 {
64 _summonedRend = false;
65 if (instance->GetBossState(DATA_GYTH) == IN_PROGRESS)
66 {
67 instance->SetBossState(DATA_GYTH, NOT_STARTED);
68 summons.DespawnAll();
69 me->DespawnOrUnsummon();
70 }
71 }
72
73 void EnterCombat(Unit* /*who*/) override
74 {
75 _EnterCombat();
76
77 events.ScheduleEvent(EVENT_CORROSIVE_ACID, urand(8000, 16000));
78 events.ScheduleEvent(EVENT_FREEZE, urand(8000, 16000));
79 events.ScheduleEvent(EVENT_FLAME_BREATH, urand(8000, 16000));
80 events.ScheduleEvent(EVENT_KNOCK_AWAY, urand(12000, 18000));
81 }
82
83 void EnterEvadeMode(EvadeReason why) override
84 {
85 instance->SetBossState(DATA_WARCHIEF_REND_BLACKHAND, FAIL);
86 BossAI::EnterEvadeMode(why);
87 }
88
89 void IsSummonedBy(Unit* /*summoner*/) override
90 {
91 events.ScheduleEvent(EVENT_SUMMONED_1, 1000);
92 }
93
94 void JustSummoned(Creature* summon) override
95 {
96 summons.Summon(summon);
97 summon->AI()->AttackStart(me->SelectVictim());
98 }
99
100 // Prevent clearing summon list, otherwise Rend despawns if the drake is killed first.
101 void JustDied(Unit* /*killer*/) override
102 {
103 instance->SetBossState(DATA_GYTH, DONE);
104 }
105
106 void DamageTaken(Unit* /*attacker*/, uint32& damage, DamageEffectType /*type*/, SpellSchoolMask /*school*/) override
107 {
108 if (!_summonedRend && me->HealthBelowPctDamaged(25, damage))
109 {
110 if (damage >= me->GetHealth())
111 {
112 // Let creature fall to 1 HP but prevent it from dying before boss is summoned.
113 damage = me->GetHealth() - 1;
114 }
115 DoCast(me, SPELL_SUMMON_REND, true);
116 me->RemoveAura(SPELL_REND_MOUNTS);
117 _summonedRend = true;
118 }
119 }
120
121 void UpdateAI(uint32 diff) override
122 {
123 if (!UpdateVictim())
124 {
125 events.Update(diff);
126
127 while (uint32 eventId = events.ExecuteEvent())
128 {
129 switch (eventId)
130 {
131 case EVENT_SUMMONED_1:
132 me->AddAura(SPELL_REND_MOUNTS, me);
133 if (GameObject* portcullis = me->FindNearestGameObject(GO_DR_PORTCULLIS, 40.0f))
134 portcullis->UseDoorOrButton();
135 events.ScheduleEvent(EVENT_SUMMONED_2, 2000);
136 break;
137 case EVENT_SUMMONED_2:
138 me->GetMotionMaster()->MovePath(GYTH_PATH_1, false);
139 break;
140 default:
141 break;
142 }
143 }
144 return;
145 }
146
147 events.Update(diff);
148
149 while (uint32 eventId = events.ExecuteEvent())
150 {
151 switch (eventId)
152 {
153 case EVENT_CORROSIVE_ACID:
154 DoCast(me, SPELL_CORROSIVE_ACID);
155 events.ScheduleEvent(EVENT_CORROSIVE_ACID, urand(10000, 16000));
156 break;
157 case EVENT_FREEZE:
158 DoCast(me, SPELL_FREEZE);
159 events.ScheduleEvent(EVENT_FREEZE, urand(10000, 16000));
160 break;
161 case EVENT_FLAME_BREATH:
162 DoCast(me, SPELL_FLAMEBREATH);
163 events.ScheduleEvent(EVENT_FLAME_BREATH, urand(10000, 16000));
164 break;
165 case EVENT_KNOCK_AWAY:
166 DoCastVictim(SPELL_KNOCK_AWAY);
167 events.ScheduleEvent(EVENT_KNOCK_AWAY, urand(14000, 20000));
168 break;
169 default:
170 break;
171 }
172 }
173 DoMeleeAttackIfReady();
174 }
175
176 private:
177 bool _summonedRend;
178 };
179

181 {
182 return GetBlackrockSpireAI<boss_gythAI>(creature);

187class spell_gyth_chromatic_protection : public AuraScript
188{
189 PrepareAuraScript(spell_gyth_chromatic_protection);
190
191 bool Validate(SpellInfo const* /*spellInfo*/) override
192 {
193 return ValidateSpellInfo({ SPELL_CORROSIVE_ACID, SPELL_FLAMEBREATH, SPELL_FREEZE });
194 }
195
196 bool CheckProc(ProcEventInfo& eventInfo)
197 {
198 if (SpellInfo const* spellInfo = eventInfo.GetSpellInfo())
199 {
200 switch (spellInfo->Id)
201 {
202 case SPELL_CORROSIVE_ACID:
203 case SPELL_FLAMEBREATH:
204 case SPELL_FREEZE:
205 return true;
206 break;
207 default:
208 break;
209 }
210 }
211 return false;
212 }
213
214 void HandleProc(AuraEffect const* /* aurEff */, ProcEventInfo& eventInfo)
215 {
216 PreventDefaultAction();
217 if (SpellInfo const* spellInfo = eventInfo.GetSpellInfo())
218 {
219 switch (spellInfo->Id)
220 {
221 case SPELL_CORROSIVE_ACID:
222 GetTarget()->CastSpell(GetTarget(), SPELL_CHROMATIC_PROTECTION_NATURE, true);
223 break;
224 case SPELL_FLAMEBREATH:
225 GetTarget()->CastSpell(GetTarget(), SPELL_CHROMATIC_PROTECTION_FIRE, true);
226 break;
227 case SPELL_FREEZE:
228 GetTarget()->CastSpell(GetTarget(), SPELL_CHROMATIC_PROTECTION_FROST, true);
229 break;
230 default:
231 break;
232 }
233 }
234 }
235
236 void Register() override
237 {
238 OnEffectProc += AuraEffectProcFn(spell_gyth_chromatic_protection::HandleProc, EFFECT_0, SPELL_AURA_DUMMY);
239 }
