95 void UpdateAI(uint32 diff) override
96 {
97 if (!UpdateVictim())
98 return;
99
100 events.Update(diff);
101 if (!events.IsInPhase(1) && me->HasUnitState(UNIT_STATE_CASTING))
102 return;
103
104 switch (events.ExecuteEvent())
105 {
106 case EVENT_SUMMON_SEEDLING:
107 if (roll_chance_i(20))
108 Talk(SAY_OOC_RANDOM);
109 me->CastSpell(me, RAND(SPELL_PLANT_WHITE, SPELL_PLANT_GREEN, SPELL_PLANT_BLUE, SPELL_PLANT_RED), false);
110 events.ScheduleEvent(EVENT_SUMMON_SEEDLING, 6000);
111 break;
112 case EVENT_TREE_FORM:
113 events.Reset();
114 events.SetPhase(1);
115 events.ScheduleEvent(EVENT_CHECK_FRAYERS, 1000);
116 events.ScheduleEvent(EVENT_TREE_FORM, 75000);
117 events.ScheduleEvent(EVENT_RESTORE_COMBAT, 46000);
118
119 Talk(SAY_TREE);
120 me->RemoveAllAuras();
121 me->GetMotionMaster()->MoveIdle();
122 me->GetMotionMaster()->Clear(false);
123
124 me->CastSpell(me, SPELL_SUMMON_FRAYER, true);
125 me->CastSpell(me, SPELL_TRANQUILITY, true);
126 me->CastSpell(me, SPELL_TREE_FORM, true);
127 break;
128 case EVENT_RESTORE_COMBAT:
129 events.SetPhase(0);
130 events.ScheduleEvent(EVENT_SUMMON_SEEDLING, 6000);
131 me->GetMotionMaster()->MoveChase(me->GetVictim());
132 break;
133 case EVENT_CHECK_FRAYERS:
134 if (!summons.HasEntry(NPC_FRAYER))
135 {
136 me->InterruptNonMeleeSpells(true);
137 me->RemoveAllAuras();
138 events.RescheduleEvent(EVENT_RESTORE_COMBAT, 0);
139 events.RescheduleEvent(EVENT_TREE_FORM, 30000);
140 break;
141 }
142 events.ScheduleEvent(EVENT_CHECK_FRAYERS, 500);
143 break;
144 }
145
146 if (!events.IsInPhase(1))
147 DoMeleeAttackIfReady();
148 }
149 };
