83 void OnCreatureCreate(Creature* creature) override
84 {
85 switch (creature->GetEntry())
86 {
87 case NPC_SKERAM:
88 SkeramGUID = creature->GetGUID();
89 break;
90 case NPC_OURO_SPAWNER:
91 if (GetBossState(DATA_OURO) != DONE)
92 creature->Respawn();
93 break;
94 case NPC_MASTERS_EYE:
95 if (GetBossState(DATA_TWIN_EMPERORS) != DONE)
96 creature->Respawn(true);
97 break;
98 case NPC_CTHUN:
99 CThunGUID = creature->GetGUID();
100 if (!creature->IsAlive())
101 {
102 for (ObjectGuid const& guid : CThunGraspGUIDs)
103 {
104 if (GameObject* cthunGrasp = instance->GetGameObject(guid))
105 {
106 cthunGrasp->DespawnOrUnsummon(1s);
107 }
108 }
109 }
110 break;
111 default:
112 break;
113 }
114
115 InstanceScript::OnCreatureCreate(creature);
116 }
117
118 void OnGameObjectCreate(GameObject* go) override
119 {
120 switch (go->GetEntry())
121 {
122 case GO_CTHUN_GRASP:
123 CThunGraspGUIDs.push_back(go->GetGUID());
124 if (Creature* CThun = instance->GetCreature(CThunGUID))
125 {
126 if (!CThun->IsAlive())
127 {
128 go->DespawnOrUnsummon(1s);
129 }
130 }
131 break;
132 default:
133 break;
134 }
135
136 InstanceScript::OnGameObjectCreate(go);
137 }
138
139 void OnUnitDeath(Unit* unit) override
140 {
141 switch (unit->GetEntry())
142 {
143 case NPC_QIRAJI_SLAYER:
144 case NPC_QIRAJI_MINDSLAYER:
145 if (Creature* creature = unit->ToCreature())
146 {
147 if (CreatureGroup* formation = creature->GetFormation())
148 {
149 scheduler.Schedule(100ms, [formation](TaskContext /*context*/)
150 {
151 if (!formation->IsAnyMemberAlive(true))
152 {
153 if (Creature* leader = formation->GetLeader())
154 {
155 if (leader->IsAlive())
156 {
157 leader->AI()->SetData(0, 1);
158 }
159 }
160 }
161 });
162 }
163 }
164 break;
165 default:
166 break;
167 }
168 }

193 void SetData(uint32 type, uint32 data) override
194 {
195 switch (type)
196 {
197 case DATA_BUG_TRIO_DEATH:
198 if (data != 0)
199 ++BugTrioDeathCount;
200 else
201 BugTrioDeathCount = 0;
202 break;
203 case DATA_CTHUN_PHASE:
204 CthunPhase = data;
205 if (data == PHASE_CTHUN_DONE)
206 {
207 for (ObjectGuid const& guid : CThunGraspGUIDs)
208 {
209 if (GameObject* cthunGrasp = instance->GetGameObject(guid))
210 {
211 cthunGrasp->DespawnOrUnsummon(1s);
212 }
213 }
214 }
215 break;
216 default:
217 break;
218 }
219 }
220
221 bool SetBossState(uint32 type, EncounterState state) override
222 {
223 if (!InstanceScript::SetBossState(type, state))
224 return false;
225
226 switch (type)
227 {
228 case DATA_OURO:
229 if (state == FAIL)
230 {
231 if (Creature* ouroSpawner = GetCreature(DATA_OURO_SPAWNER))
232 ouroSpawner->Respawn();
233 }
234 break;
235 default:
236 break;
237 }
238
239 return true;
240 }

244 scheduler.Update(diff);
