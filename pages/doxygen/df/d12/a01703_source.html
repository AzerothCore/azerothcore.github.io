48bool RotateMovementGenerator::Update(Unit* owner, uint32 diff)
49{
50 float angle = owner->GetOrientation();
51 if (m_direction == ROTATE_DIRECTION_LEFT)
52 {
53 angle += (float)diff * static_cast<float>(M_PI * 2) / m_maxDuration;
54 while (angle >= static_cast<float>(M_PI * 2)) angle -= static_cast<float>(M_PI * 2);
55 }
56 else
57 {
58 angle -= (float)diff * static_cast<float>(M_PI * 2) / m_maxDuration;
59 while (angle < 0) angle += static_cast<float>(M_PI * 2);
60 }
61
62 owner->SetFacingTo(angle);
63
64 if (m_duration > diff)
65 m_duration -= diff;
66 else
67 return false;
68
69 return true;
70}

88void DistractMovementGenerator::Finalize(Unit* owner)
89{
90 owner->ClearUnitState(UNIT_STATE_DISTRACTED);
91
92 // If this is a creature, then return orientation to original position (for idle movement creatures)
93 if (owner->GetTypeId() == TYPEID_UNIT && owner->ToCreature())
94 {
95 float angle = owner->ToCreature()->GetHomePosition().GetOrientation();
96 owner->SetFacingTo(angle);
