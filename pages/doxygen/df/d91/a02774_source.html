116 void DoSpawnAmbush(bool _first)
117 {

119 spawnId = 1;

121 me->SummonCreature(NPC_RANGER, AmbushSpawn[spawnId].posX, AmbushSpawn[spawnId].posY, AmbushSpawn[spawnId].posZ, 0.0f,

165 break;
166 case 17:

169 SetRun();

175 void UpdateEscortAI(uint32 diff) override
176 {
177 //Check if we have a current target
178 if (!UpdateVictim())
179 {
180 if (HasEscortState(STATE_ESCORT_ESCORTING) && postEventCount)
181 {
182 if (postEventTimer <= diff)
183 {
184 postEventTimer = 3000;
185
186 if (Player* player = GetPlayerForEscort())
187 {
188 switch (postEventCount)
189 {
190 case 1:
191 Talk(SAY_RIN_PROGRESS_1, player);
192 ++postEventCount;
193 break;
194 case 2:
195 Talk(SAY_RIN_PROGRESS_2, player);
196 postEventCount = 0;
197 break;
198 }
199 }
200 else
201 {
202 me->DespawnOrUnsummon();
203 return;
204 }
205 }
206 else
207 postEventTimer -= diff;
208 }
209 return;
210 }
211 DoMeleeAttackIfReady();
212 }
213

219 };
