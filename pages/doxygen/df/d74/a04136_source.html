99 //_events.ScheduleEvent(EVENT_SHAMAN_FIRESHIELD, 0);

103 }
104
105 void UpdateAI(uint32 diff) override
106 {
107 if (_initAttack)
108 {
109 if (!me->IsInCombat())
110 if (Player* owner = me->GetCharmerOrOwnerPlayerOrPlayerItself())
111 if (Unit* target = owner->GetSelectedUnit())
112 if (me->CanCreatureAttack(target))
113 AttackStart(target);
114 _initAttack = false;
115 }
116
117 if (!UpdateVictim())
118 return;
119
120 _events.Update(diff);
121 while (uint32 eventId = _events.ExecuteEvent())
122 {
123 switch (eventId)
124 {
125 case EVENT_SHAMAN_FIRENOVA:
126 me->CastSpell(me, SPELL_SHAMAN_FIRENOVA, false);
127 _events.ScheduleEvent(EVENT_SHAMAN_FIRENOVA, urand(8000, 15000));
128 break;
129 case EVENT_SHAMAN_FIREBLAST:
130 me->CastSpell(me->GetVictim(), SPELL_SHAMAN_FIREBLAST, false);
131 _events.ScheduleEvent(EVENT_SHAMAN_FIREBLAST, urand(4000, 8000));
132 break;
133 default:
134 break;
135 }
136 }
137
138 DoMeleeAttackIfReady();
139 }
140
141private:
142 EventMap _events;
143 bool _initAttack;
144};
