97
98 if (Whirlwind2Timer <= diff)
99 {
100 DoCastVictim(SPELL_WHIRLWIND2);
101 Whirlwind2Timer = urand(10 * IN_MILLISECONDS, 25 * IN_MILLISECONDS);
102 }
103 else Whirlwind2Timer -= diff;
104
105 if (MortalStrikeTimer <= diff)
106 {
107 DoCastVictim(SPELL_MORTAL_STRIKE);
108 MortalStrikeTimer = urand(10 * IN_MILLISECONDS, 30 * IN_MILLISECONDS);
109 }
110 else MortalStrikeTimer -= diff;
111
112 // check if creature is not outside of building
113 if (ResetTimer <= diff)
114 {
115 if (me->GetDistance2d(me->GetHomePosition().GetPositionX(), me->GetHomePosition().GetPositionY()) > 50)
116 {
117 EnterEvadeMode();
118 Talk(YELL_EVADE);
119 }
120 ResetTimer = 5 * IN_MILLISECONDS;
121 }
122 else ResetTimer -= diff;
123
124 DoMeleeAttackIfReady();
125 }
