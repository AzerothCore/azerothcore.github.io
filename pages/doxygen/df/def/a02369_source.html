51 static bool HandleSpectatorCommand(ChatHandler* handler)
52 {
53 handler->PSendSysMessage("Incorrect syntax.");

60 static bool HandleSpectatorVersionCommand(ChatHandler* handler, uint16 version)
61 {
62 if (version < SPECTATOR_ADDON_VERSION)
63 ArenaSpectator::SendCommand(handler->GetSession()->GetPlayer(), "%sOUTDATED", SPECTATOR_ADDON_PREFIX);
64 return true;
65 }
66
67 static bool HandleSpectatorResetCommand(ChatHandler* handler)
68 {
69 Player* p = handler->GetSession()->GetPlayer();
70 if (!p->IsSpectator())
71 return true;
72 ArenaSpectator::HandleResetCommand(p);
73 return true;
74 }
75
76 static bool HandleSpectatorLeaveCommand(ChatHandler* handler)
77 {
78 Player* player = handler->GetSession()->GetPlayer();
79 if (!player->IsSpectator() || !player->FindMap() || !player->FindMap()->IsBattleArena())
80 {
81 handler->SendSysMessage("You are not a spectator.");
82 return true;
83 }
84
85 //player->SetIsSpectator(false);
86 player->TeleportToEntryPoint();
87 return true;
88 }
89
90 static bool HandleSpectatorSpectateCommand(ChatHandler* handler, std::string const& name)
91 {
92 if (!ArenaSpectator::HandleSpectatorSpectateCommand(handler, name))
93 return false;
94
95 return true;
96 }
97
98 static bool HandleSpectatorWatchCommand(ChatHandler* handler, std::string const& name)
99 {
100 if (!ArenaSpectator::HandleSpectatorWatchCommand(handler, name))
101 return false;
102
103 return true;
104 }
