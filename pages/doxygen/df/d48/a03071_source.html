67
68 std::list<Creature*> creatureList;
69 me->GetCreatureListWithEntryInGrid(creatureList, NPC_KALDOREI_ELITE, 200.0f);
70 creatureList.remove_if([&](Creature* creature) -> bool { return !creature->IsAlive(); });
71
72 me->GetMap()->DoForAllPlayers([&, creatureList](Player* player)
73 {
74 for (uint8 i = 0; i < creatureList.size(); ++i)
75 {
76 player->CastSpell(player, SPELL_CENARION_REPUTATION, true);
77 }
78
79 if (Creature* andorov = instance->instance->GetCreature(instance->GetGuidData(DATA_ANDOROV)))

83 player->CastSpell(player, SPELL_CENARION_REPUTATION, true);
84 }
85 }
86 });
87 }

95
96 void UpdateAI(uint32 diff) override
97 {
98 if (!UpdateVictim())
99 return;
100
101 events.Update(diff);
102
103 if (me->HasUnitState(UNIT_STATE_CASTING))
104 return;
105
106 while (uint32 eventId = events.ExecuteEvent())
107 {
108 switch (eventId)
109 {
110 case EVENT_DISARM:
111 DoCastVictim(SPELL_DISARM);
112 events.ScheduleEvent(EVENT_DISARM, 22s);
113 break;
114 case EVENT_THUNDERCRASH:
115 DoCastSelf(SPELL_THUNDERCRASH);
116 me->GetThreatMgr().ResetAllThreat();
117 events.ScheduleEvent(EVENT_THUNDERCRASH, 21s);
118 break;
119 default:
120 break;
121 }
122 }
123
124 DoMeleeAttackIfReady();

192 {

306 void UpdateEscortAI(uint32 diff) override
307 {
308 if (Endwaypoint && _initialAttackTimer)
309 {
310 me->SetFacingTo(2.8772139f);
311
312 if (_initialAttackTimer <= diff)
313 {
314 _initialAttackTimer = 0;
315
316 if (Creature* queez = instance->GetCreature(DATA_QUUEZ))
317 {
318 queez->AI()->AttackStart(me);
319 }
320 }
321 else
322 {
323 _initialAttackTimer -= diff;
324 }
325 }
326
327 if (_paused)
328 {
329 _paused = false;
330 me->SetFacingTo(5.63741350f);
331 }
332
333 if (!UpdateVictim())
334 {
335 return;
336 }
337
338 events.Update(diff);
339
340 if (me->HasUnitState(UNIT_STATE_CASTING))
341 {
342 return;
343 }
344
345 while (uint32 eventId = events.ExecuteEvent())
346 {
347 switch (eventId)
348 {
349 case EVENT_BASH:
350 DoCastVictim(SPELL_BASH);
351 events.ScheduleEvent(EVENT_BASH, urand(25, 38) * IN_MILLISECONDS);
352 break;
353 case EVENT_COMMAND_AURA:
354 DoCastSelf(SPELL_AURA_OF_COMMAND, true);
355 events.ScheduleEvent(EVENT_COMMAND_AURA, urand(10, 20) * IN_MILLISECONDS);
356 break;
357 case EVENT_STRIKE:
358 DoCastVictim(SPELL_STRIKE);
359 events.ScheduleEvent(EVENT_STRIKE, urand(4, 6) * IN_MILLISECONDS);
360 break;
361 default:
362 break;
363 }
364 }
365
366 DoMeleeAttackIfReady();
367 }
