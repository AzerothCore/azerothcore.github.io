168 // Generic base class to insert elements into arbitrary containers using push_back
169 template<typename Type>
170 class ContainerInserter
171 {
172 using InserterType = void(*)(void*, Type&&);
173
174 void* ref;
175 InserterType inserter;
176
177 // MSVC workaround
178 template<typename T>
179 static void InserterOf(void* ref, Type&& type)
180 {
181 static_cast<T*>(ref)->push_back(std::move(type));
182 }
183
184 protected:
185 template<typename T>
186 ContainerInserter(T& ref_) : ref(&ref_), inserter(&InserterOf<T>) { }
187
188 void Insert(Type type)
189 {
190 inserter(ref, std::move(type));
191 }
192 };
