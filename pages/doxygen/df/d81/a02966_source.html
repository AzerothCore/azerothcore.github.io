100 void KilledUnit(Unit* /*victim*/) override

102 if (!urand(0, 1))
103 return;
104
105 Talk(SAY_SLAY);
106 }
107
108 void ExplodeGhoul()
109 {
110 for (SummonList::const_iterator itr = summons.begin(); itr != summons.end(); ++itr)
111 if (Creature* cr = ObjectAccessor::GetCreature(*me, (*itr)))
112 if (cr->IsAlive())
113 {
114 me->CastSpell(cr, DUNGEON_MODE(SPELL_EXPLODE_GHOUL_N, SPELL_EXPLODE_GHOUL_H), false);
115 return;
116 }
117 }
118
119 void UpdateAI(uint32 diff) override
120 {
121 if (!UpdateVictim())
122 return;
123
124 events.Update(diff);
125 if (me->HasUnitState(UNIT_STATE_CASTING))
126 return;
127
128 switch (events.ExecuteEvent())
129 {
130 case EVENT_SPELL_SHADOW_BOLT:
131 me->CastSpell(me->GetVictim(), DUNGEON_MODE(SPELL_SHADOW_BOLT_N, SPELL_SHADOW_BOLT_H), false);
132 events.RepeatEvent(10000);
133 break;
134 case EVENT_SPELL_STEAL_FLESH:
135 if (!urand(0, 2))
136 Talk(SAY_STEAL_FLESH);
137 me->CastSpell(me->GetVictim(), SPELL_STEAL_FLESH_CHANNEL, false);
138 events.RepeatEvent(12000);
139 break;
140 case EVENT_SPELL_SUMMON_GHOULS:
141 if (!urand(0, 2))
142 Talk(SAY_SUMMON_GHOULS);
143 me->CastSpell(me, SPELL_SUMMON_GHOULS, false);
144 events.RepeatEvent(10000);
145 break;
146 case EVENT_EXPLODE_GHOUL:
147 if (!urand(0, 2))
148 Talk(SAY_EXPLODE_GHOUL);
149 ExplodeGhoul();
150 events.RepeatEvent(15000);
151 break;
152 case EVENT_SPELL_CURSE:
153 me->CastSpell(me->GetVictim(), SPELL_CURSE_OF_TWISTED_FAITH, false);
154 events.RepeatEvent(30000);
155 break;
156 }
157
158 DoMeleeAttackIfReady();
159 }
160 };
161};
