55 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
56 {
57 ClearGossipMenuFor(player);
58 hyjalAI* ai = CAST_AI(hyjalAI, creature->AI());
59 switch (action)
60 {
61 case GOSSIP_ACTION_INFO_DEF + 1:
62 ai->StartEvent(player);
63 break;
64 case GOSSIP_ACTION_INFO_DEF + 2:
65 ai->FirstBossDead = true;
66 ai->WaveCount = 9;
67 ai->StartEvent(player);
68 break;
69 case GOSSIP_ACTION_INFO_DEF + 3:
70 ai->Retreat();
71 break;
72 case GOSSIP_ACTION_INFO_DEF:
73 ai->Debug = !ai->Debug;
74 //LOG_DEBUG("scripts", "HyjalAI - Debug mode has been toggled");
75 break;
76 }
77 return true;
78 }
79
80 bool OnGossipHello(Player* player, Creature* creature) override
81 {
82 hyjalAI* ai = CAST_AI(hyjalAI, creature->AI());
83 if (ai->EventBegun)
84 return false;
85
86 uint32 RageEncounter = ai->GetInstanceData(DATA_RAGEWINTERCHILLEVENT);
87 uint32 AnetheronEncounter = ai->GetInstanceData(DATA_ANETHERONEVENT);
88 if (RageEncounter == NOT_STARTED)
89 AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_ITEM_BEGIN_ALLY, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
90 else if (RageEncounter == DONE && AnetheronEncounter == NOT_STARTED)
91 AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_ITEM_ANETHERON, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
92 else if (RageEncounter == DONE && AnetheronEncounter == DONE)
93 AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_ITEM_RETREAT, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
94
95 if (player->IsGameMaster())
96 AddGossipItemFor(player, GOSSIP_ICON_TRAINER, GOSSIP_ITEM_GM1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
97
98 SendGossipMenuFor(player, 907, creature->GetGUID());
99 return true;
100 }
101

103 {

224 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
225 {
226 ClearGossipMenuFor(player);
227 if (action == GOSSIP_ACTION_INFO_DEF)
228 {
229 ItemPosCountVec dest;
230 uint8 msg = player->CanStoreNewItem(NULL_BAG, NULL_SLOT, dest, ITEM_TEAR_OF_GODDESS, 1);
231 if (msg == EQUIP_ERR_OK)
232 if (Item* item = player->StoreNewItem(dest, ITEM_TEAR_OF_GODDESS, true))
233 player->SendNewItem(item, 1, true, false, true);
234
235 SendGossipMenuFor(player, 907, creature->GetGUID());
236 }
237 return true;
238 }
239
240 bool OnGossipHello(Player* player, Creature* creature) override
241 {
242 hyjalAI* ai = CAST_AI(hyjalAI, creature->AI());
243 uint32 AzgalorEvent = ai->GetInstanceData(DATA_AZGALOREVENT);
244
245 // Only let them get item if Azgalor is dead.
246 if (AzgalorEvent == DONE && !player->HasItemCount(ITEM_TEAR_OF_GODDESS))
247 AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_ITEM_TYRANDE, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
248 SendGossipMenuFor(player, 907, creature->GetGUID());
249 return true;
250 }
