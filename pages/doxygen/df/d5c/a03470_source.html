219 void UpdateAI(uint32 diff) override
220 {
221 if (!IsInRoom())
222 return;
223
224 if (!UpdateVictim())
225 return;
226
227 events.Update(diff);
228 if (me->HasUnitState(UNIT_STATE_CASTING))
229 return;
230
231 switch (events.ExecuteEvent())
232 {
233 // GROUND
234 case EVENT_CURSE:
235 if (events.GetPhaseMask() == 0)
236 {
237 me->CastCustomSpell(RAID_MODE(SPELL_CURSE_OF_THE_PLAGUEBRINGER_10, SPELL_CURSE_OF_THE_PLAGUEBRINGER_25), SPELLVALUE_MAX_TARGETS, RAID_MODE(3, 10), me, false);
238 }
239 events.RepeatEvent(25000);
240 break;
241 case EVENT_SUMMON_PLAGUED_WARRIOR_ANNOUNCE:
242 Talk(SAY_SUMMON);
243 Talk(EMOTE_SUMMON);
244 events.RepeatEvent(30000);
245 events.ScheduleEvent(EVENT_SUMMON_PLAGUED_WARRIOR_REAL, 4000);
246 break;
247 case EVENT_SUMMON_PLAGUED_WARRIOR_REAL:
248 me->CastSpell(me, SPELL_SUMMON_PLAGUED_WARRIORS, true);
249 SummonHelper(NPC_PLAGUED_WARRIOR, RAID_MODE(2, 3));
250 break;
251 case EVENT_MOVE_TO_BALCONY:
252 Talk(EMOTE_TELEPORT_BALCONY);
253 me->CastSpell(me, SPELL_TELEPORT, true);
254 StartBalconyPhase();
255 break;
256 case EVENT_BLINK:
257 DoResetThreat();
258 me->CastSpell(me, RAID_MODE(SPELL_CRIPPLE_10, SPELL_CRIPPLE_25), false);
259 me->CastSpell(me, SPELL_BLINK, true);
260 Talk(EMOTE_BLINK);
261 events.RepeatEvent(30000);
262 break;
263 // BALCONY
264 case EVENT_BALCONY_SUMMON_ANNOUNCE:
265 Talk(EMOTE_SUMMON_WAVE);
266 events.RepeatEvent(30000);
267 events.ScheduleEvent(EVENT_BALCONY_SUMMON_REAL, 4000);
268 break;
269 case EVENT_BALCONY_SUMMON_REAL:
270 me->CastSpell(me, SPELL_SUMMON_PLAGUED_WARRIORS, true); // visual
271 switch (timesInBalcony)
272 {
273 case 0:
274 SummonHelper(NPC_PLAGUED_CHAMPION, RAID_MODE(2, 4));
275 break;
276 case 1:
277 SummonHelper(NPC_PLAGUED_CHAMPION, RAID_MODE(1, 2));
278 SummonHelper(NPC_PLAGUED_GUARDIAN, RAID_MODE(1, 2));
279 break;
280 default:
281 SummonHelper(NPC_PLAGUED_GUARDIAN, RAID_MODE(2, 4));
282 break;
283 }
284 break;
285 case EVENT_MOVE_TO_GROUND:
286 Talk(EMOTE_TELEPORT_BACK);
287 me->CastSpell(me, SPELL_TELEPORT_BACK, true);
288 timesInBalcony++;
289 if (timesInBalcony == 3)
290 {
291 DoCastSelf(SPELL_BERSERK);
292 }
293 StartGroundPhase();
294 break;
295 }
296 if (me->HasReactState(REACT_AGGRESSIVE))
297 DoMeleeAttackIfReady();
298 }
299 };
