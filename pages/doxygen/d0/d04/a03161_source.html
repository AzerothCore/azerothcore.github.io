87 {
88 if (CanEmote)
89 {
90 if (ResetTimer <= diff)
91 {
92 if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID))
93 {
94 if (player->GetTypeId() == TYPEID_PLAYER && player->GetQuestStatus(QUEST_SHATTERED_SALUTE) == QUEST_STATUS_INCOMPLETE)
95 player->FailQuest(QUEST_SHATTERED_SALUTE);
96 }
97 Reset();
98 }
99 else ResetTimer -= diff;
100 }
101
102 if (CanTalk && !CanEmote)
103 {
104 if (SaluteTimer <= diff)
105 {
106 me->HandleEmoteCommand(EMOTE_ONESHOT_SALUTE);
107 CanEmote = true;
108 ResetTimer = 60000;
109 }
110 else SaluteTimer -= diff;
111 }
112
113 if (!UpdateVictim())
114 return;
115
116 DoMeleeAttackIfReady();
117 }
118
119 void ReceiveEmote(Player* player, uint32 emote) override
120 {
121 if (emote == TEXT_EMOTE_SALUTE && player->GetQuestStatus(QUEST_SHATTERED_SALUTE) == QUEST_STATUS_INCOMPLETE)
122 {
123 if (CanEmote)
124 {
125 player->AreaExploredOrEventHappens(QUEST_SHATTERED_SALUTE);
126 Reset();
127 }
128 }

264 _scheduler.Schedule(2s, [this](TaskContext /*context*/)

266 Talk(SAY_THRALL_ON_QUEST_REWARD_0);

268 .Schedule(13s, [this](TaskContext /*context*/)
269 {

271 })
272 .Schedule(15s, [this](TaskContext /*context*/)
273 {
274 DoCastAOE(SPELL_WARCHIEF_BLESSING, true);
275 me->SetNpcFlag(UNIT_NPC_FLAG_GOSSIP);
276 me->GetMap()->DoForAllPlayers([&](Player* p)
277 {
278 if (p->IsAlive() && !p->IsGameMaster())
279 {
280 if (p->GetAreaId() == AREA_ORGRIMMAR || p->GetAreaId() == AREA_RAZOR_HILL || p->GetAreaId() == AREA_CROSSROADS || p->GetAreaId() == AREA_CAMP_TAURAJO)

283 }
284 }
285 });
286 });
287 }
288 }
