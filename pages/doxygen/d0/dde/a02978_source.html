75 events.Reset();

80 void JustSummoned(Creature* summon) override
81 {
82 summons.Summon(summon);
83 if (Creature* thrall = ObjectAccessor::GetCreature(*me, me->GetInstanceScript()->GetGuidData(DATA_THRALL_GUID)))
84 thrall->AI()->JustSummoned(summon);
85 summon->SetImmuneToAll(true);
86
87 if (summon->GetEntry() == NPC_SKARLOC_MOUNT)
88 return;
89
90 if (summons.size() == 1)
91 summon->GetMotionMaster()->MovePoint(0, 2060.788f, 237.301f, 63.999f);
92 else
93 summon->GetMotionMaster()->MovePoint(0, 2056.870f, 234.853f, 63.839f);
94 }
95
96 void InitializeAI() override
97 {
98 ScriptedAI::InitializeAI();
99
100 Movement::PointsArray path;
101 path.push_back(G3D::Vector3(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ()));
102 for (uint8 i = 0; i < WAYPOINTS_COUNT; ++i)
103 path.push_back(G3D::Vector3(startPath[i].GetPositionX(), startPath[i].GetPositionY(), startPath[i].GetPositionZ()));
104
105 me->GetMotionMaster()->MoveSplinePath(&path);
106 me->SetImmuneToAll(true);
107 me->Mount(SKARLOC_MOUNT_MODEL);
108 }
109
110 void MovementInform(uint32 type, uint32 id) override
111 {
112 if (type != ESCORT_MOTION_TYPE)
113 return;
114
115 // Xinef: we can rely here on internal counting
116 if (id == 1)
117 {
118 me->SummonCreature(NPC_DURNHOLDE_MAGE, 2038.549f, 273.303f, 63.420f, 5.30f, TEMPSUMMON_MANUAL_DESPAWN);
119 me->SummonCreature(NPC_DURNHOLDE_VETERAN, 2032.810f, 269.416f, 63.561f, 5.30f, TEMPSUMMON_MANUAL_DESPAWN);
120 }
121 else if (id == 2)
122 {
123 me->SummonCreature(NPC_SKARLOC_MOUNT, 2049.12f, 252.31f, 62.855f, me->GetOrientation(), TEMPSUMMON_MANUAL_DESPAWN);
124 me->Dismount();
125 me->SetWalk(true);
126 for (SummonList::const_iterator itr = summons.begin(); itr != summons.end(); ++itr)
127 if (Creature* summon = ObjectAccessor::GetCreature(*me, *itr))
128 summon->SetWalk(true);
129 }
130
131 if (me->movespline->Finalized())
132 {
133 events2.ScheduleEvent(EVENT_INITIAL_TALK, 500);
134 events2.ScheduleEvent(EVENT_START_FIGHT, 8000);
135 }
136 }
137
138 void EnterCombat(Unit* /*who*/) override
139 {
140 me->CastSpell(me, SPELL_DEVOTION_AURA, true);
141
142 events.ScheduleEvent(EVENT_SPELL_HOLY_LIGHT, 15000);
143 events.ScheduleEvent(EVENT_SPELL_CLEANSE, 6000);
144 events.ScheduleEvent(EVENT_SPELL_HAMMER, 20000);
145 events.ScheduleEvent(EVENT_SPELL_HOLY_SHIELD, 10000);
146 if (IsHeroic())
147 events.ScheduleEvent(EVENT_SPELL_CONSECRATION, 1000);
148 }
149
150 void KilledUnit(Unit* /*victim*/) override

152 Talk(SAY_SLAY);

154
155 void JustDied(Unit* /*killer*/) override
156 {
157 Talk(SAY_DEATH);
158 me->GetInstanceScript()->SetData(DATA_ESCORT_PROGRESS, ENCOUNTER_PROGRESS_SKARLOC_KILLED);
159 me->GetInstanceScript()->SetData(DATA_THRALL_ADD_FLAG, 0);
160 }
161
162 void UpdateAI(uint32 diff) override
163 {
164 events2.Update(diff);
165 switch (events2.ExecuteEvent())
166 {
167 case EVENT_INITIAL_TALK:
168 Talk(SAY_ENTER);
169 break;
170 case EVENT_START_FIGHT:
171 me->SetImmuneToAll(false);
172 me->SetInCombatWithZone();
173 for (SummonList::const_iterator itr = summons.begin(); itr != summons.end(); ++itr)
174 if (Creature* summon = ObjectAccessor::GetCreature(*me, *itr))
175 if (summon->GetEntry() != NPC_SKARLOC_MOUNT)
176 {
177 summon->SetImmuneToAll(false);
178 summon->SetInCombatWithZone();
179 }
180 break;
181 }
182
183 if (!UpdateVictim())
184 return;
185
186 events.Update(diff);
187 if (me->HasUnitState(UNIT_STATE_CASTING))
188 return;
189
190 switch (events.ExecuteEvent())
191 {
192 case EVENT_SPELL_HOLY_LIGHT:
193 me->CastSpell(me, SPELL_HOLY_LIGHT, false);
194 events.ScheduleEvent(EVENT_SPELL_HOLY_LIGHT, 20000);
195 break;
196 case EVENT_SPELL_CLEANSE:
197 if (roll_chance_i(33))
198 Talk(SAY_TAUNT);
199 me->CastSpell(me, SPELL_CLEANSE, false);
200 events.ScheduleEvent(EVENT_SPELL_CLEANSE, 10000);
201 break;
202 case EVENT_SPELL_HAMMER:
203 me->CastSpell(me->GetVictim(), SPELL_HAMMER_OF_JUSTICE, false);
204 events.ScheduleEvent(EVENT_SPELL_HAMMER, 30000);
205 break;
206 case EVENT_SPELL_HOLY_SHIELD:
207 me->CastSpell(me, SPELL_CLEANSE, false);
208 events.ScheduleEvent(SPELL_HOLY_SHIELD, 30000);
209 break;
210 case EVENT_SPELL_CONSECRATION:
211 me->CastSpell(me, SPELL_CONSECRATION, false);
212 events.ScheduleEvent(EVENT_SPELL_CONSECRATION, 20000);
213 break;
214 }
215
216 DoMeleeAttackIfReady();
217 }
218 };
