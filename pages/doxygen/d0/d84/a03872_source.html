94 events.ScheduleEvent(EVENT_YELL, urand(25000, 100000));
95 events.RescheduleEvent(EVENT_SPELL_NEEDLE, 10000);
96 events.RescheduleEvent(EVENT_SPELL_SPINE, 20001);
97 events.RescheduleEvent(EVENT_SPELL_SHIELD, 60000);

99
100 void UpdateAI(uint32 diff) override
101 {
102 if (!UpdateVictim())
103 return;
104
105 events.Update(diff);
106 if (me->HasUnitState(UNIT_STATE_CASTING))
107 return;
108
109 switch (events.ExecuteEvent())
110 {
111 case EVENT_SPELL_SHIELD:
112 me->CastSpell(me, SPELL_TIDAL_SHIELD, false);
113 events.DelayEvents(10000);
114 events.ScheduleEvent(EVENT_SPELL_SHIELD, 60000);
115 break;
116 case EVENT_SPELL_BERSERK:
117 Talk(SAY_ENRAGE);
118 me->CastSpell(me, SPELL_BERSERK, true);
119 break;
120 case EVENT_SPELL_NEEDLE:
121 me->CastCustomSpell(SPELL_NEEDLE_SPINE, SPELLVALUE_MAX_TARGETS, 3, me, false);
122 events.ScheduleEvent(EVENT_SPELL_NEEDLE, 15000);
123 break;
124 case EVENT_SPELL_SPINE:
125 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1))
126 {
127 me->CastSpell(target, SPELL_IMPALING_SPINE, false);
128 target->CastSpell(target, SPELL_SUMMON_IMPALING_SPINE, true);
129 Talk(SAY_NEEDLE);
130 }
131 events.ScheduleEvent(EVENT_SPELL_SPINE, 20000);
132 return;
133 case EVENT_YELL:
134 Talk(SAY_SPECIAL);
135 events.ScheduleEvent(EVENT_YELL, urand(25000, 100000));
136 break;
137 }
138
139 DoMeleeAttackIfReady();
140 }
141 };
