162 Creature* SelectTarget_MostHPLostFriendlyMissingBuff(uint32 spell, float range)
163 {
164 std::list<Creature*> lst = DoFindFriendlyMissingBuff(range, spell);
165 if( lst.empty() )
166 return nullptr;
167 std::list<Creature*>::const_iterator iter = lst.begin();
168 uint32 lowestHP = (*iter)->GetMaxHealth() - (*iter)->GetHealth();
169 for( std::list<Creature*>::const_iterator itr = lst.begin(); itr != lst.end(); ++itr )
170 if( ((*itr)->GetMaxHealth() - (*itr)->GetHealth()) > lowestHP )
171 {
172 iter = itr;
173 lowestHP = (*itr)->GetMaxHealth() - (*itr)->GetHealth();
174 }
175 return (*iter);
176 }

186 if( target && me->GetDistance2d(target) < distance )

194 ThreatContainer::StorageType const& tList = me->GetThreatMgr().GetThreatList();
195 Unit* target;
196 for( ThreatContainer::StorageType::const_iterator iter = tList.begin(); iter != tList.end(); ++iter )
197 {
198 target = ObjectAccessor::GetUnit((*me), (*iter)->getUnitGuid());
199 if( target && target->getPowerType() == POWER_MANA && (!casting || target->HasUnitState(UNIT_STATE_CASTING)) && me->GetExactDist(target) <= range )
200 return target;

202 return nullptr;

206 {
207 if (!me->IsInCombat())
208 return;
209
210 if( threatTimer <= diff )
211 {
212 RecalculateThreat();
213 threatTimer = urand(8750, 9250);
214 }
215 else
216 threatTimer -= diff;
217
218 if( me->getPowerType() == POWER_MANA )
219 {
220 if( powerTimer <= diff )
221 {
222 me->ModifyPower(POWER_MANA, me->GetMaxPower(POWER_MANA) / 3);
223 powerTimer = 4000;
224 }
225 else
226 powerTimer -= diff;
227 }
228 else if( me->getPowerType() == POWER_ENERGY )
