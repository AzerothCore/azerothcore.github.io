118 void AddBuddyToList(ObjectGuid CreatureGUID)
119 {
120 if (CreatureGUID == me->GetGUID())
121 return;
122
123 for (int i = 0; i < 3; ++i)
124 {
125 if (NearbyGUID[i] == CreatureGUID)

135 void GiveBuddyMyList(Creature* c)
136 {
137 aqsentinelAI* cai = CAST_AI(aqsentinelAI, (c)->AI());

139 if (NearbyGUID[i] && NearbyGUID[i] != c->GetGUID())

151 void CallBuddiesToAttack(Unit* who)

168 void AddSentinelsNear(Unit* /*nears*/)
169 {
170 std::list<Creature*> assistList;
171 me->GetCreatureListWithEntryInGrid(assistList, 15264, 100.0f);
172
173 if (assistList.empty())
174 return;
175
176 for (std::list<Creature*>::const_iterator iter = assistList.begin(); iter != assistList.end(); ++iter)
177 AddBuddyToList((*iter)->GetGUID());

180 int pickAbilityRandom(bool* chosenAbilities)
181 {
182 for (int t = 0; t < 2; ++t)
183 {
184 for (int i = !t ? (rand() % 9) : 0; i < 9; ++i)
185 {
186 if (!chosenAbilities[i])
187 {
188 chosenAbilities[i] = true;
189 return i;
190 }
191 }
192 }
193 return 0; // should never happen
194 }
195
196 void GetOtherSentinels(Unit* who)
197 {
198 bool* chosenAbilities = new bool[9];
199 memset(chosenAbilities, 0, 9 * sizeof(bool));
200 selectAbility(pickAbilityRandom(chosenAbilities));
201
202 ClearBuddyList();
203 AddSentinelsNear(me);
204 int bli;
205 for (bli = 0; bli < 3; ++bli)
206 {
207 if (!NearbyGUID[bli])
208 break;
209
210 Creature* pNearby = ObjectAccessor::GetCreature(*me, NearbyGUID[bli]);
211 if (!pNearby)
212 break;
213
214 AddSentinelsNear(pNearby);
215 CAST_AI(aqsentinelAI, pNearby->AI())->gatherOthersWhenAggro = false;
216 CAST_AI(aqsentinelAI, pNearby->AI())->selectAbility(pickAbilityRandom(chosenAbilities));
217 }
218 /*if (bli < 3)
219 DoYell("I dont have enough buddies.", LANG_NEUTRAL, 0);*/
220 SendMyListToBuddies();
221 CallBuddiesToAttack(who);
222
223 delete[] chosenAbilities;
224 }

228 void Reset() override
229 {
230 if (!me->isDead())
231 {
232 for (int i = 0; i < 3; ++i)
233 {
234 if (!NearbyGUID[i])

236 if (Creature* pNearby = ObjectAccessor::GetCreature(*me, NearbyGUID[i]))
237 {
238 if (pNearby->isDead())
239 pNearby->Respawn();

241 }
