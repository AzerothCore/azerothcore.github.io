67 void ExecuteEvent(uint32 eventId) override
68 {
69 switch (eventId)
70 {
71 case EVENT_FRENZY:
72 {
73 Talk(EMOTE_FRENZY);
74 DoCastSelf(SPELL_FRENZY);
75 events.RepeatEvent(urand(15000, 20000));
76 break;
77 }
78 case EVENT_PANIC:
79 {
80 DoCastVictim(SPELL_PANIC);
81 events.RepeatEvent(urand(31000, 38000));
82 break;
83 }
84 case EVENT_LAVA_BOMB:
85 {
86 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, MELEE_TARGET_LOOKUP_DIST, true))
87 {
88 DoCast(target, SPELL_LAVA_BOMB);
89 }
90
91 events.RepeatEvent(urand(12000, 15000));
92 break;
93 }
94 case EVENT_LAVA_BOMB_RANGED:
95 {
96 std::list<Unit*> targets;
97 SelectTargetList(targets, [this](Unit* target)
98 {
99 return target && target->IsPlayer() && target->GetDistance(me) > MELEE_TARGET_LOOKUP_DIST && target->GetDistance(me) < 100.0f;
100 }, 1, SelectTargetMethod::Random);
101
102 if (!targets.empty())
103 {
104 DoCast(targets.front() , SPELL_LAVA_BOMB_RANGED);
105 }
106 events.RepeatEvent(urand(12000, 15000));
107 break;
108 }
109 }
110 }
111 };
