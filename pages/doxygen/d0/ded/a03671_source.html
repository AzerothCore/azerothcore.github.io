56class boss_archavon : public CreatureScript
57{
58 public:
59 boss_archavon() : CreatureScript("boss_archavon") { }

61 struct boss_archavonAI : public ScriptedAI
62 {
63 boss_archavonAI(Creature* creature) : ScriptedAI(creature)
64 {
65 pInstance = me->GetInstanceScript();
66 }
67
68 InstanceScript* pInstance;
69 EventMap events;
70
71 void Reset() override
72 {
73 events.Reset();
74 if (pInstance)
75 {
76 if (pInstance->GetData(DATA_STONED))
77 {
78 if (Aura* aur = me->AddAura(SPELL_STONED_AURA, me))
79 {
80 aur->SetMaxDuration(60 * MINUTE* IN_MILLISECONDS);
81 aur->SetDuration(60 * MINUTE* IN_MILLISECONDS);
82 }
83 }
84
85 pInstance->SetData(EVENT_ARCHAVON, NOT_STARTED);
86 }
87 }
88
89 void AttackStart(Unit* who) override
90 {
91 if (me->HasAura(SPELL_STONED_AURA))
92 {
93 return;
94 }
95
96 ScriptedAI::AttackStart(who);
97 }
98
99 void EnterCombat(Unit* /*who*/) override
100 {
101 events.ScheduleEvent(EVENT_ROCK_SHARDS, 15000);
102 events.ScheduleEvent(EVENT_CHOKING_CLOUD, 30000);
103 events.ScheduleEvent(EVENT_STOMP, 45000);
104 events.ScheduleEvent(EVENT_BERSERK, 300000);
105
106 if (pInstance)
107 {
108 pInstance->SetData(EVENT_ARCHAVON, IN_PROGRESS);
109 }
110 }
111
112 void JustDied(Unit*) override
113 {
114 if (pInstance)
115 {
116 pInstance->SetData(EVENT_ARCHAVON, DONE);
117 }
118 }
119
120 void UpdateAI(uint32 diff) override
121 {
122 if (!UpdateVictim())
123 {
124 return;
125 }
126
127 events.Update(diff);
128
129 if (me->HasUnitState(UNIT_STATE_CASTING))
130 {
131 return;
132 }
133
134 switch (events.ExecuteEvent())
135 {
136 case EVENT_ROCK_SHARDS:
137 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
138 {
139 DoCast(target, SPELL_ROCK_SHARDS);
140 }
141
142 events.RepeatEvent(15000);
143 break;
144 case EVENT_CHOKING_CLOUD:
145 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1))
146 {
147 DoCast(target, RAID_MODE(SPELL_CRUSHING_LEAP_10, SPELL_CRUSHING_LEAP_25), true); //10y ~ 80y, ignore range
148 }
149
150 events.RepeatEvent(30000);
151 break;
152 case EVENT_STOMP:
153 {
154 char buffer[100];
155 sprintf(buffer, "Archavon the Stone Watcher lunges for %s!", me->GetVictim()->GetName().c_str());
156 me->TextEmote(buffer);
157
158 DoCastVictim(RAID_MODE(SPELL_STOMP_10, SPELL_STOMP_25));
159
160 events.RepeatEvent(45000);
161 events.ScheduleEvent(EVENT_IMPALE, 3000);
162 break;
163 }
164 case EVENT_IMPALE:
165 DoCastVictim(RAID_MODE(SPELL_IMPALE_10, SPELL_IMPALE_25));
166 break;
167 case EVENT_BERSERK:
168 DoCast(me, SPELL_BERSERK, true);
169 Talk(EMOTE_BERSERK);
170 break;
171 default:
172 break;
173 }
174
175 DoMeleeAttackIfReady();
176 }
177 };
178
179 CreatureAI* GetAI(Creature* creature) const override
180 {
181 return GetVaultOfArchavonAI<boss_archavonAI>(creature);
182 }
183};

191 class spell_archavon_rock_shards_SpellScript : public SpellScript
192 {
193 PrepareSpellScript(spell_archavon_rock_shards_SpellScript);
194
195 void HandleScript(SpellEffIndex effIndex)
196 {
197 PreventHitDefaultEffect(effIndex);
198
199 Unit* caster = GetCaster();
200 Unit* target = GetHitUnit();
201
202 if (!caster || !target)
203 {
204 return;
205 }
206
207 Map* map = caster->GetMap();
208 if (!map)
209 {
210 return;
211 }
212
213 caster->CastSpell(target, SPELL_ROCK_SHARDS_LEFT_HAND_VISUAL, true);
214 caster->CastSpell(target, SPELL_ROCK_SHARDS_RIGHT_HAND_VISUAL, true);
215
216 uint32 spellId = map->Is25ManRaid() ? SPELL_ROCK_SHARDS_DAMAGE_25 : SPELL_ROCK_SHARDS_DAMAGE_10;
217 caster->CastSpell(target, spellId, true);
218 }
219
220 void Register() override
221 {
222 OnEffectHitTarget += SpellEffectFn(spell_archavon_rock_shards_SpellScript::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
223 }
224 };
225
226 SpellScript* GetSpellScript() const override
227 {
228 return new spell_archavon_rock_shards_SpellScript();
229 }
