86 void UpdateAI(uint32 diff) override
87 {
88 if (!UpdateVictim())
89 return;
90
91 events.Update(diff);
92 if (me->HasUnitState(UNIT_STATE_CASTING))
93 return;
94
95 switch (events.ExecuteEvent())
96 {
97 case EVENT_VOID_BLAST:
98 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100.0f, true))
99 {
100 DoCast(target, SPELL_VOID_BLAST);
101 ++VoidBlastCounter;
102 }
103
104 if (VoidBlastCounter == 5)
105 {
106 VoidBlastCounter = 0;
107 events.RescheduleEvent(EVENT_VOID_BLAST, urand(15000, 25000));
108 }
109 else
110 {
111 events.RescheduleEvent(EVENT_VOID_BLAST, 500);
112 events.DelayEvents(EVENT_DARK_SHELL, 500);
113 }
114 break;
115 case EVENT_DARK_SHELL:
116 if (me->IsNonMeleeSpellCast(false))
117 {
118 me->InterruptNonMeleeSpells(true);
119 }
120
121 Talk(EMOTE_DARK_SHELL);
122 DoCast(me, SPELL_DARK_SHELL);
123 events.RescheduleEvent(EVENT_DARK_SHELL, 20000);
124 break;
125 default:
126 break;
127 }
128
129 DoMeleeAttackIfReady();
130 }
131
132 private:
133 uint32 VoidBlastCounter;
