141 void EnterCombat(Unit* /*who*/) override
142 {
143 _EnterCombat();
144 me->SetReactState(REACT_AGGRESSIVE);
145 // Always running events
146 events.ScheduleEvent(EVENT_THRASH, 5000);
147 // Phase one events (regular form)
148 events.ScheduleEvent(EVENT_HOLY_NOVA, urand(5000, 15000), 0, PHASE_ONE);
149 events.ScheduleEvent(EVENT_DISPEL_MAGIC, 35000, 0, PHASE_ONE);
150 events.ScheduleEvent(EVENT_HOLY_FIRE, urand(10000,20000), 0, PHASE_ONE);
151 events.ScheduleEvent(EVENT_RENEW, 30000, 0, PHASE_ONE);
152 events.ScheduleEvent(EVENT_HOLY_WRATH, urand(15000, 25000), 0, PHASE_ONE);
153
154 events.SetPhase(PHASE_ONE);
155
156 // Set zone in combat
157 DoZoneInCombat();
158 SetCombatCombras();
159 }

161 void DamageTaken(Unit*, uint32& /*damage*/, DamageEffectType, SpellSchoolMask) override
162 {
163 // check if venoxis is ready to transform
164 if (!_transformed && !HealthAbovePct(50))
165 {
166 _transformed = true;
167 // schedule the event that changes our phase
168 events.ScheduleEvent(EVENT_TRANSFORM, 100);
169 }
170 // we're losing health, bad, go frenzy
171 else if (!_frenzied && !HealthAbovePct(20))
172 {
173 _frenzied = true;
174 events.ScheduleEvent(EVENT_FRENZY, 100);
175 }
176 }

178 void UpdateAI(uint32 diff) override
179 {
180 if (!UpdateVictim())
181 return;
182
183 events.Update(diff);
184
185 // return back to main code if we're still casting
186 if (me->HasUnitState(UNIT_STATE_CASTING))
187 return;
188
189 while (uint32 eventId = events.ExecuteEvent())
190 {
191 switch (eventId)
192 {
193 // thrash is available in all phases
194 case EVENT_THRASH:
195 DoCast(me, SPELL_THRASH, true);
196 events.ScheduleEvent(EVENT_THRASH, urand(10000, 20000));
197 break;
198
199 // troll form spells and Actions (first part)
200 case EVENT_DISPEL_MAGIC:
201 DoCast(me, SPELL_DISPEL_MAGIC);
202 events.ScheduleEvent(EVENT_DISPEL_MAGIC, urand(15000, 20000), 0, PHASE_ONE);
203 break;
204 case EVENT_RENEW:
205 DoCast(me, SPELL_RENEW);
206 events.ScheduleEvent(EVENT_RENEW, urand(25000, 30000), 0, PHASE_ONE);
207 break;
208 case EVENT_HOLY_WRATH:
209 if (Unit* target = SelectTarget(SelectTargetMethod::MaxThreat))
210 DoCast(target, SPELL_HOLY_WRATH);
211 events.ScheduleEvent(EVENT_HOLY_WRATH, urand(12000, 22000), 0, PHASE_ONE);
212 break;
213 case EVENT_HOLY_FIRE:
214 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
215 DoCast(target, SPELL_HOLY_FIRE);
216 events.ScheduleEvent(EVENT_HOLY_FIRE, urand(10000, 24000), 0, PHASE_ONE);
217 break;
218 case EVENT_HOLY_NOVA:
219 DoCastSelf(SPELL_HOLY_NOVA);
220 events.ScheduleEvent(EVENT_HOLY_NOVA, urand(10000, 24000), 0, PHASE_ONE);
221 break;
222
223 //
224 // snake form spells and Actions
225 //
226
227 case EVENT_VENOM_SPIT:
228 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
229 DoCast(target, SPELL_VENOM_SPIT);
230 events.ScheduleEvent(EVENT_VENOM_SPIT, urand(5000, 15000), 0, PHASE_TWO);
231 break;
232 case EVENT_POISON_CLOUD:
233 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
234 DoCast(target, SPELL_POISON_CLOUD);
235 events.ScheduleEvent(EVENT_POISON_CLOUD, urand(15000, 20000), 0, PHASE_TWO);
236 break;
237 case EVENT_PARASITIC_SERPENT:
238 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
239 DoCast(target, SPELL_SUMMON_PARASITIC_SERPENT);
240 events.ScheduleEvent(EVENT_PARASITIC_SERPENT, 15000, 0, PHASE_TWO);
241 break;
242 case EVENT_FRENZY:
243 // frenzy at 20% health
244 DoCast(me, SPELL_FRENZY, true);
245 break;
246
247 //
248 // shape and phase-changing
249 //
250
251 case EVENT_TRANSFORM:
252 // shapeshift at 50% health
253 DoCast(me, SPELL_VENOXIS_TRANSFORM);
254 Talk(SAY_VENOXIS_TRANSFORM);
255 DoResetThreat();
256
257 // phase two events (snakeform)
258 events.ScheduleEvent(EVENT_VENOM_SPIT, 5000, 0, PHASE_TWO);
259 events.ScheduleEvent(EVENT_POISON_CLOUD, 10000, 0, PHASE_TWO);
260 events.ScheduleEvent(EVENT_PARASITIC_SERPENT, 30000, 0, PHASE_TWO);
261
262 // transformed, start phase two
263 events.SetPhase(PHASE_TWO);
264
265 break;
266 default:
267 break;
268 }
269 }
270
271 DoMeleeAttackIfReady();
272 }
273
274 private:
275 uint8 _inMeleeRange;
276 bool _transformed;
277 bool _frenzied;
278 };

282 return GetZulGurubAI<boss_venoxisAI>(creature);
283 }
