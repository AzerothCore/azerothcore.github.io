139 void UpdateAI(uint32 diff) override
140 {
141 if (!UpdateVictim())
142 return;
143
144 events.Update(diff);
145 if (me->HasUnitState(UNIT_STATE_CASTING))
146 return;
147
148 switch (events.ExecuteEvent())
149 {
150 case EVENT_CHECK_HEALTH1:
151 if (me->HealthBelowPct(70))
152 {
153 Talk(SAY_SUMMON_SNAKES);
154 events.ScheduleEvent(EVENT_SUMMON1, 1000);
155 break;
156 }
157 events.ScheduleEvent(EVENT_CHECK_HEALTH1, 1000);
158 break;
159 case EVENT_CHECK_HEALTH2:
160 if (me->HealthBelowPct(50))
161 {
162 Talk(SAY_SUMMON_CONSTRICTORS);
163 events.ScheduleEvent(EVENT_SUMMON2, 1000);
164 break;
165 }
166 events.ScheduleEvent(EVENT_CHECK_HEALTH2, 1000);
167 break;
168 case EVENT_POISON_NOVA:
169 Talk(EMOTE_NOVA);
170 me->CastSpell(me, SPELL_POISON_NOVA, false);
171 events.ScheduleEvent(EVENT_POISON_NOVA, 15000);
172 break;
173 case EVENT_POWERFULL_BITE:
174 me->CastSpell(me->GetVictim(), SPELL_POWERFULL_BITE, false);
175 events.ScheduleEvent(EVENT_POWERFULL_BITE, 10000);
176 break;
177 case EVENT_VENOM_BOLT:
178 me->CastSpell(me->GetVictim(), SPELL_VENOM_BOLT, false);
179 events.ScheduleEvent(EVENT_VENOM_BOLT, 10000);
180 break;
181 case EVENT_SUMMON1:
182 for (uint8 i = MAX_CONSTRICTOR; i < MAX_SUMMONS; ++i)
183 me->SummonCreature(NPC_SLADRAN_VIPER, SpawnLoc[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20 * IN_MILLISECONDS);
184 events.ScheduleEvent(EVENT_SUMMON1, 8000);
185 break;
186 case EVENT_SUMMON2:
187 for (uint8 i = 0; i < MAX_CONSTRICTOR; ++i)
188 me->SummonCreature(NPC_SLADRAN_CONSTRICTORS, SpawnLoc[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20 * IN_MILLISECONDS);
189 events.ScheduleEvent(EVENT_SUMMON2, urand(3000, 5000));
190 break;
191 }
192
193 DoMeleeAttackIfReady();
194 }
195
196 private:
197 bool _achievement;
198 };
