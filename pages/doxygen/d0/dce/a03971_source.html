105 void UpdateAI(uint32 diff) override
106 {
107 if (!UpdateVictim())
108 return;
109
110 events.Update(diff);
111 if (me->HasUnitState(UNIT_STATE_CASTING))
112 return;
113
114 switch (events.ExecuteEvent())
115 {
116 case EVENT_SUMMON1:
117 Talk(SAY_SUMMON);
118 me->CastSpell(me, SPELL_SUMMON_FIENDISH_HOUND, false);
119 break;
120 case EVENT_SUMMON2:
121 me->CastSpell(me, SPELL_SUMMON_FIENDISH_HOUND, false);
122 events.ScheduleEvent(EVENT_SUMMON2, 15000);
123 break;
124 case EVENT_TREACHEROUS_AURA:
125 if (roll_chance_i(33))
126 Talk(SAY_CURSE);
127 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
128 me->CastSpell(target, SPELL_TREACHEROUS_AURA, false);
129 events.ScheduleEvent(EVENT_TREACHEROUS_AURA, urand(12000, 18000));
130 break;
131 case EVENT_DEMONIC_SHIELD:
132 if (me->HealthBelowPct(21))
133 {
134 me->CastSpell(me, SPELL_DEMONIC_SHIELD, false);
135 events.ScheduleEvent(EVENT_DEMONIC_SHIELD, 15000);
136 }
137 else
138 events.ScheduleEvent(EVENT_DEMONIC_SHIELD, 1000);
139 break;
140 case EVENT_ORBITAL_STRIKE:
141 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 15.0f, true))
142 {
143 _targetGUID = target->GetGUID();
144 me->CastSpell(target, SPELL_ORBITAL_STRIKE, false);
145 events.DelayEvents(5000);
146 events.ScheduleEvent(EVENT_SHADOW_WHIP, 4000);
147 me->GetMotionMaster()->Clear();
148 }
149 events.ScheduleEvent(EVENT_ORBITAL_STRIKE, 20000);
150 break;
151 case EVENT_SHADOW_WHIP:
152 me->GetMotionMaster()->MoveChase(me->GetVictim());
153 if (Unit* target = ObjectAccessor::GetUnit(*me, _targetGUID))
154 me->CastSpell(target, SPELL_SHADOW_WHIP, false);
155 _targetGUID.Clear();
156 break;
157 }
158
159 if (!me->GetVictim() || !me->isAttackReady())
160 return;
161
162 if (me->IsWithinMeleeRange(me->GetVictim()))
163 {
164 me->GetMotionMaster()->MoveChase(me->GetVictim());
165 DoMeleeAttackIfReady();
166 }
167 else
168 {
169 me->GetMotionMaster()->Clear();
170 me->CastSpell(me->GetVictim(), SPELL_SHADOW_BOLT, false);
171 me->resetAttackTimer();
172 }
173 }
174
175 private:
176 ObjectGuid _targetGUID;
177 };
