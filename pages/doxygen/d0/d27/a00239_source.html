26template <class TO, class FROM> class Reference : public LinkedListElement
27{
28private:
29 TO* iRefTo;
30 FROM* iRefFrom;
31protected:
32 // Tell our refTo (target) object that we have a link
33 virtual void targetObjectBuildLink() = 0;
34
35 // Tell our refTo (taget) object, that the link is cut
36 virtual void targetObjectDestroyLink() = 0;
37
38 // Tell our refFrom (source) object, that the link is cut (Target destroyed)
39 virtual void sourceObjectDestroyLink() = 0;
40public:
41 Reference() { iRefTo = nullptr; iRefFrom = nullptr; }
42 virtual ~Reference() = default;
43
44 // Create new link
45 void link(TO* toObj, FROM* fromObj)
46 {
47 ASSERT(fromObj); // fromObj MUST not be nullptr
48 if (isValid())
49 {
50 unlink();
51 }
52 if (toObj != nullptr)
53 {
54 iRefTo = toObj;
55 iRefFrom = fromObj;
56 targetObjectBuildLink();
57 }
58 }
59
60 // We don't need the reference anymore. Call comes from the refFrom object
61 // Tell our refTo object, that the link is cut
62 void unlink()
63 {
64 targetObjectDestroyLink();
65 delink();
66 iRefTo = nullptr;
67 iRefFrom = nullptr;
68 }
69
70 // Link is invalid due to destruction of referenced target object. Call comes from the refTo object
71 // Tell our refFrom object, that the link is cut
72 void invalidate() // the iRefFrom MUST remain!!
73 {
74 sourceObjectDestroyLink();
75 delink();
76 iRefTo = nullptr;
77 }
78
79 [[nodiscard]] bool isValid() const // Only check the iRefTo
80 {
81 return iRefTo != nullptr;
82 }
83
84 Reference<TO, FROM>* next() { return ((Reference<TO, FROM>*) LinkedListElement::next()); }
85 [[nodiscard]] Reference<TO, FROM> const* next() const { return ((Reference<TO, FROM> const*) LinkedListElement::next()); }
86 Reference<TO, FROM>* prev() { return ((Reference<TO, FROM>*) LinkedListElement::prev()); }
87 [[nodiscard]] Reference<TO, FROM> const* prev() const { return ((Reference<TO, FROM> const*) LinkedListElement::prev()); }
88
89 Reference<TO, FROM>* nocheck_next() { return ((Reference<TO, FROM>*) LinkedListElement::nocheck_next()); }
90 [[nodiscard]] Reference<TO, FROM> const* nocheck_next() const { return ((Reference<TO, FROM> const*) LinkedListElement::nocheck_next()); }
91 Reference<TO, FROM>* nocheck_prev() { return ((Reference<TO, FROM>*) LinkedListElement::nocheck_prev()); }
92 [[nodiscard]] Reference<TO, FROM> const* nocheck_prev() const { return ((Reference<TO, FROM> const*) LinkedListElement::nocheck_prev()); }
93
94 TO* operator ->() const { return iRefTo; }
95 [[nodiscard]] TO* getTarget() const { return iRefTo; }
96
97 [[nodiscard]] FROM* GetSource() const { return iRefFrom; }
98};
