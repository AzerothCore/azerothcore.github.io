338 void AttackStart(Unit* victim) override
339 {
340 if (!victim)
341 return;
342
343 if (me->GetVictim() && me->GetVictim()->GetGUID() == victim->GetGUID() && !me->HasUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_PACIFIED))
344 {
345 if (!me->GetGuidValue(UNIT_FIELD_TARGET))
346 me->SetTarget(victim->GetGUID());
347 }
348 else if (me->Attack(victim, true))
349 {
350 if (!me->HasUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_PACIFIED))
351 me->GetMotionMaster()->MoveChase(victim);
352 else
353 me->SetTarget();
354 }
355 }
