52 npc_pet_dk_ebon_gargoyle() : CreatureScript("npc_pet_dk_ebon_gargoyle") { }

54 struct npc_pet_dk_ebon_gargoyleAI : ScriptedAI
55 {
56 npc_pet_dk_ebon_gargoyleAI(Creature* creature) : ScriptedAI(creature)
57 {
58 _despawnTimer = 36000; // 30 secs + 4 fly out + 2 initial attack timer
59 _despawning = false;
60 _initialSelection = true;
61 _targetGUID.Clear();
62 }
63
64 void MovementInform(uint32 type, uint32 point) override
65 {
66 if (type == POINT_MOTION_TYPE && point == 1)
67 {
68 me->SetCanFly(false);
69 me->SetDisableGravity(false);
70 }
71 }
72
73 void InitializeAI() override
74 {
75 ScriptedAI::InitializeAI();
76 Unit* owner = me->GetOwner();
77 if (!owner)
78 return;
79
80 // Xinef: Night of the Dead avoidance
81 if (Aura* aur = me->GetAura(SPELL_DK_NIGHT_OF_THE_DEAD))
82 if (AuraEffect* aurEff = owner->GetAuraEffect(SPELL_AURA_ADD_FLAT_MODIFIER, SPELLFAMILY_DEATHKNIGHT, 2718, 0))
83 {
84 if (aur->GetEffect(0))
85 {
86 aur->GetEffect(0)->SetAmount(-aurEff->GetSpellInfo()->Effects[EFFECT_2].CalcValue());
87 }
88 }
89
90 me->SetCanFly(true);
91 me->SetDisableGravity(true);
92
93 float tz = me->GetMapHeight(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), true, MAX_FALL_DISTANCE);
94 me->GetMotionMaster()->MoveCharge(me->GetPositionX(), me->GetPositionY(), tz, 7.0f, 1);
95 me->AddUnitState(UNIT_STATE_NO_ENVIRONMENT_UPD);
96 _selectionTimer = 2000;
97 _initialCastTimer = 0;
98 }
99
100 void MySelectNextTarget()
101 {
102 Unit* owner = me->GetOwner();
103 if (owner && owner->GetTypeId() == TYPEID_PLAYER && (!me->GetVictim() || me->GetVictim()->IsImmunedToSpell(sSpellMgr->GetSpellInfo(51963)) || !me->IsValidAttackTarget(me->GetVictim()) || !owner->CanSeeOrDetect(me->GetVictim())))
104 {
105 Unit* selection = owner->ToPlayer()->GetSelectedUnit();
106 if (selection && selection != me->GetVictim() && me->IsValidAttackTarget(selection))
107 {
108 me->GetMotionMaster()->Clear(false);
109 SetGazeOn(selection);
110 }
111
112 else if (!me->GetVictim() || !owner->CanSeeOrDetect(me->GetVictim()))
113 {
114 me->CombatStop(true);
115 me->GetMotionMaster()->Clear(false);
116 me->GetMotionMaster()->MoveFollow(owner, PET_FOLLOW_DIST, 0.0f);
117 RemoveTargetAura();
118 }
119 }
120 }
121
122 void AttackStart(Unit* who) override
123 {
124 RemoveTargetAura();
125 _targetGUID = who->GetGUID();
126 me->AddAura(SPELL_DK_SUMMON_GARGOYLE_1, who);
127 ScriptedAI::AttackStart(who);
128 }
129
130 void RemoveTargetAura()
131 {
132 if (Unit* target = ObjectAccessor::GetUnit(*me, _targetGUID))
133 target->RemoveAura(SPELL_DK_SUMMON_GARGOYLE_1, me->GetGUID());
134 }
135
136 void Reset() override
137 {
138 _selectionTimer = 0;
139 me->SetReactState(REACT_PASSIVE);
140 MySelectNextTarget();
141 }
142
143 // Fly away when dismissed
144 void FlyAway()
145 {
146 RemoveTargetAura();
147
148 // Stop Fighting
149 me->CombatStop(true);
150 me->ApplyModFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE, true);
151
152 // Sanctuary
153 me->CastSpell(me, SPELL_DK_SANCTUARY, true);
154 me->SetReactState(REACT_PASSIVE);
155
156 me->SetSpeed(MOVE_FLIGHT, 1.0f, true);
157 me->SetSpeed(MOVE_RUN, 1.0f, true);
158 float x = me->GetPositionX() + 20 * cos(me->GetOrientation());
159 float y = me->GetPositionY() + 20 * std::sin(me->GetOrientation());
160 float z = me->GetPositionZ() + 40;
161 me->DisableSpline();
162 me->GetMotionMaster()->Clear(false);
163
164 me->GetMotionMaster()->MoveCharge(x, y, z, 7.0f, 1);
165 me->SetCanFly(true);
166 me->SetDisableGravity(true);
167
168 _despawning = true;
169 }
170
171 void UpdateAI(uint32 diff) override
172 {
173 if (_initialSelection)
174 {
175 _initialSelection = false;
176 // Find victim of Summon Gargoyle spell
177 std::list<Unit*> targets;
178 Acore::AnyUnfriendlyUnitInObjectRangeCheck u_check(me, me, 50.0f);
179 Acore::UnitListSearcher<Acore::AnyUnfriendlyUnitInObjectRangeCheck> searcher(me, targets, u_check);
180 Cell::VisitAllObjects(me, searcher, 50.0f);
181 for (std::list<Unit*>::const_iterator iter = targets.begin(); iter != targets.end(); ++iter)
182 if ((*iter)->GetAura(SPELL_DK_SUMMON_GARGOYLE_1, me->GetOwnerGUID()))
183 {
184 (*iter)->RemoveAura(SPELL_DK_SUMMON_GARGOYLE_1, me->GetOwnerGUID());
185 SetGazeOn(*iter);
186 _targetGUID = (*iter)->GetGUID();
187 break;
188 }
189 }
190 if (_despawnTimer > 4000)
191 {
192 _despawnTimer -= diff;
193 if (!UpdateVictimWithGaze())
194 {
195 MySelectNextTarget();
196 return;
197 }
198
199 _initialCastTimer += diff;
200 _selectionTimer += diff;
201 if (_selectionTimer >= 1000)
202 {
203 MySelectNextTarget();
204 _selectionTimer = 0;
205 }
206 if (_initialCastTimer >= 2000 && !me->HasUnitState(UNIT_STATE_CASTING | UNIT_STATE_LOST_CONTROL) && me->GetMotionMaster()->GetMotionSlotType(MOTION_SLOT_CONTROLLED) == NULL_MOTION_TYPE)
207 me->CastSpell(me->GetVictim(), 51963, false);
208 }
209 else
210 {
211 if (!_despawning)
212 FlyAway();
213
214 if (_despawnTimer > diff)
215 _despawnTimer -= diff;
216 else
217 me->DespawnOrUnsummon();
218 }
219 }
220
221 private:
222 ObjectGuid _targetGUID;
223 uint32 _despawnTimer;
224 uint32 _selectionTimer;
225 uint32 _initialCastTimer;
226 bool _despawning;
227 bool _initialSelection;
228 };

230 CreatureAI* GetAI(Creature* creature) const override
231 {
232 return new npc_pet_dk_ebon_gargoyleAI(creature);
233 }
