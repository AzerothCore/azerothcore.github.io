61struct npc_ouro_spawner : public ScriptedAI
62{
63 npc_ouro_spawner(Creature* creature) : ScriptedAI(creature)
64 {
65 Reset();
66 }
67
68 bool hasSummoned;
69
70 void Reset() override
71 {
72 hasSummoned = false;
73 DoCastSelf(SPELL_DIRTMOUND_PASSIVE);
74 }
75
76 void MoveInLineOfSight(Unit* who) override
77 {
78 // Spawn Ouro on LoS check
79 if (!hasSummoned && who->GetTypeId() == TYPEID_PLAYER && me->IsWithinDistInMap(who, 40.0f) && !who->ToPlayer()->IsGameMaster())
80 {
81 if (InstanceScript* instance = me->GetInstanceScript())
82 {
83 Creature* ouro = instance->GetCreature(DATA_OURO);
84 if (instance->GetBossState(DATA_OURO) != IN_PROGRESS && !ouro)
85 {
86 DoCastSelf(SPELL_SUMMON_OURO);
87 hasSummoned = true;
88 }
89 }
90 }
91
92 ScriptedAI::MoveInLineOfSight(who);
93 }
94
95 void JustSummoned(Creature* creature) override
96 {
97 // Despawn when Ouro is spawned
98 if (creature->GetEntry() == NPC_OURO)
99 {
100 creature->SetInCombatWithZone();
101 me->DespawnOrUnsummon();
102 }
103 }

106struct boss_ouro : public BossAI
107{
108 boss_ouro(Creature* creature) : BossAI(creature, DATA_OURO)
109 {
110 SetCombatMovement(false);
111 me->SetControlled(true, UNIT_STATE_ROOT);
112 _scheduler.SetValidator([this] { return !me->HasUnitState(UNIT_STATE_CASTING); });
113 }
114
115 bool CanAIAttack(Unit const* victim) const override
116 {
117 return me->IsWithinMeleeRange(victim);
118 }
119
120 void DamageTaken(Unit* /*attacker*/, uint32& damage, DamageEffectType, SpellSchoolMask) override
121 {
122 if (me->HealthBelowPctDamaged(20, damage) && !_enraged)
123 {
124 DoCastSelf(SPELL_BERSERK, true);
125 _enraged = true;
126 _scheduler.CancelGroup(GROUP_PHASE_TRANSITION);
127 _scheduler.Schedule(1s, [this](TaskContext context)
128 {
129 if (!IsPlayerWithinMeleeRange())
130 DoSpellAttackToRandomTargetIfReady(SPELL_BOULDER);
131
132 context.Repeat();
133 })
134 .Schedule(20s, [this](TaskContext context)
135 {
136 DoCastSelf(SPELL_SUMMON_OURO_MOUNDS, true);
137 context.Repeat();
138 });
139 }
140 }
141
142 void Submerge()
143 {
144 if (_enraged || _submerged)
145 return;
146
147 me->AttackStop();
148 me->SetReactState(REACT_PASSIVE);
149 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
150 _submergeMelee = 0;
151 _submerged = true;
152 DoCastSelf(SPELL_OURO_SUBMERGE_VISUAL);
153 _scheduler.CancelGroup(GROUP_EMERGED);
154 _scheduler.CancelGroup(GROUP_PHASE_TRANSITION);
155
156 if (GameObject* base = me->FindNearestGameObject(GO_SANDWORM_BASE, 10.f))
157 {
158 base->Use(me);
159 base->DespawnOrUnsummon(6s);
160 }
161
162 DoCastSelf(SPELL_SUMMON_OURO_MOUNDS, true);
163 // According to sniffs, Ouro uses his mounds to respawn. The health management could be a little scuffed.
164 std::list<Creature*> ouroMounds;
165 me->GetCreatureListWithEntryInGrid(ouroMounds, NPC_DIRT_MOUND, 200.f);
166 if (!ouroMounds.empty()) // This can't be possible, but just to be sure.
167 {
168 if (Creature* mound = Acore::Containers::SelectRandomContainerElement(ouroMounds))
169 {
170 mound->AddAura(SPELL_SUMMON_OURO_AURA, mound);
171 mound->AI()->SetData(DATA_OURO_HEALTH, me->GetHealth());
172 }
173 }
174
175 me->DespawnOrUnsummon(1000);
176 }
177
178 void CastGroundRupture()
179 {
180 std::list<WorldObject*> targets;
181 Acore::AllWorldObjectsInRange checker(me, 10.0f);
182 Acore::WorldObjectListSearcher<Acore::AllWorldObjectsInRange> searcher(me, targets, checker);
183 Cell::VisitAllObjects(me, searcher, 10.0f);
184
185 for (WorldObject* target : targets)
186 {
187 if (Unit* unitTarget = target->ToUnit())
188 {
189 if (unitTarget->IsHostileTo(me))
190 DoCast(unitTarget, SPELL_GROUND_RUPTURE, true);
191 }
192 }
193 }
194
195 void SpellHitTarget(Unit* target, SpellInfo const* spellInfo) override
196 {
197 if (spellInfo->Id == SPELL_SAND_BLAST && target)
198 {
199 me->GetThreatMgr().ModifyThreatByPercent(target, -100);
200 }
201 }
202
203 void Emerge()
204 {
205 DoCastSelf(SPELL_BIRTH);
206 DoCastSelf(SPELL_SUMMON_SANDWORM_BASE, true);
207 me->SetReactState(REACT_AGGRESSIVE);
208 CastGroundRupture();
209 _scheduler
210 .Schedule(20s, GROUP_EMERGED, [this](TaskContext context)
211 {
212 if (Unit* target = SelectTarget(SelectTargetMethod::MaxThreat, 0, 0.0f, true))
213 {
214 me->SetTarget(target->GetGUID());
215 }
216
217 DoCastAOE(SPELL_SAND_BLAST);
218
219 me->m_Events.AddEventAtOffset([this]()
220 {
221 if (Unit* victim = me->GetVictim())
222 {
223 me->SetTarget(victim->GetGUID());
224 }
225 }, 3s);
226
227 context.Repeat();
228 })
229 .Schedule(22s, GROUP_EMERGED, [this](TaskContext context)
230 {
231 DoCastVictim(SPELL_SWEEP);
232 context.Repeat();
233 })
234 .Schedule(90s, GROUP_PHASE_TRANSITION, [this](TaskContext /*context*/)
235 {
236 Submerge();
237 })
238 .Schedule(3s, GROUP_PHASE_TRANSITION, [this](TaskContext context)
239 {
240 if (_enraged)
241 return;
242
243 if (!IsPlayerWithinMeleeRange() && !_submerged)
244 {
245 if (_submergeMelee < 10)
246 {
247 _submergeMelee++;
248 }
249 else
250 {
251 Submerge();
252 _submergeMelee = 0;
253 }
254 }
255 else
256 {
257 _submergeMelee = 0;
258 }
259
260 if (!_submerged)
261 context.Repeat(1s);
262 });
263 }
264
265 void Reset() override
266 {
267 instance->SetBossState(DATA_OURO, NOT_STARTED);
268 _scheduler.CancelAll();
269 _submergeMelee = 0;
270 _submerged = false;
271 _enraged = false;
272 }
273
274 void EnterEvadeMode(EvadeReason /*why*/) override
275 {
276 if (me->GetThreatMgr().GetThreatList().empty())
277 {
278 DoCastSelf(SPELL_OURO_SUBMERGE_VISUAL);
279 me->DespawnOrUnsummon(1000);
280 instance->SetBossState(DATA_OURO, FAIL);
281 if (GameObject* base = me->FindNearestGameObject(GO_SANDWORM_BASE, 200.f))
282 base->DespawnOrUnsummon();
283 }
284 }
285
286 void EnterCombat(Unit* who) override
287 {
288 Emerge();
289
290 BossAI::EnterCombat(who);
291 }
292
293 void UpdateAI(uint32 diff) override
294 {
295 UpdateVictim();
296
297 _scheduler.Update(diff, [this]
298 {
299 DoMeleeAttackIfReady();
300 });
301 }
302
303protected:
304 TaskScheduler _scheduler;
305 bool _enraged;
306 uint8 _submergeMelee;
307 bool _submerged;
308
309 bool IsPlayerWithinMeleeRange() const
310 {
311 return me->IsWithinMeleeRange(me->GetVictim());
312 }

339 {

343 DoCastSelf(SPELL_SUMMON_SCARABS, true);

347 {
348 ChaseNewTarget();

354 {
355 DoResetThreat();

357 {
358 me->AddThreat(target, 1000000.f);
359 AttackStart(target);
360 }
361 }
362
363 void UpdateAI(uint32 diff) override
364 {
365 if (!UpdateVictim())
366 return;
367
368 _scheduler.Update(diff);
369 }
370
371 void Reset() override
372 {
373 DoCastSelf(SPELL_DIRTMOUND_PASSIVE, true);
374 DoCastSelf(SPELL_DREAM_FOG, true);
375 DoCastSelf(SPELL_QUAKE, true);
376 }
377
378 void EnterEvadeMode(EvadeReason /*why*/) override
379 {
380 if (_instance)
381 {
382 _instance->SetBossState(DATA_OURO, FAIL);
383 }
384
385 if (GameObject* base = me->FindNearestGameObject(GO_SANDWORM_BASE, 200.f))
386 base->DespawnOrUnsummon();
387
388 me->DespawnOrUnsummon();
389 }
390
391protected:
392 TaskScheduler _scheduler;
393 uint32 _ouroHealth;
394 InstanceScript* _instance;
