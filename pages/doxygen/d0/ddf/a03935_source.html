109 void UpdateAI(uint32 diff) override
110 {
111 if (!UpdateVictim())
112 return;
113
114 events.Update(diff);
115 switch (events.ExecuteEvent())
116 {
117 case EVENT_CHECK:
118 float x, y, z, o;
119 me->GetHomePosition(x, y, z, o);
120 if (!me->IsWithinDist3d(x, y, z, 60))
121 {
122 EnterEvadeMode();
123 return;
124 }
125 events.RepeatEvent(5000);
126 break;
127 case EVENT_SPELL_SPORES:
128 me->CastSpell(me, SPELL_SUMMON_SPORE_STRIDER, false);
129 events.RepeatEvent(urand(10000, 15000));
130 break;
131 case EVENT_SPELL_CHAIN:
132 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
133 me->CastSpell(target, SPELL_CHAIN_LIGHTNING, false);
134 events.DelayEvents(3000);
135 events.RepeatEvent(9000);
136 break;
137 case EVENT_SPELL_STATIC:
138 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 30, true))
139 me->CastSpell(target, SPELL_STATIC_CHARGE, false);
140 events.RepeatEvent(10000);
141 break;
142 case EVENT_LEVITATE:
143 events.RepeatEvent(15000);
144 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1))
145 {
146 me->CastSpell(target, SPELL_LEVITATE, false);
147 lTarget = target->GetGUID();
148 events.DelayEvents(5000);
149 events.ScheduleEvent(EVENT_LEVITATE_TARGET_1, 2000);
150 }
151 break;
152 case EVENT_LEVITATE_TARGET_1:
153 if (Unit* target = ObjectAccessor::GetUnit(*me, lTarget))
154 {
155 if (!target->HasAura(SPELL_LEVITATE))
156 lTarget.Clear();
157 else
158 {
159 target->CastSpell(target, SPELL_MAGNETIC_PULL, true);
160 events.ScheduleEvent(EVENT_LEVITATE_TARGET_2, 1500);
161 }
162 }
163 break;
164 case EVENT_LEVITATE_TARGET_2:
165 if (Unit* target = ObjectAccessor::GetUnit(*me, lTarget))
166 {
167 if (!target->HasAura(SPELL_LEVITATE))
168 lTarget.Clear();
169 else
170 {
171 target->AddAura(SPELL_SUSPENSION, target);
172 lTarget.Clear();
173 }
174 }
175 break;
176 }
177
178 DoMeleeAttackIfReady();
179 }
180 };
