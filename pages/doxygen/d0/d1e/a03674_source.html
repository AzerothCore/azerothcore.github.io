72 struct boss_emalonAI : public ScriptedAI
73 {
74 boss_emalonAI(Creature* creature) : ScriptedAI(creature), summons(me)
75 {
76 pInstance = me->GetInstanceScript();
77 }
78
79 InstanceScript* pInstance;
80 EventMap events;
81 SummonList summons;
82
83 void ResetSummons()
84 {
85 summons.DespawnAll();
86 for (uint8 i = 0; i < MAX_TEMPEST_MINIONS; ++i)
87 me->SummonCreature(NPC_TEMPEST_MINION, TempestMinions[i], TEMPSUMMON_CORPSE_DESPAWN, 0);
88 }
89
90 void Reset() override
91 {
92 events.Reset();
93 ResetSummons();
94
95 if (pInstance)
96 {
97 if (pInstance->GetData(DATA_STONED))
98 {
99 if (Aura* aur = me->AddAura(SPELL_STONED_AURA, me))
100 {
101 aur->SetMaxDuration(60 * MINUTE * IN_MILLISECONDS);
102 aur->SetDuration(60 * MINUTE * IN_MILLISECONDS);
103 }
104 }
105 pInstance->SetData(EVENT_EMALON, NOT_STARTED);
106 }
107 }
108
109 void AttackStart(Unit* who) override
110 {
111 if (me->HasAura(SPELL_STONED_AURA))
112 return;
113
114 ScriptedAI::AttackStart(who);
115 }
116
117 void JustSummoned(Creature* summon) override
118 {
119 summons.Summon(summon);
120 }
121
122 void SummonedCreatureDies(Creature* cr, Unit*) override
123 {
124 summons.Despawn(cr);
125 events.ScheduleEvent(EVENT_SUMMON_NEXT_MINION, 4000);
126 }
127
128 void SpellHitTarget(Unit* target, SpellInfo const* spellInfo) override
129 {
130 // restore minions health
131 if (spellInfo->Id == SPELL_OVERCHARGE)
132 target->SetFullHealth();
133 }
134
135 void EnterCombat(Unit* /*who*/) override
136 {
137 events.Reset();
138 if (summons.size() < 4)
139 ResetSummons();
140
141 summons.DoZoneInCombat();
142
143 events.ScheduleEvent(EVENT_CHAIN_LIGHTNING, 5000);
144 events.ScheduleEvent(EVENT_LIGHTNING_NOVA, 40000);
145 events.ScheduleEvent(EVENT_BERSERK, 360000);
146 events.ScheduleEvent(EVENT_OVERCHARGE, 47000);
147
148 if (pInstance)
149 pInstance->SetData(EVENT_EMALON, IN_PROGRESS);
150 }
151
152 void JustDied(Unit* /*killer*/) override
153 {
154 summons.DespawnAll();
155 events.Reset();
156 if (pInstance)
157 pInstance->SetData(EVENT_EMALON, DONE);
158 }
159
160 void UpdateAI(uint32 diff) override
161 {
162 if (!UpdateVictim())
163 return;
164
165 events.Update(diff);
166 if (me->HasUnitState(UNIT_STATE_CASTING))
167 return;
168
169 switch (events.ExecuteEvent())
170 {
171 case EVENT_CHAIN_LIGHTNING:
172 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
173 me->CastSpell(target, RAID_MODE(SPELL_CHAIN_LIGHTNING_10, SPELL_CHAIN_LIGHTNING_25), false);
174 events.RepeatEvent(25000);
175 break;
176 case EVENT_LIGHTNING_NOVA:
177 me->CastSpell(me, RAID_MODE(SPELL_LIGHTNING_NOVA_10, SPELL_LIGHTNING_NOVA_25), false);
178 events.RepeatEvent(40000);
179 break;
180 case EVENT_OVERCHARGE:
181 if (!summons.empty())
182 me->CastCustomSpell(SPELL_OVERCHARGE, SPELLVALUE_MAX_TARGETS, 1, me, true);
183 Talk(EMOTE_OVERCHARGE);
184 events.RepeatEvent(40000);
185 break;
186 case EVENT_BERSERK:
187 me->CastSpell(me, SPELL_BERSERK, true);
188 Talk(EMOTE_BERSERK);
189 break;
190 case EVENT_SUMMON_NEXT_MINION:
191 me->SummonCreature(NPC_TEMPEST_MINION, TempestMinions[urand(0, 3)], TEMPSUMMON_CORPSE_DESPAWN, 0);
192 break;
193 default:
194 break;
195 }
196
197 DoMeleeAttackIfReady();
198 }
199 };
200
201 CreatureAI* GetAI(Creature* creature) const override
202 {
203 return GetVaultOfArchavonAI<boss_emalonAI>(creature);
204 }
205};
