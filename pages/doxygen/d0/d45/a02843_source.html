73 void UpdateAI(uint32 diff) override
74 {
75 if (!UpdateVictim())
76 return;
77
78 events.Update(diff);
79
80 if (me->HasUnitState(UNIT_STATE_CASTING))
81 return;
82
83 while (uint32 eventId = events.ExecuteEvent())
84 {
85 switch (eventId)
86 {
87 case EVENT_AVATAR:
88 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
89 {
90 _pursuitTargetGUID = target->GetGUID();
91 }
92 DoCast(me, SPELL_AVATAR);
93 me->SetReactState(REACT_PASSIVE);
94 DoResetThreat();
95 events.ScheduleEvent(EVENT_START_PURSUIT, 2s);
96 events.ScheduleEvent(EVENT_STOP_PURSUIT, 15s);
97 events.ScheduleEvent(EVENT_AVATAR, 45s, 50s);
98 break;
99 case EVENT_GROUND_TREMOR:
100 DoCastVictim(SPELL_GROUND_TREMOR, true);
101 events.ScheduleEvent(EVENT_GROUND_TREMOR, 12s, 16s);
102 break;
103 case EVENT_START_PURSUIT:
104 me->SetReactState(REACT_AGGRESSIVE);
105 if (Unit* pursuitTarget = ObjectAccessor::GetUnit(*me, _pursuitTargetGUID))
106 {
107 me->GetThreatMgr().AddThreat(pursuitTarget, 1000000.f);
108 }
109 break;
110 case EVENT_STOP_PURSUIT:
111 if (Unit* pursuitTarget = ObjectAccessor::GetUnit(*me, _pursuitTargetGUID))
112 {
113 _pursuitTargetGUID.Clear();
114 me->GetThreatMgr().AddThreat(pursuitTarget, -1000000.f);
115 }
116 break;
117 case EVENT_ENTANGLING_ROOTS:
118 DoCastVictim(SPELL_ENTANGLING_ROOTS);
119 events.ScheduleEvent(EVENT_ENTANGLING_ROOTS, 10s, 20s);
120 break;
121 case EVENT_SWEEPING_STRIKES:
122 DoCastSelf(SPELL_SWEEPING_STRIKES, true);
123 events.ScheduleEvent(EVENT_SWEEPING_STRIKES, 30s);
124 break;
125 default:
126 break;
127 }
128 }
129
130 DoMeleeAttackIfReady();
131 }
132
133 private:
134 ObjectGuid _pursuitTargetGUID;
135 };
