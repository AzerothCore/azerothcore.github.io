137 void SummonedCreatureDespawn(Creature* cr) override
138 {
139 int room = -1;
140 // remove the mob from the list and keep track of which room he is in.
141 for (uint8 i = 0; i < 6; i++)
142 {
143 for (uint8 j = 0; j < 3; j++)
144 {
145 if (Guardians[i][j] && Guardians[i][j]->GetGUID() == cr->GetGUID())
146 {
147 room = i;
148 Guardians[i][j] = nullptr;
149 }
150 }
151 }
152
153 // check if the room is now empty
154 if (room >= 0)
155 {
156 for (uint8 i = 0; i < 3; i++)
157 {
158 if (Guardians[room][i])
159 {
160 return;
161 }
162 }
163 // everybody is dead in there, we can open the gate.
164 SetGate(room, OPEN);
165 }
166 }
167
168 // add mob to the room of the last portal, that's the only place we can have added mobs.
169 void JustSummoned(Creature* cr) override
170 {
171 summons.Summon(cr);
172 uint32 room = GetData(GANDLING_ROOM_TO_USE);
173 if (room < 6)
174 {
175 for (uint8 i = 0; i < 3; i++)
176 {
177 if (!Guardians[room][i])
178 {
179 Guardians[room][i] = cr;
180 break;
181 }
182 }
183 }
184 }
185
186 void EnterCombat(Unit* /*who*/) override
187 {
188 DoZoneInCombat();
189 instance->SetData(DATA_DARKMASTER_GANDLING, IN_PROGRESS);
190 events.Reset();
191 events.ScheduleEvent(SPELL_ARCANE_MISSILES, TIMER_ARCANE_MIN);
192 events.ScheduleEvent(SPELL_CURSE_DARKMASTER, TIMER_CURSE_MIN);
193 events.ScheduleEvent(SPELL_SHADOW_SHIELD, TIMER_SHIELD_MIN);
194 events.ScheduleEvent(SPELL_SHADOW_PORTAL, TIMER_PORTAL);
195 }

203 void SetData(uint32 type, uint32 data) override
204 {
205 switch (type)
206 {

208 if (data < 6)

239 void Reset() override
240 {
241 _Reset();
242 instance->SetData(DATA_DARKMASTER_GANDLING, NOT_STARTED);
243 if (instance->GetData(DATA_MINI_BOSSES) != 6)
244 {
245 me->SetVisible(false);
246 me->SetFaction(35);

253 OpenAllGates();
254 summons.DespawnAll();
255 for (int i = 0; i < 6; i++)
256 {
257 for (int j =0; j < 3; j++)
258 {
259 Guardians[i][j] = nullptr;
260 }
261 }
262 }

264 // Finds a random room that is not in use
265 uint32 FindRoom()
266 {
267 uint32 attempts = 0;
268 uint32 room = urand(0, 5);
269 do
270 {
271 room = (room + 1) % 6;
272 attempts++;
273 } while ((Guardians[room][0] || Guardians[room][1] || Guardians[room][2]) && attempts < 7);
274
275 if (attempts == 7)
276 {
277 room = 7; // used as error
278 }
279
280 return room;
281 }
282
283 // spawns the 3 mobs in a given room.
284 void SpawnMobsInRoom(uint32 room)
285 {
286 for (uint8 i = 0; i < 3; ++i)
287 {
288 if (Creature* summon = me->SummonCreature(NPC_RISEN_GUARDIAN, SummonPos[room*3 + i], TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 120000))
289 {
290 summon->GetMotionMaster()->MoveRandom(8.0f);
291 }
292 }
293 }
294
295 // used for shadow portal
296 void SpellHitTarget(Unit* target, SpellInfo const* spellinfo) override
297 {
298 uint32 room = 0;
299 if (spellinfo && spellinfo->Id == SPELL_SHADOW_PORTAL && target)
300 {
301 room = GetData(GANDLING_ROOM_TO_USE);
302 SetGate(room, CLOSED);
303 SpawnMobsInRoom(room);
304 DoCast(target, GandlingPortalSpells[room], true); // needs triggered somehow.
305
306 auto victim = me->GetVictim();
307 if (victim && (target->GetGUID() == victim->GetGUID()))
308 {
309 me->AddThreat(victim, -1000000); // drop current player, add a ton to second. This should guarantee that we don't end up with both 1 and 2 in a cage...
310 if (Unit* newTarget = SelectTarget(SelectTargetMethod::MaxThreat, 1, 200.0f)) // search in whole room
311 {
312 me->AddThreat(newTarget, 1000000);
313 }
314 }
315 }
316 }
317
318 void UpdateAI(uint32 diff) override
319 {
320 uint32 room = 0;
321 if (!UpdateVictim())
322 {
323 return;
324 }
325
326 events.Update(diff);
327 if (me->HasUnitState(UNIT_STATE_CASTING))
328 {
329 return;
330 }
331
332 while (uint32 eventId = events.ExecuteEvent())
333 {
334 switch (eventId)
335 {
336 case SPELL_ARCANE_MISSILES:
337 DoCastVictim(SPELL_ARCANE_MISSILES);
338 events.ScheduleEvent(SPELL_ARCANE_MISSILES, urand(TIMER_ARCANE_MIN, TIMER_ARCANE_MAX));
339 break;
340 case SPELL_CURSE_DARKMASTER:
341 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true))
342 {
343 DoCast(target, SPELL_CURSE_DARKMASTER);
344 }
345 events.ScheduleEvent(SPELL_ARCANE_MISSILES, urand(TIMER_ARCANE_MIN, TIMER_ARCANE_MAX));
346 break;
347 case SPELL_SHADOW_SHIELD:
348 DoCastSelf(SPELL_SHADOW_SHIELD);
349 events.ScheduleEvent(SPELL_ARCANE_MISSILES, urand(TIMER_ARCANE_MIN, TIMER_ARCANE_MAX));
350 break;
351
352 case SPELL_SHADOW_PORTAL:
353 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 15.0, true))
354 {
355 room = FindRoom();
356 if (room < 6)
357 {
358 SetData(GANDLING_ROOM_TO_USE, room);
359 DoCast(target, SPELL_SHADOW_PORTAL); // don't use triggered here
360 }
361 }
362 events.ScheduleEvent(SPELL_SHADOW_PORTAL, TIMER_PORTAL);
363 break;
364 default:
365 break;
366 }
367 }
368 DoMeleeAttackIfReady();
369 }
370 };
