69 if (creature->IsQuestGiver())

71
72 uint32 creatureId = creature->GetEntry();
73
74 if (creatureId == NPC_ARATHANDRIS_SILVERSKY)
75 {
76 if (player->GetQuestRewardStatus(QUEST_CLEASING_FELWOOD_A))
77 {
78 AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_ITEM_BEACON, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
79 SendGossipMenuFor(player, 2848, creature->GetGUID());
80 }
81 else if (player->GetTeamId() == TEAM_HORDE)
82 SendGossipMenuFor(player, 2845, creature->GetGUID());
83 else
84 SendGossipMenuFor(player, 2844, creature->GetGUID());
85 }
86
87 if (creatureId == NPC_MAYBESS_RIVERBREEZE)
88 {
89 if (player->GetQuestRewardStatus(QUEST_CLEASING_FELWOOD_H))
90 {
91 AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_ITEM_BEACON, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
92 SendGossipMenuFor(player, 2849, creature->GetGUID());

95 SendGossipMenuFor(player, 2843, creature->GetGUID());

119public:

121
122 bool OnTrigger(Player* player, AreaTrigger const* /*trigger*/) override
123 {
124 if (player->IsGameMaster() || !player->IsAlive())
125 return false;
126
127 // Handle Call Ancients event start - The area trigger summons 3 ancients
128 if ((player->GetQuestStatus(QUEST_ANCIENT_LEAF) == QUEST_STATUS_COMPLETE) || (player->GetQuestStatus(QUEST_ANCIENT_LEAF) == QUEST_STATUS_REWARDED))
129 {
130 // If ancients are already spawned, skip the rest
131 if (GetClosestCreatureWithEntry(player, NPC_VARTRUS, 50.0f) || GetClosestCreatureWithEntry(player, NPC_STOMA, 50.0f) || GetClosestCreatureWithEntry(player, NPC_HASTAT, 50.0f))
132 return true;
133
134 player->GetMap()->SummonCreatureGroup(CREATURE_GROUP_ANCIENTS);
135 }
136 return false;
137 }
138};
