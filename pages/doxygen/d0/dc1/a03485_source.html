26const float HeiganPos[2] = {2796, -3707};
27const float HeiganEruptionSlope[3] =
28{
29 (-3685 - HeiganPos[1]) / (2724 - HeiganPos[0]),
30 (-3647 - HeiganPos[1]) / (2749 - HeiganPos[0]),
31 (-3637 - HeiganPos[1]) / (2771 - HeiganPos[0]),
32};
33
34inline uint8 GetEruptionSection(float x, float y)
35{
36 y -= HeiganPos[1];
37 if (y < 1.0f)
38 return 0;
39
40 x -= HeiganPos[0];
41 if (x > -1.0f)
42 return 3;
43
44 float slope = y / x;
45 for (uint32 i = 0; i < 3; ++i)
46 {
47 if (slope > HeiganEruptionSlope[i])
48 return i;
49 }
50 return 3;
51}
