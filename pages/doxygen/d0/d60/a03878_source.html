212 bool SetBossState(uint32 type, EncounterState state) override
213 {
214 if (!InstanceScript::SetBossState(type, state))
215 return false;
216
217 if (type == DATA_SHADE_OF_AKAMA && state == DONE)
218 {
219 for (ObjectGuid const& guid : ashtongueGUIDs)
220 if (Creature* ashtongue = instance->GetCreature(guid))
221 ashtongue->SetFaction(FACTION_ASHTONGUE_DEATHSWORN);
222 }
223 else if (type == DATA_ILLIDARI_COUNCIL && state == DONE)
224 {
225 if (Creature* akama = instance->GetCreature(AkamaGUID))
226 akama->SetVisible(true);
227 }
228
229 return true;
230 }

232 std::string GetSaveData() override
233 {
234 OUT_SAVE_INST_DATA;
235
236 std::ostringstream saveStream;
237 saveStream << "B T " << GetBossSaveData();
238
239 OUT_SAVE_INST_DATA_COMPLETE;
240 return saveStream.str();

242
243 void Load(char const* str) override
244 {
245 if (!str)
246 {
247 OUT_LOAD_INST_DATA_FAIL;
248 return;
249 }
250
251 OUT_LOAD_INST_DATA(str);
252
253 char dataHead1, dataHead2;
254
255 std::istringstream loadStream(str);
256 loadStream >> dataHead1 >> dataHead2;
257
258 if (dataHead1 == 'B' && dataHead2 == 'T')
259 {
260 for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
261 {
262 uint32 tmpState;
263 loadStream >> tmpState;
264 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
265 tmpState = NOT_STARTED;
266 SetBossState(i, EncounterState(tmpState));
267 }
268 }
269 else
270 OUT_LOAD_INST_DATA_FAIL;
271
272 OUT_LOAD_INST_DATA_COMPLETE;
273 }
