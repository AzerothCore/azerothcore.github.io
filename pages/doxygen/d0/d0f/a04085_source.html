105 void UpdateAI(uint32 diff) override
106 {
107 if (!UpdateVictim())
108 return;
109
110 events.Update(diff);
111 if (me->HasUnitState(UNIT_STATE_CASTING))
112 return;
113
114 switch (events.ExecuteEvent())
115 {
116 case EVENT_ARCANE_EXPLOSION:
117 me->CastSpell(me, SPELL_ARCANE_EXPLOSION, false);
118 events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, 12000);
119 break;
120 case EVENT_ARCANE_TORRENT:
121 me->RemoveAurasDueToSpell(SPELL_MANA_TAP);
122 me->ModifyPower(POWER_MANA, 5000);
123 me->CastSpell(me, SPELL_ARCANE_TORRENT, false);
124 events.ScheduleEvent(EVENT_ARCANE_TORRENT, 15000);
125 break;
126 case EVENT_MANA_TAP:
127 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, PowerUsersSelector(me, POWER_MANA, 40.0f, false)))
128 me->CastSpell(target, SPELL_MANA_TAP, false);
129 events.ScheduleEvent(EVENT_MANA_TAP, 18000);
130 break;
131 case EVENT_DOMINATION:
132 Talk(SAY_DOMINATION);
133 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 50.0f))
134 me->CastSpell(target, SPELL_DOMINATION, false);
135 events.ScheduleEvent(EVENT_DOMINATION, 30000);
136 break;
137 case EVENT_FRENZY:
138 if (me->HealthBelowPct(20))
139 {
140 summons.DespawnAll();
141 me->CastSpell(me, SPELL_DISGRUNTLED_ANGER, true);
142 Talk(SAY_ENRAGE);
143 break;
144 }
145 events.ScheduleEvent(EVENT_FRENZY, 1000);
146 break;
147 case EVENT_SUMMON:
148 for (uint8 i = 0; i < DUNGEON_MODE(3, 4); ++i)
149 me->CastSpell(me, SPELL_SUMMON_NETHER_WRAITH_1 + i, true);
150
151 Talk(SAY_SUMMON);
152 events.ScheduleEvent(EVENT_SUMMON, urand(30000, 45000));
153 break;
154 }
155
156 DoMeleeAttackIfReady();
157 }
158 };
