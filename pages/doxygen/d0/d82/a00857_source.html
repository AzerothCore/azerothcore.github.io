75void Arena::AddPlayer(Player* player)
76{
77 Battleground::AddPlayer(player);
78 PlayerScores.emplace(player->GetGUID().GetCounter(), new ArenaScore(player->GetGUID(), player->GetBgTeamId()));
79
80 if (player->GetBgTeamId() == TEAM_ALLIANCE) // gold
81 {
82 if (player->GetTeamId() == TEAM_HORDE)
83 player->CastSpell(player, SPELL_HORDE_GOLD_FLAG, true);
84 else
85 player->CastSpell(player, SPELL_ALLIANCE_GOLD_FLAG, true);
86 }
87 else // green
88 {
89 if (player->GetTeamId() == TEAM_HORDE)
90 player->CastSpell(player, SPELL_HORDE_GREEN_FLAG, true);
91 else
92 player->CastSpell(player, SPELL_ALLIANCE_GREEN_FLAG, true);
93 }
94
95 UpdateArenaWorldState();
96}

130void Arena::RemovePlayerAtLeave(Player* player)
131{
132 if (isRated() && GetStatus() == STATUS_IN_PROGRESS)
133 {
134 if (auto const& member = Acore::Containers::MapGetValuePtr(m_Players, player->GetGUID()))
135 {
136 // if the player was a match participant, calculate rating
137 auto teamId = member->GetBgTeamId();
138
139 ArenaTeam* winnerArenaTeam = sArenaTeamMgr->GetArenaTeamById(GetArenaTeamIdForTeam(GetOtherTeamId(teamId)));
140 ArenaTeam* loserArenaTeam = sArenaTeamMgr->GetArenaTeamById(GetArenaTeamIdForTeam(teamId));
141
142 if (winnerArenaTeam && loserArenaTeam && winnerArenaTeam != loserArenaTeam)
143 loserArenaTeam->MemberLost(player, GetArenaMatchmakerRating(GetOtherTeamId(teamId)));
144 }
145 }
146
147 // remove player
148 Battleground::RemovePlayerAtLeave(player);
149}

151void Arena::CheckWinConditions()
152{
153 if (!GetAlivePlayersCountByTeam(TEAM_ALLIANCE) && GetPlayersCountByTeam(TEAM_HORDE))
154 EndBattleground(TEAM_HORDE);
155 else if (GetPlayersCountByTeam(TEAM_ALLIANCE) && !GetAlivePlayersCountByTeam(TEAM_HORDE))
156 EndBattleground(TEAM_ALLIANCE);
157}

159void Arena::EndBattleground(TeamId winnerTeamId)
160{
161 if (isRated())
162 {
163 uint32 startDelay = GetStartDelayTime();
164 bool bValidArena = GetStatus() == STATUS_IN_PROGRESS && GetStartTime() >= startDelay + 15000; // pussywizard: only if arena lasted at least 15 secs
165
166 uint32 loserTeamRating = 0;
167 uint32 loserMatchmakerRating = 0;
168 int32 loserChange = 0;
169 int32 loserMatchmakerChange = 0;
170 uint32 winnerTeamRating = 0;
171 uint32 winnerMatchmakerRating = 0;
172 int32 winnerChange = 0;
173 int32 winnerMatchmakerChange = 0;
174
175 ArenaTeam* winnerArenaTeam = sArenaTeamMgr->GetArenaTeamById(GetArenaTeamIdForTeam(winnerTeamId == TEAM_NEUTRAL ? TEAM_HORDE : winnerTeamId));
176 ArenaTeam* loserArenaTeam = sArenaTeamMgr->GetArenaTeamById(GetArenaTeamIdForTeam(winnerTeamId == TEAM_NEUTRAL ? TEAM_ALLIANCE : GetOtherTeamId(winnerTeamId)));
177
178 auto SaveArenaLogs = [&]()
179 {
180 // pussywizard: arena logs in database
181 uint32 fightId = sArenaTeamMgr->GetNextArenaLogId();
182 uint32 currOnline = sWorld->GetActiveSessionCount();
183
184 CharacterDatabaseTransaction trans = CharacterDatabase.BeginTransaction();
185 CharacterDatabasePreparedStatement* stmt2 = CharacterDatabase.GetPreparedStatement(CHAR_INS_ARENA_LOG_FIGHT);
186 stmt2->SetData(0, fightId);
187 stmt2->SetData(1, GetArenaType());
188 stmt2->SetData(2, ((GetStartTime() <= startDelay ? 0 : GetStartTime() - startDelay) / 1000));
189 stmt2->SetData(3, winnerArenaTeam->GetId());
190 stmt2->SetData(4, loserArenaTeam->GetId());
191 stmt2->SetData(5, (uint16)winnerTeamRating);
192 stmt2->SetData(6, (uint16)winnerMatchmakerRating);
193 stmt2->SetData(7, (int16)winnerChange);
194 stmt2->SetData(8, (uint16)loserTeamRating);
195 stmt2->SetData(9, (uint16)loserMatchmakerRating);
196 stmt2->SetData(10, (int16)loserChange);
197 stmt2->SetData(11, currOnline);
198 trans->Append(stmt2);
199
200 uint8 memberId = 0;
201 for (auto const& [playerGuid, arenaLogEntryData] : ArenaLogEntries)
202 {
203 stmt2 = CharacterDatabase.GetPreparedStatement(CHAR_INS_ARENA_LOG_MEMBERSTATS);
204 stmt2->SetData(0, fightId);
205 stmt2->SetData(1, ++memberId);
206 stmt2->SetData(2, arenaLogEntryData.Name);
207 stmt2->SetData(3, arenaLogEntryData.Guid);
208 stmt2->SetData(4, arenaLogEntryData.ArenaTeamId);
209 stmt2->SetData(5, arenaLogEntryData.Acc);
210 stmt2->SetData(6, arenaLogEntryData.IP);
211 stmt2->SetData(7, arenaLogEntryData.DamageDone);
212 stmt2->SetData(8, arenaLogEntryData.HealingDone);
213 stmt2->SetData(9, arenaLogEntryData.KillingBlows);
214 trans->Append(stmt2);
215 }
216
217 CharacterDatabase.CommitTransaction(trans);
218 };
219
220 if (winnerArenaTeam && loserArenaTeam && winnerArenaTeam != loserArenaTeam)
221 {
222 loserTeamRating = loserArenaTeam->GetRating();
223 loserMatchmakerRating = GetArenaMatchmakerRating(GetOtherTeamId(winnerTeamId));
224 winnerTeamRating = winnerArenaTeam->GetRating();
225 winnerMatchmakerRating = GetArenaMatchmakerRating(winnerTeamId);
226
227 if (winnerTeamId != TEAM_NEUTRAL)
228 {
229 winnerMatchmakerChange = bValidArena ? winnerArenaTeam->WonAgainst(winnerMatchmakerRating, loserMatchmakerRating, winnerChange, GetBgMap()) : 0;
230 loserMatchmakerChange = loserArenaTeam->LostAgainst(loserMatchmakerRating, winnerMatchmakerRating, loserChange, GetBgMap());
231
232 sScriptMgr->OnAfterArenaRatingCalculation(this, winnerMatchmakerChange, loserMatchmakerChange, winnerChange, loserChange);
233
234 LOG_DEBUG("bg.arena", "match Type: {} --- Winner: old rating: {}, rating gain: {}, old MMR: {}, MMR gain: {} --- Loser: old rating: {}, rating loss: {}, old MMR: {}, MMR loss: {} ---",
235 GetArenaType(), winnerTeamRating, winnerChange, winnerMatchmakerRating, winnerMatchmakerChange,
236 loserTeamRating, loserChange, loserMatchmakerRating, loserMatchmakerChange);
237
238 SetArenaMatchmakerRating(winnerTeamId, winnerMatchmakerRating + winnerMatchmakerChange);
239 SetArenaMatchmakerRating(GetOtherTeamId(winnerTeamId), loserMatchmakerRating + loserMatchmakerChange);
240
241 // bg team that the client expects is different to TeamId
242 // alliance 1, horde 0
243 uint8 winnerTeam = winnerTeamId == TEAM_ALLIANCE ? PVP_TEAM_ALLIANCE : PVP_TEAM_HORDE;
244 uint8 loserTeam = winnerTeamId == TEAM_ALLIANCE ? PVP_TEAM_HORDE : PVP_TEAM_ALLIANCE;
245
246 _arenaTeamScores[winnerTeam].Assign(winnerChange, winnerMatchmakerRating, winnerArenaTeam->GetName());
247 _arenaTeamScores[loserTeam].Assign(loserChange, loserMatchmakerRating, loserArenaTeam->GetName());
248
249 LOG_DEBUG("bg.arena", "Arena match Type: {} for Team1Id: {} - Team2Id: {} ended. WinnerTeamId: {}. Winner rating: +{}, Loser rating: {}",
250 GetArenaType(), GetArenaTeamIdForTeam(TEAM_ALLIANCE), GetArenaTeamIdForTeam(TEAM_HORDE), winnerArenaTeam->GetId(), winnerChange, loserChange);
251 }
252 else // Deduct 16 points from each teams arena-rating if there are no winners after 45+2 minutes
253 {
254 // pussywizard: in case of a draw, the following is always true:
255 // winnerArenaTeam => TEAM_HORDE, loserArenaTeam => TEAM_ALLIANCE
256
257 winnerTeamRating = winnerArenaTeam->GetRating();
258 winnerMatchmakerRating = GetArenaMatchmakerRating(TEAM_HORDE);
259 loserTeamRating = loserArenaTeam->GetRating();
260 loserMatchmakerRating = GetArenaMatchmakerRating(TEAM_ALLIANCE);
261 winnerMatchmakerChange = 0;
262 loserMatchmakerChange = 0;
263 winnerChange = ARENA_TIMELIMIT_POINTS_LOSS;
264 loserChange = ARENA_TIMELIMIT_POINTS_LOSS;
265
266 _arenaTeamScores[PVP_TEAM_ALLIANCE].Assign(ARENA_TIMELIMIT_POINTS_LOSS, winnerMatchmakerRating, winnerArenaTeam->GetName());
267 _arenaTeamScores[PVP_TEAM_HORDE].Assign(ARENA_TIMELIMIT_POINTS_LOSS, loserMatchmakerRating, loserArenaTeam->GetName());
268
269 winnerArenaTeam->FinishGame(ARENA_TIMELIMIT_POINTS_LOSS, GetBgMap());
270 loserArenaTeam->FinishGame(ARENA_TIMELIMIT_POINTS_LOSS, GetBgMap());
271 }
272 }
273
274 SaveArenaLogs();
275
276 uint8 aliveWinners = GetAlivePlayersCountByTeam(winnerTeamId);
277
278 for (auto const& [playerGuid, player] : GetPlayers())
279 {
280 auto const& bgTeamId = player->GetBgTeamId();
281
282 // per player calculation
283 if (winnerArenaTeam && loserArenaTeam && winnerArenaTeam != loserArenaTeam)
284 {
285 if (bgTeamId == winnerTeamId)
286 {
287 if (bValidArena)
288 {
289 // update achievement BEFORE personal rating update
290 uint32 rating = player->GetArenaPersonalRating(winnerArenaTeam->GetSlot());
291 player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_WIN_RATED_ARENA, rating ? rating : 1);
292 player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_WIN_ARENA, GetMapId());
293
294 // Last standing - Rated 5v5 arena & be solely alive player
295 if (GetArenaType() == ARENA_TYPE_5v5 && aliveWinners == 1 && player->IsAlive())
296 {
297 player->CastSpell(player, SPELL_LAST_MAN_STANDING, true);
298 }
299
300 winnerArenaTeam->MemberWon(player, loserMatchmakerRating, winnerMatchmakerChange);
301 }
302 }
303 else
304 {
305 loserArenaTeam->MemberLost(player, winnerMatchmakerRating, loserMatchmakerChange);
306
307 // Arena lost => reset the win_rated_arena having the "no_lose" condition
308 player->ResetAchievementCriteria(ACHIEVEMENT_CRITERIA_CONDITION_NO_LOSE, 0);
309 }
310
311 player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_PLAY_ARENA, GetMapId());
312 }
313 }
314
315 // update previous opponents for arena queue
316 winnerArenaTeam->SetPreviousOpponents(loserArenaTeam->GetId());
317 loserArenaTeam->SetPreviousOpponents(winnerArenaTeam->GetId());
318
319 // save the stat changes
320 if (bValidArena)
321 {
322 winnerArenaTeam->SaveToDB();
323 winnerArenaTeam->NotifyStatsChanged();
324 }
325
326 loserArenaTeam->SaveToDB();
327 loserArenaTeam->NotifyStatsChanged();
328 }
329
330 // end battleground
331 Battleground::EndBattleground(winnerTeamId);
332}
