367 {

373 case 1:

378 break;
379 case 3:
380 cEntry = NPC_ADARRAH;
381 break;
382 case 4:
383 cEntry = NPC_FUDGERICK;
384 break;
385 case 5:
386 cEntry = NPC_DARWEN;
387 break;
388 case 6:
389 cEntry = NPC_MITZI;
390 break;
391 case 7:
392 cEntry = NPC_CHRISTIAN;
393 break;
394 case 8:
395 cEntry = NPC_BRENNAN;
396 break;
397 case 9:
398 cEntry = NPC_HOLLEE;
399 break;
400 }
401
402 if (!instance->GetData(TYPE_RAND_VENDOR_1) && roll_chance_i(10))
403 {
404 cEntry = NPC_GUNTER;
405 instance->SetData(TYPE_RAND_VENDOR_1, DONE);
406 }
407 else if (!instance->GetData(TYPE_RAND_VENDOR_2) && roll_chance_i(10))
408 {
409 cEntry = NPC_KYREN;
410 instance->SetData(TYPE_RAND_VENDOR_2, DONE);
411 }
412
413 // start generic rp
414 eventTimer = 1;
415 events.ScheduleEvent(eventTimer, 3000);
416
417 me->UpdateEntry(cEntry);
418 }
