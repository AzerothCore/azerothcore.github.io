81class boss_sacrolash : public CreatureScript
82{
83public:
84 boss_sacrolash() : CreatureScript("boss_sacrolash") { }
85
86 struct boss_sacrolashAI : public BossAI
87 {
88 boss_sacrolashAI(Creature* creature) : BossAI(creature, DATA_EREDAR_TWINS) {}
89
90 bool sisterDied;
91 void Reset() override
92 {
93 me->CastSpell(me, SPELL_SHADOWFORM, true);
94 sisterDied = false;
95 BossAI::Reset();
96 me->SetLootMode(0);
97 }
98
99 void DoAction(int32 param) override
100 {
101 if (param == ACTION_SISTER_DIED)
102 {
103 me->ResetLootMode();
104 sisterDied = true;
105 Talk(YELL_SISTER_ALYTHESS_DEAD);
106 me->CastSpell(me, SPELL_EMPOWER, true);
107
108 uint32 timer = events.GetNextEventTime(EVENT_SPELL_SHADOW_NOVA);
109 events.CancelEvent(EVENT_SPELL_SHADOW_NOVA);
110 events.ScheduleEvent(EVENT_SPELL_CONFLAGRATION, timer - events.GetTimer());
111 }
112 }
113
114 void EnterEvadeMode(EvadeReason why) override
115 {
116 BossAI::EnterEvadeMode(why);
117 if (Creature* alythess = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_GRAND_WARLOCK_ALYTHESS)))
118 {
119 if (!alythess->IsAlive())
120 alythess->Respawn(true);
121 else if (!alythess->IsInEvadeMode())
122 alythess->AI()->EnterEvadeMode(why);
123 }
124 }
125
126 void EnterCombat(Unit* who) override
127 {
128 BossAI::EnterCombat(who);
129 if (Creature* alythess = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_GRAND_WARLOCK_ALYTHESS)))
130 if (alythess->IsAlive() && !alythess->IsInCombat())
131 alythess->AI()->AttackStart(who);
132
133 events.ScheduleEvent(EVENT_SPELL_SHADOW_BLADES, 10000);
134 events.ScheduleEvent(EVENT_SPELL_SHADOW_NOVA, 36000);
135 events.ScheduleEvent(EVENT_SPELL_CONFOUNDING_BLOW, 25000);
136 events.ScheduleEvent(EVENT_SHADOW_IMAGE, 20000);
137 events.ScheduleEvent(EVENT_SPELL_ENRAGE, 360000);
138 }
139
140 void KilledUnit(Unit* victim) override
141 {
142 if (victim->GetTypeId() == TYPEID_PLAYER && urand(0, 1))
143 Talk(YELL_SAC_KILL);
144 }
145
146 void JustDied(Unit* /*killer*/) override
147 {
148 events.Reset();
149 summons.DespawnAll();
150
151 if (sisterDied)
152 {
153 Talk(YELL_SAC_DEAD);
154 instance->SetBossState(DATA_EREDAR_TWINS, DONE);
155 }
156 else if (Creature* alythess = ObjectAccessor::GetCreature(*me, instance->GetGuidData(NPC_GRAND_WARLOCK_ALYTHESS)))
157 alythess->AI()->DoAction(ACTION_SISTER_DIED);
158 }
159
160 void JustSummoned(Creature* summon) override
161 {
162 summons.Summon(summon);
163 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true))
164 {
165 summon->AI()->AttackStart(target);
166 summon->AddThreat(target, 10000000);
167 }
168 }
169
170 void UpdateAI(uint32 diff) override
171 {
172 if (!UpdateVictim())
173 return;
174
175 events.Update(diff);
176 if (me->HasUnitState(UNIT_STATE_CASTING))
177 return;
178
179 switch (events.ExecuteEvent())
180 {
181 case EVENT_SPELL_ENRAGE:
182 Talk(YELL_ENRAGE);
183 me->CastSpell(me, SPELL_ENRAGE, true);
184 break;
185 case EVENT_SPELL_CONFOUNDING_BLOW:
186 me->CastSpell(me->GetVictim(), SPELL_CONFOUNDING_BLOW, false);
187 events.ScheduleEvent(EVENT_SPELL_CONFOUNDING_BLOW, urand(20000, 25000));
188 break;
189 case EVENT_SPELL_SHADOW_BLADES:
190 me->CastSpell(me, SPELL_SHADOW_BLADES, false);
191 events.ScheduleEvent(EVENT_SPELL_SHADOW_BLADES, 10000);
192 break;
193 case EVENT_SPELL_SHADOW_NOVA:
194 {
195 Unit* target = SelectTarget(SelectTargetMethod::MaxThreat, 1, 100.0f);
196 if (!target)
197 target = me->GetVictim();
198 Talk(EMOTE_SHADOW_NOVA, target);
199 Talk(YELL_SHADOW_NOVA);
200 me->CastSpell(target, SPELL_SHADOW_NOVA, false);
201 events.ScheduleEvent(EVENT_SPELL_SHADOW_NOVA, urand(30000, 35000));
202 break;
203 }
204 case EVENT_SHADOW_IMAGE:
205 me->SummonCreature(NPC_SHADOW_IMAGE, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation(), TEMPSUMMON_TIMED_DESPAWN, 12000);
206 events.ScheduleEvent(EVENT_SHADOW_IMAGE, 6000);
207 break;
208 case EVENT_SPELL_CONFLAGRATION:
209 {
210 Unit* target = SelectTarget(SelectTargetMethod::MaxThreat, 1, 100.0f);
211 if (!target)
212 target = me->GetVictim();
213 me->CastSpell(target, SPELL_CONFLAGRATION, false);
214 events.ScheduleEvent(EVENT_SPELL_CONFLAGRATION, urand(30000, 35000));
215 break;
216 }
217 }
218
219 DoMeleeAttackIfReady();
220 }
221 };
222
223 CreatureAI* GetAI(Creature* creature) const override
224 {
225 return GetSunwellPlateauAI<boss_sacrolashAI>(creature);
226 };
