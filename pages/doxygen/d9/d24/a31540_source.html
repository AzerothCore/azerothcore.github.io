68int main(int argc, char** argv)
69{
70 Acore::Impl::CurrentServerProcessHolder::_type = SERVER_PROCESS_AUTHSERVER;
71 signal(SIGABRT, &Acore::AbortHandler);
72
73 // Command line parsing
74 auto configFile = fs::path(sConfigMgr->GetConfigPath() + std::string(_ACORE_REALM_CONFIG));
75 auto vm = GetConsoleArguments(argc, argv, configFile);
76
77 // exit if help or version is enabled
78 if (vm.count("help"))
79 return 0;
80
81 // Add file and args in config
82 sConfigMgr->Configure(configFile.generic_string(), std::vector<std::string>(argv, argv + argc));
83
84 if (!sConfigMgr->LoadAppConfigs())
85 return 1;
86
87 // Init logging
88 sLog->RegisterAppender<AppenderDB>();
89 sLog->Initialize(nullptr);
90
91 Acore::Banner::Show("authserver",
92 [](std::string_view text)
93 {
94 LOG_INFO("server.authserver", text);
95 },
96 []()
97 {
98 LOG_INFO("server.authserver", "> Using configuration file {}", sConfigMgr->GetFilename());
99 LOG_INFO("server.authserver", "> Using SSL version: {} (library: {})", OPENSSL_VERSION_TEXT, SSLeay_version(SSLEAY_VERSION));
100 LOG_INFO("server.authserver", "> Using Boost version: {}.{}.{}", BOOST_VERSION / 100000, BOOST_VERSION / 100 % 1000, BOOST_VERSION % 100);
101 });
102
103 // authserver PID file creation
104 std::string pidFile = sConfigMgr->GetOption<std::string>("PidFile", "");
105 if (!pidFile.empty())
106 {
107 if (uint32 pid = CreatePIDFile(pidFile))
108 LOG_INFO("server.authserver", "Daemon PID: {}\n", pid); // outError for red color in console
109 else
110 {
111 LOG_ERROR("server.authserver", "Cannot create PID file {} (possible error: permission)\n", pidFile);
112 return 1;
113 }
114 }
115
116 // Initialize the database connection
117 if (!StartDB())
118 return 1;
119
120 sSecretMgr->Initialize();
121
122 // Load IP Location Database
123 sIPLocation->Load();
124
125 std::shared_ptr<void> dbHandle(nullptr, [](void*) { StopDB(); });
126
127 std::shared_ptr<Acore::Asio::IoContext> ioContext = std::make_shared<Acore::Asio::IoContext>();
128
129 // Get the list of realms for the server
130 sRealmList->Initialize(*ioContext, sConfigMgr->GetOption<int32>("RealmsStateUpdateDelay", 20));
131
132 std::shared_ptr<void> sRealmListHandle(nullptr, [](void*) { sRealmList->Close(); });
133
134 if (sRealmList->GetRealms().empty())
135 {
136 LOG_ERROR("server.authserver", "No valid realms specified.");
137 return 1;
138 }
139
140 // Stop auth server if dry run
141 if (sConfigMgr->isDryRun())
142 {
143 LOG_INFO("server.authserver", "Dry run completed, terminating.");
144 return 0;
145 }
146
147 // Start the listening port (acceptor) for auth connections
148 int32 port = sConfigMgr->GetOption<int32>("RealmServerPort", 3724);
149 if (port < 0 || port > 0xFFFF)
150 {
151 LOG_ERROR("server.authserver", "Specified port out of allowed range (1-65535)");
152 return 1;
153 }
154
155 std::string bindIp = sConfigMgr->GetOption<std::string>("BindIP", "0.0.0.0");
156
157 if (!sAuthSocketMgr.StartNetwork(*ioContext, bindIp, port))
158 {
159 LOG_ERROR("server.authserver", "Failed to initialize network");
160 return 1;
161 }
162
163 std::shared_ptr<void> sAuthSocketMgrHandle(nullptr, [](void*) { sAuthSocketMgr.StopNetwork(); });
164
165 // Set signal handlers
166 boost::asio::signal_set signals(*ioContext, SIGINT, SIGTERM);
167#if AC_PLATFORM == AC_PLATFORM_WINDOWS
168 signals.add(SIGBREAK);
169#endif
170 signals.async_wait(std::bind(&SignalHandler, std::weak_ptr<Acore::Asio::IoContext>(ioContext), std::placeholders::_1, std::placeholders::_2));
171
172 // Set process priority according to configuration settings
173 SetProcessPriority("server.authserver", sConfigMgr->GetOption<int32>(CONFIG_PROCESSOR_AFFINITY, 0), sConfigMgr->GetOption<bool>(CONFIG_HIGH_PRIORITY, false));
174
175 // Enabled a timed callback for handling the database keep alive ping
176 int32 dbPingInterval = sConfigMgr->GetOption<int32>("MaxPingTime", 30);
177 std::shared_ptr<Acore::Asio::DeadlineTimer> dbPingTimer = std::make_shared<Acore::Asio::DeadlineTimer>(*ioContext);
178 dbPingTimer->expires_from_now(boost::posix_time::minutes(dbPingInterval));
179 dbPingTimer->async_wait(std::bind(&KeepDatabaseAliveHandler, std::weak_ptr<Acore::Asio::DeadlineTimer>(dbPingTimer), dbPingInterval, std::placeholders::_1));
180
181 int32 banExpiryCheckInterval = sConfigMgr->GetOption<int32>("BanExpiryCheckInterval", 60);
182 std::shared_ptr<Acore::Asio::DeadlineTimer> banExpiryCheckTimer = std::make_shared<Acore::Asio::DeadlineTimer>(*ioContext);
183 banExpiryCheckTimer->expires_from_now(boost::posix_time::seconds(banExpiryCheckInterval));
184 banExpiryCheckTimer->async_wait(std::bind(&BanExpiryHandler, std::weak_ptr<Acore::Asio::DeadlineTimer>(banExpiryCheckTimer), banExpiryCheckInterval, std::placeholders::_1));
185
186 // Start the io service worker loop
187 ioContext->run();
188
189 banExpiryCheckTimer->cancel();
190 dbPingTimer->cancel();
191
192 LOG_INFO("server.authserver", "Halting process...");
193
194 signals.cancel();
195
196 return 0;
197}

200bool StartDB()
201{
202 MySQL::Library_Init();
203
204 // Load databases
205 // NOTE: While authserver is singlethreaded you should keep synch_threads == 1.
206 // Increasing it is just silly since only 1 will be used ever.
207 DatabaseLoader loader("server.authserver");
208 loader
209 .AddDatabase(LoginDatabase, "Login");
210
211 if (!loader.Load())
212 return false;
213
214 LOG_INFO("server.authserver", "Started auth database connection pool.");
215 sLog->SetRealmId(0); // Enables DB appenders when realm is set.
216 return true;
217}

237void KeepDatabaseAliveHandler(std::weak_ptr<Acore::Asio::DeadlineTimer> dbPingTimerRef, int32 dbPingInterval, boost::system::error_code const& error)
238{
239 if (!error)
240 {
241 if (std::shared_ptr<Acore::Asio::DeadlineTimer> dbPingTimer = dbPingTimerRef.lock())
242 {
243 LOG_INFO("server.authserver", "Ping MySQL to keep connection alive");
244 LoginDatabase.KeepAlive();
245
246 dbPingTimer->expires_from_now(boost::posix_time::minutes(dbPingInterval));
247 dbPingTimer->async_wait(std::bind(&KeepDatabaseAliveHandler, dbPingTimerRef, dbPingInterval, std::placeholders::_1));
248 }
249 }
250}

252void BanExpiryHandler(std::weak_ptr<Acore::Asio::DeadlineTimer> banExpiryCheckTimerRef, int32 banExpiryCheckInterval, boost::system::error_code const& error)
253{
254 if (!error)
255 {
256 if (std::shared_ptr<Acore::Asio::DeadlineTimer> banExpiryCheckTimer = banExpiryCheckTimerRef.lock())
257 {
258 LoginDatabase.Execute(LoginDatabase.GetPreparedStatement(LOGIN_DEL_EXPIRED_IP_BANS));
259 LoginDatabase.Execute(LoginDatabase.GetPreparedStatement(LOGIN_UPD_EXPIRED_ACCOUNT_BANS));
260
261 banExpiryCheckTimer->expires_from_now(boost::posix_time::seconds(banExpiryCheckInterval));
262 banExpiryCheckTimer->async_wait(std::bind(&BanExpiryHandler, banExpiryCheckTimerRef, banExpiryCheckInterval, std::placeholders::_1));
263 }
264 }

267variables_map GetConsoleArguments(int argc, char** argv, fs::path& configFile)
268{
269 options_description all("Allowed options");
270 all.add_options()
271 ("help,h", "print usage message")
272 ("version,v", "print version build info")
273 ("dry-run,d", "Dry run")
274 ("config,c", value<fs::path>(&configFile)->default_value(fs::path(sConfigMgr->GetConfigPath() + std::string(_ACORE_REALM_CONFIG))), "use <arg> as configuration file");
275
276 variables_map variablesMap;
277
278 try
279 {
280 store(command_line_parser(argc, argv).options(all).allow_unregistered().run(), variablesMap);
281 notify(variablesMap);
282 }
283 catch (std::exception const& e)
284 {
285 std::cerr << e.what() << "\n";
286 }
287
288 if (variablesMap.count("help"))
289 {
290 std::cout << all << "\n";
291 }
292 else if (variablesMap.count("dry-run"))
293 {
294 sConfigMgr->setDryRun(true);
295 }
296
297 return variablesMap;
298}
