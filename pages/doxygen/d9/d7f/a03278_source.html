287 SphereGUID[i] = c->GetGUID();
288 break;
289 }
290 i = (i + 1) % 6;
291 } while( i != StartAt );
292 events.RepeatEvent(4000);
293 }
294 break;
295 case EVENT_SPELL_FREEZING_SLASH:
296 {
297 if( me->GetVictim() )
298 me->CastSpell(me->GetVictim(), SPELL_FREEZING_SLASH, false);
299 events.RepeatEvent(urand(15000, 20000));
300 }
301 break;
302 case EVENT_SPELL_PENETRATING_COLD:
303 {
304 me->CastCustomSpell(SPELL_PENETRATING_COLD, SPELLVALUE_MAX_TARGETS, RAID_MODE(2, 5, 2, 5));
305 events.RepeatEvent(18000);
306 }
307 break;
308 case EVENT_SUMMON_NERUBIAN:
309 {
310 me->CastCustomSpell(SPELL_SUMMON_BURROWER, SPELLVALUE_MAX_TARGETS, RAID_MODE(1, 2, 2, 4));
311 events.RepeatEvent(45000);
312 }
313 break;
314 case EVENT_SUBMERGE:
315 {
316 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
317 bool berserk = me->HasAura(SPELL_BERSERK);
318 me->RemoveAllAuras();
319 if (berserk)
320 me->CastSpell(me, SPELL_BERSERK, true);
321 Talk(EMOTE_SUBMERGE);
322 Talk(EMOTE_BURROWER);
323 me->CastSpell(me, SPELL_SUBMERGE, false);
324 events.CancelEvent(EVENT_SUMMON_NERUBIAN);
325 events.CancelEvent(EVENT_SPELL_FREEZING_SLASH);
326 events.CancelEvent(EVENT_SPELL_PENETRATING_COLD);
327 events.RescheduleEvent(EVENT_EMERGE, EMERGE_INTERVAL);
328 events.RescheduleEvent(EVENT_SPELL_SUMMON_SPIKE, 2500);
329 events.RescheduleEvent(EVENT_SUMMON_SCARAB, 3000);
330 }
331 break;
332 case EVENT_SUMMON_SCARAB:
333 {
334 uint8 i = urand(0, 3);
335 if( Creature* c = ObjectAccessor::GetCreature(*me, BurrowGUID[i]) )
336 me->CastSpell(c, SPELL_SUMMON_SCARAB, true);
337 events.RepeatEvent(4000);
338 }
339 break;
340 case EVENT_EMERGE:
341 {
342 me->CastSpell(me, SPELL_SPIKE_TELE, true);
343 summons.DespawnEntry(NPC_SPIKE);
344 events.CancelEvent(EVENT_SUMMON_SCARAB);
345 events.RescheduleEvent(EVENT_EMERGE_2, 2000);
346 }
347 break;
348 case EVENT_EMERGE_2:
349 {
350 Talk(SAY_EMERGE);
351 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
352 me->setAttackTimer(BASE_ATTACK, 3000);
353 me->RemoveAura(SPELL_SUBMERGE);
354 me->CastSpell(me, SPELL_EMERGE, false);
355 events.RescheduleEvent(EVENT_SUMMON_NERUBIAN, urand(5000, 8000));
356 events.RescheduleEvent(EVENT_SPELL_FREEZING_SLASH, urand(7000, 15000));
357 events.RescheduleEvent(EVENT_SPELL_PENETRATING_COLD, urand(15000, 20000));
358 events.RescheduleEvent(EVENT_SUBMERGE, SUBMERGE_INTERVAL);
359 }
360 break;
361 case EVENT_SPELL_SUMMON_SPIKE:
362 me->CastSpell(me, SPELL_SUMMON_SPIKE, true);
363 break;
364 }

367 }
