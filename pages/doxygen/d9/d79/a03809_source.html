132 void UpdateAI(uint32 diff) override
133 {
134 if (!UpdateVictim())
135 return;
136
137 events.Update(diff);
138 if (me->HasUnitState(UNIT_STATE_CASTING))
139 return;
140
141 switch (events.ExecuteEvent())
142 {
143 case EVENT_SPELL_FROST_NOVA:
144 me->CastSpell(me, SPELL_FROSTNOVA, false);
145 events.RepeatEvent(urand(16000, 23000));
146 events.DelayEvents(1500);
147 events.ScheduleEvent(EVENT_SPELL_BLINK, 1500);
148 break;
149 case EVENT_SPELL_FR_FI:
150 me->CastSpell(me->GetVictim(), RAND(SPELL_FROSTBOLT, SPELL_FIREBALL), false);
151 events.RepeatEvent(urand(3000, 4000));
152 break;
153 case EVENT_SPELL_BLINK:
154 me->CastSpell(me, SPELL_BLINK, false);
155 events.RescheduleEvent(EVENT_SPELL_FR_FI, 0);
156 break;
157 case EVENT_SPELL_BEACON:
158 if (!urand(0, 3))
159 Talk(SAY_SUMMON);
160
161 me->CastSpell(me, SPELL_ETHEREAL_BEACON, true);
162 events.RepeatEvent(10000);
163 break;
164 }
165
166 DoMeleeAttackIfReady();
167 }
168 };
