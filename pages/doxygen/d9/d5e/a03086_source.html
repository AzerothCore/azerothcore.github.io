234 void ScheduleTasks(bool onEngage = false)
235 {
236 _scheduler.
237 Schedule(3s, [this, onEngage](TaskContext task)
238 {
239 if (task.GetRepeatCounter() < 3 && onEngage)
240 {
241 if (Unit* target = ObjectAccessor::GetUnit(*me, _beamTarget))
242 {
243 DoCast(target, SPELL_GREEN_BEAM);
244 }
245
246 task.Repeat();
247 }
248 else
249 {
250 _scheduler.Schedule(5s, [this](TaskContext task)
251 {
252 DoCastRandomTarget(SPELL_GREEN_BEAM);
253
254 task.SetGroup(GROUP_BEAM_PHASE);
255 task.Repeat(3s);
256 });
257 }
258
259 task.SetGroup(GROUP_BEAM_PHASE);
260 })
261 .Schedule(8s, [this](TaskContext task)
262 {
263 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true))
264 {
265 if (Creature* tentacle = me->SummonCreature(NPC_CLAW_TENTACLE, *target, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 5000))
266 {
267 tentacle->AI()->AttackStart(target);
268 }
269 }
270
271 task.SetGroup(GROUP_BEAM_PHASE);
272 task.Repeat();
273 })
274 .Schedule(45s, [this](TaskContext task)
275 {
276 DoAction(ACTION_SPAWN_EYE_TENTACLES);
277 task.SetGroup(GROUP_BEAM_PHASE);
278 task.Repeat();
279 })
280 .Schedule(50s, [this](TaskContext /*task*/)
281 {
282 _scheduler.CancelGroup(GROUP_BEAM_PHASE);
283
284 me->StopMoving();
285 me->SetReactState(REACT_PASSIVE);
286 me->InterruptNonMeleeSpells(false);
287 me->SetTarget(ObjectGuid::Empty);
288
289 //Freeze animation
290 DoCast(me, SPELL_FREEZE_ANIM, true);
291
292 _scheduler.Schedule(1s, [this](TaskContext /*task*/)
293 {
294 //Select random target for dark beam to start on
295 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true))
296 {
297 //Face our target
298 DarkGlareAngle = me->GetAngle(target);
299 DarkGlareTick = 0;
300 ClockWise = RAND(true, false);
301
302 //Add red coloration to C'thun
303 DoCast(me, SPELL_RED_COLORATION, true);
304
305 me->StopMoving();
306 me->SetFacingToObject(target);
307 me->SetOrientation(DarkGlareAngle);
308 }
309
310 _scheduler.Schedule(3s, [this](TaskContext tasker)
311 {
312 me->SetTarget(ObjectGuid::Empty);
313 me->StopMoving();
314
315 float angle = ClockWise ? DarkGlareAngle + DarkGlareTick * float(M_PI) / 35 : DarkGlareAngle - DarkGlareTick * float(M_PI) / 35;
316 me->SetFacingTo(angle);
317 me->SetOrientation(angle);
318
319 DoCastSelf(SPELL_DARK_GLARE);
320
321 ++DarkGlareTick;
322
323 if (tasker.GetRepeatCounter() >= 35)
324 {
325 _scheduler.CancelAll();
326 me->SetReactState(REACT_AGGRESSIVE);
327 me->RemoveAurasDueToSpell(SPELL_RED_COLORATION);
328 me->RemoveAurasDueToSpell(SPELL_FREEZE_ANIM);
329 me->InterruptNonMeleeSpells(false);
330 ScheduleTasks();
331 }
332 else
333 tasker.Repeat(1s);
334 });
335 });
336 });
337 }

497 void UpdateAI(uint32 diff) override
498 {
499 //Check if we have a target
500 if (!UpdateVictim())
501 {
502 //No target so we'll use this section to do our random wispers instance wide
503 //WisperTimer
504 if (WisperTimer <= diff)
505 {
506 Map* map = me->GetMap();
507 if (!map->IsDungeon())
508 return;
509
510 //Play random sound to the zone
511 Map::PlayerList const& PlayerList = map->GetPlayers();
512
513 if (!PlayerList.IsEmpty())
514 {
515 for (Map::PlayerList::const_iterator itr = PlayerList.begin(); itr != PlayerList.end(); ++itr)
516 {
517 if (Player* pPlr = itr->GetSource())
518 pPlr->PlayDirectSound(RANDOM_SOUND_WHISPER, pPlr);
519 }
520 }
521
522 //One random wisper every 90 - 300 seconds
523 WisperTimer = urand(90000, 300000);
524 }
525 else WisperTimer -= diff;
526
527 return;
528 }
529
530 me->SetTarget();
531
532 _scheduler.Update(diff);
533 }
