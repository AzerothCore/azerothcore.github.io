156class spell_hun_generic_scaling : public AuraScript
157{
158 PrepareAuraScript(spell_hun_generic_scaling);
159
160 void CalculateResistanceAmount(AuraEffect const* aurEff, int32& amount, bool& /*canBeRecalculated*/)
161 {
162 // xinef: pet inherits 40% of resistance from owner and 35% of armor
163 if (Unit* owner = GetUnitOwner()->GetOwner())
164 {
165 SpellSchoolMask schoolMask = SpellSchoolMask(aurEff->GetSpellInfo()->Effects[aurEff->GetEffIndex()].MiscValue);
166 int32 modifier = schoolMask == SPELL_SCHOOL_MASK_NORMAL ? 35 : 40;
167 amount = CalculatePct(std::max<int32>(0, owner->GetResistance(schoolMask)), modifier);
168 }
169 }
170
171 void CalculateStatAmount(AuraEffect const* aurEff, int32& amount, bool& /*canBeRecalculated*/)
172 {
173 if (Unit* owner = GetUnitOwner()->GetOwner())
174 {
175 // xinef: by default pet inherits 45% of stamina
176 int32 modifier = 45;
177
178 // xinef: Wild Hunt bonus for stamina
179 if (AuraEffect* wildHuntEff = GetUnitOwner()->GetDummyAuraEffect(SPELLFAMILY_PET, 3748, EFFECT_0))
180 AddPct(modifier, wildHuntEff->GetAmount());
181
182 amount = CalculatePct(std::max<int32>(0, owner->GetStat(Stats(aurEff->GetSpellInfo()->Effects[aurEff->GetEffIndex()].MiscValue))), modifier);
183 }
184 }
185
186 void CalculateAPAmount(AuraEffect const* /*aurEff*/, int32& amount, bool& /*canBeRecalculated*/)
187 {
188 if (Unit* owner = GetUnitOwner()->GetOwner())
189 {
190 // xinef: by default 22% of RAP
191 int32 modifier = 22;
192
193 // xinef: Wild Hunt bonus for AP
194 if (AuraEffect* wildHuntEff = GetUnitOwner()->GetDummyAuraEffect(SPELLFAMILY_PET, 3748, EFFECT_1))
195 AddPct(modifier, wildHuntEff->GetAmount());
196
197 float ownerAP = owner->GetTotalAttackPowerValue(RANGED_ATTACK);
198
199 // Xinef: Hunter vs. Wild
200 if (AuraEffect* HvWEff = owner->GetAuraEffect(SPELL_AURA_MOD_ATTACK_POWER_OF_STAT_PERCENT, SPELLFAMILY_HUNTER, 3647, EFFECT_0))
201 ownerAP += CalculatePct(owner->GetStat(STAT_STAMINA), HvWEff->GetAmount());
202
203 amount = CalculatePct(std::max<int32>(0, ownerAP), modifier);
204 }
205 }
206
207 void CalculateSPAmount(AuraEffect const* /*aurEff*/, int32& amount, bool& /*canBeRecalculated*/)
208 {
209 if (Unit* owner = GetUnitOwner()->GetOwner())
210 {
211 // xinef: by default 12.87% of RAP
212 float modifier = 12.87f;
213
214 // xinef: Wild Hunt bonus for AP
215 if (AuraEffect* wildHuntEff = GetUnitOwner()->GetDummyAuraEffect(SPELLFAMILY_PET, 3748, EFFECT_1))
216 AddPct(modifier, wildHuntEff->GetAmount());
217
218 amount = CalculatePct(std::max<int32>(0, owner->GetTotalAttackPowerValue(RANGED_ATTACK)), modifier);
219
220 // xinef: Update appropriate player field
221 if (owner->GetTypeId() == TYPEID_PLAYER)
222 owner->SetUInt32Value(PLAYER_PET_SPELL_POWER, (uint32)amount);
223 }
224 }
225
226 void CalcPeriodic(AuraEffect const* /*aurEff*/, bool& isPeriodic, int32& amplitude)
227 {
228 isPeriodic = true;
229 amplitude = 2 * IN_MILLISECONDS;
230 }
231
232 void HandlePeriodic(AuraEffect const* aurEff)
233 {
234 PreventDefaultAction();
235 if (aurEff->GetAuraType() == SPELL_AURA_MOD_STAT && (aurEff->GetMiscValue() == STAT_STAMINA || aurEff->GetMiscValue() == STAT_INTELLECT))
236 {
237 int32 currentAmount = aurEff->GetAmount();
238 int32 newAmount = GetEffect(aurEff->GetEffIndex())->CalculateAmount(GetCaster());
239 if (newAmount != currentAmount)
240 {
241 if (aurEff->GetMiscValue() == STAT_STAMINA)
242 {
243 uint32 actStat = GetUnitOwner()->GetHealth();
244 GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
245 GetUnitOwner()->SetHealth(std::min<uint32>(GetUnitOwner()->GetMaxHealth(), actStat));
246 }
247 else
248 {
249 uint32 actStat = GetUnitOwner()->GetPower(POWER_MANA);
250 GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
251 GetUnitOwner()->SetPower(POWER_MANA, std::min<uint32>(GetUnitOwner()->GetMaxPower(POWER_MANA), actStat));
252 }
253 }
254 }
255 else
256 GetEffect(aurEff->GetEffIndex())->RecalculateAmount();
257 }
258
259 void Register() override
260 {
261 if (m_scriptSpellId != 34902)
262 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_hun_generic_scaling::CalculateResistanceAmount, EFFECT_ALL, SPELL_AURA_MOD_RESISTANCE);
263 else
264 {
265 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_hun_generic_scaling::CalculateStatAmount, EFFECT_ALL, SPELL_AURA_MOD_STAT);
266 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_hun_generic_scaling::CalculateAPAmount, EFFECT_ALL, SPELL_AURA_MOD_ATTACK_POWER);
267 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_hun_generic_scaling::CalculateSPAmount, EFFECT_ALL, SPELL_AURA_MOD_DAMAGE_DONE);
268 }
269
270 DoEffectCalcPeriodic += AuraEffectCalcPeriodicFn(spell_hun_generic_scaling::CalcPeriodic, EFFECT_ALL, SPELL_AURA_ANY);
271 OnEffectPeriodic += AuraEffectPeriodicFn(spell_hun_generic_scaling::HandlePeriodic, EFFECT_ALL, SPELL_AURA_ANY);
272 }

419// 53209 Chimera Shot
420class spell_hun_chimera_shot : public SpellScript
421{
422 PrepareSpellScript(spell_hun_chimera_shot);
423
424 bool Validate(SpellInfo const* /*spellInfo*/) override
425 {
426 return ValidateSpellInfo({ SPELL_HUNTER_CHIMERA_SHOT_SERPENT, SPELL_HUNTER_CHIMERA_SHOT_VIPER, SPELL_HUNTER_CHIMERA_SHOT_SCORPID });
427 }
428
429 void HandleScriptEffect(SpellEffIndex /*effIndex*/)
430 {
431 Unit* caster = GetCaster();
432 if (Unit* unitTarget = GetHitUnit())
433 {
434 uint32 spellId = 0;
435 int32 basePoint = 0;
436 Unit::AuraApplicationMap& Auras = unitTarget->GetAppliedAuras();
437 for (Unit::AuraApplicationMap::iterator i = Auras.begin(); i != Auras.end(); ++i)
438 {
439 Aura* aura = i->second->GetBase();
440 if (aura->GetCasterGUID() != caster->GetGUID())
441 continue;
442
443 // Search only Serpent Sting, Viper Sting, Scorpid Sting auras
444 flag96 familyFlag = aura->GetSpellInfo()->SpellFamilyFlags;
445 if (!(familyFlag[1] & 0x00000080 || familyFlag[0] & 0x0000C000))
446 continue;
447 if (AuraEffect* aurEff = aura->GetEffect(0))
448 {
449 // Serpent Sting - Instantly deals 40% of the damage done by your Serpent Sting.
450 if (familyFlag[0] & 0x4000)
451 {
452 int32 TickCount = aurEff->GetTotalTicks();
453 spellId = SPELL_HUNTER_CHIMERA_SHOT_SERPENT;
454 basePoint = aurEff->GetAmount();
455 ApplyPct(basePoint, TickCount * 40);
456 basePoint = unitTarget->SpellDamageBonusTaken(caster, aura->GetSpellInfo(), basePoint, DOT, aura->GetStackAmount());
457 }
458 // Viper Sting - Instantly restores mana to you equal to 60% of the total amount drained by your Viper Sting.
459 else if (familyFlag[1] & 0x00000080)
460 {
461 int32 TickCount = aura->GetEffect(0)->GetTotalTicks();
462 spellId = SPELL_HUNTER_CHIMERA_SHOT_VIPER;
463
464 // Amount of one aura tick
465 basePoint = int32(CalculatePct(unitTarget->GetMaxPower(POWER_MANA), aurEff->GetAmount()));
466 int32 casterBasePoint = aurEff->GetAmount() * unitTarget->GetMaxPower(POWER_MANA) / 50; // TODO: Caster uses unitTarget?
467 if (basePoint > casterBasePoint)
468 basePoint = casterBasePoint;
469 ApplyPct(basePoint, TickCount * 60);
470 }
471 // Scorpid Sting - Attempts to Disarm the target for 10 sec. This effect cannot occur more than once per 1 minute.
472 else if (familyFlag[0] & 0x00008000)
473 {
474 if (caster->ToPlayer()) // Scorpid Sting - Add 1 minute cooldown
475 {
476 if (caster->ToPlayer()->HasSpellCooldown(SPELL_HUNTER_CHIMERA_SHOT_SCORPID))
477 break;
478
479 caster->ToPlayer()->AddSpellCooldown(SPELL_HUNTER_CHIMERA_SHOT_SCORPID, 0, 60000);
480 }
481
482 spellId = SPELL_HUNTER_CHIMERA_SHOT_SCORPID;
483 }
484
485 // Refresh aura duration
486 aura->RefreshDuration();
487 aurEff->ChangeAmount(aurEff->CalculateAmount(caster), false);
488 }
489 break;
490 }
491
492 if (spellId)
493 caster->CastCustomSpell(unitTarget, spellId, &basePoint, 0, 0, true);
494 }
495 }
496
497 void Register() override
498 {
499 OnEffectHitTarget += SpellEffectFn(spell_hun_chimera_shot::HandleScriptEffect, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
500 }
