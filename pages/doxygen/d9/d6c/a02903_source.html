240 void UpdateAI(uint32 diff) override
241 {
242 if (timerSpawnPlate)
243 {
244 timerSpawnPlate += diff;
245 if (timerSpawnPlate >= 1000)
246 {
247 if (Vehicle* table = me->GetVehicle())
248 if (Unit* plateHolder = table->GetPassenger(6))
249 {
250 SeatMap::const_iterator itr = table->GetSeatIteratorForPassenger(me);
251 if (itr == table->Seats.end())
252 return;
253
254 uint8 vehicleSeatId = itr->first;
255 Creature* plate = me->SummonCreature(NPC_STRUDY_PLATE, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0.0f);
256 if (!plate)
257 return;
258
259 plateGUID = plate->GetGUID();
260 plate->EnterVehicle(plateHolder, vehicleSeatId);
261 timerSpawnPlate = 0;
262 timerRotateChair = 1;
263 }
264 }
265 }
266 if (timerRotateChair)
267 {
268 timerRotateChair += diff;
269 if (timerRotateChair >= 1000)
270 {
271 if (Creature* plate = GetPlate())
272 me->SetFacingToObject(plate);
273
274 timerRotateChair = 0;
275 }
276 }
277 }

467 player->CastSpell(player, spellId, true);

471 {
472 switch (spellId)
473 {
474 case SPELL_PLAYER_TURKEY:
475 player->CastSpell(player, SPELL_WELL_FED_TURKEY, true);

477 case SPELL_PLAYER_STUFFING:
478 player->CastSpell(player, SPELL_WELL_FED_STUFFING, true);
479 break;
480 case SPELL_PLAYER_PIE:
481 player->CastSpell(player, SPELL_WELL_FED_PIE, true);
482 break;
483 case SPELL_PLAYER_CRANBERRY:
484 player->CastSpell(player, SPELL_WELL_FED_CRANBERRY, true);
485 break;
486 case SPELL_PLAYER_SWEET_POTATOES:
487 player->CastSpell(player, SPELL_WELL_FED_SWEET_POTATOES, true);
488 break;
489 }
490
491 uint8 count = 0;
492 Unit::AuraEffectList const& dummyAuras = player->GetAuraEffectsByType(SPELL_AURA_DUMMY);

497 ++count;
498 }
499

502 player->CastSpell(player, SPELL_SPIRIT_OF_SHARING, true);
503 }
504 }
505 }
506 }
