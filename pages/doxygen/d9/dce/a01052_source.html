35float ThreatCalcHelper::calcThreat(Unit* hatedUnit, float threat, SpellSchoolMask schoolMask, SpellInfo const* threatSpell)
36{
37 if (threatSpell)
38 {
39 if (SpellThreatEntry const* threatEntry = sSpellMgr->GetSpellThreatEntry(threatSpell->Id))
40 if (threatEntry->pctMod != 1.0f)
41 threat *= threatEntry->pctMod;
42
43 // Energize is not affected by Mods
44 for (uint8 i = 0; i < MAX_SPELL_EFFECTS; i++)
45 if (threatSpell->Effects[i].Effect == SPELL_EFFECT_ENERGIZE || threatSpell->Effects[i].ApplyAuraName == SPELL_AURA_PERIODIC_ENERGIZE)
46 return threat;
47
48 if (Player* modOwner = hatedUnit->GetSpellModOwner())
49 modOwner->ApplySpellMod(threatSpell->Id, SPELLMOD_THREAT, threat);
50 }
51
52 return hatedUnit->ApplyTotalThreatModifier(threat, schoolMask);
53}

55bool ThreatCalcHelper::isValidProcess(Unit* hatedUnit, Unit* hatingUnit, SpellInfo const* threatSpell)
56{
57 //function deals with adding threat and adding players and pets into ThreatList
58 //mobs, NPCs, guards have ThreatList and HateOfflineList
59 //players and pets have only InHateListOf
60 //HateOfflineList is used co contain unattackable victims (in-flight, in-water, GM etc.)
61
62 if (!hatedUnit || !hatingUnit)
63 return false;
64
65 // not to self
66 if (hatedUnit == hatingUnit)
67 return false;
68
69 // not to GM
70 if (hatedUnit->GetTypeId() == TYPEID_PLAYER && hatedUnit->ToPlayer()->IsGameMaster())
71 return false;
72
73 // not to dead and not for dead
74 if (!hatedUnit->IsAlive() || !hatingUnit->IsAlive())
75 return false;
76
77 // not in same map or phase
78 if (!hatedUnit->IsInMap(hatingUnit) || !hatedUnit->InSamePhase(hatingUnit))
79 return false;
80
81 // spell not causing threat
82 if (threatSpell && threatSpell->HasAttribute(SPELL_ATTR1_NO_THREAT))
83 return false;
84
85 ASSERT(hatingUnit->GetTypeId() == TYPEID_UNIT);
86
87 return true;
88}

139void HostileReference::AddThreat(float modThreat)
140{
141 iThreat += modThreat;
142 // the threat is changed. Source and target unit have to be available
143 // if the link was cut before relink it again
144 if (!IsOnline())
145 updateOnlineStatus();
146 if (modThreat != 0.0f)
147 {
148 ThreatRefStatusChangeEvent event(UEV_THREAT_REF_THREAT_CHANGE, this, modThreat);
149 fireStatusChanged(event);
150 }
151
152 if (isValid() && modThreat >= 0.0f)
153 {
154 Unit* target = getTarget();
155 if (target->GetEntry() != NPC_EYE_OF_KILROGG) // Excluded Eye of Kilrogg
156 {
157 Unit* victimOwner = target->GetCharmerOrOwner();
158 if (victimOwner && victimOwner->IsAlive())
159 {
160 GetSource()->AddThreat(victimOwner, 0.0f); // create a threat to the owner of a pet, if the pet attacks
161 }
162 }
163 }

166void HostileReference::addThreatPercent(int32 percent)
167{
168 // Xinef: Do not allow to modify threat by percent if threat is negative (forced to big value < 0 by spells adding temporary threat)
169 // Xinef: When the temporary effect ends, temporary threat is added back which results in huge additional amount of threat
170 if (iThreat <= 0)

172

174 AddPct(tmpThreat, percent);

179// check, if source can reach target and set the status
180
181void HostileReference::updateOnlineStatus()
182{
183 bool online = false;
184
185 if (!isValid())
186 if (Unit* target = ObjectAccessor::GetUnit(*GetSourceUnit(), getUnitGuid()))
187 link(target, GetSource());
188
189 // only check for online status if
190 // ref is valid
191 // target is no player or not gamemaster
192 // target is not in flight
193 if (isValid()
194 && (getTarget()->GetTypeId() != TYPEID_PLAYER || !getTarget()->ToPlayer()->IsGameMaster())
195 && !getTarget()->IsInFlight()
196 && getTarget()->IsInMap(GetSourceUnit())
197 && getTarget()->InSamePhase(GetSourceUnit())
198 )
199 {
200 Creature* creature = GetSourceUnit()->ToCreature();
201 online = getTarget()->isInAccessiblePlaceFor(creature);
202 if (!online)
203 {
204 if (creature->IsWithinCombatRange(getTarget(), creature->m_CombatDistance))
205 online = true; // not accessible but stays online
206 }
207 }
208
209 setOnlineOfflineState(online);
210}

318HostileReference* ThreatContainer::SelectNextVictim(Creature* attacker, HostileReference* currentVictim) const
319{
320 // pussywizard: pretty much remade this whole function
321
322 HostileReference* currentRef = nullptr;
323 bool found = false;
324 bool noPriorityTargetFound = false;
325
326 // pussywizard: currentVictim is needed to compare if threat was exceeded by 10%/30% for melee/range targets (only then switching current target)
327 if (currentVictim)
328 {
329 Unit* cvUnit = currentVictim->getTarget();
330 if (!attacker->CanCreatureAttack(cvUnit)) // pussywizard: if currentVictim is not valid => don't compare the threat with it, just take the highest threat valid target
331 currentVictim = nullptr;
332 else if (cvUnit->IsImmunedToDamageOrSchool(attacker->GetMeleeDamageSchoolMask()) || cvUnit->HasNegativeAuraWithInterruptFlag(AURA_INTERRUPT_FLAG_TAKE_DAMAGE)) // pussywizard: no 10%/30% if currentVictim is immune to damage or has auras breakable by damage
333 currentVictim = nullptr;
334 }
335
336 ThreatContainer::StorageType::const_iterator lastRef = iThreatList.end();
337 --lastRef;
338
339 // pussywizard: iterate from highest to lowest threat
340 for (ThreatContainer::StorageType::const_iterator iter = iThreatList.begin(); iter != iThreatList.end() && !found;)
341 {
342 currentRef = (*iter);
343
344 Unit* target = currentRef->getTarget();
345 ASSERT(target); // if the ref has status online the target must be there !
346
347 // pussywizard: don't go to threat comparison if this ref is immune to damage or has aura breakable on damage (second choice target)
348 // pussywizard: if this is the last entry on the threat list, then all targets are second choice, set bool to true and loop threat list again, ignoring this section
349 if (!noPriorityTargetFound && (target->IsImmunedToDamageOrSchool(attacker->GetMeleeDamageSchoolMask()) || target->HasNegativeAuraWithInterruptFlag(AURA_INTERRUPT_FLAG_TAKE_DAMAGE) || target->HasAuraTypeWithCaster(SPELL_AURA_IGNORED, attacker->GetGUID())))
350 {
351 if (iter != lastRef)
352 {
353 ++iter;
354 continue;
355 }
356 else
357 {
358 noPriorityTargetFound = true;
359 iter = iThreatList.begin();
360 continue;
361 }
362 }
363
364 // pussywizard: skip not valid targets
365 if (attacker->CanCreatureAttack(target))
366 {
367 if (currentVictim) // pussywizard: if not nullptr then target must have 10%/30% more threat
368 {
369 if (currentVictim == currentRef) // pussywizard: nothing found previously was good and enough, currentRef passed all necessary tests, so end now
370 {
371 found = true;
372 break;
373 }
374
375 // pussywizard: implement 110% threat rule for targets in melee range and 130% rule for targets in ranged distances
376 if (currentRef->GetThreat() > 1.3f * currentVictim->GetThreat()) // pussywizard: enough in all cases, end
377 {
378 found = true;
379 break;
380 }
381 else if (currentRef->GetThreat() > 1.1f * currentVictim->GetThreat()) // pussywizard: enought only if target in melee range
382 {
383 if (attacker->IsWithinMeleeRange(target))
384 {
385 found = true;
386 break;
387 }
388 }
389 else // pussywizard: nothing found previously was good and enough, this and next entries on the list have less than 110% threat, and currentVictim is present and valid as checked before the loop (otherwise it's nullptr), so end now
390 {
391 currentRef = currentVictim;
392 found = true;
393 break;
394 }
395 }
396 else // pussywizard: no currentVictim, first passing all checks is chosen (highest threat, list is sorted)
397 {
398 found = true;
399 break;
400 }
401 }
402 ++iter;
403 }
404 if (!found)
405 currentRef = nullptr;
406
407 return currentRef;

437void ThreatMgr::AddThreat(Unit* victim, float threat, SpellSchoolMask schoolMask, SpellInfo const* threatSpell)
438{
439 if (!ThreatCalcHelper::isValidProcess(victim, iOwner, threatSpell))
440 return;
441
442 threat = ThreatCalcHelper::calcThreat(victim, threat, schoolMask, threatSpell);
443 if (Creature* hatingCreature = iOwner->ToCreature())
444 {
445 if (hatingCreature->IsAIEnabled)
446 {
447 hatingCreature->AI()->CalculateThreat(victim, threat, threatSpell);
448 }
449 }
450
451 DoAddThreat(victim, threat);
452}
453
454void ThreatMgr::DoAddThreat(Unit* victim, float threat)
455{
456 uint32 redirectThreadPct = victim->GetRedirectThreatPercent();
457
458 // must check > 0.0f, otherwise dead loop
459 if (threat > 0.0f && redirectThreadPct)
460 {
461 if (Unit* redirectTarget = victim->GetRedirectThreatTarget())
462 {
463 float redirectThreat = CalculatePct(threat, redirectThreadPct);
464 threat -= redirectThreat;
465 if (ThreatCalcHelper::isValidProcess(redirectTarget, GetOwner()))
466 _addThreat(redirectTarget, redirectThreat);
467 }
468 }
469
470 _addThreat(victim, threat);
471}
472
473void ThreatMgr::_addThreat(Unit* victim, float threat)
474{
475 HostileReference* ref = iThreatContainer.AddThreat(victim, threat);
476 // Ref is not in the online refs, search the offline refs next
477 if (!ref)
478 ref = iThreatOfflineContainer.AddThreat(victim, threat);
479
480 if (!ref) // there was no ref => create a new one
481 {
482 // threat has to be 0 here
483 HostileReference* hostileRef = new HostileReference(victim, this, 0);
484 iThreatContainer.addReference(hostileRef);
485 hostileRef->AddThreat(threat); // now we add the real threat
486 if (victim->GetTypeId() == TYPEID_PLAYER && victim->ToPlayer()->IsGameMaster())
487 hostileRef->setOnlineOfflineState(false); // GM is always offline
488 }
489}

566//============================================================
567// The hated unit is gone, dead or deleted
568// return true, if the event is consumed
569
570void ThreatMgr::processThreatEvent(ThreatRefStatusChangeEvent* threatRefStatusChangeEvent)
571{
572 threatRefStatusChangeEvent->setThreatMgr(this); // now we can set the threat manager
573
574 HostileReference* hostileRef = threatRefStatusChangeEvent->getReference();
575
576 switch (threatRefStatusChangeEvent->getType())
577 {
578 case UEV_THREAT_REF_THREAT_CHANGE:
579 if ((getCurrentVictim() == hostileRef && threatRefStatusChangeEvent->getFValue() < 0.0f) ||
580 (getCurrentVictim() != hostileRef && threatRefStatusChangeEvent->getFValue() > 0.0f))
581 setDirty(true); // the order in the threat list might have changed
582 break;
583 case UEV_THREAT_REF_ONLINE_STATUS:
584 if (!hostileRef->IsOnline())
585 {
586 if (hostileRef == getCurrentVictim())
587 {
588 setCurrentVictim(nullptr);
589 setDirty(true);
590 }
591 if (GetOwner() && GetOwner()->IsInWorld())
592 if (Unit* target = ObjectAccessor::GetUnit(*GetOwner(), hostileRef->getUnitGuid()))
593 if (GetOwner()->IsInMap(target))
594 GetOwner()->SendRemoveFromThreatListOpcode(hostileRef);
595 iThreatContainer.remove(hostileRef);
596 iThreatOfflineContainer.addReference(hostileRef);
597 }
598 else
599 {
600 if (getCurrentVictim() && hostileRef->GetThreat() > (1.1f * getCurrentVictim()->GetThreat()))
601 setDirty(true);
602 iThreatContainer.addReference(hostileRef);
603 iThreatOfflineContainer.remove(hostileRef);
604 }
605 break;
606 case UEV_THREAT_REF_REMOVE_FROM_LIST:
607 if (hostileRef == getCurrentVictim())
608 {
609 setCurrentVictim(nullptr);
610 setDirty(true);
611 }
612 iOwner->SendRemoveFromThreatListOpcode(hostileRef);
613 if (hostileRef->IsOnline())
614 iThreatContainer.remove(hostileRef);
615 else
616 iThreatOfflineContainer.remove(hostileRef);
617 break;
618 }
619}
