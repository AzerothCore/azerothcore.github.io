59 void UpdateAI(uint32 diff) override
60 {
61 if (!UpdateVictim())
62 return;
63
64 events.Update(diff);
65
66 if (me->HasUnitState(UNIT_STATE_CASTING))
67 return;
68
69 while (uint32 eventId = events.ExecuteEvent())
70 {
71 switch (eventId)
72 {
73 case EVENT_SHADOWFLAME:
74 DoCastVictim(SPELL_SHADOWFLAME);
75 events.ScheduleEvent(EVENT_SHADOWFLAME, urand(15000, 25000));
76 break;
77 case EVENT_WINGBUFFET:
78 DoCastVictim(SPELL_WINGBUFFET);
79 if (DoGetThreat(me->GetVictim()))
80 DoModifyThreatPercent(me->GetVictim(), -75);
81 events.ScheduleEvent(EVENT_WINGBUFFET, 30000);
82 break;
83 case EVENT_FRENZY:
84 Talk(EMOTE_FRENZY);
85 DoCast(me, SPELL_FRENZY);
86 events.ScheduleEvent(EVENT_FRENZY, 8000, 10000);
87 break;
88 }
89
90 if (me->HasUnitState(UNIT_STATE_CASTING))
91 return;
92 }
93
94 DoMeleeAttackIfReady();
95 }
96 };
