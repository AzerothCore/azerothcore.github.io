106 void UpdateAI(uint32 diff) override
107 {
108 if (!UpdateVictim())
109 return;
110
111 events.Update(diff);
112 if (me->HasUnitState(UNIT_STATE_CASTING))
113 return;
114
115 switch (events.ExecuteEvent())
116 {
117 case EVENT_MORTAL_WOUND:
118 me->CastSpell(me->GetVictim(), SPELL_MORTAL_WOUND, false);
119 events.ScheduleEvent(EVENT_MORTAL_WOUND, 8000);
120 break;
121 case EVENT_SURGE:
122 Talk(SAY_SURGE);
123 if (Unit* target = SelectTarget(SelectTargetMethod::MinDistance, 0))
124 me->CastSpell(target, SPELL_SURGE, false);
125 events.ScheduleEvent(EVENT_SURGE, 11000);
126 break;
127 case EVENT_RETALIATION:
128 if (me->HealthBelowPct(20))
129 {
130 me->CastSpell(me, SPELL_RETALIATION, false);
131 events.ScheduleEvent(EVENT_RETALIATION, 30000);
132 }
133 else
134 events.ScheduleEvent(EVENT_RETALIATION, 500);
135 break;
136 case EVENT_CHECK_HEALTH:
137 if (me->HealthBelowPct(50))
138 {
139 Talk(SAY_HEAL);
140 std::list<Creature*> clist;
141 me->GetCreaturesWithEntryInRange(clist, 100.0f, NPC_HELLFIRE_WATCHER);
142 for (std::list<Creature*>::const_iterator itr = clist.begin(); itr != clist.end(); ++itr)
143 (*itr)->AI()->SetData(NPC_HELLFIRE_WATCHER, 0);
144 break;
145 }
146 events.ScheduleEvent(EVENT_CHECK_HEALTH, 500);
147 break;
148 }
149
150 DoMeleeAttackIfReady();
151 }
152

154 bool _taunted;
155 };
