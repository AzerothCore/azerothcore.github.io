405 void UpdateAI(uint32 diff) override
406 {
407 if (!UpdateVictim())
408 {
409 EnterEvadeMode(EVADE_REASON_OTHER);
410 return;
411 }
412
413 events.Update(diff);
414 if (me->HasUnitState(UNIT_STATE_CASTING))
415 return;
416
417 switch (events.ExecuteEvent())
418 {
419 case EVENT_PREPARE_BREATH:
420 breathReady = true;
421 break;
422 case EVENT_STONE_SHOUT:
423 if (_left || _right)
424 {
425 return;
426 }
427
428 me->CastSpell(me->GetVictim(), SPELL_STONE_SHOUT, false);
429 events.ScheduleEvent(EVENT_STONE_SHOUT, 2000);
430 break;
431 case EVENT_SMASH:
432 if (_left && _right)
433 me->CastSpell(me->GetVictim(), SPELL_OVERHEAD_SMASH, false);
434 else if (_left || _right)
435 me->CastSpell(me->GetVictim(), SPELL_ONEARMED_OVERHEAD_SMASH, false);
436
437 events.DelayEvents(1000);
438 events.ScheduleEvent(EVENT_SMASH, 14000);
439 return;
440 case EVENT_SWEEP:
441 if (_left)
442 {
443 if (Creature* cr = me->FindNearestCreature(NPC_SWEEP_TRIGGER, 300))
444 cr->CastSpell(cr, SPELL_ARM_SWEEP, false);
445
446 if (urand(0, 1))
447 Talk(SAY_SHOCKWAVE);
448 }
449
450 events.DelayEvents(1000);
451 events.ScheduleEvent(EVENT_SWEEP, 17000);
452 return;
453 case EVENT_GRIP:
454 events.ScheduleEvent(EVENT_GRIP, 25000);
455 if (!_right)
456 break;
457
458 me->CastSpell(me, SPELL_STONE_GRIP, false);
459 Talk(SAY_GRAB_PLAYER);
460 Talk(EMOTE_STONE_GRIP);
461 return;
462 case EVENT_FOCUSED_EYEBEAM:
463 {
464 events.ScheduleEvent(EVENT_FOCUSED_EYEBEAM, 20000);
465
466 if ((eyebeamTarget = SelectTarget(SelectTargetMethod::MinDistance, 0, 0, true)))
467 {
468 me->CastSpell(eyebeamTarget, SPELL_FOCUSED_EYEBEAM_SUMMON, false);
469 }
470
471 Talk(EMOTE_EYES);
472 return;
473 }
474 case EVENT_RESTORE_ARM_LEFT:
475 // shouldn't happen
476 AttachLeftArm();
477 return;
478 case EVENT_RESTORE_ARM_RIGHT:
479 // shouldn't happen
480 AttachRightArm();
481 return;
482 }
483
484 //Make sure our attack is ready and we aren't currently casting before checking distance
485 if (me->isAttackReady() && me->GetVictim()) // victim could die by a spell (IMPORTANT!!!) and kologarn entered evade mode
486 {
487 //If we are within range melee the target
488 if (me->IsWithinMeleeRange(me->GetVictim()))
489 {
490 me->AttackerStateUpdate(me->GetVictim());
491 me->resetAttackTimer();
492 return;
493 }
494 else if (Unit* tgt = me->SelectNearbyTarget())
495 {
496 me->AttackerStateUpdate(tgt);
497 me->resetAttackTimer();
498 return;
499 }
500
501 if (breathReady)
502 me->CastSpell(me->GetVictim(), SPELL_PETRIFYING_BREATH, false);
503 me->resetAttackTimer();
504 }
505 }
