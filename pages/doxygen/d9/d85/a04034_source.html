83 void UpdateAI(uint32 diff) override
84 {
85 if (!UpdateVictim())
86 return;
87
88 events.Update(diff);
89 if (me->HasUnitState(UNIT_STATE_CASTING))
90 return;
91
92 switch (events.ExecuteEvent())
93 {
94 case EVENT_ARCANE_RESONANCE:
95 if (roll_chance_i(50))
96 Talk(SAY_ARCANE_RESONANCE);
97 me->CastSpell(me->GetVictim(), SPELL_ARCANE_RESONANCE, false);
98 events.ScheduleEvent(EVENT_ARCANE_RESONANCE, 27000);
99 break;
100 case EVENT_ARCANE_DEVASTATION:
101 if (roll_chance_i(50))
102 Talk(SAY_ARCANE_DEVASTATION);
103 me->CastSpell(me->GetVictim(), SPELL_ARCANE_DEVASTATION, false);
104 events.ScheduleEvent(EVENT_ARCANE_DEVASTATION, 17000);
105 break;
106 case EVENT_HEALTH_CHECK:
107 if (me->HealthBelowPct(50))
108 {
109 Talk(EMOTE_SUMMON);
110 Talk(SAY_SUMMON);
111 me->CastSpell(me, SPELL_SUMMON_REINFORCEMENTS, true);
112 break;
113 }
114 events.ScheduleEvent(EVENT_HEALTH_CHECK, 500);
115 break;
116 }
117
118 DoMeleeAttackIfReady();
119 }
120 };
