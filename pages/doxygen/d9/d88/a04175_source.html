165 amount = -1;
166 }
167
168 void Absorb(AuraEffect* /*aurEff*/, DamageInfo& dmgInfo, uint32& absorbAmount)
169 {
170 Player* target = GetTarget()->ToPlayer();
171 if (dmgInfo.GetDamage() < target->GetHealth() || target->HasSpellCooldown(SPELL_ROGUE_CHEAT_DEATH_COOLDOWN) || !roll_chance_i(absorbChance))
172 return;
173
174 target->CastSpell(target, SPELL_ROGUE_CHEAT_DEATH_COOLDOWN, true);
175 target->CastSpell(target, SPELL_ROGUE_CHEATING_DEATH, true);
176 target->AddSpellCooldown(SPELL_ROGUE_CHEAT_DEATH_COOLDOWN, 0, MINUTE * IN_MILLISECONDS);
177
178 uint32 health10 = target->CountPctFromMaxHealth(10);
179
180 // hp > 10% - absorb hp till 10%
181 if (target->GetHealth() > health10)
182 absorbAmount = dmgInfo.GetDamage() - target->GetHealth() + health10;
183 // hp lower than 10% - absorb everything
184 else
185 absorbAmount = dmgInfo.GetDamage();
186 }

195// -2818 - Deadly Poison
196class spell_rog_deadly_poison : public SpellScript
197{
198 PrepareSpellScript(spell_rog_deadly_poison);
199
200 bool Load() override
201 {
202 _stackAmount = 0;
203 // at this point CastItem must already be initialized
204 return GetCaster()->GetTypeId() == TYPEID_PLAYER && GetCastItem();
205 }
206
207 void HandleBeforeHit(SpellMissInfo missInfo)
208 {
209 if (missInfo != SPELL_MISS_NONE)
210 {
211 return;
212 }
213
214 if (Unit* target = GetHitUnit())
215 // Deadly Poison
216 if (AuraEffect const* aurEff = target->GetAuraEffect(SPELL_AURA_PERIODIC_DAMAGE, SPELLFAMILY_ROGUE, 0x10000, 0x80000, 0, GetCaster()->GetGUID()))
217 _stackAmount = aurEff->GetBase()->GetStackAmount();
218 }
219
220 void HandleAfterHit()
221 {
222 if (_stackAmount < 5)
223 return;
224
225 Player* player = GetCaster()->ToPlayer();
226
227 if (Unit* target = GetHitUnit())
228 {
229 Item* item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_MAINHAND);
230
231 if (item == GetCastItem())
232 item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_OFFHAND);
233
234 if (!item)
235 return;
236
237 // item combat enchantments
238 for (uint8 slot = 0; slot < MAX_ENCHANTMENT_SLOT; ++slot)
239 {
240 SpellItemEnchantmentEntry const* enchant = sSpellItemEnchantmentStore.LookupEntry(item->GetEnchantmentId(EnchantmentSlot(slot)));
241 if (!enchant)
242 continue;
243
244 for (uint8 s = 0; s < 3; ++s)
245 {
246 if (enchant->type[s] != ITEM_ENCHANTMENT_TYPE_COMBAT_SPELL)
247 continue;
248
249 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(enchant->spellid[s]);
250 if (!spellInfo)
251 {
252 LOG_ERROR("misc", "Player::CastItemCombatSpell Enchant {}, player (Name: {}, {}) cast unknown spell {}",
253 enchant->ID, player->GetName(), player->GetGUID().ToString(), enchant->spellid[s]);
254 continue;
255 }
256
257 // Proc only rogue poisons
258 if (spellInfo->SpellFamilyName != SPELLFAMILY_ROGUE || spellInfo->Dispel != DISPEL_POISON)
259 continue;
260
261 // Do not reproc deadly
262 if (spellInfo->SpellFamilyFlags.IsEqual(0x10000, 0x80000, 0))
263 continue;
264
265 if (spellInfo->IsPositive())
266 player->CastSpell(player, enchant->spellid[s], true, item);
267 else
268 player->CastSpell(target, enchant->spellid[s], true, item);
269 }
270 }
271 }
272 }
273
274 void Register() override
275 {
276 BeforeHit += BeforeSpellHitFn(spell_rog_deadly_poison::HandleBeforeHit);
277 AfterHit += SpellHitFn(spell_rog_deadly_poison::HandleAfterHit);
278 }
279
280 uint8 _stackAmount;
281};
