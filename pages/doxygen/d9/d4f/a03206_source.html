187 void DamageTaken(Unit* /*attacker*/, uint32& damage, DamageEffectType /*damagetype*/, SpellSchoolMask /*damageSchoolMask*/) override
188 {
189 // Do not perform insanity recast if boss is casting Insanity already
190 if (me->FindCurrentSpellBySpellId(SPELL_INSANITY))
191 {

194
195 // First insanity
196 if (insanityTimes == 0 && me->HealthBelowPctDamaged(66, damage))
197 {
198 DoCastSelf(SPELL_INSANITY, false);
199 ++insanityTimes;
200 }
201 // Second insanity
202 else if (insanityTimes == 1 && me->HealthBelowPctDamaged(33, damage))

210 void UpdateAI(uint32 diff) override
211 {
212 //Return since we have no target
213 if (!UpdateVictim())
214 {
215 return;
216 }
217
218 if (insanityPhase)
219 {
220 if (!CheckPhaseMinions())
221 {
222 return;
223 }
224
225 insanityPhase = false;
226 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
227 me->SetControlled(false, UNIT_STATE_STUNNED);
228 me->RemoveAurasDueToSpell(INSANITY_VISUAL);
229 }
230
231 events.Update(diff);
232 if (me->HasUnitState(UNIT_STATE_CASTING))
233 {
234 return;
235 }
236
237 while (uint32 const eventId = events.ExecuteEvent())
238 {
239 switch (eventId)
240 {
241 case EVENT_HERALD_MIND_FLAY:
242 {
243 DoCastVictim(SPELL_MIND_FLAY, false);
244 events.RepeatEvent(20000);
245 break;
246 }
247 case EVENT_HERALD_SHADOW:
248 {
249 DoCastVictim(SPELL_SHADOW_BOLT_VOLLEY, false);
250 events.RepeatEvent(5000);
251 break;
252 }
253 case EVENT_HERALD_SHIVER:
254 {
255 if (Unit* pTarget = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true))
256 {
257 DoCast(pTarget, SPELL_SHIVER, false);
258 }
259
260 events.RepeatEvent(15000);
261 break;
262 }
263 }
264
265 if (me->HasUnitState(UNIT_STATE_CASTING))
266 {
267 return;
268 }
269 }
270
271 DoMeleeAttackIfReady();
272 }

278 uint32 GetPlrInsanityAuraId(uint32 phaseMask) const
279 {
280 switch (phaseMask)
281 {
282 case 16:
283 return SPELL_INSANITY_PHASING_1;
284 case 32:
285 return SPELL_INSANITY_PHASING_2;
286 case 64:
287 return SPELL_INSANITY_PHASING_3;
288 case 128:
289 return SPELL_INSANITY_PHASING_4;
290 case 256:
291 return SPELL_INSANITY_PHASING_5;
292 }
293
294 return 0;
295 }
296
297 void ResetPlayersPhaseMask()
298 {
299 Map::PlayerList const& players = me->GetMap()->GetPlayers();
300 for (auto const& i : players)
301 {
302 if (Player* pPlayer = i.GetSource())
303 {
304 if (uint32 const insanityAura = GetPlrInsanityAuraId(pPlayer->GetPhaseMask()))
305 {
306 pPlayer->RemoveAurasDueToSpell(insanityAura);
307 }
308 }
309 }
310 }
311
312 bool CheckPhaseMinions()
313 {
314 summons.RemoveNotExisting();
315 if (summons.empty())
316 {
317 ResetPlayersPhaseMask();
318 return true;
319 }
320
321 uint32 phase = 1;
322 for (ObjectGuid const& summonGUID : summons)
323 {
324 if (Creature* summon = ObjectAccessor::GetCreature(*me, summonGUID))
325 {
326 phase |= summon->GetPhaseMask();
327 }
328 }
329
330 Map::PlayerList const& players = me->GetMap()->GetPlayers();
331 for (auto const& i : players)
332 {
333 Player* pPlayer = i.GetSource();
334 if (pPlayer && !(pPlayer->GetPhaseMask() & phase))
335 {
336 pPlayer->RemoveAurasDueToSpell(GetPlrInsanityAuraId(pPlayer->GetPhaseMask()));
337 }
338 }
339
340 return false;
341 }

351 void HandleDummyEffect(std::list<WorldObject*>& targets)
352 {
353 Unit* caster = GetCaster();
354 if (!caster)
355 {
356 targets.clear();
357 return;
358 }
359
360 if (!targets.empty())
361 {
362 targets.remove_if([this](WorldObject* targetObj) -> bool
363 {
364 return !targetObj || targetObj->GetTypeId() != TYPEID_PLAYER || !targetObj->ToPlayer()->IsInCombatWith(GetCaster()) ||
365 targetObj->GetDistance(GetCaster()) >= (MAX_VISIBILITY_DISTANCE * 2);
366 });
367 }
368
369 if (targets.empty())
370 {
371 return;
372 }
373
374 // Start channel visual and set self as unnattackable
375 caster->ToCreature()->AI()->Talk(SAY_INSANITY);
376 caster->CastSpell(caster, SPELL_WHISPER_INSANITY, true);
377 caster->RemoveAllAuras();
378 caster->CastSpell(caster, INSANITY_VISUAL, true);
379 caster->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
380 caster->SetControlled(true, UNIT_STATE_STUNNED);
381
382 // Handle phase effect
383 uint32 insanityCounter = 0;
384 std::list<WorldObject*>::const_iterator itr = targets.begin();
385 while (itr != targets.end() && insanityCounter < MAX_INSANITY_TARGETS)
386 {
387 WorldObject* targetObj = *itr;
388 if (!targetObj)
389 {
390 continue;
391 }
392
393 Player* plrTarget = targetObj->ToPlayer();
394 // This should never happen, spell has attribute SPELL_ATTR3_ONLY_TARGET_PLAYERS
395 if (!plrTarget)
396 {
397 continue;
398 }
399
400 // phase mask
401 plrTarget->CastSpell(plrTarget, InsanitySpells.at(insanityCounter), true);
402
403 // Summon clone
404 for (std::list<WorldObject*>::const_iterator itr2 = targets.begin(); itr2 != targets.end(); ++itr2)
405 {
406 // Should not make clone of current player target
407 Player const* plrClone = *itr2 ? (*itr2)->ToPlayer() : nullptr;
408 if (!plrClone || plrClone == plrTarget)
409 {
410 continue;
411 }
412
413 if (Unit* summon = caster->SummonCreature(NPC_TWISTED_VISAGE, plrClone->GetPosition(), TEMPSUMMON_CORPSE_DESPAWN, 0))
414 {
415 summon->AddThreat(plrTarget, 0.0f);
416 summon->SetInCombatWith(plrTarget);
417 plrTarget->SetInCombatWith(summon);
418
419 plrTarget->CastSpell(summon, SPELL_CLONE_PLAYER, true);
420 summon->SetPhaseMask(1 | (1 << (4 + insanityCounter)), true);
421 summon->SetUInt32Value(UNIT_FIELD_MINDAMAGE, plrClone->GetUInt32Value(UNIT_FIELD_MINDAMAGE));
422 summon->SetUInt32Value(UNIT_FIELD_MAXDAMAGE, plrClone->GetUInt32Value(UNIT_FIELD_MAXDAMAGE));
423 }
424 }
425
426 ++insanityCounter;
427 ++itr;
428 }
429 }
