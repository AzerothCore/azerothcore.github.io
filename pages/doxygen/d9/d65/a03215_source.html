175 std::string GetSaveData() override
176 {
177 OUT_SAVE_INST_DATA;
178
179 std::ostringstream saveStream;
180 // Encounter states
181 saveStream << "A K " << GetBossSaveData();
182

189
190 void Load(const char* in) override
191 {
192 if (!in)
193 {
194 OUT_LOAD_INST_DATA_FAIL;
195 return;
196 }
197
198 OUT_LOAD_INST_DATA(in);
199
200 char dataHead1, dataHead2;
201
202 std::istringstream loadStream(in);
203 loadStream >> dataHead1 >> dataHead2;
204
205 if (dataHead1 == 'A' && dataHead2 == 'K')
206 {
207 // Encounter states
208 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
209 {
210 uint32 tmpState;
211 loadStream >> tmpState;
212 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
213 {
214 tmpState = NOT_STARTED;
215 }
216
217 SetBossState(i, EncounterState(tmpState));
218 }
219
220 // Extra data
221 loadStream >> teldaramSpheres[0] >> teldaramSpheres[1];
222 }
223 else
224 {
225 OUT_LOAD_INST_DATA_FAIL;
226 return;
227 }
228
229 canSaveBossStates = true;
230 OUT_LOAD_INST_DATA_COMPLETE;
231 }
