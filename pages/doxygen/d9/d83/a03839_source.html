48class boss_murmur : public CreatureScript
49{
50public:
51 boss_murmur() : CreatureScript("boss_murmur") { }
52
53 CreatureAI* GetAI(Creature* creature) const override
54 {
55 return GetShadowLabyrinthAI<boss_murmurAI>(creature);

58 struct boss_murmurAI : public ScriptedAI
59 {
60 boss_murmurAI(Creature* creature) : ScriptedAI(creature)
61 {
62 SetCombatMovement(false);
63 instance = creature->GetInstanceScript();
64 }
65
66 InstanceScript* instance;
67 EventMap events;
68
69 void Reset() override
70 {
71 events.Reset();
72 me->SetHealth(me->CountPctFromMaxHealth(40));
73 me->ResetPlayerDamageReq();
74
75 if (instance)
76 instance->SetData(DATA_MURMUREVENT, NOT_STARTED);
77 }
78
79 void EnterCombat(Unit* /*who*/) override
80 {
81 events.ScheduleEvent(EVENT_SPELL_SONIC_BOOM, 30000);
82 events.ScheduleEvent(EVENT_SPELL_MURMURS_TOUCH, urand(8000, 20000));
83 events.ScheduleEvent(EVENT_SPELL_RESONANCE, 5000);
84 events.ScheduleEvent(EVENT_SPELL_MAGNETIC, urand(15000, 30000));
85 if (IsHeroic())
86 {
87 events.ScheduleEvent(EVENT_SPELL_THUNDERING, 15000);
88 events.ScheduleEvent(EVENT_SPELL_SONIC_SHOCK, 10000);
89 }
90
91 if (instance)
92 instance->SetData(DATA_MURMUREVENT, IN_PROGRESS);
93 }
94
95 void JustDied(Unit*) override
96 {
97 if (instance)
98 instance->SetData(DATA_MURMUREVENT, DONE);
99 }
100
101 void UpdateAI(uint32 diff) override
102 {
103 if (!UpdateVictim() || me->HasUnitState(UNIT_STATE_CASTING))
104 return;
105
106 events.Update(diff);
107 switch (events.ExecuteEvent())
108 {
109 case EVENT_SPELL_SONIC_BOOM:
110 Talk(EMOTE_SONIC_BOOM);
111 me->CastSpell(me, DUNGEON_MODE(SPELL_SONIC_BOOM_CAST_N, SPELL_SONIC_BOOM_CAST_H), false);
112 events.RepeatEvent(28500);
113 events.DelayEvents(1500);
114 events.ScheduleEvent(EVENT_SPELL_SONIC_BOOM_EFFECT, 0);
115 return;
116 case EVENT_SPELL_SONIC_BOOM_EFFECT:
117 me->CastSpell(me, DUNGEON_MODE(SPELL_SONIC_BOOM_EFFECT_N, SPELL_SONIC_BOOM_EFFECT_H), true);
118 break;
119 case EVENT_SPELL_MURMURS_TOUCH:
120 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 80.0f, true))
121 me->CastSpell(target, DUNGEON_MODE(SPELL_MURMURS_TOUCH_N, SPELL_MURMURS_TOUCH_H), false);
122 events.RepeatEvent(urand(25000, 35000));
123 break;
124 case EVENT_SPELL_RESONANCE:
125 if (!me->IsWithinMeleeRange(me->GetVictim()))
126 me->CastSpell(me, SPELL_RESONANCE, false);
127 events.RepeatEvent(5000);
128 break;
129 case EVENT_SPELL_MAGNETIC:
130 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 80.0f, true))
131 {
132 me->CastSpell(target, SPELL_MAGNETIC_PULL, false);
133 events.RepeatEvent(urand(15000, 30000));
134 return;
135 }
136 events.RepeatEvent(500);
137 break;
138 case EVENT_SPELL_THUNDERING:
139 me->CastSpell(me, SPELL_THUNDERING_STORM, true);
140 events.RepeatEvent(15000);
141 break;
142 case EVENT_SPELL_SONIC_SHOCK:
143 me->CastSpell(me->GetVictim(), SPELL_SONIC_SHOCK, false);
144 events.RepeatEvent(urand(10000, 20000));
145 break;
146 }
147
148 if (!me->isAttackReady())
149 return;
150
151 if (!me->IsWithinMeleeRange(me->GetVictim()))
152 {
153 ThreatContainer::StorageType threatlist = me->GetThreatMgr().GetThreatList();
154 for (ThreatContainer::StorageType::const_iterator i = threatlist.begin(); i != threatlist.end(); ++i)
155 if (Unit* target = ObjectAccessor::GetUnit(*me, (*i)->getUnitGuid()))
156 if (target->IsAlive() && me->IsWithinMeleeRange(target))
157 {
158 me->TauntApply(target);
159 break;
160 }
161 }
162
163 DoMeleeAttackIfReady();
164 }
165 };
166};
