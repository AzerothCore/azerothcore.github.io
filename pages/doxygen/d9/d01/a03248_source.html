188 if (!instance || _owner->GetMapId() != 724)

391 case EVENT_ACTIVATE_FIREWALL:
392 instance->HandleGameObject(instance->GetGuidData(GO_FLAME_RING), false, nullptr);
393 instance->HandleGameObject(instance->GetGuidData(GO_TWILIGHT_FLAME_RING), false, nullptr);
394 break;
395 case EVENT_METEOR_STRIKE:
396 _livingEmberCount = summons.GetEntryCount(NPC_LIVING_EMBER);
397 me->CastCustomSpell(SPELL_METEOR_STRIKE_TARGETING, SPELLVALUE_MAX_TARGETS, 1, me, false);
398 Talk(SAY_METEOR_STRIKE);

400 break;
401 case EVENT_FIERY_COMBUSTION:
402 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 100.0f, true, -SPELL_TWILIGHT_REALM))

404 events.ScheduleEvent(EVENT_FIERY_COMBUSTION, 25000);
405 break;
406 case EVENT_CHECK_HEALTH:
407 if (me->HealthBelowPct(75))
408 {
409 Talk(SAY_PHASE_TWO);
410 me->CastSpell(me, SPELL_TWILIGHT_PHASING, false);
411 events.DelayEvents(10000);

413 }
414 events.ScheduleEvent(EVENT_CHECK_HEALTH, 1000);
415 break;
416 }
417
418 DoMeleeAttackIfReady();
419 }

677 void UpdateCorporeality()
678 {
679 if (!_instance->IsEncounterInProgress())
680 {
681 DoAction(ACTION_RESET_ENCOUNTER);
682 return;
683 }
684
685 uint8 oldValue = _corporeality;
686 float damageRatio = float(_materialDamage) / float(_twilightDamage);
687
688 if (_twilightDamage == 1 || _materialDamage == 1)
689 _events.ScheduleEvent(EVENT_TWILIGHT_MENDING, 4000);
690
691 _twilightDamage = 1;
692 _materialDamage = 1;
693
694 CorporealityEvent action = CORPOREALITY_NONE;
695 if (damageRatio < 0.98f)
696 action = CORPOREALITY_INCREASE;
697 else if (1.02f < damageRatio)
698 action = CORPOREALITY_DECREASE;
699 else
700 return;
701
702 switch (action)
703 {
704 case CORPOREALITY_INCREASE:
705 {
706 if (_corporeality == (MAX_CORPOREALITY_STATE - 1))
707 return;
708 ++_corporeality;
709 break;
710 }
711 case CORPOREALITY_DECREASE:
712 {
713 if (_corporeality == 0)
714 return;
715 --_corporeality;
716 break;
717 }
718 default:
719 break;
720 }
721
722 _instance->DoUpdateWorldState(WORLDSTATE_CORPOREALITY_MATERIAL, _corporeality * 10);
723 _instance->DoUpdateWorldState(WORLDSTATE_CORPOREALITY_TWILIGHT, 100 - _corporeality * 10);
724
725 if (Creature* twilightHalion = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(NPC_TWILIGHT_HALION)))
726 {
727 twilightHalion->RemoveAurasDueToSpell(_corporealityReference[MAX_CORPOREALITY_STATE - 1 - oldValue]);
728 twilightHalion->CastSpell(twilightHalion, _corporealityReference[MAX_CORPOREALITY_STATE - 1 - _corporeality], true);
729 twilightHalion->AI()->Talk(oldValue < _corporeality ? EMOTE_CORPOREALITY_TOT : EMOTE_CORPOREALITY_TIT);
730 }
731
732 if (Creature* halion = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(NPC_HALION)))
733 {
734 halion->RemoveAurasDueToSpell(_corporealityReference[oldValue]);
735 halion->CastSpell(halion, _corporealityReference[_corporeality], true);
736 halion->AI()->Talk(oldValue > _corporeality ? EMOTE_CORPOREALITY_POT : EMOTE_CORPOREALITY_PIP);
737 }
738 }
