103 }
104 }

158 void Update(uint32 diff) override
159 {
160 if (RescueTimer && RescueTimer < 100 * MINUTE * IN_MILLISECONDS)
161 {
162 RescueTimer -= std::min(RescueTimer, diff);
163
164 if ((RescueTimer / IN_MILLISECONDS) == 25 * MINUTE)
165 {
166 DoRemoveAurasDueToSpellOnPlayers(SPELL_KARGATHS_EXECUTIONER_1);
167 DoCastSpellOnPlayers(SPELL_KARGATHS_EXECUTIONER_2);
168 if (Creature* prisoner = instance->GetCreature(prisonerGUID[0]))
169 Unit::Kill(prisoner, prisoner);
170 if (Creature* executioner = instance->GetCreature(executionerGUID))
171 executioner->RemoveLootMode(2);
172 }
173 else if ((RescueTimer / IN_MILLISECONDS) == 15 * MINUTE)
174 {
175 DoRemoveAurasDueToSpellOnPlayers(SPELL_KARGATHS_EXECUTIONER_2);
176 DoCastSpellOnPlayers(SPELL_KARGATHS_EXECUTIONER_3);
177 if (Creature* prisoner = instance->GetCreature(prisonerGUID[1]))
178 Unit::Kill(prisoner, prisoner);
179 }
180 else if ((RescueTimer / IN_MILLISECONDS) == 0)
181 {
182 DoRemoveAurasDueToSpellOnPlayers(SPELL_KARGATHS_EXECUTIONER_3);
183 if (Creature* prisoner = instance->GetCreature(prisonerGUID[2]))
184 Unit::Kill(prisoner, prisoner);
185 }
186 }

200 void Load(const char* strIn) override
201 {
202 if (!strIn)
203 {
204 OUT_LOAD_INST_DATA_FAIL;
205 return;
206 }
207
208 OUT_LOAD_INST_DATA(strIn);
209
210 char dataHead1, dataHead2;
211
212 std::istringstream loadStream(strIn);
213 loadStream >> dataHead1 >> dataHead2;
214
215 if (dataHead1 == 'S' && dataHead2 == 'H')
216 {
217 for (uint8 i = 0; i < ENCOUNTER_COUNT; ++i)
218 {
219 uint32 tmpState;
220 loadStream >> tmpState;
221 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
222 tmpState = NOT_STARTED;
223 SetBossState(i, EncounterState(tmpState));
224 }
225
226 loadStream >> RescueTimer;
227 }
228 else
229 OUT_LOAD_INST_DATA_FAIL;
230
231 OUT_LOAD_INST_DATA_COMPLETE;
232 }
