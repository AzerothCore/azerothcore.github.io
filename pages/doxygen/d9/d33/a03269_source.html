222 events.Reset();
223 events.RescheduleEvent(EVENT_CHECK_PLAYERS, 0);
224
225 if( !in )
226 {
227 OUT_LOAD_INST_DATA_FAIL;
228 return;
229 }

231 OUT_LOAD_INST_DATA(in);
232
233 char dataHead1, dataHead2;
234 uint16 data0, data1, data2, data3;
235 std::istringstream loadStream(in);
236 loadStream >> dataHead1 >> dataHead2 >> data0 >> data1 >> data2 >> data3;
237
238 if( dataHead1 == 'T' && dataHead2 == 'C' )
239 {
240 m_auiEncounter[0] = data0;
241 m_auiEncounter[1] = data1;
242 m_auiEncounter[2] = data2;
243 InstanceProgress = data3;
244 if( InstanceProgress == INSTANCE_PROGRESS_CHAMPIONS_UNMOUNTED )
245 InstanceProgress = INSTANCE_PROGRESS_INITIAL;
246
247 for( uint8 i = 0; i < MAX_ENCOUNTER; ++i )
248 if( m_auiEncounter[i] == IN_PROGRESS )
249 m_auiEncounter[i] = NOT_STARTED;
250 }
251 else
252 OUT_LOAD_INST_DATA_FAIL;
253
254 OUT_LOAD_INST_DATA_COMPLETE;
