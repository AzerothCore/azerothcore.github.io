115 void UpdateAI(uint32 diff) override
116 {
117 if (!UpdateVictim())
118 return;
119
120 events.Update(diff);
121 switch (uint32 eventId = events.ExecuteEvent())
122 {
123 case EVENT_SPELL_SHRINK:
124 me->CastSpell(me->GetVictim(), SPELL_SUPER_SHRINK_RAY, false);
125 events.RepeatEvent(20000);
126 break;
127 case EVENT_SPELL_SAW:
128 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1))
129 me->CastSpell(target, SPELL_SAW_BLADE, false);
130 else
131 me->CastSpell(me->GetVictim(), SPELL_SAW_BLADE, false);
132 events.RepeatEvent(15000);
133 break;
134 case EVENT_SPELL_NET:
135 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
136 me->CastSpell(target, SPELL_ELECTRIFIED_NET, false);
137 events.RepeatEvent(10000);
138 break;
139 case EVENT_CHECK_HP25:
140 case EVENT_CHECK_HP50:
141 case EVENT_CHECK_HP75:
142 if (me->HealthBelowPct(eventId * 25))
143 {
144 SummonMechanics();
145 return;
146 }
147 events.RepeatEvent(2000);
148 break;
149 }
150
151 DoMeleeAttackIfReady();
152 }
153 };

176 {

184
185 void UpdateAI(uint32 diff) override
186 {
187 repairTimer += diff;
188 if (repairTimer >= 2000)
189 {
190 repairTimer = 0;
191 if (Unit* boss = ObjectAccessor::GetUnit(*me, bossGUID))
192 {
193 if (me->IsWithinDistInMap(boss, 13.0f))
194 if (!me->HasUnitState(UNIT_STATE_CASTING))
195 me->CastSpell(me, DUNGEON_MODE(SPELL_REPAIR_N, SPELL_REPAIR_H), false);
196 }
197 return;
198 }
199
200 if (!UpdateVictim())
201 return;
202
203 DoMeleeAttackIfReady();
204 }
