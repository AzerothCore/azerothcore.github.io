59void OutdoorPvPTF::FillInitialWorldStates(WorldPacket& data)
60{
61 data << TF_UI_TOWER_SLIDER_POS << uint32(50);
62 data << TF_UI_TOWER_SLIDER_N << uint32(100);
63 data << TF_UI_TOWER_SLIDER_DISPLAY << uint32(0);
64
65 data << TF_UI_TOWER_COUNT_H << m_HordeTowersControlled;
66 data << TF_UI_TOWER_COUNT_A << m_AllianceTowersControlled;
67 data << TF_UI_TOWERS_CONTROLLED_DISPLAY << uint32(!m_IsLocked);
68
69 data << TF_UI_LOCKED_TIME_MINUTES_FIRST_DIGIT << first_digit;
70 data << TF_UI_LOCKED_TIME_MINUTES_SECOND_DIGIT << second_digit;
71 data << TF_UI_LOCKED_TIME_HOURS << hours_left;
72
73 data << TF_UI_LOCKED_DISPLAY_NEUTRAL << uint32(m_IsLocked && !m_HordeTowersControlled && !m_AllianceTowersControlled);
74 data << TF_UI_LOCKED_DISPLAY_HORDE << uint32(m_IsLocked && (m_HordeTowersControlled > m_AllianceTowersControlled));
75 data << TF_UI_LOCKED_DISPLAY_ALLIANCE << uint32(m_IsLocked && (m_HordeTowersControlled < m_AllianceTowersControlled));
76
77 for (OPvPCapturePointMap::iterator itr = m_capturePoints.begin(); itr != m_capturePoints.end(); ++itr)
78 {
79 itr->second->FillInitialWorldStates(data);
80 }
81}

200bool OutdoorPvPTF::Update(uint32 diff)
201{
202 bool changed = OutdoorPvP::Update(diff);
203
204 if (changed)
205 {
206 if (m_AllianceTowersControlled == TF_TOWER_NUM)
207 {
208 TeamApplyBuff(TEAM_ALLIANCE, TF_CAPTURE_BUFF);
209 m_IsLocked = true;
210 m_JustLocked = true;
211 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_NEUTRAL, uint32(0));
212 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_HORDE, uint32(0));
213 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_ALLIANCE, uint32(1));
214 SendUpdateWorldState(TF_UI_TOWERS_CONTROLLED_DISPLAY, uint32(0));
215 }
216 else if (m_HordeTowersControlled == TF_TOWER_NUM)
217 {
218 TeamApplyBuff(TEAM_HORDE, TF_CAPTURE_BUFF);
219 m_IsLocked = true;
220 m_JustLocked = true;
221 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_NEUTRAL, uint32(0));
222 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_HORDE, uint32(1));
223 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_ALLIANCE, uint32(0));
224 SendUpdateWorldState(TF_UI_TOWERS_CONTROLLED_DISPLAY, uint32(0));
225 }
226 else
227 {
228 TeamCastSpell(TEAM_ALLIANCE, -TF_CAPTURE_BUFF);
229 TeamCastSpell(TEAM_HORDE, -TF_CAPTURE_BUFF);
230 }
231
232 SendUpdateWorldState(TF_UI_TOWER_COUNT_A, m_AllianceTowersControlled);
233 SendUpdateWorldState(TF_UI_TOWER_COUNT_H, m_HordeTowersControlled);
234 }
235
236 if (m_IsLocked)
237 {
238 if (m_JustLocked)
239 {
240 m_JustLocked = false;
241 SaveRequiredWorldStates();
242 }
243
244 // lock timer is down, release lock
245 if (m_LockTimer < diff)
246 {
247 m_LockTimer = TF_LOCK_TIME;
248 m_LockTimerUpdate = 0;
249 m_IsLocked = false;
250
251 ResetZoneToTeamControlled(TEAM_NEUTRAL);
252 SaveRequiredWorldStates();
253
254 SendUpdateWorldState(TF_UI_TOWERS_CONTROLLED_DISPLAY, uint32(1));
255 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_NEUTRAL, uint32(0));
256 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_HORDE, uint32(0));
257 SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_ALLIANCE, uint32(0));
258 }
259 else
260 {
261 // worldstateui update timer is down, update ui with new time data
262 if (m_LockTimerUpdate < diff)
263 {
264 m_LockTimerUpdate = TF_LOCK_TIME_UPDATE;
265 RecalculateClientUILockTime();
266
267 SendUpdateWorldState(TF_UI_LOCKED_TIME_MINUTES_FIRST_DIGIT, first_digit);
268 SendUpdateWorldState(TF_UI_LOCKED_TIME_MINUTES_SECOND_DIGIT, second_digit);
269 SendUpdateWorldState(TF_UI_LOCKED_TIME_HOURS, hours_left);
270 }
271 else
272 {
273 m_LockTimerUpdate -= diff;
274 }
275
276 m_LockTimer -= diff;
277 }
278 }
279
280 return changed;
281}

330bool OutdoorPvPTF::SetupOutdoorPvP()
331{
332 m_AllianceTowersControlled = 0;
333 m_HordeTowersControlled = 0;
334
335 m_IsLocked = bool(sWorld->getWorldState(TF_UI_TOWERS_CONTROLLED_DISPLAY));
336 m_JustLocked = false;
337 m_LockTimer = TF_LOCK_TIME;
338 m_LockTimerUpdate = 0;
339 hours_left = 6;
340 second_digit = 0;
341 first_digit = 0;
342
343 // add the zones affected by the pvp buff
344 for (uint8 i = 0; i < OutdoorPvPTFBuffZonesNum; ++i)
345 RegisterZone(OutdoorPvPTFBuffZones[i]);
346
347 SetMapFromZone(OutdoorPvPTFBuffZones[0]);
348
349 AddCapturePoint(new OPvPCapturePointTF(this, TF_TOWER_NW));
350 AddCapturePoint(new OPvPCapturePointTF(this, TF_TOWER_N));
351 AddCapturePoint(new OPvPCapturePointTF(this, TF_TOWER_NE));
352 AddCapturePoint(new OPvPCapturePointTF(this, TF_TOWER_SE));
353 AddCapturePoint(new OPvPCapturePointTF(this, TF_TOWER_S));
354
355 if (m_IsLocked)
356 {
357 // Core shutdown while locked -- init from latest known data in WorldState
358 // Convert from unix
359 int32 const lockRemainingTime = int32((sWorld->getWorldState(TF_UI_LOCKED_TIME_HOURS) - GameTime::GetGameTime().count()) * IN_MILLISECONDS);
360 if (lockRemainingTime > 0)
361 {
362 m_LockTimer = lockRemainingTime;
363 RecalculateClientUILockTime();
364
365 uint32 const hordeTowers = uint32(sWorld->getWorldState(TF_UI_TOWER_COUNT_H));
366 uint32 const allianceTowers = uint32(sWorld->getWorldState(TF_UI_TOWER_COUNT_A));
367 TeamId const controllingTeam = hordeTowers > allianceTowers ? TEAM_HORDE : TEAM_ALLIANCE;
368
369 ResetZoneToTeamControlled(controllingTeam);
370 }
371 else
372 {
373 // Lock expired while core was offline
374 m_IsLocked = false;
375 SaveRequiredWorldStates();
376 }
377 }
378
379 return true;
380}

382bool OPvPCapturePointTF::Update(uint32 diff)
383{
384 // can update even in locked state if gathers the controlling faction
385 bool canupdate = ((((OutdoorPvPTF*)m_PvP)->GetAllianceTowersControlled() > 0) && m_activePlayers[0].size() > m_activePlayers[1].size()) ||
386 ((((OutdoorPvPTF*)m_PvP)->GetHordeTowersControlled() > 0) && m_activePlayers[0].size() < m_activePlayers[1].size());
387 // if gathers the other faction, then only update if the pvp is unlocked
388 canupdate = canupdate || !((OutdoorPvPTF*)m_PvP)->IsLocked();
389 return canupdate && OPvPCapturePoint::Update(diff);
390}

392void OPvPCapturePointTF::ChangeState()
393{
394 // if changing from controlling alliance to horde
395 if (m_OldState == OBJECTIVESTATE_ALLIANCE)
396 {
397 if (uint32 alliance_towers = ((OutdoorPvPTF*)m_PvP)->GetAllianceTowersControlled())
398 ((OutdoorPvPTF*)m_PvP)->SetAllianceTowersControlled(--alliance_towers);
399 sWorld->SendZoneText(OutdoorPvPTFBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(LANG_OPVP_TF_LOSE_A));
400 }
401 // if changing from controlling horde to alliance
402 else if (m_OldState == OBJECTIVESTATE_HORDE)
403 {
404 if (uint32 horde_towers = ((OutdoorPvPTF*)m_PvP)->GetHordeTowersControlled())
405 ((OutdoorPvPTF*)m_PvP)->SetHordeTowersControlled(--horde_towers);
406 sWorld->SendZoneText(OutdoorPvPTFBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(LANG_OPVP_TF_LOSE_H));
407 }
408
409 uint32 artkit = 21;
410
411 switch (m_State)
412 {
413 case OBJECTIVESTATE_ALLIANCE:
414 {
415 m_TowerState = TF_TOWERSTATE_A;
416 artkit = 2;
417 uint32 alliance_towers = ((OutdoorPvPTF*)m_PvP)->GetAllianceTowersControlled();
418 if (alliance_towers < TF_TOWER_NUM)
419 ((OutdoorPvPTF*)m_PvP)->SetAllianceTowersControlled(++alliance_towers);
420
421 sWorld->SendZoneText(OutdoorPvPTFBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(LANG_OPVP_TF_CAPTURE_A));
422
423 for (PlayerSet::iterator itr = m_activePlayers[0].begin(); itr != m_activePlayers[0].end(); ++itr)
424 if (Player* player = ObjectAccessor::FindPlayer(*itr))
425 player->AreaExploredOrEventHappens(TF_ALLY_QUEST);
426 break;
427 }
428 case OBJECTIVESTATE_HORDE:
429 {
430 m_TowerState = TF_TOWERSTATE_H;
431 artkit = 1;
432 uint32 horde_towers = ((OutdoorPvPTF*)m_PvP)->GetHordeTowersControlled();
433 if (horde_towers < TF_TOWER_NUM)
434 ((OutdoorPvPTF*)m_PvP)->SetHordeTowersControlled(++horde_towers);
435
436 sWorld->SendZoneText(OutdoorPvPTFBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(LANG_OPVP_TF_CAPTURE_H));
437
438 for (PlayerSet::iterator itr = m_activePlayers[1].begin(); itr != m_activePlayers[1].end(); ++itr)
439 if (Player* player = ObjectAccessor::FindPlayer(*itr))
440 player->AreaExploredOrEventHappens(TF_HORDE_QUEST);
441 break;
442 }
443 case OBJECTIVESTATE_NEUTRAL:
444 case OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE:
445 case OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE:
446 case OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE:
447 case OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE:
448 m_TowerState = TF_TOWERSTATE_N;
449 break;
450 }
451
452 auto bounds = sMapMgr->FindMap(530, 0)->GetGameObjectBySpawnIdStore().equal_range(m_capturePointSpawnId);
453 for (auto itr = bounds.first; itr != bounds.second; ++itr)
454 itr->second->SetGoArtKit(artkit);
455
456 UpdateTowerState();
457}
