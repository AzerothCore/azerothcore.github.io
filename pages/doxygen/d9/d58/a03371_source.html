93 void UpdateAI(uint32 diff) override
94 {
95 if (!UpdateVictim())
96 return;
97
98 events.Update(diff);
99 if (me->HasUnitState(UNIT_STATE_CASTING))
100 return;
101
102 switch (events.ExecuteEvent())
103 {
104 case EVENT_ECK_HEALTH:
105 if (me->HealthBelowPct(21))
106 {
107 events.CancelEvent(EVENT_ECK_BERSERK);
108 me->CastSpell(me, SPELL_ECK_BERSERK, false);
109 break;
110 }
111 events.ScheduleEvent(EVENT_ECK_HEALTH, 1000);
112 break;
113 case EVENT_ECK_BERSERK:
114 me->CastSpell(me, SPELL_ECK_BERSERK, false);
115 events.CancelEvent(EVENT_ECK_HEALTH);
116 break;
117 case EVENT_ECK_BITE:
118 me->CastSpell(me->GetVictim(), SPELL_ECK_BITE, false);
119 events.ScheduleEvent(EVENT_ECK_BITE, urand(8000, 12000));
120 break;
121 case EVENT_ECK_SPIT:
122 me->CastSpell(me->GetVictim(), SPELL_ECK_SPIT, false);
123 events.ScheduleEvent(EVENT_ECK_SPIT, 10000);
124 break;
125 case EVENT_ECK_SPRING:
126 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 30.0f, true))
127 {
128 me->GetThreatMgr().ResetAllThreat();
129 me->AddThreat(target, 500.0f);
130 me->CastSpell(target, SPELL_ECK_SPRING, false);
131 }
132
133 events.ScheduleEvent(EVENT_ECK_SPRING, urand(5000, 10000));
134 break;
135 }
136
137 DoMeleeAttackIfReady();
138 }
