363 void Update(uint32 diff) override
364 {
365 if (_events.Empty())
366 return;
367
368 _events.Update(diff);
369
370 while (uint32 eventId = _events.ExecuteEvent())
371 {
372 switch (eventId)
373 {
374 case EVENT_RAZOR_SPAWN:
375 if (EggEvent == IN_PROGRESS)
376 {
377 bool spawnMoreAdds = true;
378 for (uint8 i = urand(2, 5); i > 0; --i)
379 {
380 uint32 mobEntry = Entry[urand(0, 2)];
381 uint32 dragonkinsCount = addsCount[0];
382 uint32 orcsCount = addsCount[1];
383
384 // If more than 12 dragonkins...
385 if (dragonkinsCount >= 12)
386 {
387 //... and more than 40 orcs - stop spawning more adds.
388 if (orcsCount >= 40)
389 {
390 spawnMoreAdds = false;
391 break;
392 }
393 //... - stop spawning them.
394 else if (mobEntry == NPC_BLACKWING_DRAGON)
395 {
396 continue;
397 }
398 }
399 // If more than 40 orcs - stop spawning them.
400 else if (orcsCount >= 40 && mobEntry != NPC_BLACKWING_DRAGON)
401 {
402 continue;
403 }
404
405 if (Creature* summon = instance->SummonCreature(mobEntry, SummonPosition[urand(0, 7)]))
406 {
407 summon->AI()->DoZoneInCombat();
408 }
409 }
410
411 if (spawnMoreAdds)
412 {
413 _events.ScheduleEvent(EVENT_RAZOR_SPAWN, 15000);
414 }
415 }
416 break;
417 case EVENT_RAZOR_PHASE_TWO:
418 _events.CancelEvent(EVENT_RAZOR_SPAWN);
419 if (Creature* razor = instance->GetCreature(razorgoreGUID))
420 razor->AI()->DoAction(ACTION_PHASE_TWO);
421 break;
422 case EVENT_RESPAWN_NEFARIUS:
423 if (Creature* nefarius = GetCreature(DATA_LORD_VICTOR_NEFARIUS))
424 {
425 nefarius->SetPhaseMask(1, true);
426 nefarius->setActive(true);
427 nefarius->Respawn();
428 nefarius->GetMotionMaster()->MoveTargetedHome();
429 }
430 break;
431 }
432 }

446 void Load(char const* data) override
447 {
448 if (!data)
449 {
450 OUT_LOAD_INST_DATA_FAIL;
451 return;
452 }
453
454 OUT_LOAD_INST_DATA(data);
455
456 char dataHead1, dataHead2, dataHead3;
457
458 std::istringstream loadStream(data);
459 loadStream >> dataHead1 >> dataHead2 >> dataHead3;
460
461 if (dataHead1 == 'B' && dataHead2 == 'W' && dataHead3 == 'L')
462 {
463 for (uint32 i = 0; i < EncounterCount; ++i)
464 {
465 uint32 tmpState;
466 loadStream >> tmpState;
467 if (tmpState == IN_PROGRESS || tmpState == FAIL || tmpState > SPECIAL)
468 tmpState = NOT_STARTED;
469 SetBossState(i, EncounterState(tmpState));
470 }
471
472 loadStream >> NefarianLeftTunnel;
473 loadStream >> NefarianRightTunnel;
474 }
475 else
476 {
477 OUT_LOAD_INST_DATA_FAIL;
478 }
479
480 OUT_LOAD_INST_DATA_COMPLETE;
481 }
