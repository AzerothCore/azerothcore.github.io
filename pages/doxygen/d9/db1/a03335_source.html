94 void UpdateAI(uint32 diff) override
95 {
96 if (startFightTimer)
97 {
98 if (startFightTimer <= diff)
99 {
100 startFightTimer = 0;
101 me->SetInCombatWithZone();
102 }
103 else
104 startFightTimer -= diff;
105 }
106
107 if (!UpdateVictim())
108 return;
109
110 events.Update(diff);
111
112 if (me->HasUnitState(UNIT_STATE_CASTING))
113 return;
114
115 switch (events.ExecuteEvent())
116 {
117 case EVENT_QUIVERING_STRIKE:
118 me->CastSpell(me->GetVictim(), SPELL_QUIVERING_STRIKE, false);
119 events.ScheduleEvent(EVENT_QUIVERING_STRIKE, 5000);
120 break;
121 case EVENT_IMPENDING_DESPAIR:
122 if (Unit* target = SelectTargetFromPlayerList(45.0f, 0, true))
123 {
124 Talk(SAY_IMPENDING_DESPAIR);
125 me->CastSpell(target, SPELL_IMPENDING_DESPAIR, false);
126 }
127 events.ScheduleEvent(EVENT_IMPENDING_DESPAIR, 12000);
128 break;
129 case EVENT_DEFILING_HORROR:
130 Talk(SAY_DEFILING_HORROR);
131 me->CastSpell((Unit*)nullptr, SPELL_DEFILING_HORROR, false);
132 me->SetControlled(true, UNIT_STATE_ROOT);
133 events.DelayEventsToMax(5000, 0);
134 events.ScheduleEvent(EVENT_UNROOT, 4000);
135 events.ScheduleEvent(EVENT_DEFILING_HORROR, 20000);
136 break;
137 case EVENT_UNROOT:
138 me->SetControlled(false, UNIT_STATE_ROOT);
139 break;
140 }
141
142 if ((uiHopelessnessCount == 0 && HealthBelowPct(67)) || (uiHopelessnessCount == 1 && HealthBelowPct(34)) || (uiHopelessnessCount == 2 && HealthBelowPct(11)))
143 {
144 if (uiHopelessnessCount)
145 me->RemoveOwnedAura(hopelessnessId[uiHopelessnessCount - 1][DUNGEON_MODE(0, 1)]);
146 me->CastSpell((Unit*)nullptr, hopelessnessId[uiHopelessnessCount][DUNGEON_MODE(0, 1)], true);
147 ++uiHopelessnessCount;
148 }
149
150 if (!me->HasUnitState(UNIT_STATE_ROOT))
151 DoMeleeAttackIfReady();
152 }
153
154 void JustDied(Unit* /*killer*/) override
155 {

158 pInstance->SetData(DATA_FALRIC, DONE);
159 }
160
161 void KilledUnit(Unit* who) override

170 ScriptedAI::EnterEvadeMode(why);
171 if (startFightTimer)
172 Reset();
173 }
174 };
