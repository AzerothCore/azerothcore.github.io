56 template <HMACImpl::HashCreator HashCreator, size_t DigestLength>
57 class GenericHMAC
58 {
59 public:
60 static constexpr size_t DIGEST_LENGTH = DigestLength;
61 using Digest = std::array<uint8, DIGEST_LENGTH>;
62
63 template <typename Container>
64 static Digest GetDigestOf(Container const& seed, uint8 const* data, size_t len)
65 {
66 GenericHMAC hash(seed);
67 hash.UpdateData(data, len);
68 hash.Finalize();
69 return hash.GetDigest();
70 }
71
72 template <typename Container, typename... Ts>
73 static auto GetDigestOf(Container const& seed, Ts&& ... pack) -> std::enable_if_t < !(std::is_integral_v<std::decay_t<Ts>> || ...), Digest >
74 {
75 GenericHMAC hash(seed);
76 (hash.UpdateData(std::forward<Ts>(pack)), ...);
77 hash.Finalize();
78 return hash.GetDigest();
79 }

81 GenericHMAC(uint8 const* seed, size_t len) : _ctx(HMACImpl::MakeCTX())
82 {
83 int result = HMAC_Init_ex(_ctx, seed, len, HashCreator(), nullptr);
84 ASSERT(result == 1);
85 }
86 template <typename Container> GenericHMAC(Container const& container) : GenericHMAC(std::data(container), std::size(container)) {}
87
88 ~GenericHMAC()
89 {
90 if (!_ctx)
91 {
92 return;
93 }
94 HMACImpl::DestroyCTX(_ctx);
95 _ctx = nullptr;
96 }
97
98 void UpdateData(uint8 const* data, size_t len)
99 {
100 int result = HMAC_Update(_ctx, data, len);
101 ASSERT(result == 1);
102 }
103 void UpdateData(std::string_view str) { UpdateData(reinterpret_cast<uint8 const*>(str.data()), str.size()); }
104 void UpdateData(std::string const& str) { UpdateData(std::string_view(str)); } /* explicit overload to avoid using the container template */
105 void UpdateData(char const* str) { UpdateData(std::string_view(str)); } /* explicit overload to avoid using the container template */
106 template <typename Container>
107 void UpdateData(Container const& c) { UpdateData(std::data(c), std::size(c)); }
108
109 void Finalize()
110 {
111 uint32 length = 0;
112 int result = HMAC_Final(_ctx, _digest.data(), &length);
113 ASSERT(result == 1);
114 ASSERT(length == DIGEST_LENGTH);
115 HMACImpl::DestroyCTX(_ctx);
116 _ctx = nullptr;
117 }
118
119 Digest const& GetDigest() const { return _digest; }
120 private:
121 HMAC_CTX* _ctx;
122 Digest _digest = { };
123 };
