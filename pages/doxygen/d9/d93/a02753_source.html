54 void HandleScriptEffect(SpellEffIndex /* effIndex */)
55 {
56 if (Player* player = GetHitPlayer())
57 {
58 if (Group* group = player->GetGroup())
59 {
60 for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
61 if (Player* member = itr->GetSource())
62 if (member->IsWithinDistInMap(player, 20.0f) && !member->isDead())
63 member->CastSpell(member, SPELL_TELEPORT_SINGLE_IN_GROUP, true);
64 }
65 else
66 player->CastSpell(player, SPELL_TELEPORT_SINGLE, true);
67 }
68 }
