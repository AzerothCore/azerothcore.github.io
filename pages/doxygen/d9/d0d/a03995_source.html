98 {

102 switch (value)

105 if (PeonEngagedCount >= 4)
106 return;

108 if (EventStage < EVENT_STAGE_TAUNT)

180 void UpdateAI(uint32 diff) override
181 {
182 events2.Update(diff);
183 uint32 eventId = events2.ExecuteEvent();
184
185 if (EventStage < EVENT_STAGE_MAIN && instance->GetBossState(DATA_NETHEKURSE) == IN_PROGRESS)
186 {
187 if (eventId == EVENT_INTRO)
188 {
189 Talk(SAY_TAUNT);
190 EventStage = EVENT_STAGE_TAUNT;
191 me->CastSpell(me, SPELL_SHADOW_SEAR, false);
192 }
193 else if (eventId == EVENT_START_ATTACK)
194 {
195 Talk(SAY_AGGRO);
196 EventStage = EVENT_STAGE_MAIN;
197 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
198 if (Unit* target = me->SelectNearestPlayer(50.0f))
199 AttackStart(target);
200
201 events.ScheduleEvent(EVENT_SPELL_DEATH_COIL, 20000);
202 events.ScheduleEvent(EVENT_SPELL_SHADOW_FISSURE, 8000);
203 events.ScheduleEvent(EVENT_CHECK_HEALTH, 1000);
204 return;
205 }
206 }
207
208 if (!UpdateVictim())
209 return;
210
211 events.Update(diff);
212 if (EventStage < EVENT_STAGE_MAIN || me->HasUnitState(UNIT_STATE_CASTING))
213 return;
214
215 switch (events.ExecuteEvent())
216 {
217 case EVENT_SPELL_SHADOW_FISSURE:
218 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
219 me->CastSpell(target, SPELL_SHADOW_FISSURE, false);
220 events.RescheduleEvent(EVENT_SPELL_SHADOW_FISSURE, urand(7500, 10000));
221 break;
222 case EVENT_SPELL_DEATH_COIL:
223 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
224 me->CastSpell(target, DUNGEON_MODE(SPELL_DEATH_COIL_N, SPELL_DEATH_COIL_H), false);
225 events.RescheduleEvent(EVENT_SPELL_DEATH_COIL, urand(15000, 20000));
226 break;
227 case EVENT_SPELL_CLEAVE:
228 me->CastSpell(me->GetVictim(), DUNGEON_MODE(SPELL_SHADOW_CLEAVE_N, SPELL_SHADOW_SLAM_H), false);
229 events.RescheduleEvent(EVENT_SPELL_CLEAVE, urand(6000, 8000));
230 break;
231 case EVENT_CHECK_HEALTH:
232 if (me->HealthBelowPct(21))
233 {
234 events.Reset();
235 me->CastSpell(me, SPELL_DARK_SPIN, false);
236 }
237 else
238 {
239 events.RescheduleEvent(EVENT_CHECK_HEALTH, 1000);
240 }
241 break;
242 }
243
244 if (!me->HealthBelowPct(21))
245 DoMeleeAttackIfReady();
246 }

248 private:
249 uint32 PeonEngagedCount;
250 uint32 PeonKilledCount;
251 uint32 EventStage;
