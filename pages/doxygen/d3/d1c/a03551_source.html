72 struct boss_lokenAI : public ScriptedAI
73 {
74 boss_lokenAI(Creature* creature) : ScriptedAI(creature)
75 {
76 m_pInstance = creature->GetInstanceScript();
77 if (m_pInstance)
78 isActive = m_pInstance->GetData(TYPE_LOKEN_INTRO);
79 }
80
81 InstanceScript* m_pInstance;
82 EventMap events;
83
84 bool isActive;
85 uint32 IntroTimer;
86 uint8 HealthCheck;
87
88 void MoveInLineOfSight(Unit*) override { }
89
90 void Reset() override
91 {
92 events.Reset();
93 if (m_pInstance)
94 {
95 m_pInstance->DoStopTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEVEMENT_TIMELY_DEATH);
96 m_pInstance->SetData(TYPE_LOKEN, NOT_STARTED);
97 }
98
99 HealthCheck = 75;
100 IntroTimer = 0;
101 me->RemoveAllAuras();
102
103 if (!isActive)
104 {
105 me->SetControlled(true, UNIT_STATE_STUNNED);
106 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
107 }
108 else
109 {
110 me->SetControlled(false, UNIT_STATE_STUNNED);
111 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
112 }
113 }
114
115 void EnterCombat(Unit*) override
116 {
117 me->SetInCombatWithZone();
118 Talk(SAY_AGGRO);
119
120 events.ScheduleEvent(EVENT_ARC_LIGHTNING, 10000);
121 events.ScheduleEvent(EVENT_SHOCKWAVE, 3000);
122 events.ScheduleEvent(EVENT_LIGHTNING_NOVA, 15000);
123
124 if (m_pInstance)
125 {
126 m_pInstance->SetData(TYPE_LOKEN, IN_PROGRESS);
127
128 if (me->GetMap()->IsHeroic())
129 m_pInstance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEVEMENT_TIMELY_DEATH);
130 }
131 }
132
133 void JustDied(Unit*) override
134 {
135 Talk(SAY_DEATH);
136
137 if (m_pInstance)
138 m_pInstance->SetData(TYPE_LOKEN, DONE);
139 }
140
141 void LokenSpeach(bool hp)
142 {
143 if (hp)
144 {
145 switch(HealthCheck)
146 {
147 case 75:
148 Talk(SAY_75HEALTH);
149 break;
150 case 50:
151 Talk(SAY_50HEALTH);
152 break;
153 case 25:
154 Talk(SAY_25HEALTH);
155 break;
156 }
157 }
158 else
159 Talk(SAY_NOVA);
160 }
161
162 void KilledUnit(Unit* victim) override
163 {
164 if (victim->GetTypeId() != TYPEID_PLAYER)
165 return;
166
167 Talk(SAY_SLAY);
168 }
169
170 void UpdateAI(uint32 diff) override
171 {
172 if (!isActive)
173 {
174 IntroTimer += diff;
175 if (IntroTimer > 5000 && IntroTimer < 10000)
176 {
177 if (SelectTargetFromPlayerList(60))
178 {
179 Talk(SAY_INTRO_1);
180 IntroTimer = 10000;
181 }
182 else
183 IntroTimer = 0;
184 }
185
186 if (IntroTimer >= 30000 && IntroTimer < 40000)
187 {
188 Talk(SAY_INTRO_2);
189 IntroTimer = 40000;
190 }
191 if (IntroTimer >= 60000)
192 {
193 isActive = true;
194 if (m_pInstance)
195 m_pInstance->SetData(TYPE_LOKEN_INTRO, 1);
196
197 me->SetControlled(false, UNIT_STATE_STUNNED);
198 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
199
200 if (Player* target = SelectTargetFromPlayerList(80))
201 AttackStart(target);
202 }
203
204 return;
205 }
206
207 //Return since we have no target
208 if (!UpdateVictim())
209 return;
210
211 events.Update(diff);
212
213 if (me->HasUnitState(UNIT_STATE_CASTING))
214 return;
215
216 switch (events.ExecuteEvent())
217 {
218 case EVENT_CHECK_HEALTH:
219 if (HealthBelowPct(HealthCheck))
220 {
221 LokenSpeach(true);
222 HealthCheck -= 25;
223 }
224
225 events.RepeatEvent(1000);
226 break;
227 case EVENT_LIGHTNING_NOVA:
228 events.RepeatEvent(15000);
229 me->CastSpell(me, SPELL_LIGHTNING_NOVA_VISUAL, true);
230 me->CastSpell(me, SPELL_LIGHTNING_NOVA_THUNDERS, true);
231
232 events.DelayEvents(5001);
233 events.ScheduleEvent(EVENT_AURA_REMOVE, me->GetMap()->IsHeroic() ? 4000 : 5000);
234
235 me->CastSpell(me, me->GetMap()->IsHeroic() ? SPELL_LIGHTNING_NOVA_H : SPELL_LIGHTNING_NOVA_N, false);
236 break;
237 case EVENT_SHOCKWAVE:
238 me->CastSpell(me, me->GetMap()->IsHeroic() ? SPELL_PULSING_SHOCKWAVE_H : SPELL_PULSING_SHOCKWAVE_N, false);
239 break;
240 case EVENT_ARC_LIGHTNING:
241 if (Unit* target = SelectTargetFromPlayerList(100, SPELL_ARC_LIGHTNING))
242 me->CastSpell(target, SPELL_ARC_LIGHTNING, false);
243
244 events.RepeatEvent(12000);
245 break;
246 case EVENT_AURA_REMOVE:
247 me->RemoveAura(SPELL_LIGHTNING_NOVA_THUNDERS);
248 break;
249 }
250
251 DoMeleeAttackIfReady();
252 }
253 };
254};
