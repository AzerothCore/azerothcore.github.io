53void BattlegroundRV::CheckPositionForUnit(Unit* unit)
54{
55 // get height at current pos, if something is wrong (below or high above) - teleport
56 if (!unit->IsFalling() && unit->IsAlive())
57 {
58 float groundZ_vmap = unit->GetMap()->GetHeight(unit->GetPositionX(), unit->GetPositionY(), 37.0f, true, 50.0f);
59 float groundZ_dyntree = unit->GetMap()->GetDynamicMapTree().getHeight(unit->GetPositionX(), unit->GetPositionY(), 37.0f, 50.0f, unit->GetPhaseMask());
60
61 if ((groundZ_vmap > 28.0f && groundZ_vmap < 29.0f) || (groundZ_dyntree > 28.0f && groundZ_dyntree < 37.0f))
62 {
63 float groundZ = std::max<float>(groundZ_vmap, groundZ_dyntree);
64 if (unit->GetPositionZ() < groundZ - 0.2f || unit->GetPositionZ() > groundZ + 3.5f)
65 TeleportUnitToNewZ(unit, groundZ + 1.0f, true);
66 }
67 }

70void BattlegroundRV::PostUpdateImpl(uint32 diff)
71{
72 if (GetStatus() != STATUS_IN_PROGRESS)
73 return;
74
75 if (_timer < Milliseconds(diff))
76 {
77 switch (_state)
78 {
79 case BG_RV_STATE_OPEN_FENCES:
80 {
81 for (uint8 i = BG_RV_OBJECT_FIRE_1; i <= BG_RV_OBJECT_FIREDOOR_2; ++i)
82 DoorOpen(i);
83
84 _timer = BG_RV_CLOSE_FIRE_TIMER;
85 _state = BG_RV_STATE_CLOSE_FIRE;
86
87 for (auto const& [playerGuid, player] : m_Players)
88 {
89 if (!player)
90 continue;
91
92 // Demonic Circle Summon
93 if (GameObject* gObj = player->GetGameObject(48018))
94 {
95 gObj->Relocate(gObj->GetPositionX(), gObj->GetPositionY(), 28.28f);
96 gObj->UpdateObjectVisibility(true);
97 }
98
99 if (player->GetPositionZ() < 27.0f)
100 TeleportUnitToNewZ(player, 28.28f, true);
101
102 for (uint8 i = SUMMON_SLOT_TOTEM; i < MAX_TOTEM_SLOT; ++i)
103 if (player->m_SummonSlot[i])
104 if (Creature* totem = GetBgMap()->GetCreature(player->m_SummonSlot[i]))
105 if (totem->GetPositionZ() < 28.0f)
106 TeleportUnitToNewZ(totem, 28.28f, true);
107
108 for (auto itr2 = player->m_Controlled.begin(); itr2 != player->m_Controlled.end(); ++itr2)
109 {
110 if ((*itr2)->GetPositionZ() < 28.0f)
111 TeleportUnitToNewZ((*itr2), 28.28f, true);
112
113 // Xinef: override stay position
114 if (CharmInfo* charmInfo = (*itr2)->GetCharmInfo())
115 if (charmInfo->IsAtStay())
116 {
117 (*itr2)->StopMovingOnCurrentPos();
118 charmInfo->SaveStayPosition(false);
119 }
120 }
121 }
122
123 // fix ground on elevators (so aoe spells can be casted there)
124 {
125 uint32 objects[2] = { BG_RV_OBJECT_ELEVATOR_1, BG_RV_OBJECT_ELEVATOR_2 };
126 for (uint8 i = 0; i < 2; ++i)
127 if (GameObject* go = GetBGObject(objects[i]))
128 go->RemoveGameObjectFlag(GO_FLAG_TRANSPORT);
129 }
130 break;
131 }
132 case BG_RV_STATE_CLOSE_FIRE:
133 for (uint8 i = BG_RV_OBJECT_FIRE_1; i <= BG_RV_OBJECT_FIREDOOR_2; ++i)
134 DoorClose(i);
135 // Fire got closed after five seconds, leaves twenty seconds before toggling pillars
136 _timer = BG_RV_FIRE_TO_PILLAR_TIMER;
137 _state = BG_RV_STATE_SWITCH_PILLARS;
138 break;
139 case BG_RV_STATE_SWITCH_PILLARS:
140 UpdatePillars();
141 _timer = BG_RV_PILLAR_SWITCH_TIMER;
142 break;
143 }
144 }
145 else
146 _timer -= Milliseconds(diff);
147
148 if (_state == BG_RV_STATE_OPEN_FENCES)
149 return;
150
151 if (_checkPlayersTimer <= diff)
152 {
153 _checkPlayersTimer = 0;
154
155 for (auto const& itr : m_Players)
156 CheckPositionForUnit(itr.second);
157
158 // maybe for pets and m_Controlled also, but not really necessary
159 }
160 else
161 _checkPlayersTimer -= diff;
162}

264void BattlegroundRV::UpdatePillars()
265{

267 if (!test)
268 return;
269
270 if (test->GetGoState() == GO_STATE_READY)
271 {

290uint32 BattlegroundRV::GetPillarIdForPos(Position* p)
291{
