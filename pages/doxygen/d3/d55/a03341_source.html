92 bool OnGossipHello(Player* player, Creature* creature) override
93 {
94 if (!creature->HasNpcFlag(UNIT_NPC_FLAG_GOSSIP))
95 return true;
96
97 if (creature->IsQuestGiver())
98 player->PrepareQuestMenu(creature->GetGUID());
99
100 bool canStart = true;
101 if (InstanceScript* instance = creature->GetInstanceScript())
102 if (uint32 bhd = instance->GetData(DATA_BATTERED_HILT))
103 if ((bhd & BHSF_FINISHED) == 0)
104 canStart = false;
105
106 if (canStart)
107 {
108 QuestStatus status = player->GetQuestStatus(creature->GetEntry() == NPC_SYLVANAS_PART1 ? QUEST_DELIVRANCE_FROM_THE_PIT_H2 : QUEST_DELIVRANCE_FROM_THE_PIT_A2);
109 if (status == QUEST_STATUS_COMPLETE || status == QUEST_STATUS_REWARDED)
110 AddGossipItemFor(player, 0, "Can you remove the sword?", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
111
112 // once last quest is completed, she offers this shortcut of the starting event
113 status = player->GetQuestStatus(creature->GetEntry() == NPC_SYLVANAS_PART1 ? QUEST_WRATH_OF_THE_LICH_KING_H2 : QUEST_WRATH_OF_THE_LICH_KING_A2);
114 if (status == QUEST_STATUS_COMPLETE || status == QUEST_STATUS_REWARDED)
115 {
116 if (creature->GetEntry() == NPC_SYLVANAS_PART1)
117 AddGossipItemFor(player, 0, "Dark Lady, I think I hear Arthas coming. Whatever you're going to do, do it quickly.", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
118 else
119 AddGossipItemFor(player, 0, "My Lady, I think I hear Arthas coming. Whatever you're going to do, do it quickly.", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
120 }
121 }
122
123 if (creature->GetEntry() == NPC_SYLVANAS_PART1)
124 {
125 SendGossipMenuFor(player, 15215, creature->GetGUID());
126 }
127 else if (creature->GetEntry() == NPC_JAINA_PART1)
128 {
129 SendGossipMenuFor(player, 15339, creature->GetGUID());
130 }
131
132 return true;
133 }

135 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*uiSender*/, uint32 uiAction) override
136 {
137 if (!creature->HasNpcFlag(UNIT_NPC_FLAG_GOSSIP))
138 return true;
139
140 InstanceScript* instance = creature->GetInstanceScript();
141 if (!instance)
142 return true;
143
144 if (uint32 bhd = instance->GetData(DATA_BATTERED_HILT))
145 if ((bhd & BHSF_FINISHED) == 0)
146 return true;
147
148 instance->SetData(DATA_BATTERED_HILT, 1);
149
150 ClearGossipMenuFor(player);
151 switch (uiAction)
152 {
153 case GOSSIP_ACTION_INFO_DEF+1:
154 CloseGossipMenuFor(player);
155 if (creature->AI())
156 creature->AI()->DoAction(ACTION_START_INTRO);
157 creature->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP | UNIT_NPC_FLAG_QUESTGIVER);
158 break;
159 case GOSSIP_ACTION_INFO_DEF+2:
160 CloseGossipMenuFor(player);
161 if (creature->AI())
162 creature->AI()->DoAction(ACTION_SKIP_INTRO);
163 creature->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP | UNIT_NPC_FLAG_QUESTGIVER);
164 break;
165 }
166
167 return true;
168 }
