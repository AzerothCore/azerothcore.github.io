88 void Reset() override
89 {
90 BossAI::Reset();
91 instance->SetBossState(DATA_NOVOS_CRYSTALS, IN_PROGRESS);
92 instance->SetBossState(DATA_NOVOS_CRYSTALS, NOT_STARTED);
93 _crystalCounter = 0;
94 _summonTargetRightGUID.Clear();
95 _summonTargetLeftGUID.Clear();
96 _stage = 0;
97
98 me->SetUnitFlag(UNIT_FLAG_DISABLE_MOVE);
99 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
100 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
101
102 _achievement = true;
103 }
104
105 uint32 GetData(uint32 data) const override
106 {
107 if (data == me->GetEntry())
108 return uint32(_achievement);
109 return 0;
110 }
111
112 void SetData(uint32 type, uint32) override
113 {
114 if (type == me->GetEntry())
115 _achievement = false;
116 }
117
118 void MoveInLineOfSight(Unit* /*who*/) override { }
119
120 void EnterCombat(Unit* who) override
121 {
122 Talk(SAY_AGGRO);
123 BossAI::EnterCombat(who);
124
125 events.ScheduleEvent(EVENT_SUMMON_FETID_TROLL, 3000);
126 events.ScheduleEvent(EVENT_SUMMON_SHADOWCASTER, 9000);
127 events.ScheduleEvent(EVENT_SUMMON_HULKING_CORPSE, 30000);
128 events.ScheduleEvent(EVENT_SUMMON_CRYSTAL_HANDLER, 20000);
129 events.ScheduleEvent(EVENT_CHECK_PHASE, 80000);
130
131 me->CastSpell(me, SPELL_ARCANE_BLAST, true);
132 me->CastSpell(me, SPELL_ARCANE_FIELD, true);
133 me->CastSpell(me, SPELL_DESPAWN_CRYSTAL_HANDLER, true);
134
135 for (auto itr : npcSummon)
136 {
137 uint32 summonEntry;
138 Position summonPos;
139 std::tie(summonEntry, summonPos) = itr.second;
140 if (Creature* creature = me->SummonCreature(summonEntry, summonPos))
141 switch (itr.first)
142 {
143 case ROOM_LEFT:
144 _summonTargetLeftGUID = creature->GetGUID();
145 break;
146 case ROOM_RIGHT:
147 _summonTargetRightGUID = creature->GetGUID();
148 break;
149 }
150 }
151
152 me->SetGuidValue(UNIT_FIELD_TARGET, ObjectGuid::Empty);
153 me->RemoveAllAuras();
154 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
155 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
156 }
157
158 void JustDied(Unit* killer) override
159 {
160 Talk(SAY_DEATH);
161 BossAI::JustDied(killer);
162 instance->SetBossState(DATA_NOVOS_CRYSTALS, DONE);
163 }
164
165 void KilledUnit(Unit* /*victim*/) override
166 {
167 if (events.GetNextEventTime(EVENT_KILL_TALK) == 0)
168 {
169 Talk(SAY_KILL);
170 events.ScheduleEvent(EVENT_KILL_TALK, 6000);
171 }
172 }
173
174 void JustSummoned(Creature* summon) override
175 {
176 summons.Summon(summon);
177 if (me->HasUnitFlag(UNIT_FLAG_NOT_SELECTABLE) && summon->GetEntry() != NPC_CRYSTAL_CHANNEL_TARGET && summon->GetEntry() != NPC_CRYSTAL_HANDLER)
178 summon->SetReactState(REACT_DEFENSIVE);
179 else if (summon->GetEntry() != NPC_CRYSTAL_CHANNEL_TARGET)
180 summon->SetInCombatWithZone();
181 }
182
183 void UpdateAI(uint32 diff) override
184 {
185 if (!UpdateVictim())
186 return;
187
188 events.Update(diff);
189 switch (events.ExecuteEvent())
190 {
191 case EVENT_SUMMON_FETID_TROLL:
192 if (Creature* trigger = summons.GetCreatureWithEntry(NPC_CRYSTAL_CHANNEL_TARGET))
193 trigger->CastSpell(trigger, SPELL_SUMMON_FETID_TROLL_CORPSE, true, nullptr, nullptr, me->GetGUID());
194 events.ScheduleEvent(EVENT_SUMMON_FETID_TROLL, 3000);
195 break;
196 case EVENT_SUMMON_HULKING_CORPSE:
197 if (Creature* trigger = summons.GetCreatureWithEntry(NPC_CRYSTAL_CHANNEL_TARGET))
198 trigger->CastSpell(trigger, SPELL_SUMMON_HULKING_CORPSE, true, nullptr, nullptr, me->GetGUID());
199 events.ScheduleEvent(EVENT_SUMMON_HULKING_CORPSE, 30000);
200 break;
201 case EVENT_SUMMON_SHADOWCASTER:
202 if (Creature* trigger = summons.GetCreatureWithEntry(NPC_CRYSTAL_CHANNEL_TARGET))
203 trigger->CastSpell(trigger, SPELL_SUMMON_RISEN_SHADOWCASTER, true, nullptr, nullptr, me->GetGUID());
204 events.ScheduleEvent(EVENT_SUMMON_SHADOWCASTER, 10000);
205 break;
206 case EVENT_SUMMON_CRYSTAL_HANDLER:
207 if (_crystalCounter++ < 4)
208 {
209 Talk(SAY_SUMMONING_ADDS);
210 Talk(EMOTE_SUMMONING_ADDS);
211 if (Creature* target = ObjectAccessor::GetCreature(*me, _stage ? _summonTargetLeftGUID : _summonTargetRightGUID))
212 target->CastSpell(target, SPELL_SUMMON_CRYSTAL_HANDLER, true, nullptr, nullptr, me->GetGUID());
213 _stage = _stage ? 0 : 1;
214 events.ScheduleEvent(EVENT_SUMMON_CRYSTAL_HANDLER, 20000);
215 }
216 break;
217 case EVENT_CHECK_PHASE:
218 if (me->HasAura(SPELL_BEAM_CHANNEL))
219 {
220 events.ScheduleEvent(EVENT_CHECK_PHASE, 2000);
221 break;
222 }
223 events.Reset();
224 events.ScheduleEvent(EVENT_CAST_OFFENSIVE_SPELL, 3000);
225 events.ScheduleEvent(EVENT_SPELL_SUMMON_MINIONS, 10000);
226 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
227 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
228 me->InterruptNonMeleeSpells(false);
229 break;
230 case EVENT_CAST_OFFENSIVE_SPELL:
231 if (!me->HasUnitState(UNIT_STATE_CASTING))
232 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
233 me->CastSpell(target, RAND(SPELL_BLIZZARD, SPELL_FROSTBOLT, SPELL_TOUCH_OF_MISERY), false);
234
235 events.ScheduleEvent(EVENT_CAST_OFFENSIVE_SPELL, 500);
236 break;
237 case EVENT_SPELL_SUMMON_MINIONS:
238 if (me->HasUnitState(UNIT_STATE_CASTING))
239 {
240 me->CastSpell(me, SPELL_SUMMON_MINIONS, false);
241 events.ScheduleEvent(EVENT_SPELL_SUMMON_MINIONS, 15000);
242 break;
243 }
244 events.ScheduleEvent(EVENT_SPELL_SUMMON_MINIONS, 500);
245 break;
246 }
247 }
