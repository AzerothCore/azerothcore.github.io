48void SmartWaypointMgr::LoadFromDB()
49{
50 uint32 oldMSTime = getMSTime();
51
52 for (std::unordered_map<uint32, WPPath*>::iterator itr = waypoint_map.begin(); itr != waypoint_map.end(); ++itr)
53 {
54 for (WPPath::iterator pathItr = itr->second->begin(); pathItr != itr->second->end(); ++pathItr)
55 delete pathItr->second;
56
57 delete itr->second;
58 }
59
60 waypoint_map.clear();
61
62 WorldDatabasePreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_SMARTAI_WP);
63 PreparedQueryResult result = WorldDatabase.Query(stmt);
64
65 if (!result)
66 {
67 LOG_WARN("server.loading", ">> Loaded 0 SmartAI Waypoint Paths. DB table `waypoints` is empty.");
68 LOG_INFO("server.loading", " ");
69 return;
70 }
71
72 uint32 count = 0;
73 uint32 total = 0;
74 uint32 last_entry = 0;
75 uint32 last_id = 1;
76
77 do
78 {
79 Field* fields = result->Fetch();
80 uint32 entry = fields[0].Get<uint32>();
81 uint32 id = fields[1].Get<uint32>();
82 float x = fields[2].Get<float>();
83 float y = fields[3].Get<float>();
84 float z = fields[4].Get<float>();
85 Optional<float> o;
86 if (!fields[5].IsNull())
87 o = fields[5].Get<float>();
88 uint32 delay = fields[6].Get<uint32>();
89
90 if (last_entry != entry)
91 {
92 waypoint_map[entry] = new WPPath();
93 last_id = 1;
94 count++;
95 }
96
97 if (last_id != id)
98 LOG_ERROR("sql.sql", "SmartWaypointMgr::LoadFromDB: Path entry {}, unexpected point id {}, expected {}.", entry, id, last_id);
99
100 last_id++;
101 (*waypoint_map[entry])[id] = new WayPoint(id, x, y, z, o, delay);
102
103 last_entry = entry;
104 total++;
105 } while (result->NextRow());
106
107 LOG_INFO("server.loading", ">> Loaded {} SmartAI waypoint paths (total {} waypoints) in {} ms", count, total, GetMSTimeDiffToNow(oldMSTime));
108 LOG_INFO("server.loading", " ");
109}

128void SmartAIMgr::LoadSmartAIFromDB()
129{
130 uint32 oldMSTime = getMSTime();
131
132 for (uint8 i = 0; i < SMART_SCRIPT_TYPE_MAX; i++)
133 mEventMap[i].clear(); //Drop Existing SmartAI List
134
135 WorldDatabasePreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_SMART_SCRIPTS);
136 PreparedQueryResult result = WorldDatabase.Query(stmt);
137
138 if (!result)
139 {
140 LOG_WARN("server.loading", ">> Loaded 0 SmartAI scripts. DB table `smart_scripts` is empty.");
141 LOG_INFO("server.loading", " ");
142 return;
143 }
144
145 uint32 count = 0;
146
147 do
148 {
149 Field* fields = result->Fetch();
150
151 SmartScriptHolder temp;
152
153 temp.entryOrGuid = fields[0].Get<int32>();
154 if (!temp.entryOrGuid)
155 {
156 LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: invalid entryorguid (0), skipped loading.");
157 continue;
158 }
159
160 SmartScriptType source_type = (SmartScriptType)fields[1].Get<uint8>();
161 if (source_type >= SMART_SCRIPT_TYPE_MAX)
162 {
163 LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: invalid source_type ({}), skipped loading.", uint32(source_type));
164 continue;
165 }
166 if (temp.entryOrGuid >= 0)
167 {
168 switch (source_type)
169 {
170 case SMART_SCRIPT_TYPE_CREATURE:
171 {
172 if (!sObjectMgr->GetCreatureTemplate((uint32)temp.entryOrGuid))
173 {
174 LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature entry ({}) does not exist, skipped loading.", uint32(temp.entryOrGuid));
175 continue;
176 }
177 break;
178 }
179 case SMART_SCRIPT_TYPE_GAMEOBJECT:
180 {
181 if (!sObjectMgr->GetGameObjectTemplate((uint32)temp.entryOrGuid))
182 {
183 LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: GameObject entry ({}) does not exist, skipped loading.", uint32(temp.entryOrGuid));
184 continue;
185 }
186 break;
187 }
188 case SMART_SCRIPT_TYPE_AREATRIGGER:
189 {
190 if (!sObjectMgr->GetAreaTrigger((uint32)temp.entryOrGuid))
191 {
192 LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: AreaTrigger entry ({}) does not exist, skipped loading.", uint32(temp.entryOrGuid));
193 continue;
194 }
195 break;
196 }
197 case SMART_SCRIPT_TYPE_TIMED_ACTIONLIST:
198 break;//nothing to check, really
199 default:
200 LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: not yet implemented source_type {}", (uint32)source_type);
201 continue;
202 }
203 }
204 else
205 {
206 if (!sObjectMgr->GetCreatureData(uint32(std::abs(temp.entryOrGuid))))
207 {
208 LOG_ERROR("sql.sql", "SmartAIMgr::LoadSmartAIFromDB: Creature guid ({}) does not exist, skipped loading.", uint32(std::abs(temp.entryOrGuid)));
209 continue;
210 }
211 }
212
213 temp.source_type = source_type;
214 temp.event_id = fields[2].Get<uint16>();
215 temp.link = fields[3].Get<uint16>();
216 temp.event.type = (SMART_EVENT)fields[4].Get<uint8>();
217 temp.event.event_phase_mask = fields[5].Get<uint16>();
218 temp.event.event_chance = fields[6].Get<uint8>();
219 temp.event.event_flags = fields[7].Get<uint16>();
220
221 temp.event.raw.param1 = fields[8].Get<uint32>();
222 temp.event.raw.param2 = fields[9].Get<uint32>();
223 temp.event.raw.param3 = fields[10].Get<uint32>();
224 temp.event.raw.param4 = fields[11].Get<uint32>();
225 temp.event.raw.param5 = fields[12].Get<uint32>();
226
227 temp.action.type = (SMART_ACTION)fields[13].Get<uint8>();
228 temp.action.raw.param1 = fields[14].Get<uint32>();
229 temp.action.raw.param2 = fields[15].Get<uint32>();
230 temp.action.raw.param3 = fields[16].Get<uint32>();
231 temp.action.raw.param4 = fields[17].Get<uint32>();
232 temp.action.raw.param5 = fields[18].Get<uint32>();
233 temp.action.raw.param6 = fields[19].Get<uint32>();
234
235 temp.target.type = (SMARTAI_TARGETS)fields[20].Get<uint8>();
236 temp.target.raw.param1 = fields[21].Get<uint32>();
237 temp.target.raw.param2 = fields[22].Get<uint32>();
238 temp.target.raw.param3 = fields[23].Get<uint32>();
239 temp.target.raw.param4 = fields[24].Get<uint32>();
240 temp.target.x = fields[25].Get<float>();
241 temp.target.y = fields[26].Get<float>();
242 temp.target.z = fields[27].Get<float>();
243 temp.target.o = fields[28].Get<float>();
244
245 //check target
246 if (!IsTargetValid(temp))
247 continue;
248
249 // check all event and action params
250 if (!IsEventValid(temp))
251 continue;
252
253 // xinef: specific check for timed events, fix db makers
254 switch (temp.event.type)
255 {
256 case SMART_EVENT_UPDATE:
257 case SMART_EVENT_UPDATE_OOC:
258 case SMART_EVENT_UPDATE_IC:
259 case SMART_EVENT_HEALTH_PCT:
260 case SMART_EVENT_TARGET_HEALTH_PCT:
261 case SMART_EVENT_MANA_PCT:
262 case SMART_EVENT_TARGET_MANA_PCT:
263 case SMART_EVENT_FRIENDLY_HEALTH:
264 case SMART_EVENT_FRIENDLY_HEALTH_PCT:
265 case SMART_EVENT_FRIENDLY_MISSING_BUFF:
266 case SMART_EVENT_HAS_AURA:
267 case SMART_EVENT_TARGET_BUFFED:
268 if (temp.event.minMaxRepeat.repeatMin == 0 && temp.event.minMaxRepeat.repeatMax == 0)
269 temp.event.event_flags |= SMART_EVENT_FLAG_NOT_REPEATABLE;
270 break;
271 case SMART_EVENT_RANGE:
272 if (temp.event.rangeRepeat.repeatMin == 0 && temp.event.rangeRepeat.repeatMax == 0)
273 temp.event.event_flags |= SMART_EVENT_FLAG_NOT_REPEATABLE;
274 // Will only work properly if value is 0 or 1
275 if (temp.event.rangeRepeat.onlyFireOnRepeat > 1)
276 temp.event.rangeRepeat.onlyFireOnRepeat = 1;
277 break;
278 case SMART_EVENT_VICTIM_CASTING:
279 case SMART_EVENT_IS_BEHIND_TARGET:
280 if (temp.event.minMaxRepeat.min == 0 && temp.event.minMaxRepeat.max == 0)
281 temp.event.event_flags |= SMART_EVENT_FLAG_NOT_REPEATABLE;
282 break;
283 case SMART_EVENT_FRIENDLY_IS_CC:
284 if (temp.event.friendlyCC.repeatMin == 0 && temp.event.friendlyCC.repeatMax == 0)
285 temp.event.event_flags |= SMART_EVENT_FLAG_NOT_REPEATABLE;
286 break;
287 default:
288 break;
289 }
290
291 // xinef: rozpierdol tc, niedojeby ze szok
292 if (temp.action.type == SMART_ACTION_MOVE_TO_POS)
293 if (temp.target.type == SMART_TARGET_SELF && (std::fabs(temp.target.x) > 200.0f || std::fabs(temp.target.y) > 200.0f || std::fabs(temp.target.z) > 200.0f))
294 temp.target.type = SMART_TARGET_POSITION;
295
296 // creature entry / guid not found in storage, create empty event list for it and increase counters
297 if (mEventMap[source_type].find(temp.entryOrGuid) == mEventMap[source_type].end())
298 {
299 ++count;
300 SmartAIEventList eventList;
301 mEventMap[source_type][temp.entryOrGuid] = eventList;
302 }
303 // store the new event
304 mEventMap[source_type][temp.entryOrGuid].push_back(temp);
305 } while (result->NextRow());
306
307 LOG_INFO("server.loading", ">> Loaded {} SmartAI scripts in {} ms", count, GetMSTimeDiffToNow(oldMSTime));
308 LOG_INFO("server.loading", " ");
309}

359bool SmartAIMgr::IsTargetValid(SmartScriptHolder const& e)
360{
361 if (e.GetActionType() == SMART_ACTION_INSTALL_AI_TEMPLATE)
362 return true; // AI template has special handling
363 switch (e.GetTargetType())
364 {
365 case SMART_TARGET_CREATURE_DISTANCE:
366 case SMART_TARGET_CREATURE_RANGE:
367 {
368 if (e.target.unitDistance.creature && !sObjectMgr->GetCreatureTemplate(e.target.unitDistance.creature))
369 {
370 LOG_ERROR("sql.sql", "SmartAIMgr: Entry {} SourceType {} Event {} Action {} uses non-existent Creature entry {} as target_param1, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.target.unitDistance.creature);
371 return false;
372 }
373 break;
374 }
375 case SMART_TARGET_GAMEOBJECT_DISTANCE:
376 case SMART_TARGET_GAMEOBJECT_RANGE:
377 {
378 if (e.target.goDistance.entry && !sObjectMgr->GetGameObjectTemplate(e.target.goDistance.entry))
379 {
380 LOG_ERROR("sql.sql", "SmartAIMgr: Entry {} SourceType {} Event {} Action {} uses non-existent GameObject entry {} as target_param1, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType(), e.target.goDistance.entry);
381 return false;
382 }
383 break;
384 }
385 case SMART_TARGET_CREATURE_GUID:
386 {
387 if (e.target.unitGUID.entry && !IsCreatureValid(e, e.target.unitGUID.entry))
388 return false;
389 break;
390 }
391 case SMART_TARGET_GAMEOBJECT_GUID:
392 {
393 if (e.target.goGUID.entry && !IsGameObjectValid(e, e.target.goGUID.entry))
394 return false;
395 break;
396 }
397 case SMART_TARGET_PLAYER_DISTANCE:
398 case SMART_TARGET_CLOSEST_PLAYER:
399 {
400 if (e.target.playerDistance.dist == 0)
401 {
402 LOG_ERROR("sql.sql", "SmartAIMgr: Entry {} SourceType {} Event {} Action {} has maxDist 0 as target_param1, skipped.", e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType());
403 return false;
404 }
405 break;
406 }
407 case SMART_TARGET_ACTION_INVOKER:
408 case SMART_TARGET_ACTION_INVOKER_VEHICLE:
409 case SMART_TARGET_INVOKER_PARTY:
410 if (e.GetScriptType() != SMART_SCRIPT_TYPE_TIMED_ACTIONLIST && e.GetEventType() != SMART_EVENT_LINK && !EventHasInvoker(e.event.type))
411 {
412 LOG_ERROR("sql.sql", "SmartAIMgr: Entry {} SourceType {} Event {} Action {} has invoker target, but action does not provide any invoker!", e.entryOrGuid, e.GetScriptType(), e.GetEventType(), e.GetActionType());
413 return false;
414 }
415 break;
416 case SMART_TARGET_HOSTILE_SECOND_AGGRO:
417 case SMART_TARGET_HOSTILE_LAST_AGGRO:
418 case SMART_TARGET_HOSTILE_RANDOM:
419 case SMART_TARGET_HOSTILE_RANDOM_NOT_TOP:
420 AC_SAI_IS_BOOLEAN_VALID(e, e.target.hostileRandom.playerOnly);
421 break;
422 case SMART_TARGET_FARTHEST:
423 AC_SAI_IS_BOOLEAN_VALID(e, e.target.farthest.playerOnly);
424 AC_SAI_IS_BOOLEAN_VALID(e, e.target.farthest.isInLos);
425 break;
426 case SMART_TARGET_CLOSEST_CREATURE:
427 AC_SAI_IS_BOOLEAN_VALID(e, e.target.unitClosest.dead);
428 break;
429 case SMART_TARGET_CLOSEST_ENEMY:
430 AC_SAI_IS_BOOLEAN_VALID(e, e.target.closestAttackable.playerOnly);
431 break;
432 case SMART_TARGET_CLOSEST_FRIENDLY:
433 AC_SAI_IS_BOOLEAN_VALID(e, e.target.closestFriendly.playerOnly);
434 break;
435 case SMART_TARGET_OWNER_OR_SUMMONER:
436 AC_SAI_IS_BOOLEAN_VALID(e, e.target.owner.useCharmerOrOwner);
437 break;
438 case SMART_TARGET_STORED:
439 case SMART_TARGET_PLAYER_WITH_AURA:
440 case SMART_TARGET_RANDOM_POINT:
441 case SMART_TARGET_ROLE_SELECTION:
442 case SMART_TARGET_LOOT_RECIPIENTS:
443 case SMART_EVENT_SUMMONED_UNIT_DIES:
444 case SMART_TARGET_PLAYER_RANGE:
445 case SMART_TARGET_CLOSEST_GAMEOBJECT:
446 case SMART_TARGET_SELF:
447 case SMART_TARGET_VICTIM:
448 case SMART_TARGET_POSITION:
449 case SMART_TARGET_NONE:
450 case SMART_TARGET_THREAT_LIST:
451 break;
452 default:
453 LOG_ERROR("sql.sql", "SmartAIMgr: Not handled target_type({}), Entry {} SourceType {} Event {} Action {}, skipped.", e.GetTargetType(), e.entryOrGuid, e.GetScriptType(), e.event_id, e.GetActionType());
454 return false;
455 }
456
457 if (!CheckUnusedTargetParams(e))
458 {
459 return false;
460 }
461
462 return true;

837{
838 if ((e.event.type >= SMART_EVENT_TC_END && e.event.type <= SMART_EVENT_AC_START) || e.event.type >= SMART_EVENT_AC_END)
839 {
840 LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid {} using event({}) has invalid event type ({}), skipped.", e.entryOrGuid, e.event_id, e.GetEventType());
841 return false;
842 }
843 // in SMART_SCRIPT_TYPE_TIMED_ACTIONLIST all event types are overriden by core
844 if (e.GetScriptType() != SMART_SCRIPT_TYPE_TIMED_ACTIONLIST && !(SmartAIEventMask[e.event.type][1] & SmartAITypeMask[e.GetScriptType()][1]))
845 {
846 LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid {}, event type {} can not be used for Script type {}", e.entryOrGuid, e.GetEventType(), e.GetScriptType());
847 return false;
848 }
849 if (e.action.type <= 0
850 || (e.action.type >= SMART_ACTION_TC_END && e.action.type <= SMART_ACTION_AC_START)
851 || e.action.type >= SMART_ACTION_AC_END)
852 {
853 LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid {} using event({}) has an invalid action type ({}), skipped.", e.entryOrGuid, e.event_id, e.GetActionType());
854 return false;
855 }
856 switch (e.action.type)
857 {
858 case SMART_ACTION_RESERVED_16:
859 case SMART_ACTION_PLAY_ANIMKIT:
860 case SMART_ACTION_SCENE_PLAY:
861 case SMART_ACTION_SCENE_CANCEL:
862 LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid {} using event({}) has an action type that is not supported on 3.3.5a ({}), skipped.",
863 e.entryOrGuid, e.event_id, e.GetActionType());
864 return false;
865 case SMART_ACTION_SET_CAN_FLY:
866 case SMART_ACTION_REMOVE_AURAS_BY_TYPE:
867 case SMART_ACTION_REMOVE_MOVEMENT:
868 case SMART_ACTION_SPAWN_SPAWNGROUP:
869 case SMART_ACTION_DESPAWN_SPAWNGROUP:
870 case SMART_ACTION_RESPAWN_BY_SPAWNID:
871 LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid {} using event({}) has an action type that is not yet supported on AzerothCore ({}), skipped.",
872 e.entryOrGuid, e.event_id, e.GetActionType());
873 return false;
874 default:
875 break;
876 }
877 if (e.target.type < 0 || (e.target.type >= SMART_TARGET_TC_END && e.target.type < SMART_TARGET_AC_START) || e.target.type >= SMART_TARGET_AC_END)
878 {
879 LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid {} using event({}) has an invalid target type ({}), skipped.",
880 e.entryOrGuid, e.event_id, e.GetTargetType());
881 return false;
882 }
883 if (e.target.type == SMART_TARGET_LOOT_RECIPIENTS)
884 {
885 LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid {} using event({}) has a target type that is not yet supported on AzerothCore ({}), skipped.",
886 e.entryOrGuid, e.event_id, e.GetTargetType());
887 return false;
888 }
889 if (e.event.event_phase_mask > SMART_EVENT_PHASE_ALL)
890 {
891 LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid {} using event({}) has invalid phase mask ({}), skipped.", e.entryOrGuid, e.event_id, e.event.event_phase_mask);
892 return false;
893 }
894 if (e.event.event_flags > SMART_EVENT_FLAGS_ALL)
895 {
896 LOG_ERROR("sql.sql", "SmartAIMgr: EntryOrGuid {} using event({}) has invalid event flags ({}), skipped.", e.entryOrGuid, e.event_id, e.event.event_flags);
897 return false;
898 }
899 if (e.GetScriptType() == SMART_SCRIPT_TYPE_TIMED_ACTIONLIST)
900 {
901 e.event.type = SMART_EVENT_UPDATE_OOC;//force default OOC, can change when calling the script!
902 if (!IsMinMaxValid(e, e.event.minMaxRepeat.min, e.event.minMaxRepeat.max))
903 return false;
904
905 if (!IsMinMaxValid(e, e.event.minMaxRepeat.repeatMin, e.event.minMaxRepeat.repeatMax))
906 return false;
907 }
908 else
909 {
910 uint32 type = e.event.type;
911 switch (type)
912 {
913 case SMART_EVENT_UPDATE:
914 case SMART_EVENT_UPDATE_IC:
915 case SMART_EVENT_UPDATE_OOC:
916 case SMART_EVENT_HEALTH_PCT:
917 case SMART_EVENT_MANA_PCT:
918 case SMART_EVENT_TARGET_HEALTH_PCT:
919 case SMART_EVENT_TARGET_MANA_PCT:
920 case SMART_EVENT_DAMAGED:
921 case SMART_EVENT_DAMAGED_TARGET:
922 case SMART_EVENT_RECEIVE_HEAL:
923 if (!IsMinMaxValid(e, e.event.minMaxRepeat.min, e.event.minMaxRepeat.max))
924 return false;
925
926 if (!IsMinMaxValid(e, e.event.minMaxRepeat.repeatMin, e.event.minMaxRepeat.repeatMax))
927 return false;
928 break;
929 case SMART_EVENT_RANGE:
930 if (!IsMinMaxValid(e, e.event.rangeRepeat.minRange, e.event.rangeRepeat.maxRange))
931 return false;
932
933 if (!IsMinMaxValid(e, e.event.rangeRepeat.repeatMin, e.event.rangeRepeat.repeatMax))
934 return false;
935 break;
936 case SMART_EVENT_SPELLHIT:
937 case SMART_EVENT_SPELLHIT_TARGET:
938 if (e.event.spellHit.spell)
