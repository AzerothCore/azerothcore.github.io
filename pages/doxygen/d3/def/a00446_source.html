64Milliseconds randtime(Milliseconds min, Milliseconds max)
65{
66 long long diff = max.count() - min.count();
67 ASSERT(diff >= 0);
68 ASSERT(diff <= (uint32) - 1);
69 return min + Milliseconds(urand(0, diff));
