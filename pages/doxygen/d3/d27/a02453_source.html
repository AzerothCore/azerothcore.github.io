176 void ReplaceMoiraIfSaved()
177 {
178 ObjectGuid* GUIDToReplace = &PriestessGUID; // default to having Moira
179 ObjectGuid* GUIDToSpawn = &MoiraGUID;
180 uint32 NPCEntry = NPC_MOIRA;
181 bool MoiraSaved = true;
182
183 // check if all players saved her.
184 Map::PlayerList const& lPlayers = instance->GetPlayers();
185 if (!lPlayers.IsEmpty())
186 {
187 for (Map::PlayerList::const_iterator itr = lPlayers.begin(); itr != lPlayers.end(); ++itr)
188 {
189 if (Player* player = itr->GetSource())
190 {
191 // set to false if this player hasn't saved her. Another player can't put it to true.
192 MoiraSaved = MoiraSaved && ((player->GetQuestStatus(PRINCESS_QUEST_HORDE) == QUEST_STATUS_REWARDED)
193 || (player->GetQuestStatus(PRINCESS_QUEST_ALLIANCE) == QUEST_STATUS_REWARDED));
194 }
195 }
196 }
197
198 // assign correct GUIDs and spawn targets
199 if (MoiraSaved)
200 {
201 GUIDToReplace = &MoiraGUID;
202 GUIDToSpawn = &PriestessGUID;
203 NPCEntry = NPC_PRIESTESS;
204 }
205
206 if (Creature* CreatureToReplace = instance->GetCreature(*GUIDToReplace))
207 {
208 Creature* NewSpawn = instance->SummonCreature(NPCEntry, CreatureToReplace->GetPosition());
209 CreatureToReplace->RemoveFromWorld();
210 *GUIDToSpawn = NewSpawn->GetGUID();
211 }

397 void OnUnitDeath(Unit* unit) override
398 {
399 uint32 deadSenators = 0;
400 switch (unit->GetEntry())
401 {
402 case NPC_WEAPON_TECHNICIAN:
403 case NPC_DOOMFORGE_ARCANASMITH:
404 case NPC_RAGEREAVER_GOLEM:
405 case NPC_WRATH_HAMMER_CONSTRUCT:
406 ArgelmachAdds.remove(unit->GetGUID());
407 break;
408 case NPC_MAGMUS:
409 SetData(TYPE_IRON_HALL, DONE);
410 break;
411 case NPC_SHADOWFORGE_SENATOR:
412 deadSenators = 1; //hacky, but we cannot count the unit that just died through its state because OnUnitDeath() is called before the state is set.
413 for (const auto &senatorGUID: EmperorSenatorsVector)
414 {
415 if (Creature* senator = instance->GetCreature(senatorGUID))
416 {
417 if (!senator->IsAlive() || senator->isDying())
418 {
419 deadSenators++;
420 }
421 }
422 }
423
424 if (Creature* emperor = instance->GetCreature(EmperorGUID))
425 {
426 // send % of senators that died
427 emperor->AI()->SetData(0, (100 * deadSenators) / EmperorSenatorsVector.size());
428 }
429 break;
430 case NPC_ANGERREL:
431 case NPC_DOPEREL:
432 case NPC_HATEREL:
433 case NPC_VILEREL:
434 case NPC_SEETHREL:
435 case NPC_GLOOMREL:
436 case NPC_DOOMREL:
437 GhostKillCount++;
438 if (GhostKillCount >= 7)
439 {
440 SetData(TYPE_TOMB_OF_SEVEN, DONE);
441 }
442 break;
443 default:
444 break;
445 }

448 void SetData(uint32 type, uint32 data) override
449 {
450 LOG_DEBUG("scripts.ai", "Instance Blackrock Depths: SetData update (Type: {} Data {})", type, data);
451
452 switch (type)
453 {
454 case TYPE_RING_OF_LAW:
455 encounter[0] = data;
456 switch(data)
457 {
458 case IN_PROGRESS:
459 TempSummonGrimstone = instance->SummonCreature(NPC_GRIMSTONE, GrimstonePositon);
460 break;
461 case FAIL:
462 if (TempSummonGrimstone)
463 {
464 TempSummonGrimstone->RemoveFromWorld();
465 TempSummonGrimstone = nullptr;
466 timeRingFail = GameTime::GetGameTime().count();
467 }
468 SetData(TYPE_RING_OF_LAW, NOT_STARTED);
469 break;
470 case DONE:
471 for (const auto& itr : ArenaSpectators)
472 {
473 if (Creature* spectator = instance->GetCreature(itr))
474 {
475 spectator->SetFaction(FACTION_NEUTRAL);
476 spectator->SetReactState(REACT_DEFENSIVE);
477 }
478 }
479 break;
480 default:
481 break;
482 }
483 break;
484 case TYPE_VAULT:
485 encounter[1] = data;
486 break;
487 case TYPE_BAR:
488 if (data == SPECIAL)
489 ++BarAleCount;
490 else
491 encounter[2] = data;
492 break;
493 case TYPE_TOMB_OF_SEVEN:
494 encounter[3] = data;
495 switch (data)
496 {
497 case IN_PROGRESS:
498 HandleGameObject(GoTombExitGUID, false);
499 HandleGameObject(GoTombEnterGUID, false);
500 break;
501 case DONE:
502 DoRespawnGameObject(GoChestGUID, DAY);
503 HandleGameObject(GoTombExitGUID, true);
504 HandleGameObject(GoTombEnterGUID, true);
505 break;
506 }
507 break;
508 case TYPE_LYCEUM:
509 encounter[4] = data;
510 if (data == DONE)
511 {
512 HandleGameObject(GetGuidData(DATA_GOLEM_DOOR_N), true);
513 HandleGameObject(GetGuidData(DATA_GOLEM_DOOR_S), true);
514 if (Creature* magmus = instance->GetCreature(MagmusGUID))
515 {
516 magmus->AI()->Talk(0);
517 }
518 ReplaceMoiraIfSaved(); // Need to place the correct final boss, but we need her to be spawned first.
519 }
520 break;
521 case TYPE_IRON_HALL:
522 encounter[5] = data;
523 switch (data)
524 {
525 case NOT_STARTED:
526 case IN_PROGRESS:
527 for (int i = 0; i < 6; i++)
528 {
529 if (Creature* ironhand = instance->GetCreature(IronhandGUID[i]))
530 {
531 ironhand->AI()->SetData(0, data == IN_PROGRESS);
532 }
533 }
534 break;
535 case DONE:
536 HandleGameObject(GetGuidData(DATA_THRONE_DOOR), true);
537 break;
538 default:
539 break;
540 }
541 break;
542 case DATA_OPEN_COFFER_DOORS:
543 OpenedCoofers += 1;
544 if (OpenedCoofers == 12)
545 {
546 Position pos = {812.15f, -348.91f, -50.579f, 0.7f};
547 if (TempSummon* summon = instance->SummonCreature(NPC_WATCHMAN_DOOMGRIP, pos))
548 summon->SetTempSummonType(TEMPSUMMON_MANUAL_DESPAWN);
549 }
550 break;
551 case DATA_GOLEM_LORD_ARGELMACH_INIT:
552 {
553 if (Creature* argelmach = instance->GetCreature(ArgelmachGUID))
554 {
555 GuidList adds = ArgelmachAdds;
556 for (GuidList::const_iterator itr = adds.begin(); itr != adds.end();)
557 {
558 if (Creature* argelmachAdd = instance->GetCreature(*itr))
559 {
560 if (argelmachAdd->GetEntry() == NPC_WRATH_HAMMER_CONSTRUCT)
561 {
562 argelmachAdd->RemoveAurasDueToSpell(SPELL_STONED);
563 argelmachAdd->AI()->AttackStart(argelmach->GetVictim());
564 itr = adds.erase(itr);
565 }
566 else if (argelmachAdd->GetEntry() == NPC_RAGEREAVER_GOLEM)
567 {
568 if (argelmachAdd->IsWithinDist2d(argelmach, 10.f))
569 {
570 argelmachAdd->RemoveAurasDueToSpell(SPELL_STONED);
571 argelmachAdd->AI()->AttackStart(argelmach->GetVictim());
572 itr = adds.erase(itr);
573 }
574 else
575 ++itr;
576 }
577 else
578 {
579 ++itr;
580 }
581 }
582 else
583 {
584 ++itr;
585 }
586 }
587
588 if (!adds.empty())
589 {
590 argelmach->SetReactState(REACT_PASSIVE);
591 argelmach->SetTarget();
592 argelmach->AI()->SetData(DATA_GOLEM_LORD_ARGELMACH_INIT, IN_PROGRESS);
593 }
594 else
595 {
596 argelmach->AI()->SetData(DATA_GOLEM_LORD_ARGELMACH_INIT, DONE);
597 }
598 }
599 break;
600 }
601 case DATA_GOLEM_LORD_ARGELMACH_ADDS:
602 {
603 if (Creature* argelmach = instance->GetCreature(ArgelmachGUID))
604 {
605 argelmach->HandleEmoteCommand(EMOTE_ONESHOT_SHOUT);
606 argelmach->m_Events.AddEvent(new RestoreAttack(argelmach), argelmach->m_Events.CalculateTime(3000));
607
608 for (ObjectGuid const& argelmachAddGUID : ArgelmachAdds)
609 {
610 if (Creature* argelmachAdd = instance->GetCreature(argelmachAddGUID))
611 {
612 if (!argelmachAdd->IsInCombat())
613 {
614 argelmachAdd->RemoveAurasDueToSpell(SPELL_STONED);
615 argelmachAdd->AI()->AttackStart(argelmach->GetVictim());
616 }
617 }
618 }
619 }
620 break;
621 }
622 default:
623 break;
624 }
625
626 if (data == DONE || GhostKillCount >= 7)
627 {
628 OUT_SAVE_INST_DATA;
629
630 std::ostringstream saveStream;
631 saveStream << encounter[0] << ' ' << encounter[1] << ' ' << encounter[2] << ' '
632 << encounter[3] << ' ' << encounter[4] << ' ' << encounter[5] << ' ' << GhostKillCount;
633
634 str_data = saveStream.str();
635
636 SaveToDB();
637 OUT_SAVE_INST_DATA_COMPLETE;
638 }
639 }

722 void Load(const char* in) override
723 {
724 if (!in)
725 {
726 OUT_LOAD_INST_DATA_FAIL;
727 return;
728 }
729
730 OUT_LOAD_INST_DATA(in);
731
732 std::istringstream loadStream(in);
733 loadStream >> encounter[0] >> encounter[1] >> encounter[2] >> encounter[3]
734 >> encounter[4] >> encounter[5] >> GhostKillCount;
735
736 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
737 if (encounter[i] == IN_PROGRESS)
738 encounter[i] = NOT_STARTED;
739 if (GhostKillCount > 0 && GhostKillCount < 7)
740 GhostKillCount = 0;//reset tomb of seven event
741 if (GhostKillCount >= 7)
742 GhostKillCount = 7;
743

803 bool CheckTombReset(uint32 diff)
804 {
805 bool anyBossAlive = false; // status of the bosses up until the current one
806 for (uint8 i = 0; i < TombEventCounter; i++)
807 {
808 Creature* boss = instance->GetCreature(TombBossGUIDs[i]);
809 if (boss)
810 {
811 anyBossAlive |= boss->IsAlive();
812 if (boss->IsAlive() && boss->IsInCombat())
813 {
814 tombResetTimer = TIMER_TOMB_RESET;
815 return false; // any boss in combat means we shouldn't reset.
816 }
817 }
818 }
819 if (!anyBossAlive) // no boss alive, put reset timer back up
820 {
821 tombResetTimer = TIMER_TOMB_RESET;
822 }
823 tombResetTimer -= diff;
824 return tombResetTimer < diff;
825 }

834 TombOfSevenEvent();
835 }
836 else
837 {
838 TombTimer -= diff;
839 }
840
841 if (CheckTombReset(diff))
842 {
843 TombOfSevenReset();
844 }
845 }
