85 void UpdateAI(uint32 diff) override
86 {
87 if (!UpdateVictim())
88 {
89 return;
90 }
91
92 events.Update(diff);
93
94 if (me->HasUnitState(UNIT_STATE_CASTING))
95 {
96 return;
97 }
98
99 while (uint32 eventId = events.ExecuteEvent())
100 {
101 switch (eventId)
102 {
103 case EVENT_STUN_BOMB:
104 DoCastVictim(SPELL_STUNBOMB);
105 events.ScheduleEvent(EVENT_STUN_BOMB, 14000);
106 break;
107 case EVENT_HOOKED_NET:
108 if (me->IsWithinMeleeRange(me->GetVictim()))
109 {
110 DoCastVictim(SPELL_HOOKEDNET);
111 events.RepeatEvent(16000);
112 }
113 else
114 {
115 events.RepeatEvent(3000);
116 }
117 break;
118 case EVENT_SHOOT:
119 if (!me->IsWithinMeleeRange(me->GetVictim()) && me->IsWithinLOSInMap(me->GetVictim()))
120 {
121 DoCastVictim(SPELL_SHOOT);
122 me->GetMotionMaster()->Clear();
123 SetCombatMovement(false);
124 }
125 else if (!me->IsWithinLOSInMap(me->GetVictim()))
126 {
127 SetCombatMovement(true);
128 me->GetMotionMaster()->Clear();
129 me->GetMotionMaster()->MoveChase(me->GetVictim());
130 }
131 events.RepeatEvent(2000);
132 break;
133 }
134
135 if (me->HasUnitState(UNIT_STATE_CASTING))
136 {
137 return;
138 }
139 }
140
141 DoMeleeAttackIfReady();
142 }
143
144 private:
145 bool _hasDrunkPotion;
146};
