339 void UpdateAI(uint32 diff) override
340 {
341 if (EventInProgress)
342 {
343 Player* pWarrior = ObjectAccessor::GetPlayer(*me, PlayerGUID);
344 if (!pWarrior || me->GetDistance2d(pWarrior) >= 200.0f)
345 {
346 EnterEvadeMode(EVADE_REASON_OTHER);
347 return;
348 }
349
350 if (!pWarrior->IsAlive() && pWarrior->GetQuestStatus(1719) == QUEST_STATUS_INCOMPLETE)
351 {
352 Talk(SAY_TWIGGY_FLATHEAD_DOWN);
353 pWarrior->FailQuest(1719);
354 EnterEvadeMode(EVADE_REASON_OTHER);
355 return;
356 }
357
358 if (!EventGrate)
359 {
360 float x, y, z;
361 pWarrior->GetPosition(x, y, z);
362
363 if (x >= -1684 && x <= -1674 && y >= -4334 && y <= -4324)
364 {
365 pWarrior->AreaExploredOrEventHappens(1719);
366 Talk(SAY_TWIGGY_FLATHEAD_BEGIN, pWarrior);
367
368 for (uint8 i = 0; i < 6; ++i)
369 {
370 Creature* creature = me->SummonCreature(NPC_AFFRAY_CHALLENGER, AffrayChallengerLoc[i], TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 600000);
371 if (!creature)
372 continue;
373 creature->SetFaction(FACTION_FRIENDLY);
374 creature->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
375 creature->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
376 creature->HandleEmoteCommand(EMOTE_ONESHOT_ROAR);
377 AffrayChallenger[i] = creature->GetGUID();
378 }
379 WaveTimer = 5000;
380 ChallengerChecker = 1000;
381 EventGrate = true;
382 }
383 }
384 else
385 {
386 if (ChallengerChecker <= diff)
387 {
388 for (uint8 i = 0; i < 6; ++i)
389 {
390 if (AffrayChallenger[i])
391 {
392 Creature* creature = ObjectAccessor::GetCreature(*me, AffrayChallenger[i]);
393 if ((!creature || !creature->IsAlive()) && !ChallengerDown[i])
394 {
395 Talk(SAY_TWIGGY_FLATHEAD_DOWN);
396 ChallengerDown[i] = true;
397 }
398 }
399 }
400 ChallengerChecker = 1000;
401 }
402 else ChallengerChecker -= diff;
403
404 if (WaveTimer <= diff)
405 {
406 if (Wave < 6 && !EventBigWill)
407 {
408 Talk(SAY_TWIGGY_FLATHEAD_FRAY);
409 Creature* creature = ObjectAccessor::GetCreature(*me, AffrayChallenger[Wave]);
410 if (creature && creature->IsAlive())
411 {
412 creature->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
413 creature->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
414 creature->HandleEmoteCommand(EMOTE_ONESHOT_ROAR);
415 creature->SetFaction(FACTION_MONSTER);
416 creature->AI()->AttackStart(pWarrior);
417 }
418 ++Wave;
419 WaveTimer = 20000;
420 }
421 else if (Wave >= 6 && !EventBigWill)
422 {
423 if (Creature* creature = me->SummonCreature(NPC_BIG_WILL, -1722, -4341, 6.12f, 6.26f, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, 480000))
424 {
425 BigWill = creature->GetGUID();
426 creature->GetMotionMaster()->MovePoint(2, -1682, -4329, 2.79f);
427 creature->HandleEmoteCommand(EMOTE_STATE_READY_UNARMED);
428 EventBigWill = true;
429 WaveTimer = 10000;
430 }
431 }
432 else if (Wave >= 6 && EventBigWill)
433 {
434 Creature* creature = ObjectAccessor::GetCreature(*me, BigWill);
435 if (!creature || !creature->IsAlive())
436 {
437 Talk(SAY_TWIGGY_FLATHEAD_OVER);
438 EnterEvadeMode(EVADE_REASON_OTHER);
439 return;
440 }
441 else // Makes BIG WILL attackable.
442 {
443 creature->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
444 creature->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
445 creature->HandleEmoteCommand(EMOTE_ONESHOT_ROAR);
446 creature->SetFaction(FACTION_MONSTER);
447 creature->AI()->AttackStart(pWarrior);
448 }
449 WaveTimer = 2000;
450 }
451 }
452 else
453 WaveTimer -= diff;
454 }
455 }
