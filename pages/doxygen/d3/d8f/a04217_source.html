36 bool OnGossipHello(Player* player, Creature* creature) override
37 {
38 if (IsEventActive(HALLOWEEN_EVENTID) && !player->HasAura(SPELL_TRICKED_OR_TREATED))
39 {
40 AddGossipItemFor(player, GOSSIP_MENU_EVENT, 0, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + HALLOWEEN_EVENTID);
41 }
42
43 if (creature->IsQuestGiver())
44 {
45 player->PrepareQuestMenu(creature->GetGUID());
46 }
47
48 if (creature->IsVendor())
49 {
50 AddGossipItemFor(player, GOSSIP_MENU, 2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_TRADE);
51 }
52
53 if (creature->IsInnkeeper())
54 {
55 AddGossipItemFor(player, GOSSIP_MENU, 1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INN);
56 }
57
58 player->TalkedToCreature(creature->GetEntry(), creature->GetGUID());
59 SendGossipMenuFor(player, player->GetGossipTextId(creature), creature->GetGUID());
60 return true;
61 }
62
63 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
64 {
65 ClearGossipMenuFor(player);
66 if (action == GOSSIP_ACTION_INFO_DEF + HALLOWEEN_EVENTID && IsEventActive(HALLOWEEN_EVENTID) && !player->HasAura(SPELL_TRICKED_OR_TREATED))
67 {
68 player->CastSpell(player, SPELL_TRICKED_OR_TREATED, true);
69 creature->CastSpell(player, roll_chance_i(50) ? SPELL_TRICK : SPELL_TREAT, true);
70
71 CloseGossipMenuFor(player);
72 return true;
73 }
74
75 CloseGossipMenuFor(player);
76
77 switch (action)
78 {
79 case GOSSIP_ACTION_TRADE:
80 player->GetSession()->SendListInventory(creature->GetGUID());
81 break;
82 case GOSSIP_ACTION_INN:
83 player->SetBindPoint(creature->GetGUID());
84 break;
85 }
86 return true;
87 }
