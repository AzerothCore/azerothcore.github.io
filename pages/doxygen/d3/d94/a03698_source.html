81 void UpdateAI(uint32 diff) override
82 {
83 if (!UpdateVictim())
84 return;
85
86 events.Update(diff);
87
88 if (me->HasUnitState(UNIT_STATE_CASTING))
89 return;
90
91 switch(events.ExecuteEvent())
92 {
93 case 0:
94 break;
95 case EVENT_SPELL_FIREBOLT:
96 me->CastSpell(me->GetVictim(), SPELL_FIREBOLT, false);
97 events.RepeatEvent(urand(5000, 13000));
98 break;
99 case EVENT_SPELL_FLAME_BREATH:
100 me->CastSpell(me->GetVictim(), SPELL_FLAME_BREATH, false);
101 events.RepeatEvent(urand(10000, 15000));
102 break;
103 case EVENT_SPELL_LAVA_BURN:
104 me->CastSpell(me->GetVictim(), SPELL_LAVA_BURN, false);
105 events.RepeatEvent(urand(14000, 20000));
106 break;
107 case EVENT_SPELL_CAUTERIZING_FLAMES:
108 me->CastSpell((Unit*)nullptr, SPELL_FLAME_BREATH, false);
109 events.RepeatEvent(urand(10000, 16000));
110 break;
111 }
112
113 DoMeleeAttackIfReady();
114 }
