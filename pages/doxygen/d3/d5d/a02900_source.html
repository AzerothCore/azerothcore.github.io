88 npc_midsummer_torch_target(Creature* creature) : ScriptedAI(creature)

102 std::vector<Position> posVec;

104 uint8 maxCount;

106 void SetPlayerGUID(ObjectGuid guid, uint8 cnt)
107 {
108 playerGUID = guid;
109 maxCount = cnt;

111
112 bool CanBeSeen(Player const* seer) override
113 {
114 return seer->GetGUID() == playerGUID;
115 }
116
117 void SpellHit(Unit* caster, SpellInfo const* spellInfo) override
118 {
119 if (posVec.empty())
120 return;
121 // Triggered spell from torch
122 if (spellInfo->Id == 46054 && caster->GetTypeId() == TYPEID_PLAYER)
123 {
124 me->CastSpell(me, 45724, true); // hit visual anim
125 if (++counter >= maxCount)
126 {
127 caster->CastSpell(caster, (caster->ToPlayer()->GetTeamId() ? 46651 : 45719), true); // quest complete spell
128 me->DespawnOrUnsummon(1);
129 return;
130 }
131
132 teleTimer = 1;
133 }
134 }
135
136 void UpdateAI(uint32 diff) override
137 {
138 if (startTimer)
139 {
140 startTimer += diff;
141 if (startTimer >= 200)
142 {
143 startTimer = 0;
144 FillPositions();
145 SelectPosition();
146 }
147 }
148 if (teleTimer)
149 {
150 teleTimer += diff;
151 if (teleTimer >= 750 && teleTimer < 10000)
152 {
153 teleTimer = 10000;
154 SelectPosition();
155 }
156 else if (teleTimer >= 10500)
157 {
158 if (Player* plr = ObjectAccessor::GetPlayer(*me, playerGUID))
159 plr->UpdateTriggerVisibility();
160
161 teleTimer = 0;
162 }
163 }
164 }
165
166 void FillPositions()
167 {
168 std::list<GameObject*> gobjList;
169 me->GetGameObjectListWithEntryInGrid(gobjList, 187708 /*TORCH_GO*/, 30.0f);
170 for (std::list<GameObject*>::const_iterator itr = gobjList.begin(); itr != gobjList.end(); ++itr)
171 {
172 Position pos;
173 pos.Relocate(*itr);
174 posVec.push_back(pos);
175 }
176 }
177
178 void SelectPosition()
179 {
180 if (posVec.empty())
181 return;
182 int8 num = urand(0, posVec.size() - 1);
183 Position pos;
184 pos.Relocate(posVec.at(num));
185 me->m_last_notify_position.Relocate(0.0f, 0.0f, 0.0f);
186 me->m_last_notify_mstime = GameTime::GetGameTimeMS().count() + 10000;
187
188 me->NearTeleportTo(pos.GetPositionX(), pos.GetPositionY(), pos.GetPositionZ(), pos.GetOrientation());
189 }

338class spell_midsummer_fling_torch : public SpellScript
339{
340 PrepareSpellScript(spell_midsummer_fling_torch);
341
342 bool handled;
343 bool Load() override { handled = false; return true; }
344
345 SpellCastResult CheckCast()
346 {
347 GetCaster()->GetCreaturesWithEntryInRange(_crList, 100.0f, NPC_TORCH_TARGET);
348 if (_crList.empty())
349 {
350 return SPELL_FAILED_NOT_HERE;
351 }
352
353 return SPELL_CAST_OK;
354 }
355
356 void ThrowNextTorch(Unit* caster)
357 {
358 uint8 rand = urand(0, _crList.size() - 1);
359 Position pos;
360 pos.Relocate(0.0f, 0.0f, 0.0f);
361 for (std::list<Creature*>::const_iterator itr = _crList.begin(); itr != _crList.end(); ++itr, --rand)
362 {
363 if (caster->GetDistance(*itr) < 5)
364 {
365 if (!rand)
366 rand++;
367 continue;
368 }
369
370 if (!rand)
371 {
372 pos.Relocate(*itr);
373 break;
374 }
375 }
376
377 // we have any pos
378 if (pos.GetPositionX())
379 caster->CastSpell(pos.GetPositionX(), pos.GetPositionY(), pos.GetPositionZ(), SPELL_FLING_TORCH, true);
380 }
381
382 void HandleFinish()
383 {
384 Unit* caster = GetCaster();
385 if (!caster || !caster->ToPlayer()) // caster cant be null, but meh :p
386 return;
387
388 if (GetSpellInfo()->Id != SPELL_FLING_TORCH_DUMMY)
389 {
390 if (!handled)
391 if (const WorldLocation* loc = GetExplTargetDest())
392 {
393 caster->CastSpell(loc->GetPositionX(), loc->GetPositionY(), loc->GetPositionZ(), SPELL_MISSED_TORCH, true);
394 caster->RemoveAurasDueToSpell(SPELL_TORCH_COUNTER);
395 }
396 return;
397 }
398
399 ThrowNextTorch(caster);
400 }
401
402 void HandleScript(SpellEffIndex effIndex)
403 {
404 PreventHitDefaultEffect(effIndex);
405 if (Player* target = GetHitPlayer())
406 {
407 if (target->GetGUID() != GetCaster()->GetGUID())
408 return;
409
410 handled = true;
411 if (Aura* aur = target->GetAura(SPELL_TORCH_COUNTER))
412 {
413 aur->ModStackAmount(1);
414 uint8 count = 4;
415 if (target->GetQuestStatus(target->GetTeamId() ? 11925 : 11924) == QUEST_STATUS_INCOMPLETE) // More Torch Catching quests
416 count = 10;
417
418 if (aur->GetStackAmount() >= count)
419 {
420 //target->CastSpell(target, 46711, true); // Set Flag: all torch returning quests are complete
421 target->CastSpell(target, (target->GetTeamId() ? 46654 : 46081), true); // Quest completion
422 aur->SetDuration(1);
423 return;
424 }
425 }
426 else
427 target->CastSpell(target, SPELL_TORCH_COUNTER, true);
428
429 ThrowNextTorch(GetCaster());
430 }
431 }
432
433 void Register() override
434 {
435 AfterCast += SpellCastFn(spell_midsummer_fling_torch::HandleFinish);
436 if (m_scriptSpellId == 45671)
437 {
438 OnCheckCast += SpellCheckCastFn(spell_midsummer_fling_torch::CheckCast);
439 OnEffectHitTarget += SpellEffectFn(spell_midsummer_fling_torch::HandleScript, EFFECT_0, SPELL_EFFECT_SCRIPT_EFFECT);
440 }
441 }

443private:
444 std::list<Creature*> _crList;
