99 void UpdateAI(uint32 diff) override
100 {
101 if (!UpdateVictim())
102 return;
103
104 events.Update(diff);
105 if (me->HasUnitState(UNIT_STATE_CASTING))
106 return;
107
108 switch (events.ExecuteEvent())
109 {
110 case EVENT_SPELL_ACID:
111 me->CastSpell(me->GetVictim(), SPELL_ACID_SPRAY, false);
112 events.RepeatEvent(urand(15000, 23000));
113 break;
114 case EVENT_SPELL_EXPLODING:
115 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
116 me->CastSpell(target, SPELL_EXPLODING_BREAKER, false);
117 events.RepeatEvent(urand(7000, 11000));
118 break;
119 case EVENT_SPELL_DOMINATION:
120 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
121 me->CastSpell(target, SPELL_DOMINATION, false);
122 events.RepeatEvent(120000);
123 break;
124 case EVENT_SPELL_KNOCKDOWN:
125 me->CastSpell(me->GetVictim(), SPELL_KNOCKDOWN, false);
126 events.RepeatEvent(urand(4000, 12000));
127 break;
128 }
129
130 DoMeleeAttackIfReady();
131 }
132 };
