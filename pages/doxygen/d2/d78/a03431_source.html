42 bool OnGossipHello(Player* player, GameObject* go) override
43 {
44 uint32 gossipMenuId = go->GetGOInfo()->GetGossipMenuId();

56 player, gossipMenuId, ICC_TELEPORT_GOSSIP_OPT_ORATORY_OF_THE_DAMNED,
57 GOSSIP_SENDER_ICC_PORT, ORATORY_OF_THE_DAMNED_TELEPORT
58 ); // M_PI + M_PI/6
59 if (instance->GetBossState(DATA_LADY_DEATHWHISPER) == DONE && go->GetEntry() != 202243)
60 AddGossipItemFor(
61 player, gossipMenuId, ICC_TELEPORT_GOSSIP_OPT_RAMPART_OF_SKULLS,
62 GOSSIP_SENDER_ICC_PORT, RAMPART_OF_SKULLS_TELEPORT
63 ); // M_PI/6
64 if (instance->GetBossState(DATA_ICECROWN_GUNSHIP_BATTLE) == DONE && go->GetEntry() != 202244)
65 AddGossipItemFor(
66 player, gossipMenuId, ICC_TELEPORT_GOSSIP_OPT_DEATHBRINGERS_RISE,
67 GOSSIP_SENDER_ICC_PORT, DEATHBRINGER_S_RISE_TELEPORT
68 ); // M_PI/6
69 if (instance->GetData(DATA_COLDFLAME_JETS) == DONE && go->GetEntry() != 202235)
70 AddGossipItemFor(
71 player, gossipMenuId, ICC_TELEPORT_GOSSIP_OPT_UPPER_SPIRE,
72 GOSSIP_SENDER_ICC_PORT, UPPER_SPIRE_TELEPORT
73 ); // M_PI/6
74 if (instance->GetBossState(DATA_VALITHRIA_DREAMWALKER) == DONE && instance->GetBossState(DATA_SINDRAGOSA_GAUNTLET) == DONE && go->GetEntry() != 202246)
75 AddGossipItemFor(
76 player, gossipMenuId, ICC_TELEPORT_GOSSIP_OPT_SINDRAGOSAS_LAIR,
77 GOSSIP_SENDER_ICC_PORT, SINDRAGOSA_S_LAIR_TELEPORT
78 ); // M_PI*3/2 + M_PI/6
79 }
80
81 SendGossipMenuFor(player, player->GetGossipTextId(go), go->GetGUID());
82 return true;
83 }
84
85 bool OnGossipSelect(Player* player, GameObject* /*go*/, uint32 sender, uint32 action) override
86 {
87 ClearGossipMenuFor(player);
88 CloseGossipMenuFor(player);
89 SpellInfo const* spell = sSpellMgr->GetSpellInfo(action);
90 if (!spell)
91 return false;
92
93 if (player->IsInCombat())
94 {
95 Spell::SendCastResult(player, spell, 0, SPELL_FAILED_AFFECTING_COMBAT);
96 return true;
97 }
98
99 if (sender == GOSSIP_SENDER_ICC_PORT)
100 player->CastSpell(player, spell, false);
101
102 return true;
103 }

107{
108public:

110
111 bool OnTrigger(Player* player, AreaTrigger const* /*areaTrigger*/) override
112 {
113 if (player->IsInCombat())
114 {
115 if (SpellInfo const* spell = sSpellMgr->GetSpellInfo(FROZEN_THRONE_TELEPORT))
116 Spell::SendCastResult(player, spell, 0, SPELL_FAILED_AFFECTING_COMBAT);
117 return true;
118 }
119
120 if (InstanceScript* instance = player->GetInstanceScript())
121 if (instance->GetBossState(DATA_PROFESSOR_PUTRICIDE) == DONE &&
122 instance->GetBossState(DATA_BLOOD_QUEEN_LANA_THEL) == DONE &&
123 instance->GetBossState(DATA_SINDRAGOSA) == DONE &&
124 instance->GetBossState(DATA_THE_LICH_KING) != IN_PROGRESS)
125 player->CastSpell(player, FROZEN_THRONE_TELEPORT, false);
126
127 return true;
128 }
