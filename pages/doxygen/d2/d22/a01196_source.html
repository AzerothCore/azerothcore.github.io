189void Object::BuildCreateUpdateBlockForPlayer(UpdateData* data, Player* target) const
190{
191 if (!target)
192 return;
193
194 uint8 updatetype = UPDATETYPE_CREATE_OBJECT;
195 uint16 flags = m_updateFlag;
196
198 if (target == this) // building packet for yourself
199 flags |= UPDATEFLAG_SELF;
200
201 if (flags & UPDATEFLAG_STATIONARY_POSITION)
202 {
203 // UPDATETYPE_CREATE_OBJECT2 dynamic objects, corpses...
204 if (isType(TYPEMASK_DYNAMICOBJECT) || isType(TYPEMASK_CORPSE) || isType(TYPEMASK_PLAYER))
205 updatetype = UPDATETYPE_CREATE_OBJECT2;
206
207 // UPDATETYPE_CREATE_OBJECT2 for pets...
208 if (target->GetPetGUID() == GetGUID())
209 updatetype = UPDATETYPE_CREATE_OBJECT2;
210
211 // UPDATETYPE_CREATE_OBJECT2 for some gameobject types...
212 if (isType(TYPEMASK_GAMEOBJECT))
213 {
214 switch (((GameObject*)this)->GetGoType())
215 {
216 case GAMEOBJECT_TYPE_TRAP:
217 case GAMEOBJECT_TYPE_DUEL_ARBITER:
218 case GAMEOBJECT_TYPE_FLAGSTAND:
219 case GAMEOBJECT_TYPE_FLAGDROP:
220 updatetype = UPDATETYPE_CREATE_OBJECT2;
221 break;
222 default:
223 if (((GameObject*)this)->GetOwner())
224 updatetype = UPDATETYPE_CREATE_OBJECT2;
225 break;
226 }
227 }
228
229 if (isType(TYPEMASK_UNIT))
230 {
231 if (((Unit*)this)->GetVictim())
232 flags |= UPDATEFLAG_HAS_TARGET;
233 }
234 }
235
236 ByteBuffer buf(500);
237 buf << (uint8)updatetype;
238 buf << GetPackGUID();
239 buf << (uint8)m_objectTypeId;
240
241 BuildMovementUpdate(&buf, flags);
242 BuildValuesUpdate(updatetype, &buf, target);
243 data->AddUpdateBlock(buf);
244}

343void Object::BuildMovementUpdate(ByteBuffer* data, uint16 flags) const
344{
345 Unit const* unit = nullptr;
346 WorldObject const* object = nullptr;
347
348 if (isType(TYPEMASK_UNIT))
349 unit = ToUnit();
350 else
351 object = ((WorldObject*)this);
352
353 *data << uint16(flags); // update flags
354
355 // 0x20
356 if (flags & UPDATEFLAG_LIVING)
357 {
358 unit->BuildMovementPacket(data);
359
360 *data << unit->GetSpeed(MOVE_WALK)
361 << unit->GetSpeed(MOVE_RUN)
362 << unit->GetSpeed(MOVE_RUN_BACK)
363 << unit->GetSpeed(MOVE_SWIM)
364 << unit->GetSpeed(MOVE_SWIM_BACK)
365 << unit->GetSpeed(MOVE_FLIGHT)
366 << unit->GetSpeed(MOVE_FLIGHT_BACK)
367 << unit->GetSpeed(MOVE_TURN_RATE)
368 << unit->GetSpeed(MOVE_PITCH_RATE);
369
370 // 0x08000000
371 if (unit->m_movementInfo.GetMovementFlags() & MOVEMENTFLAG_SPLINE_ENABLED)
372 {
373 Movement::PacketBuilder::WriteCreate(*unit->movespline, *data);
374 }
375 }
376 else
377 {
378 if (flags & UPDATEFLAG_POSITION)
379 {
380 Transport* transport = object->GetTransport();
381
382 if (transport)
383 *data << transport->GetPackGUID();
384 else
385 *data << uint8(0);
386
387 *data << object->GetPositionX();
388 *data << object->GetPositionY();
389 *data << object->GetPositionZ();
390
391 if (transport)
392 {
393 *data << object->GetTransOffsetX();
394 *data << object->GetTransOffsetY();
395 *data << object->GetTransOffsetZ();
396 }
397 else
398 {
399 *data << object->GetPositionX();
400 *data << object->GetPositionY();
401 *data << object->GetPositionZ();
402 }
403
404 *data << object->GetOrientation();
405
406 if (GetTypeId() == TYPEID_CORPSE)
407 *data << float(object->GetOrientation());
408 else
409 *data << float(0);
410 }
411 else
412 {
413 // 0x40
414 if (flags & UPDATEFLAG_STATIONARY_POSITION)
415 {
416 *data << object->GetStationaryX();
417 *data << object->GetStationaryY();
418 *data << object->GetStationaryZ();
419 *data << object->GetStationaryO();
420 }
421 }
422 }
423
424 // 0x8
425 if (flags & UPDATEFLAG_UNKNOWN)
426 {
427 *data << uint32(0);
428 }
429
430 // 0x10
431 if (flags & UPDATEFLAG_LOWGUID)
432 {
433 switch (GetTypeId())
434 {
435 case TYPEID_OBJECT:
436 case TYPEID_ITEM:
437 case TYPEID_CONTAINER:
438 case TYPEID_GAMEOBJECT:
439 case TYPEID_DYNAMICOBJECT:
440 case TYPEID_CORPSE:
441 *data << uint32(GetGUID().GetCounter());
442 break;
445 case TYPEID_UNIT:
446 *data << uint32(0x0000000B); // unk
447 break;
448 case TYPEID_PLAYER:
449 if (flags & UPDATEFLAG_SELF)
450 *data << uint32(0x0000002F); // unk
451 else
452 *data << uint32(0x00000008); // unk
453 break;
454 default:
455 *data << uint32(0x00000000); // unk
456 break;
457 }
458 }
459
460 // 0x4
461 if (flags & UPDATEFLAG_HAS_TARGET)
462 {
463 if (Unit* victim = unit->GetVictim())
464 *data << victim->GetPackGUID();
465 else
466 *data << uint8(0);
467 }
468
469 // 0x2
470 if (flags & UPDATEFLAG_TRANSPORT)
471 {
472 GameObject const* go = ToGameObject();
473 if (go && go->ToTransport())
474 *data << uint32(go->ToTransport()->GetPathProgress());
475 else
476 *data << uint32(0);
477 }
478
479 // 0x80
480 if (flags & UPDATEFLAG_VEHICLE)
481 {
483 *data << uint32(unit->GetVehicleKit()->GetVehicleInfo()->m_ID);
484 if (unit->HasUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT))
485 *data << float(unit->GetTransOffsetO());
486 else
487 *data << float(unit->GetOrientation());
488 }
489
490 // 0x200
491 if (flags & UPDATEFLAG_ROTATION)
492 {
493 *data << int64(ToGameObject()->GetPackedLocalRotation());
494 }
495}

497void Object::BuildValuesUpdate(uint8 updateType, ByteBuffer* data, Player* target) const
498{
499 if (!target)
500 return;
501
502 ByteBuffer fieldBuffer;
503 UpdateMask updateMask;
504 updateMask.SetCount(m_valuesCount);
505
506 uint32* flags = nullptr;
507 uint32 visibleFlag = GetUpdateFieldData(target, flags);
508
509 for (uint16 index = 0; index < m_valuesCount; ++index)
510 {
511 if (_fieldNotifyFlags & flags[index] ||
512 ((updateType == UPDATETYPE_VALUES ? _changesMask.GetBit(index) : m_uint32Values[index]) && (flags[index] & visibleFlag)))
513 {
514 updateMask.SetBit(index);
515 fieldBuffer << m_uint32Values[index];
516 }
517 }
518
519 *data << uint8(updateMask.GetBlockCount());
520 updateMask.AppendToPacket(data);
521 data->append(fieldBuffer);
522}

559uint32 Object::GetUpdateFieldData(Player const* target, uint32*& flags) const
560{
561 uint32 visibleFlag = UF_FLAG_PUBLIC;
562
563 if (target == this)
564 visibleFlag |= UF_FLAG_PRIVATE;
565
566 switch (GetTypeId())
567 {
568 case TYPEID_ITEM:
569 case TYPEID_CONTAINER:
570 flags = ItemUpdateFieldFlags;
571 if (((Item*)this)->GetOwnerGUID() == target->GetGUID())
572 visibleFlag |= UF_FLAG_OWNER | UF_FLAG_ITEM_OWNER;
573 break;
574 case TYPEID_UNIT:
575 case TYPEID_PLAYER:
576 {
577 Player* plr = ToUnit()->GetCharmerOrOwnerPlayerOrPlayerItself();
578 flags = UnitUpdateFieldFlags;
579 if (ToUnit()->GetOwnerGUID() == target->GetGUID())
580 visibleFlag |= UF_FLAG_OWNER;
581
582 if (HasDynamicFlag(UNIT_DYNFLAG_SPECIALINFO))
583 if (ToUnit()->HasAuraTypeWithCaster(SPELL_AURA_EMPATHY, target->GetGUID()))
584 visibleFlag |= UF_FLAG_SPECIAL_INFO;
585
586 if (plr && plr->IsInSameRaidWith(target))
587 visibleFlag |= UF_FLAG_PARTY_MEMBER;
588 break;
589 }
590 case TYPEID_GAMEOBJECT:
591 flags = GameObjectUpdateFieldFlags;
592 if (ToGameObject()->GetOwnerGUID() == target->GetGUID())
593 visibleFlag |= UF_FLAG_OWNER;
594 break;
595 case TYPEID_DYNAMICOBJECT:
596 flags = DynamicObjectUpdateFieldFlags;
597 if (((DynamicObject*)this)->GetCasterGUID() == target->GetGUID())
598 visibleFlag |= UF_FLAG_OWNER;
599 break;
600 case TYPEID_CORPSE:
601 flags = CorpseUpdateFieldFlags;
602 if (ToCorpse()->GetOwnerGUID() == target->GetGUID())
603 visibleFlag |= UF_FLAG_OWNER;
604 break;
605 case TYPEID_OBJECT:
606 break;
607 }
608
609 return visibleFlag;
610}

612bool Object::_LoadIntoDataField(std::string const& data, uint32 startOffset, uint32 count)
613{
614 if (data.empty())
615 return false;
616
617 std::vector<std::string_view> tokens = Acore::Tokenize(data, ' ', false);
618
619 if (tokens.size() != count)
620 return false;
621
622 for (uint32 index = 0; index < count; ++index)
623 {
624 Optional<uint32> val = Acore::StringTo<uint32>(tokens[index]);
625 if (!val)
626 {
627 return false;
628 }
629
630 m_uint32Values[startOffset + index] = *val;
631 _changesMask.SetBit(startOffset + index);
632 }
633
634 return true;
635}

1207{
1208 float sizefactor = useBoundingRadius ? GetObjectSize() + obj->GetObjectSize() : 0.0f;
1209 float maxdist = dist2compare + sizefactor;
1210
1211 if (m_transport && obj->GetTransport() && obj->GetTransport()->GetGUID() == m_transport->GetGUID())
1212 {

1214 float dty = m_movementInfo.transport.pos.m_positionY - obj->m_movementInfo.transport.pos.m_positionY;

1216 if (is3D)
1217 {
1218 float dtz = m_movementInfo.transport.pos.m_positionZ - obj->m_movementInfo.transport.pos.m_positionZ;
1219 disttsq += dtz * dtz;
1220 }
1221 return disttsq < (maxdist * maxdist);
1222 }
1223
1224 float dx = GetPositionX() - obj->GetPositionX();
1225 float dy = GetPositionY() - obj->GetPositionY();
1226 float distsq = dx * dx + dy * dy;
1227 if (is3D)
1228 {
1229 float dz = GetPositionZ() - obj->GetPositionZ();
1230 distsq += dz * dz;
1231 }
1232
1233 return distsq < maxdist * maxdist;

1470bool WorldObject::IsInBetween(WorldObject const* obj1, WorldObject const* obj2, float size) const
1471{
1472 if (!obj1 || !obj2)
1473 return false;
1474
1475 if (!size)
1476 size = GetObjectSize() / 2;
1477
1478 float pdist = obj1->GetExactDist2dSq(obj2) + size / 2.0f;
1479 if (GetExactDist2dSq(obj1) >= pdist || GetExactDist2dSq(obj2) >= pdist)
1480 return false;
1481
1482 if (G3D::fuzzyEq(obj1->GetPositionX(), obj2->GetPositionX()))
1483 return GetPositionX() >= obj1->GetPositionX() - size && GetPositionX() <= obj1->GetPositionX() + size;
1484
1485 float A = (obj2->GetPositionY() - obj1->GetPositionY()) / (obj2->GetPositionX() - obj1->GetPositionX());
1486 float B = -1;
1487 float C = obj1->GetPositionY() - A * obj1->GetPositionX();
1488 float dist = std::fabs(A * GetPositionX() + B * GetPositionY() + C) / std::sqrt(A * A + B * B);
1489 return dist <= size;
1490}

1727{

1729 return true;
1730
1731 if (obj->IsNeverVisible() || CanNeverSee(obj))
1732 return false;
1733
1734 if (obj->IsAlwaysVisibleFor(this) || CanAlwaysSee(obj))
1735 return true;
1736
1737 // Creature scripts
1738 if (Creature const* cObj = obj->ToCreature())
1739 {
1740 if (Player const* player = this->ToPlayer())
1741 {
1742 if (cObj->IsAIEnabled && !cObj->AI()->CanBeSeen(player))
1743 {
1744 return false;
1745 }
1746

1748
1749 if (!sConditionMgr->IsObjectMeetToConditions((WorldObject*)this, conditions))
1750 {
1751 return false;
1752 }
1753 }
1754 }
1755
1756 // Gameobject scripts
1757 if (GameObject const* goObj = obj->ToGameObject())
1758 if (this->ToPlayer() && !goObj->AI()->CanBeSeen(this->ToPlayer()))
1759 return false;
1760
1761 // pussywizard: arena spectator
1762 if (obj->GetTypeId() == TYPEID_PLAYER)
1763 if (((Player const*)obj)->IsSpectator() && ((Player const*)obj)->FindMap()->IsBattleArena())
1764 return false;
1765

1768 {

1772 {

1782 }
1783 }
1784
1785 // our additional checks
1786 if (Unit const* target = obj->ToUnit())
1787 {

1789 if (Unit const* vehicle = target->GetVehicleBase())

1791 return false;

1802 viewpoint = thisPlayer->GetViewpoint();
1803
1804 if (thisPlayer->GetFarSightDistance() && !thisPlayer->isInFront(obj))

1806 return false;
1807 }
1808 }

1813 }

1820 return true;
1821 }
1822 else
1823 return m_serverSideVisibilityDetect.GetValue(SERVERSIDE_VISIBILITY_GM) >= obj->m_serverSideVisibility.GetValue(SERVERSIDE_VISIBILITY_GM);
1824
1825 // Ghost players, Spirit Healers, and some other NPCs
1826 if (!corpseVisibility && !(obj->m_serverSideVisibility.GetValue(SERVERSIDE_VISIBILITY_GHOST) & m_serverSideVisibilityDetect.GetValue(SERVERSIDE_VISIBILITY_GHOST)))
1827 {
1828 // Alive players can see dead players in some cases, but other objects can't do that
1829 if (Player const* thisPlayer = ToPlayer())
1830 {
1831 if (Player const* objPlayer = obj->ToPlayer())
1832 {
1833 if (thisPlayer->GetTeamId() != objPlayer->GetTeamId() || !thisPlayer->IsGroupVisibleFor(objPlayer))
1834 return false;
1835 }
1836 else
1837 return false;
1838 }
1839 else
1840 return false;
1841 }
1842
1843 if (obj->IsInvisibleDueToDespawn())
1844 return false;
1845
1846 // pussywizard: arena spectator
1847 if (this->GetTypeId() == TYPEID_PLAYER)
1848 if (((Player const*)this)->IsSpectator() && ((Player const*)this)->FindMap()->IsBattleArena() && (obj->m_invisibility.GetFlags() || obj->m_stealth.GetFlags()))
1849 return false;
1850
1851 if (!CanDetect(obj, ignoreStealth, !distanceCheck, checkAlert))
1852 return false;
1853
1854 return true;

1897bool WorldObject::CanDetectInvisibilityOf(WorldObject const* obj) const
1898{
1899 uint32 mask = obj->m_invisibility.GetFlags() & m_invisibilityDetect.GetFlags();
1900 // xinef: include invisible flags of caster in the mask, 2 invisible objects should be able to detect eachother
1901 mask |= obj->m_invisibility.GetFlags() & m_invisibility.GetFlags();
1902
1903 // Check for not detected types
1904 if (mask != obj->m_invisibility.GetFlags())
1905 return false;
1906
1907 // It isn't possible in invisibility to detect something that can't detect the invisible object
1908 // (it's at least true for spell: 66)
1909 // It seems like that only Units are affected by this check (couldn't see arena doors with preparation invisibility)
1910 if (obj->ToUnit())
1911 {
1912 uint32 objMask = m_invisibility.GetFlags() & obj->m_invisibilityDetect.GetFlags();
1913 // xinef: include invisible flags of caster in the mask, 2 invisible objects should be able to detect eachother
1914 objMask |= m_invisibility.GetFlags() & obj->m_invisibility.GetFlags();
1915 if (objMask != m_invisibility.GetFlags())
1916 return false;
1917 }
1918
1919 for (uint32 i = 0; i < TOTAL_INVISIBILITY_TYPES; ++i)
1920 {
1921 if (!(mask & (1 << i)))
1922 continue;
1923
1924 // xinef: visible for the same invisibility type:
1925 if (m_invisibility.GetValue(InvisibilityType(i)) && obj->m_invisibility.GetValue(InvisibilityType(i)))
1926 continue;
1927
1928 int32 objInvisibilityValue = obj->m_invisibility.GetValue(InvisibilityType(i));
1929 int32 ownInvisibilityDetectValue = m_invisibilityDetect.GetValue(InvisibilityType(i));
1930
1931 // Too low value to detect
1932 if (ownInvisibilityDetectValue < objInvisibilityValue)
1933 return false;
1934 }
1935
1936 return true;

1939bool WorldObject::CanDetectStealthOf(WorldObject const* obj, bool checkAlert) const
1940{
1941 // Combat reach is the minimal distance (both in front and behind),
1942 // and it is also used in the range calculation.
1943 // One stealth point increases the visibility range by 0.3 yard.
1944
1945 if (!obj->m_stealth.GetFlags())
1946 return true;
1947
1948 // dead players shouldnt be able to detect stealth on arenas
1949 if (isType(TYPEMASK_PLAYER))
1950 if (!ToPlayer()->IsAlive())
1951 return false;
1952
1953 float distance = GetExactDist(obj);
1954 float combatReach = 0.0f;
1955
1956 if (isType(TYPEMASK_UNIT))
1957 combatReach = ((Unit*)this)->GetCombatReach();
1958
1959 if (distance < combatReach)
1960 return true;
1961
1962 if (!HasInArc(M_PI, obj))
1963 return false;
1964
1965 for (uint32 i = 0; i < TOTAL_STEALTH_TYPES; ++i)
1966 {
1967 if (!(obj->m_stealth.GetFlags() & (1 << i)))
1968 continue;
1969
1970 if (isType(TYPEMASK_UNIT))
1971 if (((Unit*)this)->HasAuraTypeWithMiscvalue(SPELL_AURA_DETECT_STEALTH, i))
1972 return true;
1973
1974 // Starting points
1975 int32 detectionValue = 30;
1976
1977 // Level difference: 5 point / level, starting from level 1.
1978 // There may be spells for this and the starting points too, but
1979 // not in the DBCs of the client.
1980 detectionValue += int32(getLevelForTarget(obj) - 1) * 5;
1981
1982 // Apply modifiers
1983 detectionValue += m_stealthDetect.GetValue(StealthType(i));
1984 if (obj->isType(TYPEMASK_GAMEOBJECT))
1985 {
1986 detectionValue += 30; // pussywizard: increase detection range for gameobjects (ie. traps)
1987 if (Unit* owner = ((GameObject*)obj)->GetOwner())
1988 detectionValue -= int32(owner->getLevelForTarget(this) - 1) * 5;
1989 }
1990
1991 detectionValue -= obj->m_stealth.GetValue(StealthType(i));
1992
1993 // Calculate max distance
1994 float visibilityRange = float(detectionValue) * 0.3f + combatReach;
1995
1996 Unit const* unit = ToUnit();
1997
1998 // If this unit is an NPC then player detect range doesn't apply
1999 if (unit && unit->GetTypeId() == TYPEID_PLAYER && visibilityRange > MAX_PLAYER_STEALTH_DETECT_RANGE)
2000 visibilityRange = MAX_PLAYER_STEALTH_DETECT_RANGE;
2001
2002 if (checkAlert)
2003 visibilityRange += (visibilityRange * 0.08f) + 1.5f;
2004
2005 Unit const* targetUnit = obj->ToUnit();
2006
2007 // If checking for alert, and creature's visibility range is greater than aggro distance, No alert
2008 if (checkAlert && unit && unit->ToCreature() && visibilityRange >= unit->ToCreature()->GetAttackDistance(targetUnit) + unit->ToCreature()->m_CombatDistance)
2009 return false;
2010
2011 if (distance > visibilityRange)
2012 return false;
2013 }
2014
2015 return true;
2016}

2742 for (uint8 j = 0; j < 10; ++j)
