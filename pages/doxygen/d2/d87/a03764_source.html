70bool OutdoorPvPHP::SetupOutdoorPvP()
71{
72 m_AllianceTowersControlled = 0;
73 m_HordeTowersControlled = 0;
74 // add the zones affected by the pvp buff
75 for (int i = 0; i < OutdoorPvPHPBuffZonesNum; ++i)
76 RegisterZone(OutdoorPvPHPBuffZones[i]);
77
78 SetMapFromZone(OutdoorPvPHPBuffZones[0]);
79
80 AddCapturePoint(new OPvPCapturePointHP(this, HP_TOWER_BROKEN_HILL));
81
82 AddCapturePoint(new OPvPCapturePointHP(this, HP_TOWER_OVERLOOK));
83
84 AddCapturePoint(new OPvPCapturePointHP(this, HP_TOWER_STADIUM));
85
86 return true;

119bool OutdoorPvPHP::Update(uint32 diff)
120{
121 bool changed = OutdoorPvP::Update(diff);
122 if (changed)
123 {
124 if (m_AllianceTowersControlled == 3)
125 TeamApplyBuff(TEAM_ALLIANCE, AllianceBuff, HordeBuff);
126 else if (m_HordeTowersControlled == 3)
127 TeamApplyBuff(TEAM_HORDE, HordeBuff, AllianceBuff);
128 else
129 {
130 TeamCastSpell(TEAM_ALLIANCE, -AllianceBuff);
131 TeamCastSpell(TEAM_HORDE, -HordeBuff);
132 }
133 SendUpdateWorldState(HP_UI_TOWER_COUNT_A, m_AllianceTowersControlled);
134 SendUpdateWorldState(HP_UI_TOWER_COUNT_H, m_HordeTowersControlled);
135 }
136 return changed;
137}

171void OPvPCapturePointHP::ChangeState()
172{
173 uint32 field = 0;
174 switch (m_OldState)
175 {
176 case OBJECTIVESTATE_NEUTRAL:
177 field = HP_MAP_N[m_TowerType];
178 break;
179 case OBJECTIVESTATE_ALLIANCE:
180 field = HP_MAP_A[m_TowerType];
181 if (uint32 alliance_towers = ((OutdoorPvPHP*)m_PvP)->GetAllianceTowersControlled())
182 ((OutdoorPvPHP*)m_PvP)->SetAllianceTowersControlled(--alliance_towers);
183 sWorld->SendZoneText(OutdoorPvPHPBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(HP_LANG_LOSE_A[m_TowerType]));
184 break;
185 case OBJECTIVESTATE_HORDE:
186 field = HP_MAP_H[m_TowerType];
187 if (uint32 horde_towers = ((OutdoorPvPHP*)m_PvP)->GetHordeTowersControlled())
188 ((OutdoorPvPHP*)m_PvP)->SetHordeTowersControlled(--horde_towers);
189 sWorld->SendZoneText(OutdoorPvPHPBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(HP_LANG_LOSE_H[m_TowerType]));
190 break;
191 case OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE:
192 field = HP_MAP_N[m_TowerType];
193 break;
194 case OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE:
195 field = HP_MAP_N[m_TowerType];
196 break;
197 case OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE:
198 field = HP_MAP_A[m_TowerType];
199 break;
200 case OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE:
201 field = HP_MAP_H[m_TowerType];
202 break;
203 }
204
205 // send world state update
206 if (field)
207 {
208 m_PvP->SendUpdateWorldState(field, 0);
209 field = 0;
210 }
211 uint32 artkit = 21;
212 uint32 artkit2 = HP_TowerArtKit_N[m_TowerType];
213 switch (m_State)
214 {
215 case OBJECTIVESTATE_NEUTRAL:
216 field = HP_MAP_N[m_TowerType];
217 break;
218 case OBJECTIVESTATE_ALLIANCE:
219 {
220 field = HP_MAP_A[m_TowerType];
221 artkit = 2;
222 artkit2 = HP_TowerArtKit_A[m_TowerType];
223 uint32 alliance_towers = ((OutdoorPvPHP*)m_PvP)->GetAllianceTowersControlled();
224 if (alliance_towers < 3)
225 ((OutdoorPvPHP*)m_PvP)->SetAllianceTowersControlled(++alliance_towers);
226 sWorld->SendZoneText(OutdoorPvPHPBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(HP_LANG_CAPTURE_A[m_TowerType]));
227 break;
228 }
229 case OBJECTIVESTATE_HORDE:
230 {
231 field = HP_MAP_H[m_TowerType];
232 artkit = 1;
233 artkit2 = HP_TowerArtKit_H[m_TowerType];
234 uint32 horde_towers = ((OutdoorPvPHP*)m_PvP)->GetHordeTowersControlled();
235 if (horde_towers < 3)
236 ((OutdoorPvPHP*)m_PvP)->SetHordeTowersControlled(++horde_towers);
237 sWorld->SendZoneText(OutdoorPvPHPBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(HP_LANG_CAPTURE_H[m_TowerType]));
238 break;
239 }
240 case OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE:
241 field = HP_MAP_N[m_TowerType];
242 break;
243 case OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE:
244 field = HP_MAP_N[m_TowerType];
245 break;
246 case OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE:
247 field = HP_MAP_A[m_TowerType];
248 artkit = 2;
249 artkit2 = HP_TowerArtKit_A[m_TowerType];
250 break;
251 case OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE:
252 field = HP_MAP_H[m_TowerType];
253 artkit = 1;
254 artkit2 = HP_TowerArtKit_H[m_TowerType];
255 break;
256 }
257
258 Map* map = sMapMgr->FindMap(530, 0);
259 auto bounds = map->GetGameObjectBySpawnIdStore().equal_range(m_capturePointSpawnId);
260 for (auto itr = bounds.first; itr != bounds.second; ++itr)
261 itr->second->SetGoArtKit(artkit);
262
263 bounds = map->GetGameObjectBySpawnIdStore().equal_range(m_Objects[m_TowerType]);
264 for (auto itr = bounds.first; itr != bounds.second; ++itr)
265 itr->second->SetGoArtKit(artkit2);
266
267 // send world state update
268 if (field)
269 m_PvP->SendUpdateWorldState(field, 1);
270
271 // complete quest objective
272 if (m_State == OBJECTIVESTATE_ALLIANCE || m_State == OBJECTIVESTATE_HORDE)
273 SendObjectiveComplete(HP_CREDITMARKER[m_TowerType]);
274}

286void OPvPCapturePointHP::FillInitialWorldStates(WorldPacket& data)
287{
288 switch (m_State)
289 {
290 case OBJECTIVESTATE_ALLIANCE:
291 case OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE:
292 data << uint32(HP_MAP_N[m_TowerType]) << uint32(0);
293 data << uint32(HP_MAP_A[m_TowerType]) << uint32(1);
294 data << uint32(HP_MAP_H[m_TowerType]) << uint32(0);
295 break;
296 case OBJECTIVESTATE_HORDE:
297 case OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE:
298 data << uint32(HP_MAP_N[m_TowerType]) << uint32(0);
299 data << uint32(HP_MAP_A[m_TowerType]) << uint32(0);
300 data << uint32(HP_MAP_H[m_TowerType]) << uint32(1);
301 break;
302 case OBJECTIVESTATE_NEUTRAL:
303 case OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE:
304 case OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE:
305 default:
306 data << uint32(HP_MAP_N[m_TowerType]) << uint32(1);
307 data << uint32(HP_MAP_A[m_TowerType]) << uint32(0);
308 data << uint32(HP_MAP_H[m_TowerType]) << uint32(0);
309 break;

332void OutdoorPvPHP::HandleKillImpl(Player* player, Unit* killed)
333{
