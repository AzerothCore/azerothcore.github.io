69 void SetData(uint32 identifier, uint32 data) override
70 {
71 switch (identifier)
72 {
73 case DATA_SELIN_EVENT:
74 HandleGameObject(SelinDoorGUID, data == DONE);
75 HandleGameObject(SelinEncounterDoorGUID, data != IN_PROGRESS);
76 Encounter[identifier] = data;

78 case DATA_VEXALLUS_EVENT:

83 case DATA_DELRISSA_EVENT:
84 if (data == DONE)
85 HandleGameObject(DelrissaDoorGUID, true);
86 Encounter[identifier] = data;
87 break;
88 case DATA_KAELTHAS_EVENT:
89 HandleGameObject(KaelDoorGUID, data != IN_PROGRESS);
90 if (data == DONE)
91 if (GameObject* escapeOrb = instance->GetGameObject(EscapeOrbGUID))
92 escapeOrb->RemoveGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
93 Encounter[identifier] = data;
94 break;
95 }

164 void Load(const char* str) override
165 {
166 if (!str)
167 {
168 OUT_LOAD_INST_DATA_FAIL;
169 return;
170 }
171
172 OUT_LOAD_INST_DATA(str);
173
174 std::istringstream loadStream(str);
175
176 for (uint32 i = 0; i < MAX_ENCOUNTER; ++i)
177 {
178 uint32 tmpState;
179 loadStream >> tmpState;
180 if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
181 tmpState = NOT_STARTED;
182 SetData(i, tmpState);
183 }
184
185 OUT_LOAD_INST_DATA_COMPLETE;
186 }
