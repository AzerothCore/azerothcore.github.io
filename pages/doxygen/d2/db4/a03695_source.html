97 void Reset() override
98 {
99 globules.DespawnAll();
100 bIsExploded = false;
101 bIsFrenzy = false;
102 uiDrainedTimer = 15000;
103 uiWaterBoltVolleyTimer = urand(7000, 12000);
104 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);

108 void DoAction(int32 param) override
109 {
110 if (!me->IsAlive())
111 return;
112
113 switch(param)
114 {
115 case ACTION_WATER_ELEMENT_HIT:
116 if (pInstance)
117 pInstance->SetData(DATA_ACHIEV, 0);
118 me->ModifyHealth(int32(me->CountPctFromMaxHealth(1)));
119 if (bIsExploded)
120 DoExplodeCompleted();
121 break;
122 case ACTION_WATER_ELEMENT_KILLED:
123 uint32 damage = me->CountPctFromMaxHealth(3);
124 damage = std::min(damage, me->GetHealth() - 1);
125 me->ModifyHealth(-int32(damage));
126 me->LowerPlayerDamageReq(damage);
127 break;
128 }
129 }
130
131 void DoExplodeCompleted()
132 {
133 bIsExploded = false;
134 me->RemoveAura(SPELL_DRAINED);
135 if (!HealthBelowPct(25))
136 {
137 Talk(SAY_BUBBLE);
138 me->CastSpell(me, SPELL_PROTECTIVE_BUBBLE, true);
139 }
140
141 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
142 me->SetDisplayId(me->GetNativeDisplayId());
143 }
144
145 void IchoronDoCastToAllHostilePlayers(uint32 spellId, bool triggered)
146 {
147 Map::PlayerList const& PlayerList = me->GetMap()->GetPlayers();
148 if (PlayerList.IsEmpty())
149 return;
150
151 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
152 if (Player* plr = i->GetSource())
153 me->CastSpell(plr, spellId, triggered);
154 }
155
156 void EnterCombat(Unit* /*who*/) override
157 {
158 bIsExploded = false;
159 bIsFrenzy = false;
160 uiDrainedTimer = 15000;
161 uiWaterBoltVolleyTimer = urand(7000, 12000);
162 DoZoneInCombat();
163 Talk(SAY_AGGRO);
164 me->CastSpell(me, SPELL_PROTECTIVE_BUBBLE, true);
165 if (pInstance)
166 pInstance->SetData(DATA_ACHIEV, 1);
167 }

169 void UpdateAI(uint32 uiDiff) override
170 {
171 if (!UpdateVictim())
172 return;
173
174 if (!bIsFrenzy && !bIsExploded && HealthBelowPct(25))
175 {
176 Talk(SAY_ENRAGE);
177 me->CastSpell(me, SPELL_FRENZY, true);
178 bIsFrenzy = true;
179 }
180
181 if (!bIsFrenzy)
182 {
183 if (!bIsExploded)
184 {
185 if (!me->HasAura(SPELL_PROTECTIVE_BUBBLE))
186 {
187 me->InterruptNonMeleeSpells(false);
188 Talk(SAY_SHATTER);
189 DoZoneInCombat();
190 IchoronDoCastToAllHostilePlayers(SPELL_WATER_BLAST, true);
191 me->CastSpell(me, SPELL_DRAINED, true);
192 bIsExploded = true;
193 uiDrainedTimer = 15000;
194 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
195 me->SetDisplayId(11686);
196 for (uint8 i = 0; i < MAX_SPAWN_LOC; ++i)
197 {
198 float angle = rand_norm() * 2 * M_PI;
199 Position p1(SpawnLoc[i]), p2(SpawnLoc[i]);
200 p1.m_positionX += 2.5f * cos(angle);
201 p1.m_positionY += 2.5f * std::sin(angle);
202 p2.m_positionX -= 2.5f * cos(angle);
203 p2.m_positionY -= 2.5f * std::sin(angle);
204 DoSummon(NPC_ICHOR_GLOBULE, p1, 60000, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN);
205 DoSummon(NPC_ICHOR_GLOBULE, p2, 60000, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN);
206 }
207 }
208 }
209 else
210 {
211 if (uiDrainedTimer <= uiDiff)
212 DoExplodeCompleted();
213 else
214 {
215 uiDrainedTimer -= uiDiff;
216
217 bool bIsWaterElementsAlive = false;
218 if (!globules.empty())
219 {
220 for (ObjectGuid const& guid : globules)
221 if (Creature* pTemp = ObjectAccessor::GetCreature(*me, guid))
222 if (pTemp->IsAlive())
223 {
224 bIsWaterElementsAlive = true;
225 break;
226 }
227 }
228
229 if (!bIsWaterElementsAlive)
230 DoExplodeCompleted();
231 }
232 }
233 }
234
235 if (!bIsExploded)
236 {
237 if (uiWaterBoltVolleyTimer <= uiDiff)
238 {
239 me->CastSpell((Unit*)nullptr, SPELL_WATER_BOLT_VOLLEY, false);
240 uiWaterBoltVolleyTimer = urand(10000, 15000);
241 }
242 else uiWaterBoltVolleyTimer -= uiDiff;
243 }
244
245 DoMeleeAttackIfReady();
246 }
