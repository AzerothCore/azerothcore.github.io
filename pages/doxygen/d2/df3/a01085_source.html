34struct M2Header
35{
36 char Magic[4]; // "MD20"
37 uint32 Version; // The version of the format.
38 uint32 lName; // Length of the model's name including the trailing \0
39 uint32 ofsName; // Offset to the name, it seems like models can get reloaded by this name.should be unique, i guess.
40 uint32 GlobalModelFlags; // 0x0001: tilt x, 0x0002: tilt y, 0x0008: add 2 fields in header, 0x0020: load .phys data (MoP+), 0x0080: has _lod .skin files (MoP?+), 0x0100: is camera related.
41 uint32 nGlobalSequences;
42 uint32 ofsGlobalSequences; // A list of timestamps.
43 uint32 nAnimations;
44 uint32 ofsAnimations; // Information about the animations in the model.
45 uint32 nAnimationLookup;
46 uint32 ofsAnimationLookup; // Mapping of global IDs to the entries in the Animation sequences block.
47 uint32 nBones; // MAX_BONES = 0x100
48 uint32 ofsBones; // Information about the bones in this model.
49 uint32 nKeyBoneLookup;
50 uint32 ofsKeyBoneLookup; // Lookup table for key skeletal bones.
51 uint32 nVertices;
52 uint32 ofsVertices; // Vertices of the model.
53 uint32 nViews; // Views (LOD) are now in .skins.
54 uint32 nSubmeshAnimations;
55 uint32 ofsSubmeshAnimations; // Submesh color and alpha animations definitions.
56 uint32 nTextures;
57 uint32 ofsTextures; // Textures of this model.
58 uint32 nTransparency;
59 uint32 ofsTransparency; // Transparency of textures.
60 uint32 nUVAnimation;
61 uint32 ofsUVAnimation;
62 uint32 nTexReplace;
63 uint32 ofsTexReplace; // Replaceable Textures.
64 uint32 nRenderFlags;
65 uint32 ofsRenderFlags; // Blending modes / render flags.
66 uint32 nBoneLookupTable;
67 uint32 ofsBoneLookupTable; // A bone lookup table.
68 uint32 nTexLookup;
69 uint32 ofsTexLookup; // The same for textures.
70 uint32 nTexUnits; // possibly removed with cata?!
71 uint32 ofsTexUnits; // And texture units. Somewhere they have to be too.
72 uint32 nTransLookup;
73 uint32 ofsTransLookup; // Everything needs its lookup. Here are the transparencies.
74 uint32 nUVAnimLookup;
75 uint32 ofsUVAnimLookup;
76 G3D::AABox BoundingBox; // min/max( [1].z, 2.0277779f ) - 0.16f seems to be the maximum camera height
77 float BoundingSphereRadius;
78 G3D::AABox CollisionBox;
79 float CollisionSphereRadius;
80 uint32 nBoundingTriangles;
81 uint32 ofsBoundingTriangles; // Our bounding volumes. Similar structure like in the old ofsViews.
82 uint32 nBoundingVertices;
83 uint32 ofsBoundingVertices;
84 uint32 nBoundingNormals;
85 uint32 ofsBoundingNormals;
86 uint32 nAttachments;
87 uint32 ofsAttachments; // Attachments are for weapons etc.
88 uint32 nAttachLookup;
89 uint32 ofsAttachLookup; // Of course with a lookup.
90 uint32 nEvents;
91 uint32 ofsEvents; // Used for playing sounds when dying and a lot else.
92 uint32 nLights;
93 uint32 ofsLights; // Lights are mainly used in loginscreens but in wands and some doodads too.
94 uint32 nCameras; // Format of Cameras changed with version 271!
95 uint32 ofsCameras; // The cameras are present in most models for having a model in the Character-Tab.
96 uint32 nCameraLookup;
97 uint32 ofsCameraLookup; // And lookup-time again.
98 uint32 nRibbonEmitters;
99 uint32 ofsRibbonEmitters; // Things swirling around. See the CoT-entrance for light-trails.
100 uint32 nParticleEmitters;
101 uint32 ofsParticleEmitters; // Spells and weapons, doodads and loginscreens use them. Blood dripping of a blade? Particles.
102 uint32 nBlendMaps; // This has to deal with blending. Exists IFF (flags & 0x8) != 0. When set, textures blending is overriden by the associated array. See M2/WotLK#Blend_mode_overrides
103 uint32 ofsBlendMaps; // Same as above. Points to an array of uint16 of nBlendMaps entries -- From WoD information.};
104};
