98 struct boss_general_zarithrianAI : public BossAI

100 boss_general_zarithrianAI(Creature* creature) : BossAI(creature, DATA_GENERAL_ZARITHRIAN)
101 {
102 }
103
104 void Reset() override
105 {
106 BossAI::Reset();
107 if (instance->GetBossState(DATA_SAVIANA_RAGEFIRE) == DONE && instance->GetBossState(DATA_BALTHARUS_THE_WARBORN) == DONE)
108 {
109 me->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
110 me->SetImmuneToPC(false);
111 }
112 }
113
114 void EnterCombat(Unit* who) override
115 {
116 BossAI::EnterCombat(who);
117 Talk(SAY_AGGRO);
118
119 events.ScheduleEvent(EVENT_CLEAVE, 9000);
120 events.ScheduleEvent(EVENT_INTIDMDATING_ROAR, 14000);
121 events.ScheduleEvent(EVENT_SUMMON_ADDS1, 18000);
122 if (Is25ManRaid())
123 events.ScheduleEvent(EVENT_SUMMON_ADDS2, 20000);
124 }
125
126 void JustSummoned(Creature* summon) override
127 {
128 summons.Summon(summon);
129 }
130
131 void JustDied(Unit* killer) override
132 {
133 BossAI::JustDied(killer);
134 Talk(SAY_DEATH);
135 }
136
137 void KilledUnit(Unit* /*victim*/) override
138 {
139 if (events.GetNextEventTime(EVENT_KILL_TALK) == 0)
140 {
141 Talk(SAY_KILL);
142 events.ScheduleEvent(EVENT_KILL_TALK, 6000);
143 }
144 }
145
146 void UpdateAI(uint32 diff) override
147 {
148 if (!UpdateVictim())
149 return;
150
151 events.Update(diff);
152 if (me->HasUnitState(UNIT_STATE_CASTING))
153 return;
154
155 switch (events.ExecuteEvent())
156 {
157 case EVENT_SUMMON_ADDS1:
158 Talk(SAY_ADDS);
159 if (Creature* stalker1 = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ZARITHRIAN_SPAWN_STALKER_1)))
160 stalker1->CastSpell(stalker1, SPELL_SUMMON_FLAMECALLER, false);
161 if (Creature* stalker2 = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ZARITHRIAN_SPAWN_STALKER_2)))
162 stalker2->CastSpell(stalker2, SPELL_SUMMON_FLAMECALLER, false);
163 events.ScheduleEvent(EVENT_SUMMON_ADDS1, 40000);
164 break;
165 case EVENT_SUMMON_ADDS2:
166 if (Creature* stalker1 = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ZARITHRIAN_SPAWN_STALKER_1)))
167 stalker1->CastSpell(stalker1, SPELL_SUMMON_FLAMECALLER, false);
168 if (Creature* stalker2 = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_ZARITHRIAN_SPAWN_STALKER_2)))
169 stalker2->CastSpell(stalker2, SPELL_SUMMON_FLAMECALLER, false);
170 events.ScheduleEvent(EVENT_SUMMON_ADDS2, 40000);
171 break;
172 case EVENT_INTIDMDATING_ROAR:
173 me->CastSpell(me, SPELL_INTIMIDATING_ROAR, false);
174 events.ScheduleEvent(EVENT_INTIDMDATING_ROAR, 30000);
175 break;
176 case EVENT_CLEAVE:
177 me->CastSpell(me->GetVictim(), SPELL_CLEAVE_ARMOR, false);
178 events.ScheduleEvent(EVENT_CLEAVE, 15000);
179 break;
180 }
181
182 DoMeleeAttackIfReady();
183 }
184
185 bool CheckEvadeIfOutOfCombatArea() const override
186 {
187 return me->GetPositionX() > 3060.0f;
188 }
189 };
190
191 CreatureAI* GetAI(Creature* creature) const override
192 {
193 return GetRubySanctumAI<boss_general_zarithrianAI>(creature);
194 }

202 struct npc_onyx_flamecallerAI : public npc_escortAI
203 {
204 npc_onyx_flamecallerAI(Creature* creature) : npc_escortAI(creature), _instance(creature->GetInstanceScript())
205 {
206 npc_escortAI::SetDespawnAtEnd(false);
207 }

209 void Reset() override
210 {
211 _lavaGoutCount = 0;
212 AddWaypoints();

214 }
215
216 void EnterCombat(Unit* /*who*/) override
217 {
218 _events.Reset();
219 _events.ScheduleEvent(EVENT_BLAST_NOVA, urand(20000, 30000));
220 _events.ScheduleEvent(EVENT_LAVA_GOUT, 5000);
221 }
222
223 void EnterEvadeMode(EvadeReason /*why*/) override

225 // Prevent EvadeMode
226 }

228 void IsSummonedBy(Unit* /*summoner*/) override
229 {
230 // Let Zarithrian count as summoner. _instance cant be null since we got GetRubySanctumAI
231 if (Creature* zarithrian = ObjectAccessor::GetCreature(*me, _instance->GetGuidData(NPC_GENERAL_ZARITHRIAN)))
232 zarithrian->AI()->JustSummoned(me);
233 }
234
235 void WaypointReached(uint32 waypointId) override
236 {
237 if (waypointId == MAX_PATH_FLAMECALLER_WAYPOINTS)
238 me->SetInCombatWithZone();
239 }
240
241 void AddWaypoints()
242 {
243 if (me->GetPositionY() < 500.0f)
244 {
245 for (uint8 i = 0; i < MAX_PATH_FLAMECALLER_WAYPOINTS; i++)
246 AddWaypoint(i + 1, FlamecallerWaypointsE[i].GetPositionX(), FlamecallerWaypointsE[i].GetPositionY(), FlamecallerWaypointsE[i].GetPositionZ());
247 }
248 else
249 {
250 for (uint8 i = 0; i < MAX_PATH_FLAMECALLER_WAYPOINTS; i++)
251 AddWaypoint(i + 1, FlamecallerWaypointsW[i].GetPositionX(), FlamecallerWaypointsW[i].GetPositionY(), FlamecallerWaypointsW[i].GetPositionZ());
252 }
253 }
254
255 void UpdateEscortAI(uint32 diff) override
256 {
257 if (!UpdateVictim())
258 return;
259
260 _events.Update(diff);
261 if (me->HasUnitState(UNIT_STATE_CASTING))
262 return;
263
264 switch (_events.ExecuteEvent())
265 {
266 case EVENT_BLAST_NOVA:
267 DoCastAOE(SPELL_BLAST_NOVA);
268 _events.ScheduleEvent(EVENT_BLAST_NOVA, urand(20000, 30000));
269 break;
270 case EVENT_LAVA_GOUT:
271 if (_lavaGoutCount >= 3)
272 {
273 _lavaGoutCount = 0;
274 _events.ScheduleEvent(EVENT_LAVA_GOUT, 8000);
275 break;
276 }
277 DoCastVictim(SPELL_LAVA_GOUT);
278 _lavaGoutCount++;
279 _events.ScheduleEvent(EVENT_LAVA_GOUT, 1500);
280 break;
281 }
282
283 DoMeleeAttackIfReady();
284 }
285 private:
286 EventMap _events;
287 InstanceScript* _instance;
288 uint8 _lavaGoutCount;
289 };
290
291 CreatureAI* GetAI(Creature* creature) const override
292 {
293 return GetRubySanctumAI<npc_onyx_flamecallerAI>(creature);
294 }
