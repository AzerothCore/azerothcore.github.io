71 void Reset() override
72 {
73 events.Reset();
74 if (pInstance)
75 {
76 pInstance->DoStopTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEVEMENT_GOOD_GRIEF);
77 pInstance->SetData(BOSS_MAIDEN_OF_GRIEF, NOT_STARTED);
78 }
79 }
80
81 void EnterCombat(Unit* /*who*/) override
82 {
83 events.ScheduleEvent(EVENT_STORM, 5000);
84 events.ScheduleEvent(EVENT_SHOCK, 26000 + rand() % 6000);
85 events.ScheduleEvent(EVENT_PILLAR, 12000 + rand() % 8000);
86 events.ScheduleEvent(EVENT_PARTING, 8000);
87
88 Talk(SAY_AGGRO);
89 if (pInstance)
90 {
91 pInstance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEVEMENT_GOOD_GRIEF);
92 pInstance->SetData(BOSS_MAIDEN_OF_GRIEF, IN_PROGRESS);
93 }
94 }
95
96 void UpdateAI(uint32 diff) override
97 {
98 if(!UpdateVictim())
99 return;
100
101 events.Update(diff);
102
103 if( me->HasUnitState(UNIT_STATE_CASTING) )
104 return;
105
106 switch( events.ExecuteEvent() )
107 {
108 case EVENT_STORM:
109 {
110 me->CastSpell(me->GetVictim(), DUNGEON_MODE(STORM_OF_GRIEF, STORM_OF_GRIEF_H), true);
111 events.RepeatEvent(10000);
112 break;
113 }
114 case EVENT_SHOCK:
115 {
116 me->CastSpell(me->GetVictim(), DUNGEON_MODE(SHOCK_OF_SORROW, SHOCK_OF_SORROW_H), false);
117 Talk(SAY_STUN);
118
119 events.RepeatEvent(16000 + rand() % 6000);
120 break;
121 }
122 case EVENT_PILLAR:
123 {
124 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true, 0))
125 me->CastSpell(target, DUNGEON_MODE(PILLAR_OF_WOE, PILLAR_OF_WOE_H), false);
126
127 events.RepeatEvent(12000 + rand() % 8000);
128 break;
129 }
130 case EVENT_PARTING:
131 {
132 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true, 0))
133 me->CastSpell(target, PARTING_SORROW, false);
134
135 events.RepeatEvent(6000 + rand() % 10000);
136 break;
137 }
138 }
139
140 DoMeleeAttackIfReady();
141 }
142
143 void JustDied(Unit* /*killer*/) override
144 {
145 Talk(SAY_DEATH);
146
147 if (pInstance)
148 pInstance->SetData(BOSS_MAIDEN_OF_GRIEF, DONE);
149 }
150
151 void KilledUnit(Unit* /*victim*/) override
152 {
153 if (urand(0, 1))
154 return;
155
156 Talk(SAY_SLAY);
157 }
158 };
