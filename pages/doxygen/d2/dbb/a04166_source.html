193class spell_pal_sacred_shield_base : public AuraScript
194{
195 PrepareAuraScript(spell_pal_sacred_shield_base);
196
197 void CalculateAmount(AuraEffect const* aurEff, int32& amount, bool& /*canBeRecalculated*/)
198 {
199 if (Unit* caster = GetCaster())
200 {
201 SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(GetSpellInfo()->Effects[aurEff->GetEffIndex()].TriggerSpell);
202 amount = spellInfo->Effects[EFFECT_0].CalcValue();
203
204 // +75.00% from sp bonus
205 amount += CalculatePct(caster->SpellBaseDamageBonusDone(spellInfo->GetSchoolMask()), 75.0f);
206
207 // Xinef: removed divine guardian because it will affect triggered spell with increased amount
208 // Arena - Dampening
209 if (AuraEffect const* dampening = caster->GetAuraEffect(SPELL_GENERIC_ARENA_DAMPENING, EFFECT_0))
210 {
211 AddPct(amount, dampening->GetAmount());
212 }
213 // Battleground - Dampening
214 else if (AuraEffect const* dampening2 = caster->GetAuraEffect(SPELL_GENERIC_BATTLEGROUND_DAMPENING, EFFECT_0))
215 {
216 AddPct(amount, dampening2->GetAmount());
217 }
218 }
219 }
220
221 bool CheckProc(ProcEventInfo& eventInfo)
222 {
223 HealInfo* healinfo = eventInfo.GetHealInfo();
224 DamageInfo* damageinfo = eventInfo.GetDamageInfo();
225 return !(eventInfo.GetHitMask() & PROC_EX_INTERNAL_HOT) && ((healinfo && healinfo->GetHeal() > 0) || (damageinfo && damageinfo->GetDamage() > 0));
226 }
227
228 void HandleProc(AuraEffect const* aurEff, ProcEventInfo& eventInfo)
229 {
230 PreventDefaultAction();
231
232 if (eventInfo.GetTypeMask() & PROC_FLAG_TAKEN_SPELL_MAGIC_DMG_CLASS_POS)
233 {
234 Unit* caster = eventInfo.GetActor();
235
236 HealInfo* healinfo = eventInfo.GetHealInfo();
237
238 if (!healinfo || !healinfo->GetHeal())
239 {
240 return;
241 }
242
243 SpellInfo const* procSpell = healinfo->GetSpellInfo();
244 if (!procSpell)
245 {
246 return;
247 }
248
249 if (caster && procSpell->SpellFamilyName == SPELLFAMILY_PALADIN &&
250 procSpell->SpellFamilyFlags.HasFlag(0x40000000) && caster->GetAuraEffect(SPELL_AURA_PROC_TRIGGER_SPELL, SPELLFAMILY_PALADIN, 3021, 0)) // need infusion of light
251 {
252 int32 basepoints = int32(float(healinfo->GetHeal()) / 12.0f);
253 // Item - Paladin T9 Holy 4P Bonus (Flash of Light)
254 if (AuraEffect const* aurEffect = caster->GetAuraEffect(67191, EFFECT_0))
255 AddPct(basepoints, aurEffect->GetAmount());
256
257 caster->CastCustomSpell(eventInfo.GetActionTarget(), 66922, &basepoints, nullptr, nullptr, true, nullptr, aurEff, caster->GetGUID());
258 return;
259 }
260
261 return;
262 }
263
264 uint32 triggered_spell_id = GetSpellInfo()->Effects[aurEff->GetEffIndex()].TriggerSpell;
265 if (eventInfo.GetActionTarget()->HasSpellCooldown(triggered_spell_id))
266 return;
267
268 uint32 cooldown = eventInfo.GetProcCooldown();
269 int32 basepoints = aurEff->GetAmount();
270
271 // Item - Paladin T8 Holy 4P Bonus
272 if (Unit* caster = aurEff->GetCaster())
273 if (AuraEffect const* aurEffect = caster->GetAuraEffect(64895, 0))
274 cooldown = aurEffect->GetAmount() * IN_MILLISECONDS;
275
276 eventInfo.GetActionTarget()->AddSpellCooldown(triggered_spell_id, 0, cooldown);
277 eventInfo.GetActionTarget()->CastCustomSpell(eventInfo.GetActionTarget(), triggered_spell_id, &basepoints, nullptr, nullptr, true, nullptr, aurEff, eventInfo.GetActionTarget()->GetGUID());
278 }
279
280 void Register() override
281 {
282 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_pal_sacred_shield_base::CalculateAmount, EFFECT_0, SPELL_AURA_DUMMY);
283 DoCheckProc += AuraCheckProcFn(spell_pal_sacred_shield_base::CheckProc);
284 OnEffectProc += AuraEffectProcFn(spell_pal_sacred_shield_base::HandleProc, EFFECT_0, SPELL_AURA_DUMMY);
285 }

288// 31850 - Ardent Defender
289class spell_pal_ardent_defender : public AuraScript
290{
291 PrepareAuraScript(spell_pal_ardent_defender);
292
293public:
294 spell_pal_ardent_defender()
295 {
296 absorbPct = 0;
297 healPct = 0;
298 }
299
300private:
301 uint32 absorbPct, healPct;
302
303 enum Spell
304 {
305 PAL_SPELL_ARDENT_DEFENDER_HEAL = 66235
306 };
307
308 bool Validate(SpellInfo const* /*spellInfo*/) override
309 {
310 return ValidateSpellInfo({ PAL_SPELL_ARDENT_DEFENDER_HEAL });
311 }
312
313 bool Load() override
314 {
315 healPct = GetSpellInfo()->Effects[EFFECT_1].CalcValue();
316 absorbPct = GetSpellInfo()->Effects[EFFECT_0].CalcValue();
317 return GetUnitOwner()->GetTypeId() == TYPEID_PLAYER;
318 }
319
320 void CalculateAmount(AuraEffect const* /*aurEff*/, int32& amount, bool& /*canBeRecalculated*/)
321 {
322 // Set absorbtion amount to unlimited
323 amount = -1;
324 }
325
326 void Absorb(AuraEffect* aurEff, DamageInfo& dmgInfo, uint32& absorbAmount)
327 {
328 Unit* victim = GetTarget();
329 int32 remainingHealth = victim->GetHealth() - dmgInfo.GetDamage();
330 uint32 allowedHealth = victim->CountPctFromMaxHealth(35);
331 // If damage kills us
332 if (remainingHealth <= 0 && !victim->ToPlayer()->HasSpellCooldown(PAL_SPELL_ARDENT_DEFENDER_HEAL))
333 {
334 // Cast healing spell, completely avoid damage
335 absorbAmount = dmgInfo.GetDamage();
336
337 uint32 defenseSkillValue = victim->GetDefenseSkillValue();
338 // Max heal when defense skill denies critical hits from raid bosses
339 // Formula: max defense at level + 140 (raiting from gear)
340 uint32 reqDefForMaxHeal = victim->getLevel() * 5 + 140;
341 float pctFromDefense = (defenseSkillValue >= reqDefForMaxHeal)
342 ? 1.0f
343 : float(defenseSkillValue) / float(reqDefForMaxHeal);
344
345 int32 healAmount = int32(victim->CountPctFromMaxHealth(uint32(healPct * pctFromDefense)));
346 victim->CastCustomSpell(PAL_SPELL_ARDENT_DEFENDER_HEAL, SPELLVALUE_BASE_POINT0, healAmount, victim, true, nullptr, aurEff);
347 victim->ToPlayer()->AddSpellCooldown(PAL_SPELL_ARDENT_DEFENDER_HEAL, 0, 120000);
348 }
349 else if (remainingHealth < int32(allowedHealth))
350 {
351 // Reduce damage that brings us under 35% (or full damage if we are already under 35%) by x%
352 uint32 damageToReduce = (victim->GetHealth() < allowedHealth)
353 ? dmgInfo.GetDamage()
354 : allowedHealth - remainingHealth;
355 absorbAmount = CalculatePct(damageToReduce, absorbPct);
356 }
357 }
358
359 void Register() override
360 {
361 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_pal_ardent_defender::CalculateAmount, EFFECT_0, SPELL_AURA_SCHOOL_ABSORB);
362 OnEffectAbsorb += AuraEffectAbsorbFn(spell_pal_ardent_defender::Absorb, EFFECT_0);
363 }
