97 bool IsBetween(WorldObject* u1, WorldObject* target, WorldObject* u2) // the in-line checker
98 {
99 if (!u1 || !u2 || !target)
100 return false;
101
102 float xn, yn, xp, yp, xh, yh;
103 xn = u1->GetPositionX();
104 yn = u1->GetPositionY();
105 xp = u2->GetPositionX();
106 yp = u2->GetPositionY();
107 xh = target->GetPositionX();
108 yh = target->GetPositionY();
109
110 // check if target is between (not checking distance from the beam yet)
111 if (dist(xn, yn, xh, yh) >= dist(xn, yn, xp, yp) || dist(xp, yp, xh, yh) >= dist(xn, yn, xp, yp))
112 return false;
113 // check distance from the beam
114 return (std::abs((xn - xp) * yh + (yp - yn) * xh - xn * yp + xp * yn) / dist(xn, yn, xp, yp) < 1.5f);
115 }
116
117 float dist(float xa, float ya, float xb, float yb) // auxiliary method for distance
118 {
119 return std::sqrt((xa - xb) * (xa - xb) + (ya - yb) * (ya - yb));
120 }

133 void SummonPortals()
134 {
135 uint8 r = rand() % 4;
136 uint8 pos[3];
137 pos[RED_PORTAL] = ((r % 2) ? (r > 1 ? 2 : 1) : 0);
138 pos[GREEN_PORTAL] = ((r % 2) ? 0 : (r > 1 ? 2 : 1));
139 pos[BLUE_PORTAL] = (r > 1 ? 1 : 2); // Blue Portal not on the left side (0)
140
141 for (int i = 0; i < 3; ++i)
142 if (Creature* portal = me->SummonCreature(PortalID[i], PortalCoord[pos[i]][0], PortalCoord[pos[i]][1], PortalCoord[pos[i]][2], 0, TEMPSUMMON_TIMED_DESPAWN, 60000))
143 {
144 PortalGUID[i] = portal->GetGUID();

146 }
147 }

149 void DestroyPortals()
150 {
151 for (int i = 0; i < 3; ++i)
152 {
153 if (Creature* portal = ObjectAccessor::GetCreature(*me, PortalGUID[i]))
154 portal->DisappearAndDie();
155 if (Creature* portal = ObjectAccessor::GetCreature(*me, BeamerGUID[i]))
156 portal->DisappearAndDie();
157 PortalGUID[i].Clear();

159 }

162 void UpdatePortals() // Here we handle the beams' behavior
163 {
164 for (int j = 0; j < 3; ++j) // j = color
165 if (Creature* portal = ObjectAccessor::GetCreature(*me, PortalGUID[j]))
166 {
167 // the one who's been cast upon before
168 Unit* current = ObjectAccessor::GetUnit(*portal, BeamTarget[j]);
169 // temporary store for the best suitable beam reciever
170 Unit* target = me;
171
172 if (Map* map = me->GetMap())
173 {
174 Map::PlayerList const& players = map->GetPlayers();
175
176 // get the best suitable target
177 for (Map::PlayerList::const_iterator i = players.begin(); i != players.end(); ++i)
178 {
179 Player* p = i->GetSource();
180 if (p && p->IsAlive() // alive
181 && (!target || target->GetDistance2d(portal) > p->GetDistance2d(portal)) // closer than current best
182 && !p->HasAura(PlayerDebuff[j]) // not exhausted
183 && !p->HasAura(PlayerBuff[(j + 1) % 3]) // not on another beam
184 && !p->HasAura(PlayerBuff[(j + 2) % 3])
185 && IsBetween(me, p, portal)) // on the beam
186 target = p;
187 }
188 }
189 // buff the target
190 if (target->GetTypeId() == TYPEID_PLAYER)
191 target->AddAura(PlayerBuff[j], target);
192 else
193 target->AddAura(NetherBuff[j], target);
194 // cast visual beam on the chosen target if switched
195 // simple target switching isn't working -> using BeamerGUID to cast (workaround)
196 if (!current || target != current)
197 {
198 BeamTarget[j] = target->GetGUID();
199 // remove currently beaming portal
200 if (Creature* beamer = ObjectAccessor::GetCreature(*portal, BeamerGUID[j]))
201 {
202 beamer->CastSpell(target, PortalBeam[j], false);
203 beamer->DisappearAndDie();
204 BeamerGUID[j].Clear();
205 }
206 // create new one and start beaming on the target
207 if (Creature* beamer = portal->SummonCreature(PortalID[j], portal->GetPositionX(), portal->GetPositionY(), portal->GetPositionZ(), portal->GetOrientation(), TEMPSUMMON_TIMED_DESPAWN, 60000))
208 {
209 beamer->CastSpell(target, PortalBeam[j], false);
210 BeamerGUID[j] = beamer->GetGUID();
211 }
212 }
213 // aggro target if Red Beam
214 if (j == RED_PORTAL && me->GetVictim() != target && target->GetTypeId() == TYPEID_PLAYER)
215 me->GetThreatMgr().AddThreat(target, 100000.0f + DoGetThreat(me->GetVictim()));
216 }
217 }

265 void UpdateAI(uint32 diff) override
266 {
267 if (!UpdateVictim())
268 return;
269
270 // Void Zone
271 if (VoidZoneTimer <= diff)
272 {
273 DoCast(SelectTarget(SelectTargetMethod::Random, 1, 45, true), SPELL_VOIDZONE, true);
274 VoidZoneTimer = 15000;
275 }
276 else
277 VoidZoneTimer -= diff;
278
279 // NetherInfusion Berserk
280 if (!Berserk && NetherInfusionTimer <= diff)
281 {
282 me->AddAura(SPELL_NETHER_INFUSION, me);
283 DoCast(me, SPELL_NETHERSPITE_ROAR);
284 Berserk = true;
285 }
286 else
287 NetherInfusionTimer -= diff;
288
289 if (PortalPhase) // PORTAL PHASE
290 {
291 // Distribute beams and buffs
292 if (PortalTimer <= diff)
293 {
294 UpdatePortals();
295 PortalTimer = 1000;
296 }
297 else
298 PortalTimer -= diff;
299
300 // Empowerment & Nether Burn
301 if (EmpowermentTimer <= diff)
302 {
303 DoCast(me, SPELL_EMPOWERMENT);
304 me->AddAura(SPELL_NETHERBURN_AURA, me);
305 EmpowermentTimer = 90000;
306 }
307 else
308 EmpowermentTimer -= diff;
309
310 if (PhaseTimer <= diff)
311 {
312 if (!me->IsNonMeleeSpellCast(false))
313 {
314 SwitchToBanishPhase();
315 return;
316 }
317 }
318 else
319 PhaseTimer -= diff;
320 }
321 else // BANISH PHASE
322 {
323 // Netherbreath
324 if (NetherbreathTimer <= diff)
325 {
326 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 40, true))
327 DoCast(target, SPELL_NETHERBREATH);
328 NetherbreathTimer = urand(5000, 7000);
329 }
330 else
331 NetherbreathTimer -= diff;
332
333 if (PhaseTimer <= diff)
334 {
335 if (!me->IsNonMeleeSpellCast(false))
336 {
337 SwitchToPortalPhase();
338 return;
339 }
340 }
341 else
342 PhaseTimer -= diff;
343 }
344
345 DoMeleeAttackIfReady();
346 }
