364 void DoAction(int32 action) override
365 {
366 switch (action)
367 {
368 case ACTION_STAND_UP:
369 summons.DespawnEntry(WORLD_TRIGGER);
370 me->RemoveUnitFlag(UNIT_FLAG_PREVENT_EMOTES_FROM_CHAT_TEXT | UNIT_FLAG_NOT_SELECTABLE);
371 me->SetImmuneToAll(false);
372 me->RemoveDynamicFlag(UNIT_DYNFLAG_DEAD);
373 me->RemoveUnitFlag2(UNIT_FLAG2_FEIGN_DEATH);
374 me->SetReactState(REACT_AGGRESSIVE);
375 me->ForceValuesUpdateAtIndex(UNIT_NPC_FLAGS); // was in sniff. don't ask why
376 me->m_Events.AddEvent(new StandUpEvent(*me), me->m_Events.CalculateTime(1000));
377 DoAction(ACTION_REMOVE_INVOCATION);
378 me->SetHealth(1);
379 break;
380 case ACTION_CAST_INVOCATION:
381 Talk(SAY_KELESETH_INVOCATION);
382 me->CastSpell(me, SPELL_INVOCATION_OF_BLOOD_KELESETH, true);
383 me->CastSpell(me, SPELL_INVOCATION_VISUAL_ACTIVE, true);
384 _isEmpowered = true;
385 break;
386 case ACTION_REMOVE_INVOCATION:
387 _isEmpowered = false;
388 me->RemoveAurasDueToSpell(SPELL_INVOCATION_VISUAL_ACTIVE);
389 me->RemoveAurasDueToSpell(SPELL_INVOCATION_OF_BLOOD_KELESETH);
390 break;
391 default:
392 break;
393 }
394 }

396 void UpdateAI(uint32 diff) override
397 {
398 if (!UpdateVictim())
399 return;
400
401 events.Update(diff);
402
403 if (me->HasUnitState(UNIT_STATE_CASTING))
404 return;
405
406 switch (events.ExecuteEvent())
407 {
408 case EVENT_BERSERK:
409 me->CastSpell(me, SPELL_BERSERK, true);
410 Talk(EMOTE_KELESETH_BERSERK);
411 break;
412 case EVENT_SHADOW_RESONANCE:
413 Talk(SAY_KELESETH_SPECIAL);
414 me->CastSpell(me, SPELL_SHADOW_RESONANCE, false);
415 events.ScheduleEvent(EVENT_SHADOW_RESONANCE, urand(10000, 15000));
416 break;
417 }
418
419 DoSpellAttackIfReady(_isEmpowered ? SPELL_EMPOWERED_SHADOW_LANCE : SPELL_SHADOW_LANCE);
420 }

433 valanar->AI()->EnterEvadeMode();
434 ScriptedAI::EnterEvadeMode(why);
435 _evading = false;
436 }
