63inline bool IsValidCombatTarget(Creature* source, Player* target)
64{
65 if (target->IsGameMaster())
66 {
67 return false;
68 }
69
70 if (!source->IsInWorld() || !target->IsInWorld())
71 {
72 return false;
73 }
74
75 if (!source->IsAlive() || !target->IsAlive())
76 {
77 return false;
78 }
79
80 if (!source->InSamePhase(target))
81 {
82 return false;
83 }
84
85 if (source->HasUnitState(UNIT_STATE_IN_FLIGHT) || target->HasUnitState(UNIT_STATE_IN_FLIGHT))
86 {
87 return false;
88 }
89
90 return true;
91}

93void CreatureAI::DoZoneInCombat(Creature* creature /*= nullptr*/, float maxRangeToNearestTarget /*= 250.0f*/)
94{
95 if (!creature)
96 {
97 creature = me;
98 }
99
100 if (creature->IsInEvadeMode())
101 {
102 return;
103 }
104
105 Map* map = creature->GetMap();
106 if (!map->IsDungeon()) //use IsDungeon instead of Instanceable, in case battlegrounds will be instantiated
107 {
108 LOG_ERROR("entities.unit.ai", "DoZoneInCombat call for map {} that isn't a dungeon (creature entry = {})", map->GetId(), creature->GetTypeId() == TYPEID_UNIT ? creature->ToCreature()->GetEntry() : 0);
109 return;
110 }
111
112 Map::PlayerList const& playerList = map->GetPlayers();
113 if (playerList.IsEmpty())
114 {
115 return;
116 }
117
118 for (Map::PlayerList::const_iterator itr = playerList.begin(); itr != playerList.end(); ++itr)
119 {
120 if (Player* player = itr->GetSource())
121 {
122 if (!IsValidCombatTarget(creature, player))
123 {
124 continue;
125 }
126
127 if (!creature->IsWithinDistInMap(player, maxRangeToNearestTarget))
128 {
129 continue;
130 }
131
132 creature->SetInCombatWith(player);
133 player->SetInCombatWith(creature);
134
135 if (creature->CanHaveThreatList())
136 {
137 creature->AddThreat(player, 0.0f);
138 }
139 }
140 }
141}

156void CreatureAI::MoveInLineOfSight(Unit* who)
157{
158 if (me->IsEngaged())
159 return;
160
161 // pussywizard: civilian, non-combat pet or any other NOT HOSTILE TO ANYONE (!)
162 if (me->IsMoveInLineOfSightDisabled())
163 if (me->GetCreatureType() == CREATURE_TYPE_NON_COMBAT_PET || // nothing more to do, return
164 !who->IsInCombat() || // if not in combat, nothing more to do
165 !me->IsWithinDist(who, ATTACK_DISTANCE, true, false)) // if in combat and in dist - neutral to all can actually assist other creatures
166 return;
167

172// Distract creature, if player gets too close while stealthed/prowling
173void CreatureAI::TriggerAlert(Unit const* who) const
174{
175 // If there's no target, or target isn't a player do nothing
176 if (!who || who->GetTypeId() != TYPEID_PLAYER)
177 return;
178 // If this unit isn't an NPC, is already distracted, is in combat, is confused, stunned or fleeing, do nothing
179 if (me->GetTypeId() != TYPEID_UNIT || me->IsEngaged() || me->HasUnitState(UNIT_STATE_CONFUSED | UNIT_STATE_STUNNED | UNIT_STATE_FLEEING | UNIT_STATE_DISTRACTED))
180 return;
181 // Only alert for hostiles!
182 if (me->IsCivilian() || me->HasReactState(REACT_PASSIVE) || !me->IsHostileTo(who) || !me->_IsTargetAcceptable(who))
183 return;
184 // Only alert if target is within line of sight
185 if (!me->IsWithinLOSInMap(who))
186 return;
187 // Send alert sound (if any) for this creature
188 me->SendAIReaction(AI_REACTION_ALERT);
189 // Face the unit (stealthed player) and set distracted state for 5 seconds
190 me->SetFacingTo(me->GetAngle(who->GetPositionX(), who->GetPositionY()));
191 me->StopMoving();
192 me->GetMotionMaster()->MoveDistract(5 * IN_MILLISECONDS);
193}

195void CreatureAI::EnterEvadeMode(EvadeReason why)
196{
197 if (!_EnterEvadeMode(why))
198 return;
199
200 LOG_DEBUG("entities.unit", "Creature {} enters evade mode.", me->GetEntry());
201
202 if (!me->GetVehicle()) // otherwise me will be in evade mode forever
203 {
204 if (Unit* owner = me->GetCharmerOrOwner())
205 {
206 me->GetMotionMaster()->Clear(false);
207 me->GetMotionMaster()->MoveFollow(owner, PET_FOLLOW_DIST, me->GetFollowAngle(), MOTION_SLOT_ACTIVE);
208 }
209 else
210 {
211 // Required to prevent attacking creatures that are evading and cause them to reenter combat
212 // Does not apply to MoveFollow
213 me->AddUnitState(UNIT_STATE_EVADE);
214 me->GetMotionMaster()->MoveTargetedHome();
215 }
216 }
217
218 Reset();
219 if (me->IsVehicle()) // use the same sequence of addtoworld, aireset may remove all summons!
220 {
221 me->GetVehicleKit()->Reset(true);
222 }
223
224 // despawn bosses at reset - only verified tbc/woltk bosses with this reset type - add bosses in last line respectively (dungeon/raid) and increase array limit
225 CreatureTemplate const* cInfo = sObjectMgr->GetCreatureTemplate(me->GetEntry());
226 if (cInfo && cInfo->HasFlagsExtra(CREATURE_FLAG_EXTRA_HARD_RESET))
227 {
228 me->DespawnOnEvade();
229 me->m_Events.AddEvent(new PhasedRespawn(*me), me->m_Events.CalculateTime(20000));
230 }
231
232 sScriptMgr->OnUnitEnterEvadeMode(me, why);
233}

268bool CreatureAI::UpdateVictim()
269{
270 if (!me->IsEngaged())
271 return false;
272
273 if (!me->HasReactState(REACT_PASSIVE))
274 {
275 if (Unit* victim = me->SelectVictim())
276 AttackStart(victim);
277 return me->GetVictim();

279 // xinef: if we have any victim, just return true

291bool CreatureAI::_EnterEvadeMode(EvadeReason /*why*/)
292{
293 if (!me->IsAlive())
294 {
295 return false;
296 }
297
298 // don't remove vehicle auras, passengers aren't supposed to drop off the vehicle
299 // don't remove clone caster on evade (to be verified)
300 me->RemoveEvadeAuras();
301
302 me->ClearComboPointHolders(); // Remove all combo points targeting this unit
303 // sometimes bosses stuck in combat?
304 me->GetThreatMgr().ClearAllThreat();
305 me->CombatStop(true);
306 me->LoadCreaturesAddon(true);
307 me->SetLootRecipient(nullptr);
308 me->ResetPlayerDamageReq();
309 me->SetLastDamagedTime(0);
310 me->SetCannotReachTarget();
311
312 if (ZoneScript* zoneScript = me->GetZoneScript() ? me->GetZoneScript() : (ZoneScript*)me->GetInstanceScript())
313 zoneScript->OnCreatureEvade(me);
314
315 if (me->IsInEvadeMode())
316 {
317 return false;
318 }
319 else if (CreatureGroup* formation = me->GetFormation())
320 {
321 formation->MemberEvaded(me);
322 }
323
324 return true;
325}

327void CreatureAI::MoveCircleChecks()
328{
329 Unit *victim = me->GetVictim();
330
331 if (
332 !victim ||
333 !me->IsFreeToMove() || me->HasUnitMovementFlag(MOVEMENTFLAG_ROOT) ||
334 !me->IsWithinMeleeRange(victim) || me == victim->GetVictim() ||
335 (victim->GetTypeId() != TYPEID_PLAYER && !victim->IsPet()) // only player & pets to save CPU
336 )
337 {
338 return;
339 }
340
341 me->GetMotionMaster()->MoveCircleTarget(me->GetVictim());
