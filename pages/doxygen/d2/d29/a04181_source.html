279 void CalculateResistanceAmount(AuraEffect const* aurEff, int32& amount, bool& /*canBeRecalculated*/)

294 {
295 Stats stat = Stats(aurEff->GetSpellInfo()->Effects[aurEff->GetEffIndex()].MiscValue);
296 int32 modifier = stat == STAT_STAMINA ? 75 : 30;
297 amount = CalculatePct(std::max<int32>(0, owner->GetStat(stat)), modifier);
298 }
299 }
300
301 void CalculateAPAmount(AuraEffect const* /*aurEff*/, int32& amount, bool& /*canBeRecalculated*/)
302 {
303 // xinef: by default warlock pet inherits 57% of max(SP FIRE, SP SHADOW) as AP

307 int32 shadow = owner->SpellBaseDamageBonusDone(SPELL_SCHOOL_MASK_SHADOW);

309 amount = CalculatePct(std::max<int32>(0, maximum), 57);

335 void HandlePeriodic(AuraEffect const* aurEff)
336 {
337 PreventDefaultAction();
338 if (aurEff->GetAuraType() == SPELL_AURA_MOD_STAT && (aurEff->GetMiscValue() == STAT_STAMINA || aurEff->GetMiscValue() == STAT_INTELLECT))
339 {
340 int32 currentAmount = aurEff->GetAmount();
341 int32 newAmount = GetEffect(aurEff->GetEffIndex())->CalculateAmount(GetCaster());
342 if (newAmount != currentAmount)
343 {
344 if (aurEff->GetMiscValue() == STAT_STAMINA)
345 {
346 uint32 actStat = GetUnitOwner()->GetHealth();
347 GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
348 GetUnitOwner()->SetHealth(std::min<uint32>(GetUnitOwner()->GetMaxHealth(), actStat));
349 }
350 else
351 {
352 uint32 actStat = GetUnitOwner()->GetPower(POWER_MANA);
353 GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
354 GetUnitOwner()->SetPower(POWER_MANA, std::min<uint32>(GetUnitOwner()->GetMaxPower(POWER_MANA), actStat));
355 }
356 }
357 }
358 else
359 GetEffect(aurEff->GetEffIndex())->RecalculateAmount();
360 }
361
362 void Register() override
363 {
364 if (m_scriptSpellId != 34947)
365 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_warl_generic_scaling::CalculateResistanceAmount, EFFECT_ALL, SPELL_AURA_MOD_RESISTANCE);
366
367 if (m_scriptSpellId == 34947 || m_scriptSpellId == 34956)
368 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_warl_generic_scaling::CalculateStatAmount, EFFECT_ALL, SPELL_AURA_MOD_STAT);
369
370 if (m_scriptSpellId == 34947)
371 {
372 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_warl_generic_scaling::CalculateAPAmount, EFFECT_ALL, SPELL_AURA_MOD_ATTACK_POWER);
373 DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_warl_generic_scaling::CalculateSPAmount, EFFECT_ALL, SPELL_AURA_MOD_DAMAGE_DONE);
