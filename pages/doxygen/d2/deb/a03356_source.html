76 void Reset() override
77 {
78 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
79 events.Reset();
80 if (me->HasReactState(REACT_AGGRESSIVE)) // Reset() called by EnterEvadeMode()
81 {
82 if (!pInstance)
83 return;
84 if (Creature* c = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_MARTIN_OR_GORKUN_GUID)))

86 c->AI()->DoAction(1);

104 void DoAction(int32 param) override
105 {
106 if (param == 1)
107 {
108 Position exitPos = {1023.46f, 159.12f, 628.2f, 5.23f};

116 me->GetMotionMaster()->MoveJump(exitPos, 10.0f, 2.0f);
117
118 // start real fight
119 me->RemoveAllAuras();
120 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
121 DoZoneInCombat();
122 me->CastSpell(me, 43979, true);
123 Talk(SAY_AGGRO);
124 events.Reset();
125 events.RescheduleEvent(EVENT_SPELL_FORCEFUL_SMASH, urand(14000, 16000));
126 events.RescheduleEvent(EVENT_SPELL_OVERLORDS_BRAND, urand(4000, 6000));
127 events.RescheduleEvent(EVENT_RIMEFANG_SPELL_ICY_BLAST, 5000);
128 events.RescheduleEvent(EVENT_SPELL_MARK_OF_RIMEFANG, 25000);
129 }

132 void UpdateAI(uint32 diff) override
133 {
134 if (!UpdateVictim())
135 return;
136
137 if (me->GetVictim())
138 {
139 float x, y, z;
140 me->GetVictim()->GetPosition(x, y, z);
141 if (TSDistCheckPos.GetExactDist(x, y, z) > 100.0f || z > TSDistCheckPos.GetPositionZ() + 20.0f || z < TSDistCheckPos.GetPositionZ() - 20.0f)
142 {
143 me->SetHealth(me->GetMaxHealth());
144 EnterEvadeMode();
145 return;
146 }
147 }
148
149 events.Update(diff);
150
151 if (me->HasUnitState(UNIT_STATE_CASTING))
152 return;
153
154 switch(events.ExecuteEvent())
155 {
156 case 0:
157 break;
158 case EVENT_SPELL_FORCEFUL_SMASH:
159 if (me->IsWithinMeleeRange(me->GetVictim()))
160 {
161 me->CastSpell(me->GetVictim(), SPELL_FORCEFUL_SMASH, false);
162
163 events.RescheduleEvent(EVENT_SPELL_UNHOLY_POWER, 1000);
164 break;
165 }
166 events.RepeatEvent(3000);
167 break;
168 case EVENT_SPELL_UNHOLY_POWER:
169 Talk(SAY_SMASH);
170 Talk(EMOTE_SMASH);
171 me->CastSpell(me, SPELL_UNHOLY_POWER, false);
172
173 events.ScheduleEvent(EVENT_SPELL_FORCEFUL_SMASH, urand(40000, 48000));
174 break;
175 case EVENT_SPELL_OVERLORDS_BRAND:
176 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 95.0f, true))
177 me->CastSpell(target, SPELL_OVERLORDS_BRAND, false);
178 events.RepeatEvent(urand(11000, 12000));
179 break;
180 case EVENT_RIMEFANG_SPELL_ICY_BLAST:
181 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 190.0f, true))
182 if (Creature* c = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_RIMEFANG_GUID)))
183 c->CastSpell(target, RIMEFANG_SPELL_ICY_BLAST, false);
184 events.RepeatEvent(5000);
185 break;
186 case EVENT_SPELL_MARK_OF_RIMEFANG:
187 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 190.0f, true))
188 if (Creature* c = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_RIMEFANG_GUID)))
189 {
190 Talk(SAY_MARK);
191 c->AI()->Talk(EMOTE_RIMEFANG_ICEBOLT, target);
192 c->CastSpell(target, RIMEFANG_SPELL_HOARFROST, false);
193 }
194 events.RepeatEvent(25000);
195 events.RescheduleEvent(EVENT_RIMEFANG_SPELL_ICY_BLAST, 10000);
196 break;
197 }
198
199 DoMeleeAttackIfReady();
200 }
