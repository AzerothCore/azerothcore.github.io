81 void UpdateAI(uint32 diff) override
82 {
83 if (!UpdateVictim())
84 return;
85
86 events.Update(diff);
87 switch (events.ExecuteEvent())
88 {
89 case EVENT_CHECK_HEALTH1:
90 if (me->HealthBelowPct(67) && !health67)
91 {
92 me->CastSpell(me, SPELL_SMITE_STOMP, false);
93 events.DelayEvents(10000);
94 me->GetMotionMaster()->Clear();
95 me->GetMotionMaster()->MovePoint(EQUIP_TWO_SWORDS, 1.859f, -780.72f, 9.831f);
96 Talk(SAY_SWAP1);
97 me->SetUnitFlag(UNIT_FLAG_PACIFIED);
98 me->SetReactState(REACT_PASSIVE);
99 health67 = true;
100 break;
101 }
102 events.ScheduleEvent(EVENT_CHECK_HEALTH1, 500);
103 break;
104 case EVENT_CHECK_HEALTH2:
105 if (me->HealthBelowPct(34) && !health34)
106 {
107 me->CastSpell(me, SPELL_SMITE_STOMP, false);
108 events.DelayEvents(10000);
109 me->GetMotionMaster()->Clear();
110 me->GetMotionMaster()->MovePoint(EQUIP_MACE, 1.859f, -780.72f, 9.831f);
111 Talk(SAY_SWAP2);
112 me->SetUnitFlag(UNIT_FLAG_PACIFIED);
113 me->SetReactState(REACT_PASSIVE);
114 health34 = true;
115 break;
116 }
117 events.ScheduleEvent(EVENT_CHECK_HEALTH2, 500);
118 break;
119 case EVENT_SMITE_SLAM:
120 if (me->HealthBelowPct(33))
121 {
122 me->CastSpell(me->GetVictim(), SPELL_SMITE_SLAM, false);
123 events.ScheduleEvent(EVENT_SMITE_SLAM, 6000);
124 break;
125 }
126 events.ScheduleEvent(EVENT_SMITE_SLAM, 500);
127 break;
128 case EVENT_SWAP_WEAPON1:
129 me->LoadEquipment(EQUIP_TWO_SWORDS);
130 me->SetCanDualWield(true);
131 break;
132 case EVENT_SWAP_WEAPON2:
133 me->LoadEquipment(EQUIP_MACE);
134 me->SetCanDualWield(false);
135 break;
136 case EVENT_RESTORE_COMBAT:
137 me->SetReactState(REACT_AGGRESSIVE);
138 me->RemoveUnitFlag(UNIT_FLAG_PACIFIED);
139 me->SetStandState(UNIT_STAND_STATE_STAND);
140 if (me->GetVictim())
141 {
142 me->GetMotionMaster()->MoveChase(me->GetVictim());
143 me->SetTarget(me->GetVictim()->GetGUID());
144 }
145 break;
146 case EVENT_KNEEL:
147 me->SendMeleeAttackStop(me->GetVictim());
148 me->SetStandState(UNIT_STAND_STATE_KNEEL);
149 break;
150 }
151
152 DoMeleeAttackIfReady();
153 }
154
155 void MovementInform(uint32 type, uint32 point) override
156 {

158 return;
159
160 me->SetTarget();
161 me->SetFacingTo(5.558f);
162 me->SetStandState(UNIT_STAND_STATE_KNEEL);
163 events.ScheduleEvent(point == EQUIP_TWO_SWORDS ? EVENT_SWAP_WEAPON1 : EVENT_SWAP_WEAPON2, 1500);
164 events.ScheduleEvent(EVENT_RESTORE_COMBAT, 3000);
165 events.ScheduleEvent(EVENT_KNEEL, 0);
166 }
