37 bool OnGossipHello(Player* player, Creature* creature) override
38 {
39 if(!creature->HasNpcFlag(UNIT_NPC_FLAG_GOSSIP))
40 return true;
41
42 InstanceScript* pInstance = creature->GetInstanceScript();
43 if(!pInstance)
44 return true;
45
46 uint32 gossipTextId = 0;
47 switch(pInstance->GetData(TYPE_INSTANCE_PROGRESS))
48 {
49 case INSTANCE_PROGRESS_INITIAL:
50 gossipTextId = MSG_TESTED;
51 break;
52 case INSTANCE_PROGRESS_INTRO_DONE:
53 case INSTANCE_PROGRESS_BEASTS_DEAD:
54 case INSTANCE_PROGRESS_FACTION_CHAMPIONS_DEAD:
55 case INSTANCE_PROGRESS_VALKYR_DEAD:
56 gossipTextId = MSG_NEXT_STAGE;
57 break;
58 case INSTANCE_PROGRESS_JARAXXUS_DEAD:
59 gossipTextId = MSG_CRUSADERS;
60 break;
61 case INSTANCE_PROGRESS_DONE:
62 creature->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP);
63 return true;
64 default:
65 return true;
66 }
67
68 AddGossipItemFor(player, GOSSIP_ICON_CHAT, "We are ready!", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1337);
69 SendGossipMenuFor(player, gossipTextId, creature->GetGUID());
70 return true;
71 }
72
73 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 uiAction) override
74 {
75 if( !creature->HasNpcFlag(UNIT_NPC_FLAG_GOSSIP) )
76 return true;
77
78 InstanceScript* pInstance = creature->GetInstanceScript();
79 if( !pInstance )
80 return true;
81
82 if( uiAction == GOSSIP_ACTION_INFO_DEF + 1337 )
83 {
84 pInstance->SetData(TYPE_ANNOUNCER_GOSSIP_SELECT, 0);
85 creature->RemoveNpcFlag(UNIT_NPC_FLAG_GOSSIP);
86 }
87
88 CloseGossipMenuFor(player);
89 return true;
90 }
91};
