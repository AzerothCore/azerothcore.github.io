254 {
255 ThreatContainer::StorageType& threatList = iThreatContainer.iThreatList;
256 if (threatList.empty())
257 return;
258
259 for (auto ref : threatList)
260 {
261 if (predicate(ref->getTarget()))

263 ref->SetThreat(0);
264 setDirty(true);
