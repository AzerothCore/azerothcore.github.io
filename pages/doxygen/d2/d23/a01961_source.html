23template<typename ScriptName>
24inline Optional<bool> IsValidBoolScript(std::function<bool(ScriptName*)> executeHook)
25{
26 if (ScriptRegistry<ScriptName>::ScriptPointerList.empty())
27 return {};
28
29 for (auto const& [scriptID, script] : ScriptRegistry<ScriptName>::ScriptPointerList)
30 {
31 if (executeHook(script))
32 return true;
33 }
34
35 return false;
36}

38template<typename ScriptName, class T>
39inline T* GetReturnAIScript(std::function<T*(ScriptName*)> executeHook)
40{
41 if (ScriptRegistry<ScriptName>::ScriptPointerList.empty())
42 return nullptr;
43
44 for (auto const& [scriptID, script] : ScriptRegistry<ScriptName>::ScriptPointerList)
45 {
46 if (T* scriptAI = executeHook(script))
47 {
48 return scriptAI;
49 }
50 }
51
52 return nullptr;
53}

55template<typename ScriptName>
56inline void ExecuteScript(std::function<void(ScriptName*)> executeHook)
57{
58 if (ScriptRegistry<ScriptName>::ScriptPointerList.empty())
59 return;
60
61 for (auto const& [scriptID, script] : ScriptRegistry<ScriptName>::ScriptPointerList)
