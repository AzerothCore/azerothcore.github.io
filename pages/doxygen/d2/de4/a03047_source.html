137 void UpdateAI(uint32 diff) override
138 {
139 if (!eventInProgress)
140 return;
141
142 events.Update(diff);
143
144 while (uint32 eventId = events.ExecuteEvent())
145 {
146 switch (eventId)
147 {
148 case EVENT_CHANNEL:
149 Talk(SAY_EVENT_START);
150 DoCast(me, SPELL_IDOL_SHUTDOWN_VISUAL);
151 events.ScheduleEvent(EVENT_IDOL_ROOM_SPAWNER, 100);
152 events.ScheduleEvent(EVENT_PROGRESS, 120000);
153 me->SetReactState(REACT_PASSIVE);
154 break;
155 case EVENT_IDOL_ROOM_SPAWNER:
156 if (Creature* creature = me->SummonCreature(NPC_IDOL_ROOM_SPAWNER, PosSummonSpawner[urand(0, 2)], TEMPSUMMON_TIMED_DESPAWN, 4000))
157 creature->AI()->SetData(0, spawnerCount);
158 if (++spawnerCount < 8)
159 events.ScheduleEvent(EVENT_IDOL_ROOM_SPAWNER, 35000);
160 break;
161 case EVENT_PROGRESS:
162 {
163 switch (eventProgress)
164 {
165 case 0:
166 Talk(SAY_EVENT_THREE_MIN_LEFT);
167 ++eventProgress;
168 events.ScheduleEvent(EVENT_PROGRESS, 60000);
169 break;
170 case 1:
171 Talk(SAY_EVENT_TWO_MIN_LEFT);
172 ++eventProgress;
173 events.ScheduleEvent(EVENT_PROGRESS, 60000);
174 break;
175 case 2:
176 Talk(SAY_EVENT_ONE_MIN_LEFT);
177 ++eventProgress;
178 events.ScheduleEvent(EVENT_PROGRESS, 60000);
179 break;
180 case 3:
181 events.CancelEvent(EVENT_IDOL_ROOM_SPAWNER);
182 me->InterruptSpell(CURRENT_CHANNELED_SPELL);
183 Talk(SAY_EVENT_END);
184 events.ScheduleEvent(EVENT_COMPLETE, 3000);
185 break;
186 }
187 break;
188 }
189 case EVENT_COMPLETE:
190 {
191 DoCast(me, SPELL_IDOM_ROOM_CAMERA_SHAKE);
192 me->SummonGameObject(GO_BELNISTRASZS_BRAZIER, 2577.196f, 947.0781f, 53.16757f, 2.356195f, 0, 0, 0.9238796f, 0.3826832f, 3600);
193 std::list<WorldObject*> ClusterList;
194 Acore::AllWorldObjectsInRange objects(me, 50.0f);
195 Acore::WorldObjectListSearcher<Acore::AllWorldObjectsInRange> searcher(me, ClusterList, objects);
196 Cell::VisitAllObjects(me, searcher, 50.0f);
197 for (std::list<WorldObject*>::const_iterator itr = ClusterList.begin(); itr != ClusterList.end(); ++itr)
198 {
199 if (Player* player = (*itr)->ToPlayer())
200 {
201 if (player->GetQuestStatus(QUEST_EXTINGUISHING_THE_IDOL) == QUEST_STATUS_INCOMPLETE)
202 player->CompleteQuest(QUEST_EXTINGUISHING_THE_IDOL);
203 }
204 else if (GameObject* go = (*itr)->ToGameObject())
205 {
206 if (go->GetEntry() == GO_IDOL_OVEN_FIRE || go->GetEntry() == GO_IDOL_CUP_FIRE || go->GetEntry() == GO_IDOL_MOUTH_FIRE)
207 go->Delete();
208 }
209 }
210 instance->SetData(GO_BELNISTRASZS_BRAZIER, DONE);
211 me->DespawnOrUnsummon();
212 break;
213 }
214 case EVENT_FIREBALL:
215 if (me->HasUnitState(UNIT_STATE_CASTING) || !UpdateVictim())
216 return;
217 DoCastVictim(SPELL_FIREBALL);
218 events.ScheduleEvent(EVENT_FIREBALL, 8000);
219 break;
220 case EVENT_FROST_NOVA:
221 if (me->HasUnitState(UNIT_STATE_CASTING) || !UpdateVictim())
222 return;
223 DoCast(me, SPELL_FROST_NOVA);
224 events.ScheduleEvent(EVENT_FROST_NOVA, 15000);
225 break;
226 }
227 }
228 if (!channeling)
229 DoMeleeAttackIfReady();
230 }

232 private:
233 InstanceScript* instance;
234 EventMap events;
235 bool eventInProgress;
236 bool channeling;
237 uint8 eventProgress;
238 uint8 spawnerCount;
239 };
