201 void UpdateAI(uint32 diff) override
202 {
203 if (!UpdateVictim())
204 {
205 events.Update(diff);
206
207 while (uint32 eventId = events.ExecuteEvent())
208 {
209 switch (eventId)
210 {
211 case EVENT_RESPAWN:
212 {
213 // Respawn all Blackhand Incarcerators
214 std::list<Creature*> creatureList;
215 GetCreatureListWithEntryInGrid(creatureList, me, NPC_BLACKHAND_INCARCERATOR, 35.0f);
216 for (std::list<Creature*>::iterator itr = creatureList.begin(); itr != creatureList.end(); ++itr)
217 if (Creature* creature = *itr)
218 {
219 if (!creature->IsAlive())
220 creature->Respawn();
221
222 creature->AI()->SetData(1, 2);
223 }
224 me->AddAura(SPELL_ENCAGED_EMBERSEER, me);
225 instance->SetBossState(DATA_PYROGAURD_EMBERSEER, NOT_STARTED);
226 break;
227 }
228 case EVENT_PRE_FIGHT_1:
229 {
230 // Set data on all Blackhand Incarcerators
231 std::list<Creature*> creatureList;
232 GetCreatureListWithEntryInGrid(creatureList, me, NPC_BLACKHAND_INCARCERATOR, 35.0f);
233 for (std::list<Creature*>::iterator itr = creatureList.begin(); itr != creatureList.end(); ++itr)
234 {
235 if (Creature* creature = *itr)
236 creature->AI()->SetData(1, 1);
237 }
238 events.ScheduleEvent(EVENT_PRE_FIGHT_2, 32000);
239 break;
240 }
241 case EVENT_PRE_FIGHT_2:
242 me->CastSpell(me, SPELL_FREEZE_ANIM);
243 me->CastSpell(me, SPELL_EMBERSEER_GROWING);
244 Talk(EMOTE_ONE_STACK);
245 break;
246 case EVENT_FIRE_SHIELD:
247 // #### Spell isn't doing any damage ??? ####
248 DoCast(me, SPELL_FIRE_SHIELD);
249 events.ScheduleEvent(EVENT_FIRE_SHIELD, 3000);
250 break;
251 case EVENT_PLAYER_CHECK:
252 {
253 // Check to see if all players in instance have aura SPELL_EMBERSEER_START before starting event
254 bool _hasAura = false;
255 Map::PlayerList const& players = me->GetMap()->GetPlayers();
256 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
257 if (Player* player = itr->GetSource()->ToPlayer())
258 if (player->HasAura(SPELL_EMBERSEER_OBJECT_VISUAL))
259 _hasAura = true;
260
261 if (_hasAura)
262 {
263 events.ScheduleEvent(EVENT_PRE_FIGHT_1, 1000);
264 instance->SetBossState(DATA_PYROGAURD_EMBERSEER, IN_PROGRESS);
265 }
266 break;
267 }
268 case EVENT_ENTER_COMBAT:
269 DoZoneInCombat();
270 break;
271 default:
272 break;
273 }
274 }
275 return;
276 }
277
278 events.Update(diff);
279
280 while (uint32 eventId = events.ExecuteEvent())
281 {
282 switch (eventId)
283 {
284 case EVENT_FIRE_SHIELD:
285 DoCast(me, SPELL_FIRE_SHIELD);
286 events.ScheduleEvent(EVENT_FIRE_SHIELD, 3000);
287 break;
288 case EVENT_FIRENOVA:
289 DoCast(me, SPELL_FIRENOVA);
290 events.ScheduleEvent(EVENT_FIRENOVA, 6000);
291 break;
292 case EVENT_FLAMEBUFFET:
293 DoCast(me, SPELL_FLAMEBUFFET);
294 events.ScheduleEvent(EVENT_FLAMEBUFFET, 14000);
295 break;
296 case EVENT_PYROBLAST:
297 DoCastRandomTarget(SPELL_PYROBLAST, 0, 100.0f);
298 events.ScheduleEvent(EVENT_PYROBLAST, 15000);
299 break;
300 default:
301 break;
302 }
303 }
304 DoMeleeAttackIfReady();
305 }
306 };
