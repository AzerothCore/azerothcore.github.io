24bool Acore::PlayerCommand::HandleLearnSpellCommand(ChatHandler* handler, Player* targetPlayer, SpellInfo const* spell, Optional<EXACT_SEQUENCE("all")> allRanks)
25{
26 if (!SpellMgr::IsSpellValid(spell))
27 {
28 handler->PSendSysMessage(LANG_COMMAND_SPELL_BROKEN, spell->Id);
29 handler->SetSentErrorMessage(true);
30 return false;
31 }
32
33 if (!allRanks && targetPlayer->HasSpell(spell->Id))
34 {
35 if (targetPlayer == handler->GetPlayer())
36 {
37 handler->SendSysMessage(LANG_YOU_KNOWN_SPELL);
38 }
39 else
40 {
41 handler->PSendSysMessage(LANG_TARGET_KNOWN_SPELL, handler->GetNameLink(targetPlayer).c_str());
42 }
43
44 handler->SetSentErrorMessage(true);
45 return false;
46 }
47
48 targetPlayer->learnSpell(spell->Id, false);
49
50 if (allRanks)
51 {
52 uint32 spellId = spell->Id;
53
54 while ((spellId = sSpellMgr->GetNextSpellInChain(spellId)))
55 {
56 targetPlayer->learnSpell(spellId, false);
57 }
58 }
59
60 if (GetTalentSpellCost(spell->GetFirstRankSpell()->Id))
61 {
62 targetPlayer->SendTalentsInfoData(false);
63 }
64
65 return true;
66}

68bool Acore::PlayerCommand::HandleUnlearnSpellCommand(ChatHandler* handler, Player* target, SpellInfo const* spell, Optional<EXACT_SEQUENCE("all")> allRanks)
69{
70 uint32 spellId = spell->Id;
71
72 if (allRanks)
73 {
74 spellId = sSpellMgr->GetFirstSpellInChain(spellId);
75 }
76
77 if (target->HasSpell(spellId))

80 }

84 }

89 }
90
91 return true;
