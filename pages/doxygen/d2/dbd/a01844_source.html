29 template<>
30 inline void ForeachMaps(Map* map, std::function<void(WorldMapScript*)> executeHook)
31 {
32 auto mapEntry = map->GetEntry();
33
34 if (!mapEntry)
35 {
36 return;
37 }
38
39 if (!mapEntry->IsWorldMap())
40 {
41 return;
42 }
43
44 if (ScriptRegistry<WorldMapScript>::ScriptPointerList.empty())
45 {
46 return;
47 }
48
49 for (auto const& [scriptID, script] : ScriptRegistry<WorldMapScript>::ScriptPointerList)
50 {
51 MapEntry const* mapEntry = script->GetEntry();
52 if (!mapEntry)
53 {
54 continue;
55 }
56
57 if (mapEntry->MapID != map->GetId())
58 {
59 continue;
60 }
61
62 executeHook(script);
63 return;
64 }
65 }
