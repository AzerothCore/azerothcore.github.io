34void Totem::Update(uint32 time)
35{
36 Unit* owner = GetOwner();
37 if (!owner || !owner->IsAlive() || !IsAlive() || m_duration <= time)
38 {
39 UnSummon(); // remove self
40 return;
41 }
42
43 m_duration -= time;

47void Totem::InitStats(uint32 duration)
48{
49 // client requires SMSG_TOTEM_CREATED to be sent before adding to world and before removing old totem
50 // Xinef: Set level for Unit totems
51 if (Unit* owner = ObjectAccessor::GetUnit(*this, m_owner))
52 {
53 uint32 slot = m_Properties->Slot;
54 if (owner->GetTypeId() == TYPEID_PLAYER && slot >= SUMMON_SLOT_TOTEM && slot < MAX_TOTEM_SLOT)
55 {
56 WorldPackets::Totem::TotemCreated data;
57 data.Totem = GetGUID();
58 data.Slot = slot - SUMMON_SLOT_TOTEM;
59 data.Duration = duration;
60 data.SpellID = GetUInt32Value(UNIT_CREATED_BY_SPELL);
61 owner->ToPlayer()->SendDirectMessage(data.Write());
62
63 // set display id depending on caster's race
64 SetDisplayId(owner->GetModelForTotem(PlayerTotemType(m_Properties->Id)));
65 }
66
67 SetLevel(owner->getLevel());
68 }
69
70 Minion::InitStats(duration);
71
72 // Get spell cast by totem
73 if (SpellInfo const* totemSpell = sSpellMgr->GetSpellInfo(GetSpell()))
74 if (totemSpell->CalcCastTime()) // If spell has cast time -> its an active totem
75 m_type = TOTEM_ACTIVE;
76
77 m_duration = duration;
78}

80void Totem::InitSummon()
81{
82 Minion::InitSummon();
83
84 if (m_type == TOTEM_PASSIVE && GetSpell())
85 CastSpell(this, GetSpell(), true);
86
87 // Some totems can have both instant effect and passive spell
88 if(GetSpell(1))
89 CastSpell(this, GetSpell(1), true);
90
91 // xinef: this is better than the script, 100% sure to work
92 if(GetEntry() == SENTRY_TOTEM_ENTRY)
93 {
94 SetReactState(REACT_AGGRESSIVE);
95 GetOwner()->CastSpell(this, 6277, true);
96

99 SetVisibilityDistanceOverride(VisibilityDistanceType::Infinite);

101

108void Totem::UnSummon(uint32 msTime)
109{
110 if (msTime)
111 {
112 m_Events.AddEvent(new ForcedUnsummonDelayEvent(*this), m_Events.CalculateTime(msTime));
113 return;
114 }
115
116 CombatStop();
117 RemoveAurasDueToSpell(GetSpell(), GetGUID());
118
119 if (Unit* owner = GetOwner())
120 {
121 // clear owner's totem slot
122 for (uint8 i = SUMMON_SLOT_TOTEM; i < MAX_TOTEM_SLOT; ++i)
123 {
124 if (owner->m_SummonSlot[i] == GetGUID())
125 {
126 owner->m_SummonSlot[i].Clear();
127 break;
128 }
129 }
130
131 owner->RemoveAurasDueToSpell(GetSpell(), GetGUID());
132
133 // Remove Sentry Totem Aura
134 if (GetEntry() == SENTRY_TOTEM_ENTRY)
135 owner->RemoveAurasDueToSpell(SENTRY_TOTEM_SPELLID);
136
137 //remove aura all party members too
138 if (Player* player = owner->ToPlayer())
139 {
140 player->SendAutoRepeatCancel(this);
141
142 if (SpellInfo const* spell = sSpellMgr->GetSpellInfo(GetUInt32Value(UNIT_CREATED_BY_SPELL)))
143 player->SendCooldownEvent(spell, 0, nullptr, false);
144
145 if (Group* group = player->GetGroup())
146 {
147 for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
148 {
149 Player* target = itr->GetSource();
150 if (target && target->IsInMap(player) && group->SameSubGroup(player, target))
151 target->RemoveAurasDueToSpell(GetSpell(), GetGUID());
152 }
153 }
154 }
155 }
156
157 AddObjectToRemoveList();
158}
