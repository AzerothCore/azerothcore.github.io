114 void UpdateAI(uint32 diff) override
115 {
116 if (!UpdateVictim())
117 return;
118
119 events.Update(diff);
120 if (me->HasUnitState(UNIT_STATE_CASTING))
121 return;
122
123 switch (events.ExecuteEvent())
124 {
125 case EVENT_HATEFUL_STRIKE:
126 {
127 // Cast Hateful strike on the player with the highest amount of HP within melee distance, and second threat amount
128 std::list<Unit*> meleeRangeTargets;
129 Unit* finalTarget = nullptr;
130 uint8 counter = 0;
131 auto i = me->GetThreatMgr().GetThreatList().begin();
132 for (; i != me->GetThreatMgr().GetThreatList().end(); ++i, ++counter)
133 {
134 // Gather all units with melee range
135 Unit* target = (*i)->getTarget();
136 if (me->IsWithinMeleeRange(target))
137 {
138 meleeRangeTargets.push_back(target);
139 }
140 // and add threat to most hated
141 if (counter < RAID_MODE(2, 3))
142 {
143 me->AddThreat(target, 500.0f);
144 }
145 }
146 counter = 0;
147 std::list<Unit*, std::allocator<Unit*>>::iterator itr;
148 for (itr = meleeRangeTargets.begin(); itr != meleeRangeTargets.end(); ++itr, ++counter)
149 {
150 // if there is only one target available
151 if (meleeRangeTargets.size() == 1)
152 {
153 finalTarget = (*itr);
154 }
155 else if (counter > 0) // skip first target
156 {
157 if (!finalTarget || (*itr)->GetHealth() > finalTarget->GetHealth())
158 {
159 finalTarget = (*itr);
160 }
161 // third loop
162 if (counter >= 2)
163 break;
164 }
165 }
166 if (finalTarget)
167 {
168 me->CastSpell(finalTarget, RAID_MODE(SPELL_HATEFUL_STRIKE_10, SPELL_HATEFUL_STRIKE_25), false);
169 }
170 events.RepeatEvent(1000);
171 break;
172 }
173 case EVENT_BERSERK:
174 Talk(EMOTE_BERSERK);
175 me->CastSpell(me, SPELL_BERSERK, true);
176 events.ScheduleEvent(EVENT_SLIME_BOLT, 3000);
177 break;
178 case EVENT_SLIME_BOLT:
179 me->CastSpell(me, SPELL_SLIME_BOLT, false);
180 events.RepeatEvent(3000);
181 break;
182 case EVENT_HEALTH_CHECK:
183 if (me->GetHealthPct() <= 5)
184 {
185 Talk(EMOTE_ENRAGE);
186 me->CastSpell(me, SPELL_FRENZY, true);
187 break;
188 }
189 events.RepeatEvent(1000);
190 break;
191 }
192 DoMeleeAttackIfReady();
193 }
194 };
