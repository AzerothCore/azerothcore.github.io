74struct BoneStormMoveTargetSelector
75{
76public:
77 BoneStormMoveTargetSelector(Creature* source) : _source(source) { }
78 bool operator()(Unit const* target) const
79 {
80 if (!target)
81 return false;
82
83 if (target->GetExactDist(_source) > 175.0f)
84 return false;
85
86 if (target->GetTypeId() != TYPEID_PLAYER)
87 return false;
88
89 if (target->GetPositionX() > -337.0f)
90 return false;
91
92 return target != _source->GetVictim();
93 }
94
95private:
96 Creature const* _source;

109 bool _boneSlice;

112 void Reset() override
113 {
114 me->SetReactState(REACT_AGGRESSIVE);
115 _Reset();
116 events.ScheduleEvent(EVENT_ENABLE_BONE_SLICE, 10000);
117 events.ScheduleEvent(EVENT_SPELL_BONE_SPIKE_GRAVEYARD, urand(10000, 15000));
118 events.ScheduleEvent(EVENT_SPELL_COLDFLAME, 5000);
119 events.ScheduleEvent(EVENT_WARN_BONE_STORM, urand(45000, 50000));
120 events.ScheduleEvent(EVENT_ENRAGE, 600000);
121
122 _boneSlice = false;
123
124 for (uint8 i = 0; i < 3; ++i)
125 _lastBoneSliceTargets[i].Clear();
126
127 instance->SetData(DATA_BONED_ACHIEVEMENT, uint32(true));
128 }
129
130 void EnterCombat(Unit* /*who*/) override
131 {
132 Talk(SAY_AGGRO);
133 me->setActive(true);
134 DoZoneInCombat();
135 instance->SetBossState(DATA_LORD_MARROWGAR, IN_PROGRESS);
136 }
137
138 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
139 {
140 if (target && (spell->Id == 69055 || spell->Id == 70814)) // Bone Slice (Saber Lash)
141 for (uint8 i = 0; i < 3; ++i)
142 if (!_lastBoneSliceTargets[i])
143 {
144 _lastBoneSliceTargets[i] = target->GetGUID();
145 break;
146 }
147 }
148
149 ObjectGuid GetGUID(int32 id) const override
150 {
151 if (id >= 0 && id <= 2)
152 return _lastBoneSliceTargets[id];
153
154 return ObjectGuid::Empty;
155 }
156
157 void UpdateAI(uint32 diff) override
158 {
159 if (!UpdateVictim())
160 return;
161
162 events.Update(diff);
163
164 if (me->HasUnitState(UNIT_STATE_CASTING))
165 return;
166
167 switch (events.ExecuteEvent())
168 {
169 case 0:
170 break;
171 case EVENT_ENABLE_BONE_SLICE:
172 _boneSlice = true;
173 break;
174 case EVENT_SPELL_BONE_SPIKE_GRAVEYARD:
175 {
176 bool a = me->HasAura(SPELL_BONE_STORM);
177 if (IsHeroic() || !a)
178 me->CastSpell(me, SPELL_BONE_SPIKE_GRAVEYARD, a);
179 events.RepeatEvent(urand(15000, 20000));
180 }
181 break;
182 case EVENT_SPELL_COLDFLAME:
183 if (!me->HasAura(SPELL_BONE_STORM))
184 me->CastSpell((Unit*)nullptr, SPELL_COLDFLAME_NORMAL, false);
185 events.RepeatEvent(5000);
186 break;
187 case EVENT_SPELL_COLDFLAME_BONE_STORM:
188 me->CastSpell(me, SPELL_COLDFLAME_BONE_STORM, false);
189 break;
190 case EVENT_WARN_BONE_STORM:
191 _boneSlice = false;
192 Talk(EMOTE_BONE_STORM);
193 Talk(SAY_BONE_STORM);
194 me->FinishSpell(CURRENT_MELEE_SPELL, false);
195 me->CastSpell(me, SPELL_BONE_STORM, false);
196 me->SetReactState(REACT_PASSIVE); // to prevent chasing another target on UpdateVictim()
197 me->GetMotionMaster()->MoveIdle();
198 me->GetMotionMaster()->MovementExpired();
199 events.RepeatEvent(urand(90000, 95000));
200 events.ScheduleEvent(EVENT_BEGIN_BONE_STORM, 3050);
201 break;
202 case EVENT_BEGIN_BONE_STORM:
203 {
204 uint32 _boneStormDuration = RAID_MODE<uint32>(20000, 30000, 20000, 30000);
205 if (Aura* pStorm = me->GetAura(SPELL_BONE_STORM))
206 pStorm->SetDuration(int32(_boneStormDuration));
207 events.ScheduleEvent(EVENT_BONE_STORM_MOVE, 0);
208 events.ScheduleEvent(EVENT_END_BONE_STORM, _boneStormDuration + 1);
209 }
210 break;
211 case EVENT_BONE_STORM_MOVE:
212 {
213 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == POINT_MOTION_TYPE)
214 {
215 events.RepeatEvent(1);
216 break;
217 }
218 events.RepeatEvent(5000);
219 Unit* unit = SelectTarget(SelectTargetMethod::Random, 0, BoneStormMoveTargetSelector(me));
220 if (!unit)
221 {
222 if ((unit = SelectTarget(SelectTargetMethod::MaxThreat, 0, 175.0f, true)))
223 if (unit->GetPositionX() > -337.0f)
224 {
225 EnterEvadeMode();
226 return;
227 }
228 }
229 if (unit)
230 me->GetMotionMaster()->MoveCharge(unit->GetPositionX(), unit->GetPositionY(), unit->GetPositionZ(), 25.0f, 1337);
231 break;
232 }
233 break;
234 case EVENT_END_BONE_STORM:
235 me->StopMoving();
236 me->GetMotionMaster()->MovementExpired();
237 me->SetReactState(REACT_AGGRESSIVE);
238 DoStartMovement(me->GetVictim());
239 events.CancelEvent(EVENT_BONE_STORM_MOVE);
240 events.ScheduleEvent(EVENT_ENABLE_BONE_SLICE, 10000);
241 if (!IsHeroic())
242 events.RescheduleEvent(EVENT_SPELL_BONE_SPIKE_GRAVEYARD, urand(15000, 20000));
243 break;
244 case EVENT_ENRAGE:
245 me->CastSpell(me, SPELL_BERSERK, true);
246 Talk(SAY_BERSERK);
247 break;
248 }
249
250 if (me->HasAura(SPELL_BONE_STORM))
251 return;
252
253 if (_boneSlice && !me->GetCurrentSpell(CURRENT_MELEE_SPELL))
254 DoCastVictim(SPELL_BONE_SLICE);
255
256 if (_boneSlice && me->isAttackReady() && me->GetVictim() && !me->HasUnitState(UNIT_STATE_CASTING) && me->IsWithinMeleeRange(me->GetVictim()))
257 {
258 for (uint8 i = 0; i < 3; ++i)
259 _lastBoneSliceTargets[i].Clear();
260 }
261
262 DoMeleeAttackIfReady();
263 }
264

270 events.ScheduleEvent(EVENT_SPELL_COLDFLAME_BONE_STORM, 0);

297 }
298
299 BossAI::MoveInLineOfSight(who);
