47struct boss_twilight_corrupter : public ScriptedAI
48{
49 boss_twilight_corrupter(Creature* creature) : ScriptedAI(creature) { }
50
51 void Reset() override
52 {
53 _scheduler.CancelAll();
54 me->RemoveAurasDueToSpell(SPELL_SWELL_OF_SOULS);
55 }
56
57 void InitializeAI() override
58 {
59 // Xinef: check if copy is summoned
60 std::list<Creature*> cList;
61 me->GetCreatureListWithEntryInGrid(cList, me->GetEntry(), 50.0f);
62 for (Creature* creature : cList)
63 {
64 if (creature->IsAlive() && me->GetGUID() != creature->GetGUID())
65 {
66 me->DespawnOrUnsummon(1);
67 break;
68 }
69 }
70
71 _introSpoken = false;
72 ScriptedAI::InitializeAI();
73 }
74
75 void MoveInLineOfSight(Unit* who) override
76 {
77 if (!_introSpoken && who->GetTypeId() == TYPEID_PLAYER)
78 {
79 _introSpoken = true;
80 Talk(SAY_RESPAWN, who);
81 me->SetFaction(FACTION_MONSTER);
82 }
83 ScriptedAI::MoveInLineOfSight(who);
84 }
85
86 void EnterCombat(Unit* /*who*/) override
87 {
88 Talk(SAY_AGGRO);
89 _scheduler
90 .Schedule(12s, 18s, [this](TaskContext context)
91 {
92 DoCastRandomTarget(SPELL_CREATURE_OF_NIGHTMARE, 1, 100.f);
93 context.Repeat(35s, 45s);
94 })
95 .Schedule(9s, 16s, [this](TaskContext context)
96 {
97 DoCastVictim(SPELL_SOUL_CORRUPTION);
98 context.Repeat(5s, 9s);
99 });
100 }
101
102 void KilledUnit(Unit* victim) override
103 {
104 if (victim->GetTypeId() == TYPEID_PLAYER)
105 {
106 Talk(SAY_KILL, victim);
107 DoCastSelf(SPELL_SWELL_OF_SOULS);
108 }
109 }
110
111 void UpdateAI(uint32 diff) override
112 {
113 if (!UpdateVictim())
114 return;
115
116 _scheduler.Update(diff, [this]
117 {
118 DoMeleeAttackIfReady();
119 });
120 }
121
122private:
123 bool _introSpoken;
124 TaskScheduler _scheduler;
125};
