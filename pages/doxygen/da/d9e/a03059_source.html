57struct boss_buru : public BossAI
58{
59 boss_buru(Creature* creature) : BossAI(creature, DATA_BURU) {}
60
61 void EnterEvadeMode(EvadeReason why) override
62 {
63 BossAI::EnterEvadeMode(why);
64
65 DoCastSelf(SPELL_FULL_SPEED, true);
66
67 _phase = PHASE_EGG;
68 instance->SetData(DATA_BURU_PHASE, _phase);
69
70 ManipulateEggs(true);
71 }
72
73 void ManipulateEggs(bool respawn)
74 {
75 std::list<Creature*> eggs;
76 me->GetCreaturesWithEntryInRange(eggs, 150.0f, NPC_BURU_EGG);
77 for (Creature* egg : eggs)
78 respawn ? egg->Respawn() : Unit::Kill(me, egg);
79 }
80
81 void EnterCombat(Unit* who) override
82 {
83 BossAI::EnterCombat(who);
84 me->AddThreat(who, 1000000.f);
85 Talk(EMOTE_TARGET, who);
86 DoCastSelf(SPELL_THORNS);
87 _phase = PHASE_EGG;
88 instance->SetData(DATA_BURU_PHASE, _phase);
89 ManipulateEggs(true);
90 me->RemoveAurasDueToSpell(SPELL_FULL_SPEED);
91 events.ScheduleEvent(EVENT_DISMEMBER, 5s);
92 events.ScheduleEvent(EVENT_GATHERING_SPEED, 2s);
93 }
94
95 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
96 {
97 if (spell->Id == SPELL_CREATURE_SPECIAL)
98 ChaseNewVictim();
99 }
100
101 void JustDied(Unit* killer) override
102 {
103 instance->DoRemoveAurasDueToSpellOnPlayers(SPELL_CREEPING_PLAGUE);
104 BossAI::JustDied(killer);
105 }
106
107 void KilledUnit(Unit* victim) override
108 {
109 if (victim->GetTypeId() == TYPEID_PLAYER)
110 ChaseNewVictim();
111 }
112
113 void ChaseNewVictim()
114 {
115 if (_phase != PHASE_EGG)
116 return;
117
118 me->RemoveAurasDueToSpell(SPELL_GATHERING_SPEED);
119 events.CancelEvent(EVENT_GATHERING_SPEED);
120 events.ScheduleEvent(EVENT_GATHERING_SPEED, 2s);
121 if (Unit* victim = SelectTarget(SelectTargetMethod::Random, 0, 0.f, true))
122 {
123 DoResetThreat();
124 AttackStart(victim);
125 me->AddThreat(victim, 1000000.f);
126 Talk(EMOTE_TARGET, victim);
127 }
128 }
129
130 void DamageTaken(Unit* attacker, uint32& damage, DamageEffectType, SpellSchoolMask) override
131 {
132 if (attacker && attacker->GetEntry() == NPC_BURU_EGG)
133 {
134 me->LowerPlayerDamageReq(damage);
135 }
136
137 if (me->HealthBelowPctDamaged(20, damage) && _phase == PHASE_EGG)
138 {
139 DoCastSelf(SPELL_FULL_SPEED, true);
140 ManipulateEggs(false);
141 me->RemoveAurasDueToSpell(SPELL_THORNS);
142 me->RemoveAurasDueToSpell(SPELL_GATHERING_SPEED);
143 events.Reset();
144 _phase = PHASE_TRANSFORM;
145 instance->SetData(DATA_BURU_PHASE, _phase);
146 DoResetThreat();
147 events.ScheduleEvent(EVENT_CREEPING_PLAGUE, 2s);
148 DoCastSelf(SPELL_BURU_TRANSFORM);
149 }
150 }
151
152 void UpdateAI(uint32 diff) override
153 {
154 if (!UpdateVictim())
155 return;
156
157 events.Update(diff);
158
159 while (uint32 eventId = events.ExecuteEvent())
160 {
161 switch (eventId)
162 {
163 case EVENT_DISMEMBER:
164 DoCastVictim(SPELL_DISMEMBER);
165 events.ScheduleEvent(EVENT_DISMEMBER, 5s);
166 break;
167 case EVENT_GATHERING_SPEED:
168 DoCastSelf(SPELL_GATHERING_SPEED);
169 events.ScheduleEvent(EVENT_GATHERING_SPEED, 9s);
170 break;
171 case EVENT_CREEPING_PLAGUE:
172 DoCastAOE(SPELL_CREEPING_PLAGUE);
173 events.ScheduleEvent(EVENT_CREEPING_PLAGUE, 6s);
174 break;
175 default:
176 break;
177 }
178 }
179
180 DoMeleeAttackIfReady();
181 }
182private:
183 uint8 _phase;
184};

186struct npc_buru_egg : public ScriptedAI
187{
188 npc_buru_egg(Creature* creature) : ScriptedAI(creature)
189 {
190 _instance = me->GetInstanceScript();
191 SetCombatMovement(false);
192 me->SetReactState(REACT_PASSIVE);
193 me->SetControlled(true, UNIT_STATE_STUNNED);
194 }
195
196 void EnterCombat(Unit* attacker) override
197 {
198 if (Creature* buru = _instance->GetCreature(DATA_BURU))
199 {
200 if (!buru->IsInCombat())
201 {
202 buru->AI()->AttackStart(attacker);
203 }
204 }
205 }
206
207 void JustSummoned(Creature* who) override
208 {
209 if (who->GetEntry() == NPC_HATCHLING)
210 {
211 if (Creature* buru = _instance->GetCreature(DATA_BURU))
212 {
213 if (Unit* target = buru->AI()->SelectTarget(SelectTargetMethod::Random))
214 {
215 who->AI()->AttackStart(target);
216 }
217 }
218 }
219 }
220
221 void JustDied(Unit* killer) override
222 {
223 DoCastSelf(SPELL_SUMMON_HATCHLING, true);
224 if (killer->GetEntry() != NPC_BURU)
225 {
226 if (Creature* buru = _instance->GetCreature(DATA_BURU))
227 {
228 DoCastSelf(SPELL_EXPLODE, true);
229 DoCastSelf(SPELL_BURU_EGG_TRIGGER, true);
230 buru->CastSpell(buru, SPELL_CREATURE_SPECIAL, true);
231 }
232 }
233
234 me->DespawnOrUnsummon(5000);
235 }
236
237 void UpdateAI(uint32 /*diff*/) override { }
238
239private:
240 InstanceScript* _instance;
241};

243class spell_egg_explosion : public SpellScript
244{
245 PrepareSpellScript(spell_egg_explosion);
246
247 void HandleDummyHitTarget(SpellEffIndex /*effIndex*/)
248 {
249 if (Unit* target = GetHitUnit())
250 {
251 int32 damage = 0;
252 if (target->IsPlayer())
253 damage = -16 * GetCaster()->GetDistance(target) + 500;
254 else if (target->GetEntry() == NPC_BURU && target->HasAura(SPELL_THORNS))
255 damage = target->GetMaxHealth() * 7.f / 100;
256
257 GetCaster()->CastCustomSpell(target, SPELL_EXPLODE_2, &damage, nullptr, nullptr, true);
258 }
259 }
260
261 void Register() override
262 {
263 OnEffectHitTarget += SpellEffectFn(spell_egg_explosion::HandleDummyHitTarget, EFFECT_0, SPELL_EFFECT_DUMMY);
264 }
