305 for (Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr)
306 if (Player* plr = itr->GetSource())

309 for (uint8 i = EQUIPMENT_SLOT_START; i < EQUIPMENT_SLOT_END; ++i) // loop through equipped items
