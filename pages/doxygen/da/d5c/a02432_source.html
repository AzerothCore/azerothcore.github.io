58 void UpdateAI(uint32 diff) override
59 {
60 //Return since we have no target
61 if (!UpdateVictim())
62 {
63 return;
64 }
65 events.Update(diff);
66
67 if (me->HasUnitState(UNIT_STATE_CASTING))
68 {
69 return;
70 }
71
72 while (uint32 eventId = events.ExecuteEvent())
73 {
74 switch (eventId)
75 {
76 case SPELL_GROUNDTREMOR:
77 if (me->GetDistance2d(me->GetVictim()) < 10.0f)
78 {
79 DoCastVictim(SPELL_GROUNDTREMOR);
80 nextTremorTime = urand(TIMER_GROUNDTREMOR - 2000, TIMER_GROUNDTREMOR + 2000);
81 }
82 else
83 {
84 nextTremorTime = 0.3*urand(TIMER_GROUNDTREMOR - 2000, TIMER_GROUNDTREMOR + 2000);
85 }
86 events.ScheduleEvent(SPELL_GROUNDTREMOR, nextTremorTime);
87 break;
88 case SPELL_FRENZY:
89 DoCastSelf(SPELL_FRENZY);
90 events.ScheduleEvent(SPELL_FRENZY, urand(TIMER_FRENZY - 2000, TIMER_FRENZY + 2000));
91 Talk(EMOTE_FRENZY_KILL);
92 break;
93 default:
94 break;
95 }
96 }
97 DoMeleeAttackIfReady();
98 }
99 };
