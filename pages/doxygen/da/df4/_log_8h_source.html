130#define LOG_EXCEPTION_FREE(filterType__, level__, ...) \
131 { \
132 try \
133 { \
134 sLog->outMessage(filterType__, level__, fmt::format(__VA_ARGS__)); \
135 } \
136 catch (std::exception const& e) \
137 { \
138 sLog->outMessage("server", LogLevel::LOG_LEVEL_ERROR, "Wrong format occurred ({}) at '{}:{}'", \
139 e.what(), __FILE__, __LINE__); \
140 } \
141 }

143#ifdef PERFORMANCE_PROFILING
144#define LOG_MESSAGE_BODY(filterType__, level__, ...) ((void)0)
145#else
146#define LOG_MESSAGE_BODY(filterType__, level__, ...) \
147 do \
148 { \
149 if (sLog->ShouldLog(filterType__, level__)) \
150 LOG_EXCEPTION_FREE(filterType__, level__, __VA_ARGS__); \
151 } while (0)
152#endif
