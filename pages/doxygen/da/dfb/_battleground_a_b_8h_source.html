257struct BattlegroundABScore final : public BattlegroundScore
258{
259 friend class BattlegroundAB;
260
261protected:
262
263 explicit BattlegroundABScore(ObjectGuid playerGuid) : BattlegroundScore(playerGuid) { }
264
265 void UpdateScore(uint32 type, uint32 value) override
266 {
267 switch (type)
268 {
269 case SCORE_BASES_ASSAULTED:
270 BasesAssaulted += value;
271 break;
272 case SCORE_BASES_DEFENDED:
273 BasesDefended += value;
274 break;
275 default:
276 BattlegroundScore::UpdateScore(type, value);
277 break;
278 }
279 }

281 void BuildObjectivesBlock(WorldPacket& data) final;
282
283 uint32 GetAttr1() const override { return BasesAssaulted; }
284 uint32 GetAttr2() const override { return BasesDefended; }

287 uint32 BasesDefended = 0;
288};
