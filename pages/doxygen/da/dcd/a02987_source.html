54 void Initialize() override
55 {
56 _encounterProgress = 0;
57 _barrelCount = 0;
58 _attemptsCount = 0;
59
60 _initalFlamesSet.clear();
61 _finalFlamesSet.clear();

66 void OnPlayerEnter(Player* player) override
67 {
68 if (instance->GetPlayersCountExceptGMs() == 1)
69 CleanupInstance();
70
71 EnsureGridLoaded();
72
73 if (_encounterProgress < ENCOUNTER_PROGRESS_BARRELS)
74 player->SendUpdateWorldState(WORLD_STATE_BARRELS_PLANTED, _barrelCount);
75 }
76
77 void CleanupInstance()
78 {
79 if (_encounterProgress == ENCOUNTER_PROGRESS_NONE)
80 return;
81
82 _events.ScheduleEvent(EVENT_INITIAL_BARRELS_FLAME, 0);
83 _events.ScheduleEvent(EVENT_FINAL_BARRELS_FLAME, 0);
84
85 if (_encounterProgress == ENCOUNTER_PROGRESS_BARRELS)
86 _events.ScheduleEvent(EVENT_SUMMON_LIEUTENANT, 0);
87 else
88 SetData(DATA_THRALL_REPOSITION, 2);
89 }
90
91 void OnCreatureCreate(Creature* creature) override
92 {
93 switch (creature->GetEntry())
94 {
95 case NPC_THRALL:
96 _thrallGUID = creature->GetGUID();
97 if (_encounterProgress == ENCOUNTER_PROGRESS_FINISHED)
98 creature->SetVisible(false);
99 else
100 Reposition(creature);
101 break;
102 case NPC_ORC_PRISONER:
103 _prisonersSet.insert(creature->GetGUID());
104 break;
105 case NPC_TARETHA:
106 if (_encounterProgress == ENCOUNTER_PROGRESS_FINISHED)
107 creature->SetVisible(false);
108 _tarethaGUID = creature->GetGUID();
109 break;
110 }
111 }
112
113 void OnGameObjectCreate(GameObject* gameobject) override
114 {
115 switch (gameobject->GetEntry())
116 {
117 case GO_BARREL:
118 if (_encounterProgress >= ENCOUNTER_PROGRESS_BARRELS)
119 gameobject->SetGameObjectFlag(GO_FLAG_NOT_SELECTABLE);
120 break;
121 case GO_PRISON_DOOR:
122 if (_encounterProgress >= ENCOUNTER_PROGRESS_THRALL_ARMORED)
123 gameobject->SetGoState(GO_STATE_ACTIVE);
124 break;
125 case GO_ROARING_FLAME:
126 // Xinef: hack in DB to distinguish final / initial flames
127 if (gameobject->GetPhaseMask() & 0x2)
128 _finalFlamesSet.insert(gameobject->GetGUID());
129 else
130 _initalFlamesSet.insert(gameobject->GetGUID());
131 break;
132 }
133 }
134
135 void SetData(uint32 type, uint32 data) override
136 {
137 switch (type)
138 {
139 case DATA_THRALL_REPOSITION:
140 if (data > 1)
141 _events.ScheduleEvent(EVENT_THRALL_REPOSITION, data == 2 ? 0 : 10000);
142 else if (Creature* thrall = instance->GetCreature(_thrallGUID))
143 Reposition(thrall);
144 return;
145 case DATA_ESCORT_PROGRESS:
146 _encounterProgress = data;
147 SaveToDB();
148 break;
149 case DATA_BOMBS_PLACED:
150 {
151 if (_barrelCount >= 5 || _encounterProgress > ENCOUNTER_PROGRESS_NONE)
152 return;
153
154 DoUpdateWorldState(WORLD_STATE_BARRELS_PLANTED, ++_barrelCount);
155 if (_barrelCount == 5)
156 {
157 _events.ScheduleEvent(EVENT_INITIAL_BARRELS_FLAME, 4000);
158 _events.ScheduleEvent(EVENT_FINAL_BARRELS_FLAME, 12000);
159 _events.ScheduleEvent(EVENT_SUMMON_LIEUTENANT, 18000);
160 }
161 break;
162 }
163 case DATA_THRALL_ADD_FLAG:
164 if (Creature* thrall = instance->GetCreature(_thrallGUID))
165 thrall->SetNpcFlag(UNIT_NPC_FLAG_GOSSIP);
166 break;
167 }
168 }

170 uint32 GetData(uint32 data) const override

176 return 0;

189 void Update(uint32 diff) override
190 {
191 _events.Update(diff);
192 switch (_events.ExecuteEvent())
193 {
194 case EVENT_INITIAL_BARRELS_FLAME:
195 {
196 instance->LoadGrid(instancePositions[0].GetPositionX(), instancePositions[0].GetPositionY());
197 instance->LoadGrid(instancePositions[1].GetPositionX(), instancePositions[1].GetPositionY());
198
199 for (ObjectGuid const& guid : _prisonersSet)
200 if (Creature* orc = instance->GetCreature(guid))
201 {
202 uint8 index = orc->GetDistance(instancePositions[0]) < 80.0f ? 0 : 1;
203 Position pos(instancePositions[index]);
204 orc->MovePosition(pos, frand(1.0f, 3.0f) + 15.0f * (float)rand_norm(), (float)rand_norm() * static_cast<float>(2 * M_PI));
205 orc->GetMotionMaster()->MovePoint(1, pos);
206 orc->SetStandState(UNIT_STAND_STATE_STAND);
207 }
208
209 for (ObjectGuid const& guid : _initalFlamesSet)
210 if (GameObject* gobject = instance->GetGameObject(guid))
211 {
212 gobject->SetRespawnTime(0);
213 gobject->UpdateObjectVisibility(true);
214 }
215 break;
216 }
217 case EVENT_FINAL_BARRELS_FLAME:
218 {
219 instance->LoadGrid(instancePositions[0].GetPositionX(), instancePositions[0].GetPositionY());
220 instance->LoadGrid(instancePositions[1].GetPositionX(), instancePositions[1].GetPositionY());
221
222 if (_encounterProgress == ENCOUNTER_PROGRESS_NONE)
223 {
224 Map::PlayerList const& players = instance->GetPlayers();
225 if (!players.IsEmpty())
226 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
227 if (Player* player = itr->GetSource())
228 player->KilledMonsterCredit(NPC_LODGE_QUEST_TRIGGER);
229 }
230
231 for (ObjectGuid const& guid : _finalFlamesSet)
232 if (GameObject* gobject = instance->GetGameObject(guid))
233 {
234 gobject->SetRespawnTime(0);
235 gobject->UpdateObjectVisibility(true);
236 }
237
238 for (ObjectGuid const& guid : _prisonersSet)
239 if (Creature* orc = instance->GetCreature(guid))
240 if (roll_chance_i(25))
241 orc->HandleEmoteCommand(EMOTE_ONESHOT_CHEER);
242
243 SetData(DATA_ESCORT_PROGRESS, ENCOUNTER_PROGRESS_BARRELS);
244 DoUpdateWorldState(WORLD_STATE_BARRELS_PLANTED, 0);
245 break;
246 }
247 case EVENT_SUMMON_LIEUTENANT:
248 {
249 instance->LoadGrid(instancePositions[2].GetPositionX(), instancePositions[2].GetPositionY());
250 if (Creature* drake = instance->SummonCreature(NPC_LIEUTENANT_DRAKE, instancePositions[2]))
251 {
252 drake->AI()->Talk(0);
253 }
254 [[fallthrough]]; // TODO: Not sure whether the fallthrough was a mistake (forgetting a break) or intended. This should be double-checked.
255 }
256 case EVENT_THRALL_REPOSITION:
257 {
258 if (Creature* thrall = instance->GetCreature(_thrallGUID))
259 {
260 if (!thrall->IsAlive())
261 {
262 ++_attemptsCount;
263 EnsureGridLoaded();
264 thrall->SetVisible(false);
265 Reposition(thrall);
266 thrall->setDeathState(DEAD);
267 thrall->Respawn();
268 thrall->SetVisible(true);
269 SaveToDB();
270 }
271 else
272 thrall->AI()->Reset();
273 }
274 break;
275 }
276 }
277 }

279 void Reposition(Creature* thrall)
280 {
281 switch (uint32 data = GetData(DATA_ESCORT_PROGRESS))
282 {
283 case ENCOUNTER_PROGRESS_THRALL_ARMORED:
284 case ENCOUNTER_PROGRESS_AMBUSHES_1:
285 case ENCOUNTER_PROGRESS_SKARLOC_KILLED:
286 case ENCOUNTER_PROGRESS_TARREN_MILL:
287 case ENCOUNTER_PROGRESS_TARETHA_MEET:
288 thrall->UpdatePosition(thrallPositions[data - ENCOUNTER_PROGRESS_THRALL_ARMORED], true);
289 thrall->SetHomePosition(thrallPositions[data - ENCOUNTER_PROGRESS_THRALL_ARMORED]);
290 thrall->SetFacingTo(thrallPositions[data - ENCOUNTER_PROGRESS_THRALL_ARMORED].GetOrientation());
291 break;
292 }
293 }

299 }

312 void Load(const char* in) override
313 {
314 if (!in)
315 {
316 OUT_LOAD_INST_DATA_FAIL;
317 return;
318 }
319
320 OUT_LOAD_INST_DATA(in);
321
322 char dataHead1, dataHead2;
323 uint32 data0, data1;
324
325 std::istringstream loadStream(in);
326 loadStream >> dataHead1 >> dataHead2 >> data0 >> data1;
327
328 if (dataHead1 == 'O' && dataHead2 == 'H')
329 {
330 _encounterProgress = data0;
331 _attemptsCount = data1;
332 }
333 else
334 OUT_LOAD_INST_DATA_FAIL;
335
336 OUT_LOAD_INST_DATA_COMPLETE;
337 }
