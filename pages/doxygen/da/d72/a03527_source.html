124 {
125 if (pInstance)

127 pInstance->SetData(DATA_UROM, NOT_STARTED);

144 void EnterCombat(Unit* /*who*/) override
145 {
146 if( lock )
147 return;
148
149 uint8 phase = GetPhaseByCurrentPosition();
150 if( phase == 3 )
151 {
152 Talk(SAY_AGGRO);
153
154 if( pInstance )
155 pInstance->SetData(DATA_UROM, IN_PROGRESS);
156
157 me->SetInCombatWithZone();
158 me->SetHomePosition(cords[0][0], cords[0][1], cords[0][2], cords[0][3]);
159 if( me->FindCurrentSpellBySpellId(SPELL_EVOCATION) )
160 me->InterruptNonMeleeSpells(false);
161
162 events.RescheduleEvent(EVENT_FROSTBOMB, urand(7000, 11000));
163 events.RescheduleEvent(EVENT_TELEPORT_TO_CENTER, urand(30000, 35000));
164 events.RescheduleEvent(EVENT_TIME_BOMB, urand(20000, 25000));
165 }
166 else
167 {
168 lock = true;
169
170 switch( phase )
171 {
172 case 0:
173 Talk(SAY_SUMMON_1);
174 me->InterruptNonMeleeSpells(false);
175 me->CastSpell(me, SPELL_SUMMON_MENAGERIE_1, false);
176 break;
177 case 1:
178 Talk(SAY_SUMMON_2);
179 me->InterruptNonMeleeSpells(false);
180 me->CastSpell(me, SPELL_SUMMON_MENAGERIE_2, false);
181 break;
182 case 2:
183 Talk(SAY_SUMMON_3);
184 me->InterruptNonMeleeSpells(false);
185 me->CastSpell(me, SPELL_SUMMON_MENAGERIE_3, false);
186 break;
187 }
188 }

300 void UpdateAI(uint32 diff) override
301 {
302 if( releaseLockTimer )
303 {
304 if( releaseLockTimer >= 5000 )
305 {
306 lock = false;
307 if (me->IsInCombat())
308 {
309 LeaveCombat();
310 me->CastSpell(me, SPELL_EVOCATION, true);
311 }
312 releaseLockTimer = 0;
313 }
314 else
315 releaseLockTimer += diff;
316 }
317
318 if( !UpdateVictim() )
319 return;
320
321 events.Update(diff);
322
323 if( me->HasUnitState(UNIT_STATE_CASTING) )
324 return;
325
326 DoMeleeAttackIfReady();
327
328 switch( events.ExecuteEvent() )
329 {
330 case 0:
331 break;
332 case EVENT_FROSTBOMB:
333 if( Unit* v = me->GetVictim() )
334 me->CastSpell(v, SPELL_FROSTBOMB, false);
335 events.RepeatEvent(urand(7000, 11000));
336 break;
337 case EVENT_TIME_BOMB:
338 if( Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100.0f, true) )
339 DoCast(target, DUNGEON_MODE(SPELL_TIME_BOMB_N, SPELL_TIME_BOMB_H));
340 events.RepeatEvent(urand(20000, 25000));
341 break;
342 case EVENT_TELEPORT_TO_CENTER:
343 x = me->GetPositionX();
344 y = me->GetPositionY();
345 z = me->GetPositionZ();
346 me->CastSpell(me, SPELL_TELEPORT, false);
347 events.RepeatEvent(urand(25000, 30000));
348 events.DelayEvents(10000);
349 break;
350 case EVENT_TELE_BACK:
351 me->GetMotionMaster()->MoveIdle();
352 me->DisableSpline();
353 me->SetCanFly(false);
354 me->SetDisableGravity(false);
355 me->NearTeleportTo(x, y, z, 0.0f);
356 me->SetControlled(false, UNIT_STATE_ROOT);
357 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
358 me->GetMotionMaster()->MoveChase(me->GetVictim());
359 break;
360 }
361 }
