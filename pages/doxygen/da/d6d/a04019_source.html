115 void UpdateAI(uint32 diff) override
116 {
117 events2.Update(diff);
118 switch (events2.ExecuteEvent())
119 {
120 case EVENT_SOCCOTHRATES_DEATH:
121 Talk(SAY_SOCCOTHRATES_DEATH);
122 break;
123 }
124
125 if (!UpdateVictim())
126 return;
127
128 events.Update(diff);
129 if (me->HasUnitState(UNIT_STATE_CASTING))
130 return;
131
132 switch (events.ExecuteEvent())
133 {
134 case EVENT_GIFT_OF_THE_DOOMSAYER:
135 me->CastSpell(me->GetVictim(), SPELL_GIFT_OF_THE_DOOMSAYER, false);
136 events.ScheduleEvent(EVENT_GIFT_OF_THE_DOOMSAYER, urand(16000, 21000));
137 break;
138 case EVENT_WHIRLWIND:
139 me->CastSpell(me, SPELL_WHIRLWIND, false);
140 Talk(SAY_WHIRLWIND);
141 events.ScheduleEvent(EVENT_WHIRLWIND, urand(19000, 21000));
142 events.ScheduleEvent(EVENT_HEAL, 6000);
143 break;
144 case EVENT_HEAL:
145 me->CastSpell(me, SPELL_HEAL, false);
146 Talk(SAY_HEAL);
147 break;
148 case EVENT_SHADOW_WAVE:
149 me->CastSpell(me->GetVictim(), SPELL_SHADOW_WAVE, false);
150 events.ScheduleEvent(EVENT_SHADOW_WAVE, urand(11000, 16000));
151 break;
152 case EVENT_ME_FIRST:
153 if (Creature* soccothrates = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SOCCOTHRATES)))
154 if (soccothrates->IsAlive() && !soccothrates->IsInCombat())
155 soccothrates->AI()->Talk(SAY_AGGRO_DALLIAH_FIRST);
156 break;
157 case EVENT_CHECK_HEALTH:
158 if (HealthBelowPct(25))
159 {
160 if (Creature* soccothrates = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_SOCCOTHRATES)))
161 soccothrates->AI()->Talk(SAY_DALLIAH_25_PERCENT);
162 break;
163 }
164 events.ScheduleEvent(EVENT_CHECK_HEALTH, 1000);
165 break;
166 }
167
168 DoMeleeAttackIfReady();
169 }
170
171 private:
172 EventMap events2;
173 };
