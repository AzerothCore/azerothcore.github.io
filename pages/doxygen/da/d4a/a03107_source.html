147 void DamageTaken(Unit* attacker, uint32& damage, DamageEffectType effType, SpellSchoolMask spellSchoolMask) override
148 {
149 if (me->HealthBelowPct(5))
150 damage = 0;
151
152 if (!attacker)
153 {
154 return;
155 }
156
157 if (_phase != PHASE_MELEE)
158 {
159 if (_phase == PHASE_FROST && effType == DIRECT_DAMAGE && (spellSchoolMask & SPELL_SCHOOL_MASK_FROST) != 0)
160 {
161 ++_hitcounter;
162 }
163
164 return;
165 }
166
167 if (effType == DIRECT_DAMAGE)
168 ++_hitcounter;
169
170 if (attacker->HasUnitState(UNIT_STATE_MELEE_ATTACKING) && _hitcounter >= HITCOUNTER_EXPLODE)
171 {
172 if (me->GetHealthPct() <= 5.f)
173 {
174 Unit::Kill(attacker, me);
175 return;
176 }
177
178 Talk(EMOTE_EXPLODE);
179 me->SetReactState(REACT_PASSIVE);
180 events.Reset();
181 _phase = PHASE_GLOB;
182 me->RemoveAura(SPELL_VISCIDUS_FREEZE);
183 DoCastSelf(SPELL_STUN_SELF, true);
184 me->AttackStop();
185 me->CastStop();
186 me->HandleEmoteCommand(EMOTE_ONESHOT_FLYDEATH); // not found in sniff, this is the best one I found
187 _scheduler
188 .Schedule(2500ms, [this](TaskContext /*context*/)
189 {
190 DoCastSelf(SPELL_EXPLODE_TRIGGER, true);
191 })
192 .Schedule(3000ms, [this](TaskContext /*context*/)
193 {
194 DoCastSelf(SPELL_INVIS_SELF, true);
195 me->SetAuraStack(SPELL_VISCIDUS_SHRINKS, me, 20);
196 me->LowerPlayerDamageReq(me->GetMaxHealth());
197 me->SetHealth(me->GetMaxHealth() * 0.01f); // set 1% health
198 DoResetThreat();
199 me->NearTeleportTo(roomCenter.GetPositionX(),
200 roomCenter.GetPositionY(),
201 roomCenter.GetPositionZ(),
202 roomCenter.GetOrientation());
203 });
204 }
205 else if (_hitcounter == HITCOUNTER_SHATTER)
206 Talk(EMOTE_SHATTER);
207 else if (_hitcounter == HITCOUNTER_CRACK)
208 Talk(EMOTE_CRACK);
209 }

211 void SpellHit(Unit* caster, SpellInfo const* spellInfo) override
212 {
213 if (spellInfo->Id == SPELL_REJOIN_VISCIDUS)
214 {
215 me->RemoveAuraFromStack(SPELL_VISCIDUS_SHRINKS);
216 }
217
218 SpellSchoolMask spellSchoolMask = spellInfo->GetSchoolMask();
219 if (spellInfo->EquippedItemClass == ITEM_CLASS_WEAPON && spellInfo->EquippedItemSubClassMask & (1 << ITEM_SUBCLASS_WEAPON_WAND))
220 {
221 if (Item* pItem = caster->ToPlayer()->GetWeaponForAttack(RANGED_ATTACK))
222 {
223 spellSchoolMask = SpellSchoolMask(1 << pItem->GetTemplate()->Damage[0].DamageType);
224 }
225 }
226
227 if ((spellSchoolMask & SPELL_SCHOOL_MASK_FROST) && _phase == PHASE_FROST)
228 {
229 ++_hitcounter;
230
231 if (_hitcounter >= HITCOUNTER_FREEZE)
232 {
233 _hitcounter = 0;
234 Talk(EMOTE_FROZEN);
235 _phase = PHASE_MELEE;
236 me->RemoveAura(SPELL_VISCIDUS_SLOWED_MORE);
237 DoCastSelf(SPELL_VISCIDUS_FREEZE);
238 events.ScheduleEvent(EVENT_RESET_PHASE, 15s);
239 }
240 else if (_hitcounter == HITCOUNTER_SLOW_MORE)
241 {
242 Talk(EMOTE_FREEZE);
243 me->RemoveAura(SPELL_VISCIDUS_SLOWED);
244 DoCastSelf(SPELL_VISCIDUS_SLOWED_MORE);
245 }
246 else if (_hitcounter == HITCOUNTER_SLOW)
247 {
248 Talk(EMOTE_SLOW);
249 DoCastSelf(SPELL_VISCIDUS_SLOWED);
250 }
251 }
252 }

254 void SummonedCreatureDies(Creature* summon, Unit* killer) override
255 {
256 if (summon->GetEntry() != NPC_GLOB_OF_VISCIDUS)
257 return;
258
259 if (killer && killer->GetEntry() == NPC_GLOB_OF_VISCIDUS)
260 {
261 if (_phase == PHASE_GLOB)
262 {
263 _phase = PHASE_FROST;
264 me->RemoveAurasDueToSpell(SPELL_INVIS_SELF);
265 }
266
267 int32 heal = me->GetMaxHealth() * 0.05f;
268 me->CastCustomSpell(me, SPELL_VISCIDUS_GROWS, &heal, nullptr, nullptr, true);
269 }
270
271 if (!summons.IsAnyCreatureWithEntryAlive(NPC_GLOB_OF_VISCIDUS)) // all globs were killed
272 {
273 SoftReset();
274 InitSpells();
275 me->LowerPlayerDamageReq(me->GetMaxHealth());
276 }
277 }

285 void InitSpells()
286 {
287 events.ScheduleEvent(EVENT_TOXIN, 15s, 20s);
288 events.ScheduleEvent(EVENT_POISONBOLT_VOLLEY, 10s, 15s);
289 events.ScheduleEvent(EVENT_POISON_SHOCK, 7s, 12s);
290 }
291
292 void UpdateAI(uint32 diff) override
293 {
294 if (!UpdateVictim() || !CheckInRoom())
295 return;
296
297 events.Update(diff);
298 _scheduler.Update(diff);
299
300 while (uint32 eventId = events.ExecuteEvent())
301 {
302 switch (eventId)
303 {
304 case EVENT_POISONBOLT_VOLLEY:
305 DoCastSelf(SPELL_POISONBOLT_VOLLEY);
306 events.ScheduleEvent(EVENT_POISONBOLT_VOLLEY, 10s, 15s);
307 break;
308 case EVENT_POISON_SHOCK:
309 DoCastSelf(SPELL_POISON_SHOCK);
310 events.ScheduleEvent(EVENT_POISON_SHOCK, 7s, 12s);
311 break;
312 case EVENT_TOXIN:
313 DoCastRandomTarget(SPELL_SUMMON_TOXIN_SLIME);
314 events.ScheduleEvent(EVENT_TOXIN, 15s, 20s);
315 break;
316 case EVENT_RESET_PHASE:
317 _hitcounter = 0;
318 me->RemoveAura(SPELL_VISCIDUS_FREEZE);
319 _phase = PHASE_FROST;
320 break;
321 default:
322 break;
323 }
324 }
325
326 if (_phase != PHASE_GLOB && me->GetReactState() == REACT_AGGRESSIVE)
327 DoMeleeAttackIfReady();
328 }
329
330private:
331 uint8 _hitcounter;
332 uint8 _phase;
333 TaskScheduler _scheduler;
334};
