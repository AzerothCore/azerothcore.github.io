75void DynamicObject::RemoveFromWorld()
76{
78 if (IsInWorld())
79 {
80 if (_isViewpoint)
81 RemoveCasterViewpoint();
82
83 if (_aura)
84 RemoveAura();
85
86 // dynobj could get removed in Aura::RemoveAura
87 if (!IsInWorld())
88 return;
89
90 UnbindFromCaster();
91
92 if (Transport* transport = GetTransport())
93 transport->RemovePassenger(this, true);
94
95 WorldObject::RemoveFromWorld();
96
97 GetMap()->GetObjectsStore().Remove<DynamicObject>(GetGUID());
98 }

101bool DynamicObject::CreateDynamicObject(ObjectGuid::LowType guidlow, Unit* caster, uint32 spellId, Position const& pos, float radius, DynamicObjectType type)
102{
103 SetMap(caster->GetMap());
104 Relocate(pos);
105 if (!IsPositionValid())
106 {
107 LOG_ERROR("dyobject", "DynamicObject (spell {}) not created. Suggested coordinates isn't valid (X: {} Y: {})", spellId, GetPositionX(), GetPositionY());
108 return false;
109 }
110
111 WorldObject::_Create(guidlow, HighGuid::DynamicObject, caster->GetPhaseMask());
112
113 UpdatePositionData();
114
115 SetEntry(spellId);
116 SetObjectScale(1);
117 SetGuidValue(DYNAMICOBJECT_CASTER, caster->GetGUID());
118
119 // The lower word of DYNAMICOBJECT_BYTES must be 0x0001. This value means that the visual radius will be overriden
120 // by client for most of the "ground patch" visual effect spells and a few "skyfall" ones like Hurricane.
121 // If any other value is used, the client will _always_ use the radius provided in DYNAMICOBJECT_RADIUS, but
122 // precompensation is necessary (eg radius *= 2) for many spells. Anyway, blizz sends 0x0001 for all the spells
123 // I saw sniffed...
124 SetByteValue(DYNAMICOBJECT_BYTES, 0, type);
125 SetUInt32Value(DYNAMICOBJECT_SPELLID, spellId);
126 SetFloatValue(DYNAMICOBJECT_RADIUS, radius);
127 SetUInt32Value(DYNAMICOBJECT_CASTTIME, GameTime::GetGameTimeMS().count());
128
129 if (!GetMap()->AddToMap(this, true))
130 {
131 // Returning false will cause the object to be deleted - remove from transport
132 return false;
133 }
134
135 if (IsWorldObject())
136 {
137 setActive(true);
138 }
139
140 return true;
141}

143void DynamicObject::Update(uint32 p_time)
144{
145 // caster has to be always available and in the same map
146 ASSERT(_caster);
147 ASSERT(_caster->GetMap() == GetMap());
148
149 bool expired = false;
150
151 if (_aura)
152 {
153 if (!_aura->IsRemoved())
154 _aura->UpdateOwner(p_time, this);
155
156 // _aura may be set to null in Aura::UpdateOwner call
157 if (_aura && (_aura->IsRemoved() || _aura->IsExpired()))
158 expired = true;
159 }
160 else
161 {
162 if (GetDuration() > int32(p_time))
163 _duration -= p_time;
164 else
165 expired = true;
166 }
167
168 if (expired)
169 Remove();
170 else
171 {
172 if (_updateViewerVisibilityTimer)
173 {
174 if (_updateViewerVisibilityTimer <= p_time)
175 {
176 _updateViewerVisibilityTimer = 0;
177
178 if (Player* playerCaster = _caster->ToPlayer())
179 playerCaster->UpdateVisibilityForPlayer();
180 }
181 else
182 _updateViewerVisibilityTimer -= p_time;
183 }
184
185 sScriptMgr->OnDynamicObjectUpdate(this, p_time);
186 }
187}

235void DynamicObject::SetCasterViewpoint(bool updateViewerVisibility)
236{
237 if (Player* caster = _caster->ToPlayer())
238 {
239 // Remove old farsight viewpoint
240 if (Unit* farsightObject = ObjectAccessor::GetUnit(*caster, caster->GetGuidValue(PLAYER_FARSIGHT)))
241 {
242 _oldFarsightGUID = caster->GetGuidValue(PLAYER_FARSIGHT);
243 caster->SetViewpoint(farsightObject, false);
244 }
245
246 caster->SetViewpoint(this, true);
247 _isViewpoint = true;
248 }
249
250 _updateViewerVisibilityTimer = updateViewerVisibility ? 100 : 0;
251}
252
253void DynamicObject::RemoveCasterViewpoint()
254{
255 if (Player* caster = _caster->ToPlayer())
256 {
257 caster->SetViewpoint(this, false);
258 _isViewpoint = false;
259
260 // Restore prev farsight viewpoint
261 if (Unit* farsightObject = ObjectAccessor::GetUnit(*caster, _oldFarsightGUID))
262 {
263 caster->SetViewpoint(farsightObject, true);
264 }
265 _oldFarsightGUID.Clear();
266 }
267}
