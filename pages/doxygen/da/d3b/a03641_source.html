334 void Load(const char* in) override
335 {
336 if (!in)
337 {
338 OUT_LOAD_INST_DATA_FAIL;
339 return;
340 }
341
342 OUT_LOAD_INST_DATA(in);
343
344 char dataHead1, dataHead2;
345 uint32 data0, data1, data2, data3;
346
347 std::istringstream loadStream(in);
348 loadStream >> dataHead1 >> dataHead2 >> data0 >> data1 >> data2 >> data3;
349
350 if (dataHead1 == 'U' && dataHead2 == 'K')
351 {
352 m_auiEncounter[0] = data0;
353 m_auiEncounter[1] = data1;
354 m_auiEncounter[2] = data2;
355 ForgeEventMask = data3;
356
357 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
358 if (m_auiEncounter[i] == IN_PROGRESS)
359 m_auiEncounter[i] = NOT_STARTED;
360 }
361 else OUT_LOAD_INST_DATA_FAIL;
362
363 OUT_LOAD_INST_DATA_COMPLETE;
364 }
