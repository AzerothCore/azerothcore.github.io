111 void UpdateAI(uint32 diff) override
112 {
113 events.Update(diff);
114 uint32 eventId = events.ExecuteEvent();
115
116 if (eventId == EVENT_SPELL_INHIBIT_MAGIC)
117 {
118 Map::PlayerList const& PlayerList = me->GetMap()->GetPlayers();
119 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
120 if (Player* player = i->GetSource())
121 {
122 float dist = me->GetDistance(player);
123 if (player->IsAlive() && dist < 45.0f)
124 {
125 Aura* aura = player->GetAura(SPELL_INHIBIT_MAGIC);
126 if (!aura)
127 aura = me->AddAura(SPELL_INHIBIT_MAGIC, player);
128 else
129 aura->RefreshDuration();
130
131 if (aura)
132 aura->SetStackAmount(getStackCount(dist));
133 }
134 else
135 player->RemoveAurasDueToSpell(SPELL_INHIBIT_MAGIC);
136 }
137 events.RepeatEvent(3000);
138 return;
139 }
140
141 if (!UpdateVictim())
142 return;
143
144 switch (eventId)
145 {
146 case EVENT_SPELL_ATTRACT_MAGIC:
147 me->CastSpell(me, SPELL_ATTRACT_MAGIC, false);
148 events.RepeatEvent(30000);
149 events.RescheduleEvent(EVENT_SPELL_CARNIVOROUS, 1500);
150 break;
151 case EVENT_SPELL_CARNIVOROUS:
152 me->CastSpell(me, DUNGEON_MODE(SPELL_CARNIVOROUS_BITE_N, SPELL_CARNIVOROUS_BITE_H), false);
153 events.RepeatEvent(10000);
154 break;
155 case EVENT_SPELL_FOCUS_FIRE:
156 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 60.0f, true))
157 {
158 if (Creature* cr = me->SummonCreature(ENTRY_FOCUS_FIRE, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN, 7000))
159 focusGUID = cr->GetGUID();
160 Talk(EMOTE_FOCUSED, target);
161 }
162 events.RepeatEvent(urand(15000, 20000));
163 events.ScheduleEvent(EVENT_SPELL_FOCUS_FIRE_2, 3000);
164 events.ScheduleEvent(EVENT_SPELL_FOCUS_FIRE_2, 3500);
165 events.ScheduleEvent(EVENT_SPELL_FOCUS_FIRE_2, 4000);
166 events.ScheduleEvent(EVENT_SPELL_FOCUS_FIRE_3, 5000);
167 me->SetControlled(true, UNIT_STATE_ROOT);
168 break;
169 case EVENT_SPELL_FOCUS_FIRE_2:
170 if (Unit* flare = ObjectAccessor::GetCreature(*me, focusGUID))
171 me->CastSpell(flare, SPELL_FOCUS_CAST, true);
172 break;
173 case EVENT_SPELL_FOCUS_FIRE_3:
174 me->SetControlled(false, UNIT_STATE_ROOT);
175 break;
176 }
177
178 DoMeleeAttackIfReady();
179 }
180 };
