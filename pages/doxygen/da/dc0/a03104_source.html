103 void DamageTaken(Unit* attacker, uint32& damage, DamageEffectType, SpellSchoolMask) override
104 {
105 if (attacker)
106 {
107 if (attacker->GetEntry() == NPC_VEKLOR || attacker->GetEntry() == NPC_VEKNILASH)
108 {
109 me->LowerPlayerDamageReq(damage);
110 return;
111 }
112
113 if (Creature* twin = GetTwin())
114 {
115 float dmgPct = damage / (float)me->GetMaxHealth();
116 int32 actualDmg = dmgPct * twin->GetMaxHealth();

118 }

182 me->SetStandState(UNIT_STAND_STATE_STAND);

240 void EnterCombat(Unit* who) override
241 {
242 BossAI::EnterCombat(who);
243
244 if (!_introDone)
245 {
246 DoAction(ACTION_CANCEL_INTRO);
247 if (Creature* twin = GetTwin())
248 twin->AI()->DoAction(ACTION_CANCEL_INTRO);
249 }
250
251 if (Creature* twin = GetTwin())
252 if (!twin->IsInCombat())
253 twin->AI()->AttackStart(who);
254
255 _scheduler
256 .Schedule(15min, [this](TaskContext /*context*/)
257 {
258 if (IAmVeklor())
259 {
260 DoCastSelf(SPELL_FRENZY, true);
261 Talk(EMOTE_ENRAGE);
262 }
263 else
264 DoCastSelf(SPELL_BERSERK, true);
265 })
266 .Schedule(3600ms, [this](TaskContext context) // according to sniffs it should be casted by both emperors.
267 {
268 if (Creature* twin = GetTwin())
269 {
270 if (me->IsWithinDist(twin, 60.f))
271 DoCast(twin, SPELL_HEAL_BROTHER, true);
272 }
273
274 context.Repeat();
275 });
276 }
277
278 void UpdateAI(uint32 diff) override
279 {
280 if (!UpdateVictim() && _introDone)
281 return;
282
283 _scheduler.Update(diff, [this]
284 {
285 if (!IAmVeklor())
286 DoMeleeAttackIfReady();
287 });
288 }
289
290 virtual bool IAmVeklor() = 0;
291
292protected:
293 TaskScheduler _scheduler;
294 bool _introDone;
