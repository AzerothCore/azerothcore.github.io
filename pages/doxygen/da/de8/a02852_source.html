109 boss_jeklik(Creature* creature) : BossAI(creature, DATA_JEKLIK) { }

111 void Reset() override
112 {
113 DoCastSelf(SPELL_GREEN_CHANNELING);
114 me->SetHover(false);
115 me->SetDisableGravity(false);
116 me->SetReactState(REACT_PASSIVE);
117 _Reset();
118 SetCombatMovement(false);
119 }
120
121 void JustDied(Unit* /*killer*/) override
122 {
123 _JustDied();
124 Talk(SAY_DEATH);
125 }
126
127 void EnterEvadeMode(EvadeReason why) override
128 {
129 me->GetMotionMaster()->Clear();
130 me->SetHomePosition(homePosition);
131 me->NearTeleportTo(homePosition.GetPositionX(), homePosition.GetPositionY(), homePosition.GetPositionZ(), homePosition.GetOrientation());
132 BossAI::EnterEvadeMode(why);
133 }
134
135 void EnterCombat(Unit* /*who*/) override
136 {
137 Talk(SAY_AGGRO);
138 me->RemoveAurasDueToSpell(SPELL_GREEN_CHANNELING);
139 me->SetHover(true);
140 me->SetDisableGravity(true);
141 DoCastSelf(SPELL_BAT_FORM, true);
142
143 me->GetMotionMaster()->MovePath(PATH_JEKLIK_INTRO, false);
144 }
145
146 void PathEndReached(uint32 /*pathId*/) override
147 {
148 me->SetHover(false);
149 me->SetDisableGravity(false);
150 _EnterCombat();
151 SetCombatMovement(true);
152 me->SetReactState(REACT_AGGRESSIVE);
153 events.SetPhase(PHASE_ONE);
154 events.ScheduleEvent(EVENT_CHARGE_JEKLIK, urand(10000, 20000), PHASE_ONE);
155 events.ScheduleEvent(EVENT_PIERCE_ARMOR, urand(5000, 15000), PHASE_ONE);
156 events.ScheduleEvent(EVENT_BLOOD_LEECH, urand(5000, 15000), PHASE_ONE);
157 events.ScheduleEvent(EVENT_SONIC_BURST, urand(5000, 15000), PHASE_ONE);
158 events.ScheduleEvent(EVENT_SWOOP, 20000, PHASE_ONE);
159 events.ScheduleEvent(EVENT_SPAWN_BATS, 30000, PHASE_ONE);
160 }
161
162 void DamageTaken(Unit* /*who*/, uint32& /*damage*/, DamageEffectType, SpellSchoolMask) override
163 {
164 if (events.IsInPhase(PHASE_ONE) && !HealthAbovePct(50))
165 {
166 me->RemoveAurasDueToSpell(SPELL_BAT_FORM);
167 DoResetThreat();
168 events.SetPhase(PHASE_TWO);
169 events.CancelEventGroup(PHASE_ONE);
170
171 events.ScheduleEvent(EVENT_CURSE_OF_BLOOD, urand(5000, 15000), PHASE_TWO);
172 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(10000, 15000), PHASE_TWO);
173 events.ScheduleEvent(EVENT_PSYCHIC_SCREAM, urand(25000, 35000), PHASE_TWO);
174 events.ScheduleEvent(EVENT_MIND_FLAY, urand(10000, 30000), PHASE_TWO);
175 events.ScheduleEvent(EVENT_GREATER_HEAL, 25000, PHASE_TWO);
176 events.ScheduleEvent(EVENT_SPAWN_FLYING_BATS, 10000, PHASE_TWO);
177
178 return;
179 }
180 }

182 void UpdateAI(uint32 diff) override
183 {
184 if (!UpdateVictim())
185 return;
186
187 events.Update(diff);
188
189 if (me->HasUnitState(UNIT_STATE_CASTING))
190 return;
191
192 while (uint32 eventId = events.ExecuteEvent())
193 {
194 switch (eventId)
195 {
196 // Phase one
197 case EVENT_CHARGE_JEKLIK:
198 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
199 {
200 DoCast(target, SPELL_CHARGE);
201 AttackStart(target);
202 }
203 events.ScheduleEvent(EVENT_CHARGE_JEKLIK, urand(15000, 30000), PHASE_ONE);
204 break;
205 case EVENT_PIERCE_ARMOR:
206 DoCastVictim(SPELL_PIERCE_ARMOR);
207 events.ScheduleEvent(EVENT_PIERCE_ARMOR, urand(20000, 30000), PHASE_ONE);
208 break;
209 case EVENT_BLOOD_LEECH:
210 DoCastVictim(SPELL_BLOOD_LEECH);
211 events.ScheduleEvent(EVENT_BLOOD_LEECH, urand(10000, 20000), PHASE_ONE);
212 break;
213 case EVENT_SONIC_BURST:
214 DoCastVictim(SPELL_SONIC_BURST);
215 events.ScheduleEvent(EVENT_SONIC_BURST, urand(20000, 30000), PHASE_ONE);
216 break;
217 case EVENT_SWOOP:
218 DoCastVictim(SPELL_SWOOP);
219 events.ScheduleEvent(EVENT_SWOOP, urand(20000, 30000), PHASE_ONE);
220 break;
221 case EVENT_SPAWN_BATS:
222 Talk(EMOTE_SUMMON_BATS);
223 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
224 for (uint8 i = 0; i < 6; ++i)
225 if (Creature* bat = me->SummonCreature(NPC_BLOODSEEKER_BAT, SpawnBat[i], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 15000))
226 bat->AI()->AttackStart(target);
227 events.ScheduleEvent(EVENT_SPAWN_BATS, 30000, PHASE_ONE);
228 break;
229 //Phase two
230 case EVENT_CURSE_OF_BLOOD:
231 DoCastSelf(SPELL_CURSE_OF_BLOOD);
232 events.ScheduleEvent(EVENT_CURSE_OF_BLOOD, urand(25000, 30000), PHASE_TWO);
233 break;
234 case EVENT_PSYCHIC_SCREAM:
235 DoCastVictim(SPELL_PSYCHIC_SCREAM);
236 events.ScheduleEvent(EVENT_PSYCHIC_SCREAM, urand(35000, 45000), PHASE_TWO);
237 break;
238 case EVENT_SHADOW_WORD_PAIN:
239 DoCastRandomTarget(SPELL_SHADOW_WORD_PAIN, 0, true);
240 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(12000, 18000), PHASE_TWO);
241 break;
242 case EVENT_MIND_FLAY:
243 DoCastVictim(SPELL_MIND_FLAY);
244 events.ScheduleEvent(EVENT_MIND_FLAY, urand(20000, 40000), PHASE_TWO);
245 break;
246 case EVENT_GREATER_HEAL:
247 Talk(EMOTE_GREAT_HEAL);
248 me->InterruptNonMeleeSpells(false);
249 DoCastSelf(SPELL_GREATER_HEAL);
250 events.ScheduleEvent(EVENT_GREATER_HEAL, 25000, PHASE_TWO);
251 break;
252 case EVENT_SPAWN_FLYING_BATS:
253 Talk(SAY_CALL_RIDERS);
254 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
255 if (Creature* flyingBat = me->SummonCreature(NPC_FRENZIED_BAT, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ() + 15.0f, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 15000))
256 flyingBat->AI()->DoZoneInCombat();
257 events.ScheduleEvent(EVENT_SPAWN_FLYING_BATS, urand(10000, 15000), PHASE_TWO);
258 break;
259 default:
260 break;
261 }
262 }
263
264 DoMeleeAttackIfReady();
265 }

269struct npc_batrider : public ScriptedAI
270{
271 npc_batrider(Creature* creature) : ScriptedAI(creature)
272 {
273 _scheduler.SetValidator([this]
274 {
275 return !me->HasUnitState(UNIT_STATE_CASTING);
276 });
277 }
278
279 void Reset() override
280 {
281 _scheduler.CancelAll();
282 me->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
283 me->SetHover(true);
284 me->SetDisableGravity(true);
285 me->AddUnitState(UNIT_STATE_ROOT);
286 }
287
288 void EnterCombat(Unit* /*who*/) override
289 {
290 _scheduler.Schedule(2s, [this](TaskContext context)
291 {
292 DoCastRandomTarget(SPELL_THROW_LIQUID_FIRE);
293 context.Repeat(7s);
294 });
295 }
296
297 void UpdateAI(uint32 diff) override
298 {
299 if (!UpdateVictim())
300 return;
301
302 _scheduler.Update(diff);
303 }
304
305private:
306 TaskScheduler _scheduler;
