31 void LfgPlayerData::SetState(LfgState state)
32 {
33 if (m_State == LFG_STATE_RAIDBROWSER && state != LFG_STATE_RAIDBROWSER && !CanOverrideRBState())
34 return;
35
36 switch (state)
37 {
38 case LFG_STATE_NONE:
39 case LFG_STATE_FINISHED_DUNGEON:
40 m_Roles = 0;
41 m_SelectedDungeons.clear();
42 m_Comment = "";
43 [[fallthrough]];
44 case LFG_STATE_DUNGEON:
45 m_OldState = state;
46 [[fallthrough]];
47 default:
48 m_State = state;
49 }
50 }
51
52 void LfgPlayerData::RestoreState()
53 {
54 if (m_State == LFG_STATE_RAIDBROWSER && m_OldState != LFG_STATE_RAIDBROWSER && !CanOverrideRBState())
55 return;
56
57 if (m_OldState == LFG_STATE_NONE)
58 {
59 m_SelectedDungeons.clear();
60 m_Roles = 0;
61 }
62 m_State = m_OldState;
63 }
