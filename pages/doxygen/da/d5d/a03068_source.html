105 boss_ossirian(Creature* creature) : BossAI(creature, DATA_OSSIRIAN)
106 {
107 _saidIntro = false;
108 }
109
110 void InitializeAI() override
111 {
112 Reset();
113 }
114
115 void Reset() override
116 {
117 BossAI::Reset();
118
119 _crystalIterator = urand(1, NUM_CRYSTALS - 1);
120
121 if (!ObjectAccessor::GetGameObject(*me, _firstCrystalGUID))
122 {
123 if (Creature* trigger = me->GetMap()->SummonCreature(NPC_OSSIRIAN_TRIGGER, CrystalCoordinates[0]))
124 {
125 if (GameObject* crystal = me->SummonGameObject(GO_OSSIRIAN_CRYSTAL,
126 CrystalCoordinates[0].GetPositionX(),
127 CrystalCoordinates[0].GetPositionY(),
128 CrystalCoordinates[0].GetPositionZ(),
129 0, 0, 0, 0, 0, uint32(-1)))
130 {
131 _firstCrystalGUID = crystal->GetGUID();
132 crystal->SetOwnerGUID(ObjectGuid::Empty);
133 crystal->RemoveGameObjectFlag(GO_FLAG_IN_USE);
134 crystal->AI()->SetGUID(trigger->GetGUID(), GUID_TRIGGER_PAIR);
135 }
136 }
137 }
138 }
139
140 void JustSummoned(Creature* creature) override
141 {
142 summons.Summon(creature);
143 }
144
145 void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
146 {
147 for (uint32 weakness : spellWeakness)
148 {
149 if (spell->Id == weakness)
150 {
151 me->RemoveAurasDueToSpell(SPELL_STRENGTH_OF_OSSIRIAN);
152 }
153 }
154 }
155
156 void SetGUID(ObjectGuid guid, int32 action) override
157 {
158 if (action == ACTION_TRIGGER_WEAKNESS && guid != _firstCrystalGUID)
159 {
160 SpawnNextCrystal();
161 }
162 }
163
164 void EnterCombat(Unit* who) override
165 {
166 BossAI::EnterCombat(who);
167 events.Reset();
168 events.ScheduleEvent(EVENT_SPEEDUP, 10s);
169 events.ScheduleEvent(EVENT_SILENCE, 30s);
170 events.ScheduleEvent(EVENT_CYCLONE, 20s);
171 events.ScheduleEvent(EVENT_STOMP, 30s);
172 DoCastSelf(SPELL_STRENGTH_OF_OSSIRIAN);
173 Talk(SAY_AGGRO);
174
175 Map* map = me->GetMap();
176 if (!map->IsDungeon())
177 return;
178
179 WorldPackets::Misc::Weather weather(WEATHER_STATE_HEAVY_SANDSTORM, 1.0f);
180 map->SendToPlayers(weather.Write());
181
182 SpawnNextCrystal(3);
183
184 std::list<uint32> pathIds = { 1446800, 1446790 };
185
186 for (Position pos : VortexPositions)
187 {
188 if (Creature* vortex = me->SummonCreature(NPC_SAND_VORTEX, pos))
189 {
190 vortex->GetMotionMaster()->MovePath(pathIds.front(), true);
191 pathIds.reverse();
192 }
193 }
194 }
195
196 void SummonedCreatureDespawn(Creature* summon) override
197 {
198 summons.Despawn(summon);
199
200 if (summon->GetEntry() == NPC_OSSIRIAN_TRIGGER)
201 {
202 if (GameObject* crystal = GetClosestGameObjectWithEntry(summon, GO_OSSIRIAN_CRYSTAL, 5.0f))
203 {
204 crystal->Delete();
205 }
206 }
207 }
208
209 void KilledUnit(Unit* /*victim*/) override
210 {
211 Talk(SAY_SLAY);
212 }
213
214 void SpawnNextCrystal(uint8 count = 1)
215 {
216 for (uint8 i = 0; i < count; ++i)
217 {
218 if (_crystalIterator == NUM_CRYSTALS)
219 _crystalIterator = 1;
220
221 if (Creature* trigger = me->SummonCreature(NPC_OSSIRIAN_TRIGGER, CrystalCoordinates[_crystalIterator]))
222 {
223 if (GameObject* crystal = trigger->SummonGameObject(GO_OSSIRIAN_CRYSTAL,
224 CrystalCoordinates[_crystalIterator].GetPositionX(),
225 CrystalCoordinates[_crystalIterator].GetPositionY(),
226 CrystalCoordinates[_crystalIterator].GetPositionZ(),
227 0, 0, 0, 0, 0, uint32(-1)))
228 {
229 ++_crystalIterator;
230 crystal->SetOwnerGUID(ObjectGuid::Empty);
231 crystal->RemoveGameObjectFlag(GO_FLAG_IN_USE);
232 crystal->AI()->SetGUID(trigger->GetGUID(), GUID_TRIGGER_PAIR);
233 }
234 }
235 }
236 }
237
238 void MoveInLineOfSight(Unit* who) override
239 {
240 if (!_saidIntro)
241 {
242 Talk(SAY_INTRO);
243 _saidIntro = true;
244 }
245
246 BossAI::MoveInLineOfSight(who);
247 }
248
249 void UpdateAI(uint32 diff) override
250 {
251 if (!UpdateVictim())
252 return;
253
254 events.Update(diff);
255 bool applySupreme = true;
256 if (me->HasAura(SPELL_STRENGTH_OF_OSSIRIAN))
257 {
258 applySupreme = false;
259 }
260 else
261 {
262 for (uint32 weakness : spellWeakness)
263 {
264 if (me->HasAura(weakness))
265 {
266 applySupreme = false;
267 break;
268 }
269 }
270 }
271
272 if (applySupreme)
273 {
274 DoCastSelf(SPELL_STRENGTH_OF_OSSIRIAN);
275 Talk(SAY_SUPREME);
276 }
277
278 while (uint32 eventId = events.ExecuteEvent())
279 {
280 switch (eventId)
281 {
282 case EVENT_SPEEDUP:
283 DoCastSelf(SPELL_SPEED_BURST);
284 break;
285 case EVENT_SILENCE:
286 DoCastAOE(SPELL_CURSE_OF_TONGUES);
287 events.ScheduleEvent(EVENT_SILENCE, 20s, 30s);
288 break;
289 case EVENT_CYCLONE:
290 DoCastVictim(SPELL_ENVELOPING_WINDS);
291 events.ScheduleEvent(EVENT_CYCLONE, 20s);
292 break;
293 case EVENT_STOMP:
294 DoCastAOE(SPELL_WAR_STOMP);
295 events.ScheduleEvent(EVENT_STOMP, 30s);
296 break;
297 default:
298 break;
299 }
300 }
301 DoMeleeAttackIfReady();
302 }
303
304protected:
305 uint8 _crystalIterator;
306 ObjectGuid _firstCrystalGUID;
307 bool _saidIntro;
308};
