88void OPvPCapturePointGH::ChangeState()
89{
90 uint32 artkit = 21;
91 switch (m_State)
92 {
93 case OBJECTIVESTATE_ALLIANCE:
94 sGameEventMgr->StopEvent(GH_ALLIANCE_DEFENSE_EVENT);
95 sGameEventMgr->StopEvent(GH_HORDE_DEFENSE_EVENT);
96 sGameEventMgr->StartEvent(GH_ALLIANCE_DEFENSE_EVENT);
97 artkit = 2;
98 break;
99 case OBJECTIVESTATE_HORDE:
100 sGameEventMgr->StopEvent(GH_ALLIANCE_DEFENSE_EVENT);
101 sGameEventMgr->StopEvent(GH_HORDE_DEFENSE_EVENT);
102 sGameEventMgr->StartEvent(GH_HORDE_DEFENSE_EVENT);
103 artkit = 1;
104 break;
105 default:
106 break;
107 }
108
109 Map* map = sMapMgr->FindMap(571, 0);
110 auto bounds = map->GetGameObjectBySpawnIdStore().equal_range(m_capturePointSpawnId);
111 for (auto itr = bounds.first; itr != bounds.second; ++itr)
112 itr->second->SetGoArtKit(artkit);
113}
