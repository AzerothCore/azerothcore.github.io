119 void UpdateAI(uint32 diff) override
120 {
121 if (!UpdateVictim())
122 return;
123
124 events.Update(diff);
125
126 if (me->HasUnitState(UNIT_STATE_CASTING))
127 return;
128
129 while (uint32 eventId = events.ExecuteEvent())
130 {
131 switch (eventId)
132 {
133 case EVENT_VANISH:
134 me->SetReactState(REACT_PASSIVE);
135 _dynamicFlags = me->GetDynamicFlags();
136 me->RemoveDynamicFlag(UNIT_DYNFLAG_TRACK_UNIT);
137 DoCastSelf(SPELL_VANISH);
138 events.DelayEvents(5s);
139 events.ScheduleEvent(EVENT_AMBUSH, 5s);
140 events.ScheduleEvent(EVENT_VANISH, 38s, 45s);
141 return;
142 case EVENT_AMBUSH:
143 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1))
144 {
145 me->NearTeleportTo(target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), me->GetOrientation());
146 DoCast(target, SPELL_AMBUSH, true);
147 }
148 me->SetDynamicFlag(_dynamicFlags);
149 me->RemoveAurasDueToSpell(SPELL_VANISH);
150 me->SetReactState(REACT_AGGRESSIVE);
151 break;
152 case EVENT_GOUGE:
153 DoCastAOE(SPELL_GOUGE);
154 events.ScheduleEvent(EVENT_GOUGE, 10s, 15s);
155 return;
156 case EVENT_THOUSAND_BLADES:
157 {
158 if (_thousandBladesTargets.empty())
159 {
160 std::vector<Unit*> targetList;
161 ThreatContainer::StorageType const& threatlist = me->GetThreatMgr().GetThreatList();
162 for (ThreatContainer::StorageType::const_iterator itr = threatlist.begin(); itr != threatlist.end(); ++itr)
163 {
164 if (Unit* target = (*itr)->getTarget())
165 {
166 if (target->IsAlive() && target->IsWithinDist2d(me, 100.f))
167 {
168 targetList.push_back(target);
169 }
170 }
171 }
172
173 if (!targetList.empty())
174 {
175 Acore::Containers::RandomShuffle(targetList);
176
177 // First get ranged targets
178 for (Unit* target : targetList)
179 {
180 if (!target->IsWithinMeleeRange(me))
181 {
182 _thousandBladesTargets.push_back(target);
183 }
184 }
185
186 if (_thousandBladesTargets.size() < _thousandBladesCount)
187 {
188 // if still not enough, get melee targets
189 for (Unit* target : targetList)
190 {
191 if (target->IsWithinMeleeRange(me))
192 {
193 _thousandBladesTargets.push_back(target);
194 }
195 }
196 }
197
198 if (!_thousandBladesTargets.empty())
199 {
200 Acore::Containers::RandomResize(_thousandBladesTargets, _thousandBladesCount);
201 }
202 }
203 }
204
205 if (!_thousandBladesTargets.empty())
206 {
207 std::vector<Unit*>::iterator itr = _thousandBladesTargets.begin();
208 std::advance(itr, urand(0, _thousandBladesTargets.size() - 1));
209
210 if (Unit* target = *itr)
211 {
212 DoCast(target, SPELL_THOUSAND_BLADES, false);
213 }
214
215 if (_thousandBladesTargets.erase(itr) != _thousandBladesTargets.end())
216 {
217 events.ScheduleEvent(EVENT_THOUSAND_BLADES, 500ms);
218 }
219 else
220 {
221 _thousandBladesCount = urand(2, 5);
222 events.ScheduleEvent(EVENT_THOUSAND_BLADES, 15s, 22s);
223 }
224 }
225 else
226 {
227 _thousandBladesCount = urand(2, 5);
228 events.ScheduleEvent(EVENT_THOUSAND_BLADES, 15s, 22s);
229 }
230 break;
231 }
232 default:
233 break;
234 }
235 }
236
237 if (me->GetReactState() == REACT_AGGRESSIVE)
238 DoMeleeAttackIfReady();
239 }

241 private:
242 bool _enraged;
243 uint32 _dynamicFlags;
244 uint8 _thousandBladesCount;
245 std::vector<Unit*> _thousandBladesTargets;
246 };
