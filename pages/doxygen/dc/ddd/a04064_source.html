96 case EVENT_SPELL_BERSERK:

98 break;
99 case EVENT_SPELL_POUNDING:
100 Talk(SAY_POUNDING);
101 me->CastSpell(me, SPELL_POUNDING, false);
102 events.ScheduleEvent(EVENT_SPELL_POUNDING, 15000);
103 break;
104 case EVENT_SPELL_ARCANEORB:
105 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, -18.0f, true))
106 me->CastSpell(target, SPELL_ARCANE_ORB, false);
107 else if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 20.0f, true))
108 me->CastSpell(target, SPELL_ARCANE_ORB, false);
109 events.ScheduleEvent(EVENT_SPELL_ARCANEORB, 4000);
110 break;
111 case EVENT_SPELL_KNOCK_AWAY:
112 me->CastSpell(me->GetVictim(), SPELL_KNOCK_AWAY, false);
113 events.ScheduleEvent(EVENT_SPELL_POUNDING, 25000);
114 break;
115 }
116
117 DoMeleeAttackIfReady();
118 }
