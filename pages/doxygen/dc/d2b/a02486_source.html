111 void Reset() override
112 {
113 _Reset();
114
115 if (instance->GetBossState(DATA_GYTH) == IN_PROGRESS)
116 {
117 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_PREPARATION);
118 me->SetImmuneToAll(false);

121
122 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_PREPARATION);
123 me->SetImmuneToAll(true);
124 gythEvent = false;
125 victorGUID.Clear();
126 waveDoorGUID.Clear();
127 _currentWave = 0;
128
129 summons.DespawnAll();
130

132 victor->Respawn(true);
133
134 if (GameObject* exitDoor = me->GetMap()->GetGameObject(instance->GetGuidData(GO_GYTH_ENTRY_DOOR)))
135 exitDoor->SetGoState(GO_STATE_ACTIVE);
136
137 instance->SetBossState(DATA_WARCHIEF_REND_BLACKHAND, NOT_STARTED);
138 }

280 {
281 victor->SetFacingToObject(me);
282 victor->HandleEmoteCommand(EMOTE_ONESHOT_TALK);

302 victor->HandleEmoteCommand(EMOTE_ONESHOT_QUESTION);

305 me->HandleEmoteCommand(EMOTE_ONESHOT_ROAR);

339 victor->AI()->Talk(SAY_NEFARIUS_5);

341 events.ScheduleEvent(EVENT_SPAWN_WAVE, 3000);

343 case EVENT_WAVES_COMPLETE_TEXT_1:
344 events.ScheduleEvent(EVENT_TURN_TO_PLAYER, 0);

346 victor->AI()->Talk(SAY_NEFARIUS_6);
347 events.ScheduleEvent(EVENT_TURN_TO_FACING_1, 4000);
348 events.ScheduleEvent(EVENT_WAVES_COMPLETE_TEXT_2, 13000);

350 case EVENT_WAVES_COMPLETE_TEXT_2:

352 victor->AI()->Talk(SAY_NEFARIUS_7);

359 victor->AI()->Talk(SAY_NEFARIUS_8);
360 events.ScheduleEvent(EVENT_PATH_NEFARIUS, 1000);
361 events.ScheduleEvent(EVENT_PATH_REND, 1000);
362 break;
363 case EVENT_PATH_NEFARIUS:
364 if (Creature* victor = ObjectAccessor::GetCreature(*me, victorGUID))
365 victor->GetMotionMaster()->MovePath(NEFARIUS_PATH_1, true);
366 break;
367 case EVENT_PATH_REND:
368 me->GetMotionMaster()->MovePath(REND_PATH_1, false);
369 break;
370 case EVENT_TELEPORT_1:
371 me->NearTeleportTo(194.2993f, -474.0814f, 121.4505f, -0.01225555f);
372 events.ScheduleEvent(EVENT_TELEPORT_2, 13000);
373 break;
374 case EVENT_TELEPORT_2:
375 me->NearTeleportTo(216.485f, -434.93f, 110.888f, -0.01225555f);
376 me->SummonCreature(NPC_GYTH, 211.762f, -397.5885f, 111.1817f, 4.747295f);
377 break;
378 case EVENT_SPAWN_WAVE:
379 SummonWave();
380 break;
381 default:
382 break;
383 }
384 }
385 }
386
387 if (!UpdateVictim())
388 return;
389
390 events.Update(diff);
391
392 if (me->HasUnitState(UNIT_STATE_CASTING))
393 return;
394
395 while (uint32 eventId = events.ExecuteEvent())
396 {
397 switch (eventId)
398 {
399 case EVENT_WHIRLWIND:
400 DoCast(SPELL_WHIRLWIND);
401 events.ScheduleEvent(EVENT_WHIRLWIND, urand(13000, 18000));
402 break;
403 case EVENT_CLEAVE:
404 DoCastVictim(SPELL_CLEAVE);
405 events.ScheduleEvent(EVENT_CLEAVE, urand(10000, 14000));
406 break;
407 case EVENT_MORTAL_STRIKE:
408 DoCastVictim(SPELL_MORTAL_STRIKE);
409 events.ScheduleEvent(EVENT_MORTAL_STRIKE, urand(14000, 16000));
410 break;
411 }
412 }
413 DoMeleeAttackIfReady();
414 }
