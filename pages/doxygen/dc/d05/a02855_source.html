57struct boss_jindo : public BossAI
58{
59 boss_jindo(Creature* creature) : BossAI(creature, DATA_JINDO) { }
60
61 void EnterCombat(Unit* who) override
62 {
63 BossAI::EnterCombat(who);
64 events.ScheduleEvent(EVENT_BRAIN_WASH_TOTEM, 20000);
65 events.ScheduleEvent(EVENT_POWERFULL_HEALING_WARD, 16000);
66 events.ScheduleEvent(EVENT_HEX, 8000);
67 events.ScheduleEvent(EVENT_DELUSIONS_OF_JINDO, 10000);
68 events.ScheduleEvent(EVENT_TELEPORT, 5000);
69
70 Talk(SAY_AGGRO);
71
72 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_NONE);
73 _scheduler.CancelAll();
74 }
75
76 void JustSummoned(Creature* summon) override
77 {
78 BossAI::JustSummoned(summon);
79
80 switch (summon->GetEntry())
81 {
82 case NPC_BRAIN_WASH_TOTEM:
83 summon->SetReactState(REACT_PASSIVE);
84 if (Unit* target = SelectTarget(SelectTargetMethod::Random, me->GetThreatMgr().GetThreatListSize() > 1 ? 1 : 0))
85 {
86 summon->CastSpell(target, summon->m_spells[0], true);
87 }
88 break;
89 default:
90 break;
91 }
92 }
93
94 void EnterEvadeMode(EvadeReason evadeReason) override
95 {
96 if (_EnterEvadeMode(evadeReason))
97 {
98 Reset();
99 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_DANCE);
100
101 _scheduler.Schedule(4s, [this](TaskContext /*context*/)
102 {
103 me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_NONE);
104 me->AddUnitState(UNIT_STATE_EVADE);
105 me->GetMotionMaster()->MoveTargetedHome();
106 });
107 }
108 }
109
110 void UpdateAI(uint32 diff) override
111 {
112 _scheduler.Update(diff);
113
114 if (!UpdateVictim())
115 return;
116
117 events.Update(diff);
118
119 if (me->HasUnitState(UNIT_STATE_CASTING))
120 return;
121
122 while (uint32 eventId = events.ExecuteEvent())
123 {
124 switch (eventId)
125 {
126 case EVENT_BRAIN_WASH_TOTEM:
127 DoCastSelf(SPELL_BRAIN_WASH_TOTEM);
128 events.ScheduleEvent(EVENT_BRAIN_WASH_TOTEM, urand(18000, 26000));
129 break;
130 case EVENT_POWERFULL_HEALING_WARD:
131 DoCastSelf(SPELL_POWERFULL_HEALING_WARD, true);
132 events.ScheduleEvent(EVENT_POWERFULL_HEALING_WARD, urand(14000, 20000));
133 break;
134 case EVENT_HEX:
135 if (me->GetThreatMgr().GetThreatListSize() > 1)
136 DoCastVictim(SPELL_HEX, true);
137 events.ScheduleEvent(EVENT_HEX, urand(12000, 20000));
138 break;
139 case EVENT_DELUSIONS_OF_JINDO:
140 DoCastRandomTarget(SPELL_DELUSIONS_OF_JINDO);
141 events.ScheduleEvent(EVENT_DELUSIONS_OF_JINDO, urand(4000, 12000));
142 break;
143 case EVENT_TELEPORT:
144 DoCastRandomTarget(SPELL_BANISH);
145 events.ScheduleEvent(EVENT_TELEPORT, urand(15000, 23000));
146 break;
147 default:
148 break;
149 }
150 }
151
152 DoMeleeAttackIfReady();
153 }
154
155 bool CanAIAttack(Unit const* target) const override
156 {
157 if (me->GetThreatMgr().GetThreatListSize() > 1 && me->GetThreatMgr().GetOnlineContainer().getMostHated()->getTarget() == target)
158 return !target->HasAura(SPELL_HEX);
159
160 return true;
161 }
162
163private:
164 TaskScheduler _scheduler;
165};
