88// 50526 - Wandering Plague
89class spell_dk_wandering_plague : public SpellScript
90{
91 PrepareSpellScript(spell_dk_wandering_plague);
92
93 void FilterTargets(std::list<WorldObject*>& targets)
94 {
95 static const AuraType ccAuras[] =
96 {
97 SPELL_AURA_MOD_CONFUSE,
98 SPELL_AURA_MOD_FEAR,
99 SPELL_AURA_MOD_STUN,
100 SPELL_AURA_MOD_ROOT,
101 SPELL_AURA_TRANSFORM,
102 SPELL_AURA_NONE
103 };
104
105 for (std::list<WorldObject*>::iterator itr = targets.begin(); itr != targets.end();)
106 {
107 Unit* target = (*itr)->ToUnit();
108 if (!target)
109 {
110 targets.erase(itr++);
111 continue;
112 }
113
114 bool skip = false;
115 for (uint8 index = 0; !skip && ccAuras[index] != SPELL_AURA_NONE; ++index)
116 {
117 Unit::AuraEffectList const& auras = target->GetAuraEffectsByType(ccAuras[index]);
118 for (Unit::AuraEffectList::const_iterator i = auras.begin(); i != auras.end(); ++i)
119 if ((*i)->GetAmount())
120 {
121 skip = true;
122 break;
123 }
124 }
125
126 if (skip)
127 targets.erase(itr++);
128 else
129 ++itr;
130 }
131 }
132
133 void Register() override
134 {
135 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_dk_wandering_plague::FilterTargets, EFFECT_0, TARGET_UNIT_DEST_AREA_ENEMY);
136 }

159 ghoul->SetCreateStat(Stats(stat), float(pInfo->stats[stat]));

165 // Avoidance, Night of the Dead
166 if (Aura* aur = ghoul->AddAura(62137, ghoul))
167 if (AuraEffect* aurEff = GetCaster()->GetAuraEffect(SPELL_AURA_ADD_FLAT_MODIFIER, SPELLFAMILY_DEATHKNIGHT, 2718, 0))
168 if (aur->GetEffect(0))
169 aur->GetEffect(0)->SetAmount(-aurEff->GetSpellInfo()->Effects[EFFECT_2].CalcValue());

171 // Hit / Expertise scalling, warlock / hunter pets have this by default
172 ghoul->AddAura(SPELL_HUNTER_PET_SCALING_04, ghoul);
173
174 // DK Ghoul haste refresh
175 float val = (GetCaster()->m_modAttackSpeedPct[BASE_ATTACK] - 1.0f) * 100.0f;
176 ghoul->m_modAttackSpeedPct[BASE_ATTACK] = GetCaster()->m_modAttackSpeedPct[BASE_ATTACK];
177 ghoul->SetFloatValue(UNIT_FIELD_BASEATTACKTIME, 2000.0f);
178 ghoul->ApplyPercentModFloatValue(UNIT_FIELD_BASEATTACKTIME, val, true); // we want to reduce attack time
179
180 // Strength + Stamina
181 for (uint8 i = STAT_STRENGTH; i <= STAT_STAMINA; ++i)
182 {
183 Stats stat = Stats(i);
184 if (stat != STAT_STRENGTH && stat != STAT_STAMINA)
185 continue;
186
187 float value = 0.0f;
188 float mod = (stat == STAT_STAMINA ? 0.3f : 0.7f);
189
190 // Check just if owner has Ravenous Dead since it's effect is not an aura

197 // Glyph of the Ghoul
198 aurEff = GetCaster()->GetAuraEffect(SPELL_DK_GLYPH_OF_THE_GHOUL, EFFECT_0);
199 if (aurEff)
200 mod += CalculatePct(1.0f, aurEff->GetAmount()); // Glyph of the Ghoul adds a flat value to the scale mod
201
202 value = float(GetCaster()->GetStat(stat)) * mod;
203 value = ghoul->GetTotalStatValue(stat, value);
204 ghoul->SetStat(stat, int32(value));
205 ghoul->ApplyStatBuffMod(stat, value, true);
206 }
207
208 // Attack Power
209 ghoul->SetModifierValue(UNIT_MOD_ATTACK_POWER, BASE_VALUE, 589 + ghoul->GetStat(STAT_STRENGTH) + ghoul->GetStat(STAT_AGILITY));
210 ghoul->SetInt32Value(UNIT_FIELD_ATTACK_POWER, (int32)ghoul->GetModifierValue(UNIT_MOD_ATTACK_POWER, BASE_VALUE) * ghoul->GetModifierValue(UNIT_MOD_ATTACK_POWER, BASE_PCT));
211 ghoul->SetInt32Value(UNIT_FIELD_ATTACK_POWER_MODS, (int32)ghoul->GetModifierValue(UNIT_MOD_ATTACK_POWER, TOTAL_VALUE));
212 ghoul->SetFloatValue(UNIT_FIELD_ATTACK_POWER_MULTIPLIER, ghoul->GetModifierValue(UNIT_MOD_ATTACK_POWER, TOTAL_PCT) - 1.0f);
213

215 ghoul->SetModifierValue(UNIT_MOD_HEALTH, TOTAL_VALUE, (ghoul->GetStat(STAT_STAMINA) - ghoul->GetCreateStat(STAT_STAMINA)) * 10.0f);
216
217 // Power Energy
218 ghoul->SetModifierValue(UnitMods(UNIT_MOD_POWER_START + static_cast<uint8>(POWER_ENERGY)), BASE_VALUE, ghoul->GetCreatePowers(POWER_ENERGY));
219 ghoul->UpdateAllStats();
220 ghoul->SetFullHealth();
221
222 // Aura Immunities
223 ghoul->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_MELEE_RANGED_HASTE, true, SPELL_BLOCK_TYPE_POSITIVE);
224 ghoul->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MELEE_SLOW, true, SPELL_BLOCK_TYPE_POSITIVE);
225 ghoul->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_STAT, true, SPELL_BLOCK_TYPE_POSITIVE);

228 ghoul->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_ATTACK_POWER_PCT, true, SPELL_BLOCK_TYPE_POSITIVE);
229 }
230 }
231 }
