37struct npc_pet_pri_lightwell : public TotemAI
38{
39 npc_pet_pri_lightwell(Creature* creature) : TotemAI(creature) { }
40
41 void InitializeAI() override
42 {
43 if (Unit* owner = me->ToTempSummon()->GetSummonerUnit())
44 {
45 uint32 hp = uint32(owner->GetMaxHealth() * 0.3f);
46 me->SetMaxHealth(hp);
47 me->SetHealth(hp);
48 me->SetLevel(owner->getLevel());
49 }
50
51 me->CastSpell(me, SPELL_PRIEST_LIGHTWELL_CHARGES, false); // Spell for Lightwell Charges

56struct npc_pet_pri_shadowfiend : public PetAI
57{
58 npc_pet_pri_shadowfiend(Creature* creature) : PetAI(creature) { }
59
60 void Reset() override
61 {
62 PetAI::Reset();
63 if (!me->HasAura(SPELL_PRIEST_SHADOWFIEND_DODGE))
64 me->AddAura(SPELL_PRIEST_SHADOWFIEND_DODGE, me);
65
66 if (Unit* target = me->SelectNearestTarget(15.0f))
67 AttackStart(target);
68 }
69
70 void JustDied(Unit* /*killer*/) override
71 {
72 if (me->IsSummon())
73 if (Unit* owner = me->ToTempSummon()->GetSummonerUnit())
74 if (owner->HasAura(SPELL_PRIEST_GLYPH_OF_SHADOWFIEND))
75 owner->CastSpell(owner, SPELL_PRIEST_GLYPH_OF_SHADOWFIEND_MANA, true);
76 }
77};
