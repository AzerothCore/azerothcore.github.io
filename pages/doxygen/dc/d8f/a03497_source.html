24bool EoEDrakeEnterVehicleEvent::Execute(uint64 /*eventTime*/, uint32 /*updateTime*/)
25{
26 if (Player* p = ObjectAccessor::GetPlayer(_owner, _playerGUID))
27 if (p->IsInWorld() && !p->IsDuringRemoveFromWorld() && !p->isBeingLoaded() && p->FindMap() == _owner.FindMap())
28 {
29 p->CastCustomSpell(60683, SPELLVALUE_BASE_POINT0, 1, &_owner, true);
30 return true;
31 }
32 _owner.DespawnOrUnsummon(1);
33 return true;
34}

125 void SetData(uint32 type, uint32 data) override
126 {
127 switch(type)
128 {
129 case DATA_IRIS_ACTIVATED:
130 if (EncounterStatus == NOT_STARTED)
131 if (Creature* c = instance->GetCreature(NPC_MalygosGUID))
132 if (Player* plr = c->SelectNearestPlayer(250.0f))
133 c->AI()->AttackStart(plr);
134 break;
135 case DATA_ENCOUNTER_STATUS:
136 EncounterStatus = data;
137 switch(data)
138 {
139 case NOT_STARTED:
140 bPokeAchiev = false;
141 if (GameObject* go = instance->GetGameObject(GO_IrisGUID))
142 {
143 go->SetPhaseMask(1, true);
144 HandleGameObject(GO_IrisGUID, false, go);
145 }
146 if (GameObject* go = instance->GetGameObject(GO_ExitPortalGUID))
147 go->SetPhaseMask(1, true);
148 if (GameObject* go = instance->GetGameObject(GO_PlatformGUID))
149 {
150 go->SetDestructibleState(GO_DESTRUCTIBLE_REBUILDING, nullptr, true);
151 go->EnableCollision(true);
152 }
153 break;
154 case IN_PROGRESS:
155 bPokeAchiev = (instance->GetPlayersCountExceptGMs() < (instance->GetSpawnMode() == 0 ? (uint32)9 : (uint32)21));
156 break;
157 case DONE:
158 bPokeAchiev = false;
159 if (GameObject* go = instance->GetGameObject(GO_ExitPortalGUID))
160 go->SetPhaseMask(1, true);
161 if (Creature* c = instance->GetCreature(NPC_MalygosGUID))
162 if (c->SummonCreature(NPC_ALEXSTRASZA, 798.0f, 1268.0f, 299.0f, 2.45f, TEMPSUMMON_TIMED_DESPAWN, 604800000))
163 break;
164 }
165 if (data == DONE)
166 SaveToDB();
167 break;
168 case DATA_SET_IRIS_INACTIVE:
169 if (GameObject* go = instance->GetGameObject(GO_IrisGUID))
170 {
171 HandleGameObject(GO_IrisGUID, true, go);
172 if (Creature* c = go->SummonCreature(NPC_WORLD_TRIGGER_LAOI, *go, TEMPSUMMON_TIMED_DESPAWN, 10000))
173 c->CastSpell(c, SPELL_IRIS_ACTIVATED, true);
174 }
175 break;
176 case DATA_HIDE_IRIS_AND_PORTAL:
177 if (GameObject* go = instance->GetGameObject(GO_IrisGUID))
178 go->SetPhaseMask(2, true);
179 if (GameObject* go = instance->GetGameObject(GO_ExitPortalGUID))
180 go->SetPhaseMask(2, true);
181 break;
182 }
183 }

221 void Load(const char* in) override
222 {
223 if( !in )
224 {
225 OUT_LOAD_INST_DATA_FAIL;
226 return;
227 }
228
229 OUT_LOAD_INST_DATA(in);
230
231 char dataHead1, dataHead2;
232 uint32 data0;
233 std::istringstream loadStream(in);
234 loadStream >> dataHead1 >> dataHead2 >> data0;
235
236 if( dataHead1 == 'E' && dataHead2 == 'E' )
237 {
238 EncounterStatus = data0;
239 switch(EncounterStatus)
240 {
241 case IN_PROGRESS:
242 EncounterStatus = NOT_STARTED;
243 break;
244 case DONE:
245 // destroy platform, hide iris
246 ProcessEvent(nullptr, 20158);
247 if (GameObject* go = instance->GetGameObject(GO_IrisGUID))
248 go->SetPhaseMask(2, true);
249 break;
250 }
251 }
252 else
253 OUT_LOAD_INST_DATA_FAIL;
254
255 OUT_LOAD_INST_DATA_COMPLETE;
256 }
