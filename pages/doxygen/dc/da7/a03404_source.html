331 void UpdateAI(uint32 diff) override
332 {
333 if (!UpdateVictim() && !(events.GetPhaseMask() & PHASE_INTRO_MASK))
334 return;
335
336 events.Update(diff);
337
338 if (me->HasUnitState(UNIT_STATE_CASTING) && !(events.GetPhaseMask() & PHASE_INTRO_MASK))
339 return;
340
341 switch (events.ExecuteEvent())
342 {
343 case 0:
344 break;
345 case EVENT_INTRO_2:
346 Talk(SAY_INTRO_2);
347 break;
348 case EVENT_INTRO_3:
349 Talk(SAY_INTRO_3);
350 break;
351 case EVENT_INTRO_4:
352 Talk(SAY_INTRO_4);
353 break;
354 case EVENT_INTRO_5:
355 Talk(SAY_INTRO_5);
356 break;
357 case EVENT_INTRO_6:
358 Talk(SAY_INTRO_6);
359 break;
360 case EVENT_INTRO_7:
361 Talk(SAY_INTRO_7);
362 break;
363 case EVENT_BERSERK:
364 me->CastSpell(me, SPELL_BERSERK, true);
365 Talk(SAY_BERSERK);
366 break;
367 case EVENT_SPELL_DEATH_AND_DECAY:
368 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
369 me->CastSpell(target, SPELL_DEATH_AND_DECAY, false);
370 events.RepeatEvent(urand(22000, 30000));
371 break;
372 case EVENT_SPELL_DOMINATE_MIND_25:
373 {
374 Talk(SAY_DOMINATE_MIND);
375
376 std::vector<Player*> validPlayers;
377 Map::PlayerList const& pList = me->GetMap()->GetPlayers();
378 for (Map::PlayerList::const_iterator itr = pList.begin(); itr != pList.end(); ++itr)
379 if (Player* plr = itr->GetSource())
380 if (plr->IsAlive() && !plr->IsGameMaster() && plr->GetExactDist2dSq(me) < (150.0f * 150.0f))
381 if (!me->GetVictim() || me->GetVictim()->GetGUID() != plr->GetGUID())
382 {
383 // shouldn't be casted on any victim of summoned mobs
384 bool valid = true;
385 for (ObjectGuid const& guid : summons)
386 if (Creature* c = ObjectAccessor::GetCreature(*me, guid))
387 if (c->IsAlive() && c->GetVictim() && c->GetVictim()->GetGUID() == plr->GetGUID())
388 {
389 valid = false;
390 break;
391 }
392 if (valid)
393 validPlayers.push_back(plr);
394 }
395
396 std::vector<Player*>::iterator begin = validPlayers.begin(), end = validPlayers.end();
397
398 std::random_device rd;
399 std::shuffle(begin, end, std::default_random_engine{rd()});
400
401 for (uint8 i = 0; i < RAID_MODE<uint8>(0, 1, 1, 3) && i < validPlayers.size(); i++)
402 {
403 Unit* target = validPlayers[i];
404 me->CastSpell(target, SPELL_DOMINATE_MIND_25, true);
405 }
406
407 events.RepeatEvent(urand(40000, 45000));
408 }
409 break;
410 case EVENT_SPELL_SHADOW_BOLT:
411 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
412 me->CastSpell(target, SPELL_SHADOW_BOLT, false);
413 events.RepeatEvent(2100);
414 break;
415 case EVENT_SUMMON_WAVE_P1:
416 SummonWaveP1();
417 events.RepeatEvent(IsHeroic() ? 45000 : 60000);
418 break;
419 case EVENT_EMPOWER_CULTIST:
420 EmpowerCultist();
421 events.RepeatEvent(urand(18000, 25000));
422 break;
423 case EVENT_SPELL_FROSTBOLT:
424 me->CastSpell(me->GetVictim(), SPELL_FROSTBOLT, false);
425 events.RepeatEvent(12000);
426 break;
427 case EVENT_SPELL_FROSTBOLT_VOLLEY:
428 me->CastSpell((Unit*)nullptr, SPELL_FROSTBOLT_VOLLEY, false);
429 events.RepeatEvent(urand(13000, 15000));
430 break;
431 case EVENT_SPELL_TOUCH_OF_INSIGNIFICANCE:
432 me->CastSpell(me->GetVictim(), SPELL_TOUCH_OF_INSIGNIFICANCE, false);
433 events.RepeatEvent(urand(6000, 9000));
434 break;
435 case EVENT_SUMMON_WAVE_P2:
436 SummonWaveP2();
437 events.RepeatEvent(45000);
438 break;
439 case EVENT_SPELL_SUMMON_SHADE:
440 {
441 uint8 count = 1;
442 if (GetDifficulty() == RAID_DIFFICULTY_25MAN_NORMAL)
443 count = 2;
444 else if (GetDifficulty() == RAID_DIFFICULTY_25MAN_HEROIC)
445 count = 3;
446
447 std::list<Unit*> targets;
448 SelectTargetList(targets, NonTankTargetSelector(me, true), count, SelectTargetMethod::Random);
449 if (!targets.empty())
450 for (std::list<Unit*>::iterator itr = targets.begin(); itr != targets.end(); ++itr)
451 me->CastSpell(*itr, SPELL_SUMMON_SHADE, true);
452 }
453 events.RepeatEvent(12000);
454 break;
455 }
456
457 if (me->HasAura(SPELL_MANA_BARRIER))
458 return;
459
460 DoMeleeAttackIfReady();
461 }
