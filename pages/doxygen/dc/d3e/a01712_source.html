45enum PathType
46{
47 PATHFIND_BLANK = 0x00, // path not built yet
48 PATHFIND_NORMAL = 0x01, // normal path
49 PATHFIND_SHORTCUT = 0x02, // travel through obstacles, terrain, air, etc (old behavior)
50 PATHFIND_INCOMPLETE = 0x04, // we have partial path to follow - getting closer to target
51 PATHFIND_NOPATH = 0x08, // no valid path at all or error in generating one
52 PATHFIND_NOT_USING_PATH = 0x10, // used when we are either flying/swiming or on map w/o mmaps
53 PATHFIND_SHORT = 0x20, // path is longer or equal to its limited path length
54 PATHFIND_FARFROMPOLY_START = 0x40, // start position is far from the mmap poligon
55 PATHFIND_FARFROMPOLY_END = 0x80, // end positions is far from the mmap poligon
56 PATHFIND_FARFROMPOLY = PATHFIND_FARFROMPOLY_START | PATHFIND_FARFROMPOLY_END, // start or end positions are far from the mmap poligon
57};

98 [[nodiscard]] float getPathLength() const
99 {
100 float len = 0.0f;
101 float dx, dy, dz;
102 uint32 size = _pathPoints.size();
103 if (size)
104 {
105 dx = _pathPoints[0].x - _startPosition.x;
106 dy = _pathPoints[0].y - _startPosition.y;
107 dz = _pathPoints[0].z - _startPosition.z;
108 len += std::sqrt( dx * dx + dy * dy + dz * dz );
109 }
110 else
111 {
112 return len;
113 }
114
115 for (uint32 i = 1; i < size; ++i)
116 {
117 dx = _pathPoints[i].x - _pathPoints[i - 1].x;
118 dy = _pathPoints[i].y - _pathPoints[i - 1].y;
119 dz = _pathPoints[i].z - _pathPoints[i - 1].z;
120 len += std::sqrt( dx * dx + dy * dy + dz * dz );
121 }
122 return len;
123 }
