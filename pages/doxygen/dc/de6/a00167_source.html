50/*static*/ SessionKey SRP6::SHA1Interleave(SRP6::EphemeralKey const& S)
51{
52 // split S into two buffers
53 std::array<uint8, EPHEMERAL_KEY_LENGTH/2> buf0, buf1;
54 for (size_t i = 0; i < EPHEMERAL_KEY_LENGTH/2; ++i)
55 {
56 buf0[i] = S[2 * i + 0];
57 buf1[i] = S[2 * i + 1];
58 }
59
60 // find position of first nonzero byte
61 size_t p = 0;
62 while (p < EPHEMERAL_KEY_LENGTH && !S[p]) { ++p; }
63 if (p & 1) { ++p; } // skip one extra byte if p is odd
64 p /= 2; // offset into buffers
65
66 // hash each of the halves, starting at the first nonzero byte
67 SHA1::Digest const hash0 = SHA1::GetDigestOf(buf0.data() + p, EPHEMERAL_KEY_LENGTH / 2 - p);
68 SHA1::Digest const hash1 = SHA1::GetDigestOf(buf1.data() + p, EPHEMERAL_KEY_LENGTH / 2 - p);
69
70 // stick the two hashes back together
71 SessionKey K;
72 for (size_t i = 0; i < SHA1::DIGEST_LENGTH; ++i)
73 {
74 K[2 * i + 0] = hash0[i];
75 K[2 * i + 1] = hash1[i];
76 }
77 return K;
78}

83std::optional<SessionKey> SRP6::VerifyChallengeResponse(EphemeralKey const& A, SHA1::Digest const& clientM)
84{
85 ASSERT(!_used);
86 _used = true;
87
88 BigNumber const _A(A);
89 if ((_A % _N).IsZero())
90 {
91 return std::nullopt;
92 }
93
94 BigNumber const u(SHA1::GetDigestOf(A, B));
95 EphemeralKey const S = (_A * (_v.ModExp(u, _N))).ModExp(_b, N).ToByteArray<32>();
96
97 SessionKey K = SHA1Interleave(S);
98
99 // NgHash = H(N) xor H(g)
100 SHA1::Digest const NHash = SHA1::GetDigestOf(N);
101 SHA1::Digest const gHash = SHA1::GetDigestOf(g);
102 SHA1::Digest NgHash;
103 std::transform(NHash.begin(), NHash.end(), gHash.begin(), NgHash.begin(), std::bit_xor<>());
104
105 SHA1::Digest const ourM = SHA1::GetDigestOf(NgHash, _I, s, A, B, K);
106 if (ourM == clientM)
107 {
108 return K;
109 }
110 else
111 {
112 return std::nullopt;
113 }
