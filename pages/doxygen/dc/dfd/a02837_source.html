98 void Reset() override
99 {
100 if (events.IsInPhase(PHASE_TWO))
101 me->HandleStatModifier(UNIT_MOD_DAMAGE_MAINHAND, TOTAL_PCT, 35.0f, false); // hack
102 _Reset();
103 _summonCountA = 0;
104 _summonCountB = 0;
105 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_DAGGER));
106 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 1, uint32(WEAPON_DAGGER));
107 me->SetWalk(false);
108 me->SetHomePosition(PosMoveOnSpawn[0]);

114 _JustDied();
115 Talk(SAY_DEATH);
116 }

118 void EnterCombat(Unit* /*who*/) override
119 {
120 _EnterCombat();
121 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(7000, 9000), 0, PHASE_ONE);
122 events.ScheduleEvent(EVENT_GOUGE, urand(12000, 15000), 0, PHASE_ONE);
123 events.ScheduleEvent(EVENT_SUMMON_PROWLERS, 6000, 0, PHASE_ALL);
124 events.ScheduleEvent(EVENT_MARK_OF_ARLOKK, urand(9000, 11000), 0, PHASE_ALL);
125 events.ScheduleEvent(EVENT_TRANSFORM, 30000, 0, PHASE_ONE);
126 Talk(SAY_AGGRO);
127
128 // Sets up list of Panther spawners to cast on
129 std::list<Creature*> triggerList;
130 GetCreatureListWithEntryInGrid(triggerList, me, NPC_PANTHER_TRIGGER, 100.0f);
131 if (!triggerList.empty())
132 {
133 uint8 sideA = 0;
134 uint8 sideB = 0;
135 for (auto const& trigger : triggerList)
136 {
137 if (trigger->GetPositionY() < -1625.0f)
138 {
139 _triggersSideAGUID[sideA] = trigger->GetGUID();
140 ++sideA;
141 }
142 else
143 {
144 _triggersSideBGUID[sideB] = trigger->GetGUID();
145 ++sideB;
146 }
147 }
148 }
149 }

176 void UpdateAI(uint32 diff) override
177 {
178 if (!UpdateVictim())
179 return;
180
181 events.Update(diff);
182
183 if (me->HasUnitState(UNIT_STATE_CASTING))
184 return;
185
186 while (uint32 eventId = events.ExecuteEvent())
187 {
188 switch (eventId)
189 {
190 case EVENT_SHADOW_WORD_PAIN:
191 DoCastVictim(SPELL_SHADOW_WORD_PAIN, true);
192 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(5000, 7000), 0, PHASE_ONE);
193 break;
194 case EVENT_GOUGE:
195 DoCastVictim(SPELL_GOUGE, true);
196 break;
197 case EVENT_SUMMON_PROWLERS:
198 if (_summonCountA < MAX_PROWLERS_PER_SIDE)
199 {
200 if (Unit* trigger = ObjectAccessor::GetUnit(*me, _triggersSideAGUID[urand(0, 4)]))
201 {
202 trigger->CastSpell(trigger, SPELL_SUMMON_PROWLER);
203 ++_summonCountA;
204 }
205 }
206 if (_summonCountB < MAX_PROWLERS_PER_SIDE)
207 {
208 if (Unit* trigger = ObjectAccessor::GetUnit(*me, _triggersSideBGUID[urand(0, 4)]))
209 {
210 trigger->CastSpell(trigger, SPELL_SUMMON_PROWLER);
211 ++_summonCountB;
212 }
213 }
214 events.ScheduleEvent(EVENT_SUMMON_PROWLERS, 6000, 0, PHASE_ALL);
215 break;
216 case EVENT_MARK_OF_ARLOKK:
217 {
218 Unit* target = SelectTarget(SelectTargetMethod::MaxThreat, urand(1, 3), 0.0f, false, -SPELL_MARK_OF_ARLOKK);
219 if (!target)
220 target = me->GetVictim();
221 if (target)
222 {
223 DoCast(target, SPELL_MARK_OF_ARLOKK, true);
224 Talk(SAY_FEAST_PROWLER, target);
225 }
226 events.ScheduleEvent(EVENT_MARK_OF_ARLOKK, urand(120000, 130000));
227 break;
228 }
229 case EVENT_TRANSFORM:
230 {
231 DoCastSelf(SPELL_PANTHER_TRANSFORM); // SPELL_AURA_TRANSFORM
232 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(EQUIP_UNEQUIP));
233 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 1, uint32(EQUIP_UNEQUIP));
234 me->AttackStop();
235 DoResetThreat();
236 me->SetReactState(REACT_PASSIVE);
237 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
238 DoCastSelf(SPELL_VANISH_VISUAL);
239 DoCastSelf(SPELL_VANISH);
240 events.ScheduleEvent(EVENT_VANISH, 1000, 0, PHASE_ONE);
241 break;
242 }
243 case EVENT_VANISH:
244 DoCastSelf(SPELL_SUPER_INVIS);
245 me->SetWalk(false);
246 me->GetMotionMaster()->MovePoint(0, frand(-11551.0f, -11508.0f), frand(-1638.0f, -1617.0f), me->GetPositionZ());
247 events.ScheduleEvent(EVENT_VANISH_2, 9000, 0, PHASE_ONE);
248 break;
249 case EVENT_VANISH_2:
250 DoCastSelf(SPELL_VANISH);
251 DoCastSelf(SPELL_SUPER_INVIS);
252 events.ScheduleEvent(EVENT_VISIBLE, urand(41000, 47000), 0, PHASE_ONE);
253 break;
254 case EVENT_VISIBLE:
255 me->SetReactState(REACT_AGGRESSIVE);
256 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
257 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
258 AttackStart(target);
259 me->RemoveAura(SPELL_SUPER_INVIS);
260 me->RemoveAura(SPELL_VANISH);
261 events.ScheduleEvent(EVENT_RAVAGE, urand(10000, 14000), 0, PHASE_TWO);
262 events.ScheduleEvent(EVENT_TRANSFORM_BACK, urand(30000, 40000), 0, PHASE_TWO);
263 events.SetPhase(PHASE_TWO);
264 me->HandleStatModifier(UNIT_MOD_DAMAGE_MAINHAND, TOTAL_PCT, 35.0f, true); // hack
265 break;
266 case EVENT_RAVAGE:
267 DoCastVictim(SPELL_RAVAGE, true);
268 events.ScheduleEvent(EVENT_RAVAGE, urand(10000, 14000), 0, PHASE_TWO);
269 break;
270 case EVENT_TRANSFORM_BACK:
271 {
272 me->RemoveAura(SPELL_PANTHER_TRANSFORM); // SPELL_AURA_TRANSFORM
273 DoCast(me, SPELL_VANISH_VISUAL);
274 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_DAGGER));
275 me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 1, uint32(WEAPON_DAGGER));
276 me->HandleStatModifier(UNIT_MOD_DAMAGE_MAINHAND, TOTAL_PCT, 35.0f, false); // hack
277 events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(4000, 7000), 0, PHASE_ONE);
278 events.ScheduleEvent(EVENT_GOUGE, urand(12000, 15000), 0, PHASE_ONE);
279 events.ScheduleEvent(EVENT_TRANSFORM, 30000, 0, PHASE_ONE);
280 events.SetPhase(PHASE_ONE);
281 break;
282 }
283 default:
284 break;
285 }
286 }
287
288 DoMeleeAttackIfReady();
289 }
