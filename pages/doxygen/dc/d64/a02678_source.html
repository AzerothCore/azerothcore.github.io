58struct boss_kormok : public ScriptedAI
59{
60 boss_kormok(Creature* creature) : ScriptedAI(creature), _summons(creature) {}
61
62 void Reset() override
63 {
64 _mages = false;
65
66 _scheduler.CancelAll();
67 _scheduler.SetValidator([this]
68 {
69 return !me->HasUnitState(UNIT_STATE_CASTING);
70 });
71
72 _summons.DespawnAll();
73 }
74
75 void IsSummonedBy(Unit* /*summoner*/) override
76 {
77 Talk(TALK_SUMMON);
78
79 _scheduler.Schedule(2s, [this](TaskContext context)
80 {
81 DoCastSelf(SPELL_BONE_SHIELD);
82 context.Repeat(45s);
83 });
84 }
85
86 void EnterCombat(Unit* /*who*/) override
87 {
88 Talk(TALK_AGGRO);
89
90 _scheduler.Schedule(10s, [this](TaskContext context)
91 {
92 DoCastVictim(SPELL_SHADOWBOLT_VOLLEY);
93 context.Repeat(15s);
94 })
95 .Schedule(15s, [this](TaskContext context)
96 {
97 DoCast(SPELL_SUMMON_BONE_MINIONS);
98 context.Repeat(12s);
99 });
100 }
101
102 void JustSummoned(Creature* summon) override
103 {
104 _summons.Summon(summon);
105 DoZoneInCombat(summon);
106 }
107
108 void SummonedCreatureDespawn(Creature* summon) override
109 {
110 _summons.Despawn(summon);
111 }
112
113 void DamageTaken(Unit* /*attacker*/, uint32& damage, DamageEffectType /*damageType*/, SpellSchoolMask /*damageSchoolMask*/) override
114 {
115 if (!_mages && me->HealthBelowPctDamaged(25, damage))
116 {
117 _mages = true;
118
119 Talk(TALK_ENRAGE);
120
121 DoCast(SPELL_SUMMON_BONE_MAGES);
122 }
123 }
124
125 void JustDied(Unit* /*killer*/) override
126 {
127 Talk(TALK_DEATH);
128 }
129
130 void UpdateAI(uint32 diff) override
131 {
132 _scheduler.Update(diff);
133
134 if (!UpdateVictim())
135 {
136 return;
137 }
138
139 DoMeleeAttackIfReady();
140 }
141
