162 void EnterCombat(Unit* /*pWho*/) override
163 {
164 if (!EventStarted)
165 {
166 EventStarted = true;
167 Talk(SAY_AGGRO);
168 if (m_pInstance)
169 {
170 if (IsHeroic())
171 m_pInstance->DoStartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, ACHIEV_TIMED_LODI_DODI);
172
173 m_pInstance->SetData(DATA_SKADI_THE_RUTHLESS, IN_PROGRESS);
174 }

176 me->SetControlled(true, UNIT_STATE_ROOT);
177 me->SetInCombatWithZone();
178 events.RescheduleEvent(EVENT_SKADI_START, 2000);

181
182 void DoAction(int32 param) override
183 {
184 if (param == ACTION_PHASE2)
185 {
186 SecondPhase = true;
187 events.ScheduleEvent(EVENT_SKADI_CRUSH, 8000);
188 events.ScheduleEvent(EVENT_SKADI_SPEAR, 10000);
189 events.ScheduleEvent(EVENT_SKADI_WHIRLWIND, 15000);
190
191 if (me->GetVictim())
192 me->GetMotionMaster()->MoveChase(me->GetVictim());
193 else
194 me->SetInCombatWithZone();
195 }
196 }
197
198 void UpdateAI(uint32 diff) override
199 {
200 if (!UpdateVictim() && SecondPhase)
201 return;
202
203 events.Update(diff);
204 if (me->HasUnitState(UNIT_STATE_CASTING))
205 return;
206
207 switch (events.ExecuteEvent())
208 {
209 case EVENT_SKADI_START:
210 {
211 me->SetControlled(false, UNIT_STATE_ROOT);
212 if (Creature* cr = GetGrauf())
213 {
214 me->EnterVehicleUnattackable(cr, 0);
215 cr->AI()->DoAction(ACTION_START_EVENT);
216 }
217 else
218 EnterEvadeMode();
219
220 break;
221 }
222 case EVENT_SKADI_CRUSH:
223 {
224 me->CastSpell(me->GetVictim(), IsHeroic() ? SPELL_CRUSH_H : SPELL_CRUSH_N, false);
225 events.RepeatEvent(8000);
226 break;
227 }
228 case EVENT_SKADI_SPEAR:
229 {
230 if (Unit* tgt = SelectTarget(SelectTargetMethod::Random, 0))
231 me->CastSpell(tgt, IsHeroic() ? SPELL_POISONED_SPEAR_H : SPELL_POISONED_SPEAR_N, false);
232
233 events.RepeatEvent(10000);
234 break;
235 }
236 case EVENT_SKADI_WHIRLWIND:
237 {
238 me->CastSpell(me, IsHeroic() ? SPELL_WHIRLWIND_H : SPELL_WHIRLWIND_N, false);
239 events.RepeatEvent(15000 + rand() % 5000);
240 events.DelayEvents(10000);
241 break;
242 }
243 }
244
245 DoMeleeAttackIfReady();
246 }
247
248 void JustDied(Unit* /*pKiller*/) override
249 {

252

254 {
255 m_pInstance->SetData(DATA_SKADI_THE_RUTHLESS, DONE);

257 }
258 }
259
260 void KilledUnit(Unit* /*pVictim*/) override
261 {
262 if (urand(0, 1))
263 return;
264
265 Talk(SAY_KILL);
266 }
