105 void UpdateAI(uint32 diff) override
106 {
107 events.Update(diff);
108
109 if (events.ExecuteEvent() == EVENT_CHECK_SUMMON_AURA)

111 std::list<Creature*> tigers;

114 {

116 {

119
120 if (Unit* summoner = (*itr)->ToTempSummon()->GetSummonerUnit())
121 if (!summoner->HasAura(SPELL_NO_SUMMON_AURA) && !summoner->HasAura(SPELL_SUMMON_ZENTABRA_TRIGGER)
122 && !summoner->IsInCombat())
123 {
124 me->AddAura(SPELL_NO_SUMMON_AURA, summoner);
125 me->AddAura(SPELL_DETECT_INVIS, summoner);
126 summoner->CastSpell(summoner, SPELL_SUMMON_MATRIARCH, true);
127 Talk(SAY_MATRIARCH_AGGRO, summoner);
128 }
129 }
130 }
131
132 events.ScheduleEvent(EVENT_CHECK_SUMMON_AURA, 5000);
133 }
134 }

146class npc_tiger_matriarch : public CreatureScript
147{
148public:
149 npc_tiger_matriarch() : CreatureScript("npc_tiger_matriarch") { }
150
151 struct npc_tiger_matriarchAI : public ScriptedAI
152 {
153 npc_tiger_matriarchAI(Creature* creature) : ScriptedAI(creature)
154 {
155 }
156
157 void EnterCombat(Unit* /*target*/) override
158 {
159 _events.Reset();
160 _events.ScheduleEvent(EVENT_POUNCE, 100);
161 _events.ScheduleEvent(EVENT_NOSUMMON, 50000);
162 }
163
164 void IsSummonedBy(Unit* summoner) override
165 {
166 if (summoner->GetTypeId() != TYPEID_PLAYER || !summoner->GetVehicle())
167 return;
168
169 _tigerGuid = summoner->GetVehicle()->GetBase()->GetGUID();
170 if (Unit* tiger = ObjectAccessor::GetUnit(*me, _tigerGuid))
171 {
172 me->AddThreat(tiger, 500000.0f);
173 DoCast(me, SPELL_FURIOUS_BITE);
174 }
175 }
176
177 void KilledUnit(Unit* victim) override
178 {
179 if (victim->GetTypeId() != TYPEID_UNIT || !victim->IsSummon())
180 return;
181
182 if (Unit* vehSummoner = victim->ToTempSummon()->GetSummonerUnit())
183 {
184 vehSummoner->RemoveAurasDueToSpell(SPELL_NO_SUMMON_AURA);
185 vehSummoner->RemoveAurasDueToSpell(SPELL_DETECT_INVIS);
186 vehSummoner->RemoveAurasDueToSpell(SPELL_SPIRIT_OF_THE_TIGER_RIDER);
187 vehSummoner->RemoveAurasDueToSpell(SPELL_SUMMON_ZENTABRA_TRIGGER);
188 }
189 me->DespawnOrUnsummon();
190 }
191
192 void DamageTaken(Unit* attacker, uint32& damage, DamageEffectType, SpellSchoolMask) override
193 {
194 if (!attacker || !attacker->IsSummon())
195 return;
196
197 if (HealthBelowPct(20))
198 {
199 damage = 0;
200 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
201 if (Unit* vehSummoner = attacker->ToTempSummon()->GetSummonerUnit())
202 {
203 vehSummoner->AddAura(SPELL_SUMMON_ZENTABRA_TRIGGER, vehSummoner);
204 vehSummoner->CastSpell(vehSummoner, SPELL_SUMMON_ZENTABRA, true);
205 attacker->CastSpell(attacker, SPELL_EJECT_PASSENGERS, true);
206 vehSummoner->RemoveAurasDueToSpell(SPELL_NO_SUMMON_AURA);
207 vehSummoner->RemoveAurasDueToSpell(SPELL_DETECT_INVIS);
208 vehSummoner->RemoveAurasDueToSpell(SPELL_SPIRIT_OF_THE_TIGER_RIDER);
209 vehSummoner->RemoveAurasDueToSpell(SPELL_SUMMON_ZENTABRA_TRIGGER);
210 }
211
212 me->DespawnOrUnsummon();
213 }
214 }
215
216 void UpdateAI(uint32 diff) override
217 {
218 if (!UpdateVictim())
219 return;
220
221 if (!_tigerGuid)
222 return;
223
224 _events.Update(diff);
225
226 while (uint32 eventId = _events.ExecuteEvent())
227 {
228 switch (eventId)
229 {
230 case EVENT_POUNCE:
231 DoCastVictim(SPELL_POUNCE);
232 _events.ScheduleEvent(EVENT_POUNCE, 30000);
233 break;
234 case EVENT_NOSUMMON: // Reapply SPELL_NO_SUMMON_AURA
235 if (Unit* tiger = ObjectAccessor::GetUnit(*me, _tigerGuid))
236 {
237 if (tiger->IsSummon())
238 if (Unit* vehSummoner = tiger->ToTempSummon()->GetSummonerUnit())
239 me->AddAura(SPELL_NO_SUMMON_AURA, vehSummoner);
240 }
241 _events.ScheduleEvent(EVENT_NOSUMMON, 50000);
242 break;
243 default:
244 break;
245 }
246 }
247
248 DoMeleeAttackIfReady();
249 }
250
251 private:
252 EventMap _events;
253 ObjectGuid _tigerGuid;
254 };
255
256 CreatureAI* GetAI(Creature* creature) const override
257 {
258 return new npc_tiger_matriarchAI(creature);
259 }

332

341 me->Dismount();

344 }
