60 void UpdateAI(uint32 diff) override
61 {
62 // Return since we have no target
63 if (!UpdateVictim())
64 {
65 return;
66 }
67 events.Update(diff);
68
69 if (me->HasUnitState(UNIT_STATE_CASTING))
70 {
71 return;
72 }
73 while (uint32 eventId = events.ExecuteEvent())
74 {
75 switch (eventId)
76 {
77 case SPELL_WHIRLWIND:
78 if (me->GetDistance2d(me->GetVictim()) < 10.0f)
79 {
80 DoCastVictim(SPELL_WHIRLWIND);
81 nextWhirlwindTime = urand(TIMER_WHIRLWIND - 2000, TIMER_WHIRLWIND + 2000);
82 }
83 else
84 {
85 // reschedule sooner
86 nextWhirlwindTime = 0.3 * urand(TIMER_WHIRLWIND - 2000, TIMER_WHIRLWIND + 2000);
87 }
88 events.ScheduleEvent(SPELL_WHIRLWIND, nextWhirlwindTime);
89 break;
90 case SPELL_MORTALSTRIKE:
91 DoCastVictim(SPELL_MORTALSTRIKE);
92 events.ScheduleEvent(SPELL_MORTALSTRIKE, urand(TIMER_MORTAL - 2000, TIMER_MORTAL + 2000));
93 break;
94 case SPELL_BLOODLUST:
95 DoCastSelf(SPELL_BLOODLUST);
96 events.ScheduleEvent(SPELL_BLOODLUST, urand(TIMER_BLOODLUST - 2000, TIMER_BLOODLUST + 2000));
97 break;
98 default:
99 break;
100 }
101 }
102 DoMeleeAttackIfReady();
103 }
104 };
105};
