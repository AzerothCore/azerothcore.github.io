116 void MoveInLineOfSight(Unit* target) override
117 {
118 if (!_playerGUID || target->GetEntry() != NPC_ROCKBITER || !me->IsInRange(target, 0.f, 10.f))

120 if (!me->IsInCombat() && !me->GetVictim())

162 void FailQuest(Player* player, bool despawn)
163 {

185 me->DespawnOrUnsummon(1);
186 }
187 }

197 void sQuestAccept(Player* player, Quest const* quest) override
198 {
199 if (quest->GetQuestId() == QUEST_WANDERING_SHAY)
200 {
201 _playerGUID = player->GetGUID();
202
203 Talk(TALK_0, player);
204
205 me->RemoveNpcFlag(UNIT_NPC_FLAG_QUESTGIVER);
206 me->SetImmuneToAll(false);
207 me->SetFaction(FACTION_ESCORT_N_NEUTRAL_ACTIVE);
208 me->GetMotionMaster()->MoveFollow(player, 3.f, M_PI);
209
210 _events.ScheduleEvent(EVENT_WANDERING_START, urand(40 * IN_MILLISECONDS, 70 * IN_MILLISECONDS));
211 _events.ScheduleEvent(EVENT_CHECK_FOLLOWER, 30 * IN_MILLISECONDS);
212 }
213 }

228 void UpdateAI(uint32 diff) override
229 {
230 if (UpdateVictim())
231 {
232 DoMeleeAttackIfReady();
233 return;
234 }
235
236 _events.Update(diff);
237 while (uint32 eventId = _events.ExecuteEvent())
238 {
239 switch (eventId)
240 {
241 case EVENT_WANDERING_START:
242 {
243 Position pos = me->GetFirstCollisionPosition(15.f, rand_norm() * static_cast<float>(2 * M_PI));
244 me->GetMotionMaster()->MovePoint(0, pos);
245 Talk(TALK_2);
246 _events.ScheduleEvent(EVENT_WANDERING_START, urand(60 * IN_MILLISECONDS, 70 * IN_MILLISECONDS));
247 _events.ScheduleEvent(EVENT_WANDERING_TALK, 3 * IN_MILLISECONDS);
248 _events.ScheduleEvent(EVENT_WANDERING_RANDOM, 8 * IN_MILLISECONDS);
249 break;
250 }
251 case EVENT_WANDERING_TALK:
252 Talk(TALK_3);
253 break;
254 case EVENT_WANDERING_RANDOM:
255 me->SetHomePosition(me->GetPosition());
256 me->GetMotionMaster()->MoveRandom(15.f);
257 break;
258 case EVENT_FINAL_TALK:
259 if (Creature* robckbiter = ObjectAccessor::GetCreature(*me, _rockbiterGUID))
260 {
261 robckbiter->AI()->Talk(TALK_0, me);
262 }
263 me->DespawnOrUnsummon(10 * IN_MILLISECONDS);
264 break;
265 case EVENT_CHECK_FOLLOWER:
266 {
267 Player* player = ObjectAccessor::GetPlayer(*me, _playerGUID);
268 if (!player || !player->IsAlive() || !me->IsInRange(player, 0.f, 50.f))
269 {
270 FailQuest(player, true);
271 }
272 break;
273 }
274 default:
275 break;
276 }
277 }
278 }

283 EventMap _events;
