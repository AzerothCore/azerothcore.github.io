99template <class T>
100void LoadHelper(CellGuidSet const& guid_set, CellCoord& cell, GridRefMgr<T>& m, uint32& count, Map* map)
101{
102 for (CellGuidSet::const_iterator i_guid = guid_set.begin(); i_guid != guid_set.end(); ++i_guid)
103 {
104 T* obj = new T;
105 ObjectGuid::LowType guid = *i_guid;
106
107 if (!obj->LoadFromDB(guid, map))
108 {
109 delete obj;
110 continue;
111 }
112
113 AddObjectHelper(cell, m, count, map, obj);
114 }
115}

117template <>
118void LoadHelper(CellGuidSet const& guid_set, CellCoord& cell, GridRefMgr<GameObject>& m, uint32& count, Map* map)
119{
120 for (CellGuidSet::const_iterator i_guid = guid_set.begin(); i_guid != guid_set.end(); ++i_guid)
121 {
122 ObjectGuid::LowType guid = *i_guid;
123 GameObjectData const* data = sObjectMgr->GetGOData(guid);
124 GameObject* obj = data && sObjectMgr->IsGameObjectStaticTransport(data->id) ? new StaticTransport() : new GameObject();
125
126 if (!obj->LoadFromDB(guid, map))
127 {
128 delete obj;
129 continue;
130 }
131
132 AddObjectHelper(cell, m, count, map, obj);
133 }
134}

150void ObjectWorldLoader::Visit(CorpseMapType& /*m*/)
151{
152 CellCoord cellCoord = i_cell.GetCellCoord();
153 if (std::unordered_set<Corpse*> const* corpses = i_map->GetCorpsesInCell(cellCoord.GetId()))
154 {
155 for (Corpse* corpse : *corpses)
156 {
157 corpse->AddToWorld();
158 GridType& cell = i_grid.GetGridType(i_cell.CellX(), i_cell.CellY());
159 if (corpse->IsWorldObject())
160 cell.AddWorldObject(corpse);
161 else
162 cell.AddGridObject(corpse);
163
164 ++i_corpses;
165 }
166 }
167}

169void ObjectGridLoader::LoadN(void)
170{
171 i_gameObjects = 0;
172 i_creatures = 0;
173 i_corpses = 0;
174 i_cell.data.Part.cell_y = 0;
175 for (uint32 x = 0; x < MAX_NUMBER_OF_CELLS; ++x)
176 {
177 i_cell.data.Part.cell_x = x;
178 for (uint32 y = 0; y < MAX_NUMBER_OF_CELLS; ++y)
179 {
180 i_cell.data.Part.cell_y = y;
181
182 //Load creatures and game objects
183 {
184 TypeContainerVisitor<ObjectGridLoader, GridTypeMapContainer> visitor(*this);
185 i_grid.VisitGrid(x, y, visitor);
186 }
187
188 //Load corpses (not bones)
189 {
190 ObjectWorldLoader worker(*this);
191 TypeContainerVisitor<ObjectWorldLoader, WorldTypeMapContainer> visitor(worker);
192 i_grid.VisitGrid(x, y, visitor);
193 }
194 }
195 }
196 LOG_DEBUG("maps", "{} GameObjects, {} Creatures, and {} Corpses/Bones loaded for grid {} on map {}", i_gameObjects, i_creatures, i_corpses, i_grid.GetGridId(), i_map->GetId());
197}
