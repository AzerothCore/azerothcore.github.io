66 void UpdateAI(uint32 diff) override
67 {
68 if (!UpdateVictim())
69 return;
70
71 events.Update(diff);
72
73 if (me->HasUnitState(UNIT_STATE_CASTING))
74 return;
75
76 while (uint32 eventId = events.ExecuteEvent())
77 {
78 switch (eventId)
79 {
80 case EVENT_FLAMESTRIKE:
81 DoCastAOE(SPELL_FLAMESTRIKE);
82 events.ScheduleEvent(EVENT_FLAMESTRIKE, 10000);
83 break;
84 case EVENT_CLEAVE:
85 DoCastVictim(SPELL_CLEAVE);
86 events.ScheduleEvent(EVENT_CLEAVE, 8000);
87 break;
88 case EVENT_CONFLAGRATION:
89 DoCastVictim(SPELL_CONFLAGRATION);
90
91 if (Unit* target = me->GetVictim())
92 {
93 _conflagrateTarget = me->GetVictim()->GetGUID();
94 _conflagrateThreat = me->GetThreatMgr().GetThreat(me->GetVictim());
95 me->GetThreatMgr().ModifyThreatByPercent(target, -100);
96 }
97 events.ScheduleEvent(EVENT_CONFLAGRATION, urand(10000, 13000));
98 events.ScheduleEvent(EVENT_CHECK_CONFLAGRATION_TARGET, 10000);
99 break;
100 case EVENT_THUNDERCLAP:
101 DoCastVictim(SPELL_THUNDERCLAP);
102 events.ScheduleEvent(EVENT_THUNDERCLAP, 20000);
103 break;
104 case EVENT_PIERCE_ARMOR:
105 DoCastVictim(SPELL_PIERCE_ARMOR);
106 events.ScheduleEvent(EVENT_PIERCE_ARMOR, 40000);
107 break;
108 case EVENT_RAGE:
109 DoCastSelf(SPELL_RAGE);
110 events.ScheduleEvent(EVENT_RAGE, 35000);
111 break;
112 case EVENT_CHECK_CONFLAGRATION_TARGET:
113 if (Unit* target = ObjectAccessor::GetUnit(*me, _conflagrateTarget))
114 {
115 me->GetThreatMgr().AddThreat(target, _conflagrateThreat);
116 }
117 break;
118 }
119 }
120 DoMeleeAttackIfReady();
121 }
122
123 private:
124 float _conflagrateThreat;
125 ObjectGuid _conflagrateTarget;
126 };

128 CreatureAI* GetAI(Creature* creature) const override
129 {
130 return GetBlackrockSpireAI<boss_drakkisathAI>(creature);
131 }
