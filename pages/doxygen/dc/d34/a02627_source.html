240 void PrepareEncounter()
241 {
242 LOG_DEBUG("scripts.ai", "Barnes Opera Event - Introduction complete - preparing encounter {}", m_uiEventId);
243 uint8 index = 0;
244 uint8 count = 0;
245
246 switch (m_uiEventId)
247 {
248 case EVENT_OZ:
249 index = 0;
250 count = 4;
251 break;
252 case EVENT_HOOD:
253 index = 4;
254 count = index + 1;
255 break;
256 case EVENT_RAJ:
257 index = 5;
258 count = index + 1;
259 break;
260 }
261
262 for (; index < count; ++index)
263 {
264 uint32 entry = ((uint32)Spawns[index][0]);
265 float PosX = Spawns[index][1];
266
267 if (Creature* creature = me->SummonCreature(entry, PosX, SPAWN_Y, SPAWN_Z, SPAWN_O, TEMPSUMMON_TIMED_OR_DEAD_DESPAWN, HOUR * 2 * IN_MILLISECONDS))
268 creature->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
269 }
270
271 RaidWiped = false;
272 }

274 void UpdateAI(uint32 diff) override
275 {
276 npc_escortAI::UpdateAI(diff);
277
278 if (HasEscortState(STATE_ESCORT_PAUSED))
279 {
280 if (TalkTimer <= diff)
281 {
282 if (TalkCount > 3)
283 {
284 if (Creature* pSpotlight = ObjectAccessor::GetCreature(*me, m_uiSpotlightGUID))
285 pSpotlight->DespawnOrUnsummon();
286
287 SetEscortPaused(false);
288 return;
289 }
290
291 Talk(TalkCount);
292 ++TalkCount;
293 }
294 else TalkTimer -= diff;
295 }
296
297 if (PerformanceReady)
298 {
299 if (!RaidWiped)
300 {
301 if (WipeTimer <= diff)
302 {
303 Map* map = me->GetMap();
304 if (!map->IsDungeon())
305 return;
306
307 Map::PlayerList const& PlayerList = map->GetPlayers();
308 if (PlayerList.IsEmpty())
309 return;
310
311 RaidWiped = true;
312 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
313 {
314 if (i->GetSource()->IsAlive() && !i->GetSource()->IsGameMaster())
315 {
316 RaidWiped = false;
317 break;
318 }
319 }
320
321 if (RaidWiped)
322 {
323 RaidWiped = true;
324 EnterEvadeMode();
325 return;
326 }
327
328 WipeTimer = 15000;
329 }
330 else WipeTimer -= diff;
331 }
332 }
333 }

355 break;
356 case GOSSIP_ACTION_INFO_DEF+4:
357 CloseGossipMenuFor(player);
358 pBarnesAI->m_uiEventId = EVENT_HOOD;
359 break;
360 case GOSSIP_ACTION_INFO_DEF+5:
361 CloseGossipMenuFor(player);
362 pBarnesAI->m_uiEventId = EVENT_RAJ;
