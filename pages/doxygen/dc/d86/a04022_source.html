116 void UpdateAI(uint32 diff) override
117 {
118 if (!UpdateVictim())
119 return;
120
121 events.Update(diff);
122 if (me->HasUnitState(UNIT_STATE_CASTING))
123 return;
124
125 switch (events.ExecuteEvent())
126 {
127 case EVENT_SUMMON_IMAGE1:
128 if (HealthBelowPct(67))
129 {
130 Talk(SAY_IMAGE);
131 me->CastSpell(me, SPELL_66_ILLUSION, false);
132 break;
133 }
134 events.ScheduleEvent(EVENT_SUMMON_IMAGE1, 1000);
135 break;
136 case EVENT_SUMMON_IMAGE2:
137 if (HealthBelowPct(34))
138 {
139 Talk(SAY_IMAGE);
140 me->CastSpell(me, SPELL_33_ILLUSION, false);
141 break;
142 }
143 events.ScheduleEvent(EVENT_SUMMON_IMAGE2, 1000);
144 break;
145 case EVENT_SPELL_MIND_REND:
146 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f))
147 me->CastSpell(target, SPELL_MIND_REND, false);
148 events.ScheduleEvent(EVENT_SPELL_MIND_REND, 10000);
149 break;
150 case EVENT_SPELL_FEAR:
151 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 20.0f))
152 {
153 Talk(SAY_FEAR);
154 me->CastSpell(target, SPELL_FEAR, false);
155 }
156 events.ScheduleEvent(EVENT_SPELL_FEAR, 25000);
157 break;
158 case EVENT_SPELL_DOMINATION:
159 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 1, 30.0f))
160 {
161 Talk(SAY_MIND);
162 me->CastSpell(target, SPELL_DOMINATION, false);
163 }
164 events.ScheduleEvent(EVENT_SPELL_DOMINATION, 30000);
165 break;
166 case EVENT_SPELL_MANA_BURN:
167 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, PowerUsersSelector(me, POWER_MANA, 40.0f, false)))
168 me->CastSpell(target, SPELL_MANA_BURN, false);
169 events.ScheduleEvent(EVENT_SPELL_MANA_BURN, 30000);
170 break;
171 }
172
173 DoMeleeAttackIfReady();
174 }
175 };
