132 events.RescheduleEvent(EVENT_SPELL_ARCANE_BARRAGE, 0);

175 void UpdateAI(uint32 diff) override
176 {
177 if( !UpdateVictim() )
178 return;
179
180 if( me->HasAura(SPELL_PLANAR_SHIFT) || me->HasAura(SPELL_DRAKE_STOP_TIME) )
181 return;
182
183 events.Update(diff);
184
185 if( me->HasUnitState(UNIT_STATE_CASTING) )
186 return;
187
188 DoMeleeAttackIfReady();
189
190 switch( events.ExecuteEvent() )
191 {
192 case 0:
193 break;
194 case EVENT_SPELL_ARCANE_BARRAGE:
195 if( Unit* v = me->GetVictim() )
196 me->CastSpell(v, SPELL_ARCANE_BARRAGE, false);
197 events.RepeatEvent(2500);
198 break;
199 case EVENT_SPELL_ARCANE_VOLLEY:
200 me->CastSpell(me, SPELL_ARCANE_VOLLEY, false);
201 events.RepeatEvent(8000);
202 break;
203 case EVENT_SPELL_ENRAGED_ASSAULT:
204 Talk(SAY_ENRAGE);
205 me->CastSpell(me, SPELL_ENRAGED_ASSAULT, false);
206 events.RepeatEvent(35000);
207 break;
208 case EVENT_SUMMON_WHELPS:
209 for( uint8 i = 0; i < 5; ++i )
210 events.ScheduleEvent(EVENT_SUMMON_SINGLE_WHELP, urand(0, 8000));
211 events.RepeatEvent(40000);
212 break;
213 case EVENT_SUMMON_SINGLE_WHELP:
214 {
215 float x = rand_norm() * 50.0f - 25.0f;
216 float y = rand_norm() * 50.0f - 25.0f;
217 float z = rand_norm() * 50.0f - 25.0f;
218 me->SummonCreature(NPC_LEY_GUARDIAN_WHELP, me->GetPositionX() + x, me->GetPositionY() + y, me->GetPositionZ() + z, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 5000);
219 }
220 break;
221 case EVENT_SPELL_PLANAR_SHIFT:
222 //me->Yell(TEXT_PLANAR_SHIFT_SAY, LANG_UNIVERSAL);
223 Talk(SAY_SHIELD);
224 me->CastSpell(me, SPELL_PLANAR_SHIFT, false);
225 for( uint8 i = 0; i < 3; ++i )
226 if( Unit* t = SelectTarget(SelectTargetMethod::Random, 0, 300.0f, false) )
227 if( Creature* pa = me->SummonCreature(NPC_PLANAR_ANOMALY, *me, TEMPSUMMON_TIMED_DESPAWN, 17000) )
228 {
229 pa->SetCanFly(true);
230 pa->SetDisableGravity(true);
231 pa->SendMovementFlagUpdate();
232 pa->CastSpell(pa, SPELL_PLANAR_AURA_VISUAL, true);
233 pa->CastSpell(pa, SPELL_PLANAR_AURA_DAMAGE, true);
234 if (Aura* a = pa->GetAura(SPELL_PLANAR_AURA_DAMAGE))
235 a->SetDuration(15000);
236 if( pa->AI() )
237 {
238 pa->AI()->AttackStart(t);
239 pa->GetMotionMaster()->MoveChase(t, 0.01f);
240 }
241 }
242 break;
243 }
244 }
